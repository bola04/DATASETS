     var yScale = windowHeight / image.height;
     var scale;
     var yOffset = 0;
     var xOffset = 0;
 
     if (xScale > yScale) {
         scale = xScale;
         yOffset = (windowHeight - (image.height * scale)) / 2;

         scale = yScale;
         xOffset = (windowWidth - (image.width * scale)) / 2;

 		icotop = $(mapicons[i]).attr("data-top")*1;
 		
 		curleft = (icoleft) * scale + xOffset;
 		curtop = (icotop) * scale + yOffset;
 		
 		$(mapicons[i]).css('top', curtop);
 		$(mapicons[i]).css('left', curleft);
 		
 		textid = $(mapicons[i]).attr("data-textid");
 		$(".bt"+textid).css('top', curtop);
 		$(".bt"+textid).css('left',  curleft);
 		
 		
 	};
 	
 };
