    jQuery(function ($) {
 			$('ul.main-local-sections').each(function () {
 				var $select = $('<select class="menu-local" />');
 
 				$(this).find('a').each(function () {
 					var $option = $('<option />');
 					$option.attr('value', $(this).attr('href')).html($(this).html());
 					$select.append($option);
 				});
 
 				$(this).replaceWith($select);
 
 				var $menuLocal = $('.menu-local').html();
 				$('#toggle-menu-local').append($menuLocal);
 				$('.primary-local').remove();
 
 				// bind change event to select
 				$('#toggle-menu-local').bind('change', function () {
 					var url = $(this).val();
 					if (url) {
 						window.location = url;
 					}
 					return false;
 				});
 			});
