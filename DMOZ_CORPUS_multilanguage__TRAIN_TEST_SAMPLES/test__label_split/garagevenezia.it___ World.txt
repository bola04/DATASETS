       function GetModels(brand) {
         $.ajax({
             type: "POST",
             url: "/AjaxGetCarModels.aspx",
             data: "&brand=" + $("#marca").val(),
             success: function (msg) {
                 var response = msg.split("|");
                 var ok = response[0].split(":");
                 var retVal = ok[1];
                 if (ok[0] == 0) {
                     $("#modello").append("<option value='tutto'>Tutto</option>");
                 }
                 else {
                     $("#modello").html("");
                     $("#modello").html("<option value='tutto'>Tutto</option>");
                     for (var i = 0; i < response.length; i++) {
                         var rtype = response[i].split(":");
                         $("#modello").append("<option value='" + rtype[0] + "'>" + rtype[1] + "</option>");
                     }
                 }
             },
             error: function () {
                 alert("E avvenuto un errore di comunicazione!");
             }
         });
     }
 
     function OnBrandChange(dropdown) {
         var myindex = dropdown.selectedIndex;
         var SelValue = dropdown.options[myindex].value;
         if (SelValue != "")
             GetModels(SelValue);
         else
             $("#modello").html("<option value='tutto' selected='selected'>Tutto</option>");
         return false;
     }
