         this.hide(this.index);  
         this.index++;  
         if(this.index >= this.ITEMS.length) {  
             this.index = 0;

         this.hide(this.index);  
         this.index--;  
         if(this.index <= -1) {

         }  
         this.fadeIn(this.index);  
       
     },

         });  
         $(NEXT).bind("click",function() {  
             Scope.nextImage();  
             clearInterval(Scope.timer);  
             return false;  
         });
