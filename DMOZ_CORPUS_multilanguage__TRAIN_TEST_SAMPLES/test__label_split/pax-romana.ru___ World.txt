      function resizeImageBlock(){
     var imageBlock = $('.images-block');
     var maxBlockHeight = 495;
     var minBlockHeight = 395;
     var maxDocHeight = 782;
     var minDocHeight = 590;
     var gap = 215;
     var docHeight = $(document).height();
     var blockHeight = '';
     
     if(docHeight >  maxDocHeight){
       blockHeight = maxBlockHeight;
     }else if(docHeight <= minDocHeight || docHeight >= minDocHeight){
       blockHeight = docHeight - gap;
     }
     if(blockHeight > maxBlockHeight){
       blockHeight = maxBlockHeight;
     }else if(blockHeight < minBlockHeight){
       blockHeight = minBlockHeight;
     }
     imageBlock.css('height', blockHeight+'px');
     imageBlock.find('li').css('height', blockHeight+'px');
     //imageBlock.find('ul').css('height', blockHeight+'px');

   }
