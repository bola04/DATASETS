 function mycarousel_itemVisibleInCallback(carousel, item, i, state, evt)

 function mycarousel_getItemHTML(item)
 {
     var imgname=item.url.split("/");
     var alt=imgname[imgname.length-1].split(".");
     var alt2=alt[alt.length-2].split("_");
     var big_image_url='';
     $.each(alt, function(index, item) {
         if (index <(alt.length-2)) {
             big_image_url=big_image_url+item+'.';};
     });
     $.each(alt2, function(index, item) {
         if (index <(alt2.length-1)) {
             big_image_url=big_image_url+item+'_';};
     });  
     // big_image_url=big_image_url+'enl.'+alt[alt.length-1];
     big_image_url=item.bigurl; 
     var hugeimage=new Image();
     hugeimage.src=item.url;
     cc=hugeimage.width;
     dd=hugeimage.height;
     var rr=cc/dd; 
     
      var hh='90px';
   var ww='90px';
     if (( dd>90) & (cc>90)) 
       { if (dd>cc) { hh='90px';  ww=parseInt(parseInt(hh)*rr)+'px'; }
       else { ww='90px';  hh=parseInt(parseInt(ww)/rr)+'px'; };
       } 
     else if ( cc>90){ ww='90px';  hh=parseInt(parseInt(ww)/rr)+'px'; }
     else if (dd>90) { hh='90px';  ww=parseInt(parseInt(hh)*rr)+'px'; };
     if (parseInt(hh)<90) { style=' style="margin-top:'+((90-parseInt(hh))/2)+'px" ';} else {style='';};
   //  return '<img  title="'+item.title+'" '+style+' class="click_to_preview" id="load_big_img" src="' + item.url +'"  />';
   return '<table width="93px" height="93px"><tr><td valign="middle" align="center" style="vertical-align:middle;"><img  title="'+item.title+'" '+style+' class="click_to_preview" id="load_big_img" src="' + item.url + '"   alt="'+big_image_url+'"  /></td></tr></table>';
 
 };

               <div class="title">
                <h2>
                 <a href="https://trendybags.ru/blog/vebasist/2018/07/plyazhnye-sumki-2018-tendentsii-populyarnye-vidy-materialy-dekor/" title="Пляжные сумки 2018: тенденции, популярные виды, материалы, декор.">
                  Пляжные сумки 2018: тенденции, популярные виды, материалы, декор.
                 </a>
                </h2>
                <small>
                 3 июля 2018
                </small>
               </div>
               <div class="post">
                <p>
                 Сумка - неотъемлемый атрибут женского пляжного образа. Вместительная и стильная сумка пляжная поможет вам перенести все необходимые аксессуары для пляжа, гармонично дополнит ваш летний ансамбль и даже может послужить вам удобным лежаком или ковриком. В новом сезоне
                 <a href="https://trendybags.ru/category/novinki/">
                  пляжные сумки
                 </a>
                 2018 представлены большим разнообразием моделей и вариантов дизайна, поэтому подобрать идеальный аксессуар для себя сможет каждая модница.
                </p>
                <h4>
                 Виды пляжных сумок
                </h4>
                <p>
                 Среди всего представленного на рынке многообразия моделей пляжных сумок особенной популярностью пользуются обычные пляжные сумки-корзинки и рюкзаки женские, а также объемные сумки-мешки (торбы) и функциональные сумки-трансформеры.
                 <br/>
                 <br/>
                 <br/>
                 В 2018 году на пике популярности будут женственные сумки-корзинки. Эти аксессуары характеризуются простой трапециевидной формой с открытым верхом. Выполняются корзинки из любых поддающихся плетению материалов - соломки, ткани, шнуров, пластика и т. п.
                 <br/>
                 <br/>
                 <br/>
                 Рюкзак позволит вам освободить руки и удобно перенести все пляжные принадлежности. Обычно сумки-рюкзаки выполняются из мягких материалов, но при этом снабжаются жестким дном.
                 <br/>
                 <br/>
                 <br/>
                 Удобная сумка-торба традиционно имеет открытый верх. Форма ее может быть любой, края обычно скруглены. Вместительные мешкообразные сумки обычно имеют одно большое отделение и небольшой внутренний кармашек (обычный или непромокаемый). Выполняются подобные сумки из мягких материалов - денима, хлопка, кожзама, нейлона, льна и проч. Популярны также плетеные сумки пляжные и мешкообразные вязаные сумки: эти модели очень эффектно смотрятся в ансамбле с вязаным купальником.
                 <br/>
                 <br/>
                 <br/>
                 Сумки-трансформеры отличаются повышенной функциональностью: они послужат вам не только аксессуарами для переноса вещей, но и при необходимости превратятся в пляжные коврики, лежаки или матрасы. Выполняются подобные модели обычно из прочных водонепроницаемых материалов. Сумка-трансформер сделает ваш пляжный отдых 2018 максимально комфортным.
                </p>
                <h4>
                 Сумки пляжные 2018: расцветки, декор, стилевые решения
                </h4>
                <p>
                 Сумка для пляжа должна гармонировать с вашим пляжным образом по стилистике и расцветке. Среди популярных в теплом сезоне расцветок пляжных сумок стоит отметить оттенки воды и неба (бирюзовый, голубой, изумрудный и т. п.), солнечные тона (оранжевый, ярко-желтый), а также красный, розовый, коралловый, салатовый, зеленый и др. Отличным выбором станут сумки модные 2018 сочных и неоновых расцветок, которые не только станут ярким акцентом летнего ансамбля, но и определят ваше пляжное место. Актуальна также пастельная гамма и чистый белый цвет, который служит отличным фоном для ярких принтов и декора.
                 <br/>
                 <br/>
                 <br/>
                 Наряду с аксессуарами однотонных расцветок огромной популярностью пользуются сумки с принтами. В 2018 году стоит купить сумки пляжные с изображениями экзотических пейзажей, цветов, эффектными 3D-принтами, геометрическими и абстрактными рисунками. Очень стильно будет смотреться сумка с полосатым принтом. Причем расположение, цвет и ширина полосок могут быть любыми.
                 <br/>
                 <br/>
                 <br/>
                 Не менее разнообразным будет декор сумок 2018. Так, в тренде будут
                 <a href="https://trendybags.ru/category/vse-sumki/?price_min=&amp;price_max=&amp;brend%5B%5D=139&amp;brend%5B%5D=140&amp;sort=&amp;order=">
                  летние сумки
                 </a>
                 с милыми рюшами, кружевными элементами, объемными цветами, аппликациями и вышивкой, а также декором из тесьмы, бусин, ракушек, пайеток, деревянных элементов и проч.
                 <br/>
                 <br/>
                 <br/>
                 По стилистике пляжные сумки должны гармонировать со всеми элементами пляжного ансамбля. Среди модных в 2018 году стилевых решений сумок для пляжа стоит отметить морскую тематику, этно-стиль и кэжуал, спортивный стиль и эксклюзивный hand-made.
                 <br/>
                 <br/>
                 <br/>
                 Правильно подобранная по дизайну и форме сумка подчеркнет достоинства вашей фигуры и придаст индивидуальность вашему пляжному образу. Выбирайте модные пляжные сумки 2018 и покоряйте окружающих стильными летними ансамблями!
                </p>
