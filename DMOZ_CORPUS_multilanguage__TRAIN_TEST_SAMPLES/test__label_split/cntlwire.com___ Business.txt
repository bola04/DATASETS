     jQuery(document).ready(function($) {
   $.fn.ApplySelect = function(data) {
     var select_name = $(this).attr("name");
     
     if( select_name == "brand" ) {
       var html = "<option selected='selected' value=''>Brand</option>";  
     } else if( select_name == "industry" ) {
       var html = "<option selected='selected' value=''>Industry</option>";  
     } else if( select_name == "model" ) {
       var html = "<option selected='selected' value=''>Model</option>";  
     }
 
     if( data != null ) {
       $.each(data, function(key, value) {
         html += "<option value='"+value[0]+"'>"+value[1]+"</option>"
       });
     }
 
     $(this).html(html);
     $(this).trigger('change');
   }
 
   $(document).on("change",'select:not([name="model"])',function() {
     var elem = $(this);
     var select_name = $(this).attr("name");
     var select_value = $(this).val();
 
     var brand_val = elem.parent().parent().find('select[name="brand"]').val();
     var industry_val = elem.parent().parent().find('select[name="industry"]').val();
     var model_val = elem.parent().parent().find('select[name="model"]').val();
 
     $.post( "http://cntlwire.com/wp-admin/admin-ajax.php", {
         'action': 'product_selector',
         'brand': brand_val,  
         'industry': industry_val
         
     }, 
     function(resp){
       if( select_name == "brand" ) {
           elem.parent().parent().find('select[name="industry"]').ApplySelect(resp['industry']);  
           elem.parent().parent().find('select[name="model"]').ApplySelect(resp['model']);
 
 
       } else if( select_name == "industry" ) {
           elem.parent().parent().find('select[name="model"]').ApplySelect(resp['model']);
          
       }
 
     });
   });
 });
