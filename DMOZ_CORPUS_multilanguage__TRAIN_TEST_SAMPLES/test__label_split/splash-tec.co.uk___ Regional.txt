 					$(document).ready(function(){

 						$('.newsletter--link').on("click", function(e){
 							e.preventDefault();
 							nsltr.fadeIn();
 							nsltrolay.fadeIn();
 						});
 						
 						$(nsltrolay).on("click", function(){

 					function validateNsltr(field_id_array){
 						var nsltr = 0;
 						var validateArray = field_id_array.split(',');
 						jQuery(validateArray).each(function(index, value){
 							if(!jQuery('#' + value).val()){
 								nsltr = 1;
 							}
 						});
 						if(nsltr==1){
 							$('.error-msg').fadeIn();
 							$('.error-msg').delay(3000).fadeOut(1000);
 							return false;
 						} else {
 							return true;
 						}
 					}
 					$(function () {
 						$('#nsltr').submit(function (e) {
 							e.preventDefault();
 							$.getJSON(
 							this.action + "?callback=?",
 							$(this).serialize(),
 							function (data) {
 								if (data.Status === 400) {
 									$('.error-msg').html(data.Message);
 									$('.error-msg').fadeIn();
 									$('.error-msg').delay(5000).fadeOut(1000);
 								} else {
 									$('.msg-sent').html(data.Message);
 									$('.msg-sent').fadeIn();
 								}
 							});
 						});
 					});
