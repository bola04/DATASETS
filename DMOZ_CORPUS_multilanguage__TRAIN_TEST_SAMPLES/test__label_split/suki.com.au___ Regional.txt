     var screen_x = 0;
   var screen_y = 0;
   var left = 0;
   var limit_y = 50;
   var limit_x = 80;
   var bottom = 0;
   $(function(){
     $(window).resize();
     if($(window).width() < 768){
       $('.content').css('height', 'auto');
       $('.content').css('min-height', $(window).height()+'px');
       $('.content').css('height', $('.content').outerHeight()+'px');
       $('.content').css('min-height', $('.content').outerHeight()+'px');
     }
   });
   $(window).load(function(){
     $(window).resize();
   });
   $(window).resize(function(){
     init();
   });
   
   $('#home').mousemove(function( event ) {
     if(event.pageX > screen_x/2){
       var ratio = (screen_x/2 - event.pageX)/screen_x/2;
       var new_pos = left + ratio*limit_x;
       $('#model').css('left', new_pos+'px');
     }else{
       var ratio = (event.pageX - screen_x/2)/screen_x/2;
       var new_pos = left - ratio*limit_x;
       $('#model').css('left', new_pos+'px');
     }
     if(event.pageY > screen_y/2){
       var ratio = (event.pageY - screen_y/2)/screen_y/2;
       var new_pos = bottom + ratio*limit_y;
       $('#model').css('bottom', new_pos+'px');
     }else{
       var ratio = (screen_y/2 - event.pageY)/screen_y/2;
       var new_pos = bottom - ratio*limit_y;
       $('#model').css('bottom', new_pos+'px');
     }
     var msg = 'Handler for .mousemove() called at ';
     msg += event.pageX + ", " + event.pageY;
     console.log(msg);
   });
   
   function init(){
     screen_x = $(window).width();
     screen_y = $(window).height();
     left = (screen_x - $('#model').width())/2;
     bottom = 0 - limit_y;
     limit_x = limit_y*($(window).width()/$(window).height());
     $('#model').removeClass('active');
     $('#model').css('left', left+'px');
     $('#model').css('bottom', bottom+'px');
     $('#model').addClass('active');
   }
    </script>
