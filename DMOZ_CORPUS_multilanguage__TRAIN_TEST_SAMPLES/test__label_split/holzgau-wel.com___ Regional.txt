 function CheckScrollPosX() {

 	hoehe = $(window).height();
 	windowpos = 150;
 
 	if(height > windowpos) {
 		$('.Header').addClass('HeaderF');	
 	} else {
 		$('.Header').removeClass('HeaderF');
 	}
 
 	if(height>10) {
 		if(NavigationAktiv == true) {
 			$('#nav-icon3').toggleClass('open');
 			$('.PhotosSlider').removeClass('PhotosNav');
 			$('.Navigation').removeClass('NavigationF');
 			NavigationAktiv = false;
 			$('.NavIcon').addClass('NavIconH');
 			setTimeout(function() {
 				$('.NavIcon').attr('src','img/nav.svg');
 				$('.NavIcon').removeClass('NavIconH');
 				$('.Menu').removeClass('MenuH');
 			},400);
 		}
 	}
 	
 	if(height > hoehe) {
 			$('.Anfragebox').addClass('AnfrageboxF');			
 			$('#ui-datepicker-div').addClass('ui-datepicker-divF');
 			
 		
 	} else {
 			$('.Anfragebox').removeClass('AnfrageboxF');
 			$('#ui-datepicker-div').removeClass('ui-datepicker-divF');
 		
 	}
 	
 }
