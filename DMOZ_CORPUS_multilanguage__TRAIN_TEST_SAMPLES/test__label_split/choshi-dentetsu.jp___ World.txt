 function get_vt_spot(idx){
 
     var red_marker = "/images/marker_red.png";
 
     var latlng = new google.maps.LatLng(baseLat, baseLng);
     var _url = 'https://mw2pyafwl5.bizmw.com/api/plan/vspot/'+idx;
     var opts_virtual = {zoom: 12, center: latlng, mapTypeId: google.maps.MapTypeId.ROADMAP};
     var virtual_map = new google.maps.Map(document.getElementById("virtual_map_canvas"), opts_virtual);
     
     $.getJSON(_url, function(json){
         if (json != null && json.length > 0){
 
             for(var i = 0; i < json.length; i++){
                 var lat = json[i].latitude;
                 var lng = json[i].longitude;
                 if ( (lat == null || lat == "") && (lng == null || lng == "") ){
                     continue;
                 }
                 
                 var latlng_sp = new google.maps.LatLng(lat, lng);
                 new google.maps.Marker({position: latlng_sp, map: virtual_map, icon: red_marker});                 
             }
         }
     });
 }
