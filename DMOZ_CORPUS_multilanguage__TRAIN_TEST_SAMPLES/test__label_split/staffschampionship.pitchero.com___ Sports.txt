 {

 /**
  * Find Ad DOM elements for the specified `slotId`, if multiple
  * slots are found, work out which slot should be populated.
  * Creates or inserts adDiv with specified `slotId`;
  * Elements with the `ad-lazy-load` attribute are ignored.
  * Returns values object: id (`slotId`), slot (gpt.Slot), refreshAds (boolean),
  */
 function adjustAdDivPosition(slotId)
 {
   var values = {
     id: slotId,
     slot: getSlotById(slotId)
   }
 
   var adType = slotId.replace('div-', '');
 
   var slot = findElementsByAdType(adType);
 
   if ( slot.length > 1 ) {
     var windowWidth = $(window).width()
       , deviceType = getDeviceTypeFromScreenWidth(windowWidth);
 
     slot = findElementByAdTypeAndDeviceType(adType, deviceType);
   }
 
   if ( slot.data('ad-lazy-load') && ! slot.data('ad-lazy-loaded') ) {
     return null;
   }
 
   if ( checkIdAlreadyExistsInElement(slot, slotId) ) {
     return values;
   }
 
   var adDiv = createAdDiv(slotId);
 
   if ( findElementById(slotId).length > 0 ) {
     adDiv = detachAdDiv(findElementById(slotId));
   }
 
   appendAdDivToSlot(slot, adDiv);
   values.refreshAds = true;
 
   return values;
 }

 //Get Browser Window Dimensions -- Used in parseSizeMappings
 function getWindowDimensions() {
 	var width = window.innerWidth ||
 		document.documentElement.clientWidth ||
 		document.body.clientWidth;
 	var height = window.innerHeight ||
 		document.documentElement.clientHeight ||
 		document.body.clientHeight;
 	return [width, height];
 }
 
 /**
  * Parse mapping and create an array of sizes available at current screen resolution
  */
 function parseSizeMappings(sizeMappings) {
     try {
         // get current window dimensions
         var sd = getWindowDimensions();
         var validMappings = sizeMappings.filter(function (m) {
             return m[0][0] <= sd[0] && m[0][1] <= sd[1]
         });
         // return the leftmost mapping's sizes or an empty array
         return validMappings.length > 0 ? validMappings[0][1] : [];
     } catch (e) {
         // fallback to last size mapping supplied
         return sizeMappings[sizeMappings.length - 1][1];
     }
 }
