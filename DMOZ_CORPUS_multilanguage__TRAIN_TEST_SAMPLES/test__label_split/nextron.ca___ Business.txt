    var layer_o = new Array(100,0,0,0,0);
 
 var current = 0;
 var fadeoutpanel = 0;
 var fadeinpanel = 1;
 
 var fade_time;
 var fadein_time;
 var auto_time;
 
 function startIt() {
 	auto_time = setTimeout("startFading();",7000);
 }
 	
 function startFading() {
 	if (current == 4) {
 		fadeoutpanel = current;
 		fadeinpanel = 0;
 		//current = 0;
 	}
 	else {
 		fadeoutpanel = current;
 		fadeinpanel = current + 1;
 	}
 	//console.log("hello::current=" + current + "::fadein=" + fadeinpanel + "::fadout=" + fadeoutpanel);
 	fadeInPanel();
 	fadeOutPanel();
 	auto_time = setTimeout("startFading();",7000);
 	
 }
 
 function fadeOutPanel() {
 	//fade out first layer
 	var layername = "bigPic" + fadeoutpanel;
 	var cur_layer = getLayer(layername);
 	layer_o[fadeoutpanel] = layer_o[fadeoutpanel] - 10;
 	var d = layer_o[fadeoutpanel] / 100;
 	cur_layer.style.filter = 'alpha(opacity=' + layer_o[fadeoutpanel] + ')';
 	cur_layer.style.opacity = d;
 	//console.log(layername + ":" + d);
 	//check if done;
 	if (layer_o[fadeoutpanel] == 0) {
 		
 		stopFade();
 		//fadeInPanel();
 	}
 	else {
 		fade_time = setTimeout("fadeOutPanel();",20);
 	}
 }
 
 function fadeInPanel() {
 	//new layer fade in
 	var layername2 = "bigPic" + fadeinpanel;
 	var new_layer = getLayer(layername2);
 	layer_o[fadeinpanel] = layer_o[fadeinpanel] + 10;
 	var d2 = layer_o[fadeinpanel] / 100;
 	new_layer.style.filter = 'alpha(opacity=' + layer_o[fadeinpanel] + ')';
 	new_layer.style.opacity = d2;
 	//console.log(layername2 + ":" + d2);
 	if (layer_o[fadeinpanel] == 100) {
 		stopFadeIn();
 	}
 	else {
 		fadein_time = setTimeout("fadeInPanel();",20);
 	}
 }
 
 function stopFade() {
 	clearTimeout(fade_time);
 	if (current == 4) {
 		current = 0;
 	}
 	else {
 		current = current + 1;
 	}
 	//console.log("I stopped the fadeout");
 }
 
 function stopFadeIn() {
 	clearTimeout(fadein_time);
 	//console.log("I stopped the fadein");
 }
 
 function redirect() {
 	if (current == 0) {
 		window.location.href = "warranty.jsp";
 	}
 	else if (current == 1) {
 		window.location.href = "mastertrace_features.jsp";
 	}
 	else if (current == 2) {
 		window.location.href = "contact_us.jsp";
 	}
 	else if (current == 3) {
 		window.location.href = "contact_us.jsp";
 	}
 	else if (current == 4) {
 		window.location.href = "skid_packages.jsp";
 	}
 }
