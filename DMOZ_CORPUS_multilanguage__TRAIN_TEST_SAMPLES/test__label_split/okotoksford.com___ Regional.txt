         jQuery(document).ready(function(){
 
 		var windowWidth = jQuery(window).width();
 		jQuery(window).resize(function(){
 			windowWidth = jQuery(window).width();
 		});
 
 		// jQuery('.items-container[data-container*="cars"]').addClass('active');
 
 		jQuery('.mk-container a').click(function (e) {
 			var target = jQuery(this).attr('data-target');
 			jQuery('.mk-container a').removeClass('active');
 			jQuery(this).addClass('active');
 			jQuery('.slider-container').hide();
 			jQuery('.slider-container[data-container="'+ target +'"]').show();
 			if( windowWidth > 600 ) {
 				jQuery('.slider-container[data-container="' + target + '"]').find('.bs-option:first-child').click();
 			}
 			return false;
 		});
 
 		jQuery('.bs-select').click(function(e){
 			jQuery(this).toggleClass('down');
 			jQuery(this).next('i').toggleClass('rotate');
 
 			if( jQuery(e.target).hasClass('bs-option') && !jQuery(e.target).hasClass('active')){
 
 				if( windowWidth < 600 ){
 					jQuery(e.target).parents().closest('.bs-select').prepend(e.target);
 				}
 
 				jQuery(e.target).parents('.bs-select').children('.bs-option').removeClass('active');
 				jQuery(e.target).addClass('active');
 				var target = jQuery(e.target).attr('data-area');
 				jQuery(e.target).parents('.slider-container').children('.items-container').removeClass('active');
 				jQuery(e.target).parents('.slider-container').children('.items-container[data-container*="'+target+'"]').addClass('active');
 			}
 		});
 
 		if( windowWidth > 600){
 			jQuery('.bs-select .bs-option:first-child').first().click();
 		}
 
 		// Function for Owl Carousel and Syncing 2 owl carousels
 		var sync1 = jQuery(".opt-4 .model-container");
 		var sync2 = jQuery(".opt-4 .model-pagination");
 
 		sync1.owlCarousel({
 			items : 1,
 			singleItem: true,
 			slideSpeed : 1000,
 			navigation: true,
 			navigationText: ['<i class="icon-navigation-icons-12"></i>', '<i class="icon-navigation-icons-11"></i>'],
 			pagination : false,
 			afterAction : syncPosition,
 			responsiveRefreshRate : 200
 		});
 
 		sync2.owlCarousel({
 			itemsDesktop : [1199,8],
 			itemsDesktopSmall : [979,5],
 			itemsTablet : [1030,6],
 			itemsMobile : [768,4],
 			pagination : false,
 			navigation: true,
 			navigationText: ['<i class="icon-navigation-icons-12"></i>', '<i class="icon-navigation-icons-11"></i>'],
 			responsiveRefreshRate : 100,
 			afterInit : function(el){
 				el.find(".owl-item").eq(0).addClass("synced");
 			}
 		});
 
 		function syncPosition(el){
 			var current = this.currentItem;
 			// console.log(current);
 			jQuery(el)
 				.next()
 				.find(".owl-item")
 				.removeClass("synced")
 				.eq(current)
 				.addClass("synced");
 			if(jQuery(sync2).data("owlCarousel") !== undefined){
 				center(el, current);
 			}
 		}
 
 		jQuery(sync2).on("click", ".owl-item", function(e){
 			e.preventDefault();
 			var number = jQuery(this).data("owlItem");
 			jQuery(this).parents().closest(sync2).prev().trigger("owl.goTo",number);
 			jQuery(this).siblings().removeClass('synced');
 			jQuery(this).addClass('synced');
 		});
