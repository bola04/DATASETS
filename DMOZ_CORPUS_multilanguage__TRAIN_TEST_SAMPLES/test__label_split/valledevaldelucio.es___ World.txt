 function anyadirHorario() {
 	var dia = jQuery('#recursoDia').val();
 	var horaInicio = jQuery('#recursoHoraInicio').val();
 	var horaFin = jQuery('#recursoHoraFin').val();
 	if (dia == '' || horaInicio == '' || horaFin == '') {
 		alert('Debe rellenar los campos D\u00EDa, Hora inicio y Hora fin');
 	} else {
 		var horario = dia + " " + horaInicio  + "-" + horaFin  + ";";
 		//jQuery('#edit-field-recurso-horario-und-0-value').append(horario + '\n');
 		var horarioTexto = jQuery('<div>' + jQuery('#recursoDia option:selected').text() + ' de ' + horaInicio + ' a ' + horaFin + ' <a href="#" title="Borrar horario" onclick="return borrarHorario(jQuery(this));">Borrar horario</a></div>');
 		horarioTexto.data('horario', horario);
 		jQuery('#diasSeleccionados').append(horarioTexto);
 		reconstruirCadenaHorario();
 
 		jQuery('#recursoHoraInicio').timepicker('setDate', new Date(2000,1,1,0,0,0,0));

 }
 function reconstruirCadenaHorario() {
 	var horario = '';
 	jQuery('#diasSeleccionados div').each(function(){
 		horario += jQuery(this).data('horario') + "\n";
 	});
 	jQuery('#edit-field-recurso-horario-und-0-value').text('').text(horario.trim());
 }
 function construirDiasSeleccionados() {
 	var horarioTexto = '<div>';
 	var lineas = jQuery('#edit-field-recurso-horario-und-0-value').val().split("\n");
 	jQuery.each(lineas, function(key, linea) {
 		if (linea != '') {
 			var dia = linea.charAt(0);
 			var horas = linea.substr(2,linea.length).split('-');
 			var textoDia = '';
 
 			if (dia == 'L') {
 				textoDia = 'Lunes';
 			} else if (dia == 'M') {
 				textoDia = 'Martes';
 			} else if (dia == 'X') {
 				textoDia = 'Mi\u00E9rcoles';
 			} else if (dia == 'J') {
 				textoDia = 'Jueves';
 			} else if (dia == 'V') {
 				textoDia = 'Viernes';
 			} else if (dia == 'S') {
 				textoDia = 'S\u00E1bado';
 			} else if (dia == 'D') {
 				textoDia = 'Domingo';
 			}
 			
 
 			var horario = dia + " " + horas[0]  + "-" + horas[1];
 			var horarioTexto = jQuery('<div>' + textoDia + ' de ' + horas[0] + ' a ' + horas[1].slice(0,-1) + ' <a href="#" title="Borrar horario" onclick="return borrarHorario(jQuery(this))">Borrar horario</a></div>');
 			horarioTexto.data('horario', horario);
 			jQuery('#diasSeleccionados').append(horarioTexto);
 		}
 	});
 }
 function borrarHorario(objeto) {
 	jQuery(objeto).parent().remove();
 	reconstruirCadenaHorario();
 	return false;
 }
