 		$('#am-portal>div').width(($this.maxSlides * $this.slideWidth) + (max * 10));
 		$('#am-feat-props .next-slide').on('click', function() {

 		});
 		$('#am-feat-props .prev-slide').on('click', function() {
 			$this.prevSlide();
 
 		});
 	};
 	AMFeatPropsSliderSlider.prototype.nextSlide = function() {
 		var $this = this;
 		if ($this.current == 1)
 			return;
 		$('#am-portal>div').animate({right: '-='+(this.slideWidth + $this.shift)});
 		$this.current--;
 	};
 	AMFeatPropsSliderSlider.prototype.prevSlide = function() {
 		var $this = this;
 		if ($this.current == $this.maxSlides)
 			return;
 		$('#am-portal>div').animate({right: '+='+(this.slideWidth + $this.shift)});
 		$this.current++;
 	};
