    var checkOpacity = function() {
         $('.main_navigation_desktop.is-not-mobile').removeClass('hover');
         var bodyScroll = $(window).scrollTop(),
             mobileHeader = $('.is-mobile .menu'),
             logo = $('.is-mobile .menu .logo'),
             height = 10,
             mainNav = $('.main_navigation_desktop.is-not-mobile');
             // menu = $('.menu-icon.first');
 
         if (bodyScroll < height) {
             mainNav.removeClass('on-scroll');
             mobileHeader.addClass('-transparent').removeClass('-opaque');
         } else {
             if (mainNav.hasClass('on-scroll') == false) {
                 mainNav.addClass('on-scroll');
             }
         }
         // console.log(ScrollQuantity());
         if (bodyScroll > height) {
             mobileHeader.addClass('-opaque').removeClass('-transparent');
             logo.attr('src', '/widgets/images/logos/ud_logo.svg');
             if (ScrollQuantity() == "DOWN") {
                 mobileHeader.removeClass("-menu-visible");
                 mobileHeader.addClass("-menu-hidden");
             } else {
                 mobileHeader.removeClass("-menu-hidden");
                 mobileHeader.addClass("-menu-visible");
             }
         } else {
             logo.attr('src', '/widgets/images/logos/ud_logo_white.svg');
             mobileHeader.removeClass("-menu-hidden");
             mobileHeader.addClass("-menu-visible");
         }
     }
 
     var mousewheelevt = (/Firefox/i.test(navigator.userAgent)) ? "DOMMouseScroll" : "mousewheel" //FF doesn't recognize mousewheel as of FF3.x
     $('body').bind(mousewheelevt, checkOpacity);
     $(document).ready(function() {
         $(document).bind('scroll', checkOpacity);
         checkOpacity();
 
         var mobileMenu = $('.menu-fixed');
 
         setTimeout(function() {
             checkOpacity();
         }, 2000);
 
         $('.menu-icon').click(function() {
             $('body').css('overflow', 'hidden');

                 formContact = $('.form-contact'),
                 formProject = $('.form-project'),
                 formCandidate = $('.form-candidate'),
                 iconContact = $('.form-icon-contact'),
                 iconProject = $('.form-icon-project'),
                 iconCandidate = $('.form-icon-candidature');
 
             switch (key) {
                 case 'project':
                     if (iconProject.hasClass('active') === false) {
                         iconProject.addClass('active');
                     }
                     iconContact.removeClass('active');
                     iconCandidate.removeClass('active');
                     formContact.hide();
                     formProject.show();
                     formCandidate.hide();
                     break;
                 case 'candidate':
                     if (iconCandidate.hasClass('active') === false) {
                         iconCandidate.addClass('active');
                     }
                     iconContact.removeClass('active');
                     iconProject.removeClass('active');
                     formContact.hide();
                     formProject.hide();
                     formCandidate.show();
                     break;
                 default:
                     if (iconContact.hasClass('active') === false) {
                         iconContact.addClass('active');
                     }
                     iconProject.removeClass('active');
                     iconCandidate.removeClass('active');
                     formContact.show();
                     formProject.hide();
                     formCandidate.hide();
                     break;
             }
         });
 
 
         $('.contact-form .confirm').click(function() {
             var form = $(this).parent().parent(),
                 formType = form.parent()[0].className,
                 btn = $(this);
 
             if (btn.hasClass('loading'))
                 return;
 
             btn.addClass('loading');
 
             function nom_fonction_retour(texte_recu) {
                 // Du code pour gérer le retour de l'appel AJAX.
                 btn.removeClass('loading');
 
                 texte_recu = texte_recu.trim();
                 if (texte_recu == "") {
                     toastr["success"]("Message envoyé");
                     form[0].name.value = "";
                     form[0].firstname.value = "";
                     form[0].mail.value = "";
                     form[0].message.value = "";
                     try {
                         form[0].tel.value = "";
                     } catch (err) {}
                     try {
                         form[0].portfolio.value = "";
                     } catch (err) {}
                 } else {
                     toastr["error"](texte_recu);
                 }
             }
 
             function execute_form() {
                 switch (formType) {
                     case "form-contact":
                         var name = form[0].name.value,
                             firstName = form[0].firstname.value,
                             mail = form[0].mail.value,
                             tel = form[0].tel.value,
                             message = form[0].message.value,
                             lang = "fr",
                             type = "contact";
 
                         $.post(
                             '/form-contact.php', // Le fichier cible côté serveur.
                             {
                                 name: name,
                                 firstName: firstName,
                                 mail: mail,
                                 tel: tel,
                                 message: message,
                                 lang: lang,
                                 type: type
                             },
                             nom_fonction_retour,
                             'text'
                         );
                         break;
                     case "form-candidate":
                         var name = form[0].name.value,
                             firstName = form[0].firstname.value,
                             mail = form[0].mail.value,
                             portfolio = form[0].portfolio.value,
                             message = form[0].message.value,
                             lang = "fr",
                             type = "candidature";
 
                         $.post(
                             '/form-contact.php', // Le fichier cible côté serveur.
                             {
                                 name: name,
                                 firstName: firstName,
                                 mail: mail,
                                 portfolio: portfolio,
                                 tel: tel,
                                 message: message,
                                 lang: lang,
                                 type: type
                             },
                             nom_fonction_retour,
                             'text'
                         );
                         break;
                 }
             }
 
             execute_form();
