                 wrap.removeClass('cbp-l-filters-dropdownWrap-open');
             }
         });
 
         filtersCallback = function(me) {
             wrap.find('.cbp-filter-item').removeClass('cbp-filter-item-active');
             wrap.find('.cbp-l-filters-dropdownHeader').text(me.text());
