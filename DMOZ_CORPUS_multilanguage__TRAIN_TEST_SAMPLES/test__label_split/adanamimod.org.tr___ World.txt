    $(document).ready(function () {

                 d = date.getDate(),
                 m = date.getMonth(),
                 y = date.getFullYear(),
                 started,
                 categoryClass;
 
             var initialLocaleCode = 'tr';
 
             $('#calendar').fullCalendar({
                 header: {
                     left: 'prev,next today',
                     center: 'title',
                     right: 'month,agendaWeek,agendaDay,listMonth'
                 },
                 defaultDate: date,
                 locale: initialLocaleCode,
                 buttonIcons: false, // show the prev/next text
                 weekNumbers: true,
                 navLinks: true, // can click day/week names to navigate views

                 eventLimit: true, // allow "more" link when too many events

                     //alert(calEvent.resim);

                     $('#txtEtkinlikBasTarih').val(calEvent.start1);
                     $('#txtEtkinlikBitTarih').val(calEvent.end1);
                     $('#txtEtkinlikHeadBaslik').html(calEvent.title);
                     $('#txtEtkinlikBaslik').val(calEvent.title);
                     $('#txtEtkinlikAciklama').html(calEvent.aciklama);
                     jQuery('#myModal').modal();
                 },
                 events: "/Handler1.ashx"
             });
 
             // build the locale selector's options
             $.each($.fullCalendar.locales, function (localeCode) {
                 $('#locale-selector').append(
                     $('<option/>')
                         .attr('value', localeCode)
                         .prop('selected', localeCode == initialLocaleCode)
                         .text(localeCode)
                 );
             });
 
             // when the selected option changes, dynamically change the calendar option
             $('#locale-selector').on('change', function () {
                 if (this.value) {
                     $('#calendar').fullCalendar('option', 'locale', this.value);
                 }
             });
 
         });
