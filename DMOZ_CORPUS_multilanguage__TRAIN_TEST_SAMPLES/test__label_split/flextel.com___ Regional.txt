 function getDocumentHeight(objectHandle) {
 	var obj = objectHandle;
 	var oDoc = (obj.contentWindow || obj.contentDocument || obj.document);
 	if (oDoc.document) oDoc = oDoc.document;
 	var D = oDoc;
 	
 	var h = Math.max(
 	Math.max(D.body.scrollHeight, D.documentElement.scrollHeight),
 	Math.max(D.body.offsetHeight, D.documentElement.offsetHeight),
 	Math.max(D.body.clientHeight, D.documentElement.clientHeight)
 	);
 	return h;

 		function get_frontpagesearch_code_service(flextel_number) {
 			for (var i=flextel_number.length; i>=0; i--) {
 				if (flextel_number.indexOf('01') !=0 && flextel_number.indexOf('02') !=0) {
 				  if (typeof(CodeKeywords[flextel_number.substr(0,i)]) == 'string') return flextel_number.substr(0,i)+'-'+CodeKeywords[flextel_number.substr(0,i)];
 				}
 				else {
 				  if (typeof(flextel_labels[flextel_number.substr(0,i)]) == 'string') return flextel_number.substr(0,i)+'-'+flextel_labels[flextel_number.substr(0,i)]+'-Local-Telephone';
 				}
 			}
 		return 'Unknown-Code';
 		}
