 		if ($(this).scrollTop() > 100) {
 			$('.scrollTop').fadeIn();
 		} else {
 			$('.scrollTop').fadeOut();
 		}
 	});
 	$('.scrollTop').click(function(){
 		$('html, body').animate({scrollTop : 0},1000);
 		return false;
 	});
 	
 });
