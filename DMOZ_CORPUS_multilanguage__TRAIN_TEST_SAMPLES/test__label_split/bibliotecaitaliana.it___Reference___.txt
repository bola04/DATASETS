    <script>
     (function($) {
     $(document).ready(
       function(){
 
         var url = "/indice/autori_autocomplete";
         $("#author").autocomplete(url, {
           formatItem: function(row, i, max) {
             r = row[0] + " (" + row[1] + ")";
       			return r;
       		},
           formatResult: function(row, i, max) {
       			return row[0];
       		},    		
           minChars: "2", width: "250px", max: "50", scrollHeight: "250px"
         }).result(function(event, data) {
            $("#author_id").get(0).value =  !data ? "0" :  data[2];
            $("#submit_autore").focus();
         });
         $("#author").focus(function(){
           $("#title").val("");
           if(this.value == this.defaultValue)
           {
             this.select();
           }
         });
         // risolto bug #27 su mancato funzionamento tasto invio in IE
         $('#author').keypress(function(event) {
           if (event.keyCode == '13') {
             $("#submit_autore").focus();
           }
         });
 
 
         var title_url = "/indice/opere_autocomplete";
         $("#title").autocomplete(title_url, {
           formatItem: function(row, i, max) {
             r = row[0].substring(0, 40);
             if (r.length > 40) r = '<span title="' + row[0] + '">' + r + '...</span>';
       			return r;
       		},
           formatResult: function(row, i, max) {
       			return row[0];
       		},    		
           minChars: "2", width: "280px", max: "50", scrollHeight: "250px"
         }).result(function(event, data) {
            $("#book_id").get(0).value =  !data ? "0" :  data[1];
            $("#submit_titolo").focus();
         });
         $("#title").focus(function(){
           $("#author").val("");
           if(this.value == this.defaultValue)
           {
             this.select();
           }
         });
         // risolto bug #27 su mancato funzionamento tasto invio in IE
         $('#title').keypress(function(event) {
           if (event.keyCode == '13') {
             $("#submit_titolo").focus();
           }
         });
         
 
 
         $("ul.first-lev li span.legend").click(function(){
           $(this).parents('li').find('ul').slideToggle('fast');
           var icona = $(this).children().attr("src");
           if(icona == "/images/bullet.png"){ 
           	$(this).children().attr("src", "/images/bullet_minus.png");
       	}else{
           $(this).children().attr("src", "/images/bullet.png");
           }
         });
         
 
         function hasPlaceholderSupport() {
           var i = document.createElement('input');
           return 'placeholder' in i;
         }
         if(!hasPlaceholderSupport()){
           $("#facets").placeholder();
           $('input[autofocus=true]').focus();
         }
 
       });
   })(jQuery);

          <p>
           <strong>
            Biblioteca Italiana
           </strong>
           è una biblioteca digitale di testi rappresentativi della tradizione culturale e letteraria italiana dal Medioevo al Novecento. Il portale si articola in cinque sezioni:
           <strong>
            Bibit,
           </strong>
           più di 1600 testi consultabili e liberamente scaricabili, codificati in XML-TEI, sono in edizione integrale e si fondano sulle più autorevoli edizioni di riferimento;
           <strong>
            Scrittori d’Italia
           </strong>
           , riproduzione in formato digitale dell'omonima collana, fondata nel 1910 dall'editore Laterza di Bari sotto la direzione di Benedetto Croce;
           <strong>
            Incunaboli volgari
           </strong>
           , la digitalizzazione più estesa possibile degli incunaboli volgari esistenti nelle biblioteche italiane e straniere;
           <strong>
            Collezioni speciali
           </strong>
           , progetto scientifico di acquisizione della documentazione originaria dei testi del Cortegiano di Castiglione, e dell’Orlando Furioso di Ariosto;
           <strong>
            Risorse
           </strong>
           , numerosi progetti digitali interrogabili come risorsa esterna
