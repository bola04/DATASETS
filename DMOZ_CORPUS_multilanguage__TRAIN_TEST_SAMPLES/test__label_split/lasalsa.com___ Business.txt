    var page = jQuery('#page-wrapper, #bottom');
     var header = jQuery('header');
     var mobile_nav = jQuery('.mobile-nav');
     var move = 200;
     var duration = 50;
 
     jQuery(document).ready(function () {
         //style current page item in the nav
         if (location.pathname.split("/")[1].length > 0){$('nav a[href^="/' + location.pathname.split("/")[1] + '"]').closest('li').addClass('current_page_item'); }
         //init mobile nav functionality
         jQuery('.mobile-nav-click').click(function () {
             if (page.hasClass('show-nav')) {
                 page.animate({ left: "+=" + move }, duration);
                 header.animate({ left: "+=" + move }, duration);
                 mobile_nav.animate({ right: "-=" + move }, duration, function () { mobile_nav.removeClass('show-nav'); });
             } else {
                 jQuery(window).scrollTop(0);
                 mobile_nav.addClass('show-nav');
                 page.animate({ left: "-=" + move }, duration);
                 header.animate({ left: "-=" + move }, duration);
                 mobile_nav.animate({ right: "+=" + move }, duration);
             }
             page.toggleClass('show-nav');
         });
         jQuery('.mobile-dropdown').change(function () {
             var go_to = jQuery(this).find('option:selected').val();
             if (go_to) { window.location = go_to; }
         });
 });
   </script>
