 /*
 	Init slideshow.
 	elem: slideshow DOM element
 */
 function animate_slideshow (elem) {
 
 	if (!elem.hasAttribute ("morph_time")) { elem.setAttribute ("morph_time", "slow"); }
 	if (!elem.hasAttribute ("slide_time")) { elem.setAttribute ("slide_time", 5000); }
 	if (!elem.hasAttribute ("curr_img_index")) { elem.setAttribute ("curr_img_index", 0); }
 	if (!elem.hasAttribute ("curr_div_index")) { elem.setAttribute ("curr_div_index", 0); }
 	if (!elem.hasAttribute ("is_paused")) { elem.setAttribute ("is_paused", 0); }
 
 	{
 		var img_list = $("img.slide", $(elem));
 		var div_list = $("div.slideContent>div", $(elem));
 		var morph_time = elem.getAttribute ("morph_time");
 		var slide_time = Number (elem.getAttribute ("slide_time"));
 		var img_curr = Number (elem.getAttribute ("curr_img_index"));
 		var div_curr = Number (elem.getAttribute ("curr_div_index"));
 		var nImgs = img_list.length;
 		var nDivs = div_list.length;
 
 		var anim = function() {
 
 			if (Number (elem.getAttribute ("is_paused")) > 0) { return; }
 
 			img_curr = Number (elem.getAttribute ("curr_img_index"));
 			div_curr = Number (elem.getAttribute ("curr_div_index"));
 			nImgs = img_list.length;
 			nDivs = div_list.length;
 
 			// Slowly hide previous img and div switch to the next
 			if (nImgs > 0) { $(img_list [img_curr]).fadeOut (morph_time); img_curr = (img_curr + 1) % nImgs; elem.setAttribute ("curr_img_index", img_curr); $(img_list [img_curr]).fadeIn (morph_time); }
 			if (nDivs > 0) { $(div_list [div_curr]).fadeOut (morph_time); div_curr = (div_curr + 1) % nDivs; elem.setAttribute ("curr_div_index", div_curr); $(div_list [div_curr]).fadeIn (morph_time); }
 
 			window.setTimeout (anim, slide_time);
 		}
 
 		// Initial state
 		if (nImgs > 0) { img_list.not (img_list [img_curr]).fadeOut (0); $(img_list [img_curr]).fadeIn (0); }
 		if (nDivs > 0) { div_list.not (div_list [div_curr]).fadeOut (0); $(div_list [div_curr]).fadeIn (0); }
 
 		window.setTimeout (anim, slide_time);
 	}
 }
 
 /*
 	Pause or play slideshow.
 	elem: slideshow DOM element
 	stop: 1 - pause, 0 - play, if not set - toggle
 */
 function pause_slideshow (elem, stop) {
 
 	var img_curr = elem.hasAttribute ("curr_img_index") ? Number (elem.getAttribute ("curr_img_index")) : 0;
 	var div_curr = elem.hasAttribute ("curr_div_index") ? Number (elem.getAttribute ("curr_div_index")) : 0;
 	var isPaused = elem.hasAttribute ("is_paused") ? Number (elem.getAttribute ("is_paused")) : 0;
 	var bStop = typeof (stop) == "undefined" ? (isPaused + 1) % 2 : Number (stop);
 
 	if (isPaused != bStop) {
 
 		elem.setAttribute ("is_paused", bStop);
 
 		if (bStop == 0) { animate_slideshow (elem); }
 	}
 
 	return bStop;
 }
 
 /*
 	Rewind slideshow backward or forward by nFrames.
 	elem: slideshow DOM element
 	nFrames: -N,...,N
 */
 function rewind_slideshow (elem, nFrames) {
 
 	if (!elem.hasAttribute ("curr_img_index")) { elem.setAttribute ("curr_img_index", 0); }
 	if (!elem.hasAttribute ("curr_div_index")) { elem.setAttribute ("curr_div_index", 0); }
 	if (!elem.hasAttribute ("morph_time")) { elem.setAttribute ("morph_time", "slow"); }
 	if (!elem.hasAttribute ("slide_time")) { elem.setAttribute ("slide_time", 5000); }
 
 	{
 		var img_curr = Number (elem.getAttribute ("curr_img_index"));
 		var div_curr = Number (elem.getAttribute ("curr_div_index"));
 		var morph_time = elem.getAttribute ("morph_time");
 		var slide_time = Number (elem.getAttribute ("slide_time"));
 		var img_list = $("img.slide", $(elem));
 		var div_list = $("div.slideContent>div", $(elem));
 		var nImgs = img_list.length;
 		var nDivs = div_list.length;
 
 		// hide all backgrounds, except current
 		if (nImgs > 0) { img_list.stop (true, true); $(img_list [img_curr]).fadeOut (morph_time); img_curr = (nImgs + img_curr + nFrames) % nImgs; elem.setAttribute ("curr_img_index", img_curr); $(img_list [img_curr]).fadeIn (morph_time); }
 
 		// hide all content, except current
 		if (nDivs > 0) { div_list.stop (true, true); $(div_list [div_curr]).fadeOut (morph_time); div_curr = (nDivs + div_curr + nFrames) % nDivs; elem.setAttribute ("curr_div_index", div_curr); $(div_list [div_curr]).fadeIn (morph_time); }
 	}
 
 	return true;
 }
