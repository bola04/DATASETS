    $(document).ready(function () {
 
         $('.slidein-collapse').click(function () {
             if ($('.slidein-bar').hasClass('collapsed')) {
                 $('.slidein-bar').removeClass('collapsed');
             }
             else {
                 $('.slidein-bar').addClass('collapsed');
             }
         });
 
         $('.slidein-icon').click(function () {
             if ($(this).parent().hasClass('active')) {
                 $(this).parent().removeClass('active');
                 $(this).css({
                     'top': '0'
                 });
                 if (!$(this).closest('.slidein-bar').hasClass('align-right')) {
                     $(this).parent().css({
                         'left': ''
                     });
                     $(this).css({
                         'left': ''
                     });
                 }
                 else {
                     $(this).parent().css({
                         'right': ''
                     });
                     $(this).css({
                         'left': '',
                         'right': ''
                     });
                 }
                 $(this).parent().siblings('li').removeClass('hidden');
                 $(this).closest('.slidein-bar').removeClass('hidden');
             }
             else {
                 $(this).parent().addClass('active');
                 $(this).css({
                     'top': $(this).next('.slidein-bar.center-c .slidein-content').height() / -2
                 });
 
                 if (!$(this).closest('.slidein-bar').hasClass('align-right')) {
                     $(this).parent().css({
                         'left': (($(this).next('.slidein-content').innerWidth()) + 10) + 'px'
                     });
                     $(this).css({
                         'left': '-50px'
                     });
                 }
                 else {
                     $(this).parent().css({
                         'right': (($(this).next('.slidein-content').innerWidth()) + 10) + 'px'
                     });
                     $(this).css({
                         'left': 'inherit',
                         'right': -$(this).next('.slidein-content').innerWidth() + 'px'
                     });
                 }
                 $(this).parent().siblings('li').addClass('hidden');
                 $(this).closest('.slidein-bar').addClass('hidden');
             }
         });
 
         $('.slidein-bar.center-c .slidein-content').each(function () {
             $(this).css({
                 "top": $(this).height() / -2
             });
         });
 
         $(document).click(function (e) {
             if ($(e.target).closest('.slidein-widget').length === 0) {
                 $('.slidein-widget').removeClass('active');
                 $('.slidein-widget').removeClass('hidden');
                 $('.slidein-bar').removeClass('hidden');
                 $('.slidein-icon').css({
                     'top': '0',
                     'left': '',
                     'right': ''
                 });
                 $('.slidein-widget').css({
                     'left': '',
                     'right': ''
                 });
             }
         });
 
         function centerSlideIn() {
             var documentheight = $(window).height();
             var slidein = $('.slidein-bar.center-v');
             var slideinheight = $('.slidein-bar.center-v').innerHeight();
 
             slidein.css({
                 'top': (documentheight - slideinheight) / 2
             });
 
         };
         centerSlideIn();
         $(window).resize(function () {
             centerSlideIn();
         });
     });

    function setCookie(cname,cvalue,exdays) { 
   var d = new Date(); 
   d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000)); 
   var expires = "expires=" + d.toGMTString(); 
   document.cookie = cname + "=" + cvalue + ";"+ expires + ";path=/" 
 } 
 function getCookie(cname) { 
   var name = cname; 
   var decodedCookie = decodeURIComponent(document.cookie); 
   var ca = decodedCookie.split(';'); 
   for(var i = 0; i < ca.length; i++) { 
     var c = ca[i]; 
     while (c.charAt(0) == ' ') { 
       c = c.substring(1); 
     } 
     if (c.indexOf(name) == 0) { 
       return c.substring(name.length, c.length); 
     } 
   } 
   return  ""; 
 }
