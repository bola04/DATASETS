     (function ($, document, undefined) {
 
         var pluses = /\+/g;
 
         function raw(s) {
             return s;
         }
 
         function decoded(s) {
             return unRfc2068(decodeURIComponent(s.replace(pluses, ' ')));
         }
 
         function unRfc2068(value) {
             if (value.indexOf('"') === 0) {
                 // This is a quoted cookie as according to RFC2068, unescape
                 value = value.slice(1, -1).replace(/\\"/g, '"').replace(/\\\\/g, '\\');
             }
             return value;
         }
 
         function fromJSON(value) {
             return config.json ? JSON.parse(value) : value;
         }
 
         var config = $.cookie = function (key, value, options) {
 
             // write
             if (value !== undefined) {
                 options = $.extend({}, config.defaults, options);
 
                 if (value === null) {
                     options.expires = -1;
                 }
 
                 if (typeof options.expires === 'number') {
                     var days = options.expires, t = options.expires = new Date();
                     t.setDate(t.getDate() + days);
                 }
 
                 value = config.json ? JSON.stringify(value) : String(value);
 
                 return (document.cookie = [
                     encodeURIComponent(key), '=', config.raw ? value : encodeURIComponent(value),
                     options.expires ? '; expires=' + options.expires.toUTCString() : '', // use expires attribute, max-age is not supported by IE
                     options.path ? '; path=' + options.path : '',
                     options.domain ? '; domain=' + options.domain : '',
                     options.secure ? '; secure' : ''
                 ].join(''));
             }
 
             // read
             var decode = config.raw ? raw : decoded;
             var cookies = document.cookie.split('; ');
             var result = key ? null : {};
             for (var i = 0, l = cookies.length; i < l; i++) {
                 var parts = cookies[i].split('=');
                 var name = decode(parts.shift());
                 var cookie = decode(parts.join('='));
 
                 if (key && key === name) {
                     result = fromJSON(cookie);
                     break;
                 }
 
                 if (!key) {
                     result[name] = fromJSON(cookie);
                 }
             }
 
             return result;
         };
 
         config.defaults = {};
 
         $.removeCookie = function (key, options) {
             if ($.cookie(key) !== null) {
                 $.cookie(key, null, options);
                 return true;
             }
             return false;
         };
 
     })(jQuery, document);

     function createCookie(name, value, days) {
         if (days) {
             var date = new Date();
             date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
             var expires = "; expires=" + date.toUTCString();
         }
         else var expires = "";
         document.cookie = name + "=" + value + expires + "; path=/";
     }
 
     function readCookie(name) {
         var nameEQ = name + "=";
         var ca = document.cookie.split(';');
         for (var i = 0; i < ca.length; i++) {
             var c = ca[i];
             while (c.charAt(0) == ' ') c = c.substring(1, c.length);
             if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
         }
         return null;
     }
 
     function eraseCookie(name) {
         createCookie(name, "", -1);
     }

     if (searchKey == '') {
         alert('Please enter a keyword to search');
         $('.panel-body').unmark(searchKey, { className: "highlight" });
         $('.panel-collapse').collapse("hide");
         return;
     }
 
     $('.panel-body').unmark(searchKey, { className: "highlight" });
     searchKey = searchKey.toLowerCase();
     var atleastOneFound = false;
 
     if (searchKey.length > 2) {
         $('.panel-collapse').collapse("hide");
         $('.panel-body').find('a').each(function () {
 
             var lnkText = $(this).text().toLowerCase();
             if (lnkText.indexOf(searchKey) >= 0) {
                 var parent = $(this).parents('.panel-collapse');
                 parent.collapse("show");
                 atleastOneFound = true;
             }
 
         });
         //highlight text with "mark.js" plugin
         $('.panel-body').mark(searchKey, { className: "highlight" });
 
     }
     else {
         atleastOneFound = true;
 
     }
 
     if (!atleastOneFound) {
         // $('.panel-collapse').collapse("hide");
         alert('Sorry, no results found. Please try again');
         $('#txtSearch').focus();
     }
 };

         $("mark").removeClass("highlight").addClass("highlight1");
         $('.collapse').removeClass("in");
         $('.glyphicon').removeClass("glyphicon-minus").addClass("glyphicon-plus");
     }
 }
