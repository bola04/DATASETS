     function createMap(){

         window.map = map;//将map变量存储在全局
     }
     
     //地图事件设置函数：
     function setMapEvent(){
         map.enableDragging();//启用地图拖拽事件，默认启用(可不写)

         map.enableDoubleClickZoom();//启用鼠标双击放大，默认启用(可不写)
