           var modalTimeout;
     function newsletterSubmission(email, fname, sname, gender, year, month, day, country, url) {
 
         // Add in some form validation!
         var modelNewsletterSubscriberFormDetail = new VarienForm('model-newsletter-validate-detail');
 
         if(modelNewsletterSubscriberFormDetail.validator.validate()){
             $('newsletter-loading').show();
             $('newsletter-buttons').hide();
 
             var request = new Ajax.Request(url,
                 {
                     method:'post',
                     requestHeaders: {
                         Accept: 'application/json'
                     },
                     parameters: 'email='+email+'&fname='+fname+'&sname='+sname+'&gender='+gender+'&year='+year+'&month='+month+'&day='+day+'&country='+country,
                     onSuccess: function(transport) {
                         var json = transport.responseText.evalJSON();
 
                         if(transport.status === 200) {
                             // Only trigger Quantcast when email has been added to 3rd party
                             triggerQuantcast();
 
                             clearTimeout(modalTimeout);
                             modalTimeout = window.setTimeout(function(){
                                 document.querySelector('#model-newsletter-validate-detail').reset();
                                 jQuery('.modal__closer').trigger('click');
                                 jQuery('.untitled_global_banner_close').trigger('click');
                             }, 5000);
 
                             setMessage( json.message, json.type );
 
                             return;
                         }
 
                         setMessage( json.message, json.type );
                     },
                     onFailure: function() {
                         alert('Your request could not be completed. Please try again');
                         $('newsletter-loading').hide();
                         $('newsletter-buttons').show();
                         stopLoadingData();
                     }
                 });
         } else {
             // Just to make sure
             return false;
         }
     }
 
     function setNewsletterEmail(value) {
         var newsletterField = document.getElementById('newsletter-email');
         newsletterField.value = value;
     }
 
     function setMessage( message, type ) {
         $('newsletter-message').update('<ul class="messages"><li class="' + type +'-msg"><ul><li><span>' + message + '</span></li></ul></li></ul>');
 
         $('newsletter-loading').hide();
         $('newsletter-buttons').show();
     }
 
     function triggerQuantcast() {
         // Quantcast Event push
         if(_qevents) {
             _qevents.push(
                 {qacct:"p-S_Z3jgc9QLAx2",event:"refresh",labels:"_fp.event.Newsletter Confirmation"}
             );
         }
 
 
     }
