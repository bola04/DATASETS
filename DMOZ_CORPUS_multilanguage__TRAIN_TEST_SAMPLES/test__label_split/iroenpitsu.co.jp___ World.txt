 function GetWeekDay(yyyy, mm, dd)
 {
   var ret;
   if(mm < 3) {
     yyyy--;
     mm += 12;
   }
   ret = yyyy;
   ret += Math.round((yyyy / 4) - 0.5);
   ret -= Math.round((yyyy / 100) - 0.5);
   ret += Math.round((yyyy / 400) - 0.5);
   ret += Math.round((mm * 13 + 8) / 5 - 0.5);
   ret += dd;
   ret %= 7;
 
   return ret;
 }
