 					secondsOffset = offsets[i];
 				}
 
 				var currentSeconds = (d.getSeconds() + secondsOffset) % 60;
 				var currentMinute = d.getMinutes() + (Math.floor(secondsOffset / 60));
 
 				if (currentMinute > 59) {
 					currentMinute = (currentMinute % 60);
 				}
 
 				var setMinutes = 4 - (currentMinute % 5);
 				var setSeconds = 59 - currentSeconds;
 
 
 				timers[i] = [];
 				timers[i]['setSeconds'] = setSeconds;
 				timers[i]['setMinutes'] = setMinutes;
 
 				setTimerInterval(i, setSeconds, setMinutes);
 			}
 		}
 
 		function setTimerInterval(i) {
 			intervals[i] = setInterval(function() {
 				timers[i]['setSeconds']--;
 
 				if (timers[i]['setSeconds'] < 0) {
 					timers[i]['setMinutes']--;
 					timers[i]['setSeconds'] = 59;
 				}
 
 				if (timers[i]['setMinutes'] < 0) {
 					timers[i]['setMinutes'] = 4;
 				}
 
 				var setMinutes = timers[i]['setMinutes'];
 				var setSeconds = (timers[i]['setSeconds'] < 10 ? '0' + timers[i]['setSeconds'] : timers[i]['setSeconds']);
 
 				var timerId = '#setTimer'+i;
 				$(timerId).html(setMinutes + ' min ' + setSeconds + ' sec');
 
 			}, 1000);
 		}
 	});
