     function OcultarTooltipLogin(divMostrar) {
         var span = divMostrar.find("span");
         $(span).each(function (i, item) {
             var cssValue = $(this).css("visibility");

                     "visibility": "visible",
                     "display": ""
                 });
             } else if (cssValue == "hidden") {
                 $(this).css({
                     "visibility": "hidden",

             }
         });
     }

     function ObtenerDepartamentosLogin(Pais) {
 
         var protocol = $(location).attr('protocol');
         var url = $(location).attr('host');
         var Subsitio = $("#ctl00_g_a82c7328_21d4_412d_b16d_a20dc07b5b50_ctl00_SubSitioLogin").text();
 
         $().SPServices
             ({
                 operation: "GetListItems",
                 webURL: protocol + "//" + url + "/" + Subsitio,
                 async: false,
                 listName: "Divipola",
                 CAMLViewFields: "<ViewFields><FieldRef Name='Departamento' /></ViewFields>",
                 CAMLQuery: "<Query><Where><Eq><FieldRef Name='Pais'/><Value Type='Text'>" + Pais + "</Value> </Eq> </Where></Query>",
                 //CAMLRowLimit: 6,
                 completefunc: function (xData, Status) {
 
                     var Departamentos = [];
                     $(xData.responseXML).SPFilterNode("z:row").each(function () {
 
                         Departamentos.push($(this).attr("ows_Departamento"));
 
                     });
                     Departamentos = EliminarDuplicadosLogin(Departamentos);
                     var ListaDepartamento = $(document.getElementById('ctl00_g_a82c7328_21d4_412d_b16d_a20dc07b5b50_ctl00_DdlDepartamentoLogin'))[0]; // HTMLSelectElement
                     $.each(Departamentos, function (index, text) {
                         ListaDepartamento.options[ListaDepartamento.options.length] = new Option(text, text);
                     });
                 }
             });
         }

     function EliminarDuplicadosLogin(inputArray) {
         var outputArray = [];
 
         for (var i = 0; i < inputArray.length; i++) {
             if ((jQuery.inArray(inputArray[i], outputArray)) == -1) {
                 outputArray.push(inputArray[i]);
             }
         }
 
         return outputArray;
     }
