         $burgerMenu.click(function()
         {
           $burgerMenu.toggleClass('opened');
           $headerNav.toggleClass('opened');
           $('#menu-header .menu-item-has-children > ul').removeClass('opened');
         });
 
         $('#menu-header .menu-item-has-children').each(function()
         {
           var $self = $(this);
           var $subMenu = $self.find('> ul');
           var $backBtn = $('<li/>')
             .addClass('back-button')
             .text('НАЗАД');
           $subMenu.prepend($backBtn);
           var subClick = false;
           $self.click(function()
           {
             if(subClick)
             {
               subClick = false;
               return;
             }
             if($(window).width() < 992)
             {
               $subMenu.addClass('opened');
             }
           });
           $backBtn.click(function(e)
           {
             subClick = true;
             if($(window).width() < 992)
             {
               $subMenu.removeClass('opened');
             }
           });
         });
       });

 							layoutErrors.html(response.message);

 							layoutErrors.fadeOut(200);

 						}

 				});

 		    }
 			});
 
 			formLogin.find('.submit').click(function()
