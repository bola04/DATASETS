    function load_slide(slider) {
         // remove the slideshow icons from keyboard navigation
         jQuery('.flex-prev').attr('tabindex', -1);
         jQuery('.flex-next').attr('tabindex', -1);
         // call a flexslider callback to update the position whenever the slide changes
         jQuery('.flexslider .slides .excerpt').each(function(){
             if (jQuery(this).height() > 0) {
                 // ikg - don't do this at lower res
                 if (jQuery('.flexslider .slides').width() > 500) {
                     jQuery(this).css({'margin-top':'-'+(jQuery(this).height()+parseInt(jQuery(this).css("padding-top"))+parseInt(jQuery(this).css("padding-bottom"))-1)+'px'});
                     var $this = jQuery(this);
                     setTimeout(function() {
                         $this.fadeIn(500);
                     }, 700);
                 }
             } else {
                 jQuery(this).hide();
             }
         });
         // show the carousel if there is one - this should be hidden until the banner is loaded
         jQuery('.flexcarousel').each(function() {
             jQuery(this).show();
         });
         // highlight the current carousel selection, if appropriate
         // we don't have a slider object if we've been called by a window resize - but we shouldn't need to re-highlight
         // the current slide in that event anyway
         if (slider != undefined) {
             var i=0;
             // note we use active-slide as our class rather than flex-active-slide - this prevents the class from being
             // removed by flexslider's internal click handling
             jQuery('.flexcarousel .slides li').each(function(){
                 if (i==slider.currentSlide) {
                     jQuery(this).addClass("active-slide");
                 } else {
                     jQuery(this).removeClass("active-slide");
                 }
                 i++;
             });
         }
     }
