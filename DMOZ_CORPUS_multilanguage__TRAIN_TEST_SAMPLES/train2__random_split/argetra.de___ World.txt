 				valid=false;
 				msg = "Ungültiges Datum"
 			}
 			else{
 				var d1 = new Date();
 				d1.setDate(d1.getDate()-1);
 				var d2 = new Date();
 				d2.setMonth(d2.getMonth()+3);
 				var d = new Date(parts[2], Number(parts[1])-1, parts[0]);
