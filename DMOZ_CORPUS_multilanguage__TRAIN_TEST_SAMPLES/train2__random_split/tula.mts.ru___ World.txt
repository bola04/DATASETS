             for (var i = 0; i < maxSlidesCount; i++) {
                 loadSlide(function() {
                     sliderLoaded++;
                     if (sliderLoaded === maxSlidesCount) {
                         insertSliderToHtml();
                     }
                 }, function() {
                     maxSlidesCount--;
                 });
             }
         }
     }
 
     function insertSliderToHtml() {
         slides.sort(function(a, b) {
             if (a.sortRatio > b.sortRatio) {
                 return 1;
             } else if (a.sortRatio < b.sortRatio) {
                 return -1;
             }
             return 0;
         });
 
         var content = '';
         for (var i = 0; i < slides.length; i++) {
             content += slides[i].html;
         }
 
         if (content) {
             $(sliderContainerSelector).html(getSliderHtml(content));
             initPromoIndexSlider();
         }
     }
 
     function getSlideHtml(data) {
         var html = '<li class="promo-index__item">' +
             '<a class="cover cover_overlay cover_overlay-large" href="' + data.reference + '"';
 
         if (data.bannerAriaLabel) {
             html += ' aria-label="' + data.bannerAriaLabel + '"';
         }
 
 		html += ' target="' + data.bannerTarget + '">' +
             '<div class="cover__placeholder"><img src="' + data.bannerImage + '" class="cover__img"';
 
         if (data.bannerAlt) {
             html += ' alt="' + data.bannerAlt + '"';
         }
 			
 		html += '></div>' +
             '<div class="cover__overlay">' +
             '<div class="cover__info">';
 
         if (data.bannerTitle) {
             html += '<div class="cover__name">' + data.bannerTitle + '</div>';
         }
 
 		html += '<div class="cover__bywrap">';
 		
         if (data.bannerText) {
             html += '<div class="cover__text">' + data.bannerText + '</div>';
         }
 		
 		if (data.bannerPrice) {
             html += '<div class="cover__price">' + data.bannerPrice + '</div>';
         }
 		
 		html += '</div>' + '<div class="cover__fields-wrap">';
 
         if (data.bannerButtonText) {
             html += '<object><a href="' + (data.referenceButton ? data.referenceButton : data.reference) + '"' +
                 ' class="btn btn_large cover__fields" target="' + data.bannerTarget + '">' + data.bannerButtonText + '</a></object>';
         }
 
         html +=
             '</div>' +
             '</div>' +
             '</div>';
 			
         if (data.bannerImageLogo) {
         html += '<div class="cover__logo-wrap"><img src="' + data.bannerImageLogo + '" class="cover__logo"></div>';
         }
 
         html +=
             '</a>' +
             '</li>';
 
         return html;
     }
 
     function getSliderHtml(content) {
         return '<div class="promo-index__slider-wrap bx-slider">' +
             '<ul class="promo-index__slider">' + content + '</ul>' +
             '<div class="promo-index__slider-nav-wrap">' +
             '<div class="b-wrapper">' +
             '<div class="promo-index__slider-nav">' +
             '<div class="promo-index__slider-pager"></div>' +
             '<div class="promo-index__slider-controls">' +
             '<div class="promo-index__slider-next"></div>' +
             '<div class="promo-index__slider-prev"></div>' +
             '</div>' +
             '</div>' +
             '</div>' +
             '</div>' +
             '</div>';
     }
 
     function initSlide(data) {
         data.html = getSlideHtml(data);
         data.sortRatio = data.sortRatio ? parseInt(data.sortRatio) : 0;
         slides.push(data);
     }
 
 	function reportVisibility(index) {

             sendedRepots.push(slides[index].eventConfirmVisibility);

     }
