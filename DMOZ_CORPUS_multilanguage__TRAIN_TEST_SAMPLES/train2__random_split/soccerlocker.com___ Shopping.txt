 			$(element).addClass('variant-selection-selected');
 		}
 		$(element).parent().parent().parent().parent().find('.variant-selection-size-tabbed-with-expanded-selectors').html($(element).attr('value'));
 		$(element).parent().parent().parent().parent().find('.variant-selection-size-tabbed-with-expanded-selectors').attr('value', $(element).attr('value'));
 		
 		var selected_size = $(element).attr("value");
 		if ($('#sizeSelect').val() != selected_size) {
 			$('#sizeSelect').val(selected_size);
 			sizeChanged();
 		}
 		/* display subsize selection if there is at least one subsize */
 		var subsizes_count = $(element).parent().find('.subsize-box-tabbed-with-expanded-selectors').length;
 		// console.log('subsizes_count: ' + subsizes_count);
 
 			// var selector = prompt('Enter selector: ');
 			if (subsizes_count > 0) {
 			var selector = '.subsize-box-tabbed-with-expanded-selectors';
 			// console.log('size: ' + selected_size + ', subsize: ' + $(element).parent().find(selector).attr('value'));
 			var subsizes_list_output = "";
 			$(element).parent().find(selector).each(function(index, element) {
 				subsizes_list_output += $('<div>').append($(element).clone()).html();
 			});
 			
 			if (subsizes_count > 0) {
 				var subsize = "";
 				if ($('.color-list-tabbed-with-expanded-selectors').find('.display-subsizes-list').length > 0) {
 					// console.log('exists');
 					$('.display-subsizes-list').html(subsizes_list_output);
 					$('.display-subsizes-list').find('.subsize-box-tabbed-with-expanded-selectors').attr('onclick', 'subsize_box_click_tabbed_with_expanded_selectors(this);');
 				} else {
 					// console.log('does not exist, creating...');
 					var subsizes_label = document.createElement('label');
 					$(subsizes_label).addClass('variant-title-tabbed-with-expanded-selectors col-lg-2 col-md-3 col-sm-2 col-xs-3 control-label input-sm');
 					$(subsizes_label).html('Select subsize: ');
 					$('.sizes-container-tabbed-with-expanded-selectors').after($(subsizes_label));
 					
 					var subsizes_display = document.createElement('div');
 					$(subsizes_display).addClass('display-selected-subsize variant-selection-subsize-tabbed-with-expanded-selectors col-lg-10 col-md-9 col-sm-10 col-xs-9');
 					$(subsizes_label).after($(subsizes_display));
 					
 					var subsizes_list = document.createElement('div');
 					$(subsizes_list).html(subsizes_list_output);
 					$(subsizes_list).addClass('display-subsizes-list variant-selection-subsize-tabbed-with-expanded-selectors col-lg-10 col-md-9 col-sm-10 col-xs-9');
 					
 					$(subsizes_list).find('.subsize-box-tabbed-with-expanded-selectors').attr('onclick', 'subsize_box_click_tabbed_with_expanded_selectors(this);');
 					
 					$(subsizes_display).after($(subsizes_list));
 					// $('#subsizeSelect').val(subsize);
 					// subsizeChanged();
 				}
 			}
 		}
 		if (subsizes_count > 1) {
 			/* more than one subsize */
 			
 			/* select previously selected subsize */
 			// var subsize_previously_selected = $('.display-selected-subsize').attr('value');
 			
 			/* do not select previously selected subsize */
 			// var subsize_previously_selected = null;
 			
 			// console.log(typeof subsize_previously_selected);
 			
 			if (typeof subsize_previously_selected != 'undefined') {
 				// console.log('subsize_previously_selected is defined');
 				
 				$('.display-subsizes-list').find('.subsize-box-tabbed-with-expanded-selectors').each(function(index, element) {
 					
 					if ($(element).attr('value') == subsize_previously_selected) {
 						$(element).click();
 					}
 					
 				});
 			} else {
 				// console.log('subsize_previously_selected is not defined');
 				/* not selecting previously selected subsize so clear displayed subsize */
 				$('.display-selected-subsize').attr('value', '');
 				$('.display-selected-subsize').html('');
 			}
 			// console.log('subsize_previously_selected: ' + subsize_previously_selected);
 		} else {
 			/* only one subsize */
 			$('.subsize-box-tabbed-with-expanded-selectors').click();
 		}
 	}
 	
 	function subsize_box_click_tabbed_with_expanded_selectors(element) {
 		var size_boxes = $(element).parent().find('.subsize-box-tabbed-with-expanded-selectors').removeClass('variant-selection-selected');
 		if ($(element).hasClass('variant-selection-selected') == false) {
 			$(element).addClass('variant-selection-selected');
 		}
 				
 		var parent_size = $('.variant-selection-size-tabbed-with-expanded-selectors').attr("value");
 		// console.log("parent_size: " + parent_size);
 		// return;
 		var subsize = $(element).attr("value");
 		// console.log("subsize: " + subsize);
 		// console.log($(element).attr("value"));
 		// console.log($(element).parent().parent().parent().find('.size-box-tabbed-with-expanded-selectors').attr("value"));
 		if ($('#sizeSelect').val() != parent_size) {
 			$('#sizeSelect').val(parent_size);
 			sizeChanged();
 			$(element).click();
 		} else if ($('#sizeSelect').val() == parent_size) {
 			var subsizes_count = $('.size-box-tabbed-with-expanded-selectors[value="' + parent_size + '"]').parent().find('.subsize-box-tabbed-with-expanded-selectors').length;
 			// console.log('subsizes_count: ' + subsizes_count);
 			if (subsizes_count > 1) {
 				$('#subsizeSelect').val(subsize);
 				subsizeChanged();
 			}
 		}
 		$('.display-selected-subsize').attr('value', subsize);
 		$('.display-selected-subsize').html(subsize);
 		// $(this).parent().parent().parent().find('.size-box').attr("value");
 		
 		// console.log($(subsizeSelect).val());
 	}
