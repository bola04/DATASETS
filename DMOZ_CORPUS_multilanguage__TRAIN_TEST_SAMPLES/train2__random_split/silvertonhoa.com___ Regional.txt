    function attemptLogin() {
     var frm = document.getElementById('loginform');
     var postData = {
         UserName : frm.UserName.value, 
         Password : frm.Password.value, 
         ajax : 1 
     };
     if(frm.RedirectURL){
         postData.RedirectURL = frm.RedirectURL.value;
     }
     jQuery.post('login_check.php', postData,
         function(data) {
             var match = /^OK: (.+)/.test(data);
             if(match){
                 var redir = data.substring(4);
                 window.location = redir;
             }else{
                 var elem = document.getElementById('loginerror');
                 elem.innerHTML = data;
             }
         }
     );
 }
