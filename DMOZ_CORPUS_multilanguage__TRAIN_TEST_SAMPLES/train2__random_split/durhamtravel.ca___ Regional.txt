       var iframe = false;
 var listbox;
 $(function() {
 	var normalize = function( term ) {
 		var ret = "";
 		for ( var i = 0; i < term.length; i++ ) {
 			ret += accent_map[ term.charAt(i) ] || term.charAt(i);
 		}
 		return ret;
     };
 	
 	if( iframe || window.innerWidth >= 1025 || !Modernizr.inputtypes.date ) {
 		$('#flights-departure-date').datepicker({
 			dateFormat: 'yy-mm-dd',
 			minDate: new Date(2018, 6, 32),
 			maxDate: new Date(2019, 6, 30),
 			numberOfMonths: 1,
 			showButtonPanel: true,
 			onSelect: function(dateText, inst) {
 				var month = parseInt(inst.selectedMonth);
 				var day = parseInt(inst.selectedDay);
 				var year = parseInt(inst.selectedYear);
 				
 				$('#flights-return-date').datepicker('option', {
 					'minDate': new Date(year, month, day + 1)
 				});
 				
 				setTimeout("$('#flights-return-date').trigger('focus');", 150);
 			}
 		});
 		$('#flights-return-date').datepicker({
 			dateFormat: 'yy-mm-dd',
 			minDate: new Date(2018, 6, 32),
 			maxDate: new Date(2019, 6, 30),
 			numberOfMonths: 1,
 			showButtonPanel: true
 		});
 	}
 	
 	window.validate.flightSearch = function() {
 		$('body').addClass('validated');
 		$('.error').removeClass('error');
 		$('.error__msg').remove();
 		
 		var errors = [];
 		var error = null;
 		var valid = true;
 		
 		var departure1 = $('#flights-trip-1-departuregateway');
 		var destination1 = $('#flights-trip-1-arrivalgateway');
 		var departure2 = $('#flights-trip-2-departuregateway');
 		var destination2 = $('#flights-trip-2-arrivalgateway');
 		var departureDate = $('#flights-departure-date');
 		var returnDate = $('#flights-return-date');
 		var flightType = $('.flight-type:checked').val();
 		
 		if( departure1.val().length == 0 ) {
 			
 			error = {
 				'field': $('#flights-trip-1-departuregateway'),
 				'message': 'Departure Gateway is required'
 			};
 			errors.push(error);
 			
 		}
 		
 		if( destination1.val().length == 0 ) {
 			
 			error = {
 				'field': destination1,
 				'message': 'Departure Destination is required'
 			};
 			errors.push(error);
 			
 		}
 		
 		if( flightType == 'multi-trip' ) {
 			
 			if(departure2.val().length == 0){
 			
 				error = {
 					'field': departure2,
 					'message': 'Departure Gateway is required'
 				};
 				errors.push(error);
 				
 			}
 			
 			if(destination2.val().length == 0){
 				
 				error = {
 					'field': destination2,
 					'message': 'Departure Destination is required'
 				};
 				errors.push(error);
 				
 			}
 			
 		}
 		
 		if( departureDate.val().length == 0 ) {
 			
 			error = {
 				'field': departureDate,
 				'message': 'Departure Date is required'
 			};
 			errors.push(error);
 			
 		}
 		
 		if( returnDate.val().length == 0 ) {
 			
 			error = {
 				'field': returnDate,
 				'message': 'Departure Date is required'
 			};
 			errors.push(error);
 			
 		}
 		
 		if(errors.length){
 			
 			$.each(errors, function(index, error){
 				$(error.field).parents('li').addClass('error');
 				var div = $('<div></div>').attr({
 					'class': 'error__msg'
 				}).html(error.message);
 				$(error.field).parents('li').append(div);
 			});
 			valid = false;
 		}
 		
 		return valid;
 	}
 	
 	
 	$('.airportList').keyup(function(e) {
 		var input = $(this).val();
 		
 		switch(e.keyCode) {
 			case 13:
 				e.preventDefault();
 				select();
 			break;
 			case 38:
 				e.preventDefault();
 				if( $('.listbox').length > 0 ) {
 					if( $('.listbox').children('.selected').length > 0 ) {
 						$('.listbox').children('.selected').removeClass('selected').prev().addClass('selected');
 						if( $('.listbox').children('.selected').length == 0 ) {
 							$('.listbox').children().last().addClass('selected');
 						}
 					} else {
 						$('.listbox').children().last().addClass('selected');
 					}
 				}
 			break;
 			case 40:
 				e.preventDefault();
 				if( $('.listbox').length > 0 ) {
 					if( $('.listbox').children('.selected').length > 0 ) {
 						$('.listbox').children('.selected').removeClass('selected').next().addClass('selected');
 						if( $('.listbox').children('.selected').length == 0 ) {
 							$('.listbox').children().first().addClass('selected');
 						}
 					} else {
 						$('.listbox').children().first().addClass('selected');
 					}
 				}
 			break;
 			default:
                 $('.listbox').prev().removeClass('has-autocomplete-results');
 				$('.listbox').remove();
 				if( input.length >= 3 ) {
 					populateList(input, $(this));
 				}
 		}
 		
 	});
 	
 	$('.airportList').focus(function(e) {
 		clearTimeout(listbox);
 		var input = $(this).val();
 		
 		$(this).parent().children('.error__msg').remove();
 		
 		$('.listbox').prev().removeClass('has-autocomplete-results');
         $('.listbox').remove();
 		if( input.length >= 2 ) {
 			populateList(input, $(this));
 		}
 	});
 	
 	$('.airportList').focusout(function(e) {
 		var thisEl = $(this);
 		
 		listbox = setTimeout(function(){ 
 			$('.listbox').remove();
 			if( thisEl.val() !== '' ) {
 				if( selectArray[thisEl.next().val()] == undefined || selectArray[thisEl.next().val()][1] !== thisEl.val() ) {
 					thisEl.val('');
 					thisEl.next().val('');
 					
 					thisEl.parent().append('<div class="error__msg">Invalid Airport/Location. Please try again.</div>');
 				}
 			}
 		}, 500);
 	});

 	function populateList(input, parent) {
 		var match = false;
 		var ul = $('<ul></ul>').attr({
 			'class': 'listbox autocomplete__dropdown',
 			'id': 'listbox',
 			'aria-live': 'polite',
 			'aria-hidden': false
 		});
 		
 		for( key in selectArray ) {
 			var thisValue = normalize(selectArray[key][1].toLowerCase());
 			if( thisValue.indexOf(normalize(input.toLowerCase())) !== -1 && key.length == 3 ) {
 				if( parent.data('restriction') == 'can' ) {
 					if( selectArray[key][0] == '1' ) {
 						match = true;
 					}
 				} else {
 					match = true;
 				}
 				
 				if( match ) {
 					var a = $('<a></a>').attr({
 						'class': 'autocomplete__airport',
 						'href': '#'
 					});
 					var code = $('<div></div>').attr({
 						'class': 'autocomplete__airport__code'
 					}).html(key);
 					var airport = $('<div></div>').attr({
 						'class': 'autocomplete__airport__name'
 					}).html(selectArray[key][1]);
 					a.append(code).append(airport)
 					
 					var li = $('<li></li>').attr({'data-airport': key});
 					li.append(a);
 					
 					ul.append(li);
 				}
 			}
 		};
         
         $(parent).removeClass('has-autocomplete-results');
         
 		if( match ) {
 			ul.children().first().addClass('selected');
 			$(parent).parent().append(ul);
             $(parent).addClass('has-autocomplete-results');
 		}
         
 	}
 	
 	function select() {
 		if( $('.listbox').children('.selected').length > 0 ) {
 			var input = $('.listbox').prev().prev();
 			if( $(input).hasClass('multiport') ) {
 				$.each($('.multiport'), function() {
 					var id = $(this).data('id');
 					$(this).val($('.listbox').children('.selected').find('.autocomplete__airport__name').text());
 					$(this).next().val($('.listbox').children('.selected').data('airport'));
 				});
 			} else {
 				$(input).val($('.listbox').children('.selected').find('.autocomplete__airport__name').text());
 				var id = input.data('id');
 				$('#'+id).val($('.listbox').children('.selected').data('airport'));
 			}
 			
             $('.listbox').prev().removeClass('has-autocomplete-results');
 			$('.listbox').remove();
             
 		}
 	}

 		$('.deals--flights .deals__filter__departure__field :input').empty();
 		$.each(window.flight[group].gateways, function(gatewayCode, gatewayName) {
 			var option = $('<option></option>').attr({
 				'value': gatewayCode
 			}).html(gatewayName);
 			$('.deals--flights .deals__filter__departure__field :input').append(option);
 		});
 		
 		var gateway = $('.deals--flights .deals__filter__departure__field :input :first').val();
 		$('.deals--flights .deals__filter__departure__field :input option').each(function() {
 			if( this.value == selectedGateway ) {
 				gateway = selectedGateway;
 			}
 		});
 		$('.deals--flights .deals__filter__departure__field :input').val(gateway);
 		
 		
 		$('.deals--flights .deals__filter__departure__field :input').data('packagegroup', group);
 		
 		$('.deals--flights .deals__filter__departure__field :input').change();
 		
 		if( flightLoaded ) {
 			dataLayer.push({
 				'language'			: 'en',
 				'event'				:'PromoTabClicked',
 				'eventInfo'	: {
 					'category'		: 'PromoTab - Flights',
 					'action'		: 'Click - from ' + $('.deals--flights .deals__filter__departure__field :input :selected').text(),
 					'label'			: $(this).text()
 				}
 			});
 		}
 		
 		flightLoaded = true;
 	});
 	
 	$('#deals-flight-tab li[aria-selected="true"] a').click();
 	
 	$('.deals--flights .deals__list').on('click', '.deal__book', function() {
 		$('#flight-deals-modal [name="flightstring"]').val($(this).data('flightstring'));
 		$('#flight-deals-modal [name="tab"]').val($(this).data('tab'));
 		$('#flight-deals-modal [name="label"]').val($(this).data('label'));
 		$('#flight-deals-modal [name="value"]').val($(this).data('value'));
 	});
 	
 	$('#flight-deals-modal').on('click', 'button[type="submit"]', function(e) {
 		
 		dataLayer.push({
 			'language'		: 'en',
 			'event'			: 'BookButtonClicked',
 			'eventInfo'   	: {
 				'category'		: 'PromoTab - Book Button Clicks',
 				'action'		: 'Book - ' + $('#flight-deals-modal [name="tab"]').val() + ' - From ' + $('.deals--flights [name="flight-deals-departure"] :selected').text(),
 				'label'			: $('#flight-deals-modal [name="label"]').val() + '|' + $('#flight-deals-modal [name="number_of_adults"] :selected').val() + '|' + $('#flight-deals-modal [name="number_of_children"] :selected').val() + '|' + $('#flight-deals-modal [name="number_of_infants"] :selected').val(),
 				'value'			: $('#flight-deals-modal [name="value"]').val()
 			}
 		});
 		
 	});
 });
     </script>
