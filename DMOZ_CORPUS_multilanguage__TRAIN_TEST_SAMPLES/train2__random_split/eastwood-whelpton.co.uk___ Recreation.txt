       } else {
         // Otherwise get the tallest carousel item
         items.each(function() {
           heights.push($(this).height());
         });
         
         tallest = Math.max.apply(null, heights);
         // Animated glide
         items.stop().animate({'min-height': tallest}, 800)
       }
     };
     normalize();
 
     $(window).on('orientationchange resize', function () {
       // Reset
       tallest = 0, heights.length = 0;
       items.each(function() {
         $(this).css('min-height','0');
       });
       
       // Renormalize
       normalize();
     });
   }
 });
