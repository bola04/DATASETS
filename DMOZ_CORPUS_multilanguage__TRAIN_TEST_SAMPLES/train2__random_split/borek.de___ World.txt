         function toggleMenuOnSmallScreens(event, menuLevel, showCssClass){
             var windowWidth = jQuery(window).width();
             //only use on small screens
             if(windowWidth <= 984){
                 var subMenu = jQuery(event.currentTarget).find('ul.'+menuLevel);
                 if(subMenu.length){
                     if(!subMenu.hasClass(showCssClass)){
                         subMenu.addClass(showCssClass);
                         subMenu.show();
                     }else{
                         subMenu.removeClass(showCssClass);
                         subMenu.hide();
                     }
                     //avoid forwarding to clicked links (main-item)
                     event.preventDefault();
                     event.stopPropagation();
                 }
             }
         }
         function forwardLink(event, clickedElement){
             var parent = jQuery(clickedElement).parent();
             if(parent.hasClass('parent')){
                 toggleMenuOnSmallScreens(event, 'level1', 'show-navigation-level2');
             }
             else{
                 window.location.replace(clickedElement.href);
             }
         }

         <h2>
          Richard Borek - die ganze Welt der Briefmarken
         </h2>
         <br/>
         <h3>
          <strong>
           Briefmarken sammeln - ein spannendes Hobby voller Faszination und Abwechslung
          </strong>
         </h3>
         <p>
          Kaum ein Hobby ist so vielfältig wie das
          <a href="https://www.borek.de/briefmarkenwelt/einfuehrung-ins-briefmarken-sammeln" title="Einführung in das Sammeln von Briefmarken">
           Sammeln von Briefmarken
          </a>
          . Sie sind kleine Kunstwerke, die von historischen, geographischen und kulturellen Gegebenheiten ihrer Ausgabeländer zeugen. Sie erzählen Geschichten, laden auf Reisen in ferne Länder und zu unbekannten Kulturen ein. Ist einmal der Jagdinstinkt geweckt, wird
          <a href="https://www.borek.de/briefmarkenwelt/einfuehrung-ins-briefmarken-sammeln" title="Einführung in das Briefmarkensammeln">
           das Briefmarken-Sammeln
          </a>
          zu einer abenteuerlichen Schatzsuche voller Spannung und Abwechslung. Erleben Sie dieses besondere Glücksgefühl beim Aufbau Ihrer persönlichen Briefmarkensammlung und entdecken Sie die Sammelgebiete, an denen Sie Freude haben!
         </p>
         <p>
          <strong>
          </strong>
         </p>
         <h3>
          <strong>
           Briefmarken online kaufen - einfach und bequem
          </strong>
         </h3>
         <p>
          In unserem Online-Shop finden Sie unser gesamtes Sortiment - ob
          <a href="https://www.borek.de/deutschland" target="_self" title="Briefmarken Deutschland">
           Briefmarken aus Deutschland
          </a>
          oder
          <a href="https://www.borek.de/briefmarken-international" target="_self" title="Briefmarken International">
           international
          </a>
          , wir bieten Ihnen die ganze Vielfalt, übersichtlich nach Kategorien geordnet. Unser
          <a href="https://www.borek.de/zubehoer/briefmarkenkataloge" title="Briefmarkenkataloge">
           Briefmarkenkatalog
          </a>
          bietet Ihnen einen Überblick über die verschiedenen Sammelgebiete von den
          <a href="https://www.borek.de/deutschland/briefmarken-deutsches-reich" title="Briefmarken Deutsches Reich">
           Briefmarken des Deutschen Reiches
          </a>
          über
          <a href="https://www.borek.de/deutschland/briefmarken-ddr" title="Briefmarken DDR">
           Briefmarken der DDR
          </a>
          und
          <a href="https://www.borek.de/deutschland/briefmarken-bundesrepublik" title="Briefmarken Bundesrepublik Deutschland">
           BRD
          </a>
          bis hin zu den Themen
          <a href="https://www.borek.de/numisbriefe/2-euro-numisbriefe" target="_self" title="2 Euro Numisbriefe">
           2 Euro Numisbriefe
          </a>
          ,
          <a href="https://www.borek.de/numisbriefe/medaillenbriefe" target="_self" title="Medaillenbriefe">
           Medaillenbriefe
          </a>
          oder
          <a href="https://www.borek.de/deutschland/deutsche-ersttagsbriefe" target="_self" title="Deutsche Ersttagsbriefe">
           Deutsche Ersttagsbriefe
          </a>
          . In unserer Themenübersicht finden Sie auch viele
          <a href="https://www.borek.de/wertvolle-briefmarken" title="Seltene und wertvolle Briefmarken">
           wertvolle Briefmarken
          </a>
          , mit denen Sie Ihr Briefmarkenalbum um beständige Werte bereichern können.
         </p>
         <p>
          <strong>
          </strong>
         </p>
         <h3>
          <strong>
           Werte für Generationen: alte und seltene Briefmarken
          </strong>
         </h3>
         <p>
          Briefmarken sammeln ist nicht nur ein schönes Hobby, sondern auch eine Wertanlage aus originalen Sachwerten, die häufig über Generationen weitervererbt werden. Mit einem guten Gespür für alte, einzigartige oder
          <a href="https://www.borek.de/wertvolle-briefmarken" target="_self" title="Seltene Briefmarken">
           seltene Briefmarken
          </a>
          können Sie aus Raritäten und besonderen Sammlerstücken eine Kollektion zusammenstellen, die philatelistischen Wert besitzt und eine stabile Anlageform darstellt. Selbstverständlich spielt die Qualität der Postwertzeichen dabei eine entscheidende Rolle: Gerade bei sehr alten Briefmarken oder solchen, die in Kriegszeiten hergestellt wurden, sind überdurchschnittlich gut erhaltene Stücke selten. Solche
          <a href="https://www.borek.de/wertvolle-briefmarken" target="_self" title="Wertvolle Briefmarken">
           wertvollen Briefmarken
          </a>
          in bester Qualität präsentieren wir Ihnen in unserem Themengebiet "Wertvolle Briefmarken".
          <br/>
          <br/>
         </p>
         <h3>
          <strong>
           Deutsche Briefmarken: ganz oben auf der Beliebtheitsskala
          </strong>
         </h3>
         <p>
          Unter deutschen Sammlern sind die im eigenen Land erschienenen Briefmarken besonders beliebt - aufgrund der geringen Anzahl früher gedruckter Marken kann die Nachfrage der wachsenden Sammelgemeinde nicht immer befriedigt werden. Deutsche Sammelgebiete wie z.B. die
          <a href="https://www.borek.de/deutschland/briefmarken-altdeutsche-staaten" target="_self" title="Briefmarken Altdeutsche Staaten">
           Briefmarken der Altdeutschen Staaten
          </a>
          bis 1871 oder die
          <a href="https://www.borek.de/deutschland/briefmarken-deutsches-reich" target="_self" title="Briefmarken Deutsches Reich">
           Briefmarken des Deutschen Reiches
          </a>
          bis 1945 haben große Wertsteigerungen erfahren. Das beliebteste Kapitel der deutschen Geschichte stellen die
          <a href="https://www.borek.de/deutschland/briefmarken-bundesrepublik" target="_self" title="Briefmarken Bundesrepublik">
           Briefmarken der BRD
          </a>
          ,
          <a href="https://www.borek.de/deutschland/briefmarken-ddr" target="_self" title="Briefmarken DDR">
           DDR
          </a>
          und
          <a href="https://www.borek.de/deutschland/briefmarken-berlin" target="_self" title="Briefmarken Berlin">
           Berlin
          </a>
          dar. Hier sind zum Teil enorme Wertsteigerungen zu beobachten.
          <br/>
          <br/>
         </p>
         <h3>
          <strong>
           Auf ausgesuchte Sammelgebiete beschränken
          </strong>
         </h3>
         <p>
          Der
          <a href="https://www.borek.de/briefmarkenwelt/briefmarken-wert-ermitteln-bestimmen" title="Wert von Briefmarken">
           Wert einer Sammlung
          </a>
          zeigt sich besonders in der Beschränkung auf Sammelgebiete, die planvoll aufgebaut werden und einen möglichst kompletten Überblick über das jeweilige Thema bieten. Beschränken Sie sich also zunächst auf wenige Länder, um diese nach und nach zu vervollständigen. In der
          <a href="https://www.borek.de/briefmarkenwelt/philatelie" title="Philatelie">
           Philatelie
          </a>
          sollte man sich nicht dazu verleiten lassen, wahllos eine große Zahl von Serien aus allen Ländern der Welt zu sammeln, sondern ausgesuchte Sammelgebiete sorgfältig zu pflegen. Es versteht sich von selbst, dass ungebrauchte Briefmarken dabei wertvoller sind als eine vergleichbare Sammlung mit gebrauchten Marken.
          <br/>
          <br/>
         </p>
         <h3>
          <strong>
           Wichtig für den Qualitätserhalt: das richtige Zubehör
          </strong>
         </h3>
         <p>
          Briefmarken sind empfindlich und wollen buchstäblich mit Samthandschuhen angefasst werden. In unserer Kategorie
          <a href="https://www.borek.de/zubehoer" target="_self" title="Briefmarken Zubehör">
           Briefmarken Zubehör
          </a>
          finden Sie eine große Auswahl an
          <a href="https://www.borek.de/zubehoer/briefmarkenalben" title="Briefmarkenalben">
           hochwertigen Alben
          </a>
          vom Basic-Einsteckbuch bis zum Luxus-Ringbinder, sodass Ihre Briefmarken optimal geschützt sind. Mit weiterem Zubehör von
          <a href="https://www.borek.de/zubehoer/weiteres-zubehoer/briefmarken-lupen" title="Briefmarken Lupen">
           Lupen
          </a>
          über Pinzetten bis hin zur UV-Lampe zur Fluoreszenz-Bestimmung sind Sie perfekt ausgestattet.
          <br/>
          <br/>
          Viel Freude beim Aufbau und Vervollständigen Ihrer Sammlung!
          <br/>
          <br/>
         </p>

     <script type="text/javascript">
      document.observe("dom:loaded", function () {
 
         var CmsGridEcondaTracker = Class.create({
             /**
              * Initialize routine
              *
              * @return {void}
              */
             initialize: function (defaultCodes, extendedCodes, maxContentLength) {
                 // initialize config vars
                 this.defaultCodes = defaultCodes;
                 this.extendedCodes = extendedCodes;
                 this.maxContentLength = maxContentLength;
 
                 // check if window.emosPropertiesEvent is available
                 if (typeof window.emosPropertiesEvent !== 'function') {
                     var errorMSg = 'econda not loaded correctly';
                     throw new Error(errorMSg);
                 }
 
                 // setup markers
                 this.setupMarkers();
             },
 
             /**
              * Sets up tracking markers
              *
              * @return {void}
              */
             setupMarkers: function () {
                 var self = this;
 
                 $$('.grid-wrapper .grid-element-wrapper').each(function (element) {
                     var elementId = self.attributeData(element, 'element-id');
                     var elementCode = self.attributeData(element, 'element-code');
                     var gridId = self.attributeData(element, 'grid-id');
 
                     var isDefault = false;
                     var isExtended = false;
 
                     if (self.defaultCodes.indexOf(elementCode) !== -1) {
                         isDefault = true;
                     } else if (self.extendedCodes.indexOf(elementCode) !== -1) {
                         isExtended = true;
                     } else {
                         // exit when element is not handled
                         return;
                     }
 
                     // array of content variables
                     var contentArray = [
                         gridId,
                         elementId
                     ];
 
                     // collect and filter all links (exclude fancybox and flex links)
                     var htmlLinks = $(element).select('a').filter(function (link) {
                         return link.className.indexOf('fancybox') === -1
                             && link.className.indexOf('flex-prev') === -1
                             && link.className.indexOf('flex-next') === -1;
                     });
 
                     // collect all child buttons
                     var htmlButtons = $(element).select('button');
 
                     // merged relevant html-elements into one array
                     var mergedHtmlElements = htmlLinks.concat(htmlButtons);
 
                     if (isDefault || isExtended) {
                         mergedHtmlElements.each(function (htmlElement) {
                             // clone content array for current element context
                             var htmlElementContentArray = contentArray.slice(0);
 
                             if (isExtended) {
                                 // handle additional attribute-data for clickable elements
                                 var position = self.attributeData(htmlElement, 'position');
                                 var productId = self.attributeData(htmlElement, 'product-id');
 
 
                                 if (position !== null) {
                                     htmlElementContentArray.push('Pos:' + position);
                                 }
 
                                 if (productId !== null) {
                                     htmlElementContentArray.push(productId);
                                 }
                             }
 
                             var markerData = self.convertContentArrayToString(htmlElementContentArray);
 
                             Event.observe(htmlElement, 'click', function (event) {
                                 // fire tracking event
                                 window.emosPropertiesEvent({marker: markerData, rqtype: 'hiddenpi'});
                             });
                         });
                     }
                 });
             },
 
             /**
              * Converts content array to string and limits max content length if needed
              * @param contentArray string[]
              * @return {string}
              */
             convertContentArrayToString: function (contentArray) {
                 for (
                     var currentString = contentArray.join('/');
                     currentString.length > maxContentLength;
                     currentString = contentArray.join('/')
                 ) {
                     var insertedX = false;
                     var currentIndex = contentArray.length - 1;
 
                     while (insertedX !== true) {
                         var value = contentArray[currentIndex];
 
                         if (currentIndex === 0) {
                             // if first index is too long, take maxContentLength of it
                             console.log('trim to: ' + this.maxContentLength);
 
                             contentArray = [
                                 value.substring(0, this.maxContentLength)
                             ];
 
                             insertedX = true;
                         } else if (value !== 'X') {
                             // replace content with 'X' to shorten the content-string
                             contentArray[currentIndex] = 'X';
                             insertedX = true;
                         }
 
                         currentIndex--;
                     }
                 }
 
 
                 var contentString = contentArray.join('/');
 
                 return contentString;
             },
 
             /**
              * Reads or sets data-attributes for given element
              *
              * @param {string} element
              * @param {string} key
              * @param {any} [value]
              * @return {null|String}
              */
             attributeData: function (element, key, value) {
                 var DATA_REGEX = /data-([\w-]+)/;
                 var attributesLength = element.attributes.length;
                 if (key && value) {
                     element.setAttribute('data-' + key, value);
                 } else {
                     for (var currentIndex = 0; currentIndex < attributesLength; currentIndex++) {
                         var attr = element.attributes[currentIndex];
                         if (attr && attr.name) {
                             var m = attr.name.match(DATA_REGEX);
                             if (m && m.length > 1) {
                                 var datakey = m[1];
                                 if (datakey === key) {
                                     return attr.value;
                                 }
                             }
                         }
                     }
                 }
                 return null;
             }
         });
 
         var defaultCodes = ["n98_cmsgrid_element_image","n98_cmsgrid_element_product","n98_cmsgrid_element_category"];
         var extendedCodes = ["n98_cmsgrid_element_productlist","n98_cmsgrid_element_productstage","n98_cmsgrid_element_productslider","n98_cmsgrid_element_imagestage"];
         var maxContentLength = 128;
 
         setTimeout(function () {
             new CmsGridEcondaTracker(defaultCodes, extendedCodes, maxContentLength);
         }, 500);
     });
