          jQuery.fn.center = function(pos){
         if (! pos)
             pos = 'fixed';
 
         var w = jQuery(window);
         this.css("position", pos);
         if ('fixed' == pos){
             this.css("top" , (w.height()-this.height())/2 + "px");
             this.css("left", (w.width()-this.width())  /2 + "px");
         } else if ('absolute' == pos){
             this.css("top",(w.height()-this.height())/2+w.scrollTop() + "px");
             if(w.height() < this.height())
                 this.css("top", w.scrollTop() + "px");
             this.css("left",(w.width()-this.width())/2+w.scrollLeft() + "px");
         }
         return this;
     }

      function TranslateByGwt(lang)
     {
         var frame = jQuery('.goog-te-menu-frame').first();
 
         if (!frame.size())
         {
             alert("Error: Could not find Google translate frame.");
             return false;
         }
 
         frame.contents().find('.goog-te-menu2-item span').each(function() {
             if (IsLang(lang, jQuery(this).text()))
             {
                 jQuery(this).click();
             }
         });
 
         return false;
     }
 
     function ShowOriginal()
     {
         var frame = jQuery('.goog-te-banner-frame').first();
 
         if (!frame.size())
         {
             return false;
         }
 
         frame.contents().find('div.goog-te-button [id*="cancel"]').each(function()
         {
             jQuery(this).click();
         });
 
         frame.contents().find('div.goog-te-button [id*="restore"]').each(function()
         {
             jQuery(this).click();
         });
 
         frame.contents().find('a.goog-close-link img').each(function()
         {
             jQuery(this).click();
         });
 
         return false;
     }
 
     function IsLang(code, text)
     {
         text = text.toLowerCase();
 
         if (code == 'uk')
         {
             return (text.indexOf('') >= 0 || text.indexOf('ukr') >= 0 || text.indexOf('ucra') >= 0)
         }
 
         if (code == 'ru')
         {
             return (text.indexOf('') >= 0 || text.indexOf('') >= 0 || text.indexOf('russ') >= 0 || text.indexOf('ruso') >= 0 || text.indexOf('rusa') >= 0)
         }
 
         if (code == 'en')
         {
             return (text.indexOf('') >= 0 || text.indexOf('engl') >= 0 || text.indexOf('angl') >= 0 || text.indexOf('ingl') >= 0)
         }
 
         if (code == 'fr')
         {
             return (text.indexOf('') >= 0 || text.indexOf('fren') >= 0 || text.indexOf('fran') >= 0)
         }
 
         if (code == 'de')
         {
             return (text.indexOf('') >= 0 || text.indexOf('') >= 0 || text.indexOf('germ') >= 0 || text.indexOf('allem') >= 0 || text.indexOf('alem') >= 0 || text.indexOf('deuts') >= 0 || text.indexOf('tedesco') >= 0)
         }
 
         if (code == 'es')
         {
             return (text.indexOf('') >= 0 || text.indexOf('') >= 0 || text.indexOf('span') >= 0 || text.indexOf('espa') >= 0 || text.indexOf('spag') >= 0)
         }
 
         if (code == 'it')
         {
             return (text.indexOf('') >= 0 || text.indexOf('') >= 0 || text.indexOf('ital') >= 0)
         }
 
         if (code == 'pt')
         {
             return (text.indexOf('') >= 0 || text.indexOf('port') >= 0)
         }
 
         return false;
     }
 
     var originalLang = 'en';
 
     function TranslateByGwtUsingSelect()
     {
         var selectedLangCode = jQuery('#gwt_select').val();
 
         if (selectedLangCode == originalLang)
         {
             ShowOriginal();
         }
         else
         {
             TranslateByGwt(selectedLangCode);
         }
 
         return false;
 
     }
