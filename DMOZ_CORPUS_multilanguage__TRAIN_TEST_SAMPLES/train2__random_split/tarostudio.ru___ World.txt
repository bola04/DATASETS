 function web2py_trap_form(action,target) {
    jQuery('#'+target+' form').each(function(i){
       var form=jQuery(this);
       if(!form.hasClass('no_trap'))
         form.submit(function(obj){
          jQuery('.flash').hide().html('');
          web2py_ajax_page('post',action,form.serialize(),target);
          return false;
       });
    });
 }
 
 function web2py_ajax_page(method,action,data,target) {
   jQuery.ajax({'type':method,'url':action,'data':data,
     'beforeSend':function(xhr){
       xhr.setRequestHeader('web2py-component-location',document.location);
       xhr.setRequestHeader('web2py-component-element',target);},
     'complete':function(xhr,text){
       command=xhr.getResponseHeader('web2py-component-command');
       if(command) eval(command);
       flash=xhr.getResponseHeader('web2py-component-flash');
       if(flash) jQuery('.flash').html(flash).slideDown();
       },
     'success': function(text) {
       jQuery('#'+target).html(text);
       web2py_trap_form(action,target);
       web2py_ajax_init();
       }
     });
 }
 function web2py_component(action,target) {
     jQuery(document).ready(function(){ web2py_ajax_page('get',action,null,target); });
 }

     var st = ('0'+num);
     var char10 = st.substr(st.length-2, 1);
     var char1 = st.substr(st.length-1, 1);

   function get_countdown(datestring) {
     var current_datetime = '2018-07-30 17:48:52  +0400';
     // minutes
     var remaining_seconds = (Date.parse(datestring)-Date.parse(current_datetime))/1000;
     var remaining_minutes = Math.floor(remaining_seconds/60);
     var remaining_days = Math.floor(remaining_minutes/24/60);
     remaining_minutes -= remaining_days*24*60;
     var remaining_hours = Math.floor(remaining_minutes/60);
     remaining_minutes -= remaining_hours*60;
     remainder = '';
     if (remaining_days>0) {
       remainder += "<span class='days'>"+plural(remaining_days,"день","дня","дней") + '</span> ';
     };
     if (remaining_days<3) { // >
       remainder += "<span class='time'><span class='hours'>"+plural(remaining_hours,"час","часа","часов") +"</span> ";
       if (remaining_days<1) { // >

 Уровень I.VI. Научившись держать в себе больше энергии, чем это удавалось раньше, человек постепенно переходит в следующий Расклад, где пытается управлять собственными желаниями.
 Здесь в VI Раскладе он обучается и дальше накапливать силу и энергию, постепенно понимая, куда она девается чаще всего. Начинает более разумно смотреть на свои желания, становится уверенным в том, что не все его желания рациональны.
 Правильные желания делают его более безупречным и помогают формировать их таким образом, чтобы они укладывались в особенности Чакральной системы. Каждое желание здесь распределяется сразу по нескольким чакрам, что делает их более привлекательными и обдуманными.
 
 Уровень I.VII.
 Перестав тратить силу на бесполезные желания, человек видит, что у него увеличивается количество побед. Которые в свою очередь, даются ему намного легче, чем раньше.
 Если несколько лет назад каждая победа отнимала у него кучу сил, то теперь на реализацию целей и задач уходит намного меньше энергии. Он уже смотрим на социум совсем с другого ракурса и в его голове появляются дополнительные пространства, которые помогают больше верить в свои силы.
 
 Уровень I. XV.
 Побеждать оказывается гораздо легче, когда ты не спутан глупыми связями и свободен от бездумных обязательств. Изучая XV Расклад, практикующий анализирует качество и количество собственных привязок.
 Пересматривая ниточки, которыми он связан с другими людьми, человек понимает, что нужно их чистить, отслеживать, усиливать или ослаблять. Будучи привязан к кому-то или к чему-то, он старается улучшить качества этой связи, если видит в ней необходимость и рациональное зерно.
 
 И наоборот, осознавая неэффективность некоторых собственных «цепей», он без сожаления и страха избавляется от них, становясь более свободным.
 
 На определенном уровне он уже видит, что большинство связей делают его сильнее и мудрее и он понимает, что совсем не зря изучал XV Расклад.
 
 
 
 Уровень I. XVIII.
 
 Дальше приходит понимание того, что пришло время выходить на собственный информационный канал, двигаясь по индивидуальному Пути к «своим» Башням.
 
 В XVIII Раскладе он ищет специальную Информацию, которая поможет открыть в себе Профессионала в Таро и благодаря ей наладить собственный поток знаний. Появляются индивидуальные подключки к информационным пространствам. Усиливается связь с эгрегорами.
 
 Сознательное движение по Пути все дальше отодвигает от него старые привычки и внутренние комплексы. Развитие становится одной из его главных привычек, которую он в себе культивирует различными магическими практиками.
 
 Жизнь практикующего становится намного интересней и вновь появляется понимание того, что на этом уровне человек снова ощущает нехватку энергии.
 
 
 
 Уровень I. XVII.
 
 На прохождение предыдущих этапов Пути в Таро, человек потратил много времени и сил, стал чуточку старше и на свой организм и здоровье смотрит теперь более осознанно.
 
 Изучая себя через призму XVIII Расклада, практикующий по новому видит свои энергетические оттоки и пытается минимизировать потери силы, которая через них уходит из организма.
 
 Просматривает «хвосты», чистит энергетику и внутренние органы, овладевает навыками быть в хорошей физической форме, на данной жизненном этапе.
 
 
 
 Уровень I.X.
 
 
 
 В X Раскладе человек старается понять, что хотят от него Силы Судьбы и как сделать так, чтобы его взаимодействие с ними стало более продуктивным.
 
 Задумываясь над этим вопросом, он ощущает, что как будто бы стал немного удачливым в жизни. Ему начинает казаться, что энергий успеха и везения становится в нем больше.
 
 И постепенно ему в голову приходит мысль, что этими потоками Фортуны можно как-то управлять. И он начинает этим заниматься, собирая статистику, накапливая опыт, изучая принципы и тенденции работы с Силами Судьбы.">
