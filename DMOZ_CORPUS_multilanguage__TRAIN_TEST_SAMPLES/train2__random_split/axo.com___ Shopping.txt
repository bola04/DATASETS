            //slide function  
     function MediaCrowdSlide(where, n){
         var $ = jQuery;
         n = +n;
         //get the item width
         var item_width = $('#media-crowd ul li').outerWidth() + 10;
         item_width = item_width * n;
         
         /* using a if statement and the where variable check 
         we will check where the user wants to slide (left or right)*/
         if(where == 'left'){
             //...calculating the new left indent of the unordered list (ul) for left sliding
             var left_indent = parseInt($('#media-crowd ul').css('left')) + item_width;
         }else{
             //...calculating the new left indent of the unordered list (ul) for right sliding
             var left_indent = parseInt($('#media-crowd ul').css('left')) - item_width;
         }
         
         
         //make the sliding effect using jQuery's animate function... '
         $('#media-crowd ul:not(:animated)').animate({'left' : left_indent},500,function(){    
             
             /* when the animation finishes use the if statement again, and make an ilussion
             of infinity by changing place of last or first item*/
             if(where == 'left'){
                 //...and if it slided to left we put the last item before the first item
 
                 $('#media-crowd ul li:first').before($('#media-crowd ul li').slice(-n));
             }else{
                 //...and if it slided to right we put the first item after the last item
                 $('#media-crowd ul li:last').after($('#media-crowd ul li').slice(0,n)); 
             }
             
             //...and then just get back the default left indent
             $('#media-crowd ul').css({'left' : '-' + item_width + 'px'});
         });       
     }

         in milliseconds ( multiplied with 1000) '*/
         if( $('#media-crowd ul li').length > 7 ){
             //console.log('media crowd carousel')
             /*move he last list item before the first item. The purpose of this is 
             if the user clicks to slide left he will be able to see the last item.*/
             $('#media-crowd ul li:first').before($('#media-crowd ul li').slice(-2)); 
             
             //check if auto sliding is enabled
             if(auto_slide == 1){
                 /*set the interval (loop) to call function slide with option 'right' 
                 and set the interval time to the variable we declared previously */
                 var timer = setInterval('MediaCrowdSlide("right", 2)', auto_slide_seconds); 
                 
                 /*and change the value of our hidden field that hold info about
                 the interval, setting it to the number of milliseconds we declared previously*/
                 $('#hidden_auto_slide_seconds').val(auto_slide_seconds);
             }
       
             //check if hover pause is enabled
             if(hover_pause == 1){
                 //when hovered over the list 
                 $('#media-crowd ul').hover(function(){
                     //stop the interval
                     clearInterval(timer)
                 },function(){
                     //and when mouseout start it again
                     timer = setInterval('MediaCrowdSlide("right", 2)', auto_slide_seconds); 
                 });
       
             }
         } else {
             $('#media-crowd ul').css('left', '0px');
             $('#media-crowd #left_scroll, #media-crowd #right_scroll').hide()
         }
