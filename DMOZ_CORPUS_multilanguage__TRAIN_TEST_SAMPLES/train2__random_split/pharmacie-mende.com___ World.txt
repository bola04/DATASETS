  function no_accent(texte)
  {
   texte=texte.replace(/[ČÉĘËčéęë]/g, "e");
   texte=texte.replace(/[ĎÎďî]/g, "i");
   texte=texte.replace(/[ŮÜŰůüű]/g, "u");

   texte=texte.replace(/[ç]/g, "c");
 
   return texte;
  }
 
  function isEmail(mailObj)
  {
   Mail = mailObj.value.toLowerCase();
   if(Mail != "")
   {
    Mail = no_accent(Mail);
    if(Mail.search(/^[0-9a-z]([-_.]?[0-9a-z])*@[0-9a-z]([-.]?[0-9a-z])*\.[a-z]{2,4}$/) == -1)
    {
     document.getElementById("reponseNL").innerHTML = "E-mail non valide.";
     return "";
    }
    else
    {    
     document.getElementById("reponseNL").innerHTML = "";
     return mailObj.value;
    }
   }
   else
   {
    mailObj.value = mailObj.defaultValue;
    return " ";
   }
  }
