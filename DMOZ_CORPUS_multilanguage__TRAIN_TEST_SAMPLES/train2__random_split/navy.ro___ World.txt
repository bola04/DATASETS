 		function fillNews() {
 			$.ajax({url:"gettopnews.php", cache:false, dataType:"json", type:"get",
 				 success:function(data){               
                     if(data.err!=='')
                         alert(data.err);
                     else{
                         var str = '';
 						nrcrt = 0;
                         $.each(data.items, function(key,value){
 							nrcrt++;
 							str += '<div class="order">';
 							str += '<div class="date-order">';
 							var d = new Date(value.data);
 							var day = d.getDate();
 							var month = d.getMonth();
                             str += '<span style="font-size: 24px; font-weight: bold;">';
 							if (day<10) str += '0';
 							str += day+'</span><br/>';
 							str += luna_abr[month]+'</div>';
 							str += '<div class="detail"><a href="http://www.navy.ro/comunicat.php?id='+value.id+'">'+value.titlu+'</a></div>';
 							str += '</div>';
                         });
                         if (nrcrt===0) str += '<div>Nu există comunicate.</div>';
                         $('#bodyNews').html(str);

                     }
                 }, error:function(a,b,c){
                     alert(this.url+': '+c);
                 }
             });
 		}
 		
 		function fillEvents() {
 			$.ajax({url:"gettopevents.php", cache:false, dataType:"json", type:"get",
 				 success:function(data){               
                     if(data.err!=='')
                         alert(data.err);
                     else{
                         var str = '';
 						nrcrt = 0;
                         $.each(data.items, function(key,value){
 							nrcrt++;
 							str += '<div class="row">';
 							str += '<div class="event-photo"><a style="display:block;line-height:1;font-size:0;outline:none;" href="http://www.navy.ro/eveniment.php?id='+value.id+'">';
 							str += '<img src="events/'+value.id+'/cover.jpg" /></a></div>';
 							str += '<div class="event-detail"><a href="http://www.navy.ro/eveniment.php?id='+value.id+'" class="transition">'+value.title+'</a>';
 							str += '<p><span class="tag">'+formatDate(value.data, true, false)+'</span> <a class="decoration-none" title="Vezi album foto" href="http://www.navy.ro/eveniment.php?id='+value.id+'"><span class="tag" style="background:#008c00;">'+value.cnt+' fotografii</span></a>';
 							if (value.idcomunicat>0)
 								str += '&nbsp;<a title="Vezi comunicat de presă" class="decoration-none" href="comunicat.php?id='+value.idcomunicat+'"><span class="tag" style="background:#e51400;">Comunicat</tag></a>';
 							str += '</p></div>';
 							str += '</div>';
                         });
                         if (nrcrt===0) str += '<div>Nu există evenimente.</div>';
                         $('#bodyEvents').html(str);
                         return;
                     }
                 }, error:function(a,b,c){
                     alert(this.url+': '+c);
                 }
             });
 		}
 		function formatDate(date, showyear, showtime) {
 			var isodate = new Date(date);
 			var monthNames = ["ianuarie", "februarie", "martie", "aprilie", "mai", "iunie", "iulie", "august", "septembrie", "octombrie", "noiembrie", "decembrie"];
 			var res = '';
 			res += isodate.getDate() + ' ' + monthNames[isodate.getMonth()];
 			if (showyear) {
                 res += ' ' + isodate.getFullYear();
             }
 			if (showtime) {
 				if (isodate.getHours()<10)
 					res += ', 0' + isodate.getHours()+':';
 				else
 					res += ', ' + isodate.getHours()+':';
 				if (isodate.getMinutes()<10) res += '0';
 				res += isodate.getMinutes();
 				
             }
 			return res;
 		}
