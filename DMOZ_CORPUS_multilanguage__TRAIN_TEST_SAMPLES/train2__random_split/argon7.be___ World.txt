    <script type="text/javascript">
     $(document).ready(function() {
 
   //Get size of the image, how many images there are, then determin the size of the image reel.
   //var imageWidth = $("#rotatedivimg").width();
   //var imageSum = $(".rotatedivimgreel img").size();
   //var imageReelWidth = imageWidth * imageSum;
   
   //Adjust the image reel to its new size and show the img div
   //$(".rotatedivimgreel").css({'width' : imageReelWidth});
   $("#rotatedivimg").show();
   
   
   //Show the txt div and activate its first div
   $("#rotatedivtxt div:first").addClass("rotateactive");
   $("#rotatedivtxt").show();
   
   //Paging  and Slider Function
   rotate = function(){
       var rel = $active.attr("rel"); //Get the numbering of current active
       //var triggerID = $active.attr("rel") - 1; //Get number of times to slide
       //var image_reelPosition = triggerID * imageWidth; //Determines the distance the image reel needs to slide
   
       $("#rotatedivtxt div").removeClass('rotateactive'); //Remove all active class
       $active.addClass('rotateactive'); //Add active class (the $active is declared in the rotateSwitch function)
   
       //Slider Animation
       //$(".rotatedivimgreel").animate({ left: -image_reelPosition }, 200 );
       // hide all images, show active one
       $("#rotatedivimg a.rot").hide();
       $('#rotateimg'+rel).show(); 
   }; 
   
   //Rotation  and Timing Event
   rotateSwitch = function(){
       play = setInterval(function(){ //Set timer - this will repeat itself every 7 seconds
           $active = $('#rotatedivtxt div.rotateactive').next(); //Move to the next paging
           if ( $active.length === 0) { //If paging reaches the end...
               $active = $('#rotatedivtxt div:first'); //go back to first
           }
           rotate(); //Trigger the paging and slider function
       }, 4000); //Timer speed in milliseconds (3 seconds)
   };
   
   rotateSwitch(); //Run function on launch
   
   var myTimeout;
 
   //On Hover
   $(".rotatedivimgreel img").hover(function() {
     if(myTimeout) window.clearTimeout(myTimeout);
     clearInterval(play); //Stop the rotation
   }, function() {
     if(myTimeout) window.clearTimeout(myTimeout);
     myTimeout = window.setTimeout(function() {
       rotateSwitch(); //Resume rotation timer
     }, 500);
   });
   $("#rotatedivtxt div").hover(function() {
     if(myTimeout) window.clearTimeout(myTimeout);
     clearInterval(play); //Stop the rotation
     $active = $(this); // Activate div
     rotate(); // Trigger rotation immediately
   }, function() {
     if(myTimeout) window.clearTimeout(myTimeout);
     myTimeout = window.setTimeout(function() {
       rotateSwitch(); //Resume rotation timer
     }, 500);
   });
 
 });
    </script>
