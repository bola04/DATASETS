   function showBanner() {
             var $banner = document.getElementById('cookies-banner');
             var result = document.cookie.match ( '(^|;) ?onelogin_cafesoft_check_cookie=([^;]*)(;|$)' );
             cookie = result ? unescape(result[2]) : null;
 
             if(!cookie) {
                 var date = new Date();
                 date.setTime(date.getTime()+(7*24*60*60*1000));
                 var expires = "; expires="+date.toGMTString();
                 document.cookie = "onelogin_cafesoft_check_cookie=525be6d005b0bef18d00a41d4a33de77" + expires + "; path=/";
                 $banner.style.visibility = 'visible';
                 document.querySelector('#main-header').style.top = $banner.clientHeight + 'px';
             }
         }
 
         function closeBanner() {
             var $banner = document.getElementById('cookies-banner');
             $banner.style.visibility = 'hidden';
             document.querySelector('#main-header').style.top = 0;
 
             return false;
         }
 
         window.onload = showBanner;
