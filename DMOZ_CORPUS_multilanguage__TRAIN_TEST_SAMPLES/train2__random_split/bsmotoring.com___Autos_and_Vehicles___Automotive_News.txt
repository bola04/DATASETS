                         /*This is a navbar / full site search box code */
                             function checkForm()
                             {
                                 
                                 
                                 var query_string=$('#query1').val();
                                 if(query_string!='')
                                 {
               
                                     query_string=query_string.replace();
                                     var patt1 = /[^a-zA-Z0-9$%&+@ ]/g;
                                     query_string = query_string.replace(patt1,'');
                                     query_string=query_string.replace(/ /gi,'_');
                                     query_string=query_string.toLowerCase();
                                         //alert(query_string);
                                     window.location.replace("http://www.bsmotoring.com/search/"+encodeURI(query_string));
                                 }
                                 else
                                 {
                                     alert("Please type search query");
                                     return false;
                                 }     
                             }
                             
                             var lastcall =''; 
                             var readmore_ajax_status=0;
                             var ajax_call_variable = '';
                             $('#query1').keyup(function(e){
                                 var txtBoxValue = $(this).val();
                                 if ( e.keyCode === 27 ) { // ESC pressed hide modal
                                     $( '#search_close' ).trigger('click');
                                     $('#search_articles').html('');
                                     $('#search_images').html('');
                                     $(this).val('');
                                     }else if ( e.keyCode === 13 ) { 
                                         // Enter Key pressed hide modal and take to link
                                     $( '#search_close' ).trigger('click');    
                                     e.preventDefault();
                                     checkForm();
                                     }else if ((e.keyCode >= 37 && e.keyCode <= 40) || e.keyCode == 8 ){
                                         // if the length of textbox is empty then make results empty
                                         if(txtBoxValue.length==0){
                                             $('#search_articles').html("<li style=\"border-top:0;\">Type Search Query</li>");		    
                                             $('#search_images').html("<li style=\"border-top:0;\">Type Search Query</li>");	
                                         }
                                         // 8 : backspace , 37 : left, 38 : up , 39 : right & 40 : down
                                         // for arrow keys and backspace keys
                                         //dont make an ajax call just do noting
                                     } else{
                                         var inp = String.fromCharCode(e.keyCode);
                                         if (/[a-zA-Z0-9-_,' ]/.test(inp))
                                         {
                                         
                                         if(ajax_call_variable!='' )
                                             ajax_call_variable.abort();
                                         var currenttime = new Date();
                                         currenttime = parseInt(currenttime.getTime()/1000);
                                         //console.log(currenttime-lastcall);
                                         if (currenttime-lastcall > 0.1 && txtBoxValue.length>=3 && (parseInt(txtBoxValue.length)+1)%2 === 0  ) {
                                              ajax_call_variable = $.ajax({
 						url: 'http://www.bsmotoring.com/ajax_call/getsearchresults',
 						type:'post',
 						data:{ query: txtBoxValue},
 						dataType: 'html',			
 						beforeSend:function(){ readmore_ajax_status=1; $('#loading_div').show(); },
 						success:function(data){
 							
 						data = data.split('::::::');
                                                 if(typeof data[0] != 'undefined' && data[0].length>0){
                                                 $('#search_articles').html(data[0]);
                                                 if(typeof data[1] != 'undefined' && parseInt(data[1])==10){
                                                     $('#view_more_news').show();
                                                 }else{
                                                     $('#view_more_news').hide();
                                                 }
                                                 }else{
                                                 $('#search_articles').html("<li style=\"border-top:0;\">No results found</li>");		    
                                                 }
 						if(typeof data[2] != 'undefined' && data[2].length>0){	
 						$('#search_images').html(data[2]);		
                                                 }else{
                                                 $('#search_images').html("<li style=\"border-top:0;\">No results found</li>");		    
                                                 }	
 						    
 						$('#loading_div').hide();
 						},
 						error:function(){readmore_ajax_status=0;$('#loading_div').hide();},
 						complete:function(){readmore_ajax_status=0;$('#loading_div').hide();}						
 					 });
 						lastcall = currenttime;
                                         }
                                         }
                                     }
                                 
                             })
