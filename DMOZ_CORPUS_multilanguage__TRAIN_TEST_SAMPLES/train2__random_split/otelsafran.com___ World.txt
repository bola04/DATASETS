 function rezerve(){
 	
 var cin = $("#checkin").val();
 var cout = $("#checkout").val();
 var cocuk = $("#childrens").val();
 var adults = $("#adults").val();	
 
 if(cin!="" && cout!=""){
 
 var startd = moment(cin,'DD/MM/YYYY').format('YYYY-MM-DD');
 var endd = moment(cout,'DD/MM/YYYY').format('YYYY-MM-DD');
 
 gece = parseInt(moment(endd).diff(startd, "days"));
 
 }
 
 if(gece < 0){
 alert('Çıkış Tarihi, Giriş Tarihinden Küçük Olamaz !');	
 }else{
 	
 window.location.href = '/rezervasyon?checkin=' + cin + '&checkout=' + cout + '&children=' + cocuk + '&adults=' + adults;	
 	
 }
 }
