        $(document).ready(function(){

 		var autoTimer = "";
 
 		rotatorWidth = 0;
 		$("#logoRotator ul li").each(function(){
 			rotatorWidth = rotatorWidth + $(this).width();
 		});
 		
 		$("#logoRotator ul").css("width",rotatorWidth+"px");
 		if(rotatorWidth > 800){
 			$("#logoRotatorLeftArrow").fadeIn();
 		}
 
 		
 		//rorate right on internal
 		
 		
 		rotateRight = function(){
 			containerOffset = $("#logoRotator ul").position();
 			maxOffset = $("#logoRotator ul").width() - 800;
 			if(containerOffset.left > -maxOffset){
 				$("#logoRotator ul").animate({"left":"-=3px"},0);
 			}else{
 				//animate back to start
 				$("#logoRotatorLeftArrow").fadeOut();
 				$("#logoRotator ul").animate(
 				{
 					"left" : "0"
 				}, 500);
 			}
 		}
 		rotateLeft = function(){
 			clearInterval(autoTimer);
 			containerOffset = $("#logoRotator ul").position();
 			maxOffset = 0;
 			if(containerOffset.left < maxOffset){
 				$("#logoRotator ul").animate({"left":"+=3px"},0);
 			}else{
 				$("#logoRotatorRightArrow").fadeOut();
 			}
 		}
 
 		beginAutoScroll = function() {
 			autoTimer = setInterval("rotateRight()", 35);
 		}
 
 		beginAutoScroll();
 
 		$("#logoRotator ul li").hover(function() {
 			//pause scroll
 			clearInterval(autoTimer);
 		}, function() {
 			//resume scroll
 			beginAutoScroll();
 		});
 		
 		$("#logoRotatorRightArrow").hover(function(){
 			clearInterval(autoTimer);
 			$("#logoRotatorLeftArrow").show();
 			clearInterval(intId);
 			intId = setInterval("rotateLeft()",5);
 			return false;
 		},function(){
 			clearInterval(intId);
 			beginAutoScroll();
 		});
 		
 		$("#logoRotatorLeftArrow").hover(function(){
 			clearInterval(autoTimer);
 			$("#logoRotatorRightArrow").show();
 			clearInterval(intId);
 			intId = setInterval("rotateRight()",5);
 			return false;
 		},function(){
 			clearInterval(intId);
 			beginAutoScroll();
 		});
 	});
