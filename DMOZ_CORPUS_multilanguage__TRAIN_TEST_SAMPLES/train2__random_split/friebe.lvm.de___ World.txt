 				function startKontaktformularForUrlParam(){
 					var openContactParam = getParameterByName("openContact");
 					if(openContactParam == 1){
 						var anliegen = getParameterByName("anliegen");
 						var produktart= getParameterByName("produktart");
 						
 						if(anliegen != null && anliegen=='versicherung' && produktart == 'krankenversicherung'){
 							anliegen = 1;
 							produktart = 4;
 						}
 						
 						if(anliegen != null && jQuery.isNumeric(anliegen) && anliegen >= 0 && anliegen < 10){
 							//produktart optional
 							if(produktart != null && jQuery.isNumeric(produktart)&& produktart >= 0 && produktart < 14){
 								kontaktformularStartByAnliegenProduktart(true, anliegen, produktart);
 							}else{
 								kontaktformularStartByAnliegenProduktart(true, anliegen);
 							}
 						}else{
 							kontaktformularStart();
 						}
 					}
 				}
