 function slideSwitch() {
     var $active = $('#slideshow IMG.active');
 
     if ( $active.length == 0 ) $active = $('#slideshow IMG:last');
 
     // use this to pull the images in the order they appear in the markup
     var $next =  $active.next().length ? $active.next()
         : $('#slideshow IMG:first');
 
     // uncomment the 3 lines below to pull the images in random order
     
     // var $sibs  = $active.siblings();
     // var rndNum = Math.floor(Math.random() * $sibs.length );
     // var $next  = $( $sibs[ rndNum ] );
 
 
     $active.addClass('last-active');
 
     $next.css({opacity: 0.0})
         .addClass('active')
         .animate({opacity: 1.0}, 1500, function() {
             $active.removeClass('active last-active');
         });
 }
 
 $(function() {
     setInterval( "slideSwitch()", 5000 );
 });

      <h2>
       Pine and Everything Else...
      </h2>
      <p>
       Don’t let the name fool you. From humble beginnings Pine Door has evolved into being a premier high-quality custom manufacturer of custom solid wood doors and millwork in species ranging from traditional pine, fir, alder, and oak to more exotic species like afrimosia,  teak, mahogany, or even full natural patina reclaimed oak and fir. Our doors are available as raised panel, flat panel, flush plank, louver, or various glass option styles for both residential and commercial projects.
      </p>
      <h2>
       Wood - nature's sustainable resource...
      </h2>
      <p>
       Trees stand for centuries. Is it any wonder wood makes the most beautiful and durable doors? After milling, every board reveals a vivid record of the events of its lifetime. Growth rings, knots, grain patterns, color and shape are all influenced by powerful natural forces in the local environment. As a result, the beauty, texture, color and grain are unique to each species and to each board.
      </p>
      <h2>
       100% American Made...
      </h2>
      <p>
       Here at Pine Door old world craftsmanship is blended with new world manufacturing technologies to provide you with unsurpassed quality, quick lead times, and competitive prices.  All aspects of material grading and manufacturing for our custom  wood doors and millwork are performed in house by our craftsmen at our facility in Darby, Montana.
      </p>
