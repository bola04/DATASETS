 							curLength = innerHTMLInfo.length;
 
 								if(curLength > 0 )
 								{

 									//console.log("In Function: " + loadingDiv.attr('id') + " VAL: " + resultsFound + " LENGTH " + innerHTMLInfo.length + " HTML |" + innerHTMLInfo +"|");

 									foundResults = true;
 									$("#no-results").hide();
 
 
 								}
 								else
 								{containingDiv.hide(1000);}
 
 							});
 
 	}
