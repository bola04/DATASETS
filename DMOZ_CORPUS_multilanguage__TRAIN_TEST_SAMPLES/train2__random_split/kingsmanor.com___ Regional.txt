        <script type="text/javascript">
         $(document).ready(function(){
 
         $( "#check_in" ).datepicker({
             maxDate: 350,
             dateFormat: "dd-mm-yy"
         });
 
         $('.room_booker #room_summary').live( 'click' , function(e) {
             e.preventDefault();
             $('.room_selecter').show();
         });
 
         $('.room_booker #occupancy-close-cross, .room_booker #close_button').live( 'click' , function(e) {
             e.preventDefault();
             $('.room_selecter').hide();
         });
 
         $('.room_booker #rooms').live( 'change' , function() {
 
             var rooms = parseInt( $(this).val() , 10 );
 
             if ( rooms === 1 ) {
                 $('#room_1').show();
                 $('#room_2,#room_3').hide();
             } else if ( rooms === 2 ) {
                 $('#room_1,#room_2').show();
                 $('#room_3').hide();
             } else {
                 $('#room_1,#room_2,#room_3').show();
             }
 
         });
 
         $('.room_selection select').live( 'change' , function() {
 
             var totalRooms = parseInt( $('.room_booker #rooms').val() , 10 );
             var totalAdults = 0;
             var totalChildren = 0;
 
              if ( totalRooms === 1 ) {
                 totalAdults += parseInt( $('#room_1 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_1 .children').val() , 10 );
             } else if ( totalRooms === 2 ) {
                 totalAdults += parseInt( $('#room_1 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_1 .children').val() , 10 );
                 totalAdults += parseInt( $('#room_2 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_2 .children').val() , 10 );
             } else {
                 totalAdults += parseInt( $('#room_1 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_1 .children').val() , 10 );
                 totalAdults += parseInt( $('#room_2 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_2 .children').val() , 10 );
                 totalAdults += parseInt( $('#room_3 .adults').val() , 10 );
                 totalChildren += parseInt( $('#room_3 .children').val() , 10 );
             }
 
             $('#room_summary').val( totalRooms + " Room, " + totalAdults + " Adults, " + totalChildren + " Children" );
 
         });
 
         $('.room_booker').live( 'submit' , function(e) {
 
             e.preventDefault();
 
             // Build the url...
             var url = 'https://book.isiteplus.co.uk/room-select/property-83448';
 
             // Check in date...
             url += '/checkin-' + $('.room_booker #check_in').val();
             
             // No. nights...
             url += '/nights-' + $('.room_booker #nights').val();
 
             // Rooms
             var rooms = parseInt( $('.room_booker #rooms').val() , 10 );
 
             var room_1 = '/room-1-adults-' + $('.room_booker select[name="room[1][adults]"]').val() + '-children-' + $('.room_booker select[name="room[1][children]"]').val();
             var room_2 = '/room-2-adults-' + $('.room_booker select[name="room[2][adults]"]').val() + '-children-' + $('.room_booker select[name="room[2][children]"]').val();
             var room_3 = '/room-3-adults-' + $('.room_booker select[name="room[3][adults]"]').val() + '-children-' + $('.room_booker select[name="room[3][children]"]').val();
 
             if ( rooms === 1 ) {
                 url += room_1;
             } else if ( rooms === 2 ) {
                 url += room_1 + room_2;
             } else if ( rooms === 3 ) {
                 url += room_1 + room_2 + room_3;
             }
 
             window.open( url );
 
         });
 
 
     });
        </script>
