 function isDecimal(cadena)
 {
   var checkOK = "0123456789";
   var allValid = true;
   var allNum = "";
   for (i = 0;  i < cadena.length;  i++)
   {
     ch = cadena.charAt(i);
     for (j = 0;  j < checkOK.length;  j++)
       if (ch == checkOK.charAt(j) || ch==',')
         break;
 
     if (j == checkOK.length)
     {
       allValid = false;
       break;
     }
     allNum += ch;
   }
   
   if(allValid == true)
         return (true);
   if(allValid == false)
         return (false);
 }
 
 function ponerpuntos(cadena)
 {
  if (cadena == 0) 
  {
    return ("Consultar");
  }
 
 
  if (isDecimal(cadena)) 
  {
     if (cadena.indexOf(',') > 0)  
     {
        resultado = cadena.substr(cadena.indexOf(','));
        temporal = cadena.substr(0,(cadena.indexOf(',')));
     }
     else
     {
        resultado = "";
        temporal = cadena;
     }
     if (temporal.length - 3 < 0)
        {
          resultado = temporal.substr(0,3) + resultado;
        }
        else
        {
          resultado = temporal.substr(temporal.length - 3,3) + resultado;
        }
     temporal = temporal.substr(0,temporal.length - 3);
     for (var i = (temporal.length - 3) ; i >= -2 ; i=i-3)
     {
        if ( i < 0)
           {
              resultado = temporal.substr(0,3) + '.' + resultado;
           }
           else
           {
              resultado = temporal.substr(i,3) + '.' + resultado;
           }
        temporal = temporal.substr(0,i);
     }
     resultado = temporal + resultado;
     return (resultado);
  }
  else
  {
     return (cadena);
  }
 }

 function cambiofechatotexto(fecha)

 }

 function URLEncode( F1)
 {
 
 	// caracteres que no se tendran en cuenta
 
 	var SAFECHARS = "0123456789" + "ABCDEFGHIJKLMNOPQRSTUVWXYZ" + "abcdefghijklmnopqrstuvwxyz" + "?&=/%#@!~*'()";
 	var HEX = "0123456789ABCDEF";
 
 	var plaintext = F1;
 	var encoded = "";
 
 	for (var i = 0; i < plaintext.length; i++ ) {
 		var ch = plaintext.charAt(i);
 	    if (ch == " ") {
 		    encoded += "%20";
 		} else if (SAFECHARS.indexOf(ch) != -1) {
 		    encoded += ch;
 		} else {
 		    var charCode = ch.charCodeAt(0);
 			if (charCode > 255) {
 			    alert( "Unicode Character '" 
                         + ch 
                         + "' No puedo encode.\n" +
 				          "(URL encode solo soporta 8-bit caracteres.)\n" +
 						  "Un (+) sera sustituido." );
 				encoded += "+";
 			} else {
 				encoded += "%";
 				encoded += HEX.charAt((charCode >> 4) & 0xF);
 				encoded += HEX.charAt(charCode & 0xF);
 			}
 		}
 	} // for
 
 
 	
 	return encoded;
 }

 function isDecimal(cadena)
 {
   var checkOK = "0123456789";
   var allValid = true;
   var allNum = "";
   for (i = 0;  i < cadena.length;  i++)
   {
     ch = cadena.charAt(i);
     for (j = 0;  j < checkOK.length;  j++)
       if (ch == checkOK.charAt(j) || ch==',')
         break;
 
     if (j == checkOK.length)
     {
       allValid = false;
       break;
     }
     allNum += ch;
   }
   
   if(allValid == true)
         return (true);
   if(allValid == false)
         return (false);
 }
 
 function ponerpuntos(cadena)
 {
  if (cadena == 0) 
  {
    return ("Consultar");
  }
 
 
  if (isDecimal(cadena)) 
  {
     if (cadena.indexOf(',') > 0)  
     {
        resultado = cadena.substr(cadena.indexOf(','));
        temporal = cadena.substr(0,(cadena.indexOf(',')));
     }
     else
     {
        resultado = "";
        temporal = cadena;
     }
     if (temporal.length - 3 < 0)
        {
          resultado = temporal.substr(0,3) + resultado;
        }
        else
        {
          resultado = temporal.substr(temporal.length - 3,3) + resultado;
        }
     temporal = temporal.substr(0,temporal.length - 3);
     for (var i = (temporal.length - 3) ; i >= -2 ; i=i-3)
     {
        if ( i < 0)
           {
              resultado = temporal.substr(0,3) + '.' + resultado;
           }
           else
           {
              resultado = temporal.substr(i,3) + '.' + resultado;
           }
        temporal = temporal.substr(0,i);
     }
     resultado = temporal + resultado;
     return (resultado);
  }
  else
  {
     return (cadena);
  }
 }

 function cambiofechatotexto(fecha)
 {

   anonumero = new Number(fecha.substr(6,4));

 }

 function URLEncode( F1)
 {
 
 	// caracteres que no se tendran en cuenta
 
 	var SAFECHARS = "0123456789" + "ABCDEFGHIJKLMNOPQRSTUVWXYZ" + "abcdefghijklmnopqrstuvwxyz" + "?&=/%#@!~*'()";
 	var HEX = "0123456789ABCDEF";
 
 	var plaintext = F1;
 	var encoded = "";
 
 	for (var i = 0; i < plaintext.length; i++ ) {
 		var ch = plaintext.charAt(i);
 	    if (ch == " ") {
 		    encoded += "%20";
 		} else if (SAFECHARS.indexOf(ch) != -1) {
 		    encoded += ch;
 		} else {
 		    var charCode = ch.charCodeAt(0);
 			if (charCode > 255) {
 			    alert( "Unicode Character '" 
                         + ch 
                         + "' No puedo encode.\n" +
 				          "(URL encode solo soporta 8-bit caracteres.)\n" +
 						  "Un (+) sera sustituido." );
 				encoded += "+";
 			} else {
 				encoded += "%";
 				encoded += HEX.charAt((charCode >> 4) & 0xF);
 				encoded += HEX.charAt(charCode & 0xF);
 			}
 		}
 	} // for
 
 
 	
 	return encoded;
 }
