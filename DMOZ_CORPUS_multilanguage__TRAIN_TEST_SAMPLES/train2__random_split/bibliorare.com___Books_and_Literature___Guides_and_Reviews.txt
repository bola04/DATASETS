           (function($) {
 	$(function () {
 		$('#viewLoading').fadeOut(0);
 		$('#viewRegister').fadeOut(0);
 		$('#viewLostPassword').fadeOut(0);
 		$('.error').fadeOut(0);
 		$('.message').fadeOut(0);
 
 	    $('#fancyLogin').fancybox({
 	    	'autoDimensions': true,
 	    	'autoScale'		: true,
 			'scrolling'		: 'no',
 			'padding'		: 0,
 			'title'			: "",
 			'titleShow'		: false,
 			'titlePosition'	: 'over',
 			'onStart'		: function() {
 			    $('#viewLogin').fadeIn(0);
 			    $('#viewRegister').fadeOut(0);
 				$('#viewLostPassword').fadeOut(0);
 				$('.error').fadeOut(0);
 				$('.message').fadeOut(0);
 			},
 			'onClosed'		: function() {
 			    //$('#login_error').hide();
 			}
 		});
 
 		$('.show_viewLogin').live( 'click', function(event) {
 			event.preventDefault();
 
 			$('#viewRegister').fadeOut('fast', function() {
 				$('#viewLostPassword').fadeOut('fast', function() {
 					$('#viewLogin').fadeIn('fast');
 				});
 			});
 		});
 
 		$('.show_viewRegister').live( 'click', function(event) {
 			event.preventDefault();
 
 			$('#viewLogin').fadeOut('fast', function() {
 				$('#viewLostPassword').fadeOut('fast', function() {
 					$('#viewRegister').fadeIn('fast');
 				});
 			});
 		});
 
 		$('.show_viewLostPassword').live( 'click', function(event) {
 			event.preventDefault();
 
 			$('#viewLogin').fadeOut('fast', function() {
 				$('#viewRegister').fadeOut('fast', function() {
 					$('#viewLostPassword').fadeIn('fast');
 				});
 			});
 
 		});
 
 		/**
 		 * Surcharge de la validation du formulaire de Login
 		 * @param  {[event]} event [description]
 		 * @return 
 		 */
 		$('#viewLogin').bind('submit', function(event) {
 			event.preventDefault();
 
 			var user = $('#user_login').val();
 			var pass = $('#user_pass').val();
 			var rememberme = 0;
 			if($('input[type=checkbox][name=rememberme]:checked').val()) {
 				rememberme = 1;
 			}
 			login(user, pass, rememberme);
 		});
 		
 		/**
 		 * Surcharge de la validation du formulaire de Création de compte
 		 * @param  {[event]} event [description]
 		 * @return 
 		 */
 		$('#viewRegister').bind('submit', function(event) {
 			event.preventDefault();
 
 			var user = $('#register_user_login').val();
 			var mail = $('#register_user_email').val();
 			var newsletter = 0;
 			if($('input[type=checkbox][name=newsletter_subscribe]:checked').val()) {
 				newsletter = 1;
 			}
 			register(user, mail, newsletter);
 		});
 		
 		/**
 		 * Surcharge de la validation du formulaire de récupération de Mot de passe
 		 * @param  {[event]} event [description]
 		 * @return 
 		 */
 		$('#viewLostPassword').bind('submit', function(event) {
 			event.preventDefault();
 
 			var user = $('#lostpass_user_login').val();
 			req_password(user);
 		});
 
 	});
 
 	function login(user, password, remember) {
 		//weboordercarttoggleloading(true);
 		$('#viewLoading').fadeIn('fast', function() {
 			$.ajax({ 
 				type : 'POST',
 				url : 'http://www.bibliorare.com/wp-admin/admin-ajax.php',
 				dataType : 'json',
 				data: { action : 'gzclabs-ajax-login-submit', cmd: 'login', userlogin: user, userpwd: password, userremember: remember },
 				success : function(data) {
 					$('#viewLoading').fadeOut('fast', function() {
 						if (data.error) {
 							$('#login-error').html(data.message);
 							$('#login-error').fadeIn('medium');
 						} else {
 							// Login OK, rechargement de la page
 							window.location.reload();
 						}
 					});
 				} 
 			});
 		});
 	}
 
 
 	function register(user, mail, newsletter) {
 		//weboordercarttoggleloading(true);
 		$('#viewLoading').fadeIn('fast', function() {
 			console.log("register :: "+user+" :: "+mail+" :: "+newsletter);
 			$.ajax({ 
 				type : 'POST',
 				url : 'http://www.bibliorare.com/wp-admin/admin-ajax.php',
 				dataType : 'json',
 				data: { action : 'gzclabs-ajax-login-submit', cmd: 'register', userlogin: user, usermail: mail, usernewsletter: newsletter },
 				success : function(data) {
 					$('#viewLoading').fadeOut('fast', function() {
 						if (data.error) {
 							$('#register-error').html(data.message);
 							$('#register-message').fadeOut('fast', function() {
 								$('#register-error').fadeIn('medium');
 							});
 						} else {
 							$('#register-message').html(data.message);
 							$('#register-error').fadeOut('fast', function() {
 								$('#register-message').fadeIn('medium');
 							});
 						}
 					});
 				} 
 			});
 		});
 	}
 
 
 	function req_password(user) {
 		$('#viewLoading').fadeIn('fast', function() {
 			$.ajax({ 
 				type : 'POST',
 				url : 'http://www.bibliorare.com/wp-admin/admin-ajax.php',
 				dataType : 'json',
 				data: { action : 'gzclabs-ajax-login-submit', cmd: 'lost_password', user_login: user },
 				success : function(data) {
 					$('#viewLoading').fadeOut('fast', function() {
 						if (data.error) {
 							$('#lostpass-error').html(data.message);
 							$('#lostpass-error').fadeIn('medium');
 						} else {
 							// Login OK, rechargement de la page
 							//window.location.reload();
 							$('#lostpass-message').html(data.message);
 							$('#lostpass-error').fadeOut('fast', function() {
 								$('#lostpass-message').fadeIn('medium');
 							});
 						}
 					});
 				} 
 			});
 		});
 	}
 
 })(jQuery);

        <a class="title-link" href="http://www.bibliorare.com/category/breves/" title="Info du Bibliophile">
         <h1>
          Brèves
         </h1>
        </a>
        <div class="article">
         <p>
          <strong>
           Sympa l’info … Merci
          </strong>
          <img alt="" height="16" role="presentation" src="https://static.xx.fbcdn.net/images/emoji.php/v9/fdc/1.5/16/1f60e.png" width="16"/>
         </p>
         <p>
          <em>
           <strong>
            Une histoire, passionnante, extraordinaire… Bernard Leborgne, fantastique… et Paul Veyssière…
           </strong>
          </em>
         </p>
         <p>
          <strong>
           Le film «
           <em>
            Le faussaire qui aimait trop Galilée
           </em>
           », de Laurent Bergers et Paul-Henri Moinet, sera diffusé sur France 5 dimanche 29 avril 2018 à 9h25.
          </strong>
         </p>
         <div>
          <p>
           Comment Massimo De Caro a-t-il réussi à tromper le monde du livre rare avec une copie remarquable d’un manuscrit de Galilée ? Jusque-là, personne n’avait réussi à reproduire la typographie d’un ouvrage aussi ancien. Récit d’une imposture.
          </p>
          <p>
           <strong>
            Dans le monde du livre rare
           </strong>
           , il y a eu un avant et un après De Caro. En 2005, ce « collectionneur italien » vend à Richard Lan, un des grands experts américains en livres anciens, un exemplaire du rarissime Sidereus nuncius (traduit en français sous le titre Le Messager des étoiles) de Galilée, publié en 1610. Extraordinaire exemplaire, signé de la main de l’auteur, et illustré de quatre illustrations originales peintes et signées par lui !
           <a class="more-link" href="/le-faussaire-qui-aimait-trop-galilee-de-laurent-bergers-et-paul-henri-moinet/#more-560999">
            (suite…)
           </a>
          </p>
         </div>
        </div>
        <div class="article">
         <p>
          <strong>
           Triste nouvelle : Décès de Yannick Nexon, chef du département de la Réserve, Bibliothèque Sainte-Geneviève. Décédé le 1 février 2018
          </strong>
         </p>
         <p>
          Né en 1953, Yannick Nexon sort major de l’École nationale des chartes de la promotion 1976 avec une thèse portant sur «
          <em>
           Le mécénat du chancelier Séguier. Recherches sur un hôtel parisien au xvii
           <sup>
            e
           </sup>
           siècle, l’ac­tivité d’un cercle littéraire et religieux et son influence
          </em>
          (1588-1672) ». Il avait repris et augmenté ce travail, très utilisé par les historiens d’art, pour donner un ouvrage publié aux éditions Champ Vallon en 2016 sous le titre
          <em>
           Le Chancelier Séguier (1588-1672). Ministre, dévot et mécène au Grand Siècle
          </em>
          . Ouvrage qui obtint le Prix Thiers de l’Académie française.
         </p>
         <p>
          Yannick Nexon débute sa carrière comme conservateur à la bibliothèque municipale de Lille, en 1977.
         </p>
         <p>
          Il est ensuite conservateur à la bibliothèque de l’Académie de France à Rome, en 1978, puis à la bibliothèque municipale de Dijon, en 1980. En 1982, il devient directeur de la bibliothèque centrale de prêt de la Loire-Atlantique.
          <a class="more-link" href="/deces-de-yannick-nexon-chef-du-departement-de-la-reserve-bibliotheque-sainte-genevieve/#more-553316">
           (suite…)
          </a>
         </p>
        </div>
        <div class="article">
         <p>
          <strong>
           La nouvelle réglementation antiterroriste de l’UE visera les bibliophiles. Si la nouvelle réglementation de l’Union européenne est proposées et approuvés, les collectionneurs de livres européens et les revendeurs peuvent être dans une surprise ennuyante quand ils font leurs premiers achats importés en 2019.
           <br/>
          </strong>
         </p>
         <p>
          <strong>
           Si un livre, une gravure, un tirage, un document ou une publication d’intérêt spécial datant de plus de 250 ans est importé dans un État membre de l’UE, le propriétaire ou le
          </strong>
          «
          <em>
           détenteur des marchandises
          </em>
          » devra présenter une déclaration d’importateur signée aux autorités douanières du pays d’entrée. La déclaration doit inclure une déclaration que les livres ont été initialement exportés légalement de leur pays d’origine. Toutefois, dans les cas où le pays exportateur (distinct du pays d’origine) est une « Partie contractante à la Convention de l’UNESCO sur les biens culturels », le titulaire des livres doit fournir une déclaration attestant qu’il a été exporté de ce pays conformément aux ses lois et règlements. Il va sans dire que si la proposition précise des livres et des documents d ‘ »intérêt particulier », elle ne donne aucun critère plus explicite pour définir ce que cela signifie et la notion d’ «
          <em>
           intérêt particulier
          </em>
          » est suffisamment vague et subjective pour inclure :
          <strong>
           <em>
            en pratique, pratiquement n’importe quel livre que quelqu’un pourrait vouloir importer.
           </em>
          </strong>
         </p>
        </div>
        <div class="article">
         <p>
          <em>
           <strong>
            Ces bibliothèques qui font rêver les collectionneurs – Un monde secret et discret
           </strong>
          </em>
         </p>
         <p>
          <a href="http://www.bibliorare.com/wp-content/uploads/2018/01/bibliotheque.jpg">
           <img alt="bibliotheque" class="alignleft size-full wp-image-534253" height="396" src="http://www.bibliorare.com/wp-content/uploads/2018/01/bibliotheque.jpg" width="354"/>
          </a>
          <b>
           Pierre Berès
          </b>
          (
          <strong>
           <a href="http://www.bibliorare.com/vente-pierre_beres.htm">
            Voir les catalogues des ventes Pierre Berès entre 2005 et 2007
           </a>
          </strong>
          ) a comme transformé une fois pour toutes en une savante alchimie, appelée goût, cette masse de livres en apparence indistincte», expliquait l’expert Jean-Baptiste de Proyart, en 2005, lors de l’inauguration de la vente fleuve du fonds du célèbre libraire. Pierre Berès avait lui-même annoncé l’événement. La septième et ultime vacation en son nom, organisée les 12 et 13 décembre à Paris chez Christie’s, intervient alors que d’autres personnalités se séparent de leur prestigieuse bibliothèque.
          <b>
           Jeudi, c’est un ensemble exceptionnel de livres Art déco réunis par l’un de ses meilleurs spécialistes, Félix Marcilhac (
          </b>
          <a href="http://www.bibliorare.com/drouot/cat-vent_binoche5-12-2012.htm">
           <b>
            Voir le catalogue
           </b>
          </a>
          <b>
           )
          </b>
          , qui étaient dispersés par Binoche et Giquello, à Drouot. Il y a quelques mois, c’était la bibliothèque surréaliste du libraire-galeriste niçois Jacques Matarasso qui, à 95 ans, avait décidé pour la troisième et dernière fois, de confier ses trésors à la maison Alde.
         </p>
         <p>
          <b>
           Ces dix dernières années, alimenté par d’importantes collections, le marché de la bibliophilie a explosé.
          </b>
          Chaque saison a été marquée par des ventes mémorables de grands amateurs : celle de Pierre Leroy (
          <em>
           3,1 millions d’euros, en 2002, chez Sotheby’s
          </em>
          ), Gwenn-Aël Bolloré (
          <em>
           1,6 M€, 2002, chez Sotheby’s
          </em>
          ), Daniel Filipacchi (
          <em>
           5,9 M€ en 2004 et 5,3 M€ en 2005 chez Christie’s)
          </em>
          , Charles Hayoit (
          <em>
           1,3 M€, en 2005, chez Sotheby’s
          </em>
          ). Les vendeurs sont le plus souvent les collectionneurs mêmes. Comme Pierre Berès, Jacques Matarasso, Pierre Leroy ou Daniel Filipacchi, ils ont préféré disperser leur collection en vente publique plutôt que de les transmettre à des héritiers qui n’en apprécient pas autant la valeur. Certains ont d’autres motivations.
          <strong>
           Dominique de Villepin
          </strong>
          ,
          <strong>
           amateur éclairé, a vendu son bel ensemble de livres sur Napoléon
          </strong>
          chez Pierre Bergé (
          <a href="http://www.bibliorare.com/cat-vent_drouot19-03-08.htm">
           <strong>
            Voir le catalogue
           </strong>
          </a>
          ) pour se lancer dans une autre entreprise.
          <a class="more-link" href="/ces-bibliotheques-qui-font-rever-les-collectionneurs-un-monde-secret-et-discret/#more-534245">
           (suite…)
          </a>
         </p>
        </div>
        <div class="article">
         <p>
          <em>
           <strong>
            Le Forum des Images filme les travaux de la BnF-Richelieu depuis 2010 et la fermeture de la zone 1, comme la plupart des grands chantiers culturels de l’État.
           </strong>
          </em>
         </p>
         <p>
          A l’occasion de la réouverture de la zone 1 rénovée du site Richelieu, un nouveau film a été réalisé par Jean-François-Roudot, coproduit par le Forum des images, l’Oppic, la BnF et l’INHA. Il présente la synthèse du chantier de rénovation du site Richelieu de 2010 à 2016 et sa préfiguration pour 2020.
         </p>
         <span class="embed-youtube" style="text-align:center; display: block;">
          <iframe class="youtube-player" frameborder="0" height="930" src="http://www.youtube.com/embed/T3EZKsaSfwY?version=3&amp;rel=1&amp;fs=1&amp;showsearch=0&amp;showinfo=1&amp;iv_load_policy=1&amp;wmode=transparent" type="text/html" width="1600">
          </iframe>
         </span>
        </div>
        <div class="article">
         <p>
          <strong>
           <em>
            Que ferons-nous de nos bibliothèques
           </em>
           ? ? ?
          </strong>
         </p>
         <p>
          <a href="http://www.bibliorare.com/wp-content/uploads/2016/02/BIBLIOTHEQUE.jpg">
           <img alt="BIBLIOTHEQUE" class="alignleft size-thumbnail wp-image-354285" height="150" src="http://www.bibliorare.com/wp-content/uploads/2016/02/BIBLIOTHEQUE-150x150.jpg" width="150"/>
          </a>
         </p>
         <p>
          <strong>
           Je viens de ressentir pour la première fois l’angoisse des intellectuels dès qu’ils comprennent qu’ils mourront un jour : qu’arrivera-t-il à leur bibliothèque ? Qui en voudra ? Qui la comprendra ?
          </strong>
          <strong>
           Qui comprendra les milliers de petits émerveillements à l’origine de tel achat, de tel passage souligné, de tel signet oublié ?
          </strong>
          <strong>
           Que se passe-t-il si nos enfants n’en veulent pas ?
          </strong>
          Faudra-t-il la disperser, la vendre, la liquider ? Et que se passera-t-il si quelqu’un tombe, par hasard ou non, sur les livres que nous dissimulions en deuxième rangée, et que nous ne montrions qu’aux amis les plus intimes, qui peuvent comprendre ce qu’on y trouve ? Car il y a des livres qu’on lit sans l’avouer, puisque les valeurs de l’époque les réprouvent ou les condamnent C’est fou comme on se dévoile, d’ailleurs, quand on laisse quelqu’un entrer dans sa bibliothèque !
          <a class="more-link" href="/que-ferons-nous-de-nos-bibliotheques/#more-354281">
           (suite…)
          </a>
         </p>
        </div>
        <div class="article">
         <p>
          <strong>
           ALDE 2006-2016. Fête ses dix ans.
          </strong>
          Première maison de ventes aux enchères françaises spécialisée en livres, autographes et monnaies. 150 ventes ont été réalisées depuis sa fondation, en mai 2006. Une vente exceptionnelle aura lieu à l’occasion de cet anniversaire le 24 mai prochain à l’hôtel Ambassador à 14h30.
          <a href="http://www.bibliorare.com/pdf/ALDE-24-mai-2016-presse.pdf" title="Télécharger le communiqué de presse">
           Télécharger le communiqué de presse
          </a>
         </p>
        </div>
        <div class="article">
         <p>
          <em>
           <strong>
            La bibliothèque nationale irakienne numérise ses fonds pour se protéger de l’Etat islamique.
           </strong>
          </em>
          Estimant ses fonds menacés par l’Etat islamique (EI), la bibliothèque nationale de Bagdad, en Irak, se lance dans un vaste projet de numérisation.
         </p>
         <div>
          La bibliothèque nationale de Bagdad, en Irak, a lancé un projet de grande envergure prévoyant la numérisation de ses œuvres, après que des milliers de documents ont été perdus ou endommagés par la guerre avec les Etats-Unis, rapporte l’
          <em>
           Associated Press
          </em>
          . La menace posée par l’Etat islamique, qui veut détruire l’histoire et la culture du pays, est également à l’origine de ce projet titanesque visant à préserver des millénaires d’histoire.
          <a class="more-link" href="/la-bibliotheque-nationale-irakienne-numerise-ses-fonds-pour-se-proteger-de-letat-islamique/#more-280143">
           (suite…)
          </a>
         </div>
        </div>
        <div class="article">
         <p>
          <em>
           <strong>
            Que la bibliothèque de Jean Giono demeure !
           </strong>
          </em>
          La maison est sauve, ou presque. Un accord de rachat par la mairie de Manosque devrait être bientôt scellé. Maintenant, il ne reste plus qu’à sécuriser l’âme de cette bâtisse, autrement dit sa bibliothèque.
         </p>

          <strong>
           14 bibliothèques mondiales d’art s’associent pour diffuser en ligne plus de 30 millions de documents.

Ce portail regrouperait les archives de l’histoire de l’art gérées par 11 institutions à travers le monde associées dans un projet de numérisation conjoint
