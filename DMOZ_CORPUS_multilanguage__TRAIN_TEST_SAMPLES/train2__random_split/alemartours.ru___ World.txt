   <script>
    jQuery(document).ready(function($) {
 
       var ResponsiveMenu = {
         trigger: '#responsive-menu-button',
         animationSpeed: 500,
         breakpoint: 480,
         pushButton: 'off',
         animationType: 'slide',
         animationSide: 'left',
         pageWrapper: '',
         isOpen: false,
         triggerTypes: 'click',
         activeClass: 'is-active',
         container: '#responsive-menu-container',
         openClass: 'responsive-menu-open',
         accordion: 'off',
         activeArrow: '▲',
         inactiveArrow: '▼',
         wrapper: '#responsive-menu-wrapper',
         closeOnBodyClick: 'off',
         closeOnLinkClick: 'off',
         itemTriggerSubMenu: 'off',
         linkElement: '.responsive-menu-item-link',
         openMenu: function() {
           $(this.trigger).addClass(this.activeClass);
           $('html').addClass(this.openClass);
           $('.responsive-menu-button-icon-active').hide();
           $('.responsive-menu-button-icon-inactive').show();
           this.setWrapperTranslate();
           this.isOpen = true;
         },
         closeMenu: function() {
           $(this.trigger).removeClass(this.activeClass);
           $('html').removeClass(this.openClass);
           $('.responsive-menu-button-icon-inactive').hide();
           $('.responsive-menu-button-icon-active').show();
           this.clearWrapperTranslate();
           this.isOpen = false;
         },
         triggerMenu: function() {
           this.isOpen ? this.closeMenu() : this.openMenu();
         },
         triggerSubArrow: function(subarrow) {
           var sub_menu = $(subarrow).parent().next('.responsive-menu-submenu');
           var self = this;
           if(this.accordion == 'on') {
             /* Get Top Most Parent and the siblings */
             var top_siblings = sub_menu.parents('.responsive-menu-item-has-children').last().siblings('.responsive-menu-item-has-children');
             var first_siblings = sub_menu.parents('.responsive-menu-item-has-children').first().siblings('.responsive-menu-item-has-children');
             /* Close up just the top level parents to key the rest as it was */
             top_siblings.children('.responsive-menu-submenu').slideUp(200, 'linear').removeClass('responsive-menu-submenu-open');
             /* Set each parent arrow to inactive */
             top_siblings.each(function() {
               $(this).find('.responsive-menu-subarrow').first().html(self.inactiveArrow);
             });
             /* Now Repeat for the current item siblings */
             first_siblings.children('.responsive-menu-submenu').slideUp(200, 'linear').removeClass('responsive-menu-submenu-open');
             first_siblings.each(function() {
               $(this).find('.responsive-menu-subarrow').first().html(self.inactiveArrow);
             });
           }
           if(sub_menu.hasClass('responsive-menu-submenu-open')) {
             sub_menu.slideUp(200, 'linear').removeClass('responsive-menu-submenu-open');
             $(subarrow).html(this.inactiveArrow);
           } else {
             sub_menu.slideDown(200, 'linear').addClass('responsive-menu-submenu-open');
             $(subarrow).html(this.activeArrow);
           }
         },
         menuHeight: function() {
           return $(this.container).height();
         },
         menuWidth: function() {
           return $(this.container).width();
         },
         wrapperHeight: function() {
           return $(this.wrapper).height();
         },
         setWrapperTranslate: function() {
           switch(this.animationSide) {
             case 'left':
               translate = 'translateX(' + this.menuWidth() + 'px)'; break;
             case 'right':
               translate = 'translateX(-' + this.menuWidth() + 'px)'; break;
             case 'top':
               translate = 'translateY(' + this.wrapperHeight() + 'px)'; break;
             case 'bottom':
               translate = 'translateY(-' + this.menuHeight() + 'px)'; break;
             }
             if(this.animationType == 'push') {

       </h1>
       Мы предлагаем вам лечение, шопинг и отдых в Греции, великолепной стране с колоритной культурой, чистейшими пляжами и массой развлечений для туристов. Мы организуем индивидуальные и групповые туры в Грецию различного направления. Вы обязательно найдёте то, что искали! Полный набор услуг от бронирования билета до аренды апартаментов, вилл или номеров в отеле с экскурсиями, походами по магазинам, поездками по святым местам, охотой, рыбалкой и так далее.
       <br/>
       <br/>
       Компания Алемар-груп устроит вам самый лучший отдых в Греции, мы знаем эту страну и имеем там собственное туристическое представительство. Мы гордимся своими многочисленными клиентами. Их отзывы об отдыхе в Греции с нашим сервисом подтверждают, что нам можно доверять. Работа наших специалистов заключается в предоставлении в аренду недвижимости, яхт, машин, вертолётов; помощи в оформлении документов, организации экскурсий; решении самых разных задач для обеспечения беззаботного, интересного и приятного проживания клиентов.
       <br/>
       <br/>
       Мы предлагаем туры в Грецию по максимально выгодным ценам. Такое увлекательное времяпровождение вы не забудете никогда! Какова цена? Это будет зависеть только от вашего желания и возможностей, так как заказать тур в Грецию вы можете и с пакетом услуг эконом, и по классу люкс в отдельных апартаментах или на вилле.
       <br/>
       <br/>
       Желаете воспользоваться последними достижениями греческой медицины – медицинские туры в Грецию специально разработаны для пациентов местных клиник и их сопровождающих. Хотите побывать на святой земле – выбирайте поломнические поездки с посещениями храмов, монастырей, святой горы Афон. Лучший отдых в Греции обеспечит сама природа: ласковое море, горный воздух и чистый песчаный пляж. Хотите комфортно и весело отдохнуть на греческих островах – мы предоставим множество развлекательных программ и мероприятий.

Мы всегда будем на связи, чтобы отдых в Греции или любой другой стране стал одним из лучших ваших воспоминаний!
