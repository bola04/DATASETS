         var slide = function () {
             var nextSlider = $('.home-slider.active').next();
             if (nextSlider.length == 0)
                 nextSlider = $('.home-slider').first();
             slideTo(nextSlider);
         };
 
         var slideTo = function (el) {
             var activeSlider = $(el);
             var previousSlider = $('.home-slider.active');
             $('.home-slider').removeClass('active'); //.stop();
 
             $('.home-slider').not(previousSlider).not(activeSlider).css({ 'z-index': 50, left: '959px' });

             var myNav = $('#slider-nav-' + activeSlider.attr("data-id"));
             myNav.addClass('active');
             $('#slide-nav a').not(myNav).removeClass('active');
         };
 
         $('#slide-nav a').click(function () {
             clearInterval(window.slideInterval);
             var id = "#slider-" + $(this).attr("data-id");
