   <script type="text/javascript">
    jQuery.noConflict();
 jQuery(document).ready(function($) {
 $(function() {	
 	var process_script_url = 'http://classe-export.com/wp-content/plugins/wp-cart-for-digital-products' + '/ajax_process_cart_requests.php';
 	var save_ret_cart_msg = $('div.eStore_save_retrieve_cart_message');
 	
 	$('.eStore_save_cart_section').on("click", function (){
 		save_ret_cart_msg.html("").append("Sauvegarde de votre panier...");		
 		var dataString = 'eStore_cart_action=eStore_save_cart';		
 		$.ajax({
 			type: "POST",
 			url: process_script_url,
 			data: dataString,
 			dataType: "json",
 			success: function(data) {				
 				//var dataStr = JSON.stringify(data); console.log(dataStr);
 				if(data.status == "success"){
 					save_ret_cart_msg.html("").append('Votre panier a été sauvegardé! Numéro ID du panier : ' + data.ID);
 				}
 				else{
 					save_ret_cart_msg.html("").append('<span class="eStore_warning">' + data.details + '</span>');
 				}
 			}
 		});				
 		return false;
 	});
 
 	$('.eStore_retrieve_cart_section').on("click", function (){
 		var cart_id = prompt("Numéro ID du panier", "");	
 		if(cart_id == null){
 			return false;
 		}
 		else if(cart_id ==""){
 			save_ret_cart_msg.html("").append('<span class="eStore_warning">Le numéro ID n'est pas correct</span>');
 			return false;
 		}
 		cart_id = $.trim(cart_id);
 		save_ret_cart_msg.html("").append("Sauvegarde de votre panier...");
 		var dataString = 'eStore_cart_action=eStore_retrieve_cart&cart_id=' + cart_id;
 		$.ajax({
 			type: "POST",
 			url: process_script_url,
 			data: dataString,
 			dataType: "json",
 			success: function(data) {
 				if(data.status == "success"){	
 					save_ret_cart_msg.html("").append('Succès!');	
 					target_url = window.location.href.split("#")[0];
 					window.location.href = target_url;					
 				}
 				else{
 					save_ret_cart_msg.html("").append('<span class="eStore_warning">Le numéro ID n'est pas correct</span>');	
 				}	
 			}
 		});		
 		return false;
 	});
 });
 });
   </script>
