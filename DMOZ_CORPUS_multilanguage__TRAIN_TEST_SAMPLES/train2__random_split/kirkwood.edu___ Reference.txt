   <script>
    var resizeCount = 0;
   var resizeInterval = null;
   var desktopHeaderPush = null;
   var desktopHeader = null;
   var mobileHeaderPush = null;
   var mobileHeader = null;
   var mobileSubheader = null;
   var slidedowns = null;
   var subheaderSlidedownButtons = null;
 
   var lastFocusedSlider = null;
 
   function startResizeLoop() {
     resizeCount = 0;
 
     if (resizeInterval == null) {
       resizeInterval = setInterval(
           function () {
             resizeCount += 1;
             if (resizeCount === 11) {
               clearInterval(resizeInterval);
               resizeInterval = null;
               resizeHeaders();
             }
             resizeHeaders();
           },
           30
       );
     }
   }
 
   function resizeHeaders() {
     if (desktopHeaderPush && desktopHeader && desktopHeader.is(":visible")) {
       desktopHeaderPush.height(desktopHeader.height());
     }
 
     if (mobileHeader && mobileHeader.is(":visible")) {
       if (mobileHeaderPush) {
         mobileHeaderPush.height(mobileHeader.height());
       }
 
       if (mobileSubheader) {
         mobileSubheader.css("maxHeight", (window.innerHeight - mobileHeader.height()));
       }
     }
   }
 
   function checkSlidedowns() {
     if (slidedowns) {
       var bodyWidth = $("body").width();
 
       slidedowns.each(function () {
         var isShown = $(this).css("display");
         if (isShown == "none") {
           $(this).css("display", "block");
         }
         if ($(this).offset().left + $(this).width() > bodyWidth) {
           $(this).css("right", 0);
           // $(this).find("a").each(function () {
           //   $(this).parent().attr("align", "right");
           // });
         }
         if (isShown == "none") {
           $(this).css("display", "none");
         }
       });
     }
   }
 
   function closeSubheaderSlidedownButton(id) {
     var button = $("#" + id);
     var wrapper = button.parent().find(".slidedown-menu-wrapper");
 
     if (wrapper.hasClass("show")) {
       wrapper.removeClass("show");
       button.attr("aria-expanded", false);
       setTimeout(function () {
         wrapper.css("display", "none");
       }, 10);
     }
   }
 
   function closeSubheaderSlidedowns() {
     if (subheaderSlidedownButtons) {
       subheaderSlidedownButtons.each(function () {
         closeSubheaderSlidedownButton($(this).prop("id"));
       });
     }
   }
 
   function checkMobileMenu() {
     if (desktopHeader && desktopHeader.is(":visible") && $("#collapse_mobile_subheader_accordion").hasClass("show")) {
       $("#btn_menu_toggle").trigger("click");
     }
   }
 
   function doResizeChecks() {
     checkMobileMenu();
     resizeHeaders();
     checkSlidedowns();
   }
 
   $(window).resize(doResizeChecks);
 
   $(function () {
     // init commonly used variables
     desktopHeaderPush = $("#desktop_header_push");
     desktopHeader = $("#desktop_header");
     mobileHeaderPush = $("#mobile_header_push");
     mobileHeader = $("#mobile_header");
     mobileSubheader = $(".mobile-subheader");
     slidedowns = $(".slidedown-menu-wrapper");
     subheaderSlidedownButtons = $(".btn-subheader.slidedown-toggle");
 
 
     $("#desktop_search").click(function () {
       $("#desktop_header_white_right").addClass("in-search");
       $("#homepage_search").focus();
     });
 
     $("#desktop_cancel_search").click(function () {
       $("#desktop_header_white_right").removeClass("in-search");
     });
 
     $("#homepage_search").focusout(function () {
       $("#desktop_header_white_right").removeClass("in-search");
     });
 
     var $imgDesktopNameplate = $("#img_desktop_nameplate");
     var isSmall = false;
     $(window).scroll(function () {
       var scrollTop = $(window).scrollTop();
       var shouldSmall = (!isSmall && scrollTop > 150) || (isSmall && scrollTop >= 100);
       if (shouldSmall !== isSmall) {
         isSmall = shouldSmall;
         $imgDesktopNameplate[isSmall ? "addClass" : "removeClass"]("small");
         startResizeLoop();
       }
     });
 
     subheaderSlidedownButtons.click(function () {
       var menu = $(this).parent().find(".slidedown-menu-wrapper");
       var isShown = menu.hasClass("show");
       var $button = $(this);
 
       closeSubheaderSlidedowns();
 
       if (!isShown) {
         $button.attr("aria-expanded", true);
         menu.css("display", "block");
         setTimeout(function () {
           menu.addClass("show");
         }, 100);
       }
     });
 
     $(document).on("click", function (event) {
       if ($(event.target).parents(".slidedown.subheading-slidedown").length === 0) {
         closeSubheaderSlidedowns();
       }
 
       if ($(event.target).parents(".mobile-subheader").length === 0 && $("#collapse_mobile_subheader_accordion").hasClass("show")) {
         $("#btn_menu_toggle").trigger("click");
       }
     }).on("blur", ".slidedown", function (event) {
       var previous = $(event.target).parents(".slidedown").first();
       var current = event.relatedTarget ? $(event.relatedTarget).parents(".slidedown") : null;
       current = current ? current.first() : null;
 
       if (!previous.is(current)) {
         closeSubheaderSlidedownButton(previous.find("button.slidedown-toggle").attr("id"));
       }
     });
 
     $("#collapse_mobile_search").on("shown.bs.collapse", function () {
       $(this).find("input").focus();
     }).on("show.bs.collapse", function () {
       // startResizeLoop();
     }).on("hide.bs.collapse", function () {
       // startResizeLoop();
     });
     $("#homepage_search_mobile").focusout(function () {
       $("#collapse_mobile_search").collapse("hide");
     });
 
     $("#btn_menu_toggle").on("click", function () {
       // startResizeLoop();
       $("body").toggleClass("no-scroll");
     });
 
     $('.btn-mobile-accordion').click(function () {
       // startResizeLoop();
     });
 
     doResizeChecks();
   });
   </script>

     var splashButtonWrapper;
   var splashWrapper;
 
   function dealWithSplashButton() {
     if (splashButtonWrapper && splashWrapper) {
 
       if ($(window).scrollTop() > 0) {
         splashButtonWrapper.addClass("scrolled-by");
         setTimeout(function () {
           if (splashButtonWrapper) {
             splashButtonWrapper.remove();
             splashButtonWrapper = null;
           }
         }, 300);
       }
     }
   }
 
   function attemptToPlaceSplashButton() {
     if (splashWrapper && (splashWrapper.position().top + splashWrapper.outerHeight(true) + 10) >= window.innerHeight && $(window).scrollTop() === 0) {
       $("body").append("<div class=\"w-100 d-none d-xl-flex justify-content-center\" id=\"splash_button_wrapper\">\n" +
           "    <span class=\"text-white\" id=\"splash_button_span\"><i class=\"fas fa-chevron-circle-down\"></i></span>\n" +
           "</div>");
 
       splashButtonWrapper = $("#splash_button_wrapper");
 
       splashButtonWrapper.find("#splash_button_span").on("click", function(){
         $("html, body").animate({
           scrollTop: ($("#news_header").offset().top - 250)
         }, 500);
       });
     }
   }
 
   function doHomepageResizeChecks(){
     updateNewsSpacer();
     dealWithSplashButton();
   }
 
   function updateNewsSpacer(){
     $("#news_mirrored_spacing").css("height", $("#news_header").css("height"));
   }
 
   function toggleNewsHeader() {
     var el = $("#news_header");
     if (el.attr("align") === "center") {
       el.attr("align", "left");
     } else {
       el.attr("align", "center");
     }
   }
 
   function toggleNewsCenter() {
     var el = $("#homepage_news_carousel");
     el.data("owl.carousel").options.center = !el.data("owl.carousel").options.center;
     el.trigger("refresh.owl.carousel");
   }
