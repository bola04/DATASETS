       var theme = "none";

             var yellowTheme = "background-color: rgba(252, 246, 204, 0.4);";
             var fullSectionHeight = "false";
             var illustration = "";
             var hideTheme = "false";
 
             
             var fixModules = function(parentEl) {
             var windowWidth = window.innerWidth || document.documentElement.clientWidth || document.body.clientWidth;
 
             var modules = Array.from(parentEl.querySelectorAll('.module'));
               modules.forEach(function(moduleEl) {
                 var width = getComputedStyle(moduleEl, null).getPropertyValue('width');
                 var isModuleCallToAction = moduleEl.classList.contains('module-call-to-action');
                 var percentOfWidth = (parseInt(width) / windowWidth * 100);
                 if(isModuleCallToAction && percentOfWidth > 80) {
                   moduleEl.classList.add('module-full-width');
                 }
               });
             };
 
             var fixHeader = function() {
               if (hideTheme  === 'false' && theme !== 'none' && illustration && fullSectionHeight  === 'true' ) {
                 if (theme === 'blue') {
                     document.getElementsByClassName('gj-header')[0].style.backgroundColor = blueTheme.replace("background-color: ", "").replace(";", "");
                 } else if (theme === 'green') {
                     document.getElementsByClassName('gj-header')[0].style.backgroundColor = greenTheme.replace("background-color: ", "").replace(";", "");
                 } else if (theme === 'yellow') {
                     document.getElementsByClassName('gj-header')[0].style.backgroundColor = yellowTheme.replace("background-color: ", "").replace(";", "");
                 }
               }
             }
 
             function runFixModules() {
               Array.from(document.querySelectorAll('.flex-modules')).forEach(function(el) {
                 fixModules(el);
               });
             }
 
             document.addEventListener('DOMContentLoaded', function() {
               fixHeader();
               runFixModules();
             });
 
             var resizeTimeout;
             window.addEventListener('resize', resizeThrottler, false);
 
             function resizeThrottler() {
               if (resizeTimeout) {
                 window.clearTimeout(resizeTimeout);
                 resizeTimeout = 0;
               } else {
                 resizeTimeout = setTimeout(function() {
                   runFixModules();
                 }, 100);
               }
             }
