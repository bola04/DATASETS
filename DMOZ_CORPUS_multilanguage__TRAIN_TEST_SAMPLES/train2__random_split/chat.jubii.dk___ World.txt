                         $p.find(r.HasVoted ? '.poll-input' : '.poll-results').remove();
                         var $tpl = $p.find('.poll-item-template');
                         var votes = 0;
                         $.each(r.Answers, function (i, a) {
                             var $item = $tpl.clone();
                             $item.removeClass('poll-item-template');
                             $item.find('.text').text(a.Text);
                             $item.find('.progress-bar').data('votes', a.Votes);
                             $item.find('input[type=radio]').val(a.ID);
                             $tpl.after($item);
                             votes += a.Votes;
                         });
                         $('.poll-options-list, .poll-result-list').addClass("options-" + r.Answers.length);
                         $tpl.remove();
                         $p.data('votes', votes);
                         $p.find('.poll-question').text(r.Question);
                         $p.find('input[name=poll]').val(r.ID);
                         $p.find('.poll-votes').text(votes + ' stemme' + (votes === 1 ? '' : 'r'));
                         if ((r.Url != undefined) && (r.Url != '')) {
                             $p.find('.btn').attr('href', r.Url);
                         }
                         else {
                             $p.find('.btn').css('display', 'none');
                         }
                         $p.find('.poll-results li').each(function (i, e) {
                             var $b = $(e).find('.progress-bar'), $p = $(e).find('.percentage'), p = Math.round((parseInt($b.data('votes'), 10) / votes) * 100) + '%';
                             $b.css({ width: p });
                             $p.text(p);
                         });
                     })
                     .always(function () {
                         $p.find('.poll-loading').addBack().removeClass('poll-loading');
                     });
             });
         });
