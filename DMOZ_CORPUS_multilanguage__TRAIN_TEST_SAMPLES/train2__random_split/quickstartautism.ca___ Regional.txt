       function onChangeOther(index,oDest) {

 	if (index && divDonSpecifyAmount) {
 		if (index == "Other") {
 			setCSS(oDest, "display", "block");
 		} else {
 			setCSS(oDest, "display", "none");
 		}
 	}
 }
 function onChangeMemorial(index,oDest) {
 	var divDonSpecifyAmount = document.getElementById(oDest);
 	if (index == "Memorial Donation") {
 		setCSS(oDest, "display", "block");
 	} else {
 		setCSS(oDest, "display", "none");
 	}
 }
 
 function onSubmit() {
 	var btnSubmit = document.getElementById("btnSubmit");
 	var waittxt = document.getElementById("wait");
 	
 	if (btnSubmit) {
 		btnSubmit.style.display = "none";
 		waittxt.style.display = 'block';
 	}
 }
 
 function setCSS(id, cssParam, cssValue) {
 	var obj = document.getElementById(id);
 	var cssObj;
 	
 	if (obj) {
 		cssObj = eval("obj.style." + cssParam);
 		
 		if (cssObj != undefined) {
 			eval("obj.style." + cssParam + " = '" + cssValue + "'");
 		} else {
 			alert("CSS object does not exist: " + cssParam);
 		}
 	}
 }

        <h2>
         Mission
        </h2>
        <p>
         <span>
          QuickStart - Early Intervention for Autism is a charitable organization whose mission is to assist children with autism in reaching their full potential by helping the children obtain early intervention, at the first signs of autism.
         </span>
        </p>
        <p>
         QuickStart raises awareness and initiates programs to cut wait-times, build capacity, and promote improvements in early identification and intervention.
        </p>
