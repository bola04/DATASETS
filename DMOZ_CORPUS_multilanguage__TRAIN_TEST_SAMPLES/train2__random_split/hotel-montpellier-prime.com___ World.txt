       function validresa(f) {
 		var ddeb = f.elements['date_arr'].value;

 		var dfin = new Date(elem[2], elem[1]-1, elem[0],0,0,0,0);
 		var nbnights = parseInt(f.elements['nbnights'].value);
 		if (nbnights <= 0) nbnights = 1;
 		dfin.setDate(dfin.getDate() + nbnights);
 		var month = '' + (dfin.getMonth() + 1);
 		var day = '' + dfin.getDate();
 		var year = dfin.getFullYear();
 		if (month.length < 2) month = '0' + month;
 		if (day.length < 2) day = '0' + day;
 		dfin = [day, month, year].join('/');
 		f.elements['date_dep'].value = dfin;
 		return true;
