 function populateRegionsForCountry(){
 	var countryCategory = $('option:selected', $('#regionCountryCategories')).attr('value');
 	if(countryCategory != null || countryCategory !=''){
 		$.getJSON("/wine-finder/getRegionsFromCountryCategory.json",{categoryCode: countryCategory}, function(regionCategories){
 			$("#regionSubCategories").prop('disabled', 'disabled');
 			var options = '';
 			options += '<option value="" selected="true">Region (any)</option>';
 			if(regionCategories != null){
 				$("#regionSubCategories").removeAttr("disabled");
 				for (var i = 0; i < regionCategories.length; i++) {
 					options += '<option value="' + regionCategories[i].code + '">' + regionCategories[i].name + '</option>';
 				}
 			}
 			$("select#regionSubCategories").html(options);
 		}).done(function(){
 					searchResult();
 				});
 	}
 }
 
 function searchResult(){
 	console.log("entering wine finder method");
 	var priceVal = $('#priceRanges').val();
 	var colorVal = $('#wineColorCodes').val();
 	var regionCountryVal = $('#regionCountryCategories').val();
 	var regionVal = $('#regionSubCategories').val();
 
 	var searchQuery = ':bbr-relevance';
 	var regionCategory = 'region';
 	if(priceVal != '')
 	{
 		searchQuery = searchQuery + ':price:' + priceVal;
 	}
 	if(colorVal != '')
 	{
 		searchQuery = searchQuery + ':color:' + colorVal;
 	}
 	if(regionCountryVal != '')
 	{
 		regionCategory = regionCountryVal;
 	}
 	if(regionVal != '')
 	{
 		regionCategory = regionVal;
 	}
 	if(priceVal == '' && colorVal == '' && regionCountryVal == '' && regionVal == '')
 	{
 		$('#wine_finder_results span').html('');
 		// $('#wine_finder_button').addClass('wineFinderInactive');
 		// $('#wine_finder_button').removeClass('wineFinderActive');
 	}
 	else
 	{
 		$.ajax({
 			url : '/wine-finder/' + regionCategory,
 			data: {q : searchQuery},
 			type : 'GET',
 			cache: false,
 			beforeSubmit: function ()
 			{
                 $('#wine_finder_content').block({ message: "<img src='" + ACC.config.commonResourcePath + "/images/spinner.gif' />" });
 			},
 
 			success : function(data) {
 						$('#wine_finder_results span').html(data.pagination.totalNumberOfResults + ' results found');
 						if(data.pagination.totalNumberOfResults > 0)
 						{
 							$('#wine_finder_button').attr('href' , data.currentQuery.url);
 							$('#wine_finder_button').addClass('wineFinderActive');
 						}
 						else
 						{
 							// $('#wine_finder_button').addClass('wineFinderInactive');
 							// $('#wine_finder_button').removeClass('wineFinderActive');
 						}
 			},
 			error : function() {
 				console.log('ERROR!');
 			},
 			complete: function ()

 			}
