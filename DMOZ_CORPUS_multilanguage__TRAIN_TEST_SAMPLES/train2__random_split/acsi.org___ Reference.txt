 	(function (window, $) {
 
 		var isMobile = false,
 			mq = '(min-width: 768px)';
 
 		if (typeof window.matchMedia !== 'function' || $('.ad-container').length === 0) {
 			return;
 		}
 
 		// set up 'home base' and destinations for ad blocks
 		$('.ad-container').each(function (i) {
 			$(this).attr('data-ad-index', i)
 				.parents('#content, .mobile-nav-slide').first()
 					.append('<div class="ad-destination" data-ad-index="' + i + '"></div>');
 		});
 
 		function moveAds() {
 			if (window.matchMedia(mq).matches && isMobile) {
 				isMobile = false;
 				$('.ad-destination').each(function () {
 					var index = $(this).data('ad-index');
 					$(this).children().detach().appendTo('.ad-container[data-ad-index="' + index + '"]');
 				});
 			} else if (!window.matchMedia(mq).matches && !isMobile) {
 				isMobile = true;
 				$('.ad-origin').each(function () {
 					var index = $(this).data('ad-index');
 					$(this).children().detach().appendTo('.ad-destination[data-ad-index="' + index + '"]');
 				});
 			}
 		}
 
 		moveAds();
 		window.addEventListener('resize', moveAds);
 
 	})(window, jQuery);
