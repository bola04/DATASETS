     function drawBanner() {
 
         var originBannerOption = {	//기본옵션
             title: '',
             isOpen: true,
             url: {
                 image: '',
                 link: ''
             }
         };
 
         var chuseokBannerOption = {	//추석옵션
             title: 'SBS 2017년 추석 특집 페이지',
             isOpen: true,
             url: {
                 image: 'http://img.sbs.co.kr/event/2017_chooseok/@bnr_2017chooseok.jpg',
                 link: 'http://tv.sbs.co.kr/2017chuseok/index.jsp'
             }
         };
 
 		var d = new Date();
         var today = d.getFullYear() + ('0' + (d.getMonth() + 1)).slice(-2) + ('0' + d.getDate()).slice(-2);
         if(today > 20170929 && today < 20171010) {
             drawMainGnbContents(chuseokBannerOption);	//	- 메인 GNB html 드로잉
         }else{
             drawMainGnbContents(originBannerOption);
 		}
     }

     function setMainContents(){
         var mainTabContents      = null
             , mainPdNoteContents = null
             , mainClipContents	 = null
             , promBannerLeft     = null
             , promBannerRight    = null
             , groupPromBanner    = null
             , groupLogoBanner    = null
             , mainNewsContents	 = null;
 
         $.ajax({
             url: '/json/newMain.js',
             type: 'get',
             dataType: 'json',
             success: function(data){
                 mainTabContents		= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].mainTabContents;
                 mainPdNoteContents	= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].mainPdNoteContents;
                 mainClipContents	= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].mainClipContents;
                 promBannerLeft		= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].promotionLeft;
                 promBannerRight		= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].promotionRight;
                 groupPromBanner		= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].groupPromotion;
                 groupLogoBanner		= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].groupLogo;
                 mainNewsContents	= data.HashMap.newAllMainDisplays.newAllMainDisplayList[0].mainNewsContents;
 
                 drawMainTabContents(mainTabContents);                            	// 메인 탭 html 출력
                 drawMainPdNoteContents(mainPdNoteContents, "#mainPdNoteContents");	// 메인 피디노트 순위 html 출력
                 drawMainClipContents(mainClipContents, "#mainClipContents");		// 메인 클립 순위 html 출력
                 drawMainBannerList(promBannerLeft, '#promotionBannerLeft');        	// 메인 프로모션 좌측 html 출력
                 drawMainBannerList(promBannerRight, '#promotionBannerRight');      	// 메인 프로모션 우측 html 출력
                 drawMainBannerList(groupPromBanner, '#groupPromotionBanner');      	// 구룹사 홍보 배너 html 출력
                 drawMainBannerList(groupLogoBanner, '#groupLogoBanner');           	// 구룹사 로고 배너 html 출력
                 drawMainNewsContents(mainNewsContents);
 
             },
             complete : function(){
                 mainLayoutBookMark();		// 메인 layout 이벤트 북마크(footer까지 포함)
                 $('#container').show();
             },
             error: function (xhr, status, error) {
             }
         });
     }

     function drawRandomTabContents(obj){
         var randomArr        = obj
             , dummyObject    = []
             , probabilityArr = []
             , returnObject   = []
             , addProbability = 10;
 
         if(randomArr.length > 2){
             probabilityArr = [60, 40, 0];
 
         }else{
             probabilityArr = [100, 0];
         }
 
         for(var ii=0; ii<probabilityArr.length; ii++){
             if(randomArr[ii].indexOf('spot') > -1){
                 probabilityArr[ii] += addProbability;
             }else{
                 if(randomArr.length < 2){
                     probabilityArr[ii] -= addProbability/2;
                 }
             }
 
             for(var jj=0; jj<probabilityArr[ii]; jj++){
                 dummyObject.push(randomArr[ii]);
             }
         }
 
         returnObject.push(dummyObject[Math.floor(Math.random()*100)]);
 
         return returnObject;
     }

     function dailySnsShare(shareType){
         var url = ""
             , title = ""
             , date	= new Date()
             , year	= date.getFullYear()
             , month = addZeros(date.getMonth() + 1, 2)
             , day	= addZeros(date.getDate(), 2)
             , today = year + month + day;
 
         title = year + '.' + month + '.' + day + ' SBS Daily Conpick';
 
         title = title.replace(/replace-quotes-072/g, "\"");
         title = title.replace(/replace-one-quotes-282/g, "'");
 
         url = "http://w3.sbs.co.kr/dailySnsPage.do?searchDate=" + today;
 
         var kakaoTitle = title;
         title = encodeURIComponent(title);
 
         var kakaoUrl = url;
         url = encodeURIComponent(url);
 
         switch (shareType){
             // Facebook
             case 'F':
                 url = "http://www.facebook.com/sharer.php?u=" + url;
                 break;
             // Twitter
             case 'T':
                 url = "http://www.twitter.com/home?status="
                     + title
                     + "%0A" + url;
                 break;
             // kakao
             case 'K':
                 Kakao.Story.share({
                     url: kakaoUrl,
                     text: kakaoTitle
                 });
                 break;
             // BAND
             case 'B':
                 url = "http://share.naver.com/web/shareView.nhn?url=" + url + "&title=" + title;
                 break;
             default:
                 break;
         }
 
         if(shareType != 'K'){
             window.open(url, 'ShareonSNS', 'width=510, height=500, scrollbars=no, resizeadble=no');
         }
     }
 
     function addZeros(n, digits){
         var zero = '';
 
         n = n.toString();
 
         if(n.length < digits){
             for(var ii=0; ii<digits - n.length; ii++){
                 zero += '0';
             }
         }
 
         return zero+n;
     }
