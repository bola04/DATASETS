      <script>
       // slayt init
       var slaytSay = 0;
       var slaytSira = 0;
       var slaytTotalsure = 4;
       var slaytGerisay = slaytTotalsure;
 
       $('.slayt .img-container').each(function() {
         slaytSay++;
         $(this).attr('id', 'slayt-img-'+slaytSay);
         $(this).css({ opacity: 0.0 });
       });
 
       function slaytNext() {
         var slaytNextImg = slaytSira;
         if (slaytNextImg == slaytSay) slaytNextImg = 1; else slaytNextImg++;
 
         $('#slayt-img-'+slaytSira).css({ opacity: 0.0 });
         $('#slayt-img-'+slaytNextImg).css({ opacity: 1.0 });
 
         var imgW = $('#slayt-img-'+slaytNextImg+" IMG").width();
         var imgH = $('#slayt-img-'+slaytNextImg+" IMG").height();
         var zoomLvl = 6; // kucuk deger daha cok yakinlastirir
 
         slaytSira = slaytNextImg;
       }
 
       function slaytPrev() {
         var slaytPrevImg = slaytSira;
         if (slaytPrevImg == 1) slaytPrevImg = slaytSay; else slaytPrevImg--;
 
         $('#slayt-img-'+slaytSira).css({ opacity: 0.0 });
         $('#slayt-img-'+slaytNextImg).css({ opacity: 1.0 });
 
         var imgW = $('#slayt-img-'+slaytNextImg+" IMG").width();
         var imgH = $('#slayt-img-'+slaytNextImg+" IMG").height();
 
         $('#slayt-img-'+slaytNextImg+" IMG").animate({'width':'+='+imgW/zoomLvl,'height':'+='+imgH/zoomLvl,'left':'-='+imgW/(zoomLvl*2),'top':'-='+imgH/(zoomLvl*2)}, slaytTotalsure*1000+1000, 'linear', function() {
           $(this).css({'width':imgW, 'height':imgH, 'left':'+='+imgW/(zoomLvl*2), 'top':'+='+imgH/(zoomLvl*2)});
         });
 
         slaytSira = slaytPrevImg;
       }
 
       function welcomeOut () {
         $(".welcome").fadeOut(1500);
       }
 
       $(function(){
         $('.imagezone.slayt').waitForImages(function() {
 
           $(".welcome").fadeIn("slow");
           setTimeout("welcomeOut()", 2000);
 
           slaytNext();
           animasyonBaslat();
           if (slaytSay > 1)
             setInterval("slaytGerisayCounter()", 1000);
         });
       });
 
       function slaytGerisayCounter() {
         slaytGerisay--;
         if (slaytGerisay == 0) {
           slaytNext();
           slaytGerisay = slaytTotalsure;
         }
       }
 
       $('.slaytNav .left').click(function() {
         slaytPrev();
         slaytGerisay = slaytTotalsure;
       });
 
       $('.slaytNav .right').click(function() {
         slaytNext();
         slaytGerisay = slaytTotalsure;
       });
 
       function animasyonBaslat() {
         var slaytNextImg = slaytSira;
 
         var imgW = $('#slayt-img-'+slaytNextImg+" IMG").width();
         var imgH = $('#slayt-img-'+slaytNextImg+" IMG").height();
         var zoomLvl = 2; // kucuk deger daha cok yakinlastirir
 
         setTimeout(function() {
           $('#slayt-img-'+slaytNextImg+" IMG").css({'transition':'all 10s', 'transform':'translate('+((imgW/(zoomLvl*2))-100)+'px , '+imgH/(zoomLvl*2)+'px) scale(2,2)'});
         }, 500);
 
         setTimeout(function() {
           $('#slayt-img-'+slaytNextImg+" IMG").css({'transition':'none', 'transform':'translate(0px , 0px) scale(1,1)'});
         }, 6000);
 
       }

 function movePickerDate(selectedDate, amount) {
   var a = selectedDate.split("/");
   var data = new Date(parseInt(a[2],10), (parseInt(a[1],10) - 1), parseInt(a[0],10), 12, 0, 0);
   var giorno_data = AddDays(data, amount);
   
   return giorno_data.getDate() + '/' + (giorno_data.getMonth() + 1) + '/' + giorno_data.getFullYear();
 }
   
 function AddDays(date, amount) 
 { 
   var tzOff = date.getTimezoneOffset() * 60 * 1000; 
   
   var t = date.getTime(); 
   t += (1000 * 60 * 60 * 24) * amount; 
   var d = new Date(); 
   d.setTime(t); 
   var tzOff2 = d.getTimezoneOffset() * 60 * 1000; 
   
   if (tzOff != tzOff2) 
   { 
     var diff = tzOff2 - tzOff; 
     t += diff; 
     d.setTime(t); 
   } 
   
   return d; 
 }
