    $(document).on('click', '.home-sector-group-prev', function (e) {
             var $target = $(e.target).closest('.home-sector-group-prev');
             var currentGroup = $target.siblings('.home-sector-group.active').data('group');
             if (currentGroup > 1) {
                 $target.siblings('.home-sector-group').removeClass('active').addClass('hidden');
                 $target.siblings('.home-sector-group-' + (currentGroup - 1)).addClass('active').removeClass('hidden');
 
             }
         });
 
         $(document).on('click', '.home-sector-group-next', function (e) {
             var $target = $(e.target).closest('.home-sector-group-next');
             var currentGroup = $target.siblings('.home-sector-group.active').data('group');
             var totalGroups = $target.siblings('.home-sector-group');
             if (currentGroup < totalGroups.length) {
                 $target.siblings('.home-sector-group').removeClass('active').addClass('hidden');
                 $target.siblings('.home-sector-group-' + (currentGroup + 1)).addClass('active').removeClass('hidden');
             }
         });
 
         // Subscribe
         $(document).on('click', '#newsletter-button', function (e) {
             e.preventDefault();
             var data = {};
             var url = "http://laguilar.es/es/newsletter/send";
             var name = $('#subscribe_name').val();
             var email = $('#subscribe_email').val();
             var accept = $('#subscribe_accept').is(':checked');
             var err_string = "";
             if (!name) {
                 $('#subscribe_name').addClass("input-error");
             }
             if (!email) {
                 $('#subscribe_email').addClass("input-error");
             }
             if (!accept) {
                 $('#subscribe_accept').siblings('label').addClass("input-error");
             }
 
             data.name = name;
             data.email = email;
 
             if(!name) {
                 err_string+='El campo Nombre y apellidos es requerido.'+'<br>'
             }
 
             if(!email) {
                 err_string+='El campo E-mail es requerido.'+'<br>'
             }
 
             if(!accept) {
                 err_string+='El campo Condiciones generales es requerido.'+'<br>'
             }
 
 
 
             if (name && email && accept) {
                 $.post(url, data, function (response) {
                     $('#subscribe-response').html(response.message);
                 })
             } else {
                 $('#subscribe-response').html(err_string);
             }
         });
   </script>
