     jQuery.fn.initMenu = function() {  
     return this.each(function(){
         var theMenu = $(this).get(0);
         $('.hometext', this).hide();
         $('li.expand > .hometext', this).show();
         $('li.expand > .hometext', this).prev().addClass('active');
         $('li a', this).click(
             function(e) {
                 e.stopImmediatePropagation();
                 var theElement = $(this).next();
                 var parent = this.parentNode.parentNode;
                 if($(parent).hasClass('noaccordion')) {
                     if(theElement[0] === undefined) {
                         window.location.href = this.href;
                     }
                     $(theElement).slideToggle('normal', function() {
                         if ($(this).is(':visible')) {
                             $(this).prev().addClass('active');
                         }
                         else {
                             $(this).prev().removeClass('active');
                         }    
                     });
                     return false;
                 }
                 else {
                     if(theElement.hasClass('hometext') && theElement.is(':visible')) {
                         if($(parent).hasClass('collapsible')) {
                             $('.hometext:visible:first', parent).slideUp('normal', 
                             function() {
                                 $(this).prev().removeClass('active');
                             }
                         );
                         return false;  
                     }
                     return false;
                 }
                 if(theElement.hasClass('hometext') && !theElement.is(':visible')) {         
                     $('.hometext:visible:first', parent).slideUp('normal', function() {
                         $(this).prev().removeClass('active');
                     });
                     theElement.slideDown('normal', function() {
                         $(this).prev().addClass('active');
                     });
                     return false;
                 }
             }
         }
     );
 });
 };
