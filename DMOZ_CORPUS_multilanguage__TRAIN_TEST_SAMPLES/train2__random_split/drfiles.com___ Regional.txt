   function AnimateImage(element,offset,increment) {
     $(element).each(function() {
       var eTopPos = $(this).offset().top;
       var eTopPos = eTopPos - offset - 50;
       var eTopPos = Math.floor(eTopPos);
       $(this).attr('id',eTopPos);
       i = 1;
       var eChildren = $(this).children('.img-hldr').length;
       $(this).children('.img-hldr').each(function(i) {
         var counter = ++i;
         var speed = counter*increment;
         var eChildPos = parseFloat(eTopPos) + parseFloat(speed);
         var eChildPos = Math.floor(eChildPos);
         $(this).attr('id',eChildPos);
         var zChildren = eChildren-counter;
         var eChildTotal = eChildren*increment;
         var eChildEnd = parseFloat(eTopPos) + parseFloat(eChildTotal);
         var eChildEnd = Math.floor(eChildEnd);
         $(this).css('z-index',zChildren);
         if (eChildPos == eChildEnd) {
           $(this).attr('id','end');
         }
       })
     });
     $(window).scroll(function() {
       var winTopPos = $(window).scrollTop();
       $(element).children('.img-hldr').each(function() {
         var eChildID = $(this).attr('id');
         if(winTopPos >= eChildID) {
           $('.img-hldr#'+eChildID).addClass('hide');
         }
         if(winTopPos <= eChildID) {
           $('.img-hldr#'+eChildID).removeClass('hide');
         }
       });
     });
   };
