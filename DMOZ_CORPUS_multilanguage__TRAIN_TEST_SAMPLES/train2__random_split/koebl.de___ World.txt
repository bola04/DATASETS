 			function initXMLHttpRequest(div,doNotclassChange,after_call) {
 				if (window.XMLHttpRequest) {// code for IE7+, Firefox, Chrome, Opera, Safari
 					xmlhttp = new XMLHttpRequest();
 				} else {// code for IE6, IE5
 					xmlhttp = new ActiveXObject("Microsoft.XMLHTTP");
 				}

 					killsidebarmenus();
 				}
 				
 				if (div == null)  div = "inhaltsframe"; 
 				
 				xmlhttp.onreadystatechange = function() {
 					if (xmlhttp.readyState == 4 && xmlhttp.status == 200) {
 						document.getElementById(LoadingBar).style.visibility = "hidden";
 						
 						str = xmlhttp.responseText;
 						if (str.indexOf("#@#") > -1) { //ists nen Array
 							var array = new Array();
 							var zeile = str.split("@#@");							
 							for (index = 0; index < zeile.length; ++index) {								
 								var items = zeile[index].split("#@@#");
 								var itemarray = new Array();
 								for (itemindex = 0; itemindex < items.length; ++itemindex) {	
 									var item = items[itemindex].split("#@#");
 									itemarray[item[0]] = item[1];
 								}
 								array[index] = itemarray;
 							}
 						} else if (str.indexOf("#ERROR#") > -1) {					
 							var array = str.substring(7);
 						} else {	
 							document.getElementById(div).innerHTML	=	xmlhttp.responseText;
 						}
 
 						if (location_hash !== "") {
 							var NoResult = xmlhttp.responseText.indexOf("<!-- No Result -->");
 							if (NoResult !== 0) {
 								var DBPos = xmlhttp.responseText.indexOf("<!-- DB: ");
 								if (DBPos == 0) {
 									location_hash = location_hash + "&db=" + xmlhttp.responseText.substr(9,xmlhttp.responseText.indexOf("-->") - 9);
 								}
 								location.assign(location_hash.replace("#",""));
 							}
 						}if ((doNotclassChange == null) || doNotclassChange)
 							document.getElementById(div).className = "middle layout-1column";			if (after_call != null) {							
 							if (array != null) {
 								eval(after_call + "(array)");
 							} else {
 								eval(after_call + "()");
 							}
 						}
 					} else {
 						document.getElementById(LoadingBar).style.visibility = "visible";
 					}
 				}
 			}
