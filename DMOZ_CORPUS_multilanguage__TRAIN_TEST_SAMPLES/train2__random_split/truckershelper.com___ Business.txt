 function hasClass(elem, cl) {
   if (elem && cl) {
     if (elem.classList) return elem.classList.contains(cl);
     else if (!elem.className) return false;
     else return !!elem.className.match(new RegExp('(\\s|^)' + cl + '(\\s|$)'));
   }
 }
 
 function addClass(elem, cl) {
   if (elem && cl) {
     if (elem.classList)
       elem.classList.add(cl);
     else if (!elem.className) elem.className = cl;
     else if (!elem.className.match(new RegExp('(\\s|^)' + cl + '(\\s|$)')))
       elem.className += ' ' + cl;
   }
 }
 
 function removeClass(elem, cl) {
   if (elem && cl) {
     if (elem.classList)
       elem.classList.remove(cl);
     else if (elem.className)
       elem.className =
           elem.className.replace(new RegExp('(\\s|^)' + cl + '(\\s|$s)'), ' ');
   }
 }
     
 function swapClass(elem, cl_old, cl_new) {
   if (elem && cl_old && cl_new) {
     if (elem.classList) {
       elem.classList.remove(cl_old);
       elem.classList.add(cl_new);
     } else {
       elem.className = 
           elem.className.replace(new RegExp('(\\s|^)' + cl_old + '(\\s|$s)'),
                                  ' ');
       if (!elem.className.match(new RegExp('(\\s|^)' + cl_new + '(\\s|$)')))
         elem.className += ' ' + cl_new;
     }
   }
 }
