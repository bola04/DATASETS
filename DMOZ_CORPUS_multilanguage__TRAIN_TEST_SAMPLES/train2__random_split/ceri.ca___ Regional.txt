      $(function() {
 
 	  $(".btn-header-tab").click(function(e) {
 
 			var clickedButton = $(this);
 
 			$('#header-hr').hide();
 
 			// unclick all header buttons, then toggle the clicked button
 			$('.btn-header-tab').removeClass('toggle');
 			clickedButton.addClass("toggle");
 
 			// show the submenu container
 			$('.header-submenu').removeClass('hide-submenu');
 
 			// hide all submenu groups, then show the menu group with the matching menu-index
 			$('.submenu').addClass('hidden');
 			$('.submenu[data-menu-index="' + clickedButton.data('menu-index') + '"]').removeClass('hidden');
 
 			e.preventDefault();
 	  });
 
 	$(document).mouseup(function(e) {
 
 	    var container = $(".header-submenu");
 	    // if the target of the click isn't the container nor a descendant of the container
 	    if (!container.is(e.target) && container.has(e.target).length === 0){
 	        container.addClass('hide-submenu');
 					$('.submenu').addClass('hidden');
 					$('.btn-header-tab').removeClass('toggle');
 					$('#header-hr').show();
 	    }
 
 	    $("#btn-search").click(function(event){
 	    	event.preventDefault();
 
 	    	var frm = $('#form-search');
 
 	    	if($(frm).is(':visible')){
 	    		$('#form-search').submit();
 	    	} else {
 	    		$('#form-search').animate({width: 'toggle'});;
 	    		$('#form-search input')[0].focus();
 	    	}
 	    });
 
 	});
 
 });
