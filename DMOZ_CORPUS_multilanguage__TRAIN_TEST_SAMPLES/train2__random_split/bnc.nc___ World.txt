       jQuery(document).ready(function () {
             function setCookie(c_name, value, exdays) {
                 var exdate = new Date();
                 exdate.setDate(exdate.getDate() + exdays);
                 var c_value = escape(value) + ((exdays == null) ? "" : "; expires=" + exdate.toUTCString()) + "; path=/";
                 document.cookie = c_name + "=" + c_value;
             }
 
             function readCookie(name) {
                 var nameEQ = name + "=";
                 var ca = document.cookie.split(';');
                 for (var i = 0; i < ca.length; i++) {
                     var c = ca[i];
                     while (c.charAt(0) == ' ') c = c.substring(1, c.length);
                     if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
                 }
                 return null;
             }
 
             //fonction retournant le nom d'un cookie (nom=valeur)
             function getCookieName(cookie) {
                 var j = 0;
                 while (j < cookie.length && cookie.charAt(j) != '=') {
                     j = j + 1;
                 }
                 var nom_cookie = cookie.substring(0, j);
                 if(nom_cookie.charAt(0)==' '){
                     nom_cookie = nom_cookie.substring(1,nom_cookie.length); //suppression du blanc au début du nom
                 }
                 return nom_cookie;
             }
             // Fonction d'effacement d'un cookie à partir de son nom
             //source: https://www.cnil.fr/fr/solutions-pour-la-mesure-daudience
             function delCookie(name )   {
                 var path = ";path=" + "/";
                 var hostname = document.location.hostname;
                 if (hostname.indexOf("www.") === 0)
                     hostname = hostname.substring(4);
                 var domain = ";domain=" + "."+hostname;
                 var expiration = "Thu, 01-Jan-1970 00:00:01 GMT";
                 document.cookie = name + "=" + path + domain + ";expires=" + expiration;
             }
 
             //fonction permettant de supprimer tous les cookies du site (sauf cookieaccept)
             function deleteCookies() {
                 var ca = document.cookie.split(';');
                 for (var i = 0; i < ca.length; i++) {
                     var c = ca[i];
                     var nom = getCookieName(c);
                     if (nom != 'cookieaccept') {
                         delCookie(nom);
                     }
                 }
             }
 
             var cookieaccept = readCookie('cookieaccept');
 
             if (cookieaccept != null) {
                 //suppression du bandeau d'information à la page d'accueil
                 jQuery('.cnil_warning').css('display', 'none');
 
                 if(cookieaccept=='yes'){
                     if(jQuery("div#etat-cookies > p.alert").length<=0){
                         jQuery("div#etat-cookies").prepend("<p class=\"alert alert-success\">Vous avez accepté l'utilisation des cookies</p>");
                         jQuery("div#etat-cookies > a.accepter").hide();
                         jQuery("div#etat-cookies > a.refuser").show();
                     }
                 }else if(cookieaccept=='no'){
                     if(jQuery("div#etat-cookies > p.alert").length<=0){
                         jQuery("div#etat-cookies").prepend("<p class=\"alert alert-danger\">Vous avez refusé l'utilisation des cookies</p>");
                         jQuery("div#etat-cookies > a.refuser").hide();
                         jQuery("div#etat-cookies > a.accepter").show();
                     }
 
                 }
             }
 
             jQuery(".cnil_warning .close").click(function (event) {
                 event.preventDefault();
                 jQuery(".cnil_warning").remove();
 
                 setCookie("cookieaccept", "yes", 365);
                 jQuery.post('https://www.bnc.nc/', 'set_cookie=1', function () {
                 });
             });
 
             //module version 2
             //changer affichage de l'état des cookies, activer ou désactiver cookie
             //BOUTON ACCEPTER
             jQuery("div#etat-cookies > a.accepter").click(function () {
                 jQuery("div#etat-cookies > p.alert").remove();
                 jQuery("div#etat-cookies").prepend("<p class=\"alert alert-success\">Vous avez accepté l'utilisation des cookies</p>");
                 jQuery("div#etat-cookies > a.accepter").hide();
                 jQuery("div#etat-cookies > a.refuser").show();
 
                 //activer cookies
                 setCookie("cookieaccept", "yes", 365);
                 jQuery.post('https://www.bnc.nc/', 'set_cookie=1', function () {
                 });
 
                 //actualiser page d'accueil pour enlever le bandeau d'info sur les cookies
                 try{
                     window.parent.opener.location.reload();
                 }catch (error){
                     console.log(error);
                 }
 
             });
 
             //BOUTON REFUSER
             jQuery("div#etat-cookies > a.refuser").click(function () {
                 jQuery("div#etat-cookies > p.alert").remove();
                 jQuery("div#etat-cookies").prepend("<p class=\"alert alert-danger\">Vous avez refusé l'utilisation des cookies</p>");
                 jQuery("div#etat-cookies > a.refuser").hide();
                 jQuery("div#etat-cookies > a.accepter").show();
 
                 //désactiver cookies
                 setCookie("cookieaccept", "no", 365);
                 jQuery.post('https://www.bnc.nc/', 'set_cookie=0', function () {
                 });
