 function setCookie(cookieName, cookieValue, nDays) {
   var today = new Date();
   var expire = new Date();
   if (nDays == null || nDays == 0) nDays = 0;
 
   expire.setTime(today.getTime() + (3600000 * 24 * nDays));
 
   document.cookie = cookieName + "=" + encodeURIComponent(cookieValue)
           + ";expires=" + expire.toGMTString()
           + ";path=/"; //;HttpOnly;secure;";
 }

 // Redirects to a secure LA address version of the current page.
 function gotoSecureURL(sessionID) {
 
     // Get their currrent URL
     var loc = location.href;
 
     //If they login from the Password page, take them to the homepage.
     if (loc.toLowerCase().indexOf("/cpanel/password.asp") != -1){
     	loc = loc.toLowerCase().replace("/cpanel/password.asp", "/default.asp");
     }
 
     //If they login from SecretWord.asp, take them to the selected page
     if (loc.toLowerCase().indexOf("/secretword.asp?file=teamdocs.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=teamdocs.asp&", "/teamdocs.asp?");
     }else if (loc.toLowerCase().indexOf("/secretword.asp?file=schedule.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=schedule.asp&", "/schedule.asp?");
     }else if (loc.toLowerCase().indexOf("/secretword.asp?file=pictures.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=pictures.asp", "/pictures.asp");
     }else if (loc.toLowerCase().indexOf("/secretword.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp", "/roster.asp");
     }
     // Replace the host name portion with LeagueAthletics.com
     var host = location.hostname.toLowerCase();
     if (host.indexOf("dev.leagueathletics.com") == -1)
         loc = loc.replace(host, "LeagueAthletics.com");
 
     // Change protocal to https or add it if not present.
     loc = loc.replace("http://", "https://");
     if (loc.indexOf("https://") == -1 )
         loc = "https://" + loc;
 
     // Add org= parameter if not already there.
     if (loc.indexOf("org=") == -1) {
         if (loc.indexOf("?") == -1) {
             loc += "?";
         } else {
             loc += "&";
         }
 
         loc += "org=NYAATIGERS.ORG";
     }
 
     // Strip off any existing snid= values.
     var iSnid = loc.indexOf("snid=");
     if (iSnid >= 0) {
         var iNd = loc.indexOf("&", iSnid + 1)
         if (iNd == -1) iNd = loc.length - 1;
         loc = loc.substr(0, iSnid) + loc.substr(iNd);
     }
 
     // Redirect to secure version of the page.
     var aLoc = loc.split("#");
     var hashLocation = aLoc[1] || '';
     if (hashLocation) hashLocation = '#' + hashLocation;
     var tempLoc = aLoc[0] + "&snid=" + sessionID + hashLocation;
     location.href = tempLoc;
     //location.href = loc + "&snid=" + sessionID;
 }

                              <p>
                               The NYAA is a youth sports organization serving the Northwestern Lehigh School District area, providing educational athletic opportunities that range from recreational to competitive for pre-school through high school aged boys and girls.
                              </p>
                              <p>
                               The purpose of this organization, consisting of volunteers, shall be to sponsor, direct, teach, and coach the youth athletic programs (provided by the NYAA) which are not provided by the
                               <a href="http://www.nwlehighsd.org/" target="_blank">
                                Northwestern Lehigh School District.
                               </a>
                              </p>
                              <p>
                               It will be the intent of this organization to encourage and actively promote good sportsmanship, leadership, discipline and respect among the team members, parents, coaches and provide a drug-and alcohol-free environment for our youth. It is the goal of the NYAA to see that every child, in every league, should participate in every game and play at least 25% of each sport season available.
                              </p>
                              <p style="text-align: center;">
                               The NYAA is a non-profit 501(c)3 youth sports organization that operates through annual membership fees, sports fees, sponsorships and donations.

     function showResult(id){
         $.ajax({
             url: "/app/GetEvent/default.asp?eventid=" + id + "&org=NYAATIGERS.ORG",
             dataType: "xml",
             success: function(x){
                 var event = x.getElementsByTagName("Event");
                 event = event[0];
                 date = event.childNodes[3].childNodes[0].nodeValue;
                 $("#resultDate").text(date);
                 home_name = event.childNodes[21].childNodes[3].childNodes[0].nodeValue;
                 $("#resultHome").text(home_name);
                 if (event.childNodes[21].childNodes[11].hasChildNodes()) {
                     home_score = event.childNodes[21].childNodes[11].childNodes[0].nodeValue;
                 } else {
                     home_score = "";
                 }
                 $("#resultHomeScore").text(home_score);
                 away_name = event.childNodes[23].childNodes[3].childNodes[0].nodeValue;
                 $("#resultAway").text(away_name);
                 if (event.childNodes[23].childNodes[11].hasChildNodes()) {
                     away_score = event.childNodes[23].childNodes[11].childNodes[0].nodeValue;
                 } else {
                     away_score = "";
                 }
                 $("#resultAwayScore").text(away_score);
                 if (event.childNodes[17].hasChildNodes()){
                     commentary = event.childNodes[17].childNodes[0].nodeValue;
                 }else{
                     commentary = "";
                 }
                 $("#resultCommentary").html(commentary);
                 $("#resultModal").modal("show");
             }
         });
     }
