         function showFloatingLikeWindow() {
         var like_button = jQuery('.floating-like-button');
         if (!like_button) {
           return false;
         }
 
         if(jQuery('.liked-items-number').text() == '0') {
             like_button.hide();
         } else {
             like_button.show();
         }
     }
 
     function renderBurstAnimation(element, event) {
         var gif = jQuery('.heart-gif');
         var src = gif.prop('src');
 
         var x = element.find('.icon-compare-like-outline').offset().left;
         var diffX = +event.clientX - x;
         var posX = (+event.clientX - diffX) - 22;
 
         gif.css({'left' : posX , 'top' : +event.clientY - 26});
         
         if (x > 0) {
             gif.show();
             setTimeout(function() { gif.attr('src', src) }, 0);
         }
 
         setTimeout(function() {gif.hide()}, 300);
     }
 
     function renderLikesNumber(number, incrementer) {
         number += incrementer;
         jQuery('[data-favorites-num]').text(number);
         showFloatingLikeWindow();
     }
 
     jQuery(document).ready(function ($) {
         $.ajax({
             url: 'https://sonnyalexanderflorists.com/favorites/index/get/',
             type:'get',
             dataType: 'json',
             success: function(data, status, jqXHR) {
                 var products = data.product_ids;
                 if ($.isArray(products)) {
                     var likedProducts = products.length;
                     if (likedProducts > 0) {
                         $('.item_listings .item').each(function() {
                             var itemProductId = $(this).find('.item_listing-information').data('id');
                             if (typeof itemProductId !== 'undefined' && $.inArray(itemProductId, products) != -1) {
                                 $(this).find('.product-favorite-heart').addClass('favorite-added');
                                 $(this).find('.favorites-text').html('REMOVE FROM FAVORITES<i class="fa fa-times-thin fa-2x" aria-hidden="true"></i>');
                                 $(this).find('.icon-compare-like-outline').hide();
                                 $(this).find('.icon-compare-like-fill').show();
                             }
                         });
                         $('.liked-items-number').text(likedProducts);
                         showFloatingLikeWindow();
                     }
                 }
             }
         });
 
         $('.product-favorite-heart').on('click', function(event){
             event.preventDefault();
             var likesNumber = $('.liked-items-number').text() || $('.favorite-number').text();
             if ($(this).hasClass('favorite-added')) {
                                 $(this).removeClass('favorite-added');
                 $(this).find('.icon-compare-like-outline').show();
                 $(this).find('.icon-compare-like-fill').hide();
                 $(this).find('.favorites-text').text('ADD TO FAVORITES');
 
                 $.ajax({
                     url: 'https://sonnyalexanderflorists.com/favorites/index/remove/',
                     type:'post',
                     data: {product_id: $(this).parent().data('id')},
                     dataType: 'json',
                     success: function(data, status, jqXHR) {
                         if (data.success == true) {
                             renderLikesNumber(parseInt(likesNumber), -1);
                         }
                     }
                 });
             } else {
                 $(this).addClass('favorite-added');
                 renderBurstAnimation($(this), event);
                 $(this).find('.icon-compare-like-outline').hide();
                 $(this).find('.icon-compare-like-fill').show();
                 $(this).find('.favorites-text').html('REMOVE FROM FAVORITES<i class="fa fa-times-thin fa-2x" aria-hidden="true"></i>');
 
                 // add the chosen item to favorite list
                 $.ajax({
                     url: 'https://sonnyalexanderflorists.com/favorites/index/add/',
                     type:'post',
                     data: {product_id: $(this).parent().data('id')},
                     dataType: 'json',
                     success: function(data, status, jqXHR) {
                         if (data.success == true) {
                             renderLikesNumber(parseInt(likesNumber), 1);
                         }
                     }
                 });
             }
 
             return false;
         });
 
         $('.product-favorite-heart').hover(
             function() {

       function codeAddress(address) {
     var geocoder = new google.maps.Geocoder(),
         mapContainer = document.getElementById('map-container');
 
     geocoder.geocode({ address: address }, function(results, status) {
         var map;
 
         if (status === google.maps.GeocoderStatus.OK) {
             var results = results[0],
                 location = results.geometry.location,
                 mapOptions = {
                     zoom: 16,
                     center: location,
                     mapTypeId: google.maps.MapTypeId.ROADMAP,
                     streetViewControl: false,
                     mapTypeControl: false,
                     scrollwheel: false
                 };
 
             map = new google.maps.Map(mapContainer, mapOptions);
 
             var marker = new google.maps.Marker({
                 map: map,
                 position: results.geometry.location
             });
 
             var contentString = '8008 West Third Street Los Angeles, CA, 90048';
 
             var infowindow = new google.maps.InfoWindow({
                 content: contentString,
                 maxWidth: 200
             });
 
             google.maps.event.addListener(marker, 'click', function() {
                 infowindow.open(map, marker);
             });
 
             google.maps.event.addDomListener(window, 'resize', function() {
                 map.setCenter(location);
             });
         }
     });
 }
