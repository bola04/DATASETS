 	function majDate(locale) {
 		// create Date object for current location
 	    var d = new Date();
 
 	    // convert to msec
 	    // add local time zone offset
 	    // get UTC time in msec
 	    var utc = d.getTime() + (d.getTimezoneOffset() * 60000);
 
 	    // create new Date object en UTC-10
 	    var nd = new Date(utc + (3600000*-10));
 		
 	    var maDate = nd.toLocaleDateString();
     	document.getElementById('dd_day').innerHTML = maDate.substring(0, 1).toUpperCase() + maDate.substring(1);
     	var temps = nd.toLocaleTimeString();
     	
     	//on doit supprimer les secondes
     	//on regarde la fin si c'est un AM/PM ou pas 
     	var fin = temps.substring(temps.length - 2).toUpperCase();
     	var tempsFinal;
     	if (fin == 'AM' || fin == 'PM') {	//on est en AM/PM
     		tempsFinal = temps.substring(0, temps.length - 6) + temps.substring(temps.length - 3); //on retire les secondes
     	} else {
     		tempsFinal =  temps.substring(0, temps.length - 3);	//on est en 24h => on retire les secondes
     	}
     	document.getElementById('dd_time').innerHTML = tempsFinal;
 	}
