    var categorieElement = '.categorie';
         var marqueElement = '.marque';
         $(categorieElement).on('change', function () {
             var categorie_id = $(this).val();
             $.ajax({
                 type: "POST",
                 url: "/form",
                 data: {'categorie_id': categorie_id},
                 dataType: 'JSON',
                 timeout: 30000,
                 success: function (data) {
                     $(marqueElement).html('<option value>Marque</option>');
                     $.each(data, function (index) {
                         var id = data[index].id;
                         var nom = data[index].nom;
                         $(marqueElement).append('<option value="' + id + '">' + nom + '</option>');
                     });
                 }
             });
         });
 
         $('form').submit(function (e) {
             e.preventDefault();
             var data = $(this).serialize();
             $.ajax({
                 type: "POST",
                 url: "/recherche",
                 data: data,
                 cache: false,
                 success: function (response) {
                     if (response.error === true) {
                         $("#recherche_r").html(response.materiels);
                         $('#resultat_de_recherche').modal('show');
                     } else if (response.error === false) {
                         $("#recherche_r").html(response.materiels);
                         $('#resultat_de_recherche').modal('show');
                     }
                 }
             });
 
         });
