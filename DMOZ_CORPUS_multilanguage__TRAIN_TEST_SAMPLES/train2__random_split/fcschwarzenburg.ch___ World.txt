 var signin_submitForm = function() {
     //FgXmlHttp.regularpost('/login',$("#loginForm").serialize(),false,success_signin);
     //$.post('/login',$("#loginForm").serialize(),success_signin);
     FgXmlHttp.regularpost('/login', $("#loginForm").serialize(), false, function(data){
         if (data.status == 'success') {
             window.location.reload();
         } else {
             success_signin(data);
         }
     });
     return false;
 }
 function success_signin(data){
     $('#blank_clubidentifier').hide();
     $('#blank_username').hide();
     $('#blank_password').hide();
     var var_status = data.status;
     var club_status= data.error;
     //        alert(club_status);
     //        alert(var_status);
     var forgot_password = 0;
     if(var_status=="success"){
         if (forgot_password) {
             location.href='http://www.fcschwarzenburg.ch';
         } else {
             //$("#loginForm").submit();
             window.location.reload();
         }
     }
     else if(var_status=="success_step1"){
         if(data.club){
             location.href=data.club;
         }
         else if(data.redirect){
             location.href=data.redirect;
         }else{
             location.href = '/community/overview';
         }
     }else{
         if(club_status=='blocked'){
             location.href='/sfGuardForgotPassword/wronglink/fq/1';
             return false;
         }
         else
             location.href=data.redirect;
     }
 }
