 function CarouselSelect(id)
 {
     for (var i = 0; i < total; i++)
     {
         document.getElementById("carousellist_" + i).className = "item";
     }
     
     for (var i = 0; i < total; i++)
     {
         document.getElementById("carouselitem_" + i).style.display = "none";
     }
     
     document.getElementById("carousellist_" + id).className = "itemselected";
     document.getElementById("carouselitem_" + id).style.display = "block";
 }
 
 function ScrollCarousel()
 {
     if (scrollCarousel)
     {
         carouselID++;        
         if (carouselID == total) carouselID = 0;
         CarouselSelect(carouselID);        
         window.setTimeout("ScrollCarousel()",7000);
     }
 }
