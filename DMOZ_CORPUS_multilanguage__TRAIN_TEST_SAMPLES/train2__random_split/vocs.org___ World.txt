 	window.onscroll = function() {
 
 		function getScrollY() {
 			if( typeof( window.pageYOffset ) == 'number' ){return window.pageYOffset;}
 			else if( document.body && ( document.body.scrollLeft || document.body.scrollTop ) ){return document.body.scrollTop;}
 			else if( document.documentElement && ( document.documentElement.scrollLeft || document.documentElement.scrollTop ) ){return document.documentElement.scrollTop;}
 			return 0;
 		}
 
 		var y = getScrollY();
 		if (y > menuPos && !$('#menu_lateral').hasClass("fixed")){$('#menu_lateral').addClass('fixed');}
 		else if (y < menuPos && $('#menu_lateral').hasClass("fixed")){$('#menu_lateral').removeClass("fixed");}

    <script type="text/javascript">
     function getCookie(cname) {
 			var name = cname + "=";
 			var ca = document.cookie.split(';');
 			var caLen = ca.length;
 			for(var i=0; i<caLen; i++) {
 				var c = ca[i];
 				while (c.charAt(0)==' ') c = c.substring(1);
 				if (c.indexOf(name) == 0) return c.substring(name.length, c.length);
 			}
 			return '';
 		}
 		var vocs = function(){
 			return true;
 		};
 
 		vocs.cookie = function(cname, cvalue, exdays, cpath) {
 
 			var dcookie, options = {}, d, expires;
 
 			if(typeof cvalue === 'undefined') return getCookie(cname);
 
 			if(typeof exdays === 'object'){
 				options = exdays;
 			} else {
 				if(typeof cdomain !== 'undefined') options.domain = cdomain;
 				if(typeof cpath !== 'undefined') options.path = cpath;
 				if(typeof exdays !== 'undefined') options.expires = exdays;
 				if(typeof csecure !== 'undefined') options.secure = csecure;
 			}
 
 			if(typeof options.expires !== 'undefined'){
 				d = new Date();
 				d.setTime(d.getTime() + (options.expires*24*60*60*1000));
 				expires = d.toUTCString();
 			}
 
 			var dc1 = (typeof expires !== 'undefined')? ' expires=' + expires + ';' : '';
 			var dc2 = (typeof options.domain !== 'undefined')? ' domain=' + options.domain + ';' : '';
 			var dc3 = (typeof options.path !== 'undefined')? ' path=' + options.path + ';' : '';
 			var dc4 = (typeof options.secure !== 'undefined')? ' secure;' : '';
 
 			dcookie = cname + '=' + cvalue + ';' + dc1 + dc2 + dc3 + dc4;
 
 			document.cookie = dcookie;
 		};
 
 		vocs.removeCookie = function(cname){
 			vocs.cookie(cname, '', -1);
 		};
 
 		vocs.cookies = function() {
 			var splits = document.cookie.split(";");
 			var cookies = {};
 			var splitsLen = splits.length;
 			for (var i = 0; i < splitsLen; i++) {
 			 var split = splits[i].split("=");
 			 cookies[split[0]] = unescape(split[1]);
 			}
 			return cookies;
 		};
 
 		vocs.removeCookies = function () {
 			var cookies = vocs.cookies();
 			for (var key in cookies) {
 				if (cookies.hasOwnProperty(key)) {
 					vocs.removeCookie(key);
 				}
 			}
 		};
 
 		var cookies = $('#cookies-warning');
 		var cookiesHeight = cookies.height();
 
 
 		if(vocs.cookie('vocs_cookies') === ''){
 			cookies.addClass('show');
 			setTimeout(function(){
 				cookies.css('bottom', -1*cookiesHeight + 'px');
 			}, 10);
 			setTimeout(function(){
 				cookies.css('bottom', '0px');
 			}, 10);
 		}
 
 		$('.or-btn-entendido').click(function(){
 
 			vocs.cookie('vocs_cookies', 'aceptada', 365, '/', '.vocs.org');
 
 			cookies.css('bottom', -1*cookiesHeight + 'px');
 
 			setTimeout(function(){
 				cookies.removeClass('show');
 			}, 500);
 		});
    </script>
