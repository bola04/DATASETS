    function viewport() {
     var e = window, a = 'inner';
     if (!('innerWidth' in window )) {
         a = 'client';
         e = document.documentElement || document.body;
     }
     return { width : e[ a+'Width' ] , height : e[ a+'Height' ] };
 }
 	$(window).load(function(e) {
 		var rtime = new Date(1, 1, 2000, 12,00,00);
 		var timeout = false;
 		var delta = 200;
 		$(window).resize(function() {
 			rtime = new Date();
 			if (timeout === false) {
 				timeout = true;
 				setTimeout(resizeend, delta);
 			}
 		});
 		
 		if(viewport().width >= 1220) {
 			var oldWidth = "desktop";
 			$(window).trigger("desktop", [ "load" ]);
 		} else if(viewport().width >= 720) {
 			var oldWidth = "tablet";
 			$(window).trigger("tablet", [ "load" ]);
 		} else {
 			var oldWidth = "phone";
 			$(window).trigger("phone", [ "load" ]);
 		}
 		$(window).trigger("afterSize");
 		
 		window.currentWidth = oldWidth;
 		
 		function resizeend() {
 			if (new Date() - rtime < delta) {
 				setTimeout(resizeend, delta);
 			} else {
 				timeout = false;
 				if(viewport().width>=1220){
 					if(oldWidth != "desktop") {
 					//	$(window).trigger("desktop", [ oldWidth ]);
 					}
 					newWidth = "desktop";
 				} else {
 					if(oldWidth == "desktop") {
 						$(window).trigger("notDesktop");
 					}
 				}
 				if(viewport().width<1220 && $(window).width()>=720) {
 					if(oldWidth != "tablet") {
 					//	$(window).trigger("tablet", [ oldWidth ]);
 					}
 					newWidth = "tablet";
 				} else {
 					if(oldWidth == "tablet") {
 						$(window).trigger("notTablet");
 					}
 				}
 				if(viewport().width>=720){
 					if(oldWidth == "phone") {
 						$(window).trigger("notPhone");
 					}
 				} else {
 					if(oldWidth != "phone") {
 					//	$(window).trigger("phone", [ oldWidth ]);
 					}
 					newWidth = "phone";
 				}
 				if(oldWidth != newWidth) {
 					$(window).trigger(newWidth, [ oldWidth]);
 				}
 				$(window).trigger("newResize", [ oldWidth, newWidth ]);
 				oldWidth = newWidth;
 				window.currentWidth = newWidth;
 			}               
 		}
 	});

   <script type="text/javascript">
    function createCollapsable(targetGroup, noCloseClick) {
 	$("*[data-z-group='"+targetGroup+"'][data-z-c='collapsable']").css("display", "none");
 	if(!noCloseClick) {
 		$("*[data-z-group='"+targetGroup+"'][data-z-c='collapsable']").click(function(e) {
 			var targetSet = $(this).attr("data-z-c-set");
 		   collapseCollapsable($("*[data-z-c-set='"+targetSet+"'][data-z-c='link']"));
 		});
 	}
 	$("*[data-z-group='"+targetGroup+"'][data-z-c='link']").click(function(e) {
        if($(this).hasClass("expanded")) {
 			collapseCollapsable($(this));
 		} else {
 			expandCollapsable($(this));
 		}
     });
 }
 function destroyCollapsable(targetGroup) {
 	$("*[data-z-group='"+targetGroup+"'][data-z-c='collapsable']").attr("style", "");
 	$("*[data-z-group='"+targetGroup+"'][data-z-c='link']").off("click");
 	$("*[data-z-group='"+targetGroup+"'][data-z-c='collapsable']").off("click");
 	$("*.expanded[data-z-group='"+targetGroup+"'][data-z-c='link']").removeClass("expanded");
 }
 function expandCollapsable(targetElement) {
 	var targetGroup = targetElement.attr("data-z-group");
 	var targetSet = targetElement.attr("data-z-c-set");
 	if($("*.expanded[data-z-group='"+targetGroup+"']").length) { //Close the one that is open (if one is open)
 		collapseCollapsable($("*.expanded[data-z-group='"+targetGroup+"']"));
 	}
 	targetElement.addClass("expanded");
 	$("*[data-z-group='"+targetGroup+"'][data-z-c-set='"+targetSet+"'][data-z-c='collapsable']").slideDown(300);
 }
 function collapseCollapsable(targetElement) {
 	var targetGroup = targetElement.attr("data-z-group");
 	var targetSet = targetElement.attr("data-z-c-set");
 	targetElement.removeClass("expanded");
 	$("*[data-z-group='"+targetGroup+"'][data-z-c-set='"+targetSet+"'][data-z-c='collapsable']").slideUp(300);
 }

 		$("#header>.menu>li").removeClass("selected");
 		targetEl = $(thisEl).attr("class");
 		$(thisEl).addClass("selected");
 		if(!$("#" + targetEl).hasClass("selected")) {

 				switchExMenu(targetEl);
 			} else {
 				closeExMenu()
 			}
 		}

 			if($("#"+targetEl).length){
 				$(".expandedMenu").removeClass("selected");
 				$("#" + targetEl).addClass("selected");
 				openExMenu();
 			}
 		}
 	}

 			$(this).off(e);
 		});
 	}
