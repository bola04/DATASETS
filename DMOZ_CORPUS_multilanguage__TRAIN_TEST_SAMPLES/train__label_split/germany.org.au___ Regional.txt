 		function decryptCharcode(n,start,end,offset) {
 			n = n + offset;
 			if (offset > 0 && n > end) {
 				n = start + (n - end - 1);
 			} else if (offset < 0 && n < start) {
 				n = end - (start - n - 1);
 			}

 			var len = enc.length;
 			for(var i=0; i < len; i++) {

 					dec += enc.charAt(i);
 				}
 			}
 			return dec;
 		}
