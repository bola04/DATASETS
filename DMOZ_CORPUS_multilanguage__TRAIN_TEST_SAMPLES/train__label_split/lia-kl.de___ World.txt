            <a class="external-link" href="http://chf-kl.de/" target="_blank" title="Öffnet einen externen Link im aktuellen Fenster">
             <img alt="Opens external link in current window" src="fileadmin/styles/02_layouts_2col/img/icons/external_link.gif"/>

            </a>
