                     $('nav ul').css('display', 'none');
                     $('nav').animate({
                             height: '40px'
                         },
                         500,
                         function() {
                         }
                     );
                 }
                 else {
                     $('nav').animate({
                             height: '+='+$('nav ul').css('height')
                         },
                         500,
                         function() {
                             $('nav ul').css('display', 'block');
                         }
                     );
                 }
             });
 
             $('.sidebar-toggle').click(function() {
                 $('.main aside').css('background-color', $('.main .fond-cote').css('background-color'));
                 $('.main aside').toggle();
 
                 $('.blogmember_menu').toggle();
 
                 if($('.blogmember_menu').is(":visible")) {
                     $('#contenu').addClass('members-active');
                 }
                 else {
                     $('#contenu').removeClass('members-active');
                 }
             });
         
         });
   </script>
