 //RSS News表示
 function renderNews(){
 	$.ajax({
 		type: "POST",
 		url: "",
 		//async: false, //false=同期
 		//traditional: true,
 		dataType: "json",
 		data: {
 				"getNews": 1,
 		}
 	}).done(function(dta){
 		if (dta.length>0){
 			dta.forEach(function(v,i){
 				var article=$("<article>");
 				var time=$("<time>");
 				var a=$("<a>");
 				var h3=$("<h3>");
 				var p=$("<p>");
 				
 				time.attr("datetime",v.date1);
 				time.html(v.date2);
 				
 				a.attr("href",v.link);
 				h3.html(v.title);
 				p.html(v.description);//esc済なのでhtml可
 				
 				a.append(h3).append(p);
 				article.append(time).append(a);
 				
 				$("#rssNews1").append(article);
 			});
 			$("article:even","#rssNews1").addClass("list01-odd");
 			$("article:odd","#rssNews1").addClass("list01-even");
 		}
 	}).fail(function(jqXHR, statusText, errorThrown){
 		//alert(statusText+': '+errorThrown);
 	});
 }
