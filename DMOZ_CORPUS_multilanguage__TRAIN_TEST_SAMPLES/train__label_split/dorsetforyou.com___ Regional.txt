      $(document).ready(function () {
                 var alertTemplate = $([
                     '<div id="alert-{{id}}" class="alert alert--closed alert--{{alertClass}} clearfix row column">',
                         '<div class="alert__icon -{{alertClass}}"><span class="visuallyhidden">{{alertType}} symbol</span></div>',
                         '<div class="alert__content">',
                             '<h2 class="alert__title">{{alertHeading}}</h2>',
                             '{{alertDetails}}',
                         '</div>',
                         '<a href="#" class="alert__close close">',
                             '<span class="visuallyhidden">Close alert</span>',
                             '<span class="close__bar"></span>',
                             '<span class="close__bar"></span>',
                         '</a>',
                     '</div>'
                 ].join("\n"));
 
 
                 var alertTypes = [
                     'Global', 'Section', 'Page', 'Entry'
                 ];
 
                 for (var i = 0; i < alertTypes.length; i++) {
 
                     var alertsAPI = '/dorset-api/custom/alerts/GetAlerts/' + alertTypes[i];
 
                     $.getJSON(alertsAPI, {
                         contentPath: '/home.aspx',
                         alertReference: '',
                         entryId: ''
                     })
                     .done(function (data) {
                         $.each(data, function (i, alert) {
                             $('.alert-container').append(alertTemplate).html(function (index, html) {
                                 return html.replace(/{{id}}/g, alert.sys.id)
                                     .replace(/{{alertClass}}/g, alert.alertType.name.replace(/ /g, '').toLowerCase())
                                     .replace(/{{alertType}}/g, alert.alertType)
                                     .replace(/{{alertHeading}}/g, alert.alertHeading)
                                     .replace(/{{alertDetails}}/g, alert.alertDetails);
                             });
                         });
                         handleAlertCookies();
                     });
                 }
 
                 function handleAlertCookies() { //this is duplicated in app.js
 
                     var cookieVal = readCookie('hidden') !== null ? readCookie('hidden') : '';
 
                     $('.alert').each(function () {
 
                         var id = $(this).attr('id');
                         if (cookieVal.indexOf(id) < 0) {
                             $(this).removeClass('alert--closed');
                         }
                     });
 
                     $('.alert__close').click(function () {
                         $(this).parent().slideUp();
                         if ($.cookie('hidden') !== undefined) {
                             cookieVal = $.cookie('hidden');
                         }
                         var newcookieVal = cookieVal + $(this).parent().attr('id');
                         $.cookie('hidden', newcookieVal, {
                             expires: 7
                         });
                         return false;
                     });
                 }
             });
     </script>
