        <script type="text/javascript">
         $(document).ready(function() {
     
     
     if(!($(".responsive_menu_head").is(':visible')))
     {
         // wersja nieresponsywna
         $(".top_links .menu_level1 .level1_span").hover(function() {
             if($(".top_links #"+this.id+" .ul_menu_child").length !== 0) {
                 var submenu = $(".top_links #"+this.id+" .ul_menu_child");
                 if(!($(submenu).is(':visible'))) {
                     $(submenu).show();
                     // Roznica pozycji poczatku listy a lewego konca podmenu
                     var leftDiff = parseInt($('.top_links').offset().left - $(submenu).offset().left);
 
                     // Roznica pozycji konca listy a prawego konca podmenu
                     var rightDiff = parseInt(($(submenu).offset().left + $(submenu).width()) - ($('.top_links').offset().left + $('.top_links').width()));
 
                     // Polowa szerokosci podmenu
                     var halfSubmenuWidth = parseInt(($(submenu).width() / 2));
 
                     // Jesli podmenu znajduje sie poza lewa czesci menu
                     if(leftDiff > 1) {
                         $(submenu).css('left', "-" + (halfSubmenuWidth - leftDiff) + "px");
                     }
                     // Jesli podmenu znajduje sie poza prawa czesci menu
                     if(rightDiff > 1) {
                         $(submenu).css('left', "-" + (halfSubmenuWidth + rightDiff) + "px");
                     }
                     if($(submenu).css('width') > $(this).css('width'))
                     {
                         $(submenu).width($(this).width()-20);
                     }
                 }
             }
         },
         function() {
             $(".top_links #"+this.id+" .ul_menu_child").slideUp({duration: 200});
         });
     } else {
         // wersja responsywna
         $('.top_links .menu_level1 .level1_span a').click( function(){
             var href = $(this).attr("href");
             if(href.substring((href.length)-1, href.length) == '#'){
                 console.log($(this).parent());
                 if($(this).parent().hasClass('menu_open'))
                     $(this).parent().removeClass('menu_open');
                 else
                     $(this).parent().addClass('menu_open');
                 $(this).parent().parent().find('ul').slideToggle(300);
                 return false;
             }
         });
     }
     $(".top_links .menu_level2").hover(function() {
         if($(".top_links #"+this.id+" .ul_menu_child_l2").length !== 0) {
             var submenu = $(".top_links #"+this.id+" .menu_con_l2");
             if(!($(submenu).is(':visible'))) {
                 $(submenu).show();
 
                 var submenuWidth = parseInt(($('.top_links .menu_con').width()));
                 $(submenu).css('margin-left', (submenuWidth + 1) + 'px');
             }
         }
     },
     function() {
         $(".top_links #"+this.id+" .menu_con_l2").slideUp({duration: 200});
     });
     $('.responsive_menu_head').click(function(){
         $(this).parents('.top_links').find('div.menu_level1').slideToggle(1000, function(){
             if($(this).hasClass('expand')){
                 $(this).removeClass('expand');
             } else{
                 $(this).addClass('expand');   
             }
             $(this).css('display', '');
         });
     });
 });
