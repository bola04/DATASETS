 function handleAjaxLink(obj)
 {	
 	
 		if(jQuery(obj).hasClass('ajaxActive'))
 			return false;
 		else
 		{
 			var activeLi = jQuery('a.ajaxLink.ajaxActive');
 			if(activeLi.length > 0)
 			{
 				jQuery(activeLi).removeClass('ajaxActive');
 				var prevTargetID = jQuery(activeLi).attr('data-target');
 				jQuery('#elementID_'+prevTargetID).remove();
 			}			
 			jQuery(obj).addClass('ajaxActive');
 			var newTargetId = jQuery(obj).attr('data-target');
 			jQuery('#timelineSidebarUL').html('');			
 
 			jQuery(lastJSOnResponse).each(function(i,item){
 				var activeClass = (item.postID == newTargetId)?' ajaxActive':'';						
 				jQuery('#timelineSidebarUL').append('<li><a href="javascript:void(0);" onclick="handleAjaxLink(this);" data-target="'+item.postID+'" class="ajaxLink'+activeClass+'">'+item.title+'</a></li>');
 				if(item.postID == newTargetId)//only add the first
 				{					
 					jQuery('#timeline').append(item.print_html);				
 					$enableScrollSpy = false;
 					$timelineItems = $('#timeline div.element');
 					timeline();
 					iframeAutoWidth();
 				}
 			});
 		}
 	
 }
