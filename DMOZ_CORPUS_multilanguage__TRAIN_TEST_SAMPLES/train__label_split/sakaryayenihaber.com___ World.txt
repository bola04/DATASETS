      $(document).ready(function(){			  
         function ResimDegistir() {
         var wait = $(".GenisManset .Resimler").attr("wait"); 
         if(wait == 2){
             var ResimSayisi = $(".GenisManset .miniResim li").length;
             var AktifBul = $(".GenisManset .miniResim").find("li.aktif").attr("rel"); 
 
             if(ResimSayisi == AktifBul){
                 var AktifBul = 1;//1 aktif olcak
             }else{
                 var AktifBul = parseInt(AktifBul) + 1;
             }	
             var SonrakiResim = AktifBul;		
             for(var i = 1;i<= ResimSayisi;i++){
                 var AktifDiv = "r" + i;
                 if(i == AktifBul){							
                     $("#"+AktifDiv).addClass("aktif");
                     url = $("#"+AktifDiv).find("a").attr("href");  
 					title = $("#"+AktifDiv).find("a").attr("title"); 
                     target = $("#"+AktifDiv).find("a").attr("target"); 
                     manset = $("#"+AktifDiv).find("a").attr("manset");

                 }else{		
                     $("#"+AktifDiv).removeClass("aktif");
                 }
             }		
         }else if(wait == 3){
         
         }else{
             $(".GenisManset .Resimler").attr("wait",2);  
         }
         setTimeout(ResimDegistir, 3000);				
     }
