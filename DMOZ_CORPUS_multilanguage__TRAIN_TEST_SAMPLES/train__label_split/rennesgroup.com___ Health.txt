                     // rehab link
                     if ( $(this).hasClass('active') ) {
                         $(this).removeClass('active');
                         $(this).next('.double').slideUp();
                     } else {
                         $(this).addClass('active');
                         $(this).next('.double').slideDown();
                     }
                 } else {
                     // all links except rehab
                     if ( $(this).hasClass('active') ) {
                         $(this).removeClass('active');
                         $(this).next('.sub-navigation').slideUp();
                     } else {
                         $(this).addClass('active');
                         $(this).next('.sub-navigation').slideDown();
                     }
                 }
             }
 
         });
 
     });
