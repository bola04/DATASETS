    function validationfeedback()
 	{
 		if(document.feedbackform.name2.value =='Name :')
 			{
 				alert("Please Enter Your Name");
 				document.feedbackform.name2.focus();
 				return false;
 			}
 		if ((document.feedbackform.name2.value.length < 2) || (document.feedbackform.name2.value.length > 20))
 			{
 				alert("Please Enter Name between 2 to 20 character");
 				document.feedbackform.name2.focus();
 				return false;
 			}
 	
 		if(!isNaN(document.feedbackform.name2.value))
 			{
 				alert("Please Enter Character Only.");
 				document.feedbackform.name2.focus();
 				return false;
 			}
 			
 		if(document.feedbackform.phone2.value =='Mobile Number :')
 			{
 				alert("Please Enter Mobile No.");
 				document.feedbackform.phone2.focus();
 				return false;
 			}
 		if(isNaN(document.feedbackform.phone2.value))
 			{
 				alert("Please Enter Contact Only Digit.");
 				document.feedbackform.phone2.focus();
 				return false;
 			}
 		if ((document.feedbackform.phone2.value.length < 10) || (document.feedbackform.phone2.value.length > 13))
 			{
 				alert("Please Enter Correct Mobile No.");
 				document.feedbackform.phone2.focus();
 				return false;
 			}	
 		if(document.feedbackform.email2.value =='Email Address :')
 			{
 				alert("Please Enter Email Id");
 				document.feedbackform.email2.focus();
 				return false;
 			}
 	   else
 			{
 			var email_id_validate=/^[-0-9a-zA-Z.+_]+@[-0-9a-zA-Z.+_]+\.[a-zA-Z]{2,4}$/;
 			var email=document.feedbackform.email2.value;
 		if(!email.match(email_id_validate))
 				{
 				alert("Fill Correct Email Id");
 				document.feedbackform.email2.focus();
 				return false;
 				}	
 	
 			}
 			
 		if(document.feedbackform.location2.value =='- Location -')
 			{
 				alert("Please Select Your Location");
 				document.feedbackform.location2.focus();
 				return false;
 			}	
 			
 		if(document.feedbackform.message2.value=='Message :')
 		{
 			alert("Please Enter your Message");
 			document.feedbackform.message2.focus();
 			return false;
 		}	
 
 	}
