     var setEventListenerSearchModulemobilesearch = function () {
         var formElement = jQuery('#vmss_search_form_modulemobilesearch');
 
         if (formElement == undefined) {
             return;
         }
 
         formElement.submit(function (event) {
             var searchWord = jQuery('#vmss_search_form_modulemobilesearch input[name=searchword]').val();
             searchWord = encodeURIComponent(searchWord);
 
             event.preventDefault();
 
             var formURL = '/search?searchword=' + searchWord;
 
             formURL = formURL.replace('?', '#');
 
             
             document.location = formURL;
         });
 
         moduleSearchTextField = formElement.find('input[name=searchword]');
         moduleSearchTextField.keyup(function (event) {
             var value = jQuery(this).val();
 
             searchTextField = jQuery('#vmss_search_form_component input[name=searchword]').val(value);
 
             //update other search modules
 
             jQuery('.vmss_search_form_module_class input[name=searchword]').each(function (index, element) {
                 if (jQuery(element).prop('id') != moduleSearchTextField.prop('id')) {
                     jQuery(element).val(value);
                 }
             });
 
             //update advanced search
             try {
                 jQuery('.search-form input[name=searchword]').each(function (index, element) {
                     if (jQuery(element).prop('id') != moduleSearchTextField.prop('id')) {
                         jQuery(element).val(value);
                     }
                 });
             } catch (e) {
                 ;
             }
 
             var key = event.keyCode || event.which;
 
             if (key === 13) {
                 formElement.submit();
             }
         });
     }
