               <br/>
               De zorg en begeleiding die er inmiddels al de nodige jaren zijn voor mijn man en onze dochter, maakten dat zij op enig moment een mantelzorgcompliment mochten uitdelen.

     AantalDagenVerlopen = 20; //Na hoveeel dagen verloopt de popup en wordt hij opnieuw getoond
     PupupTimeOut = 3000;
 
     //Popup uniek ID geven
     var cookiePopupAantalKeerGezien = "popupAantalKeerGezien_ID" + 4810
     var cookiePopupTonen = "popupTonen_ID" + 4810
 
     if (Cookies.get(cookiePopupAantalKeerGezien)) {
         if (Cookies.get(cookiePopupAantalKeerGezien) <= AantalKeerTonen){
             var aantal = Cookies.get(cookiePopupAantalKeerGezien);
             aantal++; // 1 bijop tellen
             Cookies.set(cookiePopupAantalKeerGezien, aantal, { expires: AantalDagenVerlopen });
         }
         else {
             Cookies.set(cookiePopupAantalKeerGezien, AantalKeerTonen + 1, { expires: AantalDagenVerlopen });
         }
     }
     else {
         //Bij het eerste bezoek zet hij een cookie om te tellen hoe vaak hij getoond is.
         Cookies.set(cookiePopupAantalKeerGezien, 1, { expires: 14 });
     }
 
     if (Cookies.get(cookiePopupTonen) != 'false' && Cookies.get(cookiePopupAantalKeerGezien) <= AantalKeerTonen) $("body").addClass("toonMelding");
 
     $('#closepopup').off('click').on('click', function (e) {
         e.preventDefault();
         $(this).closest('.popupwrap').fadeOut(200);
         Cookies.set(cookiePopupTonen, 'false', { expires: AantalDagenVerlopen });
     });
 
     //Als je op een link in de poup klikt moet hij hem vervolgens ook niet meer tonen
     $('.tmapopup .popupwrap .popup .tekst  a').off('click').on('click', function (e) {
         Cookies.set(cookiePopupTonen, 'false', { expires: AantalDagenVerlopen });
     });
 
 
     setTimeout(function () {
         $("body").addClass("invliegenMelding");
     }, PupupTimeOut);
    </script>
