 		function setEqualHeight(selector, addTrigger) {
 		  var elements = $(selector)
 		  padTop = $(selector).css("padding-top");
 		  padBot = $(selector).css("padding-bottom");
 		  elements.css("height", "auto");
 		  var biggestOne = -1;
 		  $.each(elements, function(index, item) {
 			  if ($(item).height() > biggestOne) {
 				  biggestOne = $(item).height();
 			  }
 		  })
 		  $(selector).css("height", (parseInt(biggestOne) + parseInt(padTop) + parseInt(padBot)) + "px")
 		  if (addTrigger == true) {
 			  $(document).on("input", selector, function() {
 				  setEqualHeight(selector, false)
 			  })

 				  setEqualHeight(selector, false)
 			  })
 		  }
