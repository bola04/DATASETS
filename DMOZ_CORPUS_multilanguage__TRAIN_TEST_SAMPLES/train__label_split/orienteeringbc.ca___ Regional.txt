           <h2>
            Welcome!
           </h2>
           <p>
            <img alt="" class="alignright size-medium wp-image-159" height="300" sizes="(max-width: 200px) 100vw, 200px" src="http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic-200x300.jpg" srcset="http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic-200x300.jpg 200w, http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic.jpg 683w, http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic-230x345.jpg 230w, http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic-350x525.jpg 350w, http://orienteeringbc.ca/wp/wp-content/uploads/2017/03/Brian-May-Sprint-Camp-Pic-480x720.jpg 480w" width="200"/>
           </p>
           <p>
            Orienteering is an outdoor adventure activity involving cross-country navigation by foot over unknown terrain, using a map and a compass. It is enjoyed as a relaxing recreation, or as a challenging competitive sport.
           </p>
           <p>
            Local clubs put on regular
            <a href="http://orienteeringbc.ca/events">
             events
            </a>
            in parks, wilderness and even urban areas. At these events, you will find several orienteering courses designed to challenge anyone from recreational enthusiast to competitive orienteer. If you have a love of the outdoors and enjoy map reading, come give this exciting activity a try. All you need is a compass and a whistle.
           </p>
           <p>
            The Orienteering Association of BC is provincial body of orienteering in BC and is affiliated with Orienteering Canada and the International Orienteering Federation. Local clubs of OABC provide quality events, training session, and programming in numerous cities acorss the province.
           </p>
           <p>
            This website provides
            <a href="http://orienteeringbc.ca/wp/info">
             general information about orienteering
            </a>
            in BC and connects you with orienteering
            <a href="http://orienteeringbc.ca/clubs">
             clubs
            </a>
            and
            <a href="http://orienteeringbc.ca/events">
             events
            </a>
            in your area.
           </p>

           /* Array storing all the events from all our clubs */
 var allEvents = [];
 var maxEvents = 5;
 
 function writeEvents() {
 	var location = document.getElementById("events");
 
 	/* Remove existing nodes */
 	while (location.hasChildNodes()) {  
 	    location.removeChild(location.firstChild);
 	}
 
 	/* Add all the events */
 	for (var i = 0; i < allEvents.length && i < maxEvents;  i++) {
 		var entry = document.createElement("P");
 		var text = document.createTextNode(new Date(parseInt(allEvents[i].start) * 1000).toDateString() + " - ");
 		var link = document.createElement("A");
 		link.href = allEvents[i].url
 		link.innerHTML = allEvents[i].title;
 		entry.appendChild(text);
 		entry.appendChild(link);
 		location.appendChild(entry);
 	}
 }
 
 function addEventsToArray(jsonObject) {
 	/* Add events to the array - only care about jsonObject.title, jsonObject.start, jsonObject.end, jsonObject.color (for stylesheet), jsonObject.url */
 	for (var i = 0; i < jsonObject.length; i++) {
 		allEvents.push(jsonObject[i]);
 	}
 	allEvents.sort(function(a, b){return a.start - b.start});
 	writeEvents();
 }
 
 function getEvents(clubId, daysAdvance) {
 	var xmlhttp = new XMLHttpRequest();
 	xmlhttp.onreadystatechange = function() {
 	    if (this.readyState == 4 && this.status == 200) {
 	        /* Convert the JSON into an array and concat it to the allEvents array */
 			var json = JSON.parse(this.responseText)
 			addEventsToArray(json);
 		}
 	};
 	var startDate = Math.round(Date.now() / 1000);
 	var endDate = Math.round(new Date(Date.now() + 30 * daysAdvance * 60 * 60 * 1000).getTime() / 1000);
 
 	xmlhttp.open("GET", "https://whyjustrun.ca/events.json?prefix_club_acronym=true&start=" + startDate + "&end=" + endDate + "&club_id=" + clubId, true);
 	xmlhttp.send();
 }
 
 function getAllEvents(daysAdvance) {
 	var clubIds = new Array();
 	clubIds.push("1"); //GVOC
 	clubIds.push("43"); //VICO
 	clubIds.push("42"); //SAGE
 	clubIds.push("46"); //KOC
 	clubIds.push("80"); //Whistler
 
 	for (var i = 0; i < clubIds.length; i++) {
 		getEvents(clubIds[i], daysAdvance);
 	}
 }
