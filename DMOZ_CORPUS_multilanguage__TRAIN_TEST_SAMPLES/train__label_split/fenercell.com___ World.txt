 			msg=msg+"Adınız \n";
 			
 		if(Trim(frm.txtMusteriSoyisim.value)=="")
 			msg=msg+"Soyadınız \n";
 			
 		if(Trim(frm.txtMusteriTelefon.value)=="")

 			  if (r1.test(txteposta) == false)
 			  {
 				msg=msg+"E-posta adresi hatalı. \n";
