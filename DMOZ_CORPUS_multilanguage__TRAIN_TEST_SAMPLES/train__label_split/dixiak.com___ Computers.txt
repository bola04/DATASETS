   <script>
    var $imgwidth = 0;
 var $imgheight = 0;
 
 var $beewidth = 0;
 var $beeheight = 0;
 
 var $usableheight = 0;
 var $centerpoint = 0;
 var $containerHeight = 0;
 var $containerWidth = 0;
 var $phase = 0;
 var $nextphaseat = NaN;
 var $phaselengths = [500, 500, 250, 1000, 255, 0];
 var $phasesetups = [beesetup, beedepacksetup, beesetup, beedepacksetup, darkensetup, bouncesetup];
 var $phaserun = [empty, depacking, empty, depacking, darken, bouncing];
 
 var $darkenstart = 0;
 
 $(document).ready(function() {
 	$("#bounce").one("load", updateContainer);
 	updateContainer();
 	$(window).resize(function () {
 		updateContainer();
 	});
 	gameloop();
 });
 
 function beesetup() {
 	$("#packcontainer").empty();
 	$("#bee").show();
 	$("#bounce").hide();
 }
 
 function beedepacksetup() {
 	$("#bee").show();
 	$("#bounce").hide();
 	for(i=0; i<128; i++) {
 		$("#packcontainer").append("<div class='dash'></div>");
 	}
 }
 
 function bouncesetup() {
 	$("body").css("background-color", "rgb(0,0,0)");
 	$("#packcontainer").empty();
 	$("#bee").hide();
 	$("#bounce").show();
 }
 
 function darkensetup(timestamp) {
 	$("#packcontainer").empty();
 	$("#bee").hide();
 	$("#bounce").hide();
 	$darkenstart = timestamp;
 }
 
 function gameloop(timestamp) {
 	requestAnimationFrame(gameloop);
 
 	if ($nextphaseat!=0) {
 		if (!isNaN($nextphaseat) && (timestamp>$nextphaseat)) {
 			$nextphaseat=NaN;
 			$phase++;
 		}
 		if (isNaN($nextphaseat)) {
 			$newoffset = $phaselengths[$phase];
 			$nextphaseat = ($newoffset==0)? 0 : (timestamp + $newoffset);
 			$phasesetups[$phase](timestamp);
 		}
 	}
 	if (!isNaN(timestamp)) {
 		$phaserun[$phase](timestamp);
 	}
 }
 
 function empty(timestamp) {
 }
 
 function darken(timestamp) {
 
 	var $fract = 255 - Math.round((255.0 * (timestamp - $darkenstart))/($nextphaseat - $darkenstart));
 	var $fractext = 255 - Math.round((192.0 * (timestamp - $darkenstart))/($nextphaseat - $darkenstart));
 	$("body").css("background-color", "rgb(" + $fract + "," + $fract + "," + $fract + ")");
 	$("div.links a").css("color", "rgb(" + $fractext + "," + $fractext + "," + $fractext + ")");
 }
 
 function bouncing(timestamp) {
 	var $nowpos = 0;
 	if ($usableheight!=0) {
 		$nowpos = $usableheight * (1.0 - Math.abs(Math.sin(timestamp / 1000)));
 	} else {
 		$nowpos = $centerpoint;
 	}
 	$("#bounce").css("top", $nowpos);
 }
 
 function depacking(timestamp) {
 	$(".dash").each(function() {
 		$(this).css("left", Math.random() * $containerWidth);
 		$(this).css("top", Math.random() * $containerHeight);
 	});
 }
 
 function updateContainer() {
 	$containerHeight = $(window).outerHeight(true);
 	$containerWidth = $(window).outerWidth(true);
 	if ($imgwidth==0) {
 		$imgwidth = $("#bounce").outerWidth(true);
 	}
 	if ($imgheight==0) {
 		$imgheight = $("#bounce").outerHeight(true);
 	}
 
 	if ($beewidth==0) {
 		$beewidth = $("#bee").outerWidth(true);
 	}
 	if ($beeheight==0) {
 		$beeheight = $("#bee").outerHeight(true);
 	}
 
     var $headerHeight = $("#header").outerHeight(true);
     var $footerHeight = $("#footer").outerHeight(true);
 
     var $bodyHeight = $containerHeight - ($headerHeight + $footerHeight);
     $("#bounce").css("left", ($(window).outerWidth()-$imgwidth)/2);
     $("#bee").css("left", ($(window).outerWidth()-$beewidth)/2);
     $("#bee").css("top", ($(window).outerHeight()-$beeheight)/2);
     $("#body").outerHeight($bodyHeight);
 
 	$usableheight = $bodyHeight - $imgheight;
 	$centerpoint = $usableheight/2;
 	if ($usableheight<100) {
 		$usableheight = 0;
 	}
 	if ($centerpoint<0) {
 		$centerpoint = 0;
 	}
 }
