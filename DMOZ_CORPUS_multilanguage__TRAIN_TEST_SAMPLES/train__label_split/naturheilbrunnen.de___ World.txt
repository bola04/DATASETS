 function shopwarenkorb(produkt_id,themen_id,status,zahlungsbedingung,anzahl)
 {
 	if(status == "absenden" || status == "payment")
 	{
 		var agb = document.shopformular.agb.checked;
 		if(agb == false)
 		{
 			alert("Bitte akzeptieren Sie unsere AGB.");
 			document.shopformular.agb.focus();
 		}
 		else
 		{
 			document.shopformular.method = "post";
 			document.shopformular.action = "index.php?site=schwerpunkte&wstatus="+status+"&ppp_id="+produkt_id+"&pppid="+themen_id+"&zart="+zahlungsbedingung;
 			document.shopformular.submit();
 		}
 	}
 	else if(status == "insert")
 	{
 			document.shopformular.method = "post";

                     <h3 align="left" class="text12fettlila">
                      Hypnose für Amberg, Regensburg und Nürnberg, sowie Naturheilverfahren.                                                 „Der Grund der Arznei ist die Liebe“ Paracelsus
                     </h3>
                     <p>
                      Alle biologischen Vorgänge im Körper des Menschen, verbunden mit seiner Umwelt, werden durch übergeordnete Regulationsmechanismen gesteuert. ( z.B. Nervensystem, Akupunkturmeridiane, Hormonsystem, Erbinformation, Blutkreislauf, Immunsystem usw.)
                      <br/>
                      <br/>
                      Doch unsere heutige Lebensweise kann diese natürliche Regulation empfindlich stören - durch inneren und äußeren Streß, Überforderung, denaturierte Nahrung, Elektrosmog, Umweltgifte usw. Dieser Dauerstreß löst letztlich Fehlfunktionen im Gesamtorganismus aus. Die Folge: Ein breites Spektrum an Befindlichkeitsstörungen, Erschöpfung, Allergien, chronische Krankheiten und Schmerzen. In einer solchen Situation, hilft es wenig, einzelne Symptome oder Organe zu behandeln. Dauerhafte Heilung ist nur durch systematische Entgiftung des Körpers, Entstressung und Reaktivierung der Regulationsmechanismen möglich. Genau da setzt ganzheitliche Diagnose,Therapie und Auch Hypnose an.
                     </p>
                     <p>
                      Menschen in Amberg und im Großraum Regensburg, Nürnberg zu helfen ist meine Motivation. Viele nehmen aber auch gerne einen längeren Weg in Kauf.
                     </p>
