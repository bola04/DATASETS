           function getScrollXY(){
             var scrOfX = 0, scrOfY = 0;
             if( typeof( window.pageYOffset ) == 'number' ) {
                 //Netscape compliant
                 scrOfY = window.pageYOffset;
                 scrOfX = window.pageXOffset;
             } else if( document.body && ( document.body.scrollLeft || document.body.scrollTop ) ) {
                 //DOM compliant
                 scrOfY = document.body.scrollTop;
                 scrOfX = document.body.scrollLeft;
             } else if( document.documentElement && ( document.documentElement.scrollLeft || document.documentElement.scrollTop ) ) {
                 //IE6 standards compliant mode
                 scrOfY = document.documentElement.scrollTop;
                 scrOfX = document.documentElement.scrollLeft;
             }
             return [ scrOfX, scrOfY ];
           }
           
           function scrollCheck(){
             var aScroll = getScrollXY();
             var iScrollY = aScroll[1]; 
             var iLandingPageHeight = $("#landingPageWrapper").height();
             if(iScrollY >  iLandingPageHeight){
               clearScrollTimeouts();
             }
             
             var bIsStart = true;
             setBodyClass(aScroll[1],iLandingPageHeight, bIsStart);
           }
           
           function setBodyClass(iCurrentOffset,iLandingPageHeight, bIsStart){
               var iFixheight = 25;
               if(bIsStart){
                 iFixheight += iLandingPageHeight;
               } 
 
               var sClass = "";
               if(iCurrentOffset >= iFixheight){
                   sClass = "fixed";
               }
 
               document.getElementById("bodyId").setAttribute("class", sClass);
           }
 
           function scrollDown(){
             clearScrollTimeouts();
             scrollToElement("header",500);
           }
           
           function pulseScrollDownLink(){
               document.getElementById("landingPageCenter").setAttribute("class", "animated bounce");
               setTimeout(function(){document.getElementById("landingPageCenter").setAttribute("class", "");}, 1000);
               // REPEAT
               oTimeoutPulse = setTimeout(pulseScrollDownLink, iTimeoutPulse);
           }
           
           function clearScrollTimeouts(){
               clearTimeout(oTimeoutAutoscrollDown);
               clearTimeout(oTimeoutPulse);
           }
