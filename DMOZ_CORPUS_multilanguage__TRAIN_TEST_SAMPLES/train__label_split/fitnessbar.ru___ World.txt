 		var overlay = null;		
 		
 		function showGmapTooltip(event, gmap_object, marker_index){
 			if(!overlay){
 				overlay = new AddressOverlay(gmap_object.map, marker_index);
 			}else{
 				overlay.hideAll();
 				overlay.draw(marker_index);
 			}
 				
 		}
 		
 		function hideGmapTooltip(event, gmap_object, marker_index){
 			overlay.hide(marker_index);
 		}
 		
 		function AddressOverlay(map, marker_index){
 			this.setMap(map);
 			this.div_ = new Object();
 			this.marker_index = marker_index;
 		}
 		
 		AddressOverlay.prototype = new google.maps.OverlayView();
 	
 		AddressOverlay.prototype.onAdd = function(){
 			this.draw(this.marker_index);
 		}
 		
 		AddressOverlay.prototype.draw = function(marker_index){
 			
 			var center_default_pos = this.getProjection().fromLatLngToDivPixel(new google.maps.LatLng(59.9390373, 30.3157845));
 			
 			if(gmap.markersArray[marker_index]){
 				var pos = this.getProjection().fromLatLngToDivPixel(gmap.markersArray[marker_index]['marker'].getPosition());
 				
 				var d = document.createElement('div');
 				d.className = 'gmap-address-popup';
 				d.innerHTML = shop_address['m' + marker_index];
 				d.style.left = (pos.x + 20) + 'px';
 				//d.style.top = pos.y-135 + 'px';
 				d.setAttribute('id', 'overlay');
 				var panes = this.getPanes();
 				panes.overlayImage.appendChild(d);
 				
 				$(d).find('.icon-close-rnd').on('click', function(){
 					overlay.hide(marker_index);
 				});
 				
 				var h = $(d).outerHeight();
 				d.style.top = pos.y-h-45 + 'px';
 					
 				this.div_['m' + marker_index] = d;
 				
 				var center_pos = this.getProjection().fromLatLngToDivPixel(gmap.map.getCenter());
 				var panel_pos = $(panes.overlayLayer).offset();
 				var d_pos = $(d).offset();
 				
 				var real_x = d_pos.left - panel_pos.left + center_default_pos.x - center_pos.x;
 				var real_y = d_pos.top - panel_pos.top + center_default_pos.y - center_pos.y;
 				
 				//console.log(real_x);
 				
 				var cnt_w = $(this.div_['m' + marker_index]).width() + 20;
 				
 				var x_changed = -1;
 				if(real_x + cnt_w > $('body').width()){
 					x_changed = center_pos.x + (real_x+cnt_w-$('body').width())
 					gmap.map.panTo(this.getProjection().fromDivPixelToLatLng({'x': x_changed, 'y':center_pos.y}));
 				}
 				if(real_y < 0){
 					gmap.map.panTo(this.getProjection().fromDivPixelToLatLng({'x': x_changed > -1 ? x_changed : center_pos.x, 'y': center_pos.y - (0 - real_y + 20)}));
 				}
 					//panes.overlayLayer.appendChild(d);
 				//console.log(real_x + ',' + real_y);
 				
 			}
 		}
 		
 		AddressOverlay.prototype.hide = function(marker_index){
 			if(this.div_['m' + marker_index])
 				this.div_['m' + marker_index].style.display = 'none';
 		}
 		
 		AddressOverlay.prototype.hideAll = function(){
 			var ln = Object.keys(shop_address).length;
 			for(var i = 0; i < ln; i++){				
 				if(this.div_['m' + i])
 					this.div_['m' + i].style.display = 'none';
 			}
 		}
