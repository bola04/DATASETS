                             var form = $("#s_trailers");
                             $("#s_submit").click(function(e){
                                 e.preventDefault();
                                 var url = "";
                                 if($("#s_condition").val() == "New"){
                                     url = "/new-trailers/?f=true&models=" + $("#s_models").val() + "";
                                 }else{
                                     url = "/used-trailers/?f=true&models=" + $("#s_models").val() + "";
                                 }

                             });
 
                             var curCondition = $("#s_condition").val();
                             buildMakeSelect(curCondition);
                             buildModelSelect($("#s_makes").val());
 
                             $("#s_condition").change(function(e){
                                 curCondition = $("#s_condition").val();
                                 buildMakeSelect(curCondition);
                                 buildModelSelect($("#s_makes").val());
                             });
 
                             $("#s_makes").change(function(e){
                                 buildModelSelect($("#s_makes").val());
                             });
 
                             function buildModelSelect(c){
                                 $("#s_models").empty();
                                 var curMakes = categories[$("#s_condition").val()].makes[c];
                                 for(var i in curMakes){
                                     buildSelects($("#s_models"), curMakes[i]);
                                 }
                             };
 
                             function buildMakeSelect(c){
                                 var curMakes = "";
                                 $("#s_makes").empty();
                                 $("#s_models").empty();
                                 if(c == "New"){
                                     curMakes = returnProperties($("#s_makes"),categories["New"]);
                                 }else if(c == "Used"){
                                     curMakes = returnProperties($("#s_makes"),categories["Used"]);
                                 }
                             }
 
                             function buildSelects (elem, v){
                                 var option = $("<option value='" + v + "'>" + v + "</option>");
                                 option.appendTo(elem);
                             }
 
                             function returnProperties(e,a){
                                 for (var key in a){
                                     // skip loop if the property is from prototype
                                     if (!a.hasOwnProperty(key)) continue;
                                     var obj = a[key];
                                     for (var prop in obj) {
                                         // skip loop if the property is from prototype
                                         if(!obj.hasOwnProperty(prop)) continue;
                                         buildSelects(e,prop);
                                     }
                                 }
                             }
