    //set top padding of the workspace to the height of the ribbon
 		function setTopPadding() {

 		     //padding works better for the main window
 		     wrkElem.style.paddingTop = ribHeight + 'px';
 		   }
 		}
 		
 		// bind top padding reset to ribbon resize event so that the page always lays out correctly.
 		ExecuteOrDelayUntilScriptLoaded(function () { SP.UI.Workspace.add_resized(setTopPadding); }, "init.js");
 
         /*EMERGENCY ALERTS *******************************/
 		$(document).ready(function(){ // this waits until the DOM is fully loaded and executes before the page contents are loaded
 			var today = new Date().toISOString(); // example: 2016-01-25T21:35:12.722Z
 			var restURL = "/emergency/_api/web/lists/getByTitle('Alert')/items?$filter=(Start%20le%20datetime%27" + today + "%27)%20and%20(End%20ge%20datetime%27" + today + "%27)&$select=Heading,Description,Link,Sites";
 			$.ajax({ // query the emergency alerts list and return results of REST query
 				url: restURL,
 				type: "GET",
 				headers: { "ACCEPT": "application/json;odata=verbose" },
 				success: showAlert, 
 				error: function( req, status, err ) { console.log( 'something went wrong', status, err ); }
 			});
 		
 			function showAlert(jsondata) {
 		
 				var s = jsondata.d.results;
 				var alerts = [""];
 				var alertHTML = "";
 				var writeAlert = false; // this is set to true if the alert should be written to this page
 				var pageDomain = "https://www.tacomaschools.org/";
 				var pagePath = $(location).attr("pathname");
 				var pagePathParts = pagePath.split('/');
 				var pageURL = pageDomain + pagePathParts[1];
 				var siteURL = "";
 		
 				if (s){ // check to see if there are any results
 					for ( var key=0; key<s.length; key++ ){ // Loop through all results
 						writeAlert = false;
 						alerts[key] = {
 							heading: 	 s[key].Heading,
 							description: s[key].Description,
 							link: 		 s[key].Link,
 							sites: 		 s[key].Sites.results
 						};
 						if ( alerts[key].sites[0] == "ALL" ){ // if first thing in the Sites column for this resutl is "ALL", skip to building the alert's HTML
 							writeAlert = true; // if "ALL" is checked the alert should appear on every public-facing webpage (master pages tacoma-www.aspx, bootstrap.aspx and bootstrap-alt.aspx)
 						}
 						else{
 							pagePathParts[1] = pagePathParts[1].toLowerCase();
 							if (pagePathParts[1] === "pages" || pagePathParts[1] === "schools" || pagePathParts[1] === "parents" || pagePathParts[1] === "students" || pagePathParts[1] === "emergency" ){ 
 							// Checks if this is a "top-level" page on district site. All alerts get written to top-level district pages through this line or the "ALL" line above
 								writeAlert = true;
 							}
 							else{
 								for( var k=0; k<alerts[key].sites.length; k++ ){ // Loop through each item in the Sites column
 									siteURL = alerts[key].sites[k].slice( alerts[key].sites[k].indexOf(":")+2 , alerts[key].sites[k].length );
 									// Values look like "Blix: https://www.tacomaschools.org/blix" The slice() function takes the start and end position to extract just the URL
 									if (pageURL === siteURL){
 										writeAlert = true;
 									}
 								}						
 							}
 						}
 		
 						if (writeAlert){
 							alertHTML = alertHTML +
 										'<div class="container-fluid tps-alert alert">' +
 											'<a href="#" class="close" data-dismiss="alert" aria-label="close">×</a>' +
 										 	'<h1><a href="'+ alerts[key].link + '"><span class="container"><i class="fa fa-2x fa-chevron-circle-right"></i>' + 
 										 		alerts[key].heading + '<small>' + alerts[key].description + ' <span>Learn more »</span></small></span></a></h1>' + 
 										'</div>'; 
 						}
 					}
 				}
 		
 		
 				if ($("#tps-universal-nav")){ 	// #tps-universal-nav is on the main district website
 					$("#tps-universal-nav").after(alertHTML); 
 				}
 				if ($("#tps-header")){ 			// #tps-header is only on school websites
 					$("#tps-header").before(alertHTML); 
 				}
 		
 			}
 		});
   </script>
