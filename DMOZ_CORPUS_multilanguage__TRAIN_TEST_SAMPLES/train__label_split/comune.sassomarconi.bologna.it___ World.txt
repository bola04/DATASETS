 function Gl_overlay(status, maxHeight, mouseX, mouseY) 
 {
 
     var boxHeight = 120; // altezza box
     var boxWidth = 220; // larghezza box
     var verticalGapTop = 15; // intervallo in pixels tra y del mouse e box
     var verticalGapBottom = 35; // intervallo in pixels tra y del mouse e box
     var scrollAmount; // numero di pixel di cui Ã¨ scrollata la pagina verticalmente
     var topLimit; //limite superiore sotto al quale occorre metter il box sotto anzichÃ¨ sopra
 
     ov = document.getElementById("dvGl_Overlay");
     if(maxHeight)
     {
         p = getPageHeight();
         ov.style.height = p + 'px';
     }
     else
     {
         ov.style.height = '100%';
     }
 
     if(!document.all)
     {
         scrollAmount = -f_scrollTop(); //solo ie vuole lo scroll
         topLimit = f_scrollTop();
     }
     else
     {
         scrollAmount = 0;
         topLimit = f_scrollTop();
     }	
 
     if(status == true)
     {
         b = document.getElementById("box");
         b.style.top = mouseY - boxHeight - verticalGapBottom + scrollAmount + 'px';
         //se esce di sopra lo metto sotto
         if ((mouseY - boxHeight - verticalGapBottom) < topLimit)
         {
             b.style.top = mouseY + verticalGapTop + scrollAmount + 'px';
         }
         b.style.left = mouseX - (boxWidth/2) + 'px';
         if((mouseX - (boxWidth/2)) < 0) b.style.left = '0px';
         if((mouseX - (boxWidth/2) + boxWidth) > f_clientWidth()) b.style.left = f_clientWidth() - boxWidth + 'px';
         b.style.height = boxHeight + 'px';
         b.style.width = boxWidth + 'px';
         //document.getElementById('parola-cercata').innerHTML = 'mouseX: ' + mouseX + '<br />mouseY: ' + mouseY + '<br />top: ' + b.style.top + '<br />left: ' + b.style.left + '<br />scrollAmount: ' + scrollAmount;
     }
     ov.style.display = (status == false) ? 'none' : 'block';
 
     bv = document.getElementById("box-wrapper");
     bv.style.display = (status == false) ? 'none' : 'block';
 
     //bx = document.getElementById("box");
     //bx.style.display = (status == false) ? 'none' : 'block';
     //bx.style.zindex = (status == false) ? '-100' : '112';
 
 
     if (navigator.appVersion.indexOf("MSIE")!=-1 && navigator.appVersion.indexOf("6.0")!=-1) 
     {
         selects = document.getElementsByTagName("select");
         for (i = 0; i != selects.length; i++) 
         {
             selects[i].style.visibility = (status == false) ? 'visible' : 'hidden';
         }
     }    
 }
 
 
 	function f_scrollTop() {
 		return f_filterResults (
 			window.pageYOffset ? window.pageYOffset : 0,
 			document.documentElement ? document.documentElement.scrollTop : 0,
 			document.body ? document.body.scrollTop : 0
 		);
 	}
 	function f_filterResults(n_win, n_docel, n_body) {
 		var n_result = n_win ? n_win : 0;
 		if (n_docel && (!n_result || (n_result > n_docel)))
 			n_result = n_docel;
 		return n_body && (!n_result || (n_result > n_body)) ? n_body : n_result;
 	}
 
 	function f_clientWidth() {
 		return f_filterResults (
 			window.innerWidth ? window.innerWidth : 0,
 			document.documentElement ? document.documentElement.clientWidth : 0,
 			document.body ? document.body.clientWidth : 0
 		);
 	}

 			window.innerHeight ? window.innerHeight : 0,
 			document.documentElement ? document.documentElement.clientHeight : 0,
 			document.body ? document.body.clientHeight : 0
 		);
 	}

 		return f_filterResults (
 			window.pageXOffset ? window.pageXOffset : 0,
 			document.documentElement ? document.documentElement.scrollLeft : 0,
 			document.body ? document.body.scrollLeft : 0
 		);
 	}
 
 
 
     function yScroll()
     {
         if (window.innerHeight && window.scrollMaxY) 
         { 
             return window.innerHeight + window.scrollMaxY;
         } 
         else if (document.body.scrollHeight > document.body.offsetHeight)
         { // all but Explorer Mac
             return document.body.scrollHeight;
         } 
         else 
         { // Explorer Mac...would also work in Explorer 6 Strict, Mozilla and Safari
             return document.body.offsetHeight;
         }			
     }
 
     function getPageHeight()
     {
         var windowHeight;
         if (self.innerHeight) 
         { // all except Explorer
             windowHeight = self.innerHeight;
         } 
         else if (document.documentElement && document.documentElement.clientHeight) 
         { // Explorer 6 Strict Mode
             windowHeight = document.documentElement.clientHeight;
         } 
         else if (document.body) 
         { // other Explorers
             windowHeight = document.body.clientHeight;
         }     
         // for small pages with total height less then height of the viewport
         if(yScroll() < windowHeight)
         {
             pageHeight = windowHeight;
         } 
         else 
         { 
             pageHeight = yScroll();
         }
 
         return pageHeight;
     }
