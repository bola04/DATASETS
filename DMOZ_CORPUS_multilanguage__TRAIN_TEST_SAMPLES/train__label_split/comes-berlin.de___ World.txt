 			function getWindowHeight() {
 			  var windowHeight=0;
 			  if (typeof(window.innerHeight)=='number') {
 				windowHeight=window.innerHeight;
 			  }
 			  else {
 				if (document.documentElement&&
 				  document.documentElement.clientHeight) {
 					windowHeight=
 					  document.documentElement.clientHeight;
 				}
 				else {
 				  if (document.body&&document.body.clientHeight) {
 					windowHeight=document.body.clientHeight;
 				  }
 				}
 			  }
 			  return windowHeight;
 			}		
 			function setFooter() {
 			  if (document.getElementById) {
 				var windowHeight=getWindowHeight();
 				if (windowHeight>0) {
 				  var contentHeight;
 				  if(window.location.pathname == "/startseite/" || window.location.pathname == "/"){
 					contentHeight=document.getElementById('frontpage_content_row').offsetHeight;
 				  } else {
 					if(window.location.pathname == "/aktuelles/"){
 					  contentHeight=document.getElementById('blog_content_row').offsetHeight;
 					} else {
 					  contentHeight=document.getElementById('page_content_row').offsetHeight;
 					}
 				  }
 				  var footerElement=document.getElementById('footer_row');
 				  var footerHeight=footerElement.offsetHeight;
 				  if (windowHeight-(contentHeight+footerHeight)>=0) {
 					footerElement.style.position='relative';
 					footerElement.style.top=(windowHeight-(contentHeight+footerHeight))+'px';
 				  }
 				  else {
 					footerElement.style.position='static';
 				  }
 				}
 			  }
 			}

       <h1>
        Herzlich Willkommen bei COMES
       </h1>
       <div>
        <p>
         Wir sind ein Träger der freien Wohlfahrtspflege.
        </p>
        <p>
         Seit über 25 Jahren unterstützen wir Menschen mit Lernschwierigkeiten oder seelischen Behinderungen durch verschiedene Formen des Betreuten Wohnens in ihrer Teilhabe an der Gemeinschaft. Darüber hinaus sind wir ein Träger der ambulanten Jugendhilfe. (
         <a href="https://www.comes-berlin.de/wir-ueber-uns/" title="Wir über uns">
          Wir über uns
         </a>
         )
        </p>
        <p>
         Gemeinnützigkeit der Ziele sowie Professionalität des Vorgehens sind die miteinander verbundenen Grundlagen unserer täglichen Arbeit. (
         <a href="https://www.comes-berlin.de/qualitaet/" title="Qualität">
          Qualität
         </a>
         )
        </p>
        <p>
         Auf unserer Website können Sie sich über unsere Arbeitsweise sowie die verschiedenen Unterstützungsformen informieren. Sprechen Sie uns an. Wir machen Ihnen gerne einen Vorschlag, ab wann und wie wir Sie unterstützen können. (
         <a href="https://www.comes-berlin.de/leistungen-2/" title="Leistungen">
          Leistungen
         </a>
         )
        </p>
        <p>
        </p>
        <p>
         Wenn Sie an einer Tätigkeit bei COMES interessiert sind,  finden Sie im Bereich Arbeit eine kurze Einführung. (
         <a href="https://www.comes-berlin.de/stellenausschreibungen/" title="Stellenausschreibungen">
          Stellenausschreibungen
         </a>
         )
        </p>
        <p>
         Alles was uns derzeit besonders wichtig ist, haben wir für Sie in kurzen Beiträgen zusammengefasst. (
         <a href="https://www.comes-berlin.de/aktuelles/" title="Aktuelles">
          Aktuelles
         </a>
         )
        </p>
        <p>
         Bürgerinnen und Bürger, die unsere Arbeit durch eine Mitgliedschaft oder durch Spenden unterstützen wollen, sind herzlich willkommen. (
         <a href="https://www.comes-berlin.de/engagement/" title="Engagement">
          Engagement
         </a>
         )
        </p>
        <p>
         Für weitergehende Fragen sowie individuelle Auskünfte nutzen Sie bitte das Kontaktformular oder wenden sich an unsere Geschäftsstelle. (
         <a href="https://www.comes-berlin.de/kontakte/" title="Kontakte">
          Kontakt
         </a>
         )
        </p>
        <p>
         Und schließlich möchten wir Ihnen unsere
         <em>
          Bilder individueller Lebensmomente von Menschen mit Lernschwierigkeiten
         </em>
         empfehlen. (
         <a href="http://www.comes.berlin" rel="noopener" target="_blank" title="Fotoausstellung">
          Ausstellung
         </a>
         )
        </p>
       </div>
