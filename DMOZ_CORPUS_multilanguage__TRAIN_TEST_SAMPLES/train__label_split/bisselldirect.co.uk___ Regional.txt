                                     eventTime = parseInt(eventTime.getTime()/1000);
                                     currentTime = parseInt(currentTime.getTime()/1000);
 
                                     var leftTime = eventTime - currentTime;//Now i am passing the left time from controller itself which handles timezone stuff (UTC), just to simply question i used harcoded values.
                                     var duration = moment.duration(leftTime, 'seconds');
                                     var interval = 1000;
 
                                     if (duration.asSeconds() <= 0) {
                                         jQuery('#cd_d').text('00');
                                         jQuery('#cd_h').text('00');
                                         jQuery('#cd_m').text('00');
                                         jQuery('#cd_s').text('00');
                                     } else {
 
                                         var intervalId = setInterval(function(){
                                             // Time Out check
                                             if (duration.asSeconds() <= 0) {
                                                 clearInterval(intervalId);
                                                 days = hours = mins = secs = '00';
                                             } else {
                                                 duration = moment.duration(duration.asSeconds() - 1, 'seconds');
                                                 days = pad(duration.days(),2);
                                                 hours = pad(duration.hours(),2);
                                                 mins = pad(duration.minutes(),2);
                                                 secs = pad(duration.seconds(),2);

                                             jQuery('#cd_d').text(days);
                                             jQuery('#cd_h').text(hours);
                                             jQuery('#cd_m').text(mins);
