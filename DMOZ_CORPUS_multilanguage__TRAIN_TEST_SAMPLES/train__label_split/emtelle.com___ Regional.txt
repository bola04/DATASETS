             d.setTime(d.getTime() + (exdays*24*60*60*1000));
             var expires = "expires="+ d.toUTCString();
             document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
         }
 
         function getCookie(cname) {
             var name = cname + "=";
             var decodedCookie = decodeURIComponent(document.cookie);
             var ca = decodedCookie.split(';');
             for(var i = 0; i <ca.length; i++) {
                 var c = ca[i];
                 while (c.charAt(0) == ' ') {
                     c = c.substring(1);
                 }
                 if (c.indexOf(name) == 0) {
                     return c.substring(name.length, c.length);
                 }
             }
             return "";
         }
 
         jQuery(document).ready(function () {
             console.log('jQuery ready');
             if(getCookie('banner') != 'accepted') {
                 jQuery('#cookie-policy').show();
             }
 
             jQuery('#cookie-policy button').click(function(){
                 setCookie('banner', 'accepted', 100);
                 jQuery('#cookie-policy').hide();
             });
         })
    </script>

         <div class="article-content aboutus-article">
          <p>
           Wenn es um Erfahrung, Produktionsvolumen und Innovation geht, ist die Emtelle Gruppe Branchenführer in Kunststoffschlauch für Telekommunikation und Wasserleitungen. Als erster Großproduzent in der Welt der geblasenen Faserrohre ist Emtelle’s Know-how konkurrenzlos. Mit über 30 Jahren Geschichte als Lieferant hat die hohe Qualität unseres Produktes geformt und uns geholfen, das umfassendste und proaktive Kundenserviceprogramm zu entwickeln. Insgesamt hat die Emtelle Gruppe rund 1 Million Kilometer geblasener Faserrohr-Infrastruktur in Großbritannien und 2 Millionen Kilometern Nutzrohr für Energieunternehmen in Skandinavien geliefert.
          </p>
          <h3>
           Emtelle Group Erfahrung
          </h3>
          <p>
           Die Großprojekte der Emtelle Group umfassen ITS-Projekte in Großbritannien und Australien, hohe Aufstiege in Dubai, Häfen in den USA, den Gotthard-Tunnel in der Schweiz, das Olympiastadion in London und die Kuala Lumpur Monorail in Malaysia.
          </p>

    equalheight = function(container){

 			$el = jQuery(this); 
 			jQuery($el).height('auto') 
 			topPostion = $el.position().top; 
 
 			if (currentRowStart != topPostion) { 
 				for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) { 
 					rowDivs[currentDiv].height(currentTallest); 
 				}
 				rowDivs.length = 0; // empty the array 
 				currentRowStart = topPostion; 
 				currentTallest = $el.height(); 
 				rowDivs.push($el); 
 			} else { 
 				rowDivs.push($el); 
 				currentTallest = (currentTallest < $el.height()) ? ($el.height()) : (currentTallest); 
 			} 
 			for (currentDiv = 0 ; currentDiv < rowDivs.length ; currentDiv++) { 
 				rowDivs[currentDiv].height(currentTallest); 
 			} 
 		}); 
 	}
