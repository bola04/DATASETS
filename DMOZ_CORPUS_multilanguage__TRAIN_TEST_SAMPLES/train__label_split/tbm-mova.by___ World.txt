        function loginOpen(pr){
 			if($('#login_dv').css('display') != "none"){
 				if((pr==1 && $('#regfrm').css('display')!='none') || (!pr && $('#loginfrm').css('display')!='none')){
 					$('#login_dv').hide();
 					return;
 				}
 			}
 			$('#loginfrm').hide();
 			$('#regfrm').hide();
 			$('#forgotfrm').hide();
 			$('#login_dv').show();
 			if(pr == 1){
 				$('#regfrm').show();
 			}else if(pr == 2){
 				$('#forgotfrm').show();
 			}else{
 				$('#loginfrm').show();
 			}
         }
 
 		function doForgot(){
 			$('#forgot_err').hide();
 			$('#forgot_ok').hide();
 			if($('#forgot_email').val()=='email'){
 				$('#forgot_email').focus();
 				return;
 			}
 			$.post('?action=forgot', {email:$('#forgot_email').val()}, function (ret){
 				if(ret == 'ok'){
 					$('#forgot_ok').show();
 					setTimeout(function(){$('#forgot_ok').hide();}, 5000);
 				}else{
 					$('#forgot_err').show();
 				}
 			})
 		}
         
 		function doLogin(){
 			$('#login_err1').hide();
 			$('#login_err0').hide();
 			if(!$('#login_pass').val()){
 				$('#login_pass').focus();
 				return;
 			}
 			if($('#login_login').val()=='Email'){
 				$('#login_login').focus();
 				return;
 			}
 			$.post('?action=login', {email:$('#login_login').val(), pass: $('#login_pass').val()}, function (ret){
 				var ret_arr = ret.split('|');
 
 				if(ret_arr[0] == 'ok'){
 					$('.h_enter').html(ret_arr[1]);
 					$('#comment_not_logged').hide();
 					$('#comment_logged').show();
                     $('#codearea').hide();
 				}else if(ret_arr[0] == 'err1'){
 					$('#login_err1').show();
 				}else{
 					$('#login_err0').show();
 				}
 				
 			})
 		}
 		function doReg(){
 			$('#reg2_err').hide();
 			$('#reg_err').hide();
 			if($('#reg_email').val()=='email'){
 				$('#reg_email').focus();
 				return;
 			}
 			if($('#reg_name').val()=='имя'){
 				$('#reg_name').focus();
 				return;
 			}
 			$.post('?action=reg', {email:$('#reg_email').val(), name: $('#reg_name').val()}, function (ret){
 				var ret_arr = ret.split('|');
 				if(ret_arr[0] == 'ok'){
 					$('.h_enter').html(ret_arr[1]);
 					$('#comment_not_logged').hide();
 					$('#comment_logged').show();
                     $('#codearea').hide();
 				}else if(ret=='err1'){
 					$('#reg2_err').show();
 				}else{
 					$('#reg_err').show();
 				}
 			})
 		}
