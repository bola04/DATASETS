     $(document).ready(function(){
 
 	$('select#f_search_price').attr("disabled", true); 
 
 	$('select#f_tipo').on('change',function(){
 		
 		var valor = $(this).val();
 		//var valor = $('input:radio[name=f_operacion]:checked').val();
 		
 		//alert(valor);
 	    if(valor == 'local'){
 
 
 	    	$("#f_comodidades").val('');	    	
 	    	$('#f_comodidades').hide();
 	    	
 	    	
 		}else{
 
 			$('#f_comodidades').show();
 			
 		}
 	    
 	});
 
 	
 	$('input:radio[name=f_operacion]').on('change',function(){
 
 		var valor = $(this).val();
 		//var valor = $('input:radio[name=f_operacion]:checked').val();
 	    if(valor == 'alquiler' || valor == 'con_renta'){
 
 	    	$("#f_search_price").val('');	    	
 	    	$('#f_search_price').hide();
 			$('#f_comodidades').show();	    	
 	    	
 		}else if(valor == 'venta'){
 
 			$('select#f_search_price').attr("disabled", false); 			
 	    	$('#f_search_price').show();
 	    	$("#f_comodidades").val('');
 			$('#f_comodidades').hide();
 			
 		}else{
 
 	    	$("select#f_search_price").val('');    	
 			$('select#f_search_price').attr("disabled", true); 			
 	    	$('#f_search_price').show();		
 			$('#f_comodidades').show();
 
 		}
 	    
 	});
 	
 	/////////////////////////////////////////////////////
 	$("#btn_buscar").click(function(){
 
 		$('.site').fadeOut();
 
 		var categoria = "";
 
 		var operacion = "";
 		
 		if ($("input[name='f_operacion']:checked").length > 0){
 
 	    	operacion = $('input:radio[name=f_operacion]:checked').val();
 		
 		}
     	var tipo = $('#f_tipo').val();
     	var comodidades = $('#f_comodidades').val();	    	
     	var precio = $('#f_search_price').val();
     	var barrio = $('#f_barrio').val();
     	
     	
 
     	//alert("op: "+operacion+" tipo: "+tipo+" com: "+comodidades+ "  precio: "+precio);
 
     	
         $.ajax({
         	data	: 'categoria='+categoria+'&operacion='+operacion+'&tipo='+tipo+'&comodidades='+comodidades+'&precio='+precio+'&barrio='+barrio,
             url		: 'http://www.campiglia.com.uy/proyecto/ajax_get_proyectos',
             type	: 'post',
 	        success	: function(response){
 	            if ( response != 'false' ) {
 	            	$('.site').html(response);
 	            }
 	        }
         });
         $('.site').fadeIn();
 	});
     $('.site').fadeIn();
 
 
 ////////////////////////////////////////
