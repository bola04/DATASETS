             } else {
                 var day_curr = date.getDate();
             };
             
             if ((date.getMonth()+1).toString().length === 1) {

             };
             
             dmy = day_curr + "-" + month_curr + "-" + date.getFullYear();
             if ($.inArray(dmy, availableDates) != -1) {
