 function updateCountdown() {
 
   // Get todays date and time
   var now = new Date().getTime();
 
   // Find the distance between now an the count down date
   var distance = countDownDate - now;
 
   // Time calculations for days, hours, minutes and seconds
   var days = Math.floor(distance / (1000 * 60 * 60 * 24));
   var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

 var slideIndex = 0;
 var autoCount = 0;
 
 function rotate_banner(n) {
     showDivs(slideIndex += n);
 }
 
 function showDivs(n) {
 	autoCount = 0; //reset the autorotation
     var i;
     var x = document.getElementsByClassName("banner_slides");
 	var y = document.getElementsByClassName("banner_text");
 	var z = document.getElementsByClassName("banner_select");
 	slideIndex = n;
     if (n > x.length) {slideIndex = 1}
     if (n < 1) {slideIndex = x.length} ;
     for (i = 0; i < x.length; i++) {
         x[i].style.display = "none";
 		y[i].style.display = "none";
 		z[i].className = z[i].className.replace(" banner_selected", "");
     }
     x[slideIndex-1].style.display = "block";
 	y[slideIndex-1].style.display = "block";	
 	z[slideIndex-1].className += " banner_selected";
 }
 
 showDivs(1);
 auto_rotate();
 
 function auto_rotate() {    
 	autoCount++;
 	if(autoCount > 10) {
 		slideIndex++;	
 		showDivs(slideIndex);	
 	}
