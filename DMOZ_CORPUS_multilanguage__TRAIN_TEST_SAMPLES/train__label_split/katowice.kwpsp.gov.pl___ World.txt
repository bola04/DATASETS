 			$('.dontclose button').click(function (oEvent) {
 				oEvent.preventDefault();
 				//console.log($(this).attr('class'));
 
 				var font_class = $(this).attr('class');
 				var elements = $(".content").find($("p,em,strong,q,table th, table td,a,span, h3, li"));
 				var elements_contact = $(".contact-content").find($("p,em,strong,q,table th, table td,a,span,h3, li"));
 
 				switch (font_class) {
 					case 'standard-font':
 						{
 							if (elements.length > 0) {
 								elements.removeClass('font-big-set');
 								elements.removeClass('font-biggest-set');
 							}
 							if (elements_contact.length > 0) {
 								elements_contact.removeClass('font-big-set');
 								elements_contact.removeClass('font-biggest-set');
 							}
 						}
 						//set_font_size('90%');
 						break;
 					case 'big-font':
 						//set_font_size('113%');
 						if (elements.length > 0) {
 							elements.removeClass('font-biggest-set');
 							elements.addClass('font-big-set');
 						}
 						if (elements_contact.length > 0) {
 							elements_contact.removeClass('font-biggest-set');
 							elements_contact.addClass('font-big-set');
 						}
 						break;
 					case 'biggest-font':
 						//set_font_size('122%');
 						if (elements.length > 0) {
 							elements.removeClass('font-big-set');
 							elements.addClass('font-biggest-set');
 						}
 						if (elements_contact.length > 0) {
 							elements_contact.removeClass('font-big-set');
 							elements_contact.addClass('font-biggest-set');
 						}
 						break;
 					default:
 
 						if (elements.length > 0) {
 							elements.removeClass('font-big-set');
 							elements.removeClass('font-biggest-set');
 						}
 
 						if (elements_contact.length > 0) {
 							elements_contact.removeClass('font-big-set');
 							elements_contact.removeClass('font-biggest-set');
 						}
 				}
 
 			});
