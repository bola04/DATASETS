                 function allImagesLoaded(container, delegate)
                 {
                   var allImages = container.find("img");
                   nimages = allImages.length;
                   if(nimages == 0)
                     delegate.call();
                   else
                     allImages.load(function() {
                       nimages--;
                       if(nimages == 0) {
                         delegate.call();
                       }
                     });
                 }

                   else
                     $('.ajaxPresseLess').css("visibility","visible");
