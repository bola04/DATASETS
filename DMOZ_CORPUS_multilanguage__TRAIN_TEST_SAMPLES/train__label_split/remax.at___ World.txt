 	function checkSearchForm() {
 			if($('#search').is(':visible')) {
 				if($('#search').val() == "") {
 					/*$('#btnsearch').addClass("inactive-button");*/
 					$('#btnsearch').attr("disabled",true);
 					$('#btnsearch').addClass("inactive-button-search");
 					$('#home-error-msg').fadeIn('fast');
 				}
 				else {
 					$('#btnsearch').removeAttr("disabled");
 					$('#btnsearch').removeClass("inactive-button-search");
 					$('#home-error-msg').fadeOut('fast');
 				}
 			}
 
 			if($('#select-search').is(':visible')) {
 				if($('#select-search').val() != "" && $('#objekttyp').val() == 10) {
 					$('#btnsearch').removeAttr("disabled");
 					$('#btnsearch').removeClass("inactive-button-search");
 					$('#home-error-msg').fadeOut('fast');
 				}
 				else {
 					$('#btnsearch').attr("disabled",true);
 					$('#btnsearch').addClass("inactive-button-search");
 					$('#home-error-msg').fadeOut('fast');
 				}
 			}
 	}
