   <script>
    $(function() {
             $('#prev').click(function(e) {
                 e.preventDefault();
 
                 fetchNewGame('prev');
             });
 
             $('#next').click(function(e) {
                 e.preventDefault();
 
                 fetchNewGame('next');
             });
 
             var fetchNewGame = function(direction) {
                 var dateField = $('#hidden-date').text();
                 var locale = "nl";
 
                 $.ajax({
                     url: 'api/games/' + dateField + '/' + direction + '?locale=' + locale,
                     success: function(result) {
                         fillNewGame(result);
                     }
                 })
             };
 
             function fillNewGame(game) {
 
                 //fill data
                 $('#hidden-date').html(game.hidden_date);
 
                 $('#home-team-title').html(game.home_team.title);
                 $('#away-team-title').html(game.away_team.title);
 
                 $('#home-team-logo').css('background-image', 'url(' + game.home_team_logo + ')');
                 $('#away-team-logo').css('background-image', 'url(' + game.away_team_logo + ')');
 
                 var localeAtWord = 'om'
 
                 $('#game-date').html(game.date + " " + localeAtWord + " " + game.hour);
                 $('#matchday').html(game.matchday);
                 $('#league-title').html(game.league);
 
                 $('.match-scoreboard').hide();
 
                 if (game.next_game == false) {
                     $('#next').css({
                         visibility: 'hidden'
                     });
                 } else {
                     $('#next').css({
                         visibility: ''
                     });
                 }
 
                 if (game.previous_game == false) {
                     $('#prev').css({
                         visibility: 'hidden'
                     });
                 } else {
                     $('#prev').css({
                         visibility: ''
                     });
                 }
 
                 if (game.is_played) {
                     $('#score-home-team').html(game.home_score);
                     $('#score-away-team').html(game.away_score);
                     $('#match-not-future').show();
                 }
                 else {
                     $('#match-future').show();
                 }
 
                 if (! game.is_live) {
                     $('.timeboard').hide();
                 } else {
                     $('.timeboard').show();
                     $('.timeboard').html(game.minute + '\'');
                 }
 
                 //fill buttons
                 $('#game-buttons').empty();
                 $('.game-buttons-wrap').removeClass("multiple-mobile");
                 $('#game-buttons').removeClass("small split-mobile");
                 if (Object.keys(game.links).length > 3) {
                     $('.game-buttons-wrap').addClass("multiple-mobile");
                     $('#game-buttons').addClass("small split-mobile");
                 }
                 var translation_almost_sold_out = "Beperkt aantal vrije plaatsen";
                 for (var link in game.links) {
                     if (link == 'Uitverkocht' || link == translation_almost_sold_out) {
                         link_class = 'negative';
                         if  (game.links[link] != 'Uitverkocht' ) {
                             link_class = 'warning'; // Almost sold out
                         }
                         var button = '<li><a class="button small tt ' + link_class + '">' + link + '</a></li>';
                         if (game.free_your_seat == true) {
                             button = button + '<li><a target="_blank" href="https://tickets.tickethour.be/l0007ww/accountLogin.html" class="button small tt positive">Free your seat</a></li>';
                         }
                     } else {
                         var button = '<li><a href="' + game.links[link] + '" class="button small tt" id="button-' + link + '">' + link + '</a></li>';
                     }
                     $('#game-buttons').append(button);
                 }
             }
 
         });
   </script>
