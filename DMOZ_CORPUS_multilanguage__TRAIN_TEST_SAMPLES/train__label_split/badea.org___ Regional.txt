            function GetExchageRate(Amount, OldCurr, NewCurr) {
         //alert("none : " + document.getElementById("dvLoadingImg").style.display);
         document.getElementById("dvLoadingImg").style.display = "block";
         //alert("block : " + document.getElementById("dvLoadingImg").style.display);
         var url = "/CurrencyConvert.aspx" + "?OldCurr=" + OldCurr + "&NewCurr=" + NewCurr + "&Amount=" + Amount + "&curr=1";
         xmlHttp = GetXmlHttpObject();
         xmlHttp.onreadystatechange = responseExchangeRate
         xmlHttp.open("GET", url, true)
         xmlHttp.send("1")
         return false;
     }
 
     function responseExchangeRate() {
         if (xmlHttp.readyState == 4) {
             var CurrencyRate = xmlHttp.responseText;
             document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_OneUnitPrice').value = CurrencyRate;
             SetLables();
             SetCurrency();
            document.getElementById("dvLoadingImg").style.display = "none";
         }
     }
 
     function GetXmlHttpObject() {
         var objXMLHttp = null
         if (window.XMLHttpRequest) {
             objXMLHttp = new XMLHttpRequest()
         }
         else if (window.ActiveXObject) {
             objXMLHttp = new ActiveXObject("Microsoft.XMLHTTP")
         }
         return objXMLHttp
     }
 
     function ChangeCurrency() {
         if (document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_txtAmount').value == '') return false;
         var sourceCurrency = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlSource').value;
         var targetCurrency = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlTarget').value;
         GetExchageRate(1, sourceCurrency, targetCurrency);
         return false;
     }
 
     function SetLables() {
         var dateObjct = new Date();
         var TodayDate = dateObjct.getFullYear() + "-" + dateObjct.getMonth() + "-" + dateObjct.getDate()
         var CurrentTime = dateObjct.toLocaleTimeString();
         var sourceCurrencyText = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlSource').options[document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlSource').selectedIndex].text;
         var targetCurrencyText = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlTarget').options[document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlTarget').selectedIndex].text;
         var SourceAmount = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_txtAmount').value;
         var UnitPriceST = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_OneUnitPrice').value;
 
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblDate').innerHTML = TodayDate.toString();
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblTime').innerHTML = CurrentTime.toString();
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblSourceCurrencyText').innerHTML = sourceCurrencyText;
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblTargetCurrencyText').innerHTML = targetCurrencyText;
         //alert("hi : " + UnitPriceST);
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lbl1UnitSourceToTarget').innerHTML = parseFloat(UnitPriceST).toFixed(2).toString();
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lbl1UnitTargetToSource').innerHTML = (1 / parseFloat(UnitPriceST)).toFixed(2).toString();
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblSourceAmount').innerHTML = parseFloat(SourceAmount).toFixed(2).toString();
         document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_lblTargetAmount').innerHTML = (parseFloat(SourceAmount) * parseFloat(UnitPriceST)).toFixed(2).toString();
     }
 
     function SetCurrency() {
         var sc = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlSource').value;
         var tc = document.getElementById('ctl00_ContentPlaceHolder1_CurrencyConverter1_ddlTarget').value;
 
         var i;
         for (i = 1; i <= 3; i++) {
             var idCurSr = "curSr" + i;
             var spn1 = document.getElementById(idCurSr);
             if (spn1 != null) {
                 spn1.innerHTML = sc;
             }
 
             var idCurTar = "curTar" + i;
             spn2 = document.getElementById(idCurTar);
             if (spn2 != null) {
                 spn2.innerHTML = tc;
             }
         }
     }
