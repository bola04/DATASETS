 function go_to_link(thisform)
 {
         var e = document.getElementById("categories");
         var f = document.getElementById("manufacturer");
         strcategory="";
         strmanuf="";

                 strcategory = e.options[e.selectedIndex].value;
         if(f.selectedIndex != 0)
                 strmanuf = f.options[f.selectedIndex].value;
         e.selectedIndex = 0;
         f.selectedIndex = 0;
         document.location="../stores/"+strcategory+".asp?"+"manufacturer="+strmanuf;
 }
