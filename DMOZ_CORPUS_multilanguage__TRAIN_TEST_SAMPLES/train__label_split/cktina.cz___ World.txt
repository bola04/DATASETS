 function unselectCountry(id){
 	$('input#sm-ci-sb-'+id).attr('disabled', true);

 				}
 				$(this).remove();
 			});
 							checkbox(destination, '0', 'di');

 		$(this).remove();
 	});

 	$('input#sm-ci-sb-'+id).attr('disabled', false);
 }

 function addDestination(id, type, destinationCheck){
 	$('input#sm-ci-sb-'+id).attr('disabled', true);
 	var destCheck = '';
 	if (destinationCheck == 1){
 					destCheck = 'czowOiIiOw==';
 				}
 	$.ajax({
 		dataType : 'html',
 		data : {
             'destinationCheck': destCheck,
             'type': type,
             'offerConditions': 1,
             'dataSource': 'cesys_default'
         },
 		url : "/searchmasks/destination_list_multiselect/"+id, //naceteme destinace prislusne zeme
 		success: function(options) {
 			$('.selectBox-sm-'+type+'-sb .sm-dropdown-content').append(options);
 			if (destinationCheck == 1){
 							}
 			if (destinationCheck != 1 && options != ''){
 				var mainDestination = $(options).find('input.selectBox-destination-main');
 				checkbox(mainDestination.val(), '1', type);
 				$('#'+mainDestination.attr('id')).attr('checked', true);
 			}
 
 			//handler pro přepínání destinací - omezeno přes $parentSelect na inputy v aktuálním electu
 			$parentSelect = $('.selectBox-sm-' + type + '-sb');  //např. $('selectBox-sm-di-sb-2')	
 			$parentSelect.find('#selectBox-destinations-countryId-'+id+' input').change(function(){
 				var data = new Array();
 				if ($(this).is(':checked')){
 					data['check'] = '1';
 				}else{
 					data['check'] = '0';
 				}
 				data['value'] = $(this).val();
 				$('#sm-'+type+'-sb').multiSelectBox('check', data);
 
 				//výběr elementu selectBoxu - hledáme jen uvnitř něj
 				$parentSelect = $(this).parents('.selectBox-dropdown-menu');
 
 				//zaškrtnutí checkboxu
 				if($(this).is(':checked')){
 					//volba nerozhoduje 
 					if ($(this).attr('class') == 'sm-'+type+' selectBox-destination-main'){
 						$parentSelect.find('.'+$(this).val()).each(function(){
 							if ($(this).is(':checked')){
 																checkbox($(this).val(), '0', type);
 								$(this).attr('checked', false);
 								$(this).attr('disabled', false);
 							}
 						});
 					}
 					//zaškrtnutá běžná destinace
 					else{
 						$mainDestSelect = $parentSelect.find('.selectBox-destination-main[country_id="'+id+'"]');

 							$mainDestSelect.attr('checked', false);
 							checkbox($mainDestSelect.val(), '0', type);
 							$parentSelect.find('li#selectBox-locations-destinationId-' + $mainDestSelect.val()).each(function(){
 								checkbox($(this).find('input').val(), '0', 'lo');
 								$(this).remove();
 							});
 						}

 								checkbox($(this).val(), '0', type);
 															}
 							$(this).attr('checked', true);
 							$(this).attr('disabled', true);
 						});
 					}

 					$parentSelect.find('.'+$(this).val()).each(function(i,j){
 						$('.selectBox-sm-lo-sb li#selectBox-locations-destinationId-'+$(this).val()).remove();
 					});

 									}
 				//odškrtnutí

 						if ($(this).is(':checked')){
 							checkbox($(this).val(), '0', type);
 														$(this).attr('checked', false);
 							$(this).attr('disabled', false);
 						}
 					});
 					//ochrana proti odškrtnutí všeho
 					//pokud odškrtneme poslední destinaci - zaškrtneme možnost nerozhoduje u této destinace
 					$parentSelect = $(this).parents('.selectBox-dropdown-menu');

 										$('.selectBox-sm-lo-sb li#selectBox-locations-destinationId-'+$(this).val()).remove();

 		}
