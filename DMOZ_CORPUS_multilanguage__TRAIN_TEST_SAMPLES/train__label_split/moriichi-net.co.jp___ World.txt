 function floating_menu(AREA) {
     var AREA = $(AREA),
         AREA_H = AREA.height(),
         AREA_W = AREA.width(),
         FLOATER = $('.btn_area'),
         FLOATER_W = FLOATER.width(),
         WINDOW = $(window),
         FH = FLOATER.height(),
         AT = AREA.offset().top,
         MT = 10;
 
     WINDOW.scroll(function() {
         var WT = WINDOW.scrollTop(),
             FT = FLOATER.offset().top;
 
         ////固定
         if (WT + MT >= FT || WT <= AT + AREA_H) {
             FLOATER.one().css({
                 position: 'fixed',
                 left: AREA.offset().left + AREA_W - FLOATER_W,
                 right: 'auto',
                 top: MT,
                 bottom: 'auto'
             }).addClass('floating');
         }
 
         //固定値上
         if ((WT + MT) <= AT) {
             FLOATER.one().css({
                 position: 'absolute',
                 left: 'auto',
                 right: 0,
                 top: 0,
                 bottom: 'auto'
             }).removeClass('floating');
         }
 
         //固定値下
         if (WT >= AT + AREA_H - FH - MT) {
             FLOATER.one().css({
                 position: 'absolute',
                 left: 'auto',
                 right: 0,
                 top: 'auto',
                 bottom: 0,
             }).removeClass('floating');
         }
     });
 
     // 固定中にResize
     WINDOW.resize(function(event) {
         if (FLOATER.hasClass('floating')) {
             FLOATER.css({
                 left: AREA.offset().left + AREA_W - FLOATER_W,
             });
         }
     });
 }
