          <h2 style="text-align: center;">
           Vítejte na oficiálních webových stránkách
          </h2>
          <h2 style="text-align: center;">
           obce Albrechtice v Jizerských horách
          </h2>
          <p style="text-align: justify;">
          </p>
          <p style="text-align: justify;">
           Obec Albrechtice v Jizerských horách se nachází v Libereckém kraji nedaleko města Tanvald. Jsme obec s bohatou historií, která sahá až do roku 1670. V současné době je hlavním lákadlem pro návštěvu naší obce turistický ruch. Tanvaldský Špičák je největší lyžařské středisko v Jizerských horách a na Mariánské Hoře najdete nástupní místo na Jizerskou magistrálu. V létě si můžete vychutnávat pěší turistiku i cykloturistiku po Jizerských horách. V obci žije více jak 300 trvalých obyvatel a dále chalupáři, kteří k nám přechodně dojíždějí trávit volný čas.  Rádi bychom Vám na našich webových stránkách nabídli přehledně informace o naší obci. Najdete zde praktické informace z dění v obci, novinky z obecního úřadu, možnosti ubytování, tipy pro zimní i letní aktivity.
          </p>

        function kaPrevMonth()
 {
 	var m = kaGetSelectedMonth.apply(this);
 	var y = kaGetSelectedYear.apply(this);
 	var minY = Math.min(getSelectOptionValues(this.find('#year')));
 
 	if(m === 1){
 		m = 12;
 		y -= (y === minY ? 0 : 1);
 	}else{
 		m -= 1;
 	}
 
 	kaRedrawCalendar.apply(this, [y, m]);
 }
 
 function kaNextMonth()
 {
 	var m = kaGetSelectedMonth.apply(this);
 	var y = kaGetSelectedYear.apply(this);
 	var maxY = Math.max(getSelectOptionValues(this.find('#year')));
 
 	if(m === 12){
 		m = 1;
 		y += (y === maxY ? 0 : 1);
 	}else{
 		m += 1;
 	}
 
 	kaRedrawCalendar.apply(this, [y, m]);
 }
 
 function getSelectOptionValues($select){
 	var values = [];
 	$select.find('opiton').each(function(){
 		values.push(this.value);
 	});
 
 	return values;
 }
 
 var sShow=false;
 var bShowDetail=false;
 function ka_showdetail(id)
 {
 	oEl=document.getElementById(id);
 	if (oEl)
 	{
 		if (sShow!=false)
 		{
 			cls.add(document.getElementById(sShow),'hide');
 		}
 		sShow=id;
 		cls.remove(oEl,'hide');
 		oEl.style.zIndex=9999;
 	}
 	else
 		sShow=false;
 }
 
 function ka_hidedetail(id)
 {
 	oEl=document.getElementById(id);
 	if (oEl && !bShowDetail)
 	{
 		cls.add(oEl,'hide');
 		sShow=false;
 		bShowDetail=false;
 	}
 }
 
 function ka_hidedetails()
 {
 	if (sShow!=false)
 	{
 		oEl=document.getElementById(sShow);
 		if (oEl)
 		{
 			cls.add(oEl,'hide');
 			sShow=false;
 			bShowDetail=false;
 		}
 	}
 }
 
 function kaRedrawCalendar(year, month){
 	var $form = this;
 	if(year !== undefined){
 		$form.find('#year option[value="' + year + '"]').prop('selected', true);
 	}
 	if(month !== undefined){
 		$form.find('#month option[value="' + month + '"]').prop('selected', true);
 	}
 
 	jQuery.ajax({
 		url: 'modules/kalendar/ax_kalendar.php'
 		, data: $form.serialize()
 		, type: 'POST'
 		, beforeSend: function(xhr, settings){}
 		, success: function(data){
 			if(data.length)
 				$form.find('#kal_table table').replaceWith(data);
 		}
 		, complete: function(){}
 	});
 }
 
 function kaGetSelectedMonth(){
 	return parseInt(this.find('#month option:selected').val(), 10);
 }
 function kaGetSelectedYear(){
 	return parseInt(this.find('#year option:selected').val(), 10);
 }
 
 function getParentForm(el) {
 	return jQuery(el).closest('form');
 }
 
 jQuery(document).ready(function(){
 	jQuery('.kalendar-19').live('submit', function(e){
 		e.preventDefault();
 		return false;
 	});
 
 	jQuery('.kalendar-19 .arrows input[type=image]:eq(0)').live('click', function(e){
 		e.preventDefault();

 	jQuery('.kalendar-19 .arrows input[type=image]:eq(1)').live('click', function(e){

 	jQuery('.kalendar-19 #month, .kalendar-19 #year').live('change', function(e){
 		e.preventDefault();
 		kaRedrawCalendar.apply(getParentForm(this));

 });
