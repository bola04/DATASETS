 				var username = $('input[name=username]');
 				var password = $('input[name=password]');
 
 				usernameText = username.val().trim();
 				passwordText = password.val().trim();

 					$(this).removeClass("error");
 				});
 				password.on("keydown", function() {
 					$(this).removeClass("error");
 				});
 
 				if(usernameText && passwordText) {					
 					this.action = this.action.replace("___username___", encodeURIComponent(usernameText));
 					this.action = this.action.replace("___password___", encodeURIComponent(passwordText));
 
 					this.submit();
