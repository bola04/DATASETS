      <script>
       $(function() {
     $('#nature').change(function() {
       refreshSearchForm();
     });
 
     $('#customcity').multipleSelect({
       placeholder: "Area (All)",
       selectAllText: "Select all",
       selectAllDelimiter: "",
       width: '100%',
       minimumCountSelected: 2,
       countSelected: "# selected",
       onOptgroupClick: function(view) {
         var li = view.inputgroup.parents('li');
         var countItems = view.children.length;
         var countSelected = 0;
 
         li.nextUntil('.group').each(function() { if ($(this).hasClass('selected')) countSelected++; });
 
         if (countSelected == countItems) {
           li.nextUntil('.group').addClass('display');
         } else if (countSelected == 0) {
           li.removeClass('open');
           li.nextUntil('.group').removeClass('display');
         } else if (li.hasClass('open')) {
           li.removeClass('open');
         } else {
           li.nextUntil('.group').toggleClass('display');
         }
       },
       onInit: function() {
         $('.ms-drop-actions').appendTo('.ms-drop');
         $('.ms-drop-actions').show();
 
         $('.close').click(function(e) {
           e.preventDefault();
           $('.ms-choice').click();
         });
 
         $('.clear').click(function(e) {
           e.preventDefault();
           $('#customcity').multipleSelect('uncheckAll');
           $('.ms-drop ul > li').removeClass('open');
           $('.ms-drop ul > li').removeClass('display');
           $('.ms-choice').click();
         });
       },
       onClick: function(view) {
         var li = view.input.parents('li');
         var group = li.prev().hasClass('group') ? li.prev() : li.prevUntil('.group').last().prev();
         var countSelected = 0;
 
         group.nextUntil('.group').each(function() { if ($(this).hasClass('selected')) countSelected++; });
 
         if (countSelected == 0) {
           group.removeClass('open');
           group.nextUntil('.group').removeClass('display');
         }
       },
       onOpen: function() {
         $('.ms-drop ul > li.group').each(function() {
           var items = 0;
           var selected = 0;
           
           $(this).nextUntil('.group').each(function() {
             items++;
 
             if ($(this).hasClass('selected'))
               selected++;
           });
 
           if (selected == items)
             $(this).find('input[type="checkbox"]').prop('checked', true);
         });
 
         $('.ms-drop').find('li').not('.group').find('input[type="checkbox"]').each(function() {
           if ($(this).is(':checked')) {
             $(this).parents('li').addClass('display');
             $(this).parents('li').prevUntil('.group').last().prev().addClass('open');
 
             // add display to same group items
             $(this).parents('li').prevUntil('.group').addClass('display');
             $(this).parents('li').nextUntil('.group').addClass('display');
           }
         });
       }
     });
   });
 
   /** Update fields in search form.
    * @param (bool) submit : If true, form is submitted after update. */
   function refreshSearchForm()
   {
     var serialized = $(".search form").serialize();
 
     $.ajax({
       url: "/en/searchForm",
       type: "POST",
       dataType: "html",
       data: serialized,
       success:function(data, textStatus) {
         $(".search").html(data);
       }
     });
   }
