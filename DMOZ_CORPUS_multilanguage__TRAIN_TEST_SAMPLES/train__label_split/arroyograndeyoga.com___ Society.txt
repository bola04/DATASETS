            $(document).ready(function() {
 
         //show or hide buttons
         var show_buttons = $("#show_buttons").val();
 
         if(show_buttons == 1) {
             $(".le_mb_button").css('display', 'inline');
         }
         else {
             $(".le_mb_button").css('display', 'none');
         }
 
 
         var show_filters = $("#show_filters").val();
         //show filters option
         if(show_filters == 1) {
             $(".schedule_filters").css('display', 'inline');
         }
         else {
             $(".schedule_filters").css('display', 'none');
         }
         var show_trainers = $("#show_trainers").val();
         if(show_trainers == 1) {
             $(".trainer_name").css('display', 'inline');
         }
         else {
             $(".trainer_name").css('display', 'none');
             $(".title_staff").css('display', 'none');
         }
 
         var stay_here = $("#stay_here").val();
         var sc_filter = $("#sc_filter").val();
         var cl_filter = $("#cl_filter").val();
         var date_select_filter = $("#date_select_filter").val();
         var print_view = $("#print_view").val();
         var location_filter = $("#location_filter").val();
         var trainer_filter = $("#trainer_filter").val();
 
 
         //get our incoming filters all setup for all the arguments. these are from the element options, and can't change based on user selections (like the dropdowns can)
         var filter_url = "&show_filters=" + show_filters + "&show_trainers=" + show_trainers + "&date_select_filter=" + date_select_filter + "&sc_filter=" + sc_filter + "&cl_filter=" + cl_filter + "&show_buttons=" + show_buttons + "&stay_here=" + stay_here + "&print_view=" + print_view + "&location_filter=" + location_filter + "&trainer_filter=" + trainer_filter;
         //get all our class types that were selected from the original element!
         var class_types = $("#sched_cl").val();
         if(class_types != "") {
             var classes = jQuery.parseJSON(class_types);
             for(var i=0; i<classes.length; i++) {
                 filter_url += "&cl[]=" + classes[i];
             }
         }
 
         //get all our service categories that were selected from the original element!
         var services = $("#sched_sc").val();
         if(services != "") {
             var service_cats = jQuery.parseJSON(services);
             for(var i=0; i<service_cats.length; i++) {
                 filter_url += "&sc[]=" + service_cats[i];
             }
         }
 
         //get all the stuff setup so we only have to check the one when user makes a selection
         var selected_date = $("#hidden_date").val();
         var sess = $('#sessions option:selected').val();
         var sc = $('#service_categories option:selected').val();
         var trainer = $('#trainers option:selected').val();
         var loc = $('#locations option:selected').val();
         loc_arg = "";
         if(loc !== undefined) {
             loc_arg = "&locationid=" + loc;
         }
         var num_days = $('#num_days').val();
 
         //locations
         $("#locations").change(function() {
 
             var loc2 = $('#locations option:selected').val();
             if(loc2 == undefined) {
                 loc_arg2 = '';
             }
             else {
                 loc_arg2 = "&locationid=" + loc2;
             }
 
             window.location.replace("/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer + "&date=" + selected_date + "&num_days=" + num_days + "&service_category_id=" + sc + loc_arg2 + filter_url);
 
         });
 
 
         $("#sessions").change(function() {
 
             var sess2 = $('option:selected', this).val();
             window.location.replace("/apps/mindbody/elements/listschedule.php?sessionid=" + sess2 + "&trainerid=" + trainer + "&date=" + selected_date + "&num_days=" + num_days + "&service_category_id=" + sc + loc_arg + filter_url);
 
         });
 
         $("#trainers").change(function() {
 
             var trainer2 = $('option:selected', this).val();
 
             window.location.replace("/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer2 + "&date=" + selected_date  + "&num_days=" + num_days + "&service_category_id=" + sc + loc_arg + filter_url);
 
         });
 
         $("#service_categories").change(function() {
 
             var sc2 = $('option:selected', this).val();
 
             window.location.replace("/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer + "&date=" + selected_date  + "&num_days=" + num_days + "&service_category_id=" + sc2 + loc_arg + filter_url);
 
         });
 
 
         $('#date_filter').datepicker( {
             dateFormat: 'mm/dd/yy',
             onSelect: function() {
                 var date = $('#date_filter').val();
 
                 window.location.replace("/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer + "&date=" + date  + "&num_days=" + num_days + "&service_category_id=" + sc + loc_arg + filter_url);
 
             },
             inline: true
         });
 
         //set the hrefs of the two forward and back links!
         var start_date = $("#start_date").val();
         $("#back_sched").attr("href", "/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer + "&service_category_id=" + sc  + "&num_days=" + num_days + loc_arg + "&back=" + start_date + filter_url);
 
         $("#forward_sched").attr("href", "/apps/mindbody/elements/listschedule.php?sessionid=" + sess + "&trainerid=" + trainer + "&service_category_id=" + sc  + "&num_days=" + num_days + loc_arg + "&forward=" + start_date + filter_url);
 
         $("#printview").attr("href", "/apps/mindbody/listscheduleprint.php?sessionid=" + sess + "&trainerid=" + trainer + "&service_category_id=" + sc  + "&num_days=" + num_days + loc_arg + "&date=" + start_date + filter_url);
 
 
         $(".cancel_link").click(function() {
 
             var url = "/apps/mindbody/cancel.php";
             var link = $(this);
             var id = $(this).attr("id");
 
             var instance_id = $(this).attr("instanceID");
 
             $("#loading_" + instance_id).show();
             $(this).css('cursor', 'default');
 
             if($(this).hasClass('cancel_class')) {
                 url += "?visitID=" + id + "&instanceID=" + instance_id;
 
             }
             else if($(this).hasClass('waitlistcancel')) {
                 url += "?waitlistID=" + id;
             }
 
             //AJAX call here. if return latecancel prompt, prompt for late cancellation.
             $.getJSON(url, function(data) {
 
                 $("#late_cancel_button").attr("url", url);
                 $("#late_cancel_button").attr("id_to_delete", instance_id);
 
 
                 if(data.status == 'latecancelprompt') {
 
                     $('#late_cancel_modal').modal({
                         height: 200
                     });
                     $(".small_loader").hide();
 
                 }
                 else if(data.status == 'fail'){
                     $("#tr_" + instance_id + ".signup_message").append("<div class='error'>" + data.message + "</div>");
                 }
                 else if(data.status == 'success') {
                     $("#tr_" + instance_id + " td.row_button div.signup_message").remove();
                     $("#tr_extra_" + instance_id).remove();
                     $("#tr_" + instance_id + " td.row_button a.le_metro_silver_button").replaceWith('<div class="cancel_success">Cancellation Success!</div>');
                     $(".notice").text(data.message);
                     $(".small_loader").hide();
 
                 }
 
             });
 
         });
 
 
         $("#late_cancel_button").click(function() {
 
             var url = $(this).attr("url");
             $.getJSON(url + "&late=true", function(data) {
 
                 var id = $("#late_cancel_button").attr("id_to_delete");
 
                 if(data.status == 'fail') {
                     $("#tr_" + id + " .signup_message").append("<div class='error'>" + data.message + "</div>");
                 }
                 else if(data.status == 'success') {
                     var row = "#tr_" + id;
                     $(row + " td.row_button div.signup_message").remove();
                     $("#tr_extra_" + id).remove();
                     $(row + " td.row_button a.le_metro_silver_button").replaceWith('<div class="cancel_success">Cancellation Success!</div>');
                     $('#late_cancel_modal').modal('hide');
                     $(".small_loader").hide();
                 }
 
 
             });
 
         });
 
 
 
     });//end ready function
           </script>
