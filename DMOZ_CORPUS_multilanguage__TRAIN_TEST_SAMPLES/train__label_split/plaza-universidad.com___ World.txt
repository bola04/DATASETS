    function validateMail(inputNewsletter)
       {
       //Creamos un objeto
       object=document.getElementById(inputNewsletter);
       valueForm=object.value;
       var formulario = document.form_news;
       // Patron para el correo
       var patron=/^\\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,4})+$/;
       if(valueForm.search(patron)==0)
       {
       //Mail correcto
       object.style.color="#000";
       return;
       formulario.submit();
       }
       //Mail incorrecto
       object.style.color="#f00";
       }

         function loadCarousel() {

             $.ajax({
                 url: 'recursos/xml/promos.xml?t=4280',
                 success: function(xml){
                     var xmlDoc = $(xml);
                     var root = xmlDoc.find(' > promos[status=1]');
                     var page = root.attr('url');
                     var html = '';
                     if (root.length > 0) {
                         var promos = root.find(' > promo[status=1]');
                         if (promos.length > 0) {
                             html += '<div class="slider1">';
                             promos.each(function(){
                             var promo = $(this);
                             var url = promo.attr('url');
                             var text = promo.find(' > text').html();
                             html += '<div class="slide">';
                                 html += '<a href="'+page+url+'"><img src="'+promo.find('photo').text()+'" title="'+text+'" alt="'+text+'"></a>';
                             html += '</div>';
                             });
                             var wr = $('#promociones-wr');
                             wr.html(html);
                             wr.find('.slider1').bxSlider({
                                 slideWidth: 360,
                                 captions: true,
                                 minSlides: 2,
                                 maxSlides: 3,
                                 pager: false,
                                 slideMargin: 10
                             });
                             setTimeout(function(){
 var img = wr.first('img');
 var h = img.height() + 10;
 wr.find('.bx-viewport').css('height', h+'px');
 var caps = wr.find('.bx-caption');
 caps.each(function(){
  var $this = $(this);
  var div = $this.parent();
  var a = div.find(' > a');
  $this.html('<a href="'+a.attr('href')+'" >'+$this.html()+'</a>');
 });
 }, 500);
                             
                         }
                         
                         
                     }
                     html += '</div>';
                 },

                     alert('Ajax Error => '+err+' status => '+status);
