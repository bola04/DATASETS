    $(function() {
             $(".form-web input").focus(function(){
                 $(this).parents().removeClass('has-error');
                 $(this).prev().remove();
             });
             $("#webSubmit").click(function(){
                 var name = $('#webName').val();
                 var email = $('#webEmail').val();
                 $.ajax({
                     type: "POST",
                     url: "web.php",
                     data: $('form.form-web').serialize(),
                     success: function(msg){
                         if (msg == 1) {
                             $("#webbox").modal('hide');
                             $("[id=webThanks]").html("<p class=\"text-success\" style=\"color:#2f2f2f;\" >Thank you for submission! We will contact you soon.</p>");
                         } else {
                             if (msg == 2 || msg == 4) {
                                 if ($(".form-web .form-group:first-child").find("label").length == 0) {
                                     $(".form-web .form-group:first-child").prepend("<label class=\"control-label\" for=\"webName\">Please provide your name.</label>");
                                 }
                                 if (!($(".form-web .form-group:first-child").hasClass('has-error'))) {
                                     $(".form-web .form-group:first-child").addClass('has-error');
                                 }
                             }
                             if (msg == 3 || msg == 4) {
                                 if ($(".form-web .form-group:nth-child(2)").find("label").length == 0) {
                                     $(".form-web .form-group:nth-child(2)").prepend("<label class=\"control-label\" for=\"webEmail\">Please provide a valid e-mail address.</label>");
                                 }
                                 if (!($(".form-web .form-group:nth-child(2)").hasClass('has-error'))) {
                                     $(".form-web .form-group:nth-child(2)").addClass('has-error');
                                 }
                             }
                         }
                     },
                     error: function(){
                         alert("Error submitting admission to the server.");
                     }
                 });
             });
         });
