    <script type="text/javascript">
     jQuery(function(){          	
 
           	//imgContainer.trigger('ROTATOR_CYCLE_TO', id);
       
             var thumbContainer = jQuery('#banner-nav');
             var thumbnails = jQuery('#banner-nav li img');
             var imgContainer = jQuery('.rotator');
             var images = jQuery('.rotator li img');
             var currentID = 0;
             var isRotating = false;
 
             /**
             *  Clicking on the thumbnails will transition to 
             *  the selected image in the sequence and pause the rotator 
             */
             thumbnails.click(function(e){
               if (!isRotating) {
                 var index = jQuery.inArray(this, thumbnails);
                 if (index != currentID) {
                   isRotating = true;
                   imgContainer.trigger('ROTATOR_CYCLE_TO', [index]);                  
                 }
               }
             });
 
             /**
             *  Set the active thumbnail when we rotate to a new image
             */
             $(document).bind('ROTATOR_TRANSITION_START', function(e, oldIndex, newIndex) {
               thumbnails.removeClass('active').each(function(){
                 this.src = this.src.replace('-active', '-inactive');
               }).eq(newIndex).addClass('active');
               //alert(activeItemIndex);
               activeThumb = thumbnails.get(newIndex);
               activeThumb.src = activeThumb.src.replace('-inactive', '-active');
               currentID = newIndex;
               isRotating = false;
 						});
 
           });
    </script>
    <script type="text/javascript">
     jQuery(function() {
 	      	function updateTimestamp() {
 	        	// Timestamp code.
 	        	var daysOfWeek = new Array('Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday');
 	        	var months = new Array(
 	        		'January', 'February', 'March',
 	        		'April', 'May', 'June', 'July',
 	        		'August', 'September', 'October',
 	        		'November', 'December'
 	        	);
 	        	var date = new Date();
 	        	var dayOfWeek = daysOfWeek[date.getDay()];
 	        	var month = months[date.getMonth()];
 	        	var ampm = 'am';
 	        	var hour = date.getHours();
 	        	if (hour > 12) {
 	        		ampm = 'pm';
 	        		hour -= 12;
 	        	}
 	        	var minutes = date.getMinutes();
 	        	if (minutes < 10) {
 	        		minutes = "0" + minutes.toString();
 	        	} else {
 	        		minutes = minutes.toString();
 	        	}
 	        	jQuery("#timestamp").html(dayOfWeek + ", " + month + " " + date.getDate().toString() + ", " + date.getFullYear().toString() +
 	        		" " + hour.toString() + ":" + minutes + " " + ampm);
 	      	}
 	      	
 	      	updateTimestamp();          	
 	      	setInterval(updateTimestamp, 60000);
 	      });
    </script>
