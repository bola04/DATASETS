 function equalHeights() {
 	if ($(window).width() > 480) {
 		$('.microscope-category .text').height("auto");
 		$('.microscope-grid-wrap').each(function() {
 			var tallestHeight = 0;
 			$(this).children('.microscope-category').each(function() {
 				var currentHeight = $(this).find('.text').height();
 				if (currentHeight > tallestHeight) {
 					tallestHeight = currentHeight;
 				}
 			});
 			$(this).find('.text').height(tallestHeight);
 		});
 	} else {
 		$('.microscope-category .text').height("auto");
 	}
 }
 
 function openFirstProduct() {
 	$('ul.products li:first-child').addClass('open');
 	$('ul.products li:first-child a').addClass('open');

 }
