         function reposition() {
             var footer = document.getElementById('footerWrapper');
             if (!footer)
 			    return;
 
             var bodyHeight = document.body.clientHeight;
 
             var sidetabs = $('#sidetab-left');
             if (sidetabs.length > 0) {
                 var sidetabsBottom = sidetabs.height() + sidetabs.offset().top;
                 if (sidetabsBottom > bodyHeight) bodyHeight = sidetabsBottom;
             }
 
             var unusedHeight = document.documentElement.clientHeight - bodyHeight;
             if (unusedHeight < footer.clientHeight) {
                 footer.style.bottom = '';
                 footer.style.top = bodyHeight + 10 + 'px';
             }
             else {
                 footer.style.top = '';
                 footer.style.bottom = '0px';
             }
         }
