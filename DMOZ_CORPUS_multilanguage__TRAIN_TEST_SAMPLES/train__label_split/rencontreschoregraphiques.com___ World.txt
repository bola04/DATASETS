   <script type="text/javascript">
    jQuery(function($){
             $(document).ready(function(){
                 $('.news.carousel').carousel({
                     interval: 4000
                 });
 
                 $('#viewer').carousel({
                     interval: 3000
                 });
 
 
                 // --- DATEPICKER ---
                 var click = false;
                 var allDates = new Array();
                 var xhr;
 
                 var addClassToDatePicker = function(date){
                     var day = date.getDate();
                     var month = parseInt(date.getMonth()) + 1;
                     var year = date.getFullYear();
                     month = month <= 9 ? "0" + month : month;
                     day = day <= 9 ? "0" + day : day;
                     var compareDate = day + "/" + month + "/" + year;
                     if ($.inArray(compareDate, allDates) > -1){
                         return [true, "ui-state-has-spectacles"];
 
                     } else {
                         return [false, ""];
                     }
                 }
 
                 var loadDateForDatePicker = function(addClass){
                     $(".calendar .datepicker").next(".dates").find("input").each(function(){
                         var value = $(this).val();
                         allDates.push(value);
                     });
                 }
 
                 var onSelectDatePickerSuccess = function(html){
                     $(".cal-popup").remove();
                     $("body").append(html);
                     var offsetDatepicker = $(".ui-state-active").offset();
                     var popup = $(".cal-popup");
                     popup.css({
                         "top" : (offsetDatepicker.top-(popup.height()))-55,
                         "left" : (offsetDatepicker.left-(popup.width()/2))-70
                     });
                     popup.fadeIn();
                     $(".icon-close").click(function(){
                        $(this).parent().remove();
                     });
                 }
 
                 var onSelectDatePicker = function(date, inst){
                     if (xhr){
                         xhr.abort();
                     }
                     xhr = $.ajax({
                         url : $(".calendar .datepicker").attr("data-url"),
                         type: "POST",
                         data : {date: date},
                         success : onSelectDatePickerSuccess
                     });
                 }
 
                 loadDateForDatePicker(true);
 
                 var today = new Date();
                 $('.calendar .datepicker').datepicker({
                     onSelect: onSelectDatePicker,
                     beforeShowDay: addClassToDatePicker,
                     showOtherMonths: true,
                     minDate: new Date(today.getFullYear(), 4, 1),
                     maxDate: new Date(today.getFullYear(), 5, 30)
                 });
 
                 /*
                  * Set calendar's month
                  * if current date is between 01/07 and 31/12 set month on June
                  */
                  var secondSemester = new Date(today.getFullYear(), 5, 1);
                  if (today.getTime() > secondSemester.getTime()) {
                     $('.calendar .datepicker').datepicker("setDate", "01/06/"+today.getFullYear());
                  }
 
                 $('.set-date').click(function(e){
                     e.preventDefault();
                     // Handle datepicker
                     var month = $(this).data('month');
                     $('.calendar .datepicker').datepicker("setDate", "01/" + month + "/" + today.getFullYear());
                     // Handle class active
                     $('.set-date').each(function(){
                         if ($(this).data('month') == month) {
                             $(this).parent('span').addClass('active');
                         } else {
                             $(this).parent('span').removeClass('active');
                         }
                     })
                 });
             });
         });
