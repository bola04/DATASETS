     function jsLinkUrlPost(url, params) {
         method = "post";

         form.setAttribute("action", url);

     function removeTag(keyword) {
 	keyword = keyword.replace(/<(\/)?([a-zA-Z]*)(\s[a-zA-Z]*=[^>]*)?(\s)*(\/)?>/ig, "");
 	return keyword;
 }
 
 function getByte(text){
 
     var length = text.length;
     var tcount = 0;
     for(var i = 0; i < length; i++){
       var byteStr = text.charAt(i);
       if(escape(byteStr).length > 4){
         tcount += 2;
       }else{
         tcount += 1;
       }
     }
 
     return tcount;
 }
 
 function checkMax(text){
 	if(getByte(text)>30){
 		return false;
 	}else{
 		return true;
 	}
 }

 function keywordSearch(value) {

 		return true;
 	}

 }

     function beakerBrand(obj){
         $(".alphabetize a").removeClass("on");
         $(obj).addClass("on");
 
         var key = $(obj).attr('key');
 
         var brndShopId = "";
 
         if($("#lnbDspCtgryForm #brandShopNo").val().indexOf("BDMA09") > -1){
             brndShopId = "MCBR";
         }else if($("#lnbDspCtgryForm #brandShopNo").val().indexOf("BDMA19") > -1){
             brndShopId = "ORBR";
         }
 
         if(key != ''){
             $.ajax({
                 type : "POST",
                 url : "/public/category/beakerBrndList.json",
                 data : {'brndShopId':brndShopId, 'indexKey':key},
                 async : true,
                 success : function(data) {
 
 
                     var strHTML = "<div id='list_brand' class='mCS' data-thema='purple'><ul>";
 
                     if(data.result.length > 0){
                         for (var i = 0; i < data.result.length; i++) {
                             strHTML += "<li";
                             if(data.result[i].dspCtgryNo == $('#brandShopNo').val()){
                                 strHTML += " class='on'";
                             }
                             strHTML += ">";
                             //2017-10-16 : 전시카테고리 출력 유형이 링크이고 링크 URL 이 null일때 a 태그 없음
                             if(data.result[i].outptSectCd == 'LINK' && data.result[i].outptLinkUrl == ''){
                                 strHTML += "<span>" + data.result[i].dspCtgryNm + "</span>";
                             } else {
                                 strHTML += "<a href='#none' onclick='javascript:goAction(this); return false;'";
                                 strHTML += "outptSectCd='" + data.result[i].outptSectCd + "'";
                                 strHTML += "outptLinkUrl='" + data.result[i].outptLinkUrl + "'";
                                 strHTML += "ctgryOutptTpCd='" + data.result[i].ctgryOutptTpCd + "'";
                                 strHTML += "brndShopId='" + data.result[i].dspBrndId + "'";
                                 strHTML += "ctgryNo='"+data.result[i].dspCtgryNo+"'";
                                 strHTML += "ctgryNm='"+data.result[i].dspCtgryNm+"'";
                                 strHTML += ">"+data.result[i].dspCtgryNm+"</a>";
                             }
                             strHTML += "</li>";
                         }
                     }else{
                         strHTML += "해당 브랜드가 없습니다.";
                     }
 
                     strHTML += "</ul></div>";
 
                     //$('#beakerList').html(strHTML);
                     $('#list_brand').remove();
                     $('.alphabetize').after(strHTML);
                     mCScroll();
                 }
             });
         }else{
             var strHTML = "<div id='list_brand' class='mCS' data-thema='purple'><ul></ul></div>";
             $('#list_brand').remove();
             $('.alphabetize').after(strHTML);
             $("#list_brand ul").html($("#topBrand").html());
             mCScroll();
         }
 
         lnbHeight();
     }

 var newInFlag = true;
 function drawNewIn(data){
     var line = 10;
     var html = "";
     $('#newIn .list_goods').remove();
     html += "<div class='list_goods'>";
     html += data;
     html += "</div>";
 
     $("#newIn").append(html);
 
     if($('#newIn .list_goods ul').length < 1){
         $('#newIn').remove();
         $("#newInViewMore").remove();
     }else{
         var moreUrl = "";
         if(!$("#brandShopNo").val()){
             moreUrl = displayCategoryUrl(newInDspCtgryNo, "");
         }else{
             moreUrl = brandShopListUrl($("#brandShopNo").val(), newInDspCtgryNo);
         }
         $("#newInViewMore").attr('href', moreUrl + "?brandShopNo=" + $("#brandShopNo").val() +"&dspCtgryNo=" + newInDspCtgryNo + "&brndShopId=" + $("#brndShopId").val() + "&leftBrandNM="+$("#leftBrandNM").val());
 
         if(newInFlag){
             if(newInWomen == null || newInWomen == '' || newInWomen < line){
                 if($('#SFMA31A41').length > 0) $('#SFMA31A41').remove();
             }
             if(newInMen == null || newInMen == '' || newInMen < line){
                 if($('#SFMA31A42').length > 0) $('#SFMA31A42').remove();
             }
             if(newInKids == null || newInKids == '' || newInKids < line){
                 if($('#SFMA31A43').length > 0) $('#SFMA31A43').remove();
             }
             if(newInLifeStyle == null || newInLifeStyle == '' || newInLifeStyle < line){
                 if($('#SFMA31B84').length > 0) $('#SFMA31B84').remove();
             }
 
             // 신상품 하위의 대카테고리 필터 (전체, WOMEN, MEN, KIDS, LIFESTYLE, OUTLET)
             if ($("#newIn .sort li").length > 2) { // 해당 하위 대카테고리가 1개 이하이면 노출 안함
                 $("#newIn .sort li").click(function () {
                     var dspCtgryNo = '';
                     var isFilter = false;
                     var cnt = $("#newIn").attr("cnt");
                     if ($(this).attr('id') != "" && $(this).attr('id') != undefined) {
                         dspCtgryNo = $(this).attr('id'); // 전체가 아니면 해당 대카테고리 번호 셋팅
                         isFilter = true; // 전체가 아니면 대카테고리 필터 여부 true
                     }
                     $("#newIn .sort li").removeClass("on");
                     $(this).addClass("on");
                     newInLoading('ssf', cnt, dspCtgryNo, isFilter);
                 });
             } else {
                 $("#newIn .sort").remove();
             }
         }
         newInFlag = false;
     }
 }

     function getPostsFirstCall(){
         var gPOSTS_COUNT = insCount;
         var feed = $("#feed");
         if(feed != undefined && $("#feed").val() != undefined && $("#feed").val() != ""){
             var gAccess_Token = $("#feed").val();
             var url = 'https://api.instagram.com/v1/users/self/media/recent?access_token='+gAccess_Token+'&count='+gPOSTS_COUNT;
             getPosts(url);
         }

     function getPosts(url){
         var gPOSTS_COUNT = insCount;
 
         var checkWord = $("#tag").attr("value") ;
 
         $.ajax({
         	url: '/instagramApiResult.json',
             type: 'GET',
             dataType: 'json',
             data : {instagramApiUrl : url},
             async: true,
             success: function(getData){
             	if(getData != null && getData.instagramResult != null){
                 	var jsonData = getData.instagramResult
                 	if(jsonData != null && jsonData.data != null){
                 		loadPosts(jsonData, checkWord);
                 	} else {
                 		$("#instagramCnr").remove();
                 	}
                 } else {
             		$("#instagramCnr").remove();
             	}
             },
             error:function(data){
                 $("#instagramCnr").remove();
             }
         });
     }
 
     var receiveDataList = new Array();
 
     function loadPosts(getData, word){
         if( (word != null && word.length > 0)){
             $.each(getData.data, function(i){
                 var pData = this;
                 var postTags= pData.tags;
                 if(postTags != null){
                     $.each(postTags, function(j){
                         if(postTags[j] == word){
                             receiveDataList[postCount] = pData;
                             postCount ++;
                             return false;
                         }
                     });
                 }
                 if(postCount > 4){
                     return false;
                 }
             });
         }else{
             $.each(getData.data, function(i){
                 var pData = this;
                 receiveDataList[postCount] = pData;
                 postCount ++;
                 if(postCount > 4){
                     return false;
                 }
             });
         }
 
         totalCount += insCount;
 
         if(totalCount < 50){
             if(postCount < 4){
                 getPosts(getData.pagination.next_url);
             }else{
                 drawInstramArea(receiveDataList)
             }
         }
     }
 
     function drawInstramArea(data){
         $.each(data, function(i){
             var postImg = this.images.standard_resolution.url;
             var postUrl = this.link;
 //            var captionText = this.caption.text;
 //            var postCnt = this.likes.count;
             var postTags= this.tags;
 
 //            var commendCnt = this.comments.count;
 //            var userName = this.user.username;
 //            var userPicture = this.user.profile_picture;
             var imgElement = $('#pic'+i);
             if(postImg == '' || postImg == null){
                 $("#instagramCnr").remove();
                 return false;
             }else{
                 imgElement.attr('src', postImg);
                 imgElement.parent().attr('href', postUrl);
 
 //                if(captionText.length > 430){
 //                    captionText = captionText.substr(0, 427) + "...";
 //                }
                 var sapNos = new Array();
                 var index = 0;
                 if(postTags != null){
                     $.each(postTags, function(j){
                         // 관련 상품정보
                         if(postTags[j].indexOf('code_') > -1){
                             sapNos[index] = postTags[j].replace('code_','').toUpperCase();
                             index ++;
                         }
                     });
                     if(sapNos != null){
                         var parameters = '';
                         $.each(sapNos, function(idx){
                             if(idx == 0){
                                 parameters += sapNos[idx];
                             }
                             else{
                                 parameters += ","+sapNos[idx];
                             }
                         });
 
                         $('#sapNo'+i).val(parameters);
                         if(parameters=='') {
                         	$('#pic'+i).parent().next().hide();
                         }
 
                     }
                 }
             }
         });
     }
