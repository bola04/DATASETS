 		var totalCars = cars.length;
 		var container = cars.parent();
 		for (i = 0; i < cars.length; i++) {
 			// get node:
 			node = cars.eq(Math.floor(Math.random()*cars.length));
 
 			// extract and append:
 			node.remove().appendTo(container);
