       function toggleFindCruise() {
         scrollTo($("#find-cruise-bar"));
         $("#find-cruise-bar").toggleClass("active");
     }
 
     $(document).ready(function(){
 
         $(".select-box>div").click(function(){
 
             if($(this).siblings(".select-list").hasClass("active")){
 
                 $(".find-cruise-arrow.active").removeClass("active");
                 $(".select-list.active").removeClass("active");
                 $("#reset-done-bar").removeClass("active");
                 $("#search-bar").removeClass("active");
                 $("body").removeClass("no-scroll");
                 $("#find-cruise-bar").removeClass("active");
             }else{
                 $("#reset-done-bar").removeClass("active");
                 $("#search-bar").removeClass("active");
                 $(".find-cruise-arrow.active").removeClass("active");
                 $(".select-list.active").removeClass("active");
                 $(this).find(".find-cruise-arrow").toggleClass("active");
                 $(this).siblings(".select-list").toggleClass("active");
                 $("#find-cruise-bar").addClass("active");
 
                 if($(this).closest("#travelparty-select-box").length==0){
                     $("#search-bar").toggleClass("active");
                     $("#reset-done-bar").toggleClass("active");
                     $("body").toggleClass("no-scroll");
                 }
             }
 
         });
 
         $(".select-box a").click(function(){
             $(this).closest("li").siblings("li").find(".selected").removeClass("selected");
             $(this).addClass("selected");
             if($("#reset-done-bar").css("display")=="none"){
 
                 $(this).closest(".select-list").removeClass("active");
                 $("#reset-done-bar").removeClass("active");
                 $("#search-bar").removeClass("active");
                 $("body").removeClass("no-scroll");
                 $("#reset-done-bar").removeClass("active");
                 $("#search-bar").removeClass("active");
                 $("#find-cruise-bar").removeClass("active");
 
                 var remark = $(this).attr("remark");
                 console.log(remark);
                 if(remark!=undefined)
                     remark = ", "+remark;
                 else
                     remark = "";
 
                 $(this).closest(".select-box").find(".value").html($(this).html()+remark);
                 var id = $(this).closest(".select-box").find(".value").attr('id');
                 id = id.replace("-value","");
 
                 $("#booking-widget input[name='"+id+"']").attr("value",$(this).attr("value"));
 
             }
         });
 
         $("#reset-btn").click(function(){
 
             var id = $(".select-list.active").attr("id");
             id = id.replace("-ul","");
             $("#booking-widget input[name='"+id+"']").attr("value","");
 
             $(".select-list").removeClass("active");
             $("#reset-done-bar").removeClass("active");
             $("#search-bar").removeClass("active");
             $("body").removeClass("no-scroll");
 
         });
 
         $("#done-btn").click(function(){
 
             var remark = $(".select-list.active a.selected").attr("remark");
             if(remark!=undefined)
                 remark = ", "+remark;
             else
                 remark = "";
 
             var newHTML = $(".select-list.active a.selected").html()+remark;
 
             $(".select-list.active").closest(".select-box").find(".value").html(newHTML);
             var id = $(".select-list.active").attr("id");
             id = id.replace("-ul","");
             $("#booking-widget input[name='"+id+"']").attr("value",$(".select-list.active a.selected").attr("value"));
 
             $(".select-list").removeClass("active");
             $("#reset-done-bar").removeClass("active");
             $("#search-bar").removeClass("active");
             $("body").removeClass("no-scroll");
 
         });
     });
 
     function updateTravelParty(){
         var adultsVal = parseInt($("#adults-wrapper .value").html());
         var childrenVal = parseInt($("#children-wrapper .value").html());
         var infantsVal = parseInt($("#infants-wrapper .value").html());
         $("#travelparty-select-box>div>.value").html(adultsVal + childrenVal + infantsVal + " Guests");
 
         $("#booking-widget input[name='AdultNo']").attr("value", adultsVal);
         $("#booking-widget input[name='ChildNo']").attr("value", childrenVal);
         $("#booking-widget input[name='InfantNo']").attr("value", infantsVal);
 
     }
 
     function subtractAdults(){
         var iniVal = parseInt($("#adults-wrapper .value").html());
         if(iniVal>0){
             $("#adults-wrapper .value").html(iniVal-1);
             $("#adults-wrapper select").val(iniVal-1);
         }
         updateTravelParty();
     }

     function subscribeNewsletter() {
 
         if ($("#subscribe-checkbox.checked").length == 0) {
                 
                 alert("Please accept the terms and conditions to continue");
                 
             return;
         }
 
         var name = $("#subscribe-form input[name='name']").val();
         var email = $("#subscribe-form input[name='email']").val();
         var region = $("#subscribe-form select[name='region']").val();
 
         if (name == "" || name == undefined || name == null || name == "Your Name") {
             alert("Please input name!");
             return false;
         }
         if (email == "" || email == undefined || email == null || email == "Your Email") {
             alert("Please input email!");
             return false;
         }
 
         var data = {
             name: name,
             email: email,
             region: region
         };
 
         $.ajax({
             url: "https://media.starcruises.com" + "/web/Subscribe/submit",
             type: 'post',
             dataType: 'json',
             success: function (data) {
                 if (data.success) {
                     alert("Success!");
                     $("#subscribe-form input[name='name']").val("");
                     $("#subscribe-form input[name='email']").val("");
                 } else {
                     alert("Failed! Try again later.");
                 }
             },
             data: data
         });
     }
