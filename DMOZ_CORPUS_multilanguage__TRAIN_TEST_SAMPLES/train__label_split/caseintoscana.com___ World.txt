 		to_w=w;
 		to_h = h;
 		}

 		var percent = (to_w/from_w); 
 		var to_h=(from_h*percent);
 		}
 	
 	$jQEmbed(this).find("embed").width(to_w);
 	$jQEmbed(this).find("embed").height(to_h);
 	});
 });
