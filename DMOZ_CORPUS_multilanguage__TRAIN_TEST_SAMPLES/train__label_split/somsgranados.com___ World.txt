            function showAvailableCheckinDatesSearch(date) {

                 return [(date.getDay() === 0 || date.getDay() === 1 || date.getDay() === 2 || date.getDay() === 3 || date.getDay() === 4 || date.getDay() === 5 || date.getDay() === 6)];
             }
 
             $(function() {
 
                 // set datepicker for checkin input search
                 $("#checkin_searchsearch").datepicker($.extend({
                         dateFormat: "dd/mm/yy",
                         changeMonth: true,
                         changeYear:true,
                         numberOfMonths: 1,
                         firstDay: 1,
                         minDate: 0,
 
                         beforeShowDay: showAvailableCheckinDatesSearch
                         ,                               
 
                         onClose: function(){
                             $('#checkout_searchsearch').datepicker("show");
                             $('#checkout_searchsearch').change();
                             $('#checkout_searchsearch').datepicker("hide");
                         }
                     },$.datepicker.regional['es'])
                 );
                 
                 // set datepicker for checkout input search
                 $("#checkout_searchsearch").datepicker($.extend({
                         dateFormat: "dd/mm/yy",
                         changeMonth: true,
                         changeYear:true,
                         numberOfMonths: 1,
                         firstDay: 1,
                         minDate: 0,
 
                         beforeShow: function(input) {
                             if($('#checkout_searchsearch').val()==='' || date_ddmmyyyy_yyyymmdd($('#checkout_searchsearch').val())<=date_ddmmyyyy_yyyymmdd($('#checkin_searchsearch').val())) {
                                 var checkin_date = $('#checkin_searchsearch').datepicker("getDate");
                                 if(checkin_date)
                                 {
                                     var minimum_nights=parseInt(2);
                                     if(minimum_nights<=0) { minimum_nights=1; }
                                     var checkin_date_date=new Date(checkin_date);
                                     var checkout_date=new Date(checkin_date_date.getFullYear(),checkin_date_date.getMonth(),checkin_date_date.getDate()+minimum_nights);
                                     $('#checkout_searchsearch').val($.datepicker.formatDate('dd/mm/yy', checkout_date));
                                     $(input).datepicker("option", 'minDate', checkout_date);
                                 }
                                 else
                                 {
                                     $(input).datepicker("option", 'minDate', 0);
                                 }
                             } else if($('#checkin_searchsearch').val()!=='') {
                                 var checkin_date = $('#checkin_searchsearch').datepicker("getDate");
                                 if(checkin_date)
                                 {
                                     var minimum_nights=parseInt(2);
                                     if(minimum_nights<=0) { minimum_nights=1; }
                                     var checkin_date_date=new Date(checkin_date);
                                     var checkout_date=new Date(checkin_date_date.getFullYear(),checkin_date_date.getMonth(),checkin_date_date.getDate()+minimum_nights);
                                     $(input).datepicker("option", 'minDate', checkout_date);
                                 }
                             }
 
                             return true;
                         }
                         ,
 
                         beforeShowDay: showAvailableCheckoutDatesSearch
                     },$.datepicker.regional['es'])
                 );
