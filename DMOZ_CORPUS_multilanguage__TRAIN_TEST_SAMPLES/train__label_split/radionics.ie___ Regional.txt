                 $row.each(function(){ if ($row.length <= 1) return; var $this = $(this);
                     var style = $this.attr('style');
                     var display = $this.css('display');
                     // temporarily force a usable display value
                     if (display !== 'inline-block' && display !== 'flex' && display !== 'inline-flex') display = 'block';
                     // ensure we get the correct actual height (and not a previously set height value)
                     $this.css( { 'display': display } );
                     // find the max height (including padding, but not margin)
                     if ($this.outerHeight(false) > targetHeight) targetHeight = $this.outerHeight(false);
                     // revert styles
                     if (style) { $this.attr('style', style); } 
                     else { $this.css('display', ''); }
                 })
                 .each(function(){ var $this = $(this);
                     // parsePxVal, value parse utility function, parse value and convert NaN to 0
                     var parsePxVal = function(value) { return parseFloat(value) || 0; };
                     var verticalPadding = 0;
                     // handle padding and border correctly (required when not using border-box)
                     if ($this.css('box-sizing') !== 'border-box') {
                         verticalPadding += parsePxVal($this.css('border-top-width')) + parsePxVal($this.css('border-bottom-width'));
                         verticalPadding += parsePxVal($this.css('padding-top')) + parsePxVal($this.css('padding-bottom'));
                     }
                     // set the height (accounting for padding and border)
                     $this.css('height', (targetHeight - verticalPadding) + 'px');
                 });        
             });
