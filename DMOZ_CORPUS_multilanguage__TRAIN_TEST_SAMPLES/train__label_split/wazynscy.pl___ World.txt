     jQuery(document).ready(function(){
 		initFrame();
 		
 		jQuery('#box-news-left').click(function(){
 			var frame = jQuery('#box-news-frame');
 			var act = jQuery('.active', frame);
 			var prev = act.prev();
 			if(prev.length>0) {
 				frame.animate({marginLeft: '+=170px'}, 400, function(){
 					act.removeClass('active');
 					prev.addClass('active');
 				});				
 			}
 		});
 			
 		jQuery('#box-news-right').click(function(){
 			var frame = jQuery('#box-news-frame');
 			var act = jQuery('.active', frame);
 			var next = act.next();
 			if(next.length>0) {
 				frame.animate({marginLeft: '-=170px'}, 400, function(){
 					act.removeClass('active');
 					next.addClass('active');
 				});				
 			}
 		});	
 	});
 	
 	function initFrame() {
 		var l = jQuery('#box-news-frame > div.news-box').length;
 		jQuery('#box-news-frame').css('width',l*170);
 	}
    </script>
