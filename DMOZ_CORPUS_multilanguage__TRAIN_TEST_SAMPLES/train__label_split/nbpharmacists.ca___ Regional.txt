               function addressAutocomplete(addressId) {
 	var timer = null;
 	$(":input[id='"+addressId+".addr1']").on('keyup',function () {
 		if (timer)
 			clearTimeout( timer );
 		timer = setTimeout( function(){
     		var addr1 = $(":input[id='"+addressId+".addr1']").val();
     		if (addr1) {
     			addressAutocompleteAjax(addressId, addr1, null);
     		} else {
 				var res = $("#_results".replace(/\./g, '\\.'));
 				res.html('');
     		}
 		}, 300 );
 		return true;
 	});
 }
 function addressAutocompleteAjax(addressId, addr1, lastId) {
 	var country = $(":input[id='"+addressId+".country']").val();
 	$.ajax( {
 		url: '/client/address/find.ajax',
 		contentType: "application/x-www-form-urlencoded;charset=UTF-8",
 		data: {
 			'addr1': addr1,
 			'country': country,
 			'lastId': lastId
 		},
 		success: function( addresses ) {
 			var res = $("div[id='"+addressId+"_results']");
 			res.html('');
 			$.each(addresses, function(i, addr) {
 				var elem = $( "<a>", {id: addr.id, class: "list-group-item", style: "cursor:pointer;"} );
 				if (!addr.canRetrieve) {
                     elem.addClass("list-group-item-info");
 					elem.html("<span class='fa fa-chevron-right'>&nbsp;</span>"+ addr.text + " (" + addr.numAddresses + " Addresses)" );
 					elem.click( function() {
 						addressAutocompleteAjax(addressId, addr1, addr.id);
 					} );
 				} else {
 					elem.text( addr.text + ", " + addr.city + ", " + addr.provinceCode + ", " + addr.postalCode );
 					elem.click( function() {
 						
 						$.ajax( {
             				url: '/client/address/retrieve.ajax',
             				contentType: "application/x-www-form-urlencoded;charset=UTF-8",
                 			data: {
                 				'id': addr.id
                 			},
                 			headers: {
                 				'addr1': addr.text,
                 				'city': addr.city,
                 				'province': addr.provinceCode,
                 				'postal': addr.postalCode,
                 				'country': country
                 			},
                 			success: function( address ) {
             					$(":input[id='"+addressId+".addr1']").val( addr.text );
             					$(":input[id='"+addressId+".city']").val( addr.city );
             					$(":input[id='"+addressId+".province']").val( addr.provinceCode );
             					$(":input[id='"+addressId+".postal']").val( addr.postalCode );
                 				res.html('');
                 			},
                 			error: function( jqXHR ) {
                 				var res = $("div[id='"+addressId+"_results']");
                 				res.html( $('<div>', { class: "alert alert-danger" } ).text( jqXHR.responseText ) );
                 			}
             			} );
         				
 						
 					} );
 				}
 				res.append( elem );
 			});
 		},
 		error: function( jqXHR ) {
 			var res = $("div[id='"+addressId+"_results']");
 			res.html( $('<div>', { class: "alert alert-danger" } ).text( jqXHR.responseText ) );
 		}
 	} );
 }
 function changeCountry(addressId, selectedValue) {
 	var prov = $(":input[id='"+addressId+".province']")[0];
 	var cntry = $(":input[id='"+addressId+".country']")[0];
 	if (prov && cntry) {

 		prov.options.length = 0;
 		$(prov).children('optgroup').remove();
 		var i=0;

              <script type="text/javascript">
               var geocoder = null;
 var markers = new Array();
 var addressUpdates = new Array();
 var infowindows = new Array();
 
 function geocodeAddress(address, addressId) {
 	if (!geocoder)
 		geocoder = new google.maps.Geocoder();
 	geocoder.geocode({
 		address: address
 	}, function(responses) {
 		if (responses && responses.length > 0) {
 			var marker = markers[addressId];
             if (typeof marker != 'undefined') {
                 marker.setPosition(responses[0].geometry.location);
                 updateLocation(addressId);
             }
 		}
 	});
 }
 
 function resetMarkerPosition(addressId) {
 	$(('#'+addressId+'.modified').replace(/\./g, '\\.')).val('false');
     $(('#reset_'+addressId).replace(/\./g, '\\.')).css('display', 'none');
     _updateMarkerPosition_(addressId);
     return false;
 }
 function updateMarkerPosition(addressId) {
 	if (addressUpdates[addressId]!=null)
 		clearTimeout(addressUpdates[addressId]);
 	addressUpdates[addressId] = setTimeout("_updateMarkerPosition_('"+addressId+"')", 1000);
 }
 function getAddressString(addressId) {
 	var fullAddr = document.getElementById(addressId+'.fullAddress');
 	if (fullAddr!=null) return fullAddr.value;
 
 	var addr1 = document.getElementById(addressId+'.addr1').value;
 	var addr2 = document.getElementById(addressId+'.addr2').value;
 	var city = document.getElementById(addressId+'.city').value;
 	var province = document.getElementById(addressId+'.province');
 	var provinceOther = document.getElementById(addressId+'.provinceOther').value;
 	var postal = document.getElementById(addressId+'.postal').value;
 	var country = document.getElementById(addressId+'.country');
 	if (((addr1!='' && city!='') || postal!='') && (province.value!='' || provinceOther!='') && country.value!='') {
 		var address = '';
 		if (addr1!='') {
 			address = addr1;
 			if (addr2!='')
 				address += ', '+addr2;
 		}
 		if (city!='')
 			address += ', '+city+', ';
 		if (province.value!='')
 			address += province[province.selectedIndex].text;
 		else
 			address += provinceOther;
 		if (postal!='')
 			address += ', '+postal;
 		address += ', '+country[country.selectedIndex].text;
 		return address;
 	}
 	return '';
 }
 function _updateMarkerPosition_(addressId) {
     var dateModified = document.getElementById(addressId+'.modified');
 	if (dateModified != null && dateModified.value=='false') {
 		var address = getAddressString(addressId);
 		if (address!='') {
 			var marker = markers[addressId];
 			geocodeAddress(address, addressId);
 		}
 	}
 }
 
 function updateLocation(addressId) {
 	hideInfoWindow(addressId);
 	var marker = markers[addressId];
 	var pos = marker.getPosition();
 	marker.getMap().setCenter(pos);
 	marker.setTitle(getAddressString(addressId));
 	$(('#'+addressId+'.latitude').replace(/\./g, '\\.')).val(pos.lat());
 	$(('#'+addressId+'.longitude').replace(/\./g, '\\.')).val(pos.lng());
 }
 
 function initAddressGoogleMap(addressId, selectedValue) {
 	if (!markers[addressId]) {
 		createAddressGoogleMap(addressId, selectedValue);
 	}
 }
 function createAddressGoogleMap(addressId, selectedValue) {
 	$(function() {
         _createAddressGoogleMap_( addressId, selectedValue );
 	});
 }
 function _createAddressGoogleMap_(addressId, selectedValue) {
 	var lat = $(('#'+addressId+'.latitude').replace(/\./g, '\\.')).val();
 	var lng = $(('#'+addressId+'.longitude').replace(/\./g, '\\.')).val();
 	var latLng = new google.maps.LatLng(lat, lng);
 	var map = new google.maps.Map(document.getElementById(addressId+'_map'), {
 		zoom: 15,
 		center: latLng,
 		mapTypeId: google.maps.MapTypeId.ROADMAP,
 		mapTypeControl: true,
 		mapTypeControlOptions: {
 			style: google.maps.MapTypeControlStyle.DROPDOWN_MENU
 		},
 		navigationControl: true,
 		navigationControlOptions: {
 			style: google.maps.NavigationControlStyle.SMALL,
 			position: google.maps.ControlPosition.RIGHT
 		}
 	});
 
 	google.maps.event.addListener(map, 'zoom_changed', function() {
 		hideInfoWindow(addressId);
 	});
 	
 	var marker = new google.maps.Marker({
 		position: latLng,
 		title: getAddressString(addressId),
 		map: map,
 		draggable: true
 	});
 	marker.addressId = addressId;
 	markers[marker.addressId] = marker;
 
 	google.maps.event.addListener(marker, "click", function() {
 		showInfoWindow(marker);
 	});
 
 	google.maps.event.addListener(marker, 'dragend', function() {
 		updateLocation(addressId);
 		$(('#'+addressId+'.modified').replace(/\./g, '\\.')).val('true');
 		$(('#reset_'+addressId).replace(/\./g, '\\.')).css('display', '');
 	});
 
 	changeCountry(addressId, selectedValue);
 }
 function createGoogleMapView(addressId, lat, lng, address) {
 	var latLng = new google.maps.LatLng(lat, lng);

 			style: google.maps.NavigationControlStyle.ZOOM_PAN,
 			position: google.maps.ControlPosition.TOP_LEFT

 	});
 	google.maps.event.addListener(map, 'zoom_changed', function() {
 		hideInfoWindow(''+addressId);

 	var marker = new google.maps.Marker({
 		position: latLng,
 		title: address,
 		map: map
 	});
 	google.maps.event.addListener(marker, "click", function() {
 		showInfoWindow(marker);
 	});
 	showInfoWindow(marker);
 }
 function hideInfoWindow(addressId) {
 	if (infowindows[addressId] != null) {
 		infowindows[addressId].close();
 		infowindows[addressId] = null;
 	}
 }
 function showInfoWindow(marker) {
 	hideInfoWindow(marker.addressId);
 	infowindow = new google.maps.InfoWindow({content: marker.getTitle()});
 	infowindow.open(marker.getMap(), marker);
 	infowindows[marker.addressId] = infowindow;
 }

 function addressAutocompleteAjax(addressId, addr1, lastId) {
 	var country = $(":input[id='"+addressId+".country']").val();
 	$.ajax( {
 		url: '/client/address/find.ajax',
 		contentType: "application/x-www-form-urlencoded;charset=UTF-8",
 		data: {
 			'addr1': addr1,
 			'country': country,
 			'lastId': lastId
 		},
 		success: function( addresses ) {
 			var res = $("div[id='"+addressId+"_results']");
 			res.html('');
 			$.each(addresses, function(i, addr) {
 				var elem = $( "<a>", {id: addr.id, class: "list-group-item", style: "cursor:pointer;"} );
 				if (!addr.canRetrieve) {
                     elem.addClass("list-group-item-info");
 					elem.html("<span class='fa fa-chevron-right'>&nbsp;</span>"+ addr.text + " (" + addr.numAddresses + " Addresses)" );
 					elem.click( function() {
 						addressAutocompleteAjax(addressId, addr1, addr.id);
 					} );
 				} else {
 					elem.text( addr.text + ", " + addr.city + ", " + addr.provinceCode + ", " + addr.postalCode );
 					elem.click( function() {
 						
 						$.ajax( {
             				url: '/client/address/retrieve.ajax',
             				contentType: "application/x-www-form-urlencoded;charset=UTF-8",
                 			data: {
                 				'id': addr.id
                 			},
                 			headers: {
                 				'addr1': addr.text,
                 				'city': addr.city,

                 				'postal': addr.postalCode,
                 				'country': country
                 			},
                 			success: function( address ) {
             					$(":input[id='"+addressId+".addr1']").val( addr.text );
             					$(":input[id='"+addressId+".city']").val( addr.city );
             					$(":input[id='"+addressId+".province']").val( addr.provinceCode );

                 				var res = $("div[id='"+addressId+"_results']");

 		var provOther = document.getElementById(addressId+'.provinceOther');

 			prov.parentNode.parentNode.style.display = 'none';

 	if (((addr1!='' && city!='') || postal!='') && (province.value!='' || provinceOther!='') && country.value!='') {
 		var address = '';
 		if (addr1!='') {
 			address = addr1;
 			if (addr2!='')
 				address += ', '+addr2;
 		}
 		if (city!='')
 			address += ', '+city+', ';
 		if (province.value!='')
 			address += province[province.selectedIndex].text;
 		else
 			address += provinceOther;
 		if (postal!='')
 			address += ', '+postal;
 		address += ', '+country[country.selectedIndex].text;
 		return address;
 	}
 	return '';
 }
 function _updateMarkerPosition_(addressId) {
     var dateModified = document.getElementById(addressId+'.modified');
 	if (dateModified != null && dateModified.value=='false') {
 		var address = getAddressString(addressId);
 		if (address!='') {
 			var marker = markers[addressId];
 			geocodeAddress(address, addressId);
 		}
 	}
 }
 
 function updateLocation(addressId) {
 	hideInfoWindow(addressId);
 	var marker = markers[addressId];
 	var pos = marker.getPosition();
 	marker.getMap().setCenter(pos);
 	marker.setTitle(getAddressString(addressId));
 	$(('#'+addressId+'.latitude').replace(/\./g, '\\.')).val(pos.lat());
 	$(('#'+addressId+'.longitude').replace(/\./g, '\\.')).val(pos.lng());
 }
 
 function initAddressGoogleMap(addressId, selectedValue) {
 	if (!markers[addressId]) {
 		createAddressGoogleMap(addressId, selectedValue);
 	}
 }
 function createAddressGoogleMap(addressId, selectedValue) {
 	$(function() {
         _createAddressGoogleMap_( addressId, selectedValue );
 	});
 }
 function _createAddressGoogleMap_(addressId, selectedValue) {
 	var lat = $(('#'+addressId+'.latitude').replace(/\./g, '\\.')).val();
 	var lng = $(('#'+addressId+'.longitude').replace(/\./g, '\\.')).val();
 	var latLng = new google.maps.LatLng(lat, lng);
 	var map = new google.maps.Map(document.getElementById(addressId+'_map'), {
 		zoom: 15,
 		center: latLng,
 		mapTypeId: google.maps.MapTypeId.ROADMAP,
 		mapTypeControl: true,
 		mapTypeControlOptions: {
 			style: google.maps.MapTypeControlStyle.DROPDOWN_MENU
 		},
 		navigationControl: true,
 		navigationControlOptions: {
 			style: google.maps.NavigationControlStyle.SMALL,
 			position: google.maps.ControlPosition.RIGHT
 		}
 	});
 
 	google.maps.event.addListener(map, 'zoom_changed', function() {
 		hideInfoWindow(addressId);
 	});
 	
 	var marker = new google.maps.Marker({
 		position: latLng,
 		title: getAddressString(addressId),
 		map: map,
 		draggable: true
 	});
 	marker.addressId = addressId;
 	markers[marker.addressId] = marker;
 
 	google.maps.event.addListener(marker, "click", function() {
 		showInfoWindow(marker);
 	});
 
 	google.maps.event.addListener(marker, 'dragend', function() {
 		updateLocation(addressId);
 		$(('#'+addressId+'.modified').replace(/\./g, '\\.')).val('true');
 		$(('#reset_'+addressId).replace(/\./g, '\\.')).css('display', '');
 	});
 
 	changeCountry(addressId, selectedValue);
 }
 function createGoogleMapView(addressId, lat, lng, address) {
 	var latLng = new google.maps.LatLng(lat, lng);
 	var map = new google.maps.Map(document.getElementById(addressId+'_map'), {
 		zoom: 15,
 		center: latLng,
 		mapTypeId: google.maps.MapTypeId.ROADMAP,
 		mapTypeControl: true,
 		mapTypeControlOptions: {
 			style: google.maps.MapTypeControlStyle.DROPDOWN_MENU
 		},
 		navigationControl: true,
 		navigationControlOptions: {
 			style: google.maps.NavigationControlStyle.ZOOM_PAN,
 			position: google.maps.ControlPosition.TOP_LEFT
 		}
 	});
 	google.maps.event.addListener(map, 'zoom_changed', function() {
 		hideInfoWindow(''+addressId);
 	});
 	var marker = new google.maps.Marker({
 		position: latLng,
 		title: address,
 		map: map
 	});
 	google.maps.event.addListener(marker, "click", function() {
 		showInfoWindow(marker);
 	});
 	showInfoWindow(marker);
 }
 function hideInfoWindow(addressId) {
 	if (infowindows[addressId] != null) {
 		infowindows[addressId].close();
 		infowindows[addressId] = null;
 	}
 }
 function showInfoWindow(marker) {
 	hideInfoWindow(marker.addressId);
 	infowindow = new google.maps.InfoWindow({content: marker.getTitle()});
 	infowindow.open(marker.getMap(), marker);
 	infowindows[marker.addressId] = infowindow;
 }
