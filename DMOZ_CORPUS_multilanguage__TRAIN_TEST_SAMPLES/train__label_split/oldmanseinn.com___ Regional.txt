         jQuery( document ).ready(function() {
 			
 						// every time there is a click in the form, calculate and verify the checkout date
 						jQuery("form.availability *").click(function() {
 				
 							var arriving = jQuery("#StartDate").val();
 							var nights = jQuery("#NumNights").val();
 							var numNights = parseInt(nights);
 					
 							// extract month, day, year
 							var date = arriving.split("/");
 					
 							var year = date[2];
 							var month = parseInt(date[0], 10) - 1;
 							var day = date[1];
 					
 							var arrive = new Date(year, month, day);
 					
 							var newdate = new Date(arrive);
 							newdate.setDate(newdate.getDate() + numNights);
 
 							var nd = new Date(newdate);
 
 							var checkOut = (nd.getMonth() + 1) + "/" + nd.getDate() + "/" +  nd.getFullYear();
 					
 							jQuery("#EndDate").val(checkOut);
 					
 						});
