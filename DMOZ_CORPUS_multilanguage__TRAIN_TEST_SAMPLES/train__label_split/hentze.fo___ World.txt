   <script>
    jQuery(document).ready(function($) {
 
     var ResponsiveMenu = {
         trigger: '#responsive-menu-button',
         animationSpeed: 500,
         breakpoint: 8000,
         pushButton: 'off',
         animationType: 'slide',
         animationSide: 'left',
         pageWrapper: '',
         isOpen: false,
         triggerTypes: 'click',
         activeClass: 'is-active',
         container: '#responsive-menu-container',
         openClass: 'responsive-menu-open',
         accordion: 'off',
         activeArrow: '▲',
         inactiveArrow: '▼',
         wrapper: '#responsive-menu-wrapper',
         closeOnBodyClick: 'off',
         closeOnLinkClick: 'off',
         itemTriggerSubMenu: 'off',
         linkElement: '.responsive-menu-item-link',
         subMenuTransitionTime: 200,
         openMenu: function() {
             $(this.trigger).addClass(this.activeClass);
             $('html').addClass(this.openClass);
             $('.responsive-menu-button-icon-active').hide();
             $('.responsive-menu-button-icon-inactive').show();
             this.setButtonTextOpen();
             this.setWrapperTranslate();
             this.isOpen = true;
         },
         closeMenu: function() {
             $(this.trigger).removeClass(this.activeClass);
             $('html').removeClass(this.openClass);
             $('.responsive-menu-button-icon-inactive').hide();
             $('.responsive-menu-button-icon-active').show();
             this.setButtonText();
             this.clearWrapperTranslate();
             this.isOpen = false;
         },
         setButtonText: function() {
             if($('.responsive-menu-button-text-open').length > 0 && $('.responsive-menu-button-text').length > 0) {
                 $('.responsive-menu-button-text-open').hide();
                 $('.responsive-menu-button-text').show();
             }
         },
         setButtonTextOpen: function() {
             if($('.responsive-menu-button-text').length > 0 && $('.responsive-menu-button-text-open').length > 0) {
                 $('.responsive-menu-button-text').hide();
                 $('.responsive-menu-button-text-open').show();
             }
         },
         triggerMenu: function() {
             this.isOpen ? this.closeMenu() : this.openMenu();
         },
         triggerSubArrow: function(subarrow) {
             var sub_menu = $(subarrow).parent().siblings('.responsive-menu-submenu');
             var self = this;
             if(this.accordion == 'on') {
                 /* Get Top Most Parent and the siblings */
                 var top_siblings = sub_menu.parents('.responsive-menu-item-has-children').last().siblings('.responsive-menu-item-has-children');
                 var first_siblings = sub_menu.parents('.responsive-menu-item-has-children').first().siblings('.responsive-menu-item-has-children');
                 /* Close up just the top level parents to key the rest as it was */
                 top_siblings.children('.responsive-menu-submenu').slideUp(self.subMenuTransitionTime, 'linear').removeClass('responsive-menu-submenu-open');
                 /* Set each parent arrow to inactive */
                 top_siblings.each(function() {
                     $(this).find('.responsive-menu-subarrow').first().html(self.inactiveArrow);
                     $(this).find('.responsive-menu-subarrow').first().removeClass('responsive-menu-subarrow-active');
                 });
                 /* Now Repeat for the current item siblings */
                 first_siblings.children('.responsive-menu-submenu').slideUp(self.subMenuTransitionTime, 'linear').removeClass('responsive-menu-submenu-open');
                 first_siblings.each(function() {
                     $(this).find('.responsive-menu-subarrow').first().html(self.inactiveArrow);
                     $(this).find('.responsive-menu-subarrow').first().removeClass('responsive-menu-subarrow-active');
                 });
             }
             if(sub_menu.hasClass('responsive-menu-submenu-open')) {
                 sub_menu.slideUp(self.subMenuTransitionTime, 'linear').removeClass('responsive-menu-submenu-open');
                 $(subarrow).html(this.inactiveArrow);
                 $(subarrow).removeClass('responsive-menu-subarrow-active');
             } else {
                 sub_menu.slideDown(self.subMenuTransitionTime, 'linear').addClass('responsive-menu-submenu-open');
                 $(subarrow).html(this.activeArrow);
                 $(subarrow).addClass('responsive-menu-subarrow-active');
             }
         },
         menuHeight: function() {
             return $(this.container).height();
         },
         menuWidth: function() {
             return $(this.container).width();
         },
         wrapperHeight: function() {
             return $(this.wrapper).height();
         },
         setWrapperTranslate: function() {
             switch(this.animationSide) {
                 case 'left':
                     translate = 'translateX(' + this.menuWidth() + 'px)'; break;
                 case 'right':
                     translate = 'translateX(-' + this.menuWidth() + 'px)'; break;
                 case 'top':
                     translate = 'translateY(' + this.wrapperHeight() + 'px)'; break;
                 case 'bottom':
                     translate = 'translateY(-' + this.menuHeight() + 'px)'; break;
             }
             if(this.animationType == 'push') {
                 $(this.pageWrapper).css({'transform':translate});
                 $('html, body').css('overflow-x', 'hidden');
             }
             if(this.pushButton == 'on') {
                 $('#responsive-menu-button').css({'transform':translate});
             }
         },
         clearWrapperTranslate: function() {
             var self = this;
             if(this.animationType == 'push') {
                 $(this.pageWrapper).css({'transform':''});
                 setTimeout(function() {
                     $('html, body').css('overflow-x', '');
                 }, self.animationSpeed);
             }
             if(this.pushButton == 'on') {
                 $('#responsive-menu-button').css({'transform':''});
             }
         },
         init: function() {
             var self = this;
             $(this.trigger).on(this.triggerTypes, function(e){
                 e.stopPropagation();
                 self.triggerMenu();
             });
             $(this.trigger).mouseup(function(){
                 $(self.trigger).blur();
             });
             $('.responsive-menu-subarrow').on('click', function(e) {
                 e.preventDefault();
                 e.stopPropagation();
                 self.triggerSubArrow(this);
             });
             $(window).resize(function() {
                 if($(window).width() > self.breakpoint) {
                     if(self.isOpen){
                         self.closeMenu();
                     }
                 } else {
                     if($('.responsive-menu-open').length>0){
                         self.setWrapperTranslate();
                     }
                 }
             });
             if(this.closeOnLinkClick == 'on') {
                 $(this.linkElement).on('click', function(e) {
                     e.preventDefault();
                     /* Fix for when close menu on parent clicks is on */
                     if(self.itemTriggerSubMenu == 'on' && $(this).is('.responsive-menu-item-has-children > ' + self.linkElement)) {
                         return;
                     }
                     old_href = $(this).attr('href');
                     old_target = typeof $(this).attr('target') == 'undefined' ? '_self' : $(this).attr('target');
                     if(self.isOpen) {
                         if($(e.target).closest('.responsive-menu-subarrow').length) {
                             return;
                         }
                         self.closeMenu();
                         setTimeout(function() {
                             window.open(old_href, old_target);
                         }, self.animationSpeed);
                     }
                 });
             }
             if(this.closeOnBodyClick == 'on') {
                 $(document).on('click', 'body', function(e) {
                     if(self.isOpen) {
                         if($(e.target).closest('#responsive-menu-container').length || $(e.target).closest('#responsive-menu-button').length) {
                             return;
                         }
                     }
                     self.closeMenu();
                 });
             }
             if(this.itemTriggerSubMenu == 'on') {
                 $('.responsive-menu-item-has-children > ' + this.linkElement).on('click', function(e) {
                     e.preventDefault();
                     self.triggerSubArrow($(this).children('.responsive-menu-subarrow').first());
                 });
             }
         }
     };
     ResponsiveMenu.init();
 });
   </script>

        <header class="entry-header">
         <h1 class="entry-title">
          Konstruktiv trætuloysn
         </h1>
        </header>
        <!-- .entry-header -->
        <div class="entry-content">
         <div class="panel-layout" id="pl-2618">
          <div class="panel-grid panel-no-style" id="pg-2618-0">
           <div class="panel-grid-cell" id="pgc-2618-0-0">
            <div class="so-panel widget widget_sow-editor panel-first-child panel-last-child" data-index="0" id="panel-2618-0-0-0">
             <div class="so-widget-sow-editor so-widget-sow-editor-base">
              <div class="siteorigin-widget-tinymce textwidget">
               <figure class="wp-caption alignleft" id="attachment_1668" style="width: 235px">
                <img alt="55ade4bf1519764a21efd036d48e8e4b" class="wp-image-1668" data-attachment-id="1668" data-comments-opened="1" data-image-description="" data-image-meta='{"aperture":"0","credit":"","camera":"","caption":"","created_timestamp":"0","copyright":"","focal_length":"0","iso":"0","shutter_speed":"0","title":"","orientation":"0"}' data-image-title="55ade4bf1519764a21efd036d48e8e4b" data-large-file="https://i2.wp.com/hentze.fo/wp-content/uploads/2018/04/55ade4bf1519764a21efd036d48e8e4b-3970882180-1526599043508.jpg?fit=179%2C202&amp;ssl=1" data-medium-file="https://i2.wp.com/hentze.fo/wp-content/uploads/2018/04/55ade4bf1519764a21efd036d48e8e4b-3970882180-1526599043508.jpg?fit=179%2C202&amp;ssl=1" data-orig-file="https://i2.wp.com/hentze.fo/wp-content/uploads/2018/04/55ade4bf1519764a21efd036d48e8e4b-3970882180-1526599043508.jpg?fit=179%2C202&amp;ssl=1" data-orig-size="179,202" data-permalink="https://hentze.fo/55ade4bf1519764a21efd036d48e8e4b/" data-recalc-dims="1" height="265" src="https://i0.wp.com/konfliktmekling.fo/wp-content/uploads/2018/04/55ade4bf1519764a21efd036d48e8e4b-3970882180-1526599043508.jpg?resize=235%2C265&amp;ssl=1" width="235"/>
                <figcaption class="wp-caption-text">
                 Advokatur og meklari
                </figcaption>
               </figure>
               <p>
                Ein træta er loyst konstruktivt, tá allir partar í trætuni eru nøgdir við úrslitið.  Ein slík loysn krevur, at partarnir sjálvir leggja orku í at finna loysnina og eru sinnaðir at luttaka í gongdini allan vegin fram til loysnina.
               </p>
               <p>
                Hvør einstakur ynskir ofta eitt ávist úrslit fyri seg sjálvan - men er áhugi at varveita eitt gott samband ella samskifti við hin,  so er ikki er nóg mikið bert við eini góðari loysn fyri ein sjálvan -  loysnin skal eisini kennast fullgóð fyri hin partin.
               </p>
               <p>
                Slík konstruktiv trætuhandfaring krevur, at tosað verður saman.
                <a href="https://konfliktmekling.fo/hvat-er-konfliktmekling/">
                 Hetta er tað mekling er um
                </a>
                :  at fáa eina byrjandi ella fastlasta trætu aftur á beint. Ofta er fyrsta stigið at fá samband aftur millum partarnar. Byrjanin kann vera trupul, men gerst munandi lættari, tá uttanveltaður persóni, átekur sær uppgávuna at fáa í lag aftur mistu kontaktina,  hjálpir dialoginum á glið og heldur hesum á sporinum.
               </p>
               <p>
                Mekling er ein dialogur, har focus ikki er á at fastleggja, hvør hevur rætt ella ikki, men har dentur verður lagdur á eisini at hyggja afturum verandi íkomnu støðuna og síðani at leita frameftir eftir loysnum. Kensluligar avbjóðingar sum rættvísi og tørvandi viðurkenning eru ofta eisini partur av gongdini.
               </p>
               <p>
                Meklingin brýtur m.a. við rættarmálsførslu á hesum øki: rættarmálsførsla snýr seg um at staðfesta rættvísi(rætt ella skeivt) í mun til rættarreglur, viðgerðin snýr seg nakað sum farið er, lítur aftur í tíð og. Viðgerð av kenslum og samskifti annars er ikki partur av rættarmálsførslu. Rættarmálsførsla er ógvuliga kostnaðarmikil bæði í tíð, pengum og mentalari útlúgvan. Flestu meklingsfundir hinvegin enda við einum positivum úrslitið aftaná 2-3 fundir á 3 tímar og kostnaðurin er samsvarandi hesum.
               </p>
               <h3>
                <strong>
                 Advokatyrki
                </strong>
               </h3>
               <p>
                Yrkisøki míni sum advokatur fevna um rættarmálsførslu, revsimál og borgarlig mál, umframt løgfrøðisliga ráðgeving og gerð av skjølum knýtt at hesum. Løgfrøðilig  ráðgeving snýr seg eisini um at finna loysnir og semjur - also samráðing.
               </p>
               <p>
                Eg havi virkað sum advokatur siðani 1982, og uppgávurnar hava verið nógvar og ymiskar.
               </p>
               <p>
                Ofta hava uppgávurnar snúð seg um seg um eitt av kjarnuøkjunum innan advokatarbeiði, nevnliga  ognarviðurskiftir og broytingar í hesum : Keyp/søla, samogn, arvaviðurskifti og hjúnabandsskifti, rættindi og skyldur í samskiftum við onnur og gerð av neyðugum skjølum:
               </p>
               <p>
                -Skeyti: ráðgeving og umboðan í samband við keyp og sølu av fastogn, og kunning um tey rættindi/skyldur og tær fellur, sum har eru.
                <br/>
                -Arvur/samogn: Samognarspurningar, tá fólk eiga ognir saman, t.d. arva ella keypa ogn  saman og evt. upploysn av samogn. Samognarsáttmálar. Knýtt at hesum er ofta eisini testamentsviðgerð.
                <br/>
                -Hjúnabandsskifti: Ein neyðug fylgja við hjúnaskilnaði er broyting í ognarviðurskiftum - ein fyri partarnar stór avbjóðing, sum avgjørt krevur fakligan kunnleika, um ein vil sær sjálvum væl.
               </p>
               <h3>
                <strong>
                 Mekling
                </strong>
               </h3>
               <p>
                Í virki mínum sum advokatur, gjørdist mær støðugt meiri greitt, at handfaringin av trætum, ósemjum er knýtt at øðrum enn bert løgfrøðisligum reglum, og at einsporaði rættarvegurin oftast er ein neyðloysn, heldur enn ein góð loysn av eini trætu, konflikt.
               </p>
               <p>
                Í 2004 leyk eg prógv sum konfliktmeklari við Kbh. Universitet (master of mediation and conflict resoulution). Við tí førleikanum í sjáttuni havi eg lagt meg eftir í samskiftinum við viðskiftafólk míni eisini at lýta aftur um løgfrøðiligu trætuna. Og ofta er tað so at trætan ikki byrjaði sum ein løgfrøðilig træta um at "fáa rætt" , men gjørdist tað, tá samtalan steðgaði, tí er tann góða og varandi loysnin ofta at finna við at leita fram aftur dialogin.
               </p>
               <p>
                Havi við positivum úrlitið, sum konfliktmeklari havt nógvar ymiskar uppgávur. Fyri  arbeiðspláss: millum starvsfólk, millum starvsfólk og leiðara, sum einstaklingar og bólkar, og millum leiðarar. Innan familjurættarliga økið: ósemjur millum foreldur og børn, millum foreldur um børn, millum fosturforeldur og foreldur vm.
               </p>
               <h3>
                <strong>
                 Konstrukttiv konflikthandfaring
                </strong>
               </h3>
               <p>
                Mekling eigur altíð at roynast áðrenn farið verður víðari í juridiska sporinum alleina.
               </p>

              <h3 class="widget-title">
               Rættarvegurin - tú letur annan taka avgerð fyri teg
              </h3>
