 			var scrolltop = $(window).scrollTop();
 			if ($(window).width() < 767) {
 				if (scrolltop >= 70) {
 					$('#tray').addClass('up');
 				} else {
 					$('#tray').removeClass('up');
 				}
 			} else {
 				$('#tray').removeClass('up');
 			}
 		});

 				$(this).children('.mobile-trigger-inner').addClass('opened');
 				
 				$('.mobile-main').addClass('open');
 				$('html').addClass('disable-scrolling');

 			} else {

 				$(this).children('.mobile-trigger-inner').addClass('closed');
 				
 				$('.mobile-main').removeClass('open');
 				$('html').removeClass('disable-scrolling');
 				
 			}

      function topHeight() {
 			var windowHeight = $(window).outerHeight();
 			var navHeight = $('#nav').outerHeight();
 			var topLinks = $('#top-links').outerHeight();
 			var topImageHeight = (windowHeight - navHeight) - topLinks;
 			$('#top-image').css('height', topImageHeight+'px');
 		}
 		
 				topHeight();
 		
 				$( window ).resize(function() {
 			topHeight();
 		});
 		
 				lastScrollTop = 0;
 		keyBlock = 0;
 		$(window).on('scroll', function(){
 			if ($(window).width() > 767) {
 				var topImageHeight = $('#top-image').outerHeight() / 2;
 				var scrollTop = $(window).scrollTop();
 				if (scrollTop >= lastScrollTop){
 					if (scrollTop <= topImageHeight && keyBlock == 0) {
 						$(document).bind('mousewheel scrollstart touchmove touchstart', function(e) {
 							e.preventDefault();
 						});
 						keyBlock = 1;
 						$('html, body').animate({
 					        scrollTop: $("#content-block").offset().top - 50
 					    }, 1000, function() {
 						    keyBlock = 0;
 						    $(document).unbind('mousewheel scrollstart touchmove touchstart');
 					    });
 					}
 				}
 				lastScrollTop = scrollTop;
 			}	
 		});
 		
 				var $container =  $('#isotope').isotope({
 			layoutMode: 'packery',
 			packery: {
 				gutter: 0,
 			},
 			itemSelector: '.item'
 		});
 		
 				$('.item').on('click touch', function() {
 			var selectedCollection = $(this).data('collection');
 			var entryUrl = 'https://www.mckinleymasters.com/';
 			$('.loading').fadeIn(250);
 			$.ajax({
 				type:"GET",
 				url: "gallery-modal.html",
 				data: { collection: selectedCollection, entry: entryUrl },
 				context: document.body,
 				success: function(data) {
 					//Add data to correct tile
 					$('#gallery-modal').html(data);
 					$('.loading').fadeOut(500);
 				},
 				error: function() {
 					console.log('error');
 				}
 			});
 		});
