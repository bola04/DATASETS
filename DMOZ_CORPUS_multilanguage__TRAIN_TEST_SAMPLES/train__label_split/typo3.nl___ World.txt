    document.onreadystatechange = function () {
       if (document.readyState == "interactive") {
         if (typeof contentTeaserNavigationLinks === 'undefined') {
 
           // Register variables
           var contentTeaserNavigationLinks = document.getElementsByClassName('contentteaser-navigation-link');
           var contentTeaserNaviationTimer = 13000;
           var contentTeaserNavigationInterval = 250;
           var contentTeaserNavigationProgress = 100 / (contentTeaserNaviationTimer / contentTeaserNavigationInterval);
 
           // Add click-handler for changing active states
           for (i = 0; i < contentTeaserNavigationLinks.length; i++) {
             contentTeaserNavigationLinks[i].addEventListener(
               'click',
               function (event) {
                 event.preventDefault();
                 navigationItems = this.parentNode.parentNode.childNodes;
                 for (ni = 0; ni < navigationItems.length; ni++) {
                   navigationItems[ni].classList.remove('active');
                   navigationItems[ni].childNodes[0].setAttribute('data-progress', parseInt(0));
                   navigationItems[ni].childNodes[0].childNodes[0].childNodes[0].childNodes[1].style.strokeDashoffset = 2 * Math.PI * 54;
                 }
                 this.parentNode.classList.add('active');
                 target = document.getElementById(this.hash.slice(1));
                 contentItems = target.parentNode.childNodes;
                 for (ci = 0; ci < contentItems.length; ci++) {
                   contentItems[ci].classList.remove('active');
                 }
                 target.classList.add('active');
               }
             );
           }
 
           // Register autoplay
           var contentTeaserNavigationInterval = setInterval(function(){
             for (i = 0; i < contentTeaserNavigationLinks.length; i++) {
               if (contentTeaserNavigationLinks[i].parentNode.className.indexOf('active') != -1) {
                 var currentItem = contentTeaserNavigationLinks[i]
                 var currentParent = currentItem.parentNode;
                 var currentSpinner = currentParent.childNodes[0].childNodes[0].childNodes[0];
                 var currentSpinnerValue = currentSpinner.childNodes[1];
                 var currentProgress = parseInt(currentItem.getAttribute('data-progress'), 10);
 
                 // Progress done, find next element
                 if (currentProgress == 100) {
                   var nextElement = undefined;
                   if (currentParent.nextElementSibling) {
                     nextElement = currentParent.nextElementSibling.childNodes[0];
                   }
                   if (nextElement === undefined) {
                     nextElement = currentParent.parentNode.childNodes[0].childNodes[0];
                   }
                   nextElement.click();
                 }
 
                 // Set new Progress
                 var newProgress = parseInt(currentProgress + contentTeaserNavigationProgress);
                 if (newProgress > 95) {
                   newProgress = 100;
                 }
                 currentItem.setAttribute('data-progress', parseInt(newProgress));
 
                 // Visualize Progress
                 var progress = newProgress / 100;
                 var strokeDashoffset = 2 * Math.PI * 54 * (1 - progress);
                 currentSpinnerValue.style.strokeDashoffset = strokeDashoffset;
               }
             }
           }, contentTeaserNavigationInterval);
 
         }
       };
     }
