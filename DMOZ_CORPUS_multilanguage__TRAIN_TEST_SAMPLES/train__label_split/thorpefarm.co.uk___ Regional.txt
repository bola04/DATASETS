             theForm = $(this).parents("form");
             var error_list = "";
 
             var date = $('input[name=start]',theForm).val();
             if (date == "") {
                 error_list += "Choosing an arrival date\n";
                 $('#bbi1',theForm).addClass("form_error");
             } else {
                 $('#bbi1',theForm).removeClass("form_error");
             }
 
             var nights = $('#bbi2',theForm).val();
             if (nights == "") {
                 error_list += "Choosing the number of nights\n";
                 $('#bbi2',theForm).addClass("form_error");
             } else {
                 $('#bbi2',theForm).removeClass("form_error");
             }
 
             var adult = $('#bbi3',theForm).val();
             if (adult == "") {
                 error_list += "Choosing the number of adults\n";
                 $('#bbi3',theForm).addClass("form_error");
             } else {
                 $('#bbi3',theForm).removeClass("form_error");
             }
 
             if (error_list) {
                 alert("Please complete the availability check by:\n" + error_list)
                 return false;
             }
 
             window.open(constructURL(theForm), '_check');
         });

     function constructURL(theForm){
 
         var url = "#";
 
         try {
             var parkID = '54f1b131-5ee9-4469-94ea-cf65125b6ef1';
 
             var type = $('#bbi0',theForm).val();
 
             if (!type) {
                 type = $("input[name='type']:checked").val();
             }
 
             var pets = $('#pets',theForm).val();
             var range = $('#range',theForm).val();
 
 
             var date = $('input[name=start]',theForm).val();
             var nights = $('#bbi2',theForm).val();
             var adult = $('#bbi3',theForm).val();
             var child = $('#bbi4',theForm).val();
             var infant = $('#bbi5',theForm).val();
 
             if(type == "home"){
                 url = 'http://www.hoseasons.co.uk/holiday-parks/thorpe-farm-park-thfp?'+
                 'start='+date+'&range='+range+'&nights='+nights+'&adult='+adult+'&child='+child+'&infant='+infant+'&pets='+pets;
             }else{
                 var splitDate = date.split("-");
                 var endDate = new Date(splitDate[2],splitDate[1]-1,splitDate[0]);
                 endDate.setDate(endDate.getDate() + parseInt(nights));
                 url = 'http://www.parkbreaks.com/2/uk/Holidays/Search/Parks/Availability/?ParkID=54f1b131-5ee9-4469-94ea-cf65125b6ef1'+
                 '&SDay='+splitDate[0]+'&SMonth='+splitDate[1]+'&SYear='+splitDate[2]+
                 '&EDay='+endDate.getDate()+'&EMonth='+(endDate.getMonth()+1)+'&EYear='+endDate.getFullYear()+
                 '&AdultsN='+adult+'&ChildrenN='+child+'&InfantsN='+infant+'&PetsN='+pets+'&Type='+type;
             }
         } catch(err) {
 
         }
 
         return url;
     }
