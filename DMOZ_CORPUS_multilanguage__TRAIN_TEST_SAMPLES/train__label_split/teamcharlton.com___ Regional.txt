 					maxHeight = 0;

 				for (var i=0;i<$testimonials.length;i++){
 					var $gadget = $('<div data-idx="'+i+'"></div>').on('click',function(evt){
 						rotateTestimonials($(evt.target).data('idx'))
 					});
 					if (i===0){$gadget.addClass('selected')}
 					$('.testimonials-scroller-gadgets').append($gadget)
 				}
 
 				var totalTestimonials = $('.testimonial-review-box').length;
 				var rotateIdx = 0;
 				var rotateTmr = setTimeout(rotateTestimonials, 6000);
 				function rotateTestimonials(newIdx){
 					$($('.testimonial-review-box')[rotateIdx]).removeClass('showing')
 					$($('.testimonials-scroller-gadgets div')[rotateIdx]).removeClass('selected');
 					if (newIdx !== undefined){
 						rotateIdx = newIdx;
 					} else {
 						rotateIdx++
 					}
 					if (rotateIdx>totalTestimonials-1){
 						rotateIdx = 0;
 					}
 					$($('.testimonial-review-box')[rotateIdx]).addClass('showing')
 					$($('.testimonials-scroller-gadgets div')[rotateIdx]).addClass('selected');
 					if (rotateTmr){clearTimeout(rotateTmr)}
 					rotateTmr = setTimeout(rotateTestimonials, 6000);
