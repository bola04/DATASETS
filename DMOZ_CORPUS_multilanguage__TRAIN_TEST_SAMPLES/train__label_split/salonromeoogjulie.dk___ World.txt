         function considerTeasers() {
             var widthTeaser = Math.ceil($(".teaser").last().width());
             var widthMain = Math.floor($("#main").width());
             var hm = $("#main").height();
             var hh = $("#header").height();
             var hc = $("#ContentDiv").height();
             if (hh + hc > hm) {                
                 $("#header").addClass("borderCarrier");
                 $("#ContentDiv").addClass("borderCarrier");
                 $("#main").removeClass("borderCarrier");
             } else {
                 $("#header").removeClass("borderCarrier");
                 $("#ContentDiv").removeClass("borderCarrier");
                 $("#main").addClass("borderCarrier");                
             }
             if (widthMain > 885) {
                 // --- restore css
                 $(".teaser").attr("style", "");
                 $(".teaser").css("border-top", "0");
                 $(".teaserinner").css("width", "281px");
 
                 $(".columngrayed").attr("style", "");
                 $(".columnactive").attr("style", "");
                 // --- end restore section
 
                 var available = widthMain - (3 * widthTeaser);
                 if (available > 4) {
                     var pad = parseInt(Math.floor((available / 4)));
                     $(".teaser").css("padding-top", parseInt(pad / 2) + "px");
                     $(".teaser").css("padding-right", pad + "px");
                     $(".teaser").css("padding-bottom", "0px");
                     $(".teaser").css("padding-left", "0px");
                     $(".teaser").first().css("padding-left", (pad) + "px");
                 }
             } else {
                 $(".teaser").css("padding-top", "1%");
                 $(".teaser").css("padding-right", "1%");
                 $(".teaser").css("padding-bottom", "1%");
                 $(".teaser").css("padding-left", "1%");
                 $(".teaser").css("border-top", "2px solid #616161");
                 $(".teaser").css("width", "98%");
                 $(".teaserinner").css("width", "100%");
                 if (widthMain > 590) {
                     $("#teasercontainer").css("margin-top", "5px");
                 } else {
                     $("#teasercontainer").css("margin-top", "0px");
                 }
                 $(".columngrayed").css("border", "0");
                 $(".columnactive").css("border", "0");
             }
         }
