 //A function to set a cookie tuple with an expiration date
 function setCookie(cname, cvalue, cexpire) {
     "use strict";
     var objCurrentDate = new Date();
     objCurrentDate.setTime(objCurrentDate.getTime() + (cexpire * 24 * 60 * 60 * 1000));
     var expires = "expires=" + objCurrentDate.toUTCString();
     document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
 }
 
 //A function to get a cookie as a list in the form name=value;expiration;path
 function getCookie(cname) {
     "use strict";
     var name = cname + "=";
     var params = document.cookie.split(";");
     for(var i = 0; i < params.length; i++) {
         var c = params[i];
         while (c.charAt(0) == " ") {
             c = c.substring(1);
         }
         if (c.indexOf(name) == 0) {
             return c.substring(name.length, c.length);
         }
     }
     return "";
 }

      <h1>
       The Certified Professional Trainers Network (CPTN) Inc.
      </h1>
      <p class="print">
       Welcome to Certified Professional Trainers Network (CPTN) Inc. The
       <abbr class="normal" lang="en-CA" title="Certified Professional Trainers Network">
        CPTN
       </abbr>
       integrates current research and practical applications for education, communication, professional development and marketing opportunities for Personal Trainers to maintain a leading edge on professional training developments.
      </p>
      <p aria-hidden="true" class="print">
       CPTN is a member driven organization, and thus we strive to provide our
       <strong>
        Members
       </strong>
       and
       <strong>
        Certified Personal Trainers
       </strong>
       the assistance needed to get ahead in their professional careers. We offer several courses through
       <strong>
        distance education
       </strong>
       as an on going effort to further fitness education to a greater audience.
       <abbr class="normal" lang="en-CA" title="Certified Professional Trainers Network">
        CPTN
       </abbr>
       is
       <strong>
        committed
       </strong>
       to providing the best in health and fitness education, and we welcome new trainers to join us.
      </p>
      <p class="skiplink">
       <a href="#top" title="Return to top of page">
        Skip to Top
       </a>
      </p>
      <p class="print">
       Get prepared to become a
       <abbr class="normal" lang="en-CA" title="Certified Professional Trainers Network">
        CPTN
       </abbr>
       Certified Personal Trainer (CPTN-CPT) through the well developed
       <em>
        Art &amp; Science of Personal Training
       </em>
       workshop. This course is intended for the new fitness professional, and enthusiast interested in becoming a Personal Trainer.
      </p>
