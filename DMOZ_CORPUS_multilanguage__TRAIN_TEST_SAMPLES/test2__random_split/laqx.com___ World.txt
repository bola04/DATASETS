     /**
      * Build form to shown city forecast
      */
     function DoFcstCity(divId, fcst) {
         var dt, dtInit;
         var sHeader, sDN, sWW1, sWW2, sTemp, sWd, sWs;
         var item1, item2;
 
         dtInit = DateFromString(fcst.InitDate, "yymmddHH");
         sHeader = sDN = sWW1 = sWW2 = sTemp = sWd = sWs = "";
 
         //*
         for (var i = 0; i < fcst.Data.length; i += 2) {
             item1 = fcst.Data[i];
             item2 = fcst.Data[i + 1];
             dt = dtInit.addHour(item1.FH);
 
             item2.TMax = parseInt(item2.TMax);
             item2.TMin = parseInt(item2.TMin);
             
             sHeader += "<td colspan='2' class='title'>" + dt.format("m月d日") + "</td>";
             sDN += (dt.getHours() == 20 ? "<td>白天</td><td>夜间</td>" : "<td>夜间</td><td>白天</td>");
             sWW1 += "<td><img src='" + GUrl + "Images/wwcma/" + (dt.getHours() == 20 ? "d" : "n") + (item1.WW < 10 ? "0" : "") + item1.WW + ".gif'></img></td>" + 
                     "<td><img src='" + GUrl + "Images/wwcma/" + (dt.getHours() == 20 ? "n" : "d") + (item2.WW < 10 ? "0" : "") + item2.WW + ".gif'></img></td>";
 
             sWW2 += "<td>" + item1.WWChs + "</td><td>" + item2.WWChs + "</td>";
 
             if ( dtInit.getHours() == 20) //08时起报的最高为当日天，归低为第二日早间最低
                 sTemp += "<td colspan='2'><span style='color:blue'>" + item2.TMin + "</span>&nbsp;~&nbsp;<span style='color:red'>" + item2.TMax + "</span>℃</td>";
             else
                 sTemp += "<td colspan='2'><span style='color:red'>" + item2.TMax + "</span>&nbsp;~&nbsp;<span style='color:blue'>" + item2.TMin + "</span>℃</td>";
 
             sWd += "<td>" + item1.WDChs + "风</td><td>" + item2.WDChs + "风</td>";
             sWs += "<td>" + item1.WSChs + (item1.WS < 1 ? "":"级") + "</td><td>" + item2.WSChs + (item2.WS < 1 ? "":"级") + "</td>";
         }
 
         var s = "<table width='100%'>" +
             "<tr>" + sHeader + "</tr>" +
             "<tr>" + sDN + "</tr>" +
             "<tr>" + sWW1 + "</tr>" +
             "<tr>" + sTemp + "</tr>" +
             "<tr>" + sWd + "</tr>" +
             "<tr>" + sWs + "</tr>" +
             "</table>"
 
         $("#" + divId).html(s);
     }
