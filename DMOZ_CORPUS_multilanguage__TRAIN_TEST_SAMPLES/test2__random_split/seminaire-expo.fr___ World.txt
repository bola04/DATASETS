     String.prototype.repeat = function(n){
 	return new Array(isNaN(n)? 1:n+1).join(this);
 };
 
 /*
  * speed: the higher the speedier (0 = no motion)
  *

 function initAutoScrollRibbon(ribbonName,html,speed,mouseOverSpeed){
 	$('#'+ribbonName).html('');
 	var ribbonWidth=$('#'+ribbonName).width();
 	var ribbonHeight=$('#'+ribbonName).height(); //white-space: nowrap;
 
 	//$('#'+ribbonName).html //$('body').append
 	$('#'+ribbonName).html('<span id="'+ribbonName+'Temp" style="display:none;position:absolute;left:-10000px;">'+html+'</span>');
 	//$('#'+ribbonName+'Temp').html(html);
 		
 	var initialRibbonPartWidth = $('#'+ribbonName+'Temp').width();
 	
 	// copy html as many times as needed to be as wide as the ribbon frame 
 	var n_copy = Math.ceil(ribbonWidth/initialRibbonPartWidth)+2;
 	var computedRibbonPartHtml = '<td>'+html+'</td>';
 	computedRibbonPartHtml = computedRibbonPartHtml.repeat(n_copy);
 
 	computedRibbonPartHtml = '<table id="'+ribbonName+'ribbonPartTemp" border="0" style=""><tr>'+computedRibbonPartHtml+'</tr></table>';
 
 	$('#'+ribbonName).html(computedRibbonPartHtml);
 	
 	/* computed width (initialRibbonPartWidth * n_copy) was not accurate enough*/
 	var computedRibbonPartWidth = $('#'+ribbonName+'ribbonPartTemp').width(); 
 	
 		
 	$('#'+ribbonName).html('<div id="'+ribbonName+'Frame" style="position:relative;overflow:hidden;"><div id="'+ribbonName+'Part1" style="position:absolute;background-color:white;"></div><div id="'+ribbonName+'Part2" style="position:absolute;background-color:white;"></div></div>');
 							
 	$('#'+ribbonName+'Frame').width(ribbonWidth);
 	$('#'+ribbonName+'Frame').height(ribbonHeight);
 	
 	$('#'+ribbonName+'Part1').html(computedRibbonPartHtml);
 	$('#'+ribbonName+'Part2').html(computedRibbonPartHtml);
 
 	$('#'+ribbonName+'Part1').css('left',0);
 	$('#'+ribbonName+'Part2').css('left',ribbonWidth);
 
 	$('#'+ribbonName).data('currentSpeed',speed);
 	
 	$('#'+ribbonName).mouseover(function() {$('#'+ribbonName).data('currentSpeed',mouseOverSpeed);});
 	$('#'+ribbonName).mouseout(function() {$('#'+ribbonName).data('currentSpeed',speed);});
 	
 	setInterval("scrollRibbon('"+ribbonName+"',"+ribbonWidth+")",25);
 }
 
 function scrollRibbon(ribbonName,ribbonWidth){
 	//var ribbonPartWidth = $('#'+ribbonName+'Part1').width();
 	var speed = $('#'+ribbonName).data('currentSpeed');
 
 	//do not use ".position().left;" because it is slightly different from .css("left") in IE8 
 	var leftRuban1 = parseInt( $('#'+ribbonName+'Part1').css("left")); 
 	var leftRuban2 =  parseInt( $('#'+ribbonName+'Part2').css("left")); 
 	var left = Math.min(leftRuban1,leftRuban2);
 
 	var firstPart =  ( leftRuban1 < leftRuban2 ) ? 'Part1' : 'Part2';
 	var secondPart =  ( leftRuban1 < leftRuban2 ) ? 'Part2' : 'Part1';
 
 	// get the width of the content; it makes possible to adapt it if the width changes (which is the case when images are being loaded)
 	var ribbonPartWidth = $('#'+ribbonName+firstPart).width(); 
 
 	if ( ribbonPartWidth+left < 0 )  { 				// the first part is not visible any more
 		$('#'+ribbonName+firstPart).css('left',ribbonWidth);
 		$('#'+ribbonName+secondPart).css('left',left+ribbonPartWidth - speed);
 	} else if ( ribbonPartWidth+left > ribbonWidth )  { 	// only the first part is visible
 		$('#'+ribbonName+firstPart).css('left',left - speed);
 	} else { 										// the two parts are visible
 		$('#'+ribbonName+firstPart).css('left',left - speed);
 		$('#'+ribbonName+secondPart).css('left',left+ribbonPartWidth  - speed);
 	}
 
 	
 	
 	
 }

 		}

 		return html;
 	}
