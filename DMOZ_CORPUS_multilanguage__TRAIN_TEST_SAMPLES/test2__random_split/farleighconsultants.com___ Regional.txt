   <script type="text/javascript">
    this.randomtip = function(){
 
 	var pause = 7500; // define the pause for each tip (in milliseconds) 
 	var length = $(".testimonials blockquote").length; 
 	var temp = -1;		
 
 	this.getRan = function(){
 		// get the random number
 		var ran = Math.floor(Math.random()*length) + 1;
 		return ran;
 	};
 	this.show = function(){
 		var ran = getRan();
 		// to avoid repeating
 		while (ran == temp){
 			ran = getRan();
 		}; 
 		temp = ran;
 		$(".testimonials blockquote").hide();	
 		$(".testimonials blockquote:nth-child(" + ran + ")").fadeIn("slow");		
 	};
 	
 	show(); setInterval(show,pause);
 	
 };
 
 $(document).ready(function(){	
 	$(".testimonials blockquote").hide();
 	randomtip();
 });
   </script>
