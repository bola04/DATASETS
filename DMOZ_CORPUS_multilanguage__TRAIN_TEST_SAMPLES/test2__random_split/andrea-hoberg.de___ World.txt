       $(document).ready(function(){
 
   $('#main-nav li.level-0 a').mouseenter(function() {
     crossNavName = $(this).attr('rel');
     //console.log('rel: '+crossNavName);
     $('#'+crossNavName).mouseenter();
     goToPos($('#'+crossNavName));
   }).mouseleave(function() {  
       /*$('#'+crossNavName).mouseleave();*/
       $('#navigation').find('a.off').hide();
     });
   
   $('#main-nav ul.nav-main').mouseleave(function() {  
     //goToPos($('#navigation a.nav-path-selected').parent());
     //console.log($('#navigation a.nav-path-selected').parent().attr('id'));
     
     $('#vertical').css({
             'background-color': $('body').css('background-color'),
             'left': parseInt($('#navigation a.nav-path-selected').parent().css('left'))+100+'px'
         });
               
    $('#horizontal').css({
             'background-color': $('body').css('background-color'),
             'top': parseInt($('#navigation a.nav-path-selected').parent().css('top'))+'px'
         });
       
   });
   
   
   function goToPos(destItem) {
   //console.log(destItem.css('left'));
   
     var destX = parseInt(destItem.css('left'));
     var destY = parseInt(destItem.css('top'));
   
      $('#vertical').stop().animate({
             left: destX+100+'px'
         },400);
               
    $('#horizontal').stop().animate({
             top: destY+'px'
         },400);
   
   }
   
   //console.log('left:'+destItem.css('left'));
 
 });
