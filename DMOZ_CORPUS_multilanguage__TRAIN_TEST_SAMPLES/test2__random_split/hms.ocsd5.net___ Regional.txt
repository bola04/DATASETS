 		function SlideLink() {

 				h = Link.firstChild.href;
 				t = Link.firstChild.target;
 				if (h) { (t == "_self" | t == "_top" | t == "") ? document.location.href=h : window.open(h,'SlideWindow');}
 			}
 			if(Link.getElementsByTagName('A').length > 0){
 				h = Link.getElementsByTagName('A')[0].href;
 				t = Link.getElementsByTagName('A')[0].target;
 				if (h) { (t == "_self" | t == "_top" | t == "") ? document.location.href=h : window.open(h,'SlideWindow');}

 		}

 		startList = function() {
 			if (document.all&&document.getElementById) {
 				navRoot = document.getElementById("nav");

 				for (i=0; i<navRoot.childNodes.length; i++) {

 							this.className=this.className.replace(" over", "");

                       <span style="font-size: 18px; font-family: 'book antiqua', palatino;">
                        <strong>
                         A Day of Aviation: Robert E. Howard Taking Flight
                        </strong>
                       </span>
                      </p>
                      <p style="text-align: center;">
                       <span style="font-family: 'book antiqua', palatino;">
                        <img border="0" height="250" src="http://images.pcmac.org///images/Users/tunisha.summers@ocsd5.net//Screen_Shot_2018-05-17_at_10.15.00_AM.png" width="715"/>
                       </span>
                      </p>
                      <p>
                       <span style="font-size: 16px; font-family: 'book antiqua', palatino;">
                        The Robert E. Howard Middle School Gentlemen’s Club under the advisement of Representative Jerry Govan and Mr. Freddie Ellis (6th grade social studies teacher) participated in Orangeburg Aviation Day on Saturday, May 12, 2018. This day-long aviation camp at Columbia International Airport exposed 25 of REHMS students to a little slice of the complex world of aviation. A cross collaboration between Dr. James E. Sulton, III, Rep. Jerry Govan, and Robert E. Howard Middle School, this program afforded the gentlemen the opportunity to learn about airport operations, a visit to the air traffic control tower, insight on behind-the-scene operations at Delta Airlines and hands-on activities to learn about Aviation Maintenance Technology.
                       </span>
                      </p>
                      <p style="text-align: center;">
                       <span style="font-family: 'book antiqua', palatino;">
                       </span>
                      </p>
                      <p>
                       <span style="font-size: 16px; font-family: 'book antiqua', palatino;">
                        Coming together to make Orangeburg Aviation Day 2018 possible, the following organizations made generous donations so that all of the participants could attend for free: National Black Coalition of federal Aviation Employees (NBCFAE), The Bessie Coleman Aerospace Legacy, Columbia Metropolitan Airport, The Celebrate Freedom Foundation, Delta Airlines, the Federal Aviation Administration’s Columbia Air Traffic Control Tower and the administrative team at Robert E. Howard Middle School.
                       </span>
                      </p>
                      <p style="text-align: center;">
                       <span style="font-family: 'book antiqua', palatino;">
                       </span>
                      </p>
                      <p>
                       <span style="font-size: 16px; font-family: 'book antiqua', palatino;">
                        This day of small group rotations mimicked a perfect educational experience which included tours of the Air Traffic Control tower and Terminal Radar Approach Control (TRACON), roles of the many operational personnel, discussions of the many careers in aircraft maintenance and airline operations. The culminating events of the program encompassed guest speaker, Al Orton, Delta Airlines and a certificate of completion. These events, along with the wealth of knowledge were given to the students of REHMS and encouraged them to look towards the future and aim high to the many aviation careers. They were also enlighten that many of these careers did not require a college degree, but paid top salary following rigorous training. Many of our participants were surprised to hear this. Following the day long events, not only are the Gentlemen’s Club participants knowledgeable about aviation careers, but interested in these careers.
                       </span>
                      </p>

                        <span style="font-size: 16px;">
