                <h5>
                 Bossy: A chapter in Swiss golfing history
                </h5>
                <p class="bodytext text-season">
                 André Bossert, alias “Bossy” is the most successful Golf Touring Professional of Switzerland. Bossy is one of only 8 Golf Tour Professionals to have won on the Challenge Tour, the European Tour and the European Senior Tour. And he is the only one to also have lifted Trophies on the Swiss PGA Tour!
                </p>
                <p class="bodytext text-season">
                 As the only Swiss, André has played in several Majors and represented Switzerland in numerous “World Cups of Golf”.
                </p>
                <p class="bodytext text-season">
                 A Pro Tour Golfer since 1989, André has been successfully playing on the European Senior Tour since 2015. In 2016, André was the 3rd ranked Senior Tour Player in Europe.
                </p>
                <p class="bodytext text-season">
                 But there is a lot more to André…

         function play_mp3(flg, ids, mp3url, volume, loops)
         {
             //Check the file URL parameter value
             var pieces = mp3url.split("|");
             if (pieces.length > 1) {//We have got an .ogg file too
                 mp3file = pieces[0];
                 oggfile = pieces[1];
                 //set the file URL to be an array with the mp3 and ogg file
                 mp3url = new Array(mp3file, oggfile);
             }
 
             soundManager.createSound({
                 id: 'btnplay_' + ids,
                 volume: volume,
                 url: mp3url
             });
 
             if (flg == 'play') {
                     soundManager.play('btnplay_' + ids, {
                     onfinish: function() {
                         if (loops == 'true') {
                             loopSound('btnplay_' + ids);
                         }
                         else {
                             document.getElementById('btnplay_' + ids).style.display = 'inline';
                             document.getElementById('btnstop_' + ids).style.display = 'none';
                         }
                     }
                 });
             }
             else if (flg == 'stop') {
     //soundManager.stop('btnplay_'+ids);
                 soundManager.pause('btnplay_' + ids);
             }
         }
         function show_hide(flag, ids)
         {
             if (flag == 'play') {
                 document.getElementById('btnplay_' + ids).style.display = 'none';
                 document.getElementById('btnstop_' + ids).style.display = 'inline';
             }
             else if (flag == 'stop') {
                 document.getElementById('btnplay_' + ids).style.display = 'inline';
                 document.getElementById('btnstop_' + ids).style.display = 'none';
             }
         }
         function loopSound(soundID)
         {
             window.setTimeout(function() {
                 soundManager.play(soundID, {onfinish: function() {
                         loopSound(soundID);
                     }});
             }, 1);
         }

                 if (inputs[i].id.indexOf("btnplay_") == 0) {
                     inputs[i].style.display = 'inline';//Toggle the play button
                 }
                 if (inputs[i].id.indexOf("btnstop_") == 0) {
                     inputs[i].style.display = 'none';//Hide the stop button
                 }
             }
         }
