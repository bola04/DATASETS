 function toggleSala(a, obj) {
        
 
      if ($("#caja-valpo").attr("style")=='display: block;') {
       if (obj=="#salas .valpo"){ 
         $("#caja-valpo").attr("style","display: none");
         return false;
       } else {
         $("#caja-valpo").attr("style","display: none");
         $("#caja-stgo").attr("style","display: block");        
       }    
      }
 
      if ($("#caja-stgo").attr("style")=='display: block;') {
       if (obj=="#salas .stgo"){
         $("#caja-stgo").attr("style","display: none");
         return false;
       } else {
         $("#caja-stgo").attr("style","display: none");
         $("#caja-valpo").attr("style","display: block");      
       }
      }
 
      $("#salas .cont").hide();     
      $("#salas button").attr("style", "background:#EEE0;");
      $(obj).toggle();
      var atributo = $(obj).attr("style");
      if("display: block;".indexOf(atributo)>=0) {
         $(a).attr("style", "background:#C00; color:#FFF; border:1px solid #C00;");
                 return;
      }
  
 }
