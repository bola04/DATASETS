 	function calculateShippingTotal(el) {	
 
 			if (el.value == "Local Pickup") {
 				el.form.elements['InsuranceSelected'][1].checked = false;
 			}
 			else {
 				//tick insurance if it hasn't already been confirmed not to
 				var ins = el.form.elements['InsuranceSelected'][1];
 				if (!ins.getAttribute("seenwarning")) ins.checked = true;
 			}
 			
 			
 			var insurance = (el.form.elements['InsuranceSelected'][1].checked) ? 
 				document.getElementById('insurancePrice').innerHTML.replace(/ AUD/, "") : 0;
 
 			var price = 0;	
 			for (var i=0, opts = el.form.elements['PostageSelected']; i<opts.length; i++) {

 					price = document.getElementById('Postage' + opts[i].value + 'Price').innerHTML.replace(/ AUD/, "");
 					break;
 				}
 			}
 			
 			document.getElementById('TotalShipping').innerHTML = formatAsMoney(parseFloat(price) + parseFloat(insurance)) + " AUD"
 			
 		
 	}
 	
 	function formatAsMoney(mnt) {
     	mnt -= 0;
     	mnt = (Math.round(mnt*100))/100;
     	return (mnt == Math.floor(mnt)) ? mnt + '.00' 
               : ( (mnt*10 == Math.floor(mnt*10)) ?
