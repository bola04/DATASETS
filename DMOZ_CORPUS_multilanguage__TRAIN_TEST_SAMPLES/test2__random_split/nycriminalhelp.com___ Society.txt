   $(window).load(function() {
 	var wWidth = $(window).width();
 
 	//Check to see if we are under the right conditions and don't run if we aren't
 	//Check Three CTAs first
 	if ( ($('.cta-list ul').hasClass('items3')) && (wWidth < 985) && (wWidth > 735) ) {
 		//Grab the individual CTAs
 		var ctaThreeList = $('.items3 li'),
 			ctaTOne = ctaThreeList.eq(0).outerHeight(),
 			ctaTTwo = ctaThreeList.eq(1).outerHeight();	
 		
 		//Check all the sizes against each other to determine which is the tallest and set the min-height
 		if ( ctaTOne === ctaTTwo ) {
 			return;	
 		} else if ( ctaTOne > ctaTTwo ) {
 			ctaThreeList.eq(1).css('min-height', ctaTOne + 'px');	
 		} else {
 			ctaThreeList.eq(0).css('min-height', ctaTTwo + 'px');
 		}
 	//Check four CTAs next
 	} else if ( ($('.cta-list ul').hasClass('items4')) && (wWidth < 985) && (wWidth > 735) ) {
 		//Grab the individual CTAs
 		var ctaFourList = $('.items4 li'),
 			ctaOne = ctaFourList.eq(0).outerHeight(),
 			ctaTwo = ctaFourList.eq(1).outerHeight(),
 			ctaThree = ctaFourList.eq(2).outerHeight(),
 			ctaFour = ctaFourList.eq(3).outerHeight();
 			
 		//Check all the sizes against each other to determine which is the tallest and set the min-height
 		if ( (ctaOne >= ctaTwo) && (ctaOne >= ctaThree) && (ctaOne >= ctaFour) ) {
 			ctaFourList.css('min-height', ctaOne + 'px');	
 		} else if ( (ctaTwo >= ctaOne) && (ctaTwo >= ctaThree) && (ctaTwo >= ctaFour) ) {
 			ctaFourList.css('min-height', ctaTwo + 'px');

 			ctaFourList.css('min-height', ctaThree + 'px');	
 		} else {
 			ctaFourList.css('min-height', ctaFour + 'px');
 		}
 			
 	} else {
 		return;	
 	}	
 });
