     function isSupportedBrowser(log) {
 
         var md = new MobileDetect(window.navigator.userAgent);
 
         // allow all mobile devices
         if (md && md.mobile()) {
 
             if (log) console.log(md.mobile());
             return true;
 
         } else {
 
             var parser = new UAParser();
 
             var MIN_VERSION_CHROME = 48;
             var MIN_VERSION_IE = 11;
             var MIN_VERSION_SAFARI = 8;
             var MIN_VERSION_FIREFOX = 34;
             var MIN_VERSION_EDGE = 14;
 
             var supportedBrowsers = [];
             supportedBrowsers.push({name: 'chrome', minMajor: MIN_VERSION_CHROME});
             supportedBrowsers.push({name: 'firefox', minMajor: MIN_VERSION_FIREFOX});
             supportedBrowsers.push({name: 'ie', minMajor: MIN_VERSION_IE});
             supportedBrowsers.push({name: 'safari', minMajor: MIN_VERSION_SAFARI});
             supportedBrowsers.push({name: 'edge', minMajor:MIN_VERSION_EDGE});
 
             var name = parser.getResult().browser.name.toLowerCase();
             var major = parseInt(parser.getResult().browser.major);
             var supportedBrowser = false;
 
             if (log) {
                 console.log('Supported browsers', supportedBrowsers);
                 console.log('Current browser', name, major);
             }
 
             $.each(supportedBrowsers, function (index, value) {
                 if (value.name.toLowerCase() === name.toLowerCase() && major >= value.minMajor) {
                     supportedBrowser = true;
                 }
             });
 
             if (log) {
                 console.log('Browser supported', supportedBrowser);
             }
 
             return supportedBrowser;
         }
 
     }

     function fetchStatus(address, callback) {
         var client = new XMLHttpRequest();
         var counter = 0;
         client.onreadystatechange = function() {
             // in case of network errors this might not give reliable results
             if(this.readyState == 4) {
                 if (counter < 1) {
                     counter++;
                     callback(this.status);
                 }
             }
         };
         client.open("HEAD", address);
         client.send();
     }
