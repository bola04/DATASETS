    $(document).ready(init);
 
 function init() {
   initSlideshow();
 }
 
 function initSlideshow() {
   var slide = $("#js-slideshow");
   var timeOut = '';
 
   if (slide.length == 0) {
     return;
   }
 
   slide.append('<div id="slide-nav"></div>');
 
   var items = $("#js-slideshow .item"),
       count = items.size() - 1,
       i = 0;
 
   $(items).each(function(){
     $('<a href="#" class="slide-nav" id="nav-'+i+'">'+i+'</a>').appendTo('#slide-nav');
     i++;
   });
 
   $('#slide-nav a').click(function(){
     var slidenum = parseInt($(this).attr('id').slice(4));
     fade(slidenum, true);
     return false;
   });
 
   function fade(index, click) {
     if(click == true) {
       clearTimeout(timeOut);
     }
 
     var anim = 500,
         wait = 10000;
 
     var prev = index - 1;
     if (prev < 0) {
       prev = count;
     }
 
     $('#slide-nav a').removeClass('active').siblings('#nav-'+index).addClass('active');
 
     items.eq(index).fadeTo(anim, 1.0);
     items.not($(items).eq(index)).fadeTo(anim, 0.0);
 
     var next = index + 1;
     if (next > count) {
       next = 0;
     }
 
     timeOut = setTimeout(function() {
       fade(next)
     }, wait + anim);
   }
 
   fade(0);
 }
