     function initCheckboxClass(cBoxId, cBoxGroup, cBoxAllId) {
         $('#' + cBoxId).click(function () {
             var thisChecked = $(this).is(":checked");
             var thisVal = parseInt($(this).val());
 
             var data = $('input[name=' + cBoxGroup + ']:checked').get();
 
             var lowerChecked = null;
             var higherChecked = null;
 
             $(data).each(function () {
                 var itemVal = parseInt($(this).val());
                 if (itemVal != thisVal) {
                     if (itemVal < thisVal) {
                         lowerChecked = this;
                     }
                 }
             });
 
             $(data.reverse()).each(function () {
                 var itemVal = parseInt($(this).val());
                 if (itemVal != thisVal) {
                     if (itemVal > thisVal) {
                         higherChecked = this;
                     }
                 }
             });
 
             $('input[name=' + cBoxGroup + ']').each(function () {
                 var itemChecked = $(this).is(":checked");
                 var itemVal = parseInt($(this).val());
                 if (itemVal != thisVal & itemVal != 0) {
                     if (parseInt($(this).val()) < thisVal) {
                         if (lowerChecked != null & thisChecked) {
                         }
                         else {
                             if (higherChecked == null || (higherChecked != null & !thisChecked))
                                 $(this).prop("checked", thisChecked)
                         }
                     }
                     else {
                         if (higherChecked != null) {
                             higherVal = parseInt($(higherChecked).val());
 
                             if (itemVal < higherVal) {
                                 $(this).prop("checked", thisChecked)
                             }
                         }
                     }
                 }
             });
 
             if ($('#' + cBoxId).is(':checked')) {
                 $('#' + cBoxAllId).prop("checked", false);
             }
             $('#mainQuery').submit();
 
         });
     }
 
     function initSubCheckbox(cBoxId, cBoxAllId, cTopBoxId, cCountId) {
         $('#' + cBoxId).click(function (event) {
             var counter = window[cCountId];
             var checked = $('#' + cBoxId).prop('checked');
             if (checked) {
                 $('input[name=' + cBoxAllId + ']').prop("checked", false);
                 counter++;
             }
             else { counter--; }
             $('#' + cTopBoxId).prop('checked', counter != 0);
             window[cCountId] = counter;
             $('#mainQuery').submit();
         })
     }
 
     function toggleBox(event, ddId, ddBoxId, openClose) {
         $('#' + ddId).toggleClass("open");
         $('#' + ddBoxId).slideToggle("fast", function () {
             openClose = !openClose;
             if (openClose) {
                 $(document).bind('click.' + ddBoxId, function (e) {
                     var isInTarget = $(e.target).is('#' + ddBoxId + ', #' + ddBoxId + ' *');
                     var isOpener = $(e.target).is('#' + ddId + ', #' + ddId + ' *');
                     if (isInTarget || isOpener) return;
                     $('#' + ddBoxId).slideUp("fast");
                     $('#' + ddId).removeClass("open");
                     openClose = false;
                 })
             }
             else {
                 $(document).unbind('click.' + ddBoxId);
             }
         });
         event.preventDefault();
     }
