       function  direct_code_jump() {
         var code = jQuery.trim($("#direct_code_input").val());
         $("#direct_code_input").val(code);
         if (code != "") {
           jQuery.get("/products/9999/code".replace("9999", code.toUpperCase()),
             function(data) {
               if (data == "") {
                  alert("商品番号" + code + "の商品は見つかりません");
               } else {
                  location.href = "/products/9999".replace("9999", data);
               }
             });
         }
       };
 
       $("#direct_code_button").click(direct_code_jump);
       $("#direct_code_input").keyup(function (e) {
         if (e.keyCode == 13) {
           direct_code_jump();
         }
       });
     });
