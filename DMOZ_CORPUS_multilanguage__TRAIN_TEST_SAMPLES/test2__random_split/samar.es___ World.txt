                         value: text,
                         option: this
                       };
                     }) );          
                   },
                   select: function( event, ui ) {             
                     ui.item.option.selected = true;
                     that._trigger( "selected", event, {
                         item: ui.item.option
                     });
                       
                   },
                   change: function( event, ui ) {
                       if ( !ui.item )
                           return removeIfInvalid( this );
                   }
               })
               .addClass( "ui-widget ui-widget-content ui-corner-left" );
