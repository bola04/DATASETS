             function checkfile(file, msgtype) {
                 if (!file.size) return false;
                 if (!file.type) return false;
                 var split = file.type.split('/');
                 if ((split[0] != 'image' && msgtype == 'image') || (split[0] != 'video' && msgtype == 'video') || (split[0] != 'audio' && msgtype == 'audio')) {
                     $.growl.error({ message: "Wrong file type!" });
                     return false;
                 }
                 else if (msgtype == 'image') {
                     if (file.size > (5 * (1024 * 1024))) {
                         $.growl.error({ message: 'Image to big!' });
                         return false;
                     }
                 }
                 else if (msgtype == 'video') {
                     if (file.size > (20 * (1024 * 1024))) {
                         $.growl.error({ message: 'Video to big!' });
                         return false;
                     }
                 }
                 else if (msgtype == 'audio') {
                     if (file.size > (5 * (1024 * 1024))) {
                         $.growl.error({ message: 'Audio to big!' });
                         return false;
                     }
                 }
                 else if (msgtype == 'vcard') {
                     if (file.size > (1 * (1024 * 1024))) {
                         $.growl.error({ message: 'Vcard to big!' });
                         return false;
                     }
                 }
                 return true;
             }
