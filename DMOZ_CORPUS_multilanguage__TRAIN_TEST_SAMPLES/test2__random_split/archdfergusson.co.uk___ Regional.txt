 function runSlideShow(){

    j=(p-2)
    } else {
    j = j - 2
    if (j == -1) j=(p-1)
    }
    }
    document.images.SlideShow.src = preLoad[j].src
    if (document.all){
       document.images.SlideShow.filters.blendTrans.Play()
    }
    j = j + 1
    if (j > (p-1)) j=0
