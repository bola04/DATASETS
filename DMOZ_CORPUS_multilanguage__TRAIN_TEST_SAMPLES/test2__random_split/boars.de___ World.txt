   function PositionContainer()
   {
     var LA_div   = document.getElementById("LogoArea") ;
     var AOA_div  = document.getElementById("AddOnArea") ;
     var IA_div   = document.getElementById("InnerArea") ;
     var BT_div   = document.getElementById("ButtonTeams") ;
     var PCA_div  = document.getElementById("PopupContentArea") ;
     var PCAS_div = document.getElementById("PopupContentAreaS") ;
     // alert("\npositionContainerInProgress >" + positionContainerInProgress + "<") ;
     if (!positionContainerInProgress)
     {
       IA_div.style.left           = (document.body.clientWidth - IA_div.offsetWidth) / 2 ;
       // alert("\ndocument.body.offsetHeight [" + document.body.offsetHeight + "]\ndocument.body.clientHeight [" + document.body.clientHeight + "]\nIA_div.offsetHeight [" + IA_div.offsetHeight + "]\nminimumSpaceFromTopOfBody [" + minimumSpaceFromTopOfBody + "]") ;
       IA_div.style.top            = (document.body.clientHeight - IA_div.offsetHeight) / 2 ;
       if (IA_div.offsetTop < minimumSpaceFromTopOfBody)
         IA_div.style.top = minimumSpaceFromTopOfBody ;
       IA_div.style.visibility     = "visible" ;
 
       AOA_div.style.left          = IA_div.offsetLeft + IA_div.offsetWidth - AOA_div.offsetWidth ;
       AOA_div.style.top           = IA_div.offsetTop - AOA_div.offsetHeight - 3 ;
       AOA_div.style.visibility    = "visible" ;
 
       PCA_div.style.left          = IA_div.offsetLeft + 50 ;
       // PCA_div.style.top           = IA_div.offsetTop + 88 ;
       PCA_div.style.top           = IA_div.offsetTop + 116 ;     /* neu: Menges, 14.10.2008 */
       // PCA_div.style.visibility    = "visible" ;
       PCAS_div.style.left         = PCA_div.offsetLeft - 3 ;
       PCAS_div.style.top          = PCA_div.offsetTop - 3 ;
       // PCAS_div.style.visibility   = "visible" ;
 
       // LA_div.style.left           = IA_div.offsetLeft + BT_div.offsetLeft + BT_div.offsetWidth + 15 ;
       LA_div.style.left           = IA_div.offsetLeft + BT_div.offsetLeft + BT_div.offsetWidth + 18 ;    /* NEW LOGO !!! */
       LA_div.style.visibility     = "visible" ;
       // LA_targetTop                = IA_div.offsetTop + BT_div.offsetTop  + BT_div.offsetHeight - LA_div.offsetHeight - 10 ;
       LA_targetTop                = IA_div.offsetTop + BT_div.offsetTop  + BT_div.offsetHeight - LA_div.offsetHeight - 5 ;
       positionContainerInProgress = true ;
     }
     // alert("\nLA_div.offsetTop [" + LA_div.offsetTop + "] < LA_targetTop [" + LA_targetTop + "] ???") ;
     if (LA_div.offsetTop < LA_targetTop)
     {
       LA_div.style.top = ((LA_div.offsetTop + LA_stepValue) < LA_targetTop) ? LA_div.offsetTop + LA_stepValue : LA_targetTop ;
       // alert("\nLA_div.offsetTop - um LA_stepValue erhöht [" + LA_div.offsetTop + "]") ;
       pcThread = setTimeout('PositionContainer()',10) ;
       return ;
     }
     else
     {
       positionContainerInProgress = false ;
       clearTimeout(pcThread) ;
     }
     // -----------------------------------------------------------------------
     // Container "NextMatchArea" und "AdditionalDataArea" global bekanntmachen
     // (zusaetzlich auch die zugehoerigen Schattencontainer)
     // -----------------------------------------------------------------------
     nextMatchArea          = document.getElementById("NextMatchArea") ;
     additionalDataArea     = document.getElementById("AdditionalDataArea") ;
     popupContentArea       = document.getElementById("PopupContentArea") ;
     popupContentAreaS      = document.getElementById("PopupContentAreaS") ;
     contentArea8           = document.getElementById("ContentArea8") ;

     contentArea7S          = document.getElementById("ContentArea7S") ;
     contentArea7_MI        = document.getElementById("ContentArea7_MI") ;

     DisplayContainer(IA_div, AOA_div) ;
   }

   function SwitchFlash()
   {
     cookieString = document.cookie ;
     if (playFlash || cookieString.indexOf("NOFLASH") > -1)
     {
       playFlash = false ;
       document.getElementById("flashText").innerHTML = "Flash on/<b>off</b>" ;
       cookieString = cookieString.replace(/\[NOFLASH\]/, "") ;
       cookieString = cookieString.replace(/\[FLASH\]/, "") ;
       cookieString += "[NOFLASH]" ;
     }
     else
     {
       if (flashAvailable)
       {
         playFlash = true ;
         document.getElementById("flashText").innerHTML = "Flash <b>on</b>/off" ;
         cookieString = cookieString.replace(/\[NOFLASH\]/, "") ;
         cookieString = cookieString.replace(/\[FLASH\]/, "") ;
         cookieString += "[FLASH]" ;
       }
     }
     document.cookie = cookieString ;
   }
 
 
   function SwitchSound()
   {
     cookieString = document.cookie ;
     var sa = document.getElementById("SoundArea") ;
     if (playSound || cookieString.indexOf("NOSOUND") > -1)
     {
       playSound = false ;
       sa.innerHTML = "&nbsp;" ;
       document.getElementById("soundText").innerHTML = "Sound on/<b>off</b>" ;
       cookieString = cookieString.replace(/\[NOSOUND\]/, "") ;
       cookieString = cookieString.replace(/\[SOUND\]/, "") ;
       cookieString += "[NOSOUND]" ;
     }
     else
     {
       if (flashAvailable)
       {
         playSound = true ;
         document.getElementById("soundText").innerHTML = "Sound <b>on</b>/off" ;
         sa.innerHTML = "<pre><object data=\"./swf/webplayer.swf\" type=\"application/x-shockwave-flash\" width=\"1\" height=\"1\"><param name=\"movie\" value=\"./swf/webplayer.swf\"><param name=\"menu\" value=\"false\"><param name=\"scale\" value=\"noscale\"><param name=\"bgcolor\" value=\"#FFFFFF\"><param name=\"flashvars\" value=\"src=./sound/132_32kbit_stereo.mp3&amp;amp;autostart=yes&amp;amp;loop=no\"></object></pre>" ;
         cookieString = cookieString.replace(/\[NOSOUND\]/, "") ;
         cookieString = cookieString.replace(/\[SOUND\]/, "") ;
         cookieString += "[SOUND]" ;
       }
     }
     document.cookie = cookieString ;
   }
