             }

             } else {
                 $(this).text($(this).attr('data-original-number'));
 
             }
         });

         var hipsInput = $('.hips_input').val();

             $('.bust_input').val(!isNaN(bustInput) && bustInput !== '' ? Math.round(parseFloat(bustInput) / 2.54) : '');
