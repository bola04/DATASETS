 						$('#event_status').val('ready');

 						$('#event_status').val('work');

 					}

 			});
 			$swiper.slideTo('2', '100', false);
 		});
 		function slide_ajax()
 		{
 			first = $('#first_slide').val();
 			$.ajax({
 				type: 'POST',
 				url: './forum.html?sid=6ef75ea4630c1bd827468028364364e2',
 				data: 'event='+first,
 				success: function(data)
 					{
 						if(data !== 'false')
 						{
 							$('#first_slide').remove();
 							$('#event_ajax').remove();
 							$('.event_wrapper').prepend(data);
 							$swiper.update(true);
 							$swiper.slideNext();
 							$('#event_status').val('first');
 						}
 						else
 						{
 							$('#first_slide').remove();
 							$('#event_ajax').remove();
 							$('#event_status').val('blocked');
 							$swiper.update(true);
 						}
 					},
 					error: function(xhr, str){alert('Общая ошибка: ' + xhr.responseCode);
 				}
 			});
 		}

               Томас Андерс готовит новый сингл, который наполнен положительной энергией. Песня описывает отношение артиста к жизни: "Das Leben ist jetzt" - это гимн жизни здесь и сейчас, призыв наслаждаться моментом и жить каждым днем. Песня, напсианная Кристианом Геллером, была создана в сотрудничестве с Вольфгангом Хофером, который написал стихи для более чем 100 песен Удо Юргенса, и множества других классиков эстрады.
               <br/>
               <br/>
               Сингл "Das Leben ist jetzt" выйдет уже 8 июня 2018 года (В ЭТУ ПЯТНИЦУ), и станет предвестником нового немецкоязычного альбома Томаса Андерса.
               <br/>
               <br/>
               А вот и телеэфиры с этой песней:
               <br/>
               <br/>
               11.08. MDR „Schlager des Sommers“ (Запись шоу 09.06 (В СУББОТУ))
               <br/>
               17.08. WDR „Kölner Treff“
               <br/>
               26.08. ZDF „Fernsehgarten“
