     <h1>
      Wir planen Ihre Freiräume!
     </h1>
    </div>
   </div>
   <div class="row">
    <div class="large-4 medium-4 small-12 columns">
     <p>
      Seit über 10 Jahren entwerfen, konstruieren und überwachen wir die Realisierung von Außenanlagen. Hier steht die Besonderheit des Ortes, ob natürlich oder urban, im Vordergrund.
     </p>
    </div>
    <div class="large-4 medium-4 small-12 columns">
     <p>
      Die Einbindung des Bauwerkes in die Landschaft und die formale Fassung der Außenraumfunktionen fügen sich zu einem eigenständigen Neuen zusammen.
     </p>
    </div>

    function setCookie(cname,cvalue,exhours) {
   var d = new Date();
   d.setTime(d.getTime() + (exhours*60*60*1000));
   var expires = "expires=" + d.toGMTString();
   document.cookie = cname+"="+cvalue+"; "+expires;
 }
 
 function getCookie(cname) {
   var name = cname + "=";
   var ca = document.cookie.split(';');
   for(var i=0; i<ca.length; i++) {
     var c = ca[i];
     while (c.charAt(0)==' ') c = c.substring(1);
     if (c.indexOf(name) == 0) {
       return c.substring(name.length, c.length);
     }
   }
   return "";
 }
