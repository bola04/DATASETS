       $(document).ready(function () {
 
         $('#quick-next').click(function () {
             var index = $('#quickModal .modal-body.active').index();
             if (index == 1) {
                 $('#quicksearchsubmit').trigger('click');
             } else {
                 $('#quickModal .modal-body.active').removeClass('active').fadeOut(function () {
                     $('#quickModal .modal-body').eq(index + 1).addClass('active').show();
                 });
             }
         });
 
         $('#search-form').submit(function (e) {
             var valid = false;
             $('#search-form input[type=text]').each(function () {
                 if ($(this).val() != '') valid = true;
             });
             $('#search-form select').each(function () {
                 if ($(this).val() != '') valid = true;
             });
             if (!valid) {
                 e.preventDefault();
                 alert('Please enter search criteria.');
             }
         });
 
 
         $('.rangepicker .border').click(function (e) {
             e.preventDefault();
             $(this).children("i").toggleClass("showit");
         });
 
         $('.quicksearch .border').click(function (e) {
             e.preventDefault();
             $(this).closest(".modal-body").find("i").removeClass("showit");
             $(this).children("i").toggleClass("showit");
         });
 
         $('#quicksearchsubmit').click(function (e) {
             //e.preventDefault();
             $('.quickcheck').each(function () {
                 if ($(this).hasClass("showit")) {
                     var thisid = parseInt($(this).attr("data-optionid"));
                     var thisinput = "#" + $(this).attr("data-inputid");
                     console.log(thisid);
                     $(thisinput).val(thisid)
                 }
             });
             $('#quicksearch-form').submit();
         });
 
         $('.downloadpdf').click(function () {
             $('.modal-footer .fa-cog').addClass("fa-spin");
             $('.modal-footer .dlbutton').css("opacity", "0.3").css("cursor", "not-allowed");
             var cats = "";
             var count = 0;
             $('.categorycheck').each(function () {
                 if ($(this).hasClass("showit")) {
                     var thiscat = parseInt($(this).attr("data-categoryid"));
                     var comma = count == 0 ? "" : ","
                     cats = cats + comma + thiscat
                     count++;
                 }
             });
             $.get("index.php?route=product/category/getbottomsubcategories&categories=" + cats, function (data) {
                 console.log(data);
                 var newurl = "index.php?route=product/category/getCategoryPDF&categories=" + data;
                 console.log(newurl);
                 //var win = window.open(newurl, '_blank');
 
                 $('.downloadpdf span').html("");
                 $.get(newurl, function (data) {
                     $('.downloadpdf span').html("");
                     $('.modal-footer .fa-cog').removeClass("fa-spin");
                     $('.modal-footer .dlbutton').css("opacity", "1").css("cursor", "not-allowed");
                     $('.modal-footer .dlbutton').html('<a href="index.php?route=product/category/downloadcategorypdf&pdf=' + data + '" target="_blank"><button type="button" class="btn btn-success downloadpdf"><i class="fa fa-download"></i> </button></a>');
                 });
 
             });
             //console.log(cats);
         });
 
     });
 
     $(window).load(function () {
         resizeBlocks();
     });
 
     function resizeBlocks() {
         max_height = 0;
         $('#hp-blocks .block').css({height: 'auto'});
         $('#hp-blocks .block').each(function () {
             if ($(this).outerHeight() > max_height)
                 max_height = $(this).outerHeight();
         });
         max_height = Math.max(max_height,420);
         $('#hp-blocks .block').css({height: max_height + 'px'});
     }
 
     $(window).resize(function () {
         if ($(window).width() > 780) {
             resizeBlocks();
         } else {
             $('#hp-blocks .block').css({height: 'auto'});
         }
     });
