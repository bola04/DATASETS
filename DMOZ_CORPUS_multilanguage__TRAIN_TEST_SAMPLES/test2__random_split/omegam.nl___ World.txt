         var $accordion = $('dl.accordion');
 
         $('dd', $accordion).addClass('hide');
 
         $('dt', $accordion).find('span').text("+");
 
         $('dt', $accordion).click(function () {
             if ($(this).is('.add')) {
                 $(this).removeClass("add");
                 $('dl').has(this).find('dd').slideUp(200);
 
             } else {
                 $(this)
                         .addClass("add").find("span").text("-")
                         .end()
                         .siblings().find("span").text("+")
                         .end()
                         .removeClass("add")
                         .end()
                         .next()
                         .slideDown(200)
                         .siblings('dd')
                         .slideUp(200);
             }
         });
     });

       <h2>
        Welkom bij Eurofins Environment Testing!
       </h2>
       <p>
        Eurofins Environment Testing Nederland is onderdeel van de internationale Eurofins-laboratoriumgroep en is uw partner voor milieuanalyse, onder andere voor grond-, water-, waterbodem-, asbest-, bouwstoffen-, asfalt-, ecologisch en hydrobiologisch- en specialistisch onderzoek. Onze klantendienst staat tot uw beschikking met specialistische kennis en vergaande ervaring. Met onze efficiënte &amp; kwalitatieve analysetechnieken ondersteunen wij uw business processen. Hierbij kunt u gebruik maken van onze eigen emballage en logistieke dienst. Onze deskundige en betrouwbare koeriers verzorgen de monsteroverdracht op uw locatie en transporteren de monsters dezelfde dag nog naar het desbetreffende laboratorium.
       </p>
