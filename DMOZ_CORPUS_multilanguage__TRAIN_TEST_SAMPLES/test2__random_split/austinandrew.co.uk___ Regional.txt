     //LOOP SAVED JOBS
     $.getJSON( "/api/v1/users/saved_jobs.json?locale=en", function( data ) {
       var items = [];
   
       $.each( data['saved_jobs'], function( key, val ) {
         //IF SAVED JOBS DROPDOWN LIST EXISTS
         if ($("ul[data-saved-jobs-list]").length){
           var item = "<li data-job-id='" + val['job'].id + "'><a href='" + val['job'].job_url + "'>" + val['job'].job_title + "</a></li>";
           items.push(item);
         };
         
         if ($("a#save_job_" + val['job'].id).length){
           $("a#save_job_" + val['job'].id).html(data['unsave_text']);
           $("a#save_job_" + val['job'].id).attr("href", val['job'].job_url + "/unsave_job");
         }
         
       });
   
       //IF SAVED JOBS DROPDOWN LIST EXISTS  --> POPULATE LIST
       if ($("ul[data-saved-jobs-list]").length){
         if (items.length > 0){
           $("ul[data-saved-jobs-list]").html(items.join( "" ));
         };
         $("span[data-saved-jobs-count]").html(items.length);
       };
     }); 
   });
