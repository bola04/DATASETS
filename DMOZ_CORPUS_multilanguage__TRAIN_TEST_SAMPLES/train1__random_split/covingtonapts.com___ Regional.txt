    function parseISO8601(dateStringInRange) {
             var isoExp = /^\s*(\d{4})-(\d\d)-(\d\d)\s*$/,
                 date = new Date(NaN), month,
                 parts = isoExp.exec(dateStringInRange);
 
             if(parts) {
                 month = +parts[2];
                 date.setFullYear(parts[1], month - 1, parts[3]);
                 if(month != date.getMonth() + 1) {
                     date.setTime(NaN);
                 }
             }
             return date;
         }
