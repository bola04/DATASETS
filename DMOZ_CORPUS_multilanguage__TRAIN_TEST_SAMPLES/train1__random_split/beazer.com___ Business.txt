       (function() { 
                     var init = function() {
                         
                         // Get the dropdown and set the current metroid from the cookies
                         var $select = $('#listMetrosTop').beazerDropDown();
                         var metroid = Beazer.Cookies.GetSearchSettingsMetro();
 
                         // If their Metro cookie is for a Metro we no longer have, blank it out
                         if (metroid != null && $select.find('option[value="' + metroid + '"]').length == 0) {
                             metroid = null;
                             Beazer.Cookies.SetSearchSettingsMetro(null);
                         }
             
                         // Set Metro dropdown value
                         $select.val(metroid).change();
 
                         // Handle clicks on the GO button
                         $select.closest('.go-box').find('button').click(function(e) {
                              
                              var $selected = $select.find('option:selected');
                              var val = $selected.attr('value');
                              var $wait = $('#wait_listMetrosTop');
 
                              if ($selected.attr('data-ispartner') == 'true') {
                                 Beazer.Dialogs.ShowMetroPartnerInfo(val, $wait);
                                 e.preventDefault();
                                 return;
                              }
                              else {
 
                                 // Save the selected MetroID in cookies and navigate to the search page
                                 $wait.removeClass('hide');
                                 Beazer.Cookies.SetSearchSettingsMetro(val);
                                 window.location.href = $selected.attr('data-url');
 
                              }
 
                         });
 
                     };
                     if (document.readyState == 'complete') {
                         init();
                     }
                     else {
                         $(document).ready(init);
                     }
                 })();
      </script>

     <script>
      // Instagram Script
     var desktopInstagramMediaCount = 0;
     var desktopInstagramMediaTotal = 18;
     var mobileInstagramMediaCount = 0;
     var mobileInstagramMediaTotal = 3;
 
     $.ajax({
         type: 'GET',
         cache: false,
         url: '/SocialMedia/InstagramInit',
         data: {}
         , success: function (response, textStatus, jqXHR) {
             //Load Twitter tweets into embedded UI
             instagramInit(response);
 
         }, error: function (data, textStatus, jqXHR) {
             // Show error message
             smError('smInstagram');
         }
     });
 
     function instagramInit(response) {
 
         if (!response || !response.data || response.data.length === 0) {
             // No data provided exit init function
             smError('smInstagram');
             return;
         }
 
         var data = response.data;
 
         // Desktop 
         for (var t = 0; t < data.length && t < desktopInstagramMediaTotal; t++) {
 
             var imageUrl = data[t].images.thumbnail.url;
             var altText = 'Beazer Instagram Image';
 
             if (data[t].caption && data[t].caption.text) {
                 altText = data[t].caption.text;
             }
 
             $('.append_desktop.' + t).append('<div class="ig-media">' +
                 '<span class="bzh-icon instagram msm_watermark"></span>' +
                 '<img src="' + imageUrl + '" class="responsive-image" ' +
                 'alt="' + altText + '" />' +
                 '</div>');
             desktopInstagramMediaCount++;
 
             if (desktopInstagramMediaCount === desktopInstagramMediaTotal) {
                 $('.desktop_container .progress').css('display', 'none');
                 $('.desktop_container .smInstagram .cta').css('display', 'block');
             }
         }
 
         // Mobile 
         for (var t = 0; t < data.length && t < mobileInstagramMediaTotal; t++) {
 
             var imageUrl = data[t].images.standard_resolution.url;
             var altText = 'Beazer Instagram Image';
 
             if (data[t].caption && data[t].caption.text) {
                 altText = data[t].caption.text;
             }
 
             $('.mobile_container .smInstagram .append_container').append('<div class="ig-media append_item">' +
                 '<span class="bzh-icon instagram msm_watermark"></span>' +
                 '<img src="' + imageUrl + '" class="responsive-image" ' +
                 'alt="' + altText + '" />' +
                 '</div>');
             mobileInstagramMediaCount++;
 
             if (mobileInstagramMediaCount === mobileInstagramMediaTotal) {
                 smProcessInstagramMobile();
             }
         }
     }
 
     function smProcessInstagramMobile() {
 
         $( '.mobile_container .smInstagram .append_item' ).each(function(index) {
             if (index < 2) {
                 $(this).css('display', 'block');
             } else {
                 $(this).css('display', 'none');
             }
         });
         $('.mobile_container .smInstagram .instagram.msm_watermark').css('display', 'block');
         $('.mobile_container .smInstagram .progress').css('display', 'none');
         $('.mobile_container .smInstagram .smContent').css('display', 'block');
         $('.mobile_container .smInstagram .show_more_social').css('display', 'block');
     }

         $('.mobile_container .smInstagram .bzh_more_fadeout').css('display', 'none');
         $('.mobile_container .smInstagram .cta').css('display', 'block');
     });
