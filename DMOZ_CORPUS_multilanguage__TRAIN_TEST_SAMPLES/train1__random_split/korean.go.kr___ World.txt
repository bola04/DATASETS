 var currentPopupList;
 function mainPopup(){
     $.ajax({
         type:'POST',
         url:"/front/popup/popupList.json;front=D8FD1C58B77DB30D95F1F88AA88F992E",
         dataType : "json",
         data:"",
         success:function(data){
             if(data.popupList.length > 0){
 				currentPopupList = data;
 	            for(var i = 0; i < data.popupList.length; i++){
 	            	var co = $.cookie('mainpopup_'+data.popupList[i].popup_seq);
 	            	if (co == null){
 	            	    //팝업
 	            	    $("#dialog"+i).dialog({
 	            	        title:data.popupList[i].popup_title,
 	            	        width:data.popupList[i].popup_width+30,
 	            	        height:data.popupList[i].popup_height+45, 
 	            	        resizable : false, 
 	            	        position: [data.popupList[i].popup_x,data.popupList[i].popup_y],
 	            	        dialogClass: 'no-close',
 	            	        open: function() {
 	            	        	if(data.popupList[i].popup_controll_yn=="N"){
             	        		    $(this).after('<div class="check_close"><a href="#" class="main_pop_close">닫기');
 	            	        	}else{
 	            	                $(this).after('<div class="check_close"><input type="hidden" id="checkCookie'+data.popupList[i].popup_seq+'" name="checkCookie'+data.popupList[i].popup_seq+'" value="'+data.popupList[i].popup_seq+'"><input type="checkbox" id="dialog2_check'+data.popupList[i].popup_seq+'"><label for="dialog2_check'+data.popupList[i].popup_seq+'">오늘은 더 이상 열리지 않음.<a href="#" class="main_pop_close">닫기');
 	            	        	}
 	            	        }
 	            	    });	            		
 	            	}
 	            }
             }
         }
     });
 }
 
 // 다시 열지 않기 기능이 정상적으로 동작하지 않아 직접적으로 팝업창의 체크박스로 접근하여 제어하도록 변경
 $('.main_pop_close').live('click',function () {
 	for(var i = 0; i < currentPopupList.popupList.length; i++){
 		if($("input[type='hidden'][id='checkCookie"+currentPopupList.popupList[i].popup_seq+"']").val()==currentPopupList.popupList[i].popup_seq){
 			if($("#dialog2_check"+currentPopupList.popupList[i].popup_seq).attr("checked")=="checked"){
 				$.cookie('mainpopup_'+currentPopupList.popupList[i].popup_seq, 'close', {expires:1, path:'/'});
 			}
 		}
 	}
     $(this).closest('.check_close').prev('.main_pop').dialog('close');
 });
