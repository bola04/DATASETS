   <script type="text/javascript">
    var loginBegin = function() {
         var submitButton = $('#loginForm button[role=loginSubmit]');
 
         submitButton.data('old-html', submitButton.html());
         submitButton.html('<img src="/Images/ajax-loader-small.gif" alt="Loading" /> Loading...');
         submitButton.attr('disabled', 'disabled');
     };
     var loginSuccess = function (data) {
         if(data.Result == true) {
             // Login valid!
                 
             // Now check if there is a redirect
             if (data.Redirect != null) {
                 if (data.UseModalWindow == true) {
                     $('#loginModal').load(data.Redirect);
                 }
                 else {
                     location.href = location.href = data.Redirect;
                 }
             }
             else if (data.IsPasswordResetMode) {
                 $('.modal-body').text('Le istruzioni per il recupero password sono state inviate via mail!');
                 $('#loginForm').attr('action', '#');
                 $('#loginForm button[role=loginSubmit]').text('Ok');
                 $('#loginForm button[role=loginSubmit]').removeAttr('disabled');
                 $('#loginForm button[role=loginSubmit]').click(function () {
                     $('#loginModal').modal('hide');
                 });
             }
             else {
                 if (location.href.indexOf('#')) {
                     location.href = location.href.substring(0, location.href.indexOf('#'));
                 } else {
                     location.href = location.href;
                 }
             }
         } else {
             // Login invalid!
                 
             var submitButton = $('#loginForm button[role=loginSubmit]'),
                 loginModal = $('#loginModal');
                 
             submitButton.html(submitButton.data('old-html'));
             submitButton.removeAttr('disabled');
             shakeElement(loginModal);
 
             // Print the error message
             for (var error in data.Errors) {
                 $(createAlertElement(data.Errors[error], 'error')).prependTo('#loginModal .modal-body');
             }
         }
     };
   </script>
