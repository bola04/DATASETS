                 serviceUrl: '/cms/render/live/en/modules/delta-templates/2.2.48/templates/base/navigation/main-navbar.json',
                 paramName: 'q',
                 transformResult: function(response, originalQuery) {
                     response = JSON.parse(response);
                     $.map(response.products, function(product) {
                     	if(originalQuery.toUpperCase()==product.name){
                     		matchedProduct=product.name;
                     	}
                     });
                     return {
                         suggestions: $.map(response.products, function(product) {
                             return { value: product.name + ' - ' + product.descriptionWithCollection, data: { grouping: 'Products', name: product.name } };
                         }).concat($.map(response.parts, function(part) {
                                 return { value: part.name + ' - ' + part.descriptionWithCollection, data: { grouping: 'Repair Parts', name: part.name } };
                             })
                         )
                     };
                 },
                 formatResult: function (suggestion, currentValue) {
                     var pattern = '(' + currentValue.replace(/[\-\[\]\/\{\}\(\)\*\+\?\.\\\^$\|]/g, "\$&") + ')';
 
                     return suggestion.value
                         .replace(new RegExp(pattern, 'gi'), '<strong>$1<\/strong>')
                         //                            .replace(/&/g, '&amp;')
                         //                            .replace(/</g, '&lt;')
                         //                            .replace(/>/g, '&gt;')
                         .replace(/"/g, '&quot;')
                         .replace(/&lt;(\/?strong)&gt;/g, '<$1>');
                 },
                 groupBy: 'grouping',
                 appendTo: $autocompletInput.closest('form'),
                 containerPosition: 'absolute',
                 maxHeight: 'auto',
                 onSelect: function (suggestion) {
                     gtmNavSearchClick(suggestion, '/home.goto2.do?type=product&value=' + suggestion.data.name)
                 },
                 width: 'auto !important'
             });
         });

             		window.location.href = gotoUrl+'?type=product&value='+matchedProduct;
             		return false;
             	}
             }
          });
     });
