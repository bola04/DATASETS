   <script type="text/javascript">
    $(function () {
 
         $("#action-achat").click(function () {
             $("#input-type").val("vente")
         });
 
         $("#action-location").click(function () {
             $("#input-type").val("loca")
         });
 
         var cities = [];
 
         var updateVilles = function () {
             $("#input-villes").val(cities.join(","))
         }
 
         var addCity = function(city){
             var cityCtn = $('<span style="cursor:pointer"><a class="remove-city-bt" data-city="' + city + '"><i class="icon-remove"></i></a>' + city + '</span>')
             cityCtn.click(function () {
                 $(this).remove();
                 indexOfCity = cities.indexOf(city);
                 cities.splice(indexOfCity);
                 updateVilles();
             })
             $(".city-add").append(cityCtn);
             cities.push(city);
 
             updateVilles();
         };
 
         $("#appendedInputButton").typeahead({
             source: ["Plaisir", "LES CLAYES SOUS BOIS", "Les Clayes-sous-Bois", "Bailly", "Neauphle-le-Château", "PLAISIR", "BOIS D' ARCY", "Houdan", "Fontenay-le-Fleury", "BEYNES", "Villepreux", "JOUARS PONTCHARTRAIN", "LE PECQ", "MARCQ", "AULNAY SUR MAULDRE"],
             updater: addCity
         });
 
     })
   </script>
