        $(document).ready(function() {
         var $slider = $('#' + 'carousel-home-banners' + ' .slick-carousel');
         $slider.on('init', function (event, slick) {
             // Inhalt ist initial versteckt damit nicht alle Slides untereinander angezeigt werden
             $slider.find('.slide').removeClass('hidden');
             // Bilder der benachbarte Slides laden
             slickLoadSlideNeighbors($slider, slick, slick.currentSlide);
         });
         $slider.on('beforeChange', function (event, slick, currentSlide, nextSlide) {
             slickLoadSlideNeighbors($slider, slick, nextSlide);
             if ($slider.find('img.slick-loading[data-lazy]').length === 0) {
                 // wenn alle Bilder geladen wurden müssen wir nicht weiter prüfen
                 $slider.off('beforeChange');
             }
         });
         $slider.slick({
             lazyLoad: 'ondemand',
             arrows: true,
             appendArrows: $('#' + 'carousel-home-banners' + ' .carousel-control-container'),
             draggable: false,
             dots: true,
             touchMove: true,
             infinite: true,
             cssEase: 'ease',
             slidesToShow: 1,
             slidesToScroll: 1,
             touchThreshold: 5,
             autoplay: true,
             autoplaySpeed: 5000,
         });
     });
 
     /**
      * Lädt die Bilder der benachbarten Slides nach
      *
      * @param $slider
      * @param slick
      * @param targetSlide
      */
     function slickLoadSlideNeighbors($slider, slick, targetSlide) {
         var leftSlide = targetSlide - 1;
         if (leftSlide < 0) {
             leftSlide = slick.slideCount - 1;
         }
         var rightSlide = targetSlide + 1;
         if (rightSlide > slick.slideCount) {
             rightSlide = 0;
         }
         slickLoadSlideImage($slider, leftSlide);
         slickLoadSlideImage($slider, rightSlide);
     }
 
     /**
      * Lädt das Bild zum angegebenen Slide nach
      *
      * @param $slider
      * @param targetSlide
      */
     function slickLoadSlideImage($slider, targetSlide) {
         var $targetImage = $slider.find('.slide[data-slick-index="' + targetSlide + '"] img.slick-loading[data-lazy]');
         $targetImage
             .attr('src', $targetImage.data('lazy'))
             .removeAttr('data-lazy')
             .removeClass('slick-loading');
     }
