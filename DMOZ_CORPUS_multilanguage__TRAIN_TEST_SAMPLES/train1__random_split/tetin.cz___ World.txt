  function vodnitoky_chart_mini1_data()
  {
 	if (jQuery.browser.msie && parseInt(jQuery.browser.version,10)<=7)
 		var sServer='http://www.tetin.cz/modules/vodnitoky/stanice_data.php?';
 	else
 		var sServer='https://vodnitoky.gc-system.cz/modules/vodnitoky/graf_data_mini.php?';
 	sServer+='stanice=476&typ=full&zalozka=mini&lang=cs&time=2018-07-29T11:30:00';
 
 	jQuery.ajax({
 		url: sServer,
 		type: 'GET',
 		dataType: 'html',
 		success: function(tsv){
 			var lines = [],
 			listen = false,
 			datum,
 			vysky = [],
 			vyska_min = 200,
 			vyska_max = -200;
 
 			var control1=/^476>/;
 			var control2=/<476$/;
 			if (control1.test(tsv) && control2.test(tsv))
 			{
 				tsv = tsv.replace(control1,'');
 				tsv = tsv.replace(control2,'');
 				tsv = tsv.split(/\n/g);
 
 				jQuery.each(tsv, function(i, line) {
 					data = line.split(/\|/);
 					if (data.length==2)
 					{
 						var reg=/([0-9]+)/g;
 						var parse_date=data[0].match(reg);
 						datum=Date.UTC(parseInt(parse_date[0],10), parseInt(parse_date[1],10)-1, parseInt(parse_date[2],10), parseInt(parse_date[3],10), parseInt(parse_date[4],10), parseInt(parse_date[5],10));
 						if (data[1].length>0)
 						{
 							var fVyska=parseFloat(data[1], 10);
 							if (vyska_min>fVyska) { vyska_min=fVyska; }
 							if (vyska_max<fVyska) { vyska_max=fVyska; }
 
 							vysky.push([
 								datum,
 								fVyska
 							]);
 						}
 					}
 				});
 			}
 			vodnitoky_chart_mini1.hideLoading();
 			fMin=0;
 			fMax=10*Math.ceil(vyska_max/10);
 	 		vodnitoky_chart_mini1.yAxis[0].setExtremes(fMin,fMax,false,false);
 			vodnitoky_chart_mini1.series[0].setData(vysky);
 		},
 		error: function(jqXHR, textStatus){
 			alert(textStatus);
 		}
 	});
 }
          </script>

          function kaPrevMonth()
 {
 	var m = kaGetSelectedMonth.apply(this);
 	var y = kaGetSelectedYear.apply(this);
 	var minY = Math.min(getSelectOptionValues(this.find('#year')));
 
 	if(m === 1){
 		m = 12;
 		y -= (y === minY ? 0 : 1);
 	}else{
 		m -= 1;
 	}
 
 	kaRedrawCalendar.apply(this, [y, m]);
 }
 
 function kaNextMonth()
 {
 	var m = kaGetSelectedMonth.apply(this);
 	var y = kaGetSelectedYear.apply(this);
 	var maxY = Math.max(getSelectOptionValues(this.find('#year')));
 
 	if(m === 12){
 		m = 1;
 		y += (y === maxY ? 0 : 1);
 	}else{
 		m += 1;
 	}
 
 	kaRedrawCalendar.apply(this, [y, m]);
 }
 
 function getSelectOptionValues($select){
 	var values = [];
 	$select.find('opiton').each(function(){
 		values.push(this.value);
 	});
 
 	return values;
 }

 function kaRedrawCalendar(year, month){
 	var $form = this;
 	if(year !== undefined){
 		$form.find('#year option[value="' + year + '"]').prop('selected', true);
 	}
 	if(month !== undefined){
 		$form.find('#month option[value="' + month + '"]').prop('selected', true);
 	}
 
 	jQuery.ajax({
 		url: 'modules/kalendar/ax_kalendar.php'
 		, data: $form.serialize()
 		, type: 'POST'
 		, beforeSend: function(xhr, settings){}
 		, success: function(data){
 			if(data.length)
 				$form.find('#kal_table table').replaceWith(data);
 		}
 		, complete: function(){}
 	});
 }
