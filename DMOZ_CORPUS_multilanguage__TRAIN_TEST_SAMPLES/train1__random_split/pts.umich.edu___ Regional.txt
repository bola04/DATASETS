  var currentTime =

  // Setup From/To input field interaction.
  function setupGFormInput(field, message) {
    var onFocus = function() {
      if (this.value == message) {
        this.className = 'g-wp';
        this.value = '';
      }
    };
    var onBlur = function() {
      if (this.value == '') {
        this.className = 'g-empty';
        this.value = message;
      }
    };
    field.onblur = onBlur;
    field.onfocus = onFocus;
    onBlur.apply(field);
  };
  // Builds the destination URL
  function submitGForm() {
    var loc = '//www.google.com/maps?ie=UTF8&f=d&';
    var dirflg = 'r';
    for (var i = 0; (i < document.gForm.length - 1); i++) {
      var e = document.gForm[i];
      if (e.name == 'saddr') {
        loc += e.name + '=' +
               encodeURIComponent(e.value) + (zip ? '+near+' + zip : '') + '&';
      } else if (e.name == 'pref') {
        dirflg += e.value;
      } else {
        loc += e.name + '=' + encodeURIComponent(e.value) + '&';
      }
    }
    loc += 'dirflg=' + encodeURIComponent(dirflg);
 
    location.href=loc;
    return false;
  }

The department oversees Waste Management, Moving and Trucking and Material Services.  LTP also operates and maintains over 27,000 parking spaces, provides campus bus service 360 days a year with roughly 55 buses serving 12 routes, maintains a fleet of approximately 1,000 vehicles of various types and uses, and offers vanpool and other transportation alternatives to our campus community.
