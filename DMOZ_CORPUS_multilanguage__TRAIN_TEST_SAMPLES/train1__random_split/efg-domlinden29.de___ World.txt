         var ImageCount = 6
         var ImageActivePosition = 1

           if(LastWidth != $(this).width()){
             LastWidth = $(this).width()
             $("#Layout_Bildermenu").css("width", (LastWidth - 2) + "px")
             if(LastWidth - 2 > ImageWidth + (ImageCount - 1) * 50) ImageWidth1 = 310
             else ImageWidth1 = LastWidth - 2 - (ImageCount - 1) * 50
             ImageWidth2 = (LastWidth - 2 - ImageWidth1) / (ImageCount - 1)
             
             $("div#Layout_Bildermenu ul li a").css("width", ImageWidth2 + "px")
             $("div#Layout_Bildermenu ul li:nth-child(" + ImageActivePosition + ") a").css("width", ImageWidth1 + "px")
           }
         });
       
         $('#Layout_gesamt').resize();

           if($("div#Layout_Bildermenu ul li:nth-child(" + ImageActivePosition + ") a").text() != $(this).text()){
