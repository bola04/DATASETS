 	function position_images() {
 
 		if (num_slides > 5) {
   		for (var i = 0; i < 3*num_slides; i++) {
         $('#gcui_book_widget-image' + i).css('left', distance * i + 270 - (num_slides * distance) );
       }
   		fix_positions();
 		} else {
 			for (var i = 0; i < 3; i++) {
         $('#gcui_book_widget-image' + i).css('left', distance * i + 270 );
       }
 			for (var i = 3; i < num_slides; i++) {
 				$('#gcui_book_widget-image' + i).css('left', '-200px');
 			}
 		}
 	}
 	
 	function fix_positions() {
 
 		for (var i = 0; i < 3*num_slides; i++) {
 			
 			// Get the current left position.
       var this_pos = $('#gcui_book_widget-image' + i).position();
 
       // If too low, move it to the other end.
       if (this_pos.left < -(num_slides/2)*distance) {
     	  $('#gcui_book_widget-image' + i).css('left', this_pos.left + 3*num_slides*distance );
       }
 
       // If too high, move it to the other end.
       if (this_pos.left > num_slides*distance) {
     	  $('#gcui_book_widget-image' + i).css('left', this_pos.left - 3*num_slides*distance );
       }
 
     }
   }
 
 	
 	
 	function pop_window(nnum) {
 		
 		// Get the left position of the image div.
     var left_pos2 = $('#' + nnum).position();
     var left_pos = left_pos2.left;
     
     // Get the width's calculated.
 		var my_width = $('#' + nnum + ' img').width();
 		var popup_width = $('.gcui_book_widget-popup').width();
 		
 		// This is gonna be the pop-up's left position.
 		var new_left = Math.floor(left_pos + (my_width/2) - (popup_width/2) - 45);
 		
 		// If it would stick out, put it on the other side.
 		if ( new_left + popup_width > (860-224) ) {
 			new_left = (860-224) - 65;
 		}
 
 		// Show the popup.
 		$('.gcui_book_widget-popup').css('left', new_left  + "px");
 		$('.gcui_book_widget-popup').show();
 
 		// Fill out the blurb.
 		var nnnum = parseInt(nnum.substring(22));
 		document.getElementById('gcui_book_widget-popup').innerHTML = unescape(blurbs[(nnnum % num_slides)]);
 		
 	}
