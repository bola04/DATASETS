         function QuickSearch() {
             var isAny = false;
             var beds = $('#bedrooms').find(":selected").text();
             //if studio is selected - set value to 0
             if (beds == "Studio") beds = 0;
             if (beds == "Any") {
                 isAny = true;
             }
             var location = $('#location').val();
             var locationId = $('#locationId').val();
             var rent = $('#rent').val();
             if (rent == "") rent = "500";
 
             //check location is valid first
             var isValidPostcode = IsValidPostcode(location);
             //a valid location has a location id or is a postcode
             if ((locationId == null || locationId == "") && !isValidPostcode) {
                 if ($('.alert-danger').length == 0) {
                     $(".well").prepend("<div class=\"alert alert-danger\"><p>Please enter a location</p></div>");
                 }
                 return false;
             }
 
             //for mixpanel
             var quickSearchOptions = {
                 Location: location,
                 Bedrooms: beds,
                 MaxRent: rent
             };
             sendEvent("Quick Search From Home Page Event Desktop", quickSearchOptions, null);
 
             if (!isAny) {
                 url = "/Search/Results/?BedroomsMin=" + beds + "&BedroomsMax=" + beds + "&locationName=" + location + "&locationId=" + locationId + "&rentMax=" + rent + "&distance=20.0" + "&latitude=null&longitude=null";
             } else {
                 url = "/Search/Results/?BedroomsMin=" + 0 + "&BedroomsMax=" + 9 + "&locationName=" + location + "&locationId=" + locationId + "&rentMax=" + rent + "&distance=20.0" + "&latitude=null&longitude=null";
             }
 
 
             window.location.href = url;
             return false;
         }
