        jQuery(document).ready(function() {
 		
 		jQuery('.iosSlider').iosSlider({
 			autoSlide: true,
 			snapToChildren: true,
 			desktopClickDrag: true,
 			infiniteSlider: true,
 			snapSlideCenter: true,
 			onSlideChange: slideChange,
             onSliderLoaded:slideLoad,
 			navPrevSelector:'.prev',
 			navNextSelector: '.next'
 		});

     function slideLoad(args) {
         
         jQuery('.slider .item'+(args.currentSlideNumber)).find('.text2').show();
         jQuery('.slider .item'+(args.currentSlideNumber)).find('.selectorShadow').hide();
         
         if (args.data.numberOfSlides > 1) {
             jQuery('.slider .item'+(args.data.numberOfSlides)).find('.selectorShadow').show();
             jQuery('.slider .item'+(args.data.numberOfSlides)).find('.text2').hide();
         }
         
         jQuery('.slider .item'+(args.currentSlideNumber + 1)).find('.selectorShadow').show();
         jQuery('.slider .item'+(args.currentSlideNumber + 1)).find('.text2').hide();
         
         jQuery('.iosSlider').css('visibility','hidden');
         setTimeout(function(){            
            jQuery(window).resize();
            jQuery('.iosSlider').css('visibility','visible'); 
         },2000);
         
     }
 	function slideChange(args) {
 	   
         //console.log(args);
        
 		jQuery('.indicators .item').removeClass('selected');
 		jQuery('.indicators .item:eq(' + (args.currentSlideNumber - 1) + ')').addClass('selected');
         
         jQuery('.slider .item').find('.text2').hide();
         
         // Current
         var currentNumber = args.currentSlideNumber;
         if (args.currentSlideNumber <= 0) {
             currentNumber = args.data.numberOfSlides+currentNumber;
         }
         
         jQuery('.slider .item'+(currentNumber)).find('.text2').show();
         jQuery('.slider .item'+(currentNumber)).find('.selectorShadow').hide();
         //console.log('C ='+currentNumber);
         
         // Next
         var prevSlideNumber = args.prevSlideNumber; //currentNumber-1;
         if (prevSlideNumber == 0) {
             prevSlideNumber = args.data.numberOfSlides;
         }
         jQuery('.slider .item'+prevSlideNumber).find('.selectorShadow').show();
         jQuery('.slider .item'+prevSlideNumber).find('.text2').hide();
         //console.log('Prev ='+prevSlideNumber);
         
         // Prev
         if (currentNumber == 1) { //args.data.numberOfSlides) {
             currentNumber = args.data.numberOfSlides;
         } else {
            currentNumber = currentNumber-1; 
         }
         jQuery('.slider .item'+(currentNumber)).find('.selectorShadow').show();
         jQuery('.slider .item'+(currentNumber)).find('.text2').hide();
         
 	
 	}
