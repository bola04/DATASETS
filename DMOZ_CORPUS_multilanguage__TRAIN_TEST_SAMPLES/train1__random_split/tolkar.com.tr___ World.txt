             function AnimasyonBasla() {
                 animasyon = setInterval(function () {
                     var index = $(".divSlider .divSlides .divSlide").filter(function () { return $(this).css("display") == "block" }).index();
                     nesne.eq(index).fadeOut(zamanEffekt);
                     //$(".divSlider .divSlogan").eq(index).fadeOut(zamanEffekt);
                     if (index == nesne.length - 1) {
                         index = 0;
                     } else {
                         index++;
                     }
                     nesne.eq(index).fadeIn(zamanEffekt);
                     //$(".divSlider .divSlogan").eq(index).fadeIn(zamanEffekt);
 
                     $(".spanSayac").css({ "background": "#0a2e4b" });
                     $(".spanSayac").eq(index).css({ "background": "#ad1a2a" });
                 }, zamanAnimasyon);
             }
             function AnimasyonHareket(tipi) {
                 var index = $(".divSlider .divSlides .divSlide").filter(function () { return $(this).css("display").toLowerCase() == "block" }).index();
                 if (tipi == 'SolaGit') {
                     indexEski = index;
                     if (index - 1 < 0)
                         index = nesne.length - 1;
                     else
                         index--;
                 }
                 else if (tipi == 'SagaGit') {
                     indexEski = index;
                     if (index + 1 >= nesne.length)
                         index = 0;
                     else
                         index++;
                 }
                 clearInterval(animasyon);
                 $(".spanSayac").css("background", "#0a2e4b");
                 nesne.eq(indexEski).fadeOut(zamanEffekt);
                 nesne.eq(index).fadeIn(zamanEffekt);
                 $(".spanSayac").eq(index).css("background", "#ad1a2a");
                 AnimasyonBasla();
             }
