 		$('html').css("overflow", "hidden");
 		$('#overlay').show();

 	});
 
 	// Navigation Slide Out
 	$('#overlay, .responsive-nav-close').click(function() {
 		$('nav').removeClass('slide-in');

 		return false;
 	});
