     function getQueryVariable(variable) {
         var query = window.location.search.substring(1);
         var vars = query.split("&");
         for (var i = 0; i < vars.length; i++) {
             var pair = vars[i].split("=");
             if (pair[0] == variable) {
                 return pair[1] == null ? "" : pair[1];
             }
         }
         return null;
     }

     function getNodeText(node) {
         var text = "";

         if (node.firstChild) text = node.firstChild.nodeValue;

       <h3>
        Features
       </h3>
       <p>
        <strong>
         Compatibility:
        </strong>
        iPhoneModem has been tested on all iPhoneOS and iOS devices since version
                 2.0
                 (September 2008)
       </p>
       <p>
        <strong>
         Sharing:
        </strong>
        the internet connection can be used by several devices.
       </p>
       <p>
        <strong>
         Security:
        </strong>
        you can use WiFi security to protect the shared internet connection.
       </p>
       <p>
        <strong>
         Speed:
        </strong>
        full 7.2 Mbps 3.5G speeds can be achieved (with optimal network reception).
       </p>
       <p>
        <strong>
         Usage:
        </strong>
        iPhoneModem and the helper app on the computer provide a simple one-click
                 solution to
                 internet sharing.
       </p>
       <h3>
        How does it work
       </h3>
       <p>
        iPhoneModem has two sides: the Modem application on the iPhone and the iPhoneModem helper for the Mac or
                 the PC.
       </p>
       <p>
        On the computer, the helper application creates a new computer-to-computer (or ad-hoc) WiFi network and
                 configures the system preferences to use the iPhone as an Internet gateway and proxy.
       </p>
       <p>
        On the iPhone, the application opens a routing engine, DHCP, DNS, HTTP, HTTPS and SOCKS proxies and
                 connects to the helper on the computer.
       </p>
