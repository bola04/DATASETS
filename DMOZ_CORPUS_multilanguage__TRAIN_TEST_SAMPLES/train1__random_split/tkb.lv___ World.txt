 		data = {
 			"_token": $(this).find('[name=_token]').val(),
 			"name": $(this).find('[name=name]').val(),
 			"phone": $(this).find('[name=phone]').val(),
 			"email": $(this).find('[name=email]').val(),
 			"message": $(this).find('[name=message]').val(),
 		};
 		if ($(this).find('[name=agree]:checked').length) {
 			data.agree = 1;
 		}
 		if ($(this).find('[name=title]').val()) {
 			data.title = $(this).find('[name=title]').val();
 		}
 	    $this = $(this);
 	    $this.find('.error').removeClass('error')
 	    $this.find('.title-error').removeClass('title-error')
 	    $this.next().removeClass('display');
 	    $this.find('[name="agree"]').parent().parent().css('color', 'inherit');
 		$.ajax({
 			url: 'http://www.tkb.lv/ajax/contact', 
 			data: data,
 			method: 'POST', 
 			success: function(data) {
 		    	$this.parent().parent().addClass('success').removeClass('error');
 			}, 
 			error: function(xhr) {
 				data = JSON.parse(xhr.responseText);
 			    for(i in data) {
 			    	$item = $this.find('[name="'+i+'"]');
 			    	if (i == 'title') {
 			    		$item.next().addClass('title-error');
 			    	}
 			    	if (i == 'agree') {
 			    		$this.find('[name="agree"]').parent().parent().css('color', '#9C0E33');
 			    	}
 			    	$item.addClass('error');
 			    }
 	    		$this.next().addClass('display');
 			}
 		});
 	})
   </script>
