 						},
 						switchTo: function (index) {
 							var target = $j(BigTop.container).find('.banner-' + parseInt(index));
 							if (!target.length || target.hasClass('active')) {
 								return
 							}
 
 							var active = $j(BigTop.container).find('.banner.active');
 
 							active.css({"z-index": 200});
 							target.css({"z-index": 100});
 
 							active.fadeOut(function () {
 								$j(this).removeClass('active');
 							});
 							target.fadeIn(function () {
 								$j(this).addClass('active');
 							});
 
 							var activeDot = $j(BigTop.container).find('.dot.active').first();
 							var nextDot = $j(BigTop.container).find('.dot-' + parseInt(index)).first();
 
 							activeDot.removeClass('active');
 							nextDot.addClass('active');
 						},
 						screenSizeChanged: function () {
 							BigTop.fixSizes();
 						},
 						fixSizes: function () {
 							$j(BigTop.container).height($j(BigTop.banners + ":visible").first().height());
