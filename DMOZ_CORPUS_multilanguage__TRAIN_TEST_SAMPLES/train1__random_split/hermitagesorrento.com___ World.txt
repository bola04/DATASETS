    function getList() {
   var cemail = document.getElementById('smail').value;
   sendReq("hwndsignup.php", "cid="+cemail, "post", loadList)
 }
 function loadList()
 {
 	//alert(areq.readyState);
 	if (areq.readyState == 4) 
 	{
 		if (areq.status == 200) 
 		{
 			//alert(areq.responseText);
 			var questions = areq.responseText;
 			if(questions=="yes")
 			{
 				document.getElementById("msgdiv").style.display='';
 			}
 		}
 	}
 }
 function sendReq(url,params,HttpMethod, callBack)
 {
 	if (!HttpMethod)
 	{
 		HttpMethod="POST";
 	}
 	areq=initXMLHTTPRequest();
 	if (areq)
 	{
 		areq.onreadystatechange=callBack;
 		areq.open(HttpMethod,url,true);
 		areq.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
 		areq.send(params);
 	}
 }
 //****************************creating the XMLHTTPRequest object
 function initXMLHTTPRequest()
 {
 	var xRequest=null;
 	if (window.XMLHttpRequest)
 	{
 		xRequest=new XMLHttpRequest();
 	} else if (window.ActiveXObject)
 	{
 		xRequest=new ActiveXObject
 		("Microsoft.XMLHTTP");
 	}
 	return xRequest;
 }
 function nfcheck()
 {
 	var bval=0;
 	if(document.nform.smail.value=="")
 	{
 		document.getElementById("smail").style.border='thin solid #FF0000';
 		bval=bval+1;
 	}else{ 	
 		document.getElementById("smail").style.border='';
 	}
 	if(document.nform.smail.value!='') 
 	{
 		var emailPat=/^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/
 		var matchArray;	
 		var emailStr = document.nform.smail.value;	
 		if(emailStr.length > 0)
 		{  	  	
 			matchArray = emailStr.match(emailPat);
 			if (matchArray == null) 
 			{
 				document.getElementById("smail").style.border='thin solid #FF0000';
