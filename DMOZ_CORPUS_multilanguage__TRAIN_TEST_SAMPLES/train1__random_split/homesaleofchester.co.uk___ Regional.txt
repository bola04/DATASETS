            function remember( selector ){
 $(selector).each(
 function(){
 //if this item has been cookied, restore it
 var name = $(this).attr('name');
 if( $.cookie( name ) ){
 $(this).val( $.cookie(name) );
 }
 //assign a change function to the item to cookie it
 $(this).change(
 function(){
 $.cookie(name, $(this).val(), { path: '/', expires: 365 });
 }
 );
 }
 );
 }
