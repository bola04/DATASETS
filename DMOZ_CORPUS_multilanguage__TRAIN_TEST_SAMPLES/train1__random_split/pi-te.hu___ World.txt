        function opentura(turadir) {
 
         $.modal('<iframe src="./modules/rendezvenyeink/?mod='+turadir+'" height="540" width="1120" style="border:0;overflow:hidden">', {
             onOpen: function (dialog) {
 	            dialog.overlay.fadeIn('slow', function () {

 		            dialog.container.fadeIn('slow', function () {
 			        dialog.data.slideDown('slow');
             		});
             	});
 
                 $("#advertisement_tura").load("./modules/advertisement/index.php?mod="+turadir+"&style=showmore&style2=5&_="+Math.random()).delay(4000).fadeIn('slow');
 
             },
             onClose: function(dialog) {
                 $("#advertisement_tura").fadeOut('slow');
                 this.close();
 
             },
         	containerCss:{
         		height:550,
         		padding:0,
         		width:1120
         	},
