 								$GMap.empty();
 								
 								$GMap.gmap3
 								({
 									
 									longitude         : longitude,
 									latitude          : latitude,
 									zoom              : zoom,
 									
 									navigationControl : true,
 									mapTypeControl    : true,
 									scaleControl      : true,
 									scrollwheel       : false,
 									
 									allEvent          : function ()
 									{
 										if(!load_map) GmapMakerSwitch();
 										
 										// 中心の座標を取得
 										var position = this.getCenter();
 										latitude     = position.lat();
 										longitude    = position.lng();
 										
 										$("input[name='" + form_name + "[Map1]']:hidden").val(longitude);
 										$("input[name='" + form_name + "[Map2]']:hidden").val(latitude);
 										$("input[name='" + form_name + "[Zoom]']:hidden").val(this.getZoom());
 										
 										// 表示地図の表示範囲(左下(西南)と右上(東南))の座標を取得
 										var latlngBounds = this.getBounds();
 										var SWLatLng     = latlngBounds.getSouthWest();
 										var SWlat        = SWLatLng.lat();
 										var SWlng        = SWLatLng.lng();
 										
 										var NELatLng = latlngBounds.getNorthEast();
 										var NElat    = NELatLng.lat();
 										var NElng    = NELatLng.lng();

 										$("input[name='" + form_name + "[NElat]']:hidden").val(NElat);
 										$("input[name='" + form_name + "[NElng]']:hidden").val(NElng);
 										
 										load_map = true;
 									}
 									
 									
 																			
 								});

 							// Google Map マーカー表示切替
 							var GmapMakerSwitch = function()
 							{
 								// チェックが付いている種別を取得
 								var group_chk_ary = Array();
 								
 								$checks.each
 								(
 									function(i)
 									{
 										if($(this).attr('checked')) group_chk_ary.push($(this).val());
 									}
 								);
 								
 								// マーカーの要素を取得
 								var makerAry = $GMap.data("markers");
 								
 								// マーカーの表示切替処理
 								for(var i=0; i < makerAry.length; i++)
 								{
 									var on_flg = false;
 									
 									for(var j=0; j < group_chk_ary.length; j++)
 									{
 										if(makerAry[i].group_name == group_chk_ary[j]) on_flg = true;
 									}
 									
 									if(on_flg)
 									{
 										makerAry[i].setVisible(true);
 									} else {
 										makerAry[i].setVisible(false);
 									}
 								}
 							}
