     function checkIfParentItemExists(value) {
         var allParents = $(".breadcrumb span a");
         for (i = allParents.length - 1; i >= 0; i--) {
             var anchor = allParents[i];
             var anchorDecodedText = $("<div/>").html($.trim(anchor.innerHTML.toUpperCase())).text();
             if (anchorDecodedText == $.trim(value.toUpperCase())) {
                 return true;
             }
         }
         return false;
     }
 
     function changeCurrPageBreadcrum(value, className) {
         var breadcrm = $(".breadcrumb span");
         for (i = breadcrm.length - 1; i > 0; i--) {
             var span = breadcrm[i];
             if ($.trim(span.innerHTML) == '&gt;') {
                 breadcrm[i + 1].innerHTML = value;
                 breadcrm[i + 1].className = className;
                 return;
             }
         }
     }
 
     function changeLeftNavigation(title) {
 
         var leftNavTextItems = $('.left-nav-bar .menu .root .static ul li .menu-item-text');
         var leftNavAnchor = $('.left-nav-bar .menu .root .static ul li a');
         var leftNavli = $('.left-nav-bar .menu .root .static ul li');
 
         for (var i = 0; i < leftNavli.length; i++) {
 
             var textToMatch = '';
             if ($.browser.msie) {
                 textToMatch = leftNavTextItems[i].innerText;
             }
             else {
                 textToMatch = leftNavTextItems[i].textContent;
             }
 
             if (textToMatch.toLowerCase() == title.toLowerCase()) {
                 leftNavli.removeClass('selected');
                 leftNavAnchor.removeClass('selected');
                 leftNavli[i].className += ' selected';
                 leftNavAnchor[i].className += ' selected';
             }
         }
     }
