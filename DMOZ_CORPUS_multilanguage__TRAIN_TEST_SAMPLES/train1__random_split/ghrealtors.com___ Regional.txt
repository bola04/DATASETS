      function LocationSearchOnBlur(e) {
         var locationValue = $(this).val();
 
         if (locationValue.length <= 6) {
             return;
         }
 
         if (locationValue.indexOf(' and ', locationValue.length - 5) !== -1) {
             $(this).val(locationValue.substring(0, locationValue.length - 5))
         }
     }
 
     function GetAutoCompleteTypeWithArea(type, area) {
         var displayValue = '';
 
         if (type && type.length > 0) {
             displayValue = type;
         }
 
         if (area && area.length > 0) {
             displayValue += ' in ' + area;
         }
 
         if (displayValue.length > 0) {
             return '<span class="sugType">' + displayValue + '</span>';
         }
 
         return '';
     }
