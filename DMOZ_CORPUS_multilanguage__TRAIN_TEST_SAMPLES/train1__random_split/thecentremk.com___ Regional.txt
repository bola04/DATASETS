    function displayResults(results) {
           var ul = $("#nameSearchResults");
           $(ul).empty();
           for(var i = 0; i < results.length; i++) {
               var catResult = results[i];
               ul.append("<li><b>" + catResult.Category + "</b></li>");
               for(var j = 0; j < catResult.Results.length; j++) {
                   var found = catResult.Results[j];
                   ul.append("<li><a href=\"" + found.Url + "\">" + found.Name + "</a></li>");
               }
           }
           ul.append("<li><a href=\"search?searchTerms=" + encodeURIComponent($("#nameSearch").val()) + "\">Search all Information</a></li>");
           $('body').on('click', function(e){
             if ($(e.target).not('input#nameSearch')){
                 $("ul#nameSearchResults").empty();
                 $("input#nameSearch").empty();
             }
           });
       }
    
         $(function () {
 
             $("#searchBtn").click(function () {
                 window.location.href = "/search?searchTerms=" + encodeURIComponent($("#nameSearch").val()) + "";
             });
 
 
           $('#nameSearch')
               .on('keyup change',
                   function() {
                       var search = $(this).val();
                       var searchLen = search.length;
                       // Wait for at least 3 characters
                       if (searchLen >= 3) {
                           var searchTerms = encodeURIComponent(search);
   
                           $.get("/connect/api/search/querysummary", { "searchTerms": searchTerms })
                               .done(function(data) {
                                   displayResults(data);
                               });
   
                       }
   
                   });
       });
