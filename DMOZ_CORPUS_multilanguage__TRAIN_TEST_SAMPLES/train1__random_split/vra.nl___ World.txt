     return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, delimiter);
   }
 
   // Call API for search
   status = $.get(
     // Endpoint
     'https://api.raisely.com/v2/profiles/vracc?key=336bab90-6748-11e7-adb7-39ce7eff48b8',
 
     // Success
     function (data) {
       var goal, total, deadline;
 
       // Check if non-empty response
       if (!$.isEmptyObject(data.data)) {
         // Get goal field
         var goal = data.data.goal;
 
         // Get total field
         var total = data.data.total;
       }
 
       // Calculate deadline
       var max = new Date('September 30, 2017').getTime();
       var now = new Date().getTime();
       var diff = max - now;
       var deadline = Math.ceil(diff / (1000 * 60 * 60 * 24));
 
       // Determine delimiter for humanized numbers by language
       var delimiter = ($('html').attr('lang') == 'en') ? ',' : '.';
 
       // Update UI
