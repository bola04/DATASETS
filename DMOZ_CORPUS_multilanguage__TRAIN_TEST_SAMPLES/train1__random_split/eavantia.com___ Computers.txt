                 var input = $j(this);
                 if (input.val() == input.attr('placeholder')) {
                     input.val('');
                     input.removeClass('placeholder');
                 }
             }).blur(function () {
                 var input = $j(this);
                 if (input.val() == '' || input.val() == input.attr('placeholder')) {
