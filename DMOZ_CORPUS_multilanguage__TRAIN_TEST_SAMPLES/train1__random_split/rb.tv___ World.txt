         var type = 1;
     var formBlack;
     var searchInput;
     var timeOutId;
     $(function () {
         formBlack = $(".form-black");
         searchInput = $("#textSearch");
         searchInput.keyup(function (e) {
             if (e.which == 13) {
                 goSearch();
             }
             if (typeof (timeOutId) != 'undefined')
                 clearTimeout(timeOutId);
             timeOutId = setTimeout(function () {
                 Search.searchString = searchInput.val();
                 Search.Start();
             }, 400);
         });
 
     });
 
     function goSearch() {
         if ($("#textSearch").val().length == 0) return;
         $("#go-to-full-search").submit();
     }
 
     var Search = {
         searchString: "",
         type: null,
         resultHolder: null,
         resultHolderWrapper: null,
         url: "",
         languagePrefix: "ru",
         oldResultsArray: [],
         localization: [],
         isSameArray: false,
         SetOptions: function (_wrapper, _type, _url, _langPrfix, _localization) {
             Search.resultHolderWrapper = _wrapper;
             Search.type = _type;
             Search.languagePrefix = _langPrfix;
             Search.url = _url;
             Search.localization = _localization;
         },
         Start: function () {
             $.ajax({
                 type: 'Post',
                 url: Search.url,
                 data: {
                     s: Search.searchString,
                     type: Search.type,
                     take: 5
                 },
                 success: function (_data) {
                     Search.Init(_data);
                 }
             });
         },
         Init: function (_data) {
             Search.isSameArray = Search.IsSameAnswerArray(_data.SearchResult);
             Search.Text = _data.Text;
             if (Search.isSameArray) {
                 Search.CreateResultingHtml(Search.oldResultsArray, true);
                 Search.resultHolderWrapper.html($(Search.resultHolder).css('display', 'block'));
             }
             if (!Search.isSameArray) {
                 Search.oldResultsArray = [];
 
             }
             if (!Search.isSameArray || Search.oldResultsArray.length == 0) {
                 Search.CreateResultingHtml(_data.SearchResult, false);
             }
 
 
             if (!Search.isSameArray) {
                 var searchResultObj = $("#search-result-holder");
                 if (searchResultObj.length != 0) {
                     searchResultObj.show();
                     Search.SlideDown();
                 } else {
                     Search.SlideDown();
                 }
             } else {
                 if (!Search.resultHolderHasChild()) {
                     Search.CreateResultingHtml(Search.oldResultsArray, true);
                     Search.SlideDown();
                 }
             }
             Search.isSameArray = false;
         },
         FindResultingText: function (_resultingText, _match, _counter) {
             var substring = _match[0];
             var firstPart = _resultingText.substring(0, _match.index + _counter);
             var secondPart = _resultingText.substring(_match.index + _match[0].length + _counter);
             return firstPart + "<span class='HighLighted'>" + substring + "</span>" + secondPart;
         },
         HighLightMatchingResults: function (_text) {
             var splittedSearchInput = Search.Text.trim().split(" ");
             var matchingIndexArray = [];
             var temp = [];
             var step = 33;
             $.map(splittedSearchInput, function (_element) {
                 if ($.inArray(_element, temp) == -1) {
                     temp.push(_element);
                 }
             });
             splittedSearchInput = temp.sort(function (a, b) {
                 return a.length >= b.length ? -1 : 1;
             });
             var resultingText = _text;
             $.each(splittedSearchInput, function (_index, _word) {
                 if (_word.length == 0) return;
                 var regexp = new RegExp(_word, "gi");
                 var match;
                 while ((match = regexp.exec(_text)) != null) {
                     var counter = 0;
                     var greaterMatch = false;
 
                     $.each(matchingIndexArray, function (_index, _matchIndex) {
                         if (greaterMatch) return;
                         if (match.index >= _matchIndex.index && match.index <= _matchIndex.length && match[0].toLowerCase() != _matchIndex.word) {
                             greaterMatch = true;
 
                         }
                     });
                     if (!greaterMatch) {
                         matchingIndexArray.push({
                             index: match.index,
                             length: match.index + _word.length - 1,
                             word: _word
                         });
                         $.each(matchingIndexArray, function (_index, _matchIndex) {
                             if (match.index > _matchIndex.index) counter++;
                         });
                         resultingText = Search.FindResultingText(resultingText, match, counter * step);
                     }
                 }
             });
             return resultingText;
         },
         CreateResultingHtml: function (_results, _offPush) {
             Search.resultHolder = $("<div></div>");
             Search.resultHolder.attr("id", "search-result-holder");
             $.each(_results, function (_index, _result) {
                 if (!_offPush) {
                     Search.oldResultsArray.push(_result);
                 }
                 var searchResultHref = $("<a href='" + Search.Redirect(_result.Id, _result.Type) + "'></a>");
                 searchResultHref.css("color", "white");
                 var searchResult = $("<div></div>");
                 searchResult.addClass("searchResult");
                 var searchResultType = $("<span></span>");
                 searchResultType.addClass("search-result-type");
                 searchResultType.text(Search.GetLocalization(_result.Type)).appendTo(searchResult);
                 $("<div></div>").addClass("col s1 searchResultImage")
                                 .css({ "background-image": "url('" + _result.ImageUrl + "')" })//result.ImageUrl
                                 .appendTo(searchResult);
 
                 var searchResultWrapper = $("<div></div>").addClass("col s10");
                 var searchResultWrapperSpan = $("<span></span>")
                     .css({
                         "font-weight": "bold",
                         "max-width": "90%",
                         "display": "block"
                     })
                     .html(Search.HighLightMatchingResults(_result.Title));
                 var searchDescription = $("<div></div>").addClass('descriptionHolder').html(Search.HighLightMatchingResults(_result.Description));
                 var authorsLength = _result.Authors.length;
                 var categorysLength = _result.Categories.length;
                 var authors = "<span style='color:#999'>" + (_result.Type == 1 ? Search.GetLocalization("Speakers") : Search.GetLocalization("Authors")) + "</span> ";
                 //Авторы
                 $.each(_result.Authors, function (_index, _author) {
                     authors += Search.HighLightMatchingResults(_author);
                     if (authorsLength != 1 && _index != authorsLength - 1)
                         authors += ', ';
                 });
                 $("<div></div>").html(authors).appendTo(searchDescription);
                 //Категории
                 var categories = "<span style='color:#999'>" + Search.GetLocalization("Categories") + "</span> ";
                 $.each(_result.Categories, function (_index, _category) {
                     categories += _category.toLowerCase();
                     if (categorysLength != 1 && _index != categorysLength - 1)
                         categories += ', ';
                 });
                 $("<div></div>").html(categories).appendTo(searchDescription);
 
                 searchResultWrapperSpan.appendTo(searchResultWrapper);
                 searchDescription.appendTo(searchResultWrapper);
                 searchResultWrapper.appendTo(searchResult);
                 searchResult.appendTo(searchResultHref);
                 searchResultHref.appendTo(Search.resultHolder);
                 return;
 
             });
         },
         resultHolderHasChild: function () {
             return $("#search-result-holder").children().length != 0;
         },
         SlideDown: function () {
             Search.resultHolderWrapper.html(Search.resultHolder);
             if (Search.resultHolderHasChild()) {
                 Search.resultHolder.show();
             }
         },
         IsSameAnswerArray: function (_resultsArray) {
             if (Search.oldResultsArray.length == 0) { return false; }
             var tempNewResultsStringArray = $.map(_resultsArray, function (_result) {
                 return _result.Id + "(" + _result.Type + ")";
             });
             var tempOldResultsStringArray = $.map(Search.oldResultsArray, function (_result) {
                 return _result.Id + "(" + _result.Type + ")";
             });
             if (Search.oldResultsArray.length != tempNewResultsStringArray.length) return false;
             for (var i = 0; i < tempNewResultsStringArray.length - 1; i++) {
                 if ($.inArray(tempNewResultsStringArray[i], tempOldResultsStringArray) == -1) return false;
             }
             return true;
         },
         Redirect: function (_id, _type) {
             switch (_type) {
                 case 1:
                     {
                         return '/' + Search.languagePrefix + '/products/' + _id;
                     }
                 case 2:
                     {
                         return '/' + Search.languagePrefix + '/book/' + _id;
                     }
                 case 3:
                     {
                         return '/' + Search.languagePrefix + '/articles/' + _id;
                     }
                 case 4:
                     {
                         return '/' + Search.languagePrefix + "QA/Question/" + _id;
                     }
             }
         },
         GetLocalization: function (_text) {
             var defaultLocalization;
             var localization = $.map(Search.localization, function (_localicationObject) {
                 if (_localicationObject.type == "Default") defaultLocalization = _localicationObject.text;
                 if (_localicationObject.type == _text) return _localicationObject.text;
             });
             return localization.length == 0 ? defaultLocalization : localization;
         }
     }
