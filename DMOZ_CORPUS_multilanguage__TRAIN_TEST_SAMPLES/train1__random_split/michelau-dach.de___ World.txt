            Energetische Dachsanierungen gehören in die Hände von Spezialisten.           var firsttime;
 
     function isScrolledIntoView(elem) {
         var $window = jQuery(window),
             docViewTop = $window.scrollTop(),
             docViewBottom = docViewTop + $window.height(),
             elemTop = jQuery(elem).offset().top,
             elemBottom = elemTop + jQuery(elem).outerHeight();
         return ((elemBottom <= docViewBottom) && (elemTop >= docViewTop));
     }
 
     window.onload=function() {
       upTime('feb,15,1962,07:00:00');  // Month,Day,Year,Hour,Minute,Second
     };
 
     function upTime(countTo) {
       now = new Date();
       countTo = new Date(countTo);
       difference = (now-countTo);
 
       days=Math.floor(difference/(60*60*1000*24)*1);
       years = Math.floor(days / 365);
 	  
 	  // Schaltjahre
 	  leapyears = Math.round(years / 4);
 		
       if (years > 1){ days = days - (((years - leapyears) * 365) + (leapyears * 366))}
       hours=Math.floor((difference%(60*60*1000*24))/(60*60*1000)*1);
       mins=Math.floor(((difference%(60*60*1000*24))%(60*60*1000))/(60*1000)*1);
       secs=Math.floor((((difference%(60*60*1000*24))%(60*60*1000))%(60*1000))/1000*1);
 
       //Count Up Effekt or normal clock
       if (isScrolledIntoView('#years')) {
         if (firsttime){
             document.getElementById('years').firstChild.nodeValue = years;
             document.getElementById('days').firstChild.nodeValue = days;
             document.getElementById('hours').firstChild.nodeValue = hours;
             document.getElementById('minutes').firstChild.nodeValue = mins;
             document.getElementById('seconds').firstChild.nodeValue = secs;
         }else{
           var options = {
             useEasing : true,
             useGrouping : false,
             separator : ',',
             decimal : '.',
             prefix : '',
             suffix : ''
           };
 
           years = new CountUp('years', 0, years, 0 ,3, options);
           days = new CountUp('days', 0, days, 0, 3, options);
           hours = new CountUp('hours', 0, hours, 0, 3, options);
           minutes = new CountUp('minutes', 0, mins, 0, 3, options);
           seconds = new CountUp('seconds', 0, secs, 0, 3, options);
 
           years.start();
           days.start();
           hours.start();
           minutes.start();
           seconds.start(firsttime=true);
         }
       }
 
       clearTimeout(upTime.to);
 
       upTime.to=setTimeout(function(){ upTime(countTo); },1000);
     }
