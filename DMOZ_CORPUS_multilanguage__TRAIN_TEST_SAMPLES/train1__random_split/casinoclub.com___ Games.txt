         function incrementValue(item, end) {
             var duration = 60000;
             var current = parseFloat(end.replace(/,/g, '').substr(1)) - 150;
             var step = 0.01;
             var stepTime = step * duration / current;
             var timer = setInterval(function () {
                 current += step;
                 $(item).text("€" + numberWithCommas(current.toFixed(2)));
                 if (current >= parseFloat(end.replace(/,/g, '').substr(1))) {
                     clearInterval(timer);
                     $(item).text(end);
                 }
             }, stepTime);
         }
 
         function numberWithCommas(x) {
             return x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
         }

             function animm(elm) {
                 TweenMax.to(elm, R(5, 25), { y: h, ease: Linear.easeNone, repeat: -1, delay: -20 });
                 TweenMax.to(elm, R(4, 10), { x: '+=50', rotationZ: R(0, 180), repeat: -1, yoyo: true, ease: Sine.easeInOut });
                 TweenMax.to(elm, R(2, 8), { rotationX: R(0, 360), rotationY: R(0, 360), repeat: -1, yoyo: true, ease: Sine.easeInOut, delay: -1 });
             };
             function animm_sec(elm) {
                 TweenMax.to(elm, R(4, 12), { y: h, ease: Linear.easeNone, repeat: -1, delay: -10 });
                 TweenMax.to(elm, R(100, 200), { x: '+=20', rotationZ: R(0, 180), repeat: -1, yoyo: true, ease: Sine.easeInOut });
                 TweenMax.to(elm, R(2, 4), { rotationX: R(0, 360), rotationY: R(0, 360), repeat: -1, yoyo: true, ease: Sine.easeInOut, delay: -1 });
             };
             function R(min, max) { return min + Math.random() * (max - min) };
