     $(function () {
         var flow_news_animate = function () {
             this.wrapper = $(".flow_news_box");
             this.inner = $(".flow_news_box .inner");
             this.speed = 0.3;
             this.innerTop = 0;
             this.height = 0;
             this.isPause = false;
         }
         flow_news_animate.prototype.init = function () {
             this.inner.append(this.inner.html());
             this.height = this.inner.height();
             this.move();
         }
         flow_news_animate.prototype.move = function () {
             var _this = this;
             if (!this.isPause)
                 this.innerTop -= this.speed;
             if (Math.round(this.innerTop + this.height / 2) < 1) this.innerTop = 0;
             this.inner.css({"transform": "translate(0," + this.innerTop + "px)"});
 
             requestAnimationFrame(function () {
                 _this.move();
             })
         }
 
         flow_news = new flow_news_animate();
         flow_news.init();
 
         $(".flow_news_box .inner").hover(function () {
             flow_news.isPause = true;
         }, function () {
             flow_news.isPause = false;
         })
 
     })
    </script>
