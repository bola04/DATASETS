                 inputs = $("#steps fieldset.current input,#steps fieldset.current textarea");
                 var er = false;
                 inputs.each(function(index, el) {
                     if ($(this).hasClass("validate['required']")) {
                         if ($(this).val()=='') {
                             $(this).addClass('error')
                             $('#etapesFormulaire>ul>li.selected').addClass('error');
                             er = true;
                         }; 
                     }
                     if ($(this).hasClass("validate['required','email']")) {
                         if ($(this).val()=='') {
                             $(this).addClass('error')
                             $('#etapesFormulaire>ul>li.selected').addClass('error');
                             er = true;
                         }; 
                     }
                     if ($(this).hasClass("validate['required','phone']")) {
                         if ($(this).val()=='') {
                             $(this).addClass('error')
                             $('#etapesFormulaire>ul>li.selected').addClass('error');
                             er = true;
                         }; 
                     }
                     
                 });
                 if (!er) {
                     $('#etapesFormulaire>ul>li.selected').removeClass('error');
                     $("#steps fieldset.current input").removeClass('error');
                     $("#steps fieldset.current textarea").removeClass('error');
 
                 };
             }
             function validateall() {
                 var er = false;
                 $("#steps fieldset").each(function(index, el) {
                     var inputs = $(this).find('input,textarea');
                     var fieldset = $(this);
                     var legend = fieldset.find('legend').text();
                     var etape;
                     $('#etapesFormulaire>ul>li').each(function(index, el) {
                         if ($(this).find('a').text() == legend) {
                             etape = $(this);
                         };
                     });
                     
                     inputs.each(function(index, el) {
                         if ($(this).hasClass("validate['required']")) {
                             if ($(this).val()=='') {
                                 $(this).addClass('error');
                                 if (etape) {etape.addClass('error');};
                                 er = true;
                             }; 
                         }
                         if ($(this).hasClass("validate['required','email']")) {
                             if ($(this).val()=='') {
                                 $(this).addClass('error')
                                 if (etape) {etape.addClass('error');};
                                 er = true;
                             }; 
                         }
                         if ($(this).hasClass("validate['required','phone']")) {
                             if ($(this).val()=='') {
                                 $(this).addClass('error')
                                 if (etape) {etape.addClass('error');};
                                 er = true;
                             }; 
                         }
                         
                     });
                 });
                 return er;
             }
             $('#steps').parent().append('<div id="etapesFormulaire"><ul></ul></div>');
             $('#steps>form>fieldset').first().addClass('current');

             });

                 $('#steps fieldset').each(function(index, el) {
