             event.preventDefault();
 
             var formdata = $(this).serializeArray();
 
             var infos = {};
             for (var i = 0; i < formdata.length; i++) {
                 infos[formdata[i].name] = formdata[i].value;
             }
 
             var action = $(this).attr('action');
             var generateKikiURL =  action + '/1';
 
             if(infos.type_transaction != 'default'){
                 generateKikiURL += '/transaction/' + infos.type_transaction;
             }
 
             if(infos.reference && infos.reference != 'default'){
                 generateKikiURL += '/reference/' + infos.reference;
             }
 
             if(infos.type_bien != 'default'){
                 generateKikiURL += '/bien/' + infos.type_bien;
             }
 
             if(infos.localisation != 'default'){
                 generateKikiURL += '/localisation/' + infos.localisation;
             }
 
             if(infos.price_min && infos.price_min != 'default'){
                 generateKikiURL += '/prix-minimum/' + infos.price_min;
             }
 
             if(infos.price_max && infos.price_max != 'default'){
                 generateKikiURL += '/prix-maximum/' + infos.price_max;
             }
 
             if(infos.keywordor){
                 generateKikiURL += '/mot-cles/' + infos.keywordor;
             }
 
             if(infos.keywordand){
                 generateKikiURL += '/mot-clesn/' + infos.keywordand;
             }
 
             $(location).attr('href',generateKikiURL);
         });
