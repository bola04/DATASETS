 	          var duration = 0.3;
 	          var currentIndex = 0;
 	          var nextIndex = 0;

 	             if(slides[i].visible())   {
 	                currentIndex = i;
 	                break;
 	             }

 			  nextIndex = currentIndex + 1;
 	
 			  if(nextIndex == slides.length)   {
 	             nextIndex = 0;
 			  }
