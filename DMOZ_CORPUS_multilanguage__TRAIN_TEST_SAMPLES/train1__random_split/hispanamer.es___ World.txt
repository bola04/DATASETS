 function abrir_ventana (pag,tam_x,tam_y,vent,scroll,resize,menu,tool,status) {

  resize = (resize) ? "yes" : "no";

  tam_y = (tam_y==null) ? 300 : tam_y;
 
  if (ns4 || ie4) {
     var alto=window.screen.height;
     var ancho=window.screen.width;
     pos_x = (ancho-tam_x)/2;
     pos_y = (alto-tam_y)/2-20;
  } else {
     pos_x = 0;
     pos_y = 0;

 }

 function cambiarClase (src,accion,estiloOn, estiloOff) {
  if (accion) {
  	src.style.cursor = 'hand';
  	src.className = estiloOn;
  } else {
  	src.style.cursor = 'default';
  	src.className = estiloOff;
  }
 }
