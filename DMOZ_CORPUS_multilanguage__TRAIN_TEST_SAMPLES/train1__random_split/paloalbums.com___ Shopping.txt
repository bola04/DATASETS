   <script>
    // Set cookie to hide cookie notice.
         function cookie_notice_dismiss() {
             $('#cookie-notice').remove();
             var date = new Date();
             date.setTime(new Date().getTime()+(365*24*60*60*1000));
             document.cookie = 'cookie_notice_dismiss=1; expires=' + date.toGMTString() + '; path=/';
         }
         // Hide notice on page load if cookie set.
         if (document.cookie.match(/cookie_notice_dismiss/)) {
             $('#cookie-notice').remove();
         }
   </script>
