     var m = arr[1];
     var s = arr[2];
     if (s == 0) {
       if (m == 0) {
         if (h == 0) {
           alert("Время вышло");
           window.location.reload();
           return;
         }

         m = 60;
         if (h < 10) h = "0" + h;
       }
       m--;
       if (m < 10) m = "0" + m;
       s = 59;
