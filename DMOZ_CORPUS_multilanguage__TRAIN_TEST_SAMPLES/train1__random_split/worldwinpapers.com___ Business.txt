     function getProductInfoFromDataLayer(sku)
     {
         var product = false;
         jQuery(window.dataLayer).each(function(i,element) {
             if (jQuery(element).prop('ecommerce') != undefined) {
                 var impressions = element.ecommerce.impressions;
                 jQuery(impressions).each(function(i,tag) {
                     if (sku == tag.id) {
                         product = tag;
                     }
                 });
             }
         });
         return product;
     }
 
     function addClickedProductToDataLayer(skuText)
     {
         var sku = jQuery.trim(skuText.split(':')[1]);
         var product = getProductInfoFromDataLayer(sku);
         if (product) {
             window.dataLayer.push(
                 {
                     'event': 'productClick',
                     'ecommerce': {'click': {'actionField': {'list': 'category'}, 'products': [product]}}
                 }
             );
         }
     }
 
     function addToCartProductToDataLayer(skuText, samplePack = false)
     {
         var sku = jQuery.trim(skuText.split(':')[1]);
         var product = false;
         product = getProductInfoFromDataLayer(sku, samplePack);
 
         var productData = product;
         if (samplePack) {
             var sampleSkuCategory = productData.category + '/' + productData.id;
             productData.id = 'SAMPLEPACK';
             productData.category = sampleSkuCategory;
             productData.price = 1.00;
         }
 
         if (product) {
             window.dataLayer.push({
                 'event': 'addToCart',
                 'ecommerce': {
                     'currencyCode': 'USD',
                     'add': {
                         'products': [productData]
                     }
                 }
             });
         }
     }
