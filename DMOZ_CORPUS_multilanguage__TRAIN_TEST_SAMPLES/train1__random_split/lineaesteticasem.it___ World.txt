    $('.addtocart,.amico').click(function () {

         var i = $(this).data('pid');
         var qta = $('input[name="quantita_' + i + '"]').val();
         var tc = $(this).attr('class');
         if (tc == 'amico') {
             var str = 'pid=' + pid + '&qta=' + qta + '&tc=' + tc;
         } else {
             var str = 'pid=' + pid + '&qta=' + qta;
         }
 
 
 //alert(str);
         $.ajax({
             url: 'cart_adderr.php',
             type: 'POST',
             data: str,
             success: function () {
 
                 alert('Prodotto aggiunto al Carrello!');
                 document.location.assign("carrello.html");
             }
 
         })
     })
 
 
     $('button[data-action="remove"]').click(function () {
 
         $.ajax({
             url: 'remove_to_cart.php',
             type: 'POST',
