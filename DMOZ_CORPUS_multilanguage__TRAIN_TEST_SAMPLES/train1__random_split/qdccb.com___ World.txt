 //首页banner切换
 $(function () {
 	var zIndex = 0;
 	var aLen = $('.bannerImg li').length;
 	var bannerNav = '';
 	var timer = null;
 	
 	$('.bannerImg li').eq(0).show();
 	for (var j=0; j<aLen; j++) {
 		bannerNav += '<span></span>'
 	}
 	$('.bannerNav').html(bannerNav).find('span').eq(zIndex).addClass('bannerNavOn');
 	
 	$('.banner').hover(function () {
 		clearInterval(timer);
 	}, function () {
 		bannerMove ();
 	});
 	
 	$('.bannerNav span').each(function (index, element) {
 		$(element).mouseover(function () {
 			zIndex = index;
 			bannerFn (zIndex);
 		});
 	});
 	
 	bannerMove ();
 	function bannerMove () {
 		timer = setInterval(function () {
 			zIndex++;
 			if (zIndex >= aLen) {
 				zIndex = 0;
 			}
 			bannerFn (zIndex);
 		}, 3000);
 	}
 	function bannerFn (Index) {
 		$('.bannerNav span').eq(Index).addClass('bannerNavOn').siblings().removeClass('bannerNavOn');
 		$('.bannerImg li').eq(Index).fadeIn().siblings().hide();
 	}
 });
 
 //首页产品左右点击切换
 $(function () {
 	libFn ('#con_one_1');
 });
 
 function libFn (element) {
 	var i = $(element).attr('data-val') || 0;
 	var aUl = $(element).find('ul');
 	var aLi = aUl.find('li');
 	var aliWidth = aLi.width();
 	var aliLen = aLi.length;
 	var aliLenVal = aliLen - 5;
 	var btn = $(element).find('.lib_Btn');
 	var btnLeft = $(element).find('.lib_BtnLeft');
 	var btnRight = $(element).find('.lib_BtnRight');
 	aUl.width(aliWidth * aliLen);
 	
 	if (aliLen > 5) {
 		btn.show();
 	};
 	
 	btnLeft.click(function () {
 		i--;
 		if (i < 0) {
 			i = aliLenVal;
 		}
 		libFn (i);
 	});
 	btnRight.click(function () {
 		i++;
 		if (i > aliLenVal) {
 			i = 0;
 		}
 		libFn (i);
 	});
 	
 	function libFn (Index) {
 		$(element).attr('data-val', Index);
 		aUl.stop().animate({
 			'left': -Index * aliWidth
 		});
 	};
 };
 
 function setTabMore(name, cursel, n) {
 	for ( i = 1; i <= n; i++) {
 		var menu = document.getElementById(name + i);
 		var con = document.getElementById("con_" + name + "_" + i);
 		var co = document.getElementById("more_" + name + "_" + i);
 		menu.className = i == cursel ? "hover" : "";
 		con.style.display = i == cursel ? "block" : "none";
 		co.style.display = i == cursel ? "block" : "none";
 	};
 	libFn ("#con_" + name + "_" + cursel);
 };
 function setTab(name, cursel, n) {

 		menu.className = i == cursel ? "hover" : "";
 		con.style.display = i == cursel ? "block" : "none";
 	}
 };
