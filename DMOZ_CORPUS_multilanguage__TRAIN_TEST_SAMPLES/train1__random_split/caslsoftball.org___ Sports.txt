 function fetchsomepics() { //called every time want another batch of pics... this is staggering procedure to create user responsiveness.
                            //just needs to fetch each pic once, but avoid doing all in one shot or every second one since high latency and would otherwise miss high thruput opportunity (but without overdoing it and hurting user responsiveness at startup).
                            //note, we aren't setting up tag style here. only fetch from server pics data.
                            //done after get core data since pics subservient; however, start this process even without page loaded.
                            //but setting up element style is only after loaded and is staggered during first pass of slideshow in xxxx() below for reasons just explained.
   var ii,jj,img,skp;
   for (ii=0, skp=skipahead, getcnt=GETCNTINIT; ii<generalpicdata.length; ii++) {
     if (skp--<=0&&getcnt-->0) {  //alert(skp)//getcnt-- iff skp <=0
       generalpicobj[ii]=new Image();
       generalpicobj[ii].cached=-1;
       generalpicobj[ii].onload=
       (function(i,pN){
           return function (){  //alert(i)
               this.cached=0;
           };
       })(ii,null);  //ii not needed and img could be global to avoid this construct
       generalpicobj[ii].width= 363;//img.width;
       generalpicobj[ii].height= 363;//img.height;
       generalpicobj[ii].src=generalpicdata[ii];
     }
   }
   for (ii=0; ii<7; ii++) {  ///not tested since 0 pics for now
     if (getcnt>0) {
       if (daypicobj[ii]===undefined) { //?
           daypicobj[ii]=[];
       }
       for (jj=0; jj<daypicdata[ii].length; jj++) {
         if (skp--<=0&&getcnt-->0) {
           daypicobj[ii][jj]=new Image();
           daypicobj[ii][jj].cached=-1;
           daypicobj[ii][jj].onload=
           function(){
               this.cached=0;
           };
           daypicobj[ii][jj].width=PIC_W;//img.width; 363
           daypicobj[ii][jj].height=PIC_H;//img.height; 363
           daypicobj[ii][jj].src=daypicdata[ii][jj];
         }
       }
     }
   }
   skipahead+=GETCNTINIT;  //may be off a bit on last round
   if (getcnt>0) {
       getcnt=-1; //-1 means done
   } else {
       getcnt=0; //0 will mean cnt upward normally;
   }
 //alert(generalpicobj.length)
 }

 function adjusttextareacolumnsforbrowsers(box1, box2) {
     //adjust textareas width/col since col are ready. the cols for each one may differ and is related to dim of slideshow. later put the main numbers at top together. 363,363,64col, etc. and init col above this step.

  //alert (box1.cols);
  //alert(navigator.userAgent.search(/firefox/i))
  if (navigator.userAgent.search(/firefox/i)>=0) {
      if (navigator.userAgent.search(/edge/i)>=0) {
          ;
      } else {
          box1.cols=Math.round(box1.cols*.74*1.00);  // .74?xxxxcalibrated to work when turning 63 into 47 at size1 (8pt?).. decent results in most other cases i assume. otherwise include some font dependency (?) and factor out the right scrollbar also.

      }
  } else if (navigator.userAgent.search(/ozilla/)>=0) {
      if (navigator.userAgent.search(/edge/i)>=0) {
          ;
      } else {
          box1.cols=Math.round(box1.cols*.89);
          box2.cols=Math.round(box2.cols*.85);//*.9*.96);
      }
  }
 }
