 function setCookie(cookieName, cookieValue, nDays) {
   var today = new Date();
   var expire = new Date();
   if (nDays == null || nDays == 0) nDays = 0;
 
   expire.setTime(today.getTime() + (3600000 * 24 * nDays));
 
   document.cookie = cookieName + "=" + encodeURIComponent(cookieValue)
           + ";expires=" + expire.toGMTString()
           + ";path=/"; //;HttpOnly;secure;";
 }

 // Redirects to a secure LA address version of the current page.
 function gotoSecureURL(sessionID) {
 
     // Get their currrent URL
     var loc = location.href;
 
     //If they login from the Password page, take them to the homepage.
     if (loc.toLowerCase().indexOf("/cpanel/password.asp") != -1){
     	loc = loc.toLowerCase().replace("/cpanel/password.asp", "/default.asp");
     }
 
     //If they login from SecretWord.asp, take them to the selected page
     if (loc.toLowerCase().indexOf("/secretword.asp?file=teamdocs.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=teamdocs.asp&", "/teamdocs.asp?");
     }else if (loc.toLowerCase().indexOf("/secretword.asp?file=schedule.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=schedule.asp&", "/schedule.asp?");
     }else if (loc.toLowerCase().indexOf("/secretword.asp?file=pictures.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp?file=pictures.asp", "/pictures.asp");
     }else if (loc.toLowerCase().indexOf("/secretword.asp") != -1){
         loc = loc.toLowerCase().replace("/secretword.asp", "/roster.asp");
     }
     // Replace the host name portion with LeagueAthletics.com
     var host = location.hostname.toLowerCase();
     if (host.indexOf("dev.leagueathletics.com") == -1)
         loc = loc.replace(host, "LeagueAthletics.com");
 
     // Change protocal to https or add it if not present.
     loc = loc.replace("http://", "https://");
     if (loc.indexOf("https://") == -1 )
         loc = "https://" + loc;
 
     // Add org= parameter if not already there.
     if (loc.indexOf("org=") == -1) {
         if (loc.indexOf("?") == -1) {
             loc += "?";
         } else {
             loc += "&";
         }
 
         loc += "org=WESTPORTSOCCER.ORG";
     }
 
     // Strip off any existing snid= values.
     var iSnid = loc.indexOf("snid=");
     if (iSnid >= 0) {
         var iNd = loc.indexOf("&", iSnid + 1)
         if (iNd == -1) iNd = loc.length - 1;
         loc = loc.substr(0, iSnid) + loc.substr(iNd);
     }
 
     // Redirect to secure version of the page.
     var aLoc = loc.split("#");
     var hashLocation = aLoc[1] || '';
     if (hashLocation) hashLocation = '#' + hashLocation;
     var tempLoc = aLoc[0] + "&snid=" + sessionID + hashLocation;
     location.href = tempLoc;
     //location.href = loc + "&snid=" + sessionID;
 }

                              <b>
                               WSA HEADING POLICY
                              </b>
                             </div>
                             <div style="padding: 8px; ">
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                Dear WSA Families,
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: "Times New Roman";'>
                                The Connecticut Junior Soccer Association (“CJSA”) has adopted the US Soccer and US Youth Soccer (national governing bodies for soccer in the US) rules regarding heading.  These rules take effect on April 1, 2016 and must be enforced by coaches, referees and club administrators.
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; tab-stops: .5in; mso-pagination: none; mso-layout-grid-align: none;">
                               <b>
                                <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                 Compliance is mandatory for all CJSA members (the WSA is a member of the CJSA).
                                </span>
                               </b>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <b>
                                <i>
                                 <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                  Heading Restrictions which apply to games and practices
                                 </span>
                                </i>
                               </b>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <b>
                                <i>
                                 <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                 </span>
                                </i>
                               </b>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <i>
                                <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                 Players who are 10 and younger
                                </span>
                               </i>
                              </p>
                              <ul style="list-style-type: disc; direction: ltr;">
                               <li style="color: rgb(0, 0, 0); font-size: 11pt; font-style: normal; font-weight: normal;">
                                <p style="color: rgb(0, 0, 0); font-size: 12pt; font-style: normal; font-weight: normal; margin-top: 0in; margin-bottom: 0pt; mso-list: l2 level1 lfo4; mso-pagination: none; mso-layout-grid-align: none;">
                                 <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                  No player who is 10 or younger may deliberately head the ball, regardless of the age group of the team they are playing on. This includes all activities where CJSA insurance applies, such as practices, scrimmages, and competitions at all levels.
                                 </span>
                                </p>
                               </li>
                               <li style='color: rgb(0, 0, 0); font-family: "Comic Sans MS"; font-size: 11pt; font-style: normal; font-weight: normal;'>
                                <p style='color: rgb(0, 0, 0); font-family: "Cambria","serif"; font-size: 12pt; font-style: normal; font-weight: normal; margin-top: 0in; margin-bottom: 0pt; mso-list: l2 level1 lfo4; tab-stops: 11.0pt .5in; mso-pagination: none; mso-layout-grid-align: none;'>
                                 <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                  No player on a U11 team or a combined team including the U11 Age Group, (e.g. U11/12), even if the player has turned 11, may head the ball deliberately.
                                 </span>
                                </p>
                               </li>
                              </ul>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                For the WSA this means that headers are not allowed on any of our U11 teams, regardless if the player is 11 or older.  Imagine trying to run a practice or referee a game where some players are able to head and some are not.  Therefore no heading for our U-11 groups or any individual player who might be playing up an age group to U12.
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                               </span>
                              </p>
                              <p style="margin: 0in 0in 0pt; mso-pagination: none; mso-layout-grid-align: none;">
                               <span style='font-family: "Comic Sans MS"; font-size: 11pt; mso-bidi-font-family: Calibri;'>
                                This restriction will be enforced in competitions for teams U11 and younger (including combined age brackets including U11 and younger) by award of an indirect free kick for the opposing team. Coaches and parents are responsible for applying this restriction for players who are 10 or younger and playing up.
                               </span>
                              </p>
