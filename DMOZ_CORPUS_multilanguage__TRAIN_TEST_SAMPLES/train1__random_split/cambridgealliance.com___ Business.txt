    document.addEventListener("DOMContentLoaded", function () {
         var modalShown = {};
         $(".alert-close").click(function () {
             var alert = $(this).closest("*[data-alert-id]");
             if (alert.length < 1) {
                 alert = $(this).children("*[data-alert-id]");
             }
             var alertid = alert.data("alert-id");
             var alertData = getAlertData(alertid, alert.data("alert-storage"));
             alertData.hidden = true;
             setAlertData(alertid, alertData, alert.data("alert-storage"));
             stuff.apply(alert);
         });
         $(".alert-title, .alert-show").click(function () {
             var shown = $(this).closest(".alert-bar").find(".alert-message").toggleClass("shown").hasClass("shown");
             var statusElm = $(this).closest(".alert-bar").find(".fa-angle-double-down, .fa-angle-double-up");
             if (statusElm.hasClass("fa-angle-double-down")) {
                 statusElm.removeClass("fa-angle-double-down").addClass("fa-angle-double-up");
             } else {
                 statusElm.addClass("fa-angle-double-down").removeClass("fa-angle-double-up");
             }
         });
         $("div[data-alert-id]").each(stuff);
 
         function stuff(initial) {
             var $this = $(this);
             var thingtodo = getAlertData($this.data("alert-id"), $this.data("alert-storage"));
 
             if (thingtodo.hidden) {
                 $this.toggleClass("shown", !thingtodo.hidden);
             } else {
                 if ($.isNumeric($this.data("time-based"))) {
                     var delay = 100;
                     try {
                         delay = parseInt($this.data("time-based") || "0.1");
                         delay = delay * 1000;
                         delay = delay - (thingtodo.time || 0);
                         //console.log("Time remaining is: " + delay);
                     } catch (e) { }
                     var loadTime = new Date().getTime();
                     setTimeout(function () {
                         $this.toggleClass("shown", !thingtodo.hidden);
                         appendModalOverlay($this.data("alert-id"));
                     }, delay);
                     if (delay > 0) {
                         window.addEventListener("unload", function () {
                             var data = getAlertData($this.data("alert-id"), $this.data("alert-storage"));
                             if (!data.hidden) {
                                 var timePassed = new Date().getTime() - loadTime;
                                 data.time = (data.time || 0) + timePassed;
                                 setAlertData($this.data("alert-id"), data, $this.data("alert-storage"));
                             }
                         });
                     }
                 } else if ($this.data("time-based") == "exit") {
                     $(document).mouseleave(function (e) {
                         if (e.clientY <= 0) {
                             var thingtodo = getAlertData($this.data("alert-id"), $this.data("alert-storage"));
                             $this.toggleClass("shown", !thingtodo.hidden);
                             appendModalOverlay($this.data("alert-id"));
                         }
                     });
                 }
             }
         }
         function getAlertData(alertid, storage) {
             if (window[storage]) {
                 try{
                     var stuff = window[storage].getItem(alertid);
                     if (stuff) {
                         try {
                             return JSON.parse(stuff);
                         } catch (e) { }
                     }
                 } catch (e) {
                     //issue persisting
                 }
             }
             return {};
         }
         function setAlertData(alertid, alertData, storage) {
             if (window[storage]) {
                 window[storage].setItem(alertid, JSON.stringify(alertData));
             }
             
         }
         //if ($(".content-hero, .intro-coverage, .intro, .intro-image").length > 0 && $(".cta-links").length < 1) {

         //}
         //$(".alert-modal-overlay").appendTo(document.body);
         function appendModalOverlay(alertId) {
             if (!modalShown[alertId] && $("#overlay-" + alertId).length) {
                 modalShown[alertId] = true;
                 //console.log("appendModalOverlay alertId = overlay-" + alertId);
                 $("#overlay-" + alertId).appendTo(document.body);
             }
         }
     });
