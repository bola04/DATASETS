 						}
 						     
 					],
 					eventMouseover: function(calEvent, domEvent) {
 										var layer = "<div id='events-layer'  style='background-color:#E2EFF8; color:#000; position:relative; width:250px; height:100%; top:-1px; text-align:left; z-index:-1'>"+ calEvent.datiAggiuntivi +"</div>";

 						$(this).css("z-index",(valZindex + 1)*1);
 						//alert(valZindex);
 						$(this).append(layer);
 					},
 					 eventMouseout: function(calEvent, domEvent) {
 						//alert(valZindex);
 						$(this).css("z-index",valZindex);
 						$('#events-layer').remove();
 					},
 					eventClick: function(event) {
 						if (event.url) {
 							window.open(event.url);
 							return false;
 						}
 					}
