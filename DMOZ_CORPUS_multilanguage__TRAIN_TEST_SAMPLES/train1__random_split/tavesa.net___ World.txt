     $(document).ready(function() {
         $('.call_you_form_container').find('#submit').click(function () {
             $('.call_you_form_container').find('form').submit(function(e){
                 e.stopImmediatePropagation();
 
                 $('.call_you_form_container #submit').hide();
                 $('#call_you_loader_container').show();
 
                 postForm($(this), function(response){
                     removeErrors();
 
                     if(response['requireError'].length > 0 || response['notValidError'].length > 0){
                         onError(response);
                     } else {
                         onSuccess(response['response'])
                     }
                 });
 
                 return false;
             });

         function postForm($form, callback){
             var formValues = {};
             $.each($form.serializeArray(), function(i, field){
                 formValues[field.name] = field.value;
             });
 
             $.ajax({
                 type        : 'POST',
                 url         : $form.attr( 'action' ),
                 data        : formValues,
                 success     : function(data){
                     callback(data);
                 }
             });
         }
 
         function removeErrors(){
             $("#call_you_name").removeClass("validateError");
             $("#call_you_phone").removeClass("validateError");
             $("#call_you_firstSurname").removeClass("validateError");
             $("#call_you_reason").removeClass("validateError");
             $("#call_you_policy").removeClass("validateError");
             $(".mandatory_fields").removeClass("validateError");
         }
 
         function onError(errors){
 
             $('.call_you_form_container #submit').show();
             $('#call_you_loader_container').hide();
 
             if(errors['requireError'].length > 0) {
                 onRequireError(errors['requireError']);
             }
 
             if(errors['notValidError'].length > 0) {
                 onNotValidError(errors['notValidError']);
             }
         }
 
         function onRequireError(errors) {
             errors.forEach(function(error) {
                 var inputID= "#call_you_";
                 inputID = inputID.concat(error);
                 $(inputID).addClass("validate_error");
             });
 
             $(".mandatory_fields").addClass("validate_error");
         }
 
         function onNotValidError(errors){
             errors.forEach(function(error) {
                 var inputID= "#call_you_";
                 inputID = inputID.concat(error);
                 $(inputID).addClass("validate_error");
             });
         }
 
         function onSuccess(data){
             $(".call_you_form_container").replaceWith(data);
         }
     });
