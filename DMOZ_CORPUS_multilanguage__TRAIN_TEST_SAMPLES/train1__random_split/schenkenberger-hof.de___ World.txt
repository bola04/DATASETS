    $(document).ready(function(){
             
             var dayarray = ['So','Mo','Di','Mi','Do','Fr','Sa']
             var weatherHTML = [];
             
             $.ajax({
                 url:'http://api.openweathermap.org/data/2.5/forecast/daily?id=2938376',
                 type:'post',
                 async:true,
                 dataType:'jsonp',                    
                 success:function(xR){
                     
                     var weatherdata = xR;
                     
                     var temperatures = weatherdata.list;
                     
                     var daycount = 0;
                     
                     var temp_offset = -273.15;
                     
                     
                     for (key in temperatures) {
                         
                         var day = new Date((temperatures[key].dt*1000));
                        
                         var temp_min = Math.round(temp_offset+temperatures[key].temp.min);
                         var temp_max = Math.round(temp_offset+temperatures[key].temp.max);
                         var temp_icon = temperatures[key].weather[0].icon+'.png';
                         var weather_desc = temperatures[key].weather[0].description;
                        
                         var datum = day.getDate()+'.'+day.getMonth()+'.'+day.getFullYear();
                        
                         $('#wetter').append('<div class="weather_item"><table><tr><td class="col_1"><img src="image/icons/wetter/'+temp_icon+'" alt="'+weather_desc+'" width="36px"/></td><td class="col_2">'+dayarray[day.getDay()]+'</td><td class="col_3">'+temp_min+'° / '+temp_max+'°</td></tr></table></div>');
                        
                         daycount++;
                         
                         if (daycount>2) {
                             break;
                         }
                         
                     }
                     
                    
                    
                     
             }});
             
             // NEWS BOX
             $('#news_box .news_item').hide();
                 
             var news_size = $('#news_box .news_item').size();
             
             var news_navigation = $('<div id="news_navigation" />').css({'display':'block','height':'16px','text-align':'right','position':'absolute','bottom':'0','right':'4px'});
             
             var interval = 0;
             
             for(var i=0,k=news_size;i<k;i++)
             {
                 
                 var link = $('<a />',{'href':'javascript:void(0)','data-show-item':i});
                 
                 link.on('click',function(){
                     
                     $('#news_box .news_item:visible').hide();
                     
                     $('#news_box .news_item:eq('+$(this).attr('data-show-item')+')').fadeIn(650);
                     
                     $('#news_navigation a').css('background','#666').removeClass('active');
                     
                     $(this).css('background','#970311').addClass('active');
                     
                     autoSlide();
                     
                 });
                 
                 
                 link.css({'display':'inline','width':'6px','height':'6px','font-size':'12px','background':'#666','margin':'2px','padding':'0','float':'left'});
                 news_navigation.append(link);
                 
             }
             
             $('#news_box').append(news_navigation);
             
             news_navigation.find('a').eq(0).trigger('click');
             
             
             //$('#some_selector').parents('li').reverse();
             //:)
             
             function autoSlide(){
             
                 clearInterval(interval);
             
                 interval = setInterval(function(){
                     
                     // Check the indexes
                     var cIndex = $('#news_navigation a.active').index();
                     var linkSize = $('#news_navigation a').size()-1;
                     
                     var nextIndex = (cIndex+1)>linkSize?0:(cIndex+1);
                     
                     $('#news_navigation a:eq('+nextIndex+')').trigger('click');
                     
                 },7000);
             
             }
                 
                 
             
                     
         });
