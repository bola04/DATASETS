    function close_popup(){
 						$('#popup_frame').fadeOut('fast', function(){

 								$(this).remove();
 							});

 						if (!element.is('iframe')){
 							var element = element.clone();
 						}
 						$("body").append('<div id="popup_overlay"></div>');
 
 						$("body").append('<div id="popup_frame"></div>');

 						element.appendTo('#popup_frame');
 						element.show();
 						
 						$("#popup_overlay").fadeTo('fast', 0.7, function() {
 							var w = element.width();
 							var h = element.height();
 							
 							if (w>0 && h>0){
 								$("#popup_frame").width(w+"px");
 								$("#popup_frame").height(h+"px");
 							}
 							else{
 								element.width("100px");
 								element.height("200px");
 							}
 							
 							var t = (($(window).height()/2)-(h/2));
 							var l = ($(window).width()/2)-(w/2);							
 							
 							$("#popup_frame").css({ top: t, left: l });
 							$("#popup_frame").fadeTo('fast', 1.0, function() {
 								// done
 							});
 							
 							$(this).click(function(){
 								close_popup();
 							});
 							
 							$('#popup_close_btn').click(function(){
 								close_popup();
 							});
 						});
 					}

      function createCookie(name, value, days){
     var d = new Date();
     d.setTime(d.getTime() + (days*24*60*60*1000));
     var expires = "expires="+d.toUTCString();
     document.cookie = name + "=" + value + "; " + expires;
 }
 
 function deleteCookie(name){
    document.cookie = name + '=; expires=Thu, 01 Jan 1970 00:00:01 GMT;';
 }
 
 function readCookie(cname){
     var name = cname + "=";
     var ca = document.cookie.split(';');
     for(var i=0; i<ca.length; i++) {
         var c = ca[i];
         while (c.charAt(0)==' ') c = c.substring(1);
         if (c.indexOf(name) == 0) return c.substring(name.length,c.length);
     }
     return "";
 }
