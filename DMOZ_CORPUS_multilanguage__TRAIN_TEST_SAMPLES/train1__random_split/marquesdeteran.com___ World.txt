     function validateEmail(email){
         jQuery(".error").hide();
         var hasError = true;
         var emailReg = /^([\w-\.]+@([\w-]+\.)+[\w-]{2,4})?$/;
 
         var emailaddressVal = email;
         if(emailaddressVal == '') {
             hasError = false;
         }
 
         else if(!emailReg.test(emailaddressVal)) {
             hasError = false;
         }
 
         return hasError;
     };
