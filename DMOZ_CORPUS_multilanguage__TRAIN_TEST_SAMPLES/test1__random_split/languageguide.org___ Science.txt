 function keyListener(evt)
 {
    // if (ie) evt = window.event;
     //write any code here that you want
     //to perform when a key is pressed.
 // alert(evt.keyCode);
     //here is one example for de-activating the 'esc'
 // keyPressInfo += evt.keyCode;
    // alert(evt.keyCode);
 	if(evt.keyCode == 40) {
        if (selectedDiv == null){
 			var selectedParentDiv = activeLangDisplay.firstChild;
 			while (selectedParentDiv.nodeType == 3){
 				 selectedParentDiv = selectedParentDiv.nextSibling;
 		   }
 		   selectedDiv = selectedParentDiv.firstChild;
 	   }
 	   else{
 		   selectedDiv.className = "option";
 		   selectedDiv = selectedDiv.parentNode.nextSibling.firstChild; 
 		   if (selectedDiv == null){
 			   return;
 		   }
 	   }
 	   selectedDiv.className += " selected";
 	   var topPos = selectedDiv.offsetTop;
 	   if (topPos > 400){
 	   	activeLangDisplay.scrollTop = 30;
 	   }
 
     }
 	else if(evt.keyCode == 38) {
        if (selectedDiv == null){
 	   	selectedDiv = activeLangDisplay.lastChild.firstChild;
 	   }
 	   else{
 		   selectedDiv.className = "option";
 		   selectedDiv = selectedDiv.parentNode.previousSibling.firstChild;
 	   }
 	   if (selectedDiv != null){
 	   		selectedDiv.className += " selected";
 	   }
 		event.preventDefault();
     }
 	else if(evt.keyCode == 13) {
 		if (selectedDiv != null){
 			goTo(selectedDiv.parentNode.href);
 		}
 	}
 }

 function setDimensions(div){
 	//alert(document.body.height);
 	var sHeight = document.documentElement.clientHeight;
 	var sWidth = document.documentElement.clientWidth;
 	var rHeight = parseInt(sHeight/2);
 	//div.style.maxHeight = rHeight + "px";
 	 //= rHeight + 40 + "px";
 	
 	var computed;
 	if (largeScreen){
 		var center = document.getElementById("center");
 		computed = getComputedStyle(center);
 		div.style.left = (sWidth / 2) - 250 + "px";
 		var top = parseInt(computed.top, 10);
 		var height = parseInt(computed.height, 10);
 		div.style.top = (top + 15)  +"px";
 		var heightDropdown = (sHeight - top) - 25;
 		div.style.maxHeight = heightDropdown + "px";
 	}
 	else{
 		var outer = document.getElementById("outer-text");
 		computed = getComputedStyle(outer);
 		div.style.left = 0;
 		div.style.top = "54px";
 		var heightSmallScreen = sHeight - 54;
 		div.style.maxHeight = heightSmallScreen + "px";
 	}
 }
