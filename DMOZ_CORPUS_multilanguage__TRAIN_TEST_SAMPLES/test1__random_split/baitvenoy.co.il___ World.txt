 	findCookies: function (cookName) {

 		var pos = allcookies.indexOf(cookName + "=");
 		if (pos != -1) {
 			var start = pos + cookName.length + 1;
 			var end = allcookies.indexOf(";", start);
 			if (end == -1) end = allcookies.length;
 			var value = allcookies.substring(start, end);
