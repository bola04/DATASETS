              <script>
               require(["jquery", "domReady!"], function($) {
 
 	var $root = $('#fcd0f8d0-0434-4ce0-a410-1561848aeddd');
 	init_search_handler();
 	init_form($root); // mobile search js is handled with mobile nav
 
 	function init_search_handler()
 	{
 		var $cnt_search = $('#cnt_search');
 
 		if($cnt_search)
 		{
 			var $btn_toggle = $cnt_search.find('.btn_toggle');
 
 			var $search_form = $cnt_search.find('.search_form');
 			var $btn_search = $search_form.find('.btn_search');
 			var $txt_search = $search_form.find('[data-sv-searchInput]');
 
 			$btn_toggle.off('click').on('click', function()
 			{
 				if(!$search_form.is(':visible'))
 				{
 					$search_form.fadeIn('fast', function()
 					{
 						$txt_search.focus();
 					});
 				}
 				else
 				{
 					$search_form.fadeOut('fast');
 				}
 
 				return false;
 			});
 
 			$btn_search.off('click').on('click', function()
 			{
 				if($txt_search.val() == "")
 				{
 					$btn_toggle.trigger('click');
 
 					return false;
 				}
 			});
 
 			$txt_search.on('blur', function()
 			{
 				if($(this).val() == "")
 				{
 					$btn_toggle.trigger('click');
 				}
 			});
 		}
 	}
 
 
 	function init_form($root){
 		var $input = $root.find("input[type='search']");
 		var $form = $root.find("form");
 		$form.find('[data-sv-searchBtn]').click(function(){ $form.submit(); })
 		$form.submit(function(e){ if ($input.val().length === 0) { 
 			e.preventDefault(); 
 			if($(window).width() > 1024){
 				$root.find(".search-form").fadeOut('fast');
 			} 
 		} });
 	}
 });

 			var d = new Date();
 			d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
 			var expires = "expires=" + d.toUTCString();
 			document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";

 				while (c.charAt(0) == ' ') {
 					c = c.substring(1);
 				}
