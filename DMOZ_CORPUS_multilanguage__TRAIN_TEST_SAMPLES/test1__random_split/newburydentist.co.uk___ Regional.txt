 // Perform transition is run after the timer expires or when the next/prev arrows are clicke
 SlideShow3171005.PerformTransition = function(trans_dir) 
 {
 	if (SlideShow3171005.transitioning == true) return;    
     var is_fwd = (trans_dir === 'next');
     var is_selected_index = (trans_dir === 'selected');
     var next_img = '';
 
     if (!is_selected_index) {
         // next_img only references the correct element before setContainers is called!
         next_img = is_fwd ? SlideShow3171005.nextContentContainer : SlideShow3171005.previousContentContainer;
     } else {
         SlideShow3171005.selectedContentContainer.html(SlideShow3171005.contents[SlideShow3171005.currentContentIndex]);
         next_img = SlideShow3171005.selectedContentContainer;
     }
 
     var next_caption_container = next_img.find('.captionContainer');

 		next_caption_container.children('img').unbind('load.LCMSslideshow').bind('load.LCMSslideshow', SlideShow3171005.resizeCaptionContainer);
 	} else {
 		next_caption_container.children().children('img').unbind('load.LCMSslideshow').bind('load.LCMSslideshow', SlideShow3171005.resizeCaptionContainer);
 	}
 
     SlideShow3171005.updateIndexes();
     
     
     	
     	
     	SlideShow3171005.transitioning = true;
     	
     	if (is_fwd || is_selected_index) {
 	    	//Set the sliding transition. append the next slide to the current container
 	    	SlideShow3171005.container.append(next_img.children(":first").clone(true));	
     	} else {
 	    	//Set the margin of the previous contents to negative of the parent container width so it can slide in from the right.

 	    	//prepend the next image in front of the current image

     	}
     	
         
         //make sure the next slide is visible
         SlideShow3171005.container.children('.SlideShowImage').width(SlideShow3171005.container.parent().parent().width()).css('float', 'left');    
         
         if (!is_fwd && !is_selected_index) {
 	        //since we have now prepended the image we want to the current container we set the margin of the previous container back to 0
 	        next_img.children('.SlideShowImage').css('margin-left', '0px')
         }
         
         // first_div is the current slide for 'next' and the new slide for 'prev'
         var first_div = SlideShow3171005.container.children('.SlideShowImage').eq(0), 
         	second_div = SlideShow3171005.container.children('.SlideShowImage').eq(1);
         
         
         // if 'next' transition, animate the transition by setting the left margin to the negative width of the parent container and animating it
         first_div.animate(
