 	function order(fio,state,address,phone,fio2,city,zip,email,delivery,pay,other){
 		$.ajax({
 			url: '/cart/ajax_order',
 			type: 'post',
 			data: 'fio=' + fio + '&state=' + state + '&address=' + address + '&phone=' + phone + '&fio2=' + fio2 + '&city=' + city + '&zip=' + zip + '&email=' + email + '&delivery=' + delivery + '&pay=' + pay + '&other=' + other,
 			dataType: 'json',
 			success: function(data) {
 				// Если отправка прошла успешно, отображаем сообщение об успехе
 				if (data.success == 'success') {
 					if (pay == 'банковской картой') {
 						// Направляем покупателя на страницу подтверждения суммы к оплате картой

 						document.getElementById('tablo').innerHTML = data.tablo;
 						// Данные для окна корзины (сообщение об отправке)
 						document.getElementById('cart-content').innerHTML = data.carttab;

 				}
