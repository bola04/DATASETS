 	function showinfoBox(triggerTime, cookieName, infoBox, expireDate, notNowTime, crossClickedTime, slideinName) {
 		// uses setTimout to trigger the slide in after a set period (triggerTime)
 		setTimeout(function() {
 			$('#info_box_container').html('<div id="info_box" class="info_box" data-slidein="' + slideinName + ': Shown"><div id="info_box_close" class="info_box_close info_box_btn" data-slidein="' + slideinName + ': Close" title="Hide">&times;</div><div id="info_box_content" class="info_box_content">' + infoBox + '</div.</div>'); // Define the HTML for the slide in
 			$('.editor-alert-outer').hide(); // hides any admin messages so slide in will look OK to admin as well as public
 			$('#info_box').animate({bottom: "0"}, 'slow');  // slides the slide in
 			$(document).on('click', '.info_box_close', function(e) {  // X clicked
 				e.preventDefault();
 				var inXhours = 1/24 * crossClickedTime; // set cookie expiry time to 1 hour x crossClickeTime (i.e. 30 minutes)
 				Cookies.set(cookieName, false, { expires: inXhours }); // set cookie to expire in time specified above
 				$('#info_box_container').animate({bottom: "-300px"}, 'slow'); // close slide in
 			});

 			});

 				e.preventDefault();				
 				var inXhours = 1/24 * notNowTime; // set cookie expiry time to 1 hour x notNowTime (i.e. 1 hour)
 				Cookies.set(cookieName, false, { expires: inXhours }); // set cookie to expire in time specified above
