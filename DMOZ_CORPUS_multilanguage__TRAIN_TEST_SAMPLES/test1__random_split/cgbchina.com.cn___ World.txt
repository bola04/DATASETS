 		function stratMove(obj,json,fn)

 	function doMove(obj,json,fn)
 	{
 		var iCur=0;
 		var attr=null;
 		var dope=true;
 		for(attr in json)
 		{
 			if(attr=="opacity")
 			{
 				iCur=parseInt(getStyle(obj,attr)*100);
 			}
 			else
 			{
 				iCur=parseInt(getStyle(obj,attr));

 			if(isNaN(iCur)){iCur=0;}
 			
 			var iSpeed=parseInt(json[attr]-iCur)/6;
