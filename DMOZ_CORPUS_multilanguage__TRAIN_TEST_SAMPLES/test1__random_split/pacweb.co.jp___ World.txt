 				var fetchNews = function (callback) {
 					var xhr = new XMLHttpRequest();
 					xhr.onload = function () {
 						var result = JSON.parse(this.responseText);
 						console.log(result);
 						callback(result);
 					};
 					xhr.open('GET', '/pp/wp-json/wp/v2/posts?_embed&per_page=4');
 					xhr.send();
 				};
 
 				var formatDate = function (t) {
 					var d = new Date(t);
 					var Y = d.getFullYear();
 					var M = d.getMonth() + 1;
 					var D = d.getDate();
 					var W = ['日','月','火','水','木','金','土'][d.getDay()];
 					return Y + '年' + M + '月' + D + '日（' + W + '）';
 				}
 
 				var bindData = function (tpl, data) {
 					return tpl.replace(/\{.+?\}/gm, function(s){
 						var key = s.slice(1, -1).trim();
 						return data[key] ? data[key] : '';
 					});
 				};
 
 				var render = function (result) {
 					el.innerHTML = result.map(function (item) {
 						return bindData(tpl, {
 							link: item.link,
 							title: item.title.rendered,
 							label: item.news_label,
 							date: formatDate(item.date_gmt),
 							img: item.featured_image_src,
 							bg: 'style="background-image: url(' + item.featured_image_src + ')"',
 							category: item.cat_name
 						});
 					}).join('');
 				};
 
 				fetchNews(render);
 
 			})();
