    <script>
     $(function () {
 		// horni reklama
 		var $topBanner = $('#top-banner');
 		// je na strance horni reklama
 		var topBannerPresent = $topBanner.length > 0;
 
 		// úvodní fotka uzivatele
 		var $userImage = $('#userImage');
 		// je na strance úvodní fotka uzivatele?
 		var userImagePresent = $userImage.length > 0;
 
 		// horni pruh s menu
 		var $topBar = $('#topBar');
 
 		var onScroll = function (param) {
 			if ($('body').hasClass('mikro')) {
 				return;
 			}
 			var scrollTop = $(window).scrollTop() || document.documentElement.scrollTop;
 
 			var lastScrollTop = $topBanner.data('lastScrollTop');
 			if (lastScrollTop == scrollTop) {
 				return;
 			}
 			$topBanner.data('lastScrollTop', scrollTop);
 
 			var aboveAdvertH = 0;
 			var advertH = 0;
 
 			if (userImagePresent) {
 				aboveAdvertH = $('#userImagePlaceholder').height();
 			}
 
 			if (topBannerPresent) {
 				advertH = $topBanner.height();
 			}
 
 			// stránka s úvodní fotkou, bez reklamy, s původním scrollem = zobraz průhledné nemu
 			if (userImagePresent && !topBannerPresent && aboveAdvertH == scrollTop) {
 				$topBar.attr('class', 'homepage');
 				$('#topBarGradient').show();
 				return;
 			}
 
 			//reklama neni videt, je zasunuta nahore
 			if (scrollTop >= aboveAdvertH + advertH) {
 				if ($topBar.hasClass('homepage') || $topBar.hasClass('inPlace')) {
 					if ($topBar.hasClass('homepage')) {
 						$topBar.hide();
 						$topBar.fadeIn();
 						$('#topBarGradient').hide();
 					}
 					$topBar.attr('class', '');
 				}
 			}
 
 			// scroll dolu, reklama zajizdi za horni okraj
 			else if (scrollTop >= aboveAdvertH) {
 				var $topBannerContent = $('#top-banner .content');
 
 				// pokud horní reklama není schovaná
 				if (!$topBannerContent.data('hiding')) {
 					if (!$topBar.hasClass('inPlace')) {
 						$topBar.attr('class', 'inPlace');
 						$('#topBarGradient').hide();
 					}
 				}
 				// pokud uz je schovaná, tak nastav pruhledne menu
 				else {
 					// nastav pruhledne menu
 					if (!$topBar.hasClass('homepage')) {
 						$topBar.hide();
 						$topBar.fadeIn();
 						$('#topBarGradient').fadeIn();
 						$topBar.attr('class', 'homepage');
 					}
 				}
 			}
 
 			// scroll nad reklamu = schovani reklamy
 			else if (userImagePresent) {
 				// nastav pruhledne menu
 				if (!$topBar.hasClass('homepage')) {
 					$topBar.hide();
 					$topBar.fadeIn();
 					$('#topBarGradient').fadeIn();
 					$topBar.attr('class', 'homepage');
 
 					var $topBannerContent = $('#top-banner .content');
 					if (!$topBannerContent.data('hiding')) {
 						$topBannerContent.fadeOut();
 						$topBannerContent.data('hiding', true);
 					}
 				}
 			}
 		};
 
 		onScroll();
 		$(window).scroll(onScroll);
 	});
    </script>
