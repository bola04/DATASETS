 	function Mira_Habitaciones()
 	{
             if (Comprueba_Fechas(2))
                 {
 			var hab;
 			var cadena;
 			hab = document.form1.num_habitaciones.options[document.form1.num_habitaciones.selectedIndex].value;
 			contenedor = document.getElementById("form_habitaciones");
 
 			if (hab != 999)
 			{
 
                                 ajax404=nuevoAjax();
 
 
 				var randomnumber=Math.floor(Math.random()*11);
 				ajax404.open("GET", "reservas/procesos1.php?fes=sel&nhab="+hab+ "&rdm=" + randomnumber,true);
 				ajax404.onreadystatechange=function() {
 
 					if (ajax404.readyState==4) {
 						contenedor.innerHTML = ajax404.responseText;
 						cadena = contenedor.innerHTML;
 
 						if (cadena.search('No hay habitaciones disponibles') > 0)
 						   {
 							   NoPasar = 1;
 							}
 						else
 						   {
 							   NoPasar = 0;
 							}
 					}
 				}
 				ajax404.send(null);
 			}
 			else
 			{
 				alert("Para ofertas de grupo pongase en contacto con nosotros.");
 				contenedor = document.getElementById("num_habitaciones");
 				contenedor.value="1";
 			}
 		}
 	}
