     //convert DateTime (dd-mm-yyyy hh-mm) to javascript DateTIme
     //Ex: 16-11-2015 16:05
     function toJSDate( dateTime ) {
 
         var dateTime = dateTime.split(" ");//dateTime[0] = date, dateTime[1] = time
 
         var date = dateTime[0].split("-");
         var time = dateTime[1].split(":");
 
     //(year, month, day, hours, minutes, seconds, milliseconds)
     //subtract 1 from month because Jan is 0 and Dec is 11
         return new Date(date[2], (date[1]-1), date[0], time[0], time[1], 0, 0);
 
     }
 
     //Check to see if the DateTime is in the future
     //param: dateTime must be a JS Date Object
     //return True if DateTime is after Now
     //return False if DateTIme is before Now
     function futureDateTime( dateTime ) {
         var now = new Date();
         var future = false;
         var dateNow = Date.parse(now);
         var second=1000, minute=second*60, hour=minute*60, day=hour*24, week=day*7;
         var dateArray = [0,0,0,0,0, false];
         if( dateNow < Date.parse(dateTime) ) {
             var timediff = Date.parse(dateTime) - dateNow;
             dateArray[0] = dateTime.getFullYear() - now.getFullYear();
             dateArray[1] = ( dateTime.getFullYear() * 12 + dateTime.getMonth() ) - ( now.getFullYear() * 12 + now.getMonth() );
             dateArray[2] = Math.floor(timediff / day);
             dateArray[3] = Math.floor(timediff / hour);
             dateArray[4] = Math.floor(timediff / minute);
             dateArray[5] = true;
 
         }
 
         return dateArray;
     }
     var setTime = function(langx){
         var dt = new Date();
         var mo =  (dt.getMonth() < 10 ? '0' : '') +  (dt.getMonth() + 1);
         var da =  (dt.getDate() < 10 ? '0' : '') + dt.getDate();
         var daNext = (dt.getDate() < 10 ? '0' : '') + (dt.getDate() + 1);
         var time13 = da + "-" + mo + "-" + dt.getFullYear() + " 13:00";
         var time16 =  da + "-" + mo + "-" + dt.getFullYear() + " 16:30";
         var time16Next =  daNext + "-" + mo + "-" + dt.getFullYear() + " 13:00";
         var is13 = futureDateTime(toJSDate(time13));
         var is16 = futureDateTime(toJSDate(time16));
         var is16Next = futureDateTime(toJSDate(time16Next));
         var inText = "";
         var preText = "Vaduz erleben in";
         if(is13[5]){
             //vor 13:00
             inText += getText(is13[4], langx);
         }else{
             if(is16[5]){
                 //vor 16:30
                 inText += getText(is16[4], langx);
             }else{
                 //nächster Tag
                 inText += getText(is16Next[4], langx);
                 //preText = "Heute keine öffentliche Fahrten mehr";
                 //inText = "";
             }
         }
 
         if(inText != ""){

             $( ".preText" ).replaceWith( "<div class='preText'>" + preText + "&nbsp;" + "</div>" );

     }
     function getText(inMinutes, lang){
         var inHours = 0;
         var inText = "";
         var inDays =  Math.floor(inMinutes / (60*24));
         if((inDays > 0)){
             if(inDays == 1){
                 inText +=  inDays + " Tag";
             }else{
                 inText +=  inDays + " Tagen";
             }
             inMinutes = inMinutes - (inDays * (60 * 24));
         }
 
         inHours = Math.floor(inMinutes / 60);
         if((inDays > 0 && inHours != 0)){
             inText += " und ";
         }
         if((inHours > 0)){
             if(inHours == 1){
                 inText +=  inHours + " Stunde";
             }else{
                 inText +=  inHours + " Stunden";
             }
             inMinutes = inMinutes - (inHours * 60);
         }
         if((inHours > 0 && inMinutes != 0)){
             inText += " und ";
         }
 
         if(inMinutes > 0){
             if(inMinutes == 1){
                 inText +=  inMinutes + " Minute";
             }else{
                 inText +=  inMinutes + " Minuten";
             }
         }
         if(lang == "en"){
             inText = inText.replace("Stunden", "hours").replace("Minuten", "minutes");
             inText = inText.replace("Tag", "day").replace("Stunde", "hour").replace("Minute", "minute").replace("und" , "and");
         }
         return inText;
     }
