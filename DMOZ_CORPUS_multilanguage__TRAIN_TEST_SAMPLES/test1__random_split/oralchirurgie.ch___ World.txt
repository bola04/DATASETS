             function AjaxEmail(object, anyway)
             {
                 $(".message").removeClass("open");
 
                 var date = object.find(".datetimepicker").val(),
                     name = object.find(".callback__name").val(),
                     email = object.find(".callback__email").val(),
                     message = object.find("textarea").val();
 
                 $.ajax({
                     type: "POST",
                     url: "/mail.php",
                     dataType: 'json',
                     data:
                     {
                         lang: "de",
                         name: name,
                         email: email,
                         message: message,
                         date: date,
                         anyway: anyway
                     },
                     beforeSend: function()
                     {
                         object.find("i").addClass("show");
                     },
                     success: function(data)
                     {
                         object.find("i").removeClass("show");
 
                         if (data.flag == true)
                         {
                             $("#email_success").addClass("open").find("p").text(data.message);
 
                             setTimeout(function()
                             {
                                 $(".message").removeClass("open");
                             }, 6000);
                         }
                         else
                         {
                             $("#email_appointment_late").addClass("open").find("p").text(data.message);
                         }
 
                         return false;
                     },
                     error: function()
                     {
                         object.find("i").removeClass("show");
                         $("#email_error").addClass("open").find("p").text("Ein unbekannter Fehler ist aufgetreten. Bitte versuchen Sie es erneut");
 
                         setTimeout(function()
                         {
                             $(".message").removeClass("open");
                         }, 6000);
                     }
                 });
             }

             function DisplayNotice(elem, data, position)
             {
                 var parent = $(elem).parent(),
                     notice = '<div class="notice notice-' + position + '"> \
                                 <div class="triangle-' + position + '">' + data + '</div> \
                               </div>';
 
                 parent.find(".notice").remove();
                 parent.css("position", "relative").append(notice);
                 notice = parent.find(".notice");
                 notice.fadeIn(200);
 
                 switch (position)
                 {
                     case "top":
                         notice.css("top", $(elem).outerHeight() + 10);
                         break;
 
                     case "bottom":
                         notice.css("top", -notice.height() - 10);
                         break;
 
                     case "right":
                         notice.css("left", -notice.width() - 20);
                         break;
 
                     case "left":
                         notice.css("right", -notice.width() - 20);
                         break;
                 }
             }
