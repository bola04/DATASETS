    function checkWidth(init) {
             /*If browser resized, check width again */
 
             if ($(window).width() >= 768) {
                 $('body').addClass('desk-large');
             }
             else {
                 if (!init) {
                     $('body').removeClass('desk-large');
                 }
             }
 
             if ($(window).width() <= 1024) {
                 $('body').addClass('desk');
                 $('body').removeClass('mobile');
             }
             else {
                 if (!init) {
                     $('body').removeClass('desk');
                 }
             }
 
             if ($(window).width() <= 767) {
                 $('body').addClass('mobile');
             }
             else {
                 if (!init) {
                     $('body').removeClass('mobile');
                 }
             }
 
             if ($(window).width() >= 1025) {
                 $('body').addClass('menu-large');
             }
             else {
                 if (!init) {
                     $('body').removeClass('menu-large');
                 }
 
             }
         }
         $(document).ready(function () {
             checkWidth(true);
 
             $(window).resize(function () {
                 checkWidth(false);
             });
