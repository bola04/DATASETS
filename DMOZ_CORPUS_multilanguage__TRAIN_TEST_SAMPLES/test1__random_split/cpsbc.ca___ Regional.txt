     // Get the category, action and label from the element and send it to GA. The action is optional, because mostly it will be a click event.
     var trackClickEvent = function (event) {
 //          event.preventDefault();
         var eventCategory = this.getAttribute("data-event-category");
         var eventAction = this.getAttribute("data-event-action");
         // Concatenate internal to the eventAction variable if users are internal
         if (Drupal.settings.custom_settings.internalUser === 'true') {
             eventAction = this.getAttribute("data-event-action") + ' - Internal';
         } else {
             eventAction = this.getAttribute("data-event-action");
         }
 //        console.log(eventAction);
         var eventLabel = this.getAttribute("data-event-label");
         var eventValue = this.getAttribute("data-event-value");
 
         // Tracking events using Matomo Javascript trackEvent()
         _paq.push(['trackEvent', eventCategory, (eventAction != undefined && eventAction != '' ? eventAction : 'click'), eventLabel, eventValue]);
 
         // Tracking events using Google Analytics
         ga('send', 'event', eventCategory, (eventAction != undefined && eventAction != '' ? eventAction : 'click'), eventLabel, eventValue);
     };
