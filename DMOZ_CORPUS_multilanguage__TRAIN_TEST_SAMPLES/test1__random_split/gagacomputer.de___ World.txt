 var xmlHttp = null;
 var HistoryIndex = 0;
 
 
 /**
 * Ajax initialisation
 *
 * check Browser and set Requstmethod
 **/
 function ajaxInit() 
 { 
 	try {
 		// Mozilla, Opera, Safari sowie Internet Explorer (ab v7)
 		if (window.XMLHttpRequest) {
 				xmlHttp = new XMLHttpRequest();				
 				if (xmlHttp.overrideMimeType) {
 					xmlHttp.overrideMimeType("text/html; charset=utf-8");
 				}
 		} else	if (!xmlHttp) {
 				// Internet Explorer 6 und älter
 				try {
 						xmlHttp  = new ActiveXObject("Msxml2.XMLHTTP");
 				} catch(e) {
 						try {
 								xmlHttp  = new ActiveXObject("Microsoft.XMLHTTP");
 						} catch(e) {
 								xmlHttp  = null;
 						}
 				}
 		} else {
 			//alert("Kein Ajax Support !")			
 		}
 	} catch(e) { 
 		alert("Fehler: " + e); 
 	} 
 }
 
 
 /**
 * @param string	URL String i.e. http://www.test.de?test=1&counter=9
 * @return void
 */
 function ajaxGo(url, content, message) 
 {
 	// alert(url);
 	// alert(content);
 	
 	// ersetzen des + in urlencodeformat %2B
 	
 	if(!message) message = 'lade Content';
 	ajaxInit(); 
 	if (xmlHttp) { 
 		xmlHttp.open("POST", url, true); 
 		xmlHttp.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
 		xmlHttp.onreadystatechange = function () {_ajaxCallback(url, message);}; 
 		xmlHttp.send(content); 
 	} 
 }
 
 
 function ajaxFormGo(url, formname, message) 
 {
 	var ajaxstring = '';
 	
 	for (var i = 0; i < document.forms[formname].length; i++) {
 		var value= '';
 		
 		switch(document.forms[formname].elements[i].type) {
 			case 'checkbox':
 					if(document.forms[formname].elements[i].checked == true)
 					{
 						if(document.forms[formname].elements[i].value != null)
 						{ 
 							//value = document.forms[formname].elements[i].value;
 							if(document.forms[formname].elements[i].name != '')
 							{
 								// ersetzen des + in urlencodeformat %2B
 								content = document.forms[formname].elements[i].value.replace(/\+/g,'%2B');
 								ajaxstring = ajaxstring + '&' + document.forms[formname].elements[i].name + '=' + escape(content);
 							}
 						}else //value = document.forms[formname].elements[i].checked;
 							if(document.forms[formname].elements[i].name != '')
 							ajaxstring = ajaxstring + '&' + document.forms[formname].elements[i].name + '=' + document.forms[formname].elements[i].checked;
 					}
 				break;
 			case 'radio':
 					if(document.forms[formname].elements[i].checked)
 						//value = document.forms[formname].elements[i].value;
 						if(document.forms[formname].elements[i].name != '')
 						{
 							// ersetzen des + in urlencodeformat %2B
 							content = document.forms[formname].elements[i].value.replace(/\+/g,'%2B');
 							ajaxstring = ajaxstring + '&' + document.forms[formname].elements[i].name + '=' + escape(content);
 						}
 				break;
 			default:
 						if(document.forms[formname].elements[i].name != '')
 						{
 							// ersetzen des + in urlencodeformat %2B
 							content = document.forms[formname].elements[i].value.replace(/\+/g,'%2B');
 							ajaxstring = ajaxstring + '&' + document.forms[formname].elements[i].name + '=' + escape(content);
 						}
 						//value = document.forms[formname].elements[i].value;
 		}
 		
 		//if(document.forms[formname].elements[i].name != '') {
 		//	ajaxstring = ajaxstring + '&' + document.forms[formname].elements[i].name + '=' + value;
 		//}
 	}
 	
 	if(!message) message = 'lade Content';
 	ajaxInit(); 
 	if (xmlHttp) { 
 		xmlHttp.open("POST", url, true); 
 		xmlHttp.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
 		xmlHttp.onreadystatechange = function () {_ajaxCallback(url, message);}; 
 		xmlHttp.send(ajaxstring); 
 	} 
 }
 
 
 
 function trim (zeichenkette) {
   // Erst führende, dann Abschließende Whitespaces entfernen
   // und das Ergebnis dieser Operationen zurückliefern
   return zeichenkette.replace (/^\s+/, '').replace (/\s+$/, '');
 }
 
 
 
 /**
 *
 * onLoading = What to do when Ajax starts sending the request
 * onLoaded = What to do when Ajax is finished sending the request
 * onInteractive = What to do when Ajax is waiting for response.
 * onCompletion = what to do when ajax is finished with it's request
 *
 * 0 = uninitialized
 * 1 = loading(onLoading method)
 * 2 = loaded(onLoaded method)
 * 3 = interactive(onInteractive method)
 * 4 = complete(onCompletion method)
 *
 **/
 function _ajaxCallback(url,message)
 {
 	switch (xmlHttp.readyState){
 		case 1:
 			setTimeout("SystemMessage.ShowLoadScreen('"+message+"')");
 			//alert("onLoading");
 		break;
 		case 2:
 			//alert("onLoaded");
 		break;
 		case 3:
 			//alert("onInteractive");
 		break;
 		case 4:
 			if (200 != xmlHttp.status) { 
 				//alert("Error " + xmlHttp.status + ": " + xmlHttp.statusText); 
 			} else { 
 				if (xmlHttp.responseText) {
 					_ajaxSetContent(xmlHttp.responseText.split("<!-- ajax splitter -->"));
 				}
 			} 
 			setTimeout("SystemMessage.HideSplash()");
 		break;
 	}
 	//setTimeout("SystemMessage.HideSplash()");
 }
 
 
 /**
 *  set the  Contents to the
 *  div-Tags
 *
 **/
 function _ajaxSetContent(aresponseText)
 {
 	for (var i = 0; i < aresponseText.length; i++) {
 		var sContent = aresponseText[i].split("____");
 		if (sContent[0] == "cmd") {
 			eval(sContent[1]);
 		} else if (sContent[0] == "message") {
 			setTimeout("SystemMessage.Load('"+sContent[1]+"')");
 		} else if (document.getElementById(sContent[0]) && sContent[0] && sContent[1]) {
 			document.getElementById(sContent[0]).innerHTML = sContent[1];	
 			
 		} else if (document.getElementById('ajax_search_result') && sContent[0] && sContent[1]) {
 			document.getElementById('ajax_search_result').innerHTML = sContent[1];	
 		  //addHistoryEvent(HistoryIndex,sContent[1]);
 		  //HistoryIndex++;
 		} else if (document.getElementById('ajax_search_result')) {
 			document.getElementById('ajax_search_result').innerHTML = "";
 		} else {
 			alert(sContent[0] + '------------------------' + sContent[1]);
 		}
 	}						
 }
 
 
 function urlencode(str) {
 	return escape(str).replace('+', '%2B').replace('%20', '+').replace('*', '%2A').replace('/', '%2F').replace('@', '%40');
 }
 
 
 function urldecode(str) {
 	return unescape(str.replace('+', ' '));
 }
