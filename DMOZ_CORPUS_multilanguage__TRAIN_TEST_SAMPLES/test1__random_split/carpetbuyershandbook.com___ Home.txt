     var qsPrivacyPolicyLinkFound = false; /* Was Privacy Policy link detected? */
 
     /* Hide ribbon if cookie is set */
     if (document.cookie.replace(/(?:(?:^|.*;\s*)qs_privacy_agreed\s*\=\s*([^;]*).*$)|^.*$/, "$1") == "true") {
         document.getElementById('sitePrivacyModal').style.display = "none";

     function managePrivacyPolicyRibbon()
     {
         /* No need to continue if ribbon is hidden */
         if (!qsPrivacyPolicyRibbon) {
             return;
         }
 
         /* Find Privacy Policy link in reverse order and copy its href. */
         var links        = document.links;
         var linkFound    = document.getElementById("privacyLinkFound");
         var linkNotFound = document.getElementById("privacyLinkNotFound");
         var privacyLinks = ['privacy', 'privacy policy', 'privacy & security', 'privacy statement'];
         for(var i = links.length-1; i >= 0; i--) {
             var linktext = (links[i].innerText || links[i].textContent).toLowerCase().trim();
             if (links[i].id != "privacyLinkFound" && privacyLinks.indexOf(linktext) > -1)
             {
                 linkFound.href = links[i].href;
                 linkNotFound.style.display = 'none';
                 linkFound.style.display = 'inline';
                 qsPrivacyPolicyLinkFound = true;
                 break;
             }
         }
 
         /* Only show popup on first page of session */    
         if (document.cookie.replace(/(?:(?:^|.*;\s*)qs_privacy_agreed\s*\=\s*([^;]*).*$)|^.*$/, "$1") != "true") {
             var d = new Date();d.setTime(d.getTime() + (365*24*60*60*1000));
             var cookieExp = "expires="+ d.toUTCString();
             var cookieName = 'qs_privacy_agreed'; 
             var cookieValue = 'true';
             document.cookie = cookieName + "=" + cookieValue + ";" + cookieExp + ";path=/";
         }
     }
 
     document.getElementById("privacyAgreed").onclick = function() {
         document.getElementById('sitePrivacyModal').style.display = "none";
