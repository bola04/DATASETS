 			formLabelsAndValues=$j(form).serialize();
 			target=$j("body").offset().top;
 			$j(".v65-widgetModalCart-dropdown").hide();
 			$j('html,body').animate({scrollTop: target},400,function() { 
 				//Stuff placed in here will execute twice//
 			});
 			d = new Date();
 			$j(".v65-widgetModalCart-dropdown").show().html(vin65remote.loadingGraphic);
 			$j.getJSON(website + '/index.cfm?method=cart.addToCart&modalLayout=1&' + formLabelsAndValues + '&timeStamp=' + d.getTime() + '&?callback=?', function(response) {
 				if($j("status",response).text() == "2") return;
 				$j("[v65remotejs=modalCart]").replaceWith('<div v65remotejs="modalCart">' + response + '</div>').hide();
 				$j(".v65-widgetModalCart-dropdown").slideDown();
 				$j(".v65-widgetModalCart-itemMessage").fadeIn().slideDown().delay(3000);
 				$j(".v65-widgetModalCart-itemMessage").slideUp().fadeOut();
 			});
 			return false;
 		}
 	}
