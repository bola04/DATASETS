     function getLoginCheck() {
         var actionUrl = 'http://www.worldcamera.co.th/member/checklogin/';
         var loginEmail = $("#loginEmail").val();
         var loginPass = $("#loginPass").val();
         if (loginEmail === '' && loginPass === '') {
             alert("Error! กรุณากรอกข้อมูล");
             return false;
         } else {
             document.getElementById('submit').style.display = 'none';
             document.getElementById('loading').style.display = 'block';
             var dataSet = {username: loginEmail, password: loginPass};
             $.ajax({
                 type: "post",
                 url: actionUrl,
                 data: dataSet,
                 success: function (data) {
                     loginresponmse(data);
                 },
                 error: function () {
                     alert("ผิดพลาด! กรุณาตรวจสอบอินเตอร์เน็ตของท่าน\nหรือติดต่อเจ้าหน้าที่ของเรา");
                     document.getElementById('submit').style.display = 'block';
                     document.getElementById('loading').style.display = 'none';
                 }
             });
             return false;
         }
     }
 
     function loginresponmse(a) {
         console.log(a);
         if (a == 'true') {
             window.location.href = 'http://www.worldcamera.co.th/home/index'; //window.location.href;
         } else {
             alert("ผิดพลาด! หากคุณยังไม่เป็นสมาชิกกรุณาสมัครสมาชิก\nหรือหรอกข้อมูลให้ถูกต้อง");
             document.getElementById('submit').style.display = 'block';
             document.getElementById('loading').style.display = 'none';
         }
     }
