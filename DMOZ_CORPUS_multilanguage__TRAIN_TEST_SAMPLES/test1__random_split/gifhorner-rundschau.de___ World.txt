       !(function($) {
             $(document).ready(function(){
 
                 var $document = $(document),
                     $loginButton = $("#header-top-abo-text"),
                     $loginLink = $("#header-top-abo"),
                     $overlay = $(".header-overlay"),
                     overlayVisibleClass = "header-overlay--visible",
                     ssoCookieName = "sso_data",
                     ssoRegisteredCookieName = "sso_uie_biscuit",
                     aboCookieName = "fpcacc",
                     aboMessageCookieName = "aboMessage",
                     unknownErrorMsg = "Es ist ein Fehler aufgetreten. Bitte versuchen Sie es erneut.";
 
                 function isLoggedIn() {
                     return !!window.readCookie( ssoCookieName );
                 }
 
                 function isRegistered() {
                     return !!window.readCookie( ssoRegisteredCookieName );
                 }
 
                 function hasAboCookie() {
                     return !!window.readCookie( aboCookieName) ;
                 }
 
                 function hasAboMessageCookie() {
                     return !!window.readCookie( aboMessageCookieName );
                 }
 
                 function getAboCookie() {
                     return window.readCookie( aboCookieName );
                 }
 
                 function hasAbo() {
                     var hasAbo = false, aboCookieObj;
 
                     if (hasAboCookie()) {
                         aboCookieObj = JSON.parse(getAboCookie());
                         if (aboCookieObj.hasOwnProperty("secret")) {
                             hasAbo = true;
                         }
                     }
 
                     return hasAbo;
                 }
 
                 function shouldConfirmAboMessage() {
                     return isLoggedIn() && hasAboCookie() && !hasAbo() && !hasAboMessageCookie();
                 }
 
                 function showOverlay() {
                     $overlay.addClass(overlayVisibleClass);
                 }
 
                 function hideOverlay() {
                     $overlay.removeClass(overlayVisibleClass);
                 }
 
                 function hideAllBoxes() {
                     $(".abo-area").hide();
                     $(".pw-area").hide();
                     $("#aboMessageBox").hide();
                     $("#profileBox").hide();
                     $('#reg-success').hide();
                     $('#reg-error').hide();
                     $('#reg-outdated').hide();
                     $('#reg-passwordreset').hide();
                     if ('reg='.indexOf(window.location.hash.substring(1))) {
                         history.pushState('', document.title, window.location.pathname);
                     }
                     hideOverlay();
                 }
 
                 function reloadPage() {
                     location.reload();
 
                     var oldHash = window.location.hash;
                     if (oldHash) {
                         window.location.hash = "";
                         window.location.hash = oldHash;
                     }
                 }
 
                 function setAboCookie(setPermanentCookie) {
                     var ajaxUrl = "https://" + window.location.host + "/secure/sso/dispatch/paidcontent-UserCanRead",
                         getCookieLiveTime = function (cLiveTime) {
                             return setPermanentCookie ? cLiveTime : null;
                         };
 
                     if (isLoggedIn() || isRegistered()) {
                         $.ajax({
                             url: ajaxUrl,
                             type: "POST",
                             data: JSON.stringify({}),
                             dataType: "json",
                             contentType: 'application/json;charset=UTF-8',
                             xhrFields: {
                                 withCredentials: true
                             },
                             success: function(result) {
                                 var canRead = result.canRead;
                                 var currentProductName = result.currentProductName || "";
                                 var lastProductName = result.lastProductName || "";
                                 var isEmployee = result.isEmployee || false;
                                 var secret = result.secret;
                                 var lastProductCreateTime = result.lastProductCreateTime || "";
                                 var validityPeriodMS = result.validityPeriodMS || 0;
                                 var validityPeriodDays = validityPeriodMS / 24 / 60 / 60 / 1000;
                                 var cookieValue;
 
                                 if (canRead == true) {
                                     /* wenn user lesen darf, dann nur secret in cookie */
                                     cookieValue = JSON.stringify({"secret": secret , "currentProductName": currentProductName, "isEmployee": isEmployee });
                                 } else if (canRead == false && lastProductName) {
                                     /* nicht lesen, und Produkt vorher hatte -> productName und Zeitstempel */
                                     cookieValue = JSON.stringify({"lastProductCreateTime":  lastProductCreateTime , "lastAboName":  encodeURI(lastProductName) });
                                 } else {
                                     /* user nicht lesen, und vorher kein Abo (lastProduct leer) -> kein Wert */
                                     cookieValue = JSON.stringify({});
                                 }
 
                                 window.createCookie(aboCookieName, cookieValue, getCookieLiveTime(validityPeriodDays));
                                 reloadPage();
                             },
                             error: function(e) {
                                 var j = JSON.parse(e.responseText);
                                 console.log("setAboCookie failed: " + j.error);
                                 window.createCookie(aboCookieName, '{}', getCookieLiveTime(1));
                                 reloadPage();
                             }
                         });
                     }
                 }
 
                 function deleteCookies() {
                     window.eraseCookie(ssoCookieName);
                     window.eraseCookie(ssoRegisteredCookieName);
                     window.eraseCookie(aboCookieName);
                 }
 
                 function logout() {
                     $.ajax({
                         url: "https://" + window.location.hostname + "/secure/sso/logout",
                         type: "post",
                         xhrFields: {
                             withCredentials: true
                         },
                         complete: function(r) {
                             deleteCookies();
                             window.location.reload();
                         }
                     });
                 }
 
                 function showLoginForm(success) {
                     var url = "https://www.gifhorner-rundschau.de/?widgetid=210926731&view=login_box_bzv&sec=13521";
                     if (success) {

                             showOverlay();

                         error: function(exception) {
                             console.log("Error requesting login form:", exception);
                         }
                     });
                 }
 
                 function ssoSuccessCallback(result, rememberLogin) {
                     function imgCallback() {
                         if ( ++pixelCounter >= numberOfPixels ) {
                             setAboCookie(rememberLogin);
                         }
                     }
 
                     if ( typeof result.evolverCookie === "string" ) {
                         var $pixels = $(result.evolverCookie),
                                 numberOfPixels = $pixels.length,
                                 pixelCounter = 0;
 
                         $("body").append( $pixels );
 
                         $pixels.on({
                             "load": imgCallback,
                             "error": imgCallback
                         });
                     }
                 }
 
                 if (!hasAboCookie()) {
                     setAboCookie();
                 }
 
                 $(document).ready(function() {

                         $(this).removeClass("form__input--error");

                         showLoginForm();

                     $document.on("click", "#header-top-abo", function(e) {
                         if (e) {
                             e.preventDefault();
                         }
                         if (isLoggedIn()) {
                             if ($.utils.isMobile()) {
                                 $("#profileBox").show();
                                 showOverlay();
                             } else {
                                 logout();
                             }
                         } else {
                             showLoginForm();
                         }
                     });
 
                     $document.on("click", ".form__logout", function(e) {
                         e.preventDefault();
                         logout();
                     });
 
                     $document.on("submit", "#loginForm", function(e) {
                         e.preventDefault();
                         var  $form = $(this),
                              emailField = $('#email'),
                              passwordField = $('#password'),
                              rememberLogin = $("#remember", $form).is(':checked'),
                              hasSsoUieBiscuitCookie = document.cookie.indexOf(ssoRegisteredCookieName) != -1;
                         $.ajax({
                             type: "POST",
                             xhrFields: {
                                 withCredentials: true
                             },
                             data: {
                                 userName: $("#email", $form).val(),
                                 password: $("#password", $form).val(),
                                 remember: rememberLogin
                             },
                             url: "https://" + window.location.hostname + "/secure/sso/login",
                             success: function(result) {
                             	if (hasSsoUieBiscuitCookie) {
                             		window.eraseCookie(ssoRegisteredCookieName);
                             	}
                                 ssoSuccessCallback(result, rememberLogin);
                             },
                             error: function(e) {
                                 var errorMsg = e.responseJSON ? e.responseJSON.message : unknownErrorMsg;
                                 /* FIXME: work with error codes and not messages */
                                 if (errorMsg.indexOf('password') > -1) {
                                     passwordField.addClass('form__input--error');
                                 } else {
                                     emailField.addClass('form__input--error');
                                 }
                                 $(".form__error", $form).text(errorMsg);
                             }
                         });
                     });
 
                     $document.on("click", ".form__label--reset-pw", function(e) {
                         e.preventDefault();
                         $(".abo-area").hide();
                         $.ajax({
                             type: "POST",
                             url: "https://www.gifhorner-rundschau.de/?widgetid=210926731&view=pw_box_bzv&sec=13521",
                             success: function(data) {
                                 $(".pw-area").html(data).show();
                                 showOverlay();
                             },
                             error: function(exception) {
                                 console.log("Error requesting password reset:", exception);
                             }
                         });
 
                     });
 
                     $document.on("submit", "#pwForgottenForm", function(e) {
                         e.preventDefault();
                         var $form = $(this).closest("form"),
                             $error = $(".form__error", $form),
                             $success = $(".form__success", $form);
                         $error.hide();
                         $success.hide();

                             url: "https://" + window.location.hostname + "/secure/sso/resetPassword",
                             xhrFields: {
                                 withCredentials: true
                             },
                             type: "post",
                             data: $form.serialize(),
                             success: function() {
                                 $success.text("Ihre E-Mail wurde erfolgreich versendet!").show();
                             },
                             error: function(e) {
                                 var errorMsg = e.responseJSON ? e.responseJSON.message : unknownErrorMsg;
                                 $error.text(errorMsg).show();
                             }
                         });
                     });
 
                     $document.on("click", ".form__close, .header-overlay", function(e) {
                         e.preventDefault();
                         hideAllBoxes();
                         /* Remember that the abo message box was noticed (closed) */
                         if ($(this).closest("#aboMessageBox").length > 0) {
                             window.createCookie(aboMessageCookieName, "message-confirmed", 1);
                         }
                     });
 
                     /* displaying box after registration */
                     (function() {
                         var hash = window.location.hash.substring(1);
                         if (hash == "reg=suc") {
                             showLoginForm('success');
                         } else if (hash == "reg=err") {
                             showLoginForm('error');
                         } else if (hash == "reg=out") {
                             showLoginForm('outdated');
                         } else if (hash == "reg=pw") {
                             showLoginForm('passwordreset');
                         }
 
                     })();
 
                 });
             });
 
         })(jQuery);
