     function buyukKontrol(id)
 {
 	var a=$("#"+id).val();
 	
 	var turk=["ı","ğ","ü","ş","ç","ö","‘","’","‚","“","”","„","†","‡","‰","‹","›","‾","←","↑","→","↓","!","\"","#","$","%","&","'","(",")","*","+",",","-",".",",","/",":",";","<",">","=","?","@","[","]","&#92;","_","{","}","~","^","¢","£","¤","¥","¦","§","¨","©","«","¬","®","±","¿","À","Á","Â"];
 	var eng =["i","g","u","s","c","o","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","","",""];
 	 
 	a=a.replace("İ","i");
 	a=a.replace("Ğ","g");
 	a=a.replace("Ü","u");
 	a=a.replace("Ç","c");
 	a=a.replace("Ö","o");
 	a=a.replace("Ş","s");
 	a = a.toLowerCase();
 	for(x = 0; x < turk.length; x++){
 		a= a.replace(turk[x],eng[x]);
 	}
 	
 	document.getElementById(id).value=a;
 	
 }
