 		inputRow.find('input').parent().removeClass('error-input');

 		inputRow.find(' .message-text').removeClass('message-success');
 
 		inputRow.find(' .message-text').show();
 		inputRow.find(' .message-text').text(text);
 		inputRow.find('input').parent().addClass('message-input');
 		if (status == 'error') {

 	var  submitLoginFormEvent = function() {
 		$('.mfp-content .auth-popup-section form').submit(function(event) {
     		var openenForm = $('.mfp-content .auth-popup-section form');
     		var formData = {};
 	        $(".mfp-content .auth-popup-section form input").each(function() {
 				formData[$( this ).attr('name')] = $( this ).val();
 			});
     		clearFormAndLoading();
 	        
 	        $.ajax({
 	            type        : 'POST', // define the type of HTTP verb we want to use (POST for our form)
 	            url         : openenForm.attr('action'),
 	            data        : formData
 			}).success(function(data) {
 				var answerLogin = $(data);
 				var errorMes = answerLogin.find("form#login .error-auth");
 				if ( errorMes.length ) {
 					showErrorMessage(errorMes.html());
 
 				    var limitLogin = answerLogin.find("form#login #answer");
 				    if ( limitLogin.length ) {
 				    	openenForm.find(".answer-caprcha").append('<p class="hint">'+ answerLogin.find('form#login dl').eq(0).find('dt span').html() +'</p>').append(answerLogin.find("form#login input#qa_confirm_id")).append( '<input type="text" name="qa_answer" class="auth-inp answer" />' ).append('<div class="question-text">'+ answerLogin.find('form#login dl').eq(0).find('dt label').html() +'</div>');
 				    }
 
     				$(".mfp-content .auth-popup-section #floatingCirclesG").hide();
 				    $(".mfp-content .auth-popup-section input[type='submit']").show();
 				} else {
 					showAuthorisedMenus(answerLogin);
 					$.magnificPopup.close();
 					showSuccess('Вы успешно вошли в систему.');
 				}
 			});
 	        event.preventDefault();
 	    });
 	}

 	            data        : formData,

 			    processData:false,
 			    contentType:false

 					showErrorMessage(errorMes.html());

 				}
 
 			});

 	    });

 	function send_sms_code() {
         var phone = $('.send-sms-phone-input').val();
 
         if (!phone) {
             return;
         }
 
         $.ajax(
             '/api/v1/sms/validate/' + phone,
             {
                 method: 'POST',
                 dataType: 'json',
                 error: function (jqXHR, textStatus, errorThrown) {
                     showMessageAuthSms('Произошла ошибка при отправке запроса к серверу. Попробуйте позже или сообщите администрации сайта');
         			$('#send-sms-button').show();
         			$('#floatingcircles-sms').hide();
                 },
                 success: function (data, textStatus, jqXHR) {
                     if (data.success) {
                         sent_phone = phone;
                         showMessageAuthSms('СМС отправлена!');
 	        			$('.send-sms-phone-input').hide();
 	        			$('#floatingcircles-sms').hide();
                         $('#check-sms-phone-input').show();
                         $('#login-with-sms-button').show();
                     } else {
                         showMessageAuthSms(data.message);
                         $('#send-sms-button').show();
         				$('#floatingcircles-sms').hide();
                     }
                 }
             }
         );
     }
 
     function check_sms_code() {
         var code = $('#check-sms-phone-input').val();
 
         if (!code) {
             return;
         }
 
         $.ajax(
             '/function_po.php',
             {
                 method: 'POST',
                 data: {
                     "cmd": "check_sms_code",
                     "phone": sent_phone || $('.send-sms-phone-input').val(),
                     "code": code
                 },
                 dataType: 'json',
                 error: function (jqXHR, textStatus, errorThrown) {
                     showErrorAuth('Произошла ошибка при отправке запроса к серверу. Попробуйте позже или сообщите администрации сайта');
                 },
                 success: function (data, textStatus, jqXHR) {
                     if (data.success) {
                         replaceNewDesignAuthHeader();
 						$.magnificPopup.close();
 						showSuccess('Вы успешно вошли в систему.');
 						$('#login-with-sms-button').show();
     					$('#floatingcircles-sms').hide();
     					showMessageAuthSms('Вы успешно вошли в систему.');
                     } else {
 
                         if (data.message) {
                             showMessageAuthSms('Произошла ошибка: ' + data.message);
                         } else {
                             showMessageAuthSms('Неправильный код');
                         }
                     }
                 }
             }
         );
 	}
