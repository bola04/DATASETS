 	function CheckPhoneNumber(TheNumber) {
 		var valid = 0;
 		var GoodChars = "0123456789";		
 		for (var i=0; i <= TheNumber.length -1; i++) {
 			if (GoodChars.indexOf(TheNumber.charAt(i)) != -1) valid++;			
 		}
 		
 		if(valid<10) return false; 
 		else return true;
 	}

 	function CheckPostCode(postcode) {
 		var lengthI = 0; var lengthL = 0;
 		postcode = postcode.toUpperCase();
 		var IChars = "0123456789";
 		var LChars = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";		
 		for (var i=0; i <= postcode.length -1; i++) {
 			if (IChars.indexOf(postcode.charAt(i)) != -1) lengthI++;
 			if (LChars.indexOf(postcode.charAt(i)) != -1) lengthL++;
 		}
 		
 		if(lengthI == 4 && lengthL == 2) return true; 
 		else return false;
 	}
