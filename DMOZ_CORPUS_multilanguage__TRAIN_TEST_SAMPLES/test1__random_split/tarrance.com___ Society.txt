                 var _textHeight = _text.height();
                 var padding = Math.round((maxheight - _textHeight) / 2);

                 _text.css('padding-bottom', padding);

         function get_sliderHeight(minHeight, maxHeight) {
             var mobileMenu = jQuery('.mobile-menu');
             var isMobileMenuVisible = mobileMenu.is(':visible');
             //console.log('Mobile: ', isMobileMenuVisible);
             var windowHeight = jQuery(window).height();
             console.log("Window Height: ", windowHeight);
 
             if (isMobileMenuVisible) {
                 return 0;
             }
 
             var result = Math.round(windowHeight * 0.60);

                 result = maxHeight;
