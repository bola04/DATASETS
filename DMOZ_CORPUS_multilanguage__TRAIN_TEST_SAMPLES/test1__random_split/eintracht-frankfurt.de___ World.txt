         if ($('body').hasClass('landing-page')) {
           window.location = $url;
         }
       }
 
 
       // analyse checkbox
       $('[data-switch-direction]').click(function() {
         $that = $(this);
 
         // if checked save cookie
         if ($('.landing-switch input[type="checkbox"]').is(":checked")) {
           Cookies.set('eintracht_landing_switch', $that.attr('data-switch-direction'), { expires: 365, path: '/'});

       });
 
       // h1 headline from txt
       $.ajax({
         url: "https://www.eintracht.de/fileadmin/design/landing/headline.txt",
         // Work with the response
         success: function( response ) {
           if (response && response !== '' && response.trim() !== '') {
             $('h1.introtext').html(response.trim().replace(/\n/g, '<br>'));

             $('h1.introtext').hide();

       });
