         function showSliderchel(){
 	$container = $('#slider_chel');
 	$loader = $container.find('.loader');
 	$imgscont = $container.find('.adv-slider-container ul');
 	$caplinks = $('#slider_chel table.cap-links');
 	if (!$caplinks.length)
 		return;
 
 	$elements = $('#slider_chel table.cap-links tr.elements');
 	var actid = parseInt($caplinks.find('td:first-child').attr('id').replace('chel_link-',''));
 	var clear = 1;
 	var loading = 0;
 
 	var interval = setInterval( function(){
 		if(!$caplinks.find('td.active').next().get(0)){
 			actid = parseInt($caplinks.find('td:first-child').attr('id').replace('chel_link-',''));
 		}else{
 			actid = parseInt($caplinks.find('td.active').next().attr('id').replace('chel_link-',''));
 		}
 		$aparent = $caplinks.find('td.active').next();
 		$caplinks.find('td.active').removeClass('active');
 		$caplinks.find('td#chel_link-'+actid).addClass('active');
 
 		$elements.find('td.active').removeClass('active');
 		$elements.find('td#chel_el-'+actid).addClass('active');
 
 		$aparent.addClass('active');
 		$elements.find('td#chel_el-'+actid).addClass('active');
 		clear = 0;
 		ChangeImagechel (actid,clear);
 	}, 16000);
 
 	ChangeImagechel (actid,clear);
 
 	$caplinks.find('a').click(function(){
 		if($(this).hasClass('active')){
 			return
 		} else{
 			$caplinks.find('td.active').removeClass('active');
 			$aparent = $(this).parent();
 			$aparent.addClass('active');
 			$elements.find('td.active').removeClass('active');
 			actid = parseInt($aparent.attr('id').replace('chel_link-',''));
 			$elements.find('td#chel_el-'+actid).addClass('active');
 			clear = 1;
 			ChangeImagechel (actid,clear);
 		}
 	});
 
 	function ChangeImagechel(actid,clear,interval){
 		if($caplinks.find('active')){
 			$(this).removeClass('active');
 		}
 		ImgLoaderchel (actid,clear,interval);
 	}
 
 	function ImgLoaderchel(actid,clear,interval){
 		loading = 1;
 		$imgscont.find('li.active').removeClass('active');
 		$loader.show();
 		if (!$imgscont.find('#chel_adv_'+actid+' img.cheked').get(0)){
 			$imgscont.find('li.active').removeClass('active');
 			var loadimg = $imgscont.find('#chel_adv_'+actid+' img');
 			var loader = new Image();
 			var imgSrc = loadimg.attr('src')+($.browser.msie ? '?'+Math.round():'');
 			loader.src = imgSrc;
 
 			$(loader).load(function (){
 				$imgscont.find('#chel_adv_'+actid).parent().addClass('active');
 				$imgscont.find('#chel_adv_'+actid+' img').addClass('cheked');
 				$loader.hide();
 				loading = 0;
 			});
 		}else {
 			$imgscont.find('#chel_adv_'+actid).parent().addClass('active');
 			$loader.hide();
 			loading = 0;
 		}
 		if(clear==1){
 			clearInterval (interval);
 		}
 	}
 
 };
