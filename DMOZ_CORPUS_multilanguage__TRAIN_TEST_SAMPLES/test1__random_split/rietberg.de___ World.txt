             if(windowWidth > mobile_break_point){
 
                 if($(this).hasClass("mega-5"))

                 if($(this).hasClass("mega-4"))
                     count =  4
 
                 if($(this).hasClass("mega-3"))
                     count =  3
 
                 if($(this).hasClass("mega-2"))

                     $(this).children('ul.sub-menu').css("left", "initial" );
                     $(this).children('ul.sub-menu').css("right", "50px" );
                 }else{
                     mmWidth = 280 * count;
                     $(this).children('ul.sub-menu').css("margin-left", -mmWidth/2 );

                 //Insert a <br> to break the Mega Menu
                 $("nav > ul > li > ul > li:nth-child(" + count + ")", this).next('br').remove();
                 $("nav > ul > li > ul > li:nth-child(" + count + ")", this).after("<br>");
             }
