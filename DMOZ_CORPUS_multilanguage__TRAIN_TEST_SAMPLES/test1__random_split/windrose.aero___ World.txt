                 if($("#DepAp").val() == '' || $("#ArrAp").val() == '') {
                     return false;
                 }
                 var set_date = '';
                 if($("#DepDate").val()=='') {
                     var set_schedule_date = schedule_data[$('#DepAp').val() + $('#ArrAp').val()][0].split("-");
                     set_date = set_schedule_date[2] + "-" + set_schedule_date[1] + "-" + set_schedule_date[0];
                 }else{
                     var dep_date = $("#DepDate").val().split(".");
                     if (0 > jQuery.inArray(dep_date[2] + "-" + dep_date[1] + "-" + dep_date[0], schedule_data[$('#DepAp').val() + $('#ArrAp').val()])) {
                         var d = new Date(dep_date[2] + "-" + dep_date[1] + "-" + dep_date[0]);
                         var it = 0;
                         for (; it < 60; it++) {
                             d.setDate(d.getDate() + 1);
                             if(jQuery.inArray( d.toISOString().substr(0, 10), schedule_data[$('#DepAp').val() + $('#ArrAp').val()]) >= 0){
                                 var set_schedule_date = d.toISOString().substr(0, 10).split("-");
                                 set_date = set_schedule_date[2] + "-" + set_schedule_date[1] + "-" + set_schedule_date[0];
                                 break;
                             }
                         }
                         if(set_date==''){
                             var set_schedule_date = schedule_data[$('#DepAp').val() + $('#ArrAp').val()][0].split("-");
                             set_date = set_schedule_date[2] + "-" + set_schedule_date[1] + "-" + set_schedule_date[0];
                         }
                     }else{
                         console.log('dep date in array');
                     }
                 }
                 if(set_date) {
                     $('#DepDate').datepicker("update", set_date);
                     $('#ArrDate').datepicker('setStartDate', set_date);
                     $('#ArrDate').datepicker("update", set_date);
                     console.log(set_date);
                 }else{
                     console.log('no day need to be setuped');
                 }
             }
