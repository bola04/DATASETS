                 if($.cookie('stickySelect_acfrm_type') == null) {
                     cat = category.val();
                 }
 
                 $.each( $('#acfrm_type_price').find('option'), function() {
                     if(!$(this).hasClass('sub_'+cat)) {
                         $(this).hide();
                     } else {
                         $(this).show();
                     }
                 });

             region.change(function (evt) {
                 var reg = $.cookie('stickySelect_acfrm_location');
                 var location = form.prop('action').split('/');
 
                 if(evt.currentTarget.value != reg) {
                     reg = evt.currentTarget.value;
                 }
                 // Clean the URL of all the nonsense
                 for(var i=0; i < location.length; i++) {
                     if(location[i].indexOf('http') >= 0) {
                         location.splice(i, 1);
                     }
                     if(location[i].length <= 0) {
                         location.splice(i, 1);
                     }
                 }
                 location.shift();
                 (location.length <= 2) ? location[(location.length)] = reg : location[(location.length-1)] = reg;
                 form.prop({'action' : '/'+location.join('/')});
             });
 
             category.change(function(evt) {
                 updatePrices();
             });
 
             updatePrices();
 
             form_button.click(function(evt) {
                 evt.stopPropagation();
                 evt.preventDefault();
                 form.unbind('submit');
 
                 if(form.prop('action') !== "booking") {
                     form.submit();
                 }
 
                 if(town.val().length <= 0) {
                     submit = false;
                 }
 
                 if(submit) {
                     form.submit();
                 } else {
                     town.css({'border-color' : 'firebrick'});
                     town.css({'background-color' : 'rgba(178, 34, 34, 0.7)'});
                     setTimeout(function() {
                         town.css({'border-color' : '#FFF'});

             if($results_container_data.height() > parseInt($results_container.css('max-height').replace('px', ''))) {
                 $results_container.css({'height' : $results_container.css('max-height')});
             }
 
             $results_container_data.find('li').fadeOut('slow', function() {
                 $results_container_data.find('li').remove();
             });
 
             call = setTimeout(function() {
                 $.get(  'http://api.welcometoscotland.com/town',
                         {'query' : town_search},
                         function(response) {
                             $.each(response.towns, function() {
                                 var town = $(this)[0];
                                 $results_container_data.append(
                                         $('<li>', {'data-type' : 'text'}).append($('<a>', {'text' : town.name, 'data-town' : town.slug, 'href' : 'javascript:void(-1);', 'onclick' : 'applyTownToSearch(this);'}))
                                 );
                                 $results = $results_container_data.find('li');
                             });
                         }
                 );
             }, 500);
 
             $('input.autocomplete').on('blur', function(evt) {
                 $results_container.slideUp();
             });
 
         });
    </script>
