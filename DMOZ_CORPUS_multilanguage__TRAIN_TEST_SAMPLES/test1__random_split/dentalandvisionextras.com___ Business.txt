             var zip = $("#ZipCode").val();
             var ageRange = $("#AgeRangeID").val();
             var tier = $("#TierID").val();
             var covStartDate = $("#CoverageStartDate").val();
 
             if (zip == '' || tier == '' || ageRange == 0 || tier == 0 || tier == 'WHO WILL NEED COVERAGE' || zip == 'ENTER ZIP CODE') {
                 dcpAlert("Missing Fields", "You must enter a zip code and select your coverage, age range, and effective date to continue with a quote.");
                 $("#btnQuote").data("kendoButton").enable(true);
                 return;}
 
             var data = {
                 zipCode: zip,
                 ageRangeID: ageRange,
                 tierID: tier,
                 covStartDate: covStartDate
             }
 
             console.log(data);
 
             $.ajax({
                 url: submitURL,
                 type: 'POST',
                 cache: false,
                 data: JSON.stringify(data),
                 contentType: 'application/json; charset=utf-8',
                 success: function(result) { validateResult(result); },
                 error: function(result) {
                     dcpAlert('Invalid Zip code.', 'An Error has occurred. Please check your entries and try again.');
                     $("#ErrorMessage").innerHTML = result.message;
                     $("#btnQuote").data("kendoButton").enable(true);
                 }
             });

         function validateResult(response) {
             switch (response.Status) {
             case "Success":
                 url = 'https://www.mydentalcareplus.com/Quote/SelectAPlan';
                 $(location).attr('href', url);
                 break;
             case "Failure":
                 url = 'https://www.mydentalcareplus.com/Quote/NoPlanAvailable';
                 $(location).attr('href', url);
                 break;
             case "Error":
                 dcpAlert("Invalid Zip Code.", response.Message);
                 $("#btnQuote").data("kendoButton").enable(true);
                 break;
             }
         }
