 function doIframe(){
 	o = document.getElementsByTagName('iframe');
 	for(i=0;i<o.length;i++){
 		if (/\bautoHeight\b/.test(o[i].className)){
 			setHeight(o[i]);
 			addEvent(o[i],'load', doIframe);
 		}
 	}
 }
 
 function setHeight(e){
 	if(e.contentDocument){
 		e.height = e.contentDocument.body.offsetHeight + 35;
 	} else {
 		e.height = e.contentWindow.document.body.scrollHeight;
 	}
 }
 
 function addEvent(obj, evType, fn){
 	if(obj.addEventListener)
 	{
 	obj.addEventListener(evType, fn,false);
 	return true;
 	} else if (obj.attachEvent){
 	var r = obj.attachEvent("on"+evType, fn);
 	return r;
 	} else {
 	return false;
 	}
 }

    function Reset_input(id_ensemble)
 {
   // Sélection du bloc contenant les boutons radio dans notre exemple) 
   var id_ensemble = document.getElementById(id_ensemble);
   if(id_ensemble)
     {
       //Initialisation dune variable pour contenir un tableau.
       var tab = new Array();
 
       tab = id_ensemble.getElementsByTagName('input');
       // Liste les éléments du tableau
       for (i=0; i < tab.length; i++)        
           {
             //Si balise de type radio on met checked = false
             	if(tab[i].type == 'radio')  tab[i].checked = false;
 				if(tab[i].type == 'checkbox')  tab[i].checked = false;
           }			
     }
 }

    <h2 class="gras">
     Servoz l'authentique
    </h2>
    <p class="closed">
     Porte d’entrée de la Vallée de Chamonix, Servoz est situé plein sud face au massif du Mont-Blanc.
     <br/>
     Ne manquez pas de goûter au charme et à l’atmosphère de ce village qui a su préserver intacte son architecture, ses cultures et sa nature.
     <br/>
     <br/>
     Entre les pages de "Premier de cordée" tous les amateurs d'alpinisme se souviennent de Servoz. "Gardien" de la vallée, ce village pétri d'authenticité vit sereinement au pied de la magnifique chaîne des Fiz, en gardant un oeil sur son grand frère le mont Blanc, sur lequel Servoz offre une vue majestueuse.
     <br/>
     <br/>
     Authentique et traditionnel village de montagne combinant douceur, convivialité et art de vivre, nous vous invitons à venir partager l’hospitalité des habitants qui vous accueillent toute l'année.
     <br/>
     Découvrez-y ces hameaux ensoleillés et les nombreux points de vue accessibles depuis la commune. Une vue à 360° vous permet d’observer le massif du Mont-Blanc, Pormenaz, la chaine des Fiz, la chaine des Aravis, Tête Noire et le Prarion.
     <br/>
     <br/>
     Profitez également d’un moment gourmand en découvrant la cuisine de qualité des restaurants du village.
     <br/>
     <br/>
     Riche d’oratoires, de fours à pain, de fontaines et bachals, qui participent au charme et à l’ambiance du village, Servoz est classé au patrimoine des Pays de Savoie et peut se découvrir lors des visites guidées qui ont lieu chaque année.
    </p>
