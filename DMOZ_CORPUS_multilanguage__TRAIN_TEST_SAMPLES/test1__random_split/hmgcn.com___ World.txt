         function returnNowTime() {
             var testDate = new Date();
 
             var Year = testDate.getFullYear();
             var Month = testDate.getMonth() + 1;
             if (Month < 10) {
                 Month = "0" + Month;
             }
 
             var Day = parseInt(testDate.getDate());
             if (Day < 10) {
                 Day = "0" + Day;
             }
 
             var hour = parseInt(testDate.getHours());
             if (hour < 10) {
                 hour = "0" + hour;
             }
             var min = parseInt(testDate.getMinutes());
             if (min < 10) {
                 min = "0" + min;
             }
             var second = parseInt(testDate.getSeconds());
             if (second < 10) {
                 second = "0" + second;
             }
