             function setWidth(times, width)
             {
                 theWidth = (width + 4 + 6) * times + 6; 
                 
                 if(theWidth < jQuery(".slider-wrapper").width())
                     jQuery(".thumbs-bottom-left .nivo-thumbs-enabled, .thumbs-bottom-right .nivo-thumbs-enabled, .thumbs-bottom-center .nivo-thumbs-enabled").width(jQuery(".slider-wrapper").width());

             function testThumbs(totest, speed)
             {
                     if(!jQuery(".thumbs-left .nivo-control.active img, .thumbs-right .nivo-control.active img, .thumbs-bottom-left .nivo-control.active img, .thumbs-bottom-right .nivo-control.active img,.thumbs-bottom-center .nivo-control.active img").length)

                         a = totest.children("img").eq(0);

                             o = a.offset().top + a.height() - (t.offset().top + t.height()) + 15;
                             jQuery(".nivo-control").animate({top:"-=" + o + "px"},speed);

                             o =  t.offset().top - a.offset().top + 15;
                             jQuery(".nivo-control").animate({top:"+=" + o + "px"},speed);
                         }
                     }
                     else {
