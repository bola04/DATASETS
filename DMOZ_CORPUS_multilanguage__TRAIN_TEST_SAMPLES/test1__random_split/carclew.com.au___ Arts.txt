      $(document).ready(function() {
     var carousel = $(".carousel");
     var details = $("#details");
     var detailsHeading = $(details).find('h1');
     var detailsSubhead = $(details).find('.sub-head');
 
     // Function to change the text in the details box
     function detailsChange(heading, subhead) {
         detailsHeading.fadeOut(200, function() {
             $(this).text(heading).fadeIn(200);
         });
 
         detailsSubhead.fadeOut(200, function() {
             $(this).text(subhead).fadeIn(200);
         });
     }
 
     // Initialise the Owl Carousel
     carousel.owlCarousel({
         items: 1,
         loop: true,
         nav: false,
         smartSpeed: 1000,
         autoplay: true,
         autoplayTimeout: 5000,
         autoplaySpeed: 1000,
         autoplayHoverPause: true,
         responsiveRefreshRate: 100,
         onInitialized: setupDetails,
     });
 
     // Put text from first slide in the details box
     function setupDetails() {
 	    
         var heading = $('#feature .owl-item.active .text-holder h1').text();
         var subhead = $('#feature .owl-item.active .text-holder .sub-head').text();
 
         detailsChange(heading, subhead);
     }
 
     // Function for carousel change
     // See here http://stackoverflow.com/questions/25769366/owl-carousel-2-current-item-when-page-loads
     carousel.on('initialized.owl.carousel changed.owl.carousel', function(e) {
         if ( ! e.namespace || e.type != 'initialized' && e.property.name != 'position' ) {
             return;
         }
 
         // Get the current carousel index
         var current = e.relatedTarget.current();
 
         // Load items for the staged (active) item
         var items   = $(this).find('.owl-stage').children();
 
         // Get the lowest common ancestor of the elements we need to use
         var item    = items.eq(e.relatedTarget.normalize(current)).find('.text-holder');
 
         var heading = item.find('h1').text();
         var subhead = item.find('.sub-head').text();
 
         detailsChange(heading, subhead);
     });
 
     carousel.append("");
 });
