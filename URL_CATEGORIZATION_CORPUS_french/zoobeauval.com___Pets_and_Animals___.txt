   <script>
    var Slider = (function ($, window) {
         var Defaults = {
           slideTime: 200,
           duration: 4000,
           ratio: 0.3657
         }
         var timer, resizeTimer, SELF
 
         function Slider ($el, options) {
           this.options = $.extend({}, Defaults, options || {})
           this.$el = $el
           this.ruban = this.$el.find('.sl-ruban')
           this.slides = this.$el.find('.sl-slide')
           this.ratio = this.options.ratio
           this.currentSlide = -1
           if (this.slides.length < 1) {
             this.loadPoster()
             return this
           }
           SELF = this
           this.init()
         }
 
         Slider.prototype = {
           init: function () {
             this.update()
             if (this.slides.length > 1) {
               this.createNavigation()
               this.start()
             }
             this.events()
           },
           createNavigation: function () {
             var nav = $('<ul/>', {'class': 'sl-nav'})
             var i = 0, l = this.slides.length
             for (; i < l; i++) {
               var li = $('<li/>')
               if (i == 0) {
                 li.addClass('is-active')
               }
               var a = $('<a/>', {'class': 'sl-bullet'})
 
               a.on('click', function (evt) {
                 evt.preventDefault()
                 SELF.stop()
                 var idx = $(this).parent().index()
                 if (idx == 0) {
                   idx = SELF.slides.length
                 }
                 SELF.currentSlide = idx
                 SELF.goto()
               })
 
               li.append(a)
               nav.append(li)
             }
             this.$el.append(nav)
           },
           update: function () {
             this.ruban.find('.sl-end-slide').remove()
             this.slideWidth = this.$el.width()
             var h = Math.floor(this.$el.width() * this.ratio)
             this.$el.height(h)
             this.slides.each(function (i, el) {
               $(el).width(SELF.slideWidth).height(h)
             })
             this.ruban.append(this.slides.eq(0).clone().addClass('sl-end-slide'))
             this.ruban.width((this.slides.length + 1) * this.slideWidth).height(h)
           },
           updateBullet: function (idx) {
             var bullets = this.$el.find('.sl-bullet')
             var curr = idx
             if (idx == this.slides.length) {
               curr = 0
             }
             bullets.each(function (i, e) {
               $(e).parent().removeClass('is-active')
             })
             bullets.eq(curr).parent().addClass('is-active')
           },
           start: function () {
             timer = window.setTimeout(function () {
               SELF.goto(++SELF.currentSlide)
             }, this.options.duration)
           },
           stop: function () {
             window.clearTimeout(timer)
           },
           goto: function (idx) {
             var index = isNaN(idx) ? this.currentSlide : idx
             if (index < 0) {
               this.currentSlide = index = this.slides.length
               this.ruban.css('left', -(this.slides.length * this.slideWidth))
             }
             if (index > this.slides.length) {
               this.currentSlide = index = 1
               this.ruban.css('left', 0)
             }
             this.ruban.animate({
               left: -index * this.slideWidth
             }, this.slideTime, function () {
               SELF.updateBullet(index)
             })
             if (this.slides.length > 1) {
               this.start()
             }
           },
           events: function () {
             var hc = new Hammer(this.$el.get(0))
             this.$el.on('mouseover', function (evt) {
               evt.preventDefault()
               SELF.stop()
             })
             this.$el.on('mouseleave', function (evt) {
               evt.preventDefault()
               SELF.start()
             })
             $(window).on('resize', function () {
               window.clearTimeout(resizeTimer)
               SELF.stop()
               resizeTimer = window.setTimeout(function () {
                 SELF.currentSlide = 0
                 SELF.update()
                 SELF.goto()
               }, 250)
             })
             hc.on('swipeleft', function (evt) {
               evt.preventDefault()
               SELF.stop()
               SELF.currentSlide += 1
               SELF.goto()
               return false
             })
             hc.on('swiperight', function (evt) {
               evt.preventDefault()
               SELF.stop()
               SELF.currentSlide -= 1
               SELF.goto()
               return false
             })
           },
           loadPoster: function () {
             var poster = this.$el.find('#sl-poster')
             if (poster.length) {
               var imgUrl = poster.attr('data-slider-poster')
               var i = $('<img/>')
               i.on('load', function () {
                 poster.append(i)
                 poster.show()
               }).attr('src', imgUrl)
             }
           }
         }
         return Slider
       })(jQuery, this)
 
       var slider = new Slider($('#slider'))
