         function validateRegistrationNum(url_site,urlImmat) {
             var RegistrationNum = $('#RegistrationNum').val().trim().replace(/\-/g, '');
             if(RegistrationNum.length > 0){
                 var Regex1 = new RegExp('^[a-zA-Z]{2}-?[0-9]{3}-?[a-zA-Z]{2}$');
                 var Regex2 = new RegExp('^[0-9]{1,4}-?[a-zA-Z]{1,3}-?[0-9]{2,3}$');
                 var Regex3 = new RegExp('^[0-9]{1,4}-?[a-zA-Z]{1,3}-?2[a-bA-B]$');
 
                 if (Regex1.test(RegistrationNum) || Regex2.test(RegistrationNum) || Regex3.test(RegistrationNum)) {
                     $('#ErrorRegistrationNum').hide();
                     $('#RegistrationNum').removeClass('errorInput');

                 }

                 $('#RegistrationNum').removeClass('errorInput');
                 window.open(url_site,'_blank');
             }

         }

     $(function(){
         $('body a').on('click',function(e){
             if(url_supplements.length>0
                 && $(this).attr('href').indexOf(url_supplements)==-1
                 && $(this).attr('href').substring(0,1)!='/'
                 && $(this).attr('href').substring(0,1)!='#'
             ){
                 var url_supp = url_supplements;
                 var url = $(this).attr('href');
                 if(url.indexOf(location.href)==-1){
                     var pos_query = url.indexOf('?');
                     if(pos_query==-1){
                         var pos_achor = url.indexOf('#');
                         if(pos_achor==-1){
                             pos_query = url.length-1;
                         }else{
                             pos_query = pos_achor-1;
                         }
                         url_supp = '?'+url_supp;
                     }else{
                         url_supp = url_supp+'&';
                     }
                     url = url.substr(0,pos_query).concat(url_supp,url.substr(pos_query+1));
                     $(this).attr('href',url);
                 }
             }
         });
     });
