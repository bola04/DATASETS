 function PrixTimer(heures,minutes,secondes) {
 	if(heures==null) var heures = jQuery(".decompte ul li.heures span").text();
 	if(minutes==null) var minutes = jQuery(".decompte ul li.minutes span").text();
 	if(secondes==null) var secondes = jQuery(".decompte ul li.secondes span").text();
 	else var secondes = secondes-1;
 	if(secondes == -1) {var secondes = 59;var minutes = minutes -1;}
 	if(minutes == -1) {var minutes = 59;var heures = heures -1;}
 	jQuery(".decompte ul li.heures").css({ backgroundPosition: "0px "+(heures* -21)+"px" });
 	jQuery(".decompte ul li.minutes").css({ backgroundPosition: "0px "+(minutes* -21)+"px" });
 	jQuery(".decompte ul li.secondes").css({ backgroundPosition: "0px "+(secondes* -21)+"px" });
 	if(heures+minutes+secondes>0) setTimeout("PrixTimer("+heures+","+minutes+","+secondes+")",1000);
 }
 
 function DigitalTimer(heures,minutes,secondes) {
 	if(heures==null) var heures = jQuery(".compte ul li.heures span").text();
 	if(minutes==null) var minutes = jQuery(".compte ul li.minutes span").text();
 	if(secondes==null) var secondes = jQuery(".compte ul li.secondes span").text();
 	else var secondes = secondes+1;
 	if(secondes == 60) {var secondes = 0;var minutes = minutes +1;}
 	if(minutes == 60) {var minutes = 0;var heures = heures +1;}
 	if(heures == 24) {var heures = 0;}
 	jQuery(".compte ul li.heures").css({ backgroundPosition: "0px "+(heures* -21)+"px" });
 	jQuery(".compte ul li.minutes").css({ backgroundPosition: "0px "+(minutes* -21)+"px" });
 	jQuery(".compte ul li.secondes").css({ backgroundPosition: "0px "+(secondes* -21)+"px" });
 	if(heures+minutes+secondes>0) setTimeout("DigitalTimer("+heures+","+minutes+","+secondes+")",1000);
 }
