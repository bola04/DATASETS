        var accountInfoSessionMobile;
     jQuery(function() {
         var requestAccountInfoUrl = "https://www.alinea.fr/checkout/session/getSessionInfo/";
         // On vérifie si les cookies sont activés.
         if (true == isCookiesEnabled()) {
             var accountInfoSession = getCookie('customerInfo');
             if (0 != accountInfoSession.length) {
                 updateInfoCustomer(JSON.parse(accountInfoSession));
             }
         } else {
             jQuery.ajax({
                 url: requestAccountInfoUrl,
                 cache: false,
                 dataType: 'json',
                 success: function (accountInfoSession) {
                     updateInfoCustomer(accountInfoSession);
                 }
             });

         accountInfoSessionMobile = accountInfoSession;
     });
     function updateInfoCustomer(accountInfoSession) {
         var header = jQuery('#header-main');
         var headerLogged = jQuery(header).find('.isLogged');
         var headerNotLogged = jQuery(header).find('.isNotLogged');
         if (true == accountInfoSession.isLogged) {
             // The user is logged in
             jQuery(headerLogged).removeClass('hidden');
             jQuery(headerNotLogged).addClass('hidden');
             // Display firstname and lastname
             jQuery(headerLogged).find('.lastname').html(accountInfoSession.lastname);
             jQuery(headerLogged).find('.firstname').html(accountInfoSession.firstname);
         } else {
             // The user is not logged in
             jQuery(headerLogged).addClass('hidden');
             jQuery(headerNotLogged).removeClass('hidden');
         }
         var msgCartCountArticle = "article";
         if (accountInfoSession.nbProductInCart > 1) {
             msgCartCountArticle = "articles";
         }
         jQuery(header).find('.cart-qty').html(accountInfoSession.nbProductInCart);
         jQuery(header).find('.cart-article').html(msgCartCountArticle);
     }
     function isCookiesEnabled() {
         var cookieEnabled = navigator.cookieEnabled;
         // When cookieEnabled flag is present and false then cookies are disabled.
         if (cookieEnabled === false) {
             return false;
         }
         // try to set a test cookie if we can't see any cookies and we're using
         // either a browser that doesn't support navigator.cookieEnabled
         // or IE (which always returns true for navigator.cookieEnabled)
         if (!document.cookie && (cookieEnabled === null || /*@cc_on!@*/false)) {
             document.cookie = "testcookie=1";
             if (!document.cookie) {
                 return false;
             } else {
                 document.cookie = "testcookie=; expires=" + new Date(0).toUTCString();
             }
         }
         return true;
     }

     function addQASAddressesSuggestion()
     {
         //On récupère tous les champs data-qas="street1"
         var streets = jQuery('*[data-qas="street1"]');
         //Pour chaque elecment strret, on definit l'autocomplete
         streets.each(function () {
             var currentform = jQuery(this).closest('form');
             var currentcountry = currentform.find('*[data-qas="country"]');
             //Reafficher tous les champs à la sortie du champs street.
             jQuery(this).blur(function () {
                 if (jQuery(this).val().length > 0) {
                     showAddressInputs(currentform);
                 }
             });
             // Si le pays n'est pas FR, on affiche les autres champs
             if (currentcountry.val() !== 'FR')
             {
                 showAddressInputs(currentform);
             }
             
             var selector = jQuery(this);
             //On definit l'autocomplete pour appel du WS uniquement si country == 'FR'
             jQuery(this).autocomplete({
                 minLength: 5,
                 source: function (request, response) {
                     // Revérification du pays lors de la saisie du formulaire
                     if (currentcountry.val() === 'FR') {
                         callSuggestAddresses(selector, response);
                     }
                 },
                 select: function (event, object) {
                     // Revérification du pays lors de la saisie du formulaire
                     if (currentcountry.val() === 'FR') {
                         fillAddressFields(jQuery(this), object);
                         return false; //Pour éviter que l'autocomplete decide de lui meme de vider les champs
                     }
                 }
             });
         });
     }
     //Fonction d'appel du WS
     function callSuggestAddresses(street1selector, response)
     {
         jQuery.ajax({
             cache: 'false',
             url: '//www.alinea.fr/qas/intuitiveSearch/getSuggestAddresses/',
             dataType: 'json',
             type: "POST",
             data: {
                 search: street1selector.val(),
                 maxRows: 15
             },
             beforeSend: function () {
                 //jQuery("#loader").show();
             },
             success: function (data) {
                 response(jQuery.map(data.suggestResults, function (objet) {
                     return {
                         label: objet.PartialAddress,
                         value: ' ',
                         moniker: objet.Moniker
                     };
                 }));
             },
             complete: function (data) {
                 //jQuery("#loader").hide();
             },
             error: function (data) {
                 //on affiche les champs rempli grâce à la recherche intuitive QAS
                 showAddressInputs(street1selector.closest('form'));
             }
         });
     }
     //Fonction de remplissage des champs suite à la selection
     function fillAddressFields(street1selector, object)
     {
         // Appel AJAX pour récupérer la sélection
         var moniker = object.item.moniker;
         var closestform = street1selector.closest('form');
         var street1 = street1selector.val();
         if (moniker !== null && moniker !== 'manual') {
             jQuery.ajax({
                 cache: 'false',
                 data: {moniker: moniker},
                 type: "POST",
                 url: '//www.alinea.fr/qas/intuitiveSearch/getAddressFromSuggest/',
                 dataType: 'json',
                 success: function (data) {
                     // On affiche les champs rempli grâce à la recherche intuitive QAS
                     if (data.isValid === true) {
                         jQuery.each(data, function (key, value) {
                             switch (key) {
                                 case 'street1':
                                     closestform.find('*[data-qas="street1"]').val(value);
                                     break;
                                 case 'street2':
                                     closestform.find('*[data-qas="street2"]').val(value);
                                     break;
                                 case 'postcode':
                                     closestform.find('*[data-qas="postcode"]').val(value);
                                     break;
                                 case 'city':
                                     closestform.find('*[data-qas="city"]').val(value);
                                     break;
                             }
                         });
                         // On passe la fiabilité à 0 (= adresse fiable)
                         closestform.find('*[data-qas="reliability"]').val(0);
                     } else {
                         street1selector.val(street1);
                     }
                     showAddressInputs(closestform);
                 },
                 complete: function (data) {
                     street1selector.blur();
                 }
             });
         } else {
             street1selector.val(street1);
             closestform.find('*[data-qas="reliability"]').val(1);
             showAddressInputs(closestform);
         }
     }
