 	            var day = $(form).find("select[name=" + selectName + "_day] option:selected").val();
 	            var month = $(form).find("select[name=" + selectName + "_month] option:selected").val();
 	            var year = $(form).find("select[name=" + selectName + "_year] option:selected").val();
 
 	            var birthday = "";
 	            if(day != "" && month != "" && year != "") {
 	                birthday = Tc.Utils.padNumber(day,2) + Tc.Utils.padNumber(month,2) + Tc.Utils.padNumber(year,2);
