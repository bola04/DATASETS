     Cycon.GUI.registerElement('homeSlider330', $('home-page-slider-330'));
 
 		// diese Funktion kümmert sich um die saubere Darstellung der einzelnen SliderPages (Vollbild)
 		Cycon.GUI.registerFunction('onStart,onResize', function(gui){
 			var container	= gui.getRegisteredElement('homeSlider330');
 			var viewport	= document.viewport.getDimensions();
 			var measures	= new $H({
 				'maxWidth': 1024,		// maximale Breite der Stage
 				'maxHeight': 478,		// maximale Höhe der Stage
 				'maxWidthCS': 1920,		// maximale Breite des Coffee Smile
 				'maxHeightCS': 414,		// maximale Höhe des Coffee Smile
 				'standardWidth': 1296	// standard Breite der Stage für Elemente
 			});
 			var newRatio	= Math.max(
 					viewport.height / measures.get('maxHeight'),
 					viewport.width / measures.get('maxWidth')
 			);
 
 			container.setStyle({ 'height': ''.concat(viewport.height, 'px'), 'width': ''.concat(viewport.width, 'px') });
 
 			// Ausführung pro SliderPage
 			container.select('.slider-page').each(function(page){
 				var background			= page.select('img.background').first() || new Element('img');
 				var previewSliderLeft	= page.select('.preview-slider-left').first() || new Element('div');
 				var previewSliderRight	= page.select('.preview-slider-right').first() || new Element('div');
 
 				var focusElement			= page.select('.focus-element').first() || new Element('div');
 				var focusElementStandard 	= page.select('.focus-element-standard').first() || new Element('div');
 				var feLayout 				= new Element.Layout(focusElement);
 				var feLayoutStandard		= new Element.Layout(focusElementStandard);
 
 				page.setStyle({ 'width': ''.concat(viewport.width, 'px'), 'height': ''.concat(viewport.height, 'px') });
 
 				previewSliderLeft.hide();
 				previewSliderRight.hide();
 
 
 				// Focus Element
 				if (viewport.width > measures.get('standardWidth')) {
 					focusElement.setStyle({
 						'left': ''.concat(feLayoutStandard.get('left') + (Math.floor(viewport.width - measures.get('standardWidth')) / 2), 'px')
 					});
 				}
 
 				else {
 					focusElement.setStyle({
 						'left': ''.concat(feLayoutStandard.get('left') - (Math.floor(measures.get('standardWidth') - viewport.width) / 2), 'px')
 					});
 				}
 
 				// Hintergrund
 				if (viewport.height > measures.get('maxHeight') || viewport.width > measures.get('maxWidth')) {
 					var newWidth = measures.get('maxWidth') * newRatio + 2;
 					var newMarginLeft = -1 * Math.floor(measures.get('maxWidth') * newRatio / 2) - 1;
 					background.setStyle({
 						'width': ''.concat(newWidth, 'px'),
 						'marginLeft': ''.concat(newMarginLeft, 'px'),
 						'height': ''.concat(measures.get('maxHeight') * newRatio, 'px'),
 						'marginTop': ''.concat(-1 * Math.floor(measures.get('maxHeight') * newRatio / 2), 'px')
 					});
 				}
 
 				else {
 					background.setStyle({
 						'width': ''.concat(measures.get('maxWidth'), 'px'),
 						'marginTop': ''.concat(-1 * Math.floor(measures.get('maxHeight') / 2), 'px'),
 						'height': ''.concat(measures.get('maxHeight'), 'px'),
 						'marginLeft': ''.concat(-1 * Math.floor(measures.get('maxWidth') / 2), 'px')
 					});
 				}
 
 			});
 
 			var slider = gui.getRegisteredElement('pageSliderHome');
 
 			if (slider)
 				slider.update();
 
 		});
 
 		// diese Funktion erzeugt den Slider selbst
 		Cycon.GUI.registerFunction('onStart', function(gui){
 
 			var slider = new PageSlider('home-page-slider-330', {
 				'cssSelectorPage': '.slider-page',
 				'btnSlideLeft': new Element('a', { 'class': 'page-back', 'href': 'javascript:void(0)' }).update(new Element('div', { 'class': 'preview' })).insert({ 'bottom': new Element('div', { 'class': 'arrow' }) }),
 				'btnSlideRight': new Element('a', { 'class': 'page-next', 'href': 'javascript:void(0)' }).update(new Element('div', { 'class': 'preview' })).insert({ 'top': new Element('div', { 'class': 'arrow' }) }),
 				'carousel': true,
 				'duration': .4,
 
 				'onUpdate': function(pageslider){
 					var offset = ''.concat(parseInt((pageslider.container.getHeight() - pageslider.options.btnSlideLeft.getHeight()) / 2), 'px');
 
 					pageslider.options.btnSlideLeft.setStyle({ 'top': offset });
 					pageslider.options.btnSlideRight.setStyle({ 'top': offset });
 				},
 
 				'onStart': function(pageslider){
 					var page = pageslider.getActualPage();
 					var prvL = page.select('.preview-slider-left').first();
 					var prvR = page.select('.preview-slider-right').first();
 
 					var btnLeft = pageslider.options.btnSlideLeft;
 					var btnRight = pageslider.options.btnSlideRight;
 
 					if (prvL){
 						btnLeft.down('.preview').setStyle({ 'background': ''.concat('url(', prvL.readAttribute('src'), ') no-repeat right center') });
 						btnLeft.observe('mouseover', function(event, element){
 							var effect = element.retrieve('effect');
 							if (effect)
 								effect.cancel();
 
 							element.addClassName('page-back-hover');
 
 							effect = new Effect.Morph(element.down('.preview'), { 'duration': .2, 'style': 'width: 97px' });
 							element.store('effect', effect);
 						}.bindAsEventListener(this, btnLeft));
 						btnLeft.observe('mouseout', function(event, element){
 							var effect = element.retrieve('effect');
 							if (effect)
 								effect.cancel();
 
 							effect = new Effect.Morph(element.down('.preview'), { 'duration': .2, 'style': 'width: 0', 'afterFinish': function(){ element.removeClassName('page-back-hover'); }.bind(element) });
 							element.store('effect', effect);
 						}.bindAsEventListener(this, btnLeft));
 					}
 
 					if (prvR){
 						btnRight.down('.preview').setStyle({ 'background': ''.concat('url(', prvR.readAttribute('src'), ') no-repeat left center') });
 						btnRight.observe('mouseover', function(event, element){
 							var effect = element.retrieve('effect');
 							if (effect)
 								effect.cancel();
 
 							element.addClassName('page-next-hover');
 
 							effect = new Effect.Morph(element.down('.preview'), { 'duration': .2, 'style': 'width: 97px' });
 							element.store('effect', effect);
 						}.bindAsEventListener(this, btnRight));
 						btnRight.observe('mouseout', function(event, element){
 							var effect = element.retrieve('effect');
 							if (effect)
 								effect.cancel();
 
 							effect = new Effect.Morph(element.down('.preview'), { 'duration': .2, 'style': 'width: 0', 'afterFinish': function(){ element.removeClassName('page-next-hover'); }.bind(element) });
 							element.store('effect', effect);
 						}.bindAsEventListener(this, btnRight));
 					}
 				},
 
 				'onClick': function(pageslider, direction){
 					var startAnimation = gui.getRegisteredElement('startAnimation');
 
 					if (startAnimation)
 						startAnimation.stop();
 				},
 
 				'afterScroll': function(pageslider){
 					var page = pageslider.getActualPage();
 					var prvL = page.select('.preview-slider-left').first();
 					var prvR = page.select('.preview-slider-right').first();
 
 					pageslider.options.btnSlideLeft.down('.preview').setStyle({ 'background': ''.concat('url(', prvL.readAttribute('src'), ') no-repeat right center') });
 					pageslider.options.btnSlideRight.down('.preview').setStyle({ 'background': ''.concat('url(', prvR.readAttribute('src'), ') no-repeat left center') });
 				}
 			});
 
 			// Bei Start erst einmal noch das ganze Animieren
 			var pe = new PeriodicalExecuter(function(pe){
 				slider.calculateIndexToTheRight();
 				slider.scroll();
 			}, 10);
 
 			gui.registerElement('pageSliderHome', slider);
 			gui.registerElement('startAnimation', pe);
 		});
 
 		// Fuer das swipen benoetigt
 		function processingRoutine() {
 			var slider = Cycon.GUI.getRegisteredElement('pageSliderHome');
 			if (swipeDirection == 'left') {
 				slider.scrollRight();
 			}
 			else if (swipeDirection == 'right') {
 				slider.scrollLeft();
 			}
 		}

 					viewport.width / measures.get('maxWidth')
