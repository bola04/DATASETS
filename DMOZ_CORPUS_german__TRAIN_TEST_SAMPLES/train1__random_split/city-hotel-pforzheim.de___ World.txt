   // redirect to TLD
     function redirectToTLD(tld) {
       var loc = window.location;
       var hostParts = loc.hostname.split(".");
       var host = "";
       for(var i=0; i<hostParts.length-1; i++)
         host += hostParts[i]+".";
       host += tld;
       var newURL = loc.protocol+"//"+host+loc.pathname+loc.search;
       window.location.href = newURL;
     }
 
     // EXCHANGE IMAGES WITH HIGH QUALITY
     if(screen.width > 480) { // 480 vs 1024 ?
       // non-mobile device
       var images = document.getElementsByTagName("img");
       for(var i = 0; i<images.length; i++) {
         var hq_urlsegments = images[i].src.split(".");
         var hq_url = hq_urlsegments[0];
         for(var j=1; j<hq_urlsegments.length-1; j++)
           hq_url += "."+hq_urlsegments[j];
         hq_url += "_hq."+hq_urlsegments[hq_urlsegments.length-1];
         images[i].src = hq_url;
       }
     }
 
     // MENUES
     // get the menu targeted
     function getMenu(target) {
       for(var i = 0; i<menus.length; i++) {
         var menu = document.getElementById(menus[i]);
         for(var parent = target.parentElement;
               parent.parentElement != null; parent = parent.parentElement)
           if(parent == menu)
             return menu;
       }
       return null;
     }
 
     // opens a new menu
     function openMenu(event) {
       var menu = getMenu(event.target);
       // close open menus
       if(menu != currentOpenMenu) {
         closeMenu(event);
         // open a new one
         if(null != menu) {
           for(var i=0; i<menu.childElementCount; i++)
             menu.children[i].style.display = "block";
           event.preventDefault();
         }
       }
       currentOpenMenu = menu;
     }
 
     // closes the current open menu
     function closeMenu(event) {
       var menu = getMenu(event.target);
       if(menu != currentOpenMenu && currentOpenMenu != null) {
         for(var i=0; i<currentOpenMenu.childElementCount; i++) {
           var entry = currentOpenMenu.children[i];
           if(!entry.classList.contains("current"))
             entry.style.display = "none";
         }
       }
       currentOpenMenu = null;
     }
 
     // mobile function registration
     var currentOpenMenu = null;
     var mobileEvents = false;
     var menus;
     if(screen.width <= 480) {
       mobileEvents = true;
       var menus = ["pages", "languages"];
     } else if(screen.width <= 1024) {
       mobileEvents = true;
       var menus = ["languages"];
     }
     if(mobileEvents)
       document.addEventListener(
         ("ontouchstart" in window)?"touchstart":"click", openMenu);
