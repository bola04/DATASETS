    var agi_stop_ajax = false;
 $(document).ready(function() {

 function send_ajax_basket()
 {
   if (agi_stop_ajax)
     return false;
   var post = '';
   var pID  = 0;
   var form = document.getElementById('cart_quantity');
   for (var i = 0; i<form.elements.length; i++)
   {
     if (form.elements[i].name == 'products_id')
       pID = form.elements[i].value;
     if (form.elements[i].type == 'radio')
     {
       if (form.elements[i].checked)
       {
         post += post == '' ? '' : '&';
         post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].value);
       }
     }
     else if (form.elements[i].type == 'checkbox')
     {
       if (form.elements[i].checked)
       {
         post += post == '' ? '' : '&';
         if (form.elements[i].value)
           post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].value);
         else
           post += form.elements[i].name+'=1';
       }
     }
     else if (form.elements[i].type == 'select')
     {
       if (form.elements[i].multiple)
       {
         for (var k = 0; form.elements[i].options.length; k++)
         {
           if (form.elements[i].options[k].selected)
           {
             post += post == '' ? '' : '&';
             post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].options[k].value);
           }
         }
       }
       else
       {
         post += post == '' ? '' : '&';
         post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].options[k][form.elements[i].selectedIndex].value);
       }
     }
     else if (form.elements[i].type == 'select-one')
     {
       var idx = form.elements[i].selectedIndex;
       post += post == '' ? '' : '&';
       post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].options[form.elements[i].selectedIndex].value);
     }
     else if (form.elements[i].type == 'select-multiple')
     {
       for (var k = 0; form.elements[i].options.length; k++)
       {
         if (form.elements[i].options[k].selected)
         {
           post += post == '' ? '' : '&';
           post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].options[k].value);
         }
       }
     }
     else
     {
       post += post == '' ? '' : '&';
       post += form.elements[i].name+'='+encodeURIComponent(form.elements[i].value);
     }
   }
 
   if ( form.elements['submit_target'].value != 'cart')
     return true;
   
   if (post != '')
   {
     agi_stop_ajax = true;
     $.ajax({
       type: "POST",
       url: "/ajax_flying_cart.php",
       contentType: 'application/x-www-form-urlencoded; charset=ISO-8859-15',
       data: post,
       beforeSend: function(data) { send_ajax_basket_prepare(data, pID) },
       success: function(data) { send_ajax_basket_success(data, pID) },
       error: function(data) { send_ajax_basket_error(data, pID) }
     });
   }
   return false;
 }
 
 function send_ajax_basket_prepare(data, pID)
 {
   $("#products_main_image_wrapper img")
     .clone()
     .prependTo("#products_main_image_wrapper")
     .attr("id", "products_main_image_clone_" + pID)
     .css({'position' : 'absolute'})
     .animate({opacity: 0.8, marginLeft: '10px', marginTop: '-10px'}, 400 )
 }
 
 function send_ajax_basket_success(data, pID)
 {
   agi_stop_ajax = false;
   if (data == '')
   {
     send_ajax_basket_error(data, pID);
     return;
   }
   if (document.getElementById('box_cart'))
   {
     var cloneImageX = $("#products_main_image_clone_" + pID).offset().left;
     var cloneImageY = $("#products_main_image_clone_" + pID).offset().top;
     var cartX       = $("#box_cart").offset().left;
     var cartY       = $("#box_cart").offset().top;
     
     var newWidth   = $("#products_main_image_clone_" + pID).width() / 4;
     var newHeight  = $("#products_main_image_clone_" + pID).height() / 4;
 
     var gotoX           = cartX - cloneImageX;
     var gotoY           = cartY - cloneImageY;
     
     $("#products_main_image_clone_" + pID)
       .animate({opacity: 0, marginLeft: gotoX, marginTop: gotoY, width: newWidth, height: newHeight}, 1200)
       .toggle( function () { 
         $("#products_main_image_clone_" + pID).remove();
         $("#box_cart").empty();
         $("#box_cart").html(data);
       })
   }
 }
 
 function send_ajax_basket_error(data, pID)
 {
   agi_stop_ajax = false;
   $("#products_main_image_clone_" + pID).fadeOut();
 }
