           <h2>
            Herzlich willkommen
           </h2>
           <p>
            Bambali ist die
            <strong>
             100% kostenlose Tauschbörse
            </strong>
            für Waren und Dienstleistungen. Hier verwandeln sich
            <a href="https://www.bambali.de/tauschboerse/2-sammler-fanartikel/">
             alte Schätze
            </a>
            von Dachboden,
            <a href="https://www.bambali.de/tauschboerse/7-b%C3%BCcher-magazine/">
             Bücherregal
            </a>
            oder
            <a href="https://www.bambali.de/tauschboerse/8-mode-accessoires/">
             Kleiderschrank
            </a>
            in etwas
 Nützliches. Online tauschen bei Bambali ist aktives Recycling, schützt die Umwelt,
 spart Geld und bringt Spaß.
            <a href="/register/">
             <u>
              Jetzt anmelden und ausprobieren...
             </u>
            </a>
           </p>
          </section>
          <section class="clearfix">
           <a href="https://www.bambali.de/tauschboerse/6-musik/">
            <img src="https://www.bambali.de/images/dashboard/musik-medien.jpg" title="CDs &amp; Schallplatten tauschen"/>
           </a>
           <h2>
            Medien, Kleidung, Raritäten... einfach alles!
           </h2>
           <p>
            Wir tauschen alles, von
            <a href="https://www.bambali.de/tauschboerse/42-schallplatten/">
             alten Schallplatten
            </a>
            bis
            <a href="https://www.bambali.de/tauschboerse/38-dvd/">
             DVDs
            </a>
            , von Second-Hand-Kleidung bis
 selbstgemachtem
            <a href="https://www.bambali.de/tauschboerse/50-accessoires/">
             Schmuck
            </a>
            . Die lokale Suche unserer Tauschbörse macht
            <a href="https://www.bambali.de/tauschboerse/249-helfen-teilen/">
             Nachbarschaftshilfe
            </a>
            ,
 Fahrgemeinschaften und Tausch von Unterkunft möglich
 - denn Bambalianer leben nachhaltig und bewusst.
           </p>
          </section>
          <section class="clearfix">
           <a href="https://www.bambali.de/tauschboerse/7-b%C3%BCcher-magazine/">

           </a>
           <h2>
            Stöbern, verhandeln, Schnäppchen machen
           </h2>
           <p>
            Bambali ist Online-Flohmarkt, Nachbarschaftshilfe, Tauschring, Verschenkmarkt und Community in einem. Beteilige Dich
 im
            <a href="https://www.bambali.de/forum/">
             Tauschforum
            </a>
            und den
            <a href="https://www.bambali.de/blog/">
             Mitglieder-Blogs
            </a>
            , um die Mitglieder kennen zu lernen. Knüpfe
 Kontakte zu Gleichgesinnten. Tausche, leihe oder verschenke Sachen die Du selten brauchst. Der persönliche
 Kontakt macht unsere Tauschbörse einzigartig.
           </p>

 // Hide Header on on scroll down
 var didScroll;
 var lastScrollTop = 0;
 var delta = 5;
 var navbarHeight = $('#topMenu').outerHeight();
 
 $(window).scroll(function(event){
     didScroll = true;
 });
 
 setInterval(function() {
     if (didScroll) {
         hasScrolled();
         didScroll = false;
     }
 }, 250);
 
 function hasScrolled() {
     var st = $(this).scrollTop();
     
     // Make sure they scroll more than delta
     if(Math.abs(lastScrollTop - st) <= delta)
         return;
     
     // If they scrolled down and are past the navbar, add class .nav-up.
     // This is necessary so you never see what is "behind" the navbar.
     if (st > lastScrollTop && st > navbarHeight){
         // Scroll Down
         $('#topMenu').removeClass('nav-down').addClass('nav-up');
     } else {
         // Scroll Up
         if(st + $(window).height() < $(document).height()) {
             $('#topMenu').removeClass('nav-up').addClass('nav-down');
         }
     }
     
     lastScrollTop = st;
 }
 });
 
 function addLink() {
 	    //Get the selected text and append the extra info
 	    var selection = window.getSelection();
 	    if (selection.toString().length<128) return;
 	    pagelink = " (Quelle: " + document.location.href + ")";
 	    copytext = selection + pagelink;
 	    //Create a new div to hold the prepared text
 	    newdiv = document.createElement('div');
 	    //hide the newly created container
 	    newdiv.style.position = 'absolute';
 	    newdiv.style.left = '-99999px';
 	    //insert the container, fill it with the extended text, and define the new selection
 	    document.body.appendChild(newdiv);
 	    newdiv.innerHTML = copytext;
 	    selection.selectAllChildren(newdiv);
 	    window.setTimeout(function () {
 	        document.body.removeChild(newdiv);
 	    }, 100);
 	}
 	if (!WCF.User.userID) document.addEventListener('copy', addLink);
