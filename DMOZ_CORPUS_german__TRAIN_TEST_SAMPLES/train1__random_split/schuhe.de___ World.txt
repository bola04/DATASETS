     function getSearchResult(start,bSearch,initCall){
         var qBrand,qSeason,qColor, xTrade, usort, qCurrentMin, qCurrentMax;
         var mNewLocation         =   $("#locationSelectInput" ).val();
         if (typeof mNewLocation != 'undefined'){
             if(mNewLocation.trim() != ''){
                 paURL	=	"/ajax.handler.php?axmodule=setLocation&location="+mNewLocation;
                 $.ajax({type: 'GET',url: paURL,async:false});
             }
         }
 //        console.log(mNewLocation)
         qCurrentMin  =   parseInt($('.qCurrentMin').val());
         qCurrentMax  =   parseFloat($('.qCurrentMax').val());
         maxPrice     =   typeof ($('.max-price').attr('data-maxPrice')) == 'undefined' ? 0 : parseFloat($('.max-price').attr('data-maxPrice'));
         qBrand  = {"items":getSelectedFilters('.brand-selector')};
         qSeason = {"items":getSelectedFilters('.season-selector')};
         qColor  = {"items":getSelectedFilters('.color-selector')};
         qAbsatz = {"items":getSelectedFilters('.absatz-selector')};
         qCat    = {"items":getSelectedFilters('.cat-selector')};
         qSize   = {"items":getSelectedFilters('.size-selector')};
         usort    = {"items":getSelectedFilters('.sort-selector')};
         xTrade =            $('input[name=xtrade]').prop('checked') ? 1 : 0;
         qLocal       =   $("#qLocal" ).val();
         idCatSearch     =   parseInt($("#justCatSearch" ).val());
         isConnectCat    =   parseInt($("#isConnectCat" ).val());
         rootUri         =   $("#rootUri" ).val();
 
         initCall = typeof initCall !== 'undefined' ? initCall : 'true';
         bSearch = typeof bSearch !== 'undefined' ? bSearch : '';
         q = trim(document.fSearchForm.qSearch.value);
         qJustCat    =   (idCatSearch > 0 && q == "") ? '&justCat=' + idCatSearch + '&isConnectCat=' + isConnectCat + '&rootUri=' + rootUri : '';
         if(q != '' || (q == "" && idCatSearch > 0))
         {
             if(start == "newSearch")
             {}
             else{
                 $.ajax({
                     type: 'post',
                     url: '/ajax.handler.php?axmodule=searchSolr&q='+q+qJustCat+'&start='+start+'&bSearch='+bSearch,
                     data: {qCurrentMin:qCurrentMin,qCurrentMax:qCurrentMax,qBrand:qBrand,qSize:qSize,qColor:qColor,qAbsatz:qAbsatz,qCat:qCat,qLocal:qLocal,maxPrice:maxPrice,qSeason:qSeason,usort:usort, initCall:initCall,xTrade:xTrade},
                     async: true,

                             {

                                 {

                                     {

                                         if(obj.forceRedirect == 'true'){window.location.href = obj.redirectURI;}
                                         return false;
                                     }
                                     catch(err){}
                                     $('#solCatResults').html(html);
                                     reloadScript("/reconnect/assets/js/lib/components/components.min.js");
                                     reloadScript("/reconnect/assets/js/lib/plugins/plugins.min.js");
                                 }
                                 else
                                 {
                                     $('#page-header').html('');
                                     $('#mainContent').html(html);
                                 }
                                 initSearchPagination();
                                 if(start != "init"){
                                     _gaq.push(['_trackEvent', 'Pagination', 'search','Seite ' + (parseInt(start)+1)]);
                                     _gaq.push(['_trackPageview']);
                                 }
                                 document.getElementById("topDiv").scrollIntoView();
                             }
                         }
                     }
                 });
                 return false;
             }
             return true;
         }
         else{return false;}
     }

            Damenschuhe
           </a>
