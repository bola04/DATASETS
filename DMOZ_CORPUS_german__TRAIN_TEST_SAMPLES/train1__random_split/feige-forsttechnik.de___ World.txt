 }
 
 function FadeIn(){

     FadePos += FadeStep;
     document.getElementById('Popuplayer').style.filter = 'alpha(opacity=' + FadePos + ')';
     document.getElementById('Popuplayer').style.opacity = FadePos / 100;
     window.setTimeout('FadeIn()',1);
   }
 }
