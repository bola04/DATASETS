 			  , isFixed = 0
 
 			processScroll();

 			function processScroll() {
 			  var i, scrollTop = $win.scrollTop()

 				isFixed = 1
 				$nav.addClass('subnav-fixed')
 			  } else if (scrollTop <= navTop && isFixed) {
 				isFixed = 0
 				$nav.removeClass('subnav-fixed')
 			  }
 			}
