    function heute()
  {
          var Datum_string;
          var Tag, Monat, Jahr;
          Wochentagname = new Array("Sonntag","Montag","Dienstag","Mittwoch","Donnerstag","Freitag","Samstag");
          heute=new Date();
          Tag=heute.getDate();
          if (heute.getDate()<10) Tag = "0"+heute.getDate();
