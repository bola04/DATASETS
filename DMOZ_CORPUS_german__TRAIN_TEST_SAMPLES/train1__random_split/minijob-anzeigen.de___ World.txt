       <script>
        function setErrorState(error) {
 	if (error === true) {
 		$( "input#where" ).addClass("error");
 	} else {
 		$( "input#where" ).removeClass("error");
 	}
 }
 
 function customItemRenderer(ul, item) {
     return $("<li>")
     .attr( "data-value", item.value )
     .appendTo( ul )
     .append($("<div class=\"place\">"+item.label+"</div>"), $("<div class=\"state\">"+item.state+"</div>"))
 
 }
 
 $( function() {
     var cache = {};
 	$( "input#where" ).autocomplete({
 	    source: function (request, response) {
 		    var term = request.term;
 		    if (term in cache) {
 			    response(cache[term]);
 			    return;
 		    }
 		    $.ajax({
 			    url: '/ajax/geolocate.php',
 			    data: { "term" : request.term},
 			    dataType: 'json',
 			    success: function(data){
 				    var result = $.map(data, function(item){
 					    value = item.plz + ' ' + item.ort;
 					    label = value;
 					    state = item.bundesland;
 					    return {label: label, value: value, state: state}
 				    });
 				    cache[term] = result;
 				    response(result);
 			    }
 		    });
 	    },
 	    response: function( event, ui ) {
 		    setErrorState((ui.content.length === 0));
 	    },
 	    change : function( event, ui ) {
 	    	if (ui.content) {setErrorState((ui.content.length===0) || (ui.content.length >= 0));}
 	    },
 	    open: function(event, ui) { }, // avoid width bug
     	minLength: 1,
     	autoFocus: true
 	}).data("ui-autocomplete")._renderItem = customItemRenderer;
 
 	$("input#where").on('copy paste cut', function() {
 	    if (!this.value) { setErrorState(false); }
 	});
 } );
       </script>
