   <script>
    function openGallery(caller, className) {
 			var pswpElement = document.querySelectorAll('.pswp')[0];
 			var items = getItemsFromDOM();
 			var options = {
 				index: parseInt(caller.getAttribute("index")),
 				getThumbBoundsFn: function (index) {
 					var pageYScroll = window.pageYOffset || document.documentElement.scrollTop;
 					var rect = caller.getBoundingClientRect();
 
 					return { x: rect.left, y: rect.top + pageYScroll, w: rect.width };
 				},
 				shareButtons: [
 					{ id: 'download', label: 'Bild herunterladen', url: '{{raw_image_url}}', download: true }
 				]
 			};
 
 			var gallery = new PhotoSwipe(pswpElement, PhotoSwipeUI_Default, items, options);
 			gallery.init();
 
 			function getItemsFromDOM() {
 				var items = [];
 				var elements = document.getElementsByClassName(className);
 				for (i = 0; i < elements.length; i++) {
 					items.push({
 						src: elements[i].getAttribute("src"),
 						w: elements[i].getAttribute("galleryWidth"),
 						h: elements[i].getAttribute("galleryHeight")
 					});
 				}
 				return items;
 			}
 		}
   </script>
