 function setBrowser() {
 	//1 - Default bzw. IE; 3 - Firefox; 2 - Safari; 4 - Opera; 5 - Google Chrome
 	var sBrowser = window.navigator.userAgent.toLowerCase();
 
 	if ( sBrowser.indexOf("opera") >= 0 ) {
 		nBrowser = 4;
 	}
 	else if ( sBrowser.indexOf("chrome") >= 0 ) {
 		nBrowser = 5;
 	}
 	else if ( sBrowser.indexOf("safari") >= 0 ) {
 		nBrowser = 2;
 	}
 	else if ( sBrowser.indexOf("firefox") >= 0 ) {
 		nBrowser = 3;
 	}
 	var b_version = navigator.appVersion;
 	sBrowsVers = parseFloat(b_version);
 }

 function Fensterweite() {
 	if (window.innerWidth) {
 		return window.innerWidth;
 	} else if (document.body && document.body.offsetWidth) {
 		return document.body.offsetWidth;
 	} else if (document.body && document.body.clientWidth) {
 		return document.body.clientWidth;
 	} else if (document.documentElement.clientWidth) {
 		return document.documentElement.clientWidth;
 	} else {
 		return 999;
 	}
 }
 
 function Fensterhoehe() {
 	if (window.innerHeight) {
 		return window.innerHeight;
 	} else if (document.body && document.body.offsetHeight) {
 		return document.body.offsetHeight;
 	} else if (document.body && document.body.clientHeight) {
 		return document.body.clientHeight;
 	} else if (document.documentElement.clientHeight) {
 		return document.documentElement.clientHeight;
 	} else {
 		return 481;
 	}
 }

 function setTab(id, n, sLink) {
 	if(id != selected) {
 		document.getElementById("logo").style.backgroundImage = "url('images/layout/" + aLogo[n] + "_logo.png')";
 		if(selected != "") {
 			document.getElementById(selected).className = "";
 			var seldiv = selected.split("_");
 			document.getElementById(seldiv[1]).style.visibility = "hidden";
 		}
 		var newdiv = id.split("_");
 		selected = id;
 		document.getElementById(newdiv[1]).style.visibility = "visible";
 
 		document.getElementById(id).className = "active";
 
 		var sFrameArrName = "";
 		var sFrameCurName = "";
 		if(selframe < 3 && aActive[selframe] == 1) {
 			//this.frames[selframe].setFocus(false);
 			sFrameArrName = aFrames[selframe];
 			for (var i=0;i<this.frames.length;i++) {
 				if (this.frames[i].name != null) {
 					sFrameCurName = this.frames[i].name;
 					if (sFrameArrName.indexOf(sFrameCurName) >= 0) {
 						this.frames[i].setFocus(false);
 						i = this.frames.length;
 					}
 				}
 			}
 		}
 		if(aActive[n] == 1) {
 			//this.frames[n].setFocus(true);
 			sFrameArrName = aFrames[n];
 			for (var j=0;j<this.frames.length;j++) {
 				if (this.frames[j].name != null) {
 					sFrameCurName = this.frames[j].name;
 					if (sFrameArrName.indexOf(sFrameCurName) >= 0) {
 						this.frames[j].setFocus(true);
 						j = this.frames.length;
 					}
 				}
 			}
 		}
 
 		if(aActive[n] == 0) {
 			aActive[n] = 1;
 			if(n==2) {
 				if ((Fensterweite() > 1024) && (Fensterhoehe() > 768)) {
 					document.getElementById(aFrames[n]).src = aSrc[n] + "sites/start_galerie.asp?gallery=13&stufe=2";
 					ngalstufe = 2;
 				}
 				else {
 					document.getElementById(aFrames[n]).src = aSrc[n] + "sites/start_galerie.asp?gallery=13&stufe=1";
 					ngalstufe = 1;
 				}
 			}
 			else if (n==0) {
 				if (sLink.length > 0)
 					document.getElementById(aFrames[n]).src = aSrc[n] + "?" + sLink;
 				else if (sFormQuery.length > 0)
 					document.getElementById(aFrames[n]).src = aSrc[n] + "?" + sFormQuery;
 				else
 					document.getElementById(aFrames[n]).src = aSrc[n];
 			}
 			else
 				document.getElementById(aFrames[n]).src = aSrc[n];
 		}
 		selframe = n;
 		setSrollbar();
 		//setFrameSize();
 	}
 }
