   function switchSearch(state)
 {
 if (state) {document.getElementById("searchButton").className = 'searchButtonCollapse'; document.getElementById("search").style.visibility="visible";}
 else {document.getElementById("searchButton").className = 'suchen'; document.getElementById("search").style.visibility="hidden";}
 }
 
 function pingpong()
 {
 if(document.getElementById("search").style.visibility == "visible") switchSearch(false); else switchSearch(true);
 }
 
 searchArea = false;
 AVisible = true;
 newEnter = true;
 last= 0;
 timer = new Array();minOpacity = 0; maxOpacity = 100; timeStepIn = 1; timeStepOut = 1; fadeStepIn = 5; fadeStepOut = 5;
 function setOpacity(oID,value) {oID.style.opacity = (value/100); oID.style.filter = 'alpha(opacity=' + value + ')';}
 function fadeIn(oID){ if (timer[oID.id]) clearTimeout(timer[oID.id]); 
 if (oID.style.opacity == null || oID.style.opacity == ""){setOpacity(oID,maxOpacity);} transformOpacity(oID.id,true);}
 function fadeOut(oID){ clearTimeout(timer[oID.id]); if (oID.style.opacity == null || oID.style.opacity == ""){setOpacity(oID,maxOpacity);} 
 transformOpacity(oID.id,false);}
 function transformOpacity(ID,direct) {obID = document.getElementById(ID);
 if ( direct==true  && obID.style.opacity<(maxOpacity/100)){setOpacity(obID,obID.style.opacity*100+fadeStepIn); timer[ID] = setTimeout("transformOpacity('"+ID+"',"+direct+")",timeStepIn);}
 else if (direct==false && obID.style.opacity>(minOpacity/100)){setOpacity(obID,obID.style.opacity*100-fadeStepOut);timer[ID] = setTimeout("transformOpacity('"+ID+"',"+direct+")",timeStepOut);}	}
 
 
 
 function get(ID){
 
 
 return document.getElementById(ID);}
