     // ----- Input-Element mit Select-Funktion -----

     var focusX = 0;     
     
     // Strom oder Gas
      $(".calcType").click( function() {
         var calc_type = $("input[name=calcType]:checked").val();
         if (calc_type == "electricity") {
             $(".con-gas").addClass("con-inactive");
             $(".con-electricity").removeClass("con-inactive");
         }
         else {
             $(".con-electricity").addClass("con-inactive");
             $(".con-gas").removeClass("con-inactive");
               }
     });
           
     // Toggle-Switch
       $(".con-toggle").click( function() {
 
           focusX = 1;
           $("#consumption").focus();
           focusX = 0;
 
           if (toggle == 0) {
               $(this).addClass("con-active");
               toggle = 1;
           }
           else {
               $(this).removeClass("con-active");
               toggle = 0;
           }
 
       });
 
     // Aufklappen bei Focus          
       $("#consumption").focus( function() {
             if((toggle == 0) && (focusX == 0)) {
                 $(".con-toggle").addClass("con-active");
                 toggle = 1;
             }
       });
 
     // Einklappen bei Blur          
       $("#consumption").blur( function() {
           if(toggle == 1) {
               setTimeout( function(){ $(".con-toggle").removeClass("con-active"); }, 100 );
               setTimeout( function(){ toggle = 0 }, 400 );
           }              
       });
 
     // Auswahl bei Mousedown, Click       
         $(".con-select").mousedown( function() {
             var con_select = $(this).attr("data-value");
             $("#consumption").val(con_select); 
             toggle = 0;
       });
         $(".con-select").click( function() {
             focusX = 1;
             $("#consumption").focus();
             focusX = 0;
             $(".con-select").removeClass("con-select-active");
             $(this).addClass("con-select-active");
             $(".con-toggle").removeClass("con-active");
             toggle = 0;       
       });
 
     // Deselect bei Keydown         
         $("#consumption").keydown( function() {
             $(".con-select").removeClass("con-select-active");
       });

         error = true;
       }
