     $(document).ready(function () {
         'use strict';
 
         // Adjust Height
         $(".modal-wide").on("show.bs.modal", function() {
             var height = $(window).height() - 200;
             $(this).find(".modal-body").css("max-height", height);
         });
 
         /**
          * Function to Handle Answers
          */
 
         // Hide all
         hideAllItems();
 
         // Display first Item
         displayItem(0);
 
         // Mark all Next Buttons to inactiv
         $('.next').attr('disabled','disabled');
 
         // Hide third button next
         $('.button-id-2').css('display','none');
 
         // click to set Answer
         $('.set-answer').click(function () {
 
             // Set all inactive
             setAllInactive();
 
             // set Current to Active
             $(this).addClass('active');
 
             setSelectedAnswer($(this).data("associateid"), $(this).data("questionid"));
             $('.button-id-' + $(this).data("currentid")).removeAttr('disabled');
         });
 
         // click to next
         $('.next').click(function () {
             hideAllItems();
             displayItem($(this).data("nextid"));
 
         });
 
         /**
          * hide All Items
          */
         function hideAllItems() {
             $(".question-item").css('display', 'none');
         }
 
         /**
          * display Item
          * @param itemNumber
          */
         function displayItem(itemNumber) {
             $(".itemkey-" + itemNumber).css('display', 'block');
         }
 
 
         function setAllInactive() {
             $('.set-answer').removeClass('active');
         }
 
         /**
          * set Selected Answer
          * @param answerid
          */
         function setSelectedAnswer(answerid, questionid) {
 
             var content = {
                 'action': 'setSelectedAnswer',
                 'answerid': answerid,
                 'questionid': questionid,
             };
 
             $.ajax(window.location.href, {
                 type: 'POST',
                 contentType: 'application/json',
                 dataType: 'json',
                 data: JSON.stringify(content),
                 beforeSend: function() {
                 },
                 success: function(response) {
 
                     $('.result-list-content').empty();
                     $('.result-list-content').append(response.output);
 
                 }
             }).done(function(response) {
             });
 
 
 
         };
 
     });
