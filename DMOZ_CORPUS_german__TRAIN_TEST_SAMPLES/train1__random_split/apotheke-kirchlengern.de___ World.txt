 			var typ	= "#" +$(this).attr('id').split('_')[1];
 			
 			$(".openclose").each(function() {
 			
 				var inneritem = "#" +$(this).attr('id');
 				var outeritem = "#mobil_" +$(this).attr('id');
 				
 				if(inneritem != typ){
 				
 					if( $(inneritem).hasClass( "open" )){
 						
 						$(inneritem).removeClass( "open" ).animate({ height: 0 }, 1000 , 'easeInOutCirc');
 						$(outeritem).removeClass( "openmenu" );
 					  
 					}
 				
 				}else{
 
 					if( $(typ).hasClass( "open" )){
 					
 						$(typ).removeClass( "open" ).animate({ height: 0 }, 1000 , 'easeInOutCirc');
 						$(item).removeClass( "openmenu" );
 					  
 					}else{
 					
 						$(typ).css('height', 'auto');
 						autoHeight = $(typ).height();
 						$(typ).css('height', '0');
 						$(typ).addClass('open').animate({ height: autoHeight }, 1000 , 'easeInOutCirc');
 						$(item).addClass( "openmenu" );
 						
 					}
 				}
 					
 			});
 			
 		});
 		
 		$(".akkordeon").click(function() { //akkordeonmenü
 			
 			var akkbtnact 	= "#" +$(this).attr('id');
 			var akkdivact 	= "#" +$(this).attr('id').split('_')[1];
 			
 			$(".akkdiv").each(function() {
 				
 				var akkbtn 	= "#akk_" +$(this).attr('id');
 				var akkdiv 	= "#" +$(this).attr('id');
 				
 				if(akkbtnact != akkbtn){
 					
 					if( $(akkbtn).hasClass( "akkopen" )){
 						
 						$(akkbtn).removeClass( "akkopen" );
 						$(akkdiv).animate({ height: 0 }, 1000 , 'easeInOutCirc');
 						
 					};
 					
 				}else{
 					
 					if( $(akkbtn).hasClass( "akkopen" )){
 						
 						$(akkbtn).removeClass( "akkopen" );
 						$(akkdiv).animate({ height: 0 }, 1000 , 'easeInOutCirc');
 						
 					}else{
 						
 						$(akkbtn).addClass( "akkopen" );
 						$(akkdiv).css('height', 'auto');
 						akkHeight = $(akkdiv).height();
 						$(akkdiv).css('height', '0');
 						$(akkdiv).animate({ height: akkHeight }, 1000 , 'easeInOutCirc');
 						
 					};
 					
 				}
 				
 			});
 			
 		});
