                         urlMatch = $(this).data('style').match(/url\('?([^'\)]+)'?\)/);
                         
                         // preload images -> backgrounds for hidden elements are not going to be loaded
                         if(typeof urlMatch[1] !== 'undefined')
                         {
                             preloadImage(urlMatch[1], loadedCallback);
                         }
                         
                         typeof callback === 'function' && callback($(this));
                     });
                 },
                 getControls = function(slider){
                     var nivoVars = slider.data('nivo:vars');
                     
                     return slider.find('.nivo-directionNav').add(typeof nivoVars !== 'undefined' ? nivoVars.controlNavEl : slider.find('.nivo-prevNav, .nivo-nextNav'));
                 },
                 loadImages = function(){
                     if (!loaded){
                         loaded = 1;
                        
                         // obtain remaining images
                         $('.nivoSlider').each(function(){
                             var _this = $(this), controls = getControls(_this), hasImg = _this.find('.nivo-imageLink img').length;
                             !_this.data('img_cached') && controls.hide();
                         
                             loadImage($(this), null, function() {
                                 !hasImg && controls.show();
                                 portunaHelper.cookie.set('slider_cached', 1, 60);
                             });
                         });
                     }
                 },
                 initNivoSlider = function(){
                     var w = $(window), prevTop = w.scrollTop();
                     
                     // retrigger scrolling due to potentially missing height of banner-picture (.trigger doesn't work)
                     w.scrollTop(prevTop + 1).scrollTop(prevTop);
                     
                     $('.nivoSlider').each(function(){
                         var _this = $(this);
                         
                         if (!_this.data('img_cached'))
                         {
                             var controls    = getControls(_this),
                                 images      = _this.find('.nivo-imageLink img'), 
                                 imageLength = images.length, 
                                 loaded      = 0;
                             
                             if (imageLength)
                             {
                                 controls.hide();
                                 
                                 var handle = function() {
                                     loaded++;
                                     
                                     if (loaded === imageLength)
                                     {
                                         controls.show();
                                     }
                                 };
                                 
                                 images.load(handle).error(handle);
                                 
                                 // retrigger load of already provided image-sources
                                 images.each(function(){
                                     var img = $(this), src = img.attr('src');
                                     
                                     typeof src !== 'undefined' && img.attr('src', src);
                                 });
                             }
                         }
                     });
                 }
                 ;
                 
                 window.setTimeout(loadImages, 9000);$('#slider').nivoSliderBg({effect: "fade", animSpeed: 1000, slices: 15, boxCols: 8, boxRows: 4, pauseTime: 10000, startSlide: 0, directionNav: true, controlNav: false, controlNavThumbs: false, pauseOnHover: true, manualAdvance: false, prevText: "Zurück", nextText: "Weiter", beforeChange: loadImages, afterLoad: initNivoSlider, randomStart: false});$(".nivo-prevNav").attr( "title", "Zurück").end();$(".nivo-nextNav").attr( "title", "Weiter");});
      </script>
