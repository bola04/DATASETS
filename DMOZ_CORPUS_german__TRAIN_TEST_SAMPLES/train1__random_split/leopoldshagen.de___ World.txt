 function createCookie(name, value, days) {
     if (days) {
         var date = new Date();
         date.setTime(date.getTime() + (days * 60 * 1000));
         var expires = '; expires=' + date.toGMTString();
     } else {
         var expires = '';
     }
     
     document.cookie = name + '=' + value + expires + '; path=/';
 }
 
 function readCookie(name) {
     var nameEQ = name + '=';
     var ca = document.cookie.split(';');
     for (var i = 0; i < ca.length; i++) {
         var c = ca[i];
         while (c.charAt(0) == ' ') c = c.substring(1, c.length);
         if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
     }
     return null;
 }

         <h1>
          Herzlich willkommen
         </h1>
         <p class="tiny_p">
          Der
          <strong>
           Ferienort Leopoldshagen
          </strong>
          liegt am Stettiner Haff in Mecklenburg – Vorpommern.
         </p>
         <p class="tiny_p">
         </p>
         <p class="tiny_p">
          Durch Wald- und Wassernähe ist die Lage besonders günstig. Vom Radwanderweg bis hin zum Badestrand wird die Gegend in Vorpommern zu einem beliebten Ferien- und Erholungsgebiet. Der Ort ist 2 km lang und ca. 500 m breit. Im Norden des Ortes liegt in etwa 3 km Entfernung das Stettiner Haff, wo man bei guter Sicht die Insel Usedom sehen kann. Das Stettiner Haff, mit 903 Quadratkilometern der größte Bodden der Ostsee (der Bodensee hat zum Vergleich eine Fläche von nur 538 Quadratkilometern) ist gegen die rauhe Ostsee durch die beiden Inseln Usedom und Wollin abgeschirmt.
         </p>
         <p class="tiny_p">
         </p>
         <p class="tiny_p">
          <img alt="Ortsschild Leopoldshagen" data-magnifier="false" data-scaleable="https://fotos.verwaltungsportal.de/seitengenerator/gross/17d56706a9ed2a063c832924e16c13d0_dsc00879.jpg" height="233" src="https://fotos.verwaltungsportal.de/seitengenerator/gross/17d56706a9ed2a063c832924e16c13d0_dsc00879.jpg" style="float: right; margin: 5px;" title="Ortsschild Leopoldshagen" width="311"/>
          Es bietet Sportschiffern ein ideales Betätigungsfeld, Fischern gute Fänge, Badegästen herrliche Strände, Touristen angenehme Ausflüge auf Fahrgastschiffen und Naturliebhabern schilfbewachsene Uferstreifen voller Wasservögel. Ein 260 Quadratkilometer großes Landschaftsschutzgebiet vereint die Küstenregion mit dem großen Waldgebiet südlich von Leopoldshagen, der Anklamer Heide. Auf mehreren gut ausgeschilderten Rad- und Wanderwegen (ca. 100 km Gesamtlänge), lässt sich die Natur hier pur erleben. Es gibt auch Schutzhütten. Der Radwanderweg führt westlich in Richtung Bugewitz – Rosenhagen – Kamp, wo noch die alte Eisenbahnbrücke zu sehen ist, die bis April des Jahres 1945 die Zugverbindung  Berlin – Swinemünde herstellte.
         </p>
         <p class="tiny_p">
         </p>
         <p class="tiny_p">
          Östlich von Leopoldshagen führt der Radwanderweg nach Mönkebude (ca. 4 km), wo sich ein sehr schöner Badestrand befindet. Dort werden auch Bootsfahrten angeboten. Östlich von Mönkebude geht es dann weiter in Richtung Grambin, wo man auch einen sehr schönen Badestrand findet. Der nächste Ort ist die kleine Haffstadt Ueckermünde, die auch über Strand, Schiffsfahrten, Tierpark, Museum u.s.w. verfügt. Zu den Sehenswürdigkeiten gehören u.a. die Heimatstube und die 250 Jahre alte   Leopoldshagener Fachwerkkirche. (Offene Kirche) Die breite Dorfstraße mit der schönen Lindenallee lädt zu herrlichen Spaziergängen ein. Feriengäste können folgende Freizeitangebote nutzen: Schiffsfahrten, Segelsport, Wassersport (Surfen), Angelsport, Pferdesport (Reiten), Rad- und Wandertouren durch Wiesen, Wald und Flur, Besuche von Konzerten, Besichtigung der Heimatstube und der sehr schönen alten Fachwerkkirche (jeden Mittwoch mit Führung).
         </p>
         <p class="tiny_p">
         </p>
         <p class="tiny_p">
          Genießen Sie einfach die ruhige Lage in der Gegend am schönen Stettiner Haff!
         </p>
         <p class="tiny_p">
         </p>
         <p class="tiny_p">
          Erholen Sie sich in MV und schöpfen Sie neue Kraft für die Zeit nach Ihrem Urlaub!
         </p>
