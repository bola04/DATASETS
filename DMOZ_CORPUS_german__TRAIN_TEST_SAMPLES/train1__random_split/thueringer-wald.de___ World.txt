             function HeadBanner() {

 			var floatbanner = jQuery('.floatbanner');
 
 			var headbannerposition = headbanner.position();
 			var headbannerout = headbannerposition.top+topadder;

 			var rightposition = right.position();
 			var rightout = rightposition.top+right.height()+topadder;
 
 
 			// Oben / Seite
 			if(jQuery(window).scrollTop() > headbannerout && jQuery(window).scrollTop() < rightout) {
