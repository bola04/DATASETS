 // Extrahiert aus einer Zeichenkette den Teil from bis to (Zeichenketten, exklusiv).
 // Legt die Position to in die globale Variable extract_offset ab, so, dass diese uebergeben werden kann um das Suchen von vorne zu verhindern (inhaltlich oder aus Performancegruenden).
 var cl = cl || {};
 cl.extract_offset = 0;
 cl.extract = function (in_, from, to, case_sensitive, offset)
  {
   if (typeof to == 'undefined') to = '';
   if (typeof case_sensitive == 'undefined') case_sensitive = true; // wird derzeit ignoriert
   if (typeof offset == 'undefined') offset = 0;
 
   //console.log (in_, from, offset);
   if (offset)
    {
     in_ = in_.substr (offset);
    }
 
   var pos_start = in_.indexOf (from);
   //console.log (pos_start);
   if (pos_start == -1) //;; Anfang nicht gefunden: nichts zurueck geben
    {
     return '';
    }
   pos_start = pos_start + from.length;
   var len = 0;
   var out;
   if (! to)
    {
     //;;Debug(A_LineNumber, in, from, pos_start)
     out = in_.substr (pos_start);
     len = pos_start + out.length;
     cl.extract_offset = offset + pos_start + len;
    }
   else
    {
     //console.log (in_, pos_start, to, in_.substr (pos_start).indexOf (to));
     //len = in_.substr (pos_start).indexOf (to) - pos_start;
     len = in_.substr (pos_start).indexOf (to);
     //console.log (len);
     if (len < 0) //;; Ende nicht gefunden: nichts zurueck geben
      {
       return '';
      }
     //;;Debug(A_LineNumber, in, from, to, pos_end, len)
     out = in_.substr (pos_start, len)
     cl.extract_offset = offset + pos_start + len + to.length;
    }
   return out;
  };

 cl.benno.mailingwork_parse_mw_response = function (html)
  {
   // parst alle <meta-Tags>
   var metas = [];
   cl.extract_offset = 0;
   for (var i in html.split ('<meta'))
    {
     var found = cl.extract (html, '<meta', '>', true, cl.extract_offset);
     if (! found) break;
     metas.push (found);
    }
   //console.log (metas);
 
   // <meta irgendwas="abc" content="def"> -> ignoriere
   // <meta mw-status="success" content="Vielen Dank für Ihre Anmeldung."> -> {'mw-status': 'success', 'content': 'Vielen Dank für Ihre Anmeldung.'}
   // <meta mw-status="error" content="Vielen Dank für Ihre Anmeldung. Leider..."> -> {'mw-status': 'error', 'content': 'Vielen Dank für Ihre Anmeldung. Leider...'}
   for (var i in metas)
    {
     var meta = metas[i];
 
     if (! meta.indexOf) continue; // Schleife bringt mehr als sie soll
 
     if (meta.indexOf ('mw-status') >= 0)
      {
       metas[i] =
        {
         'mw-status': cl.extract (meta, 'mw-status="', '"'),
         'content': cl.extract (meta, 'content="', '"'),
        }
      }
    }
   //console.log (metas);
   // {'mw-status': 'success', 'content': 'Vielen Dank für Ihre Anmeldung.'}
   // {'mw-status': 'error', 'content': 'Vielen Dank für Ihre Anmeldung. Leider...'}
   /// -> {'success': 'Vielen Dank für Ihre Anmeldung.', 'error': 'Vielen Dank für Ihre Anmeldung. Leider...'}
   var mw_response = {};
   for (var i in metas)
    {
     var meta = metas[i];
 
     if (meta['mw-status'])
      {
       mw_response[meta['mw-status']] = meta['content'];
      }
    }
   //console.log (mw_response);
 
   //mw-status
   //
   //console.log (cl.extract (html, '<meta', '>'));
   ////console.log (cl.extract (cl.extract (meta, '<', '>'), 'content=');
   return mw_response;
  }

       jQuery (mailingworkform).submit ();

       setTimeout (cl.benno.submitRegisterformAndMailingworkform, 1000, event, fn_submit);
