 	function insertParam(url, parameterName, parameterValue, atStart){
 	    replaceDuplicates = true;
 	    if(url.indexOf('#') > 0){
 	        var cl = url.indexOf('#');
 	        urlhash = url.substring(url.indexOf('#'),url.length);
 	    } else {
 	        urlhash = '';
 	        cl = url.length;
 	    }
 	    sourceUrl = url.substring(0,cl);
 
 	    var urlParts = sourceUrl.split("?");
 	    var newQueryString = "";
 
 	    if (urlParts.length > 1)
 	    {
 	        var parameters = urlParts[1].split("&");
 	        for (var i=0; (i < parameters.length); i++)
 	        {
 	            var parameterParts = parameters[i].split("=");
 	            if (!(replaceDuplicates && parameterParts[0] == parameterName))
 	            {
 	                if (newQueryString == "")
 	                    newQueryString = "?";
 	                else
 	                    newQueryString += "&";
 	                newQueryString += parameterParts[0] + "=" + (parameterParts[1]?parameterParts[1]:'');

 		function getFrameID(id){
 		    var elem = document.getElementById(id);
 		    if (elem) {
 		        if(/^iframe$/i.test(elem.tagName)) return id; //Frame, OK
 		        // else: Look for frame
 		        var elems = elem.getElementsByTagName("iframe");
 		        if (!elems.length) return null; //No iframe found, FAILURE
 		        for (var i=0; i<elems.length; i++) {
 		           if (/^https?:\/\/(?:www\.)?youtube(?:-nocookie)?\.com(\/|$)/i.test(elems[i].src)) break;
 		        }
 		        elem = elems[i]; //The only, or the best iFrame
 		        if (elem.id) return elem.id; //Existing ID, return it
 		        // else: Create a new ID
 		        do { //Keep postfixing `-frame` until the ID is unique

 		        } while (document.getElementById(id));

             in diesem Sinne möchten wir uns in unserer Praxis für Innere Medizin, Gastroenterologie und Kardiologie Zeit nehmen, uns auf Sie einzustellen, Ihre Anliegen und Wünsche zu verstehen und Ihnen schließlich ein optimales Therapiekonzept vorzuschlagen. Wir widmen uns vor allem der Gesundheit der Verdauungsorgane, der Leber und des Herzens. Ein weiterer Schwerpunkt ist die Behandlung von Bluthochdruck und Herzrhythmusstörungen. Dabei orientieren wir uns an den Behandlungsleitlinien renommierter Fachgesellschaften und bilden uns regelmäßig fort. Unsere hochmodernen Geräte ermöglichen eine präzise und schonende Diagnostik. Neben endoskopischen Untersuchungen des Magen-Darm-Traktes sind wir auf die Ultraschalldiagnostik der Bauchorgane und des Herzens spezialisiert. Unsere Mitarbeiterinnen und Mitarbeiter sind stets bemüht, Ihren Besuch bei uns so angenehm wie möglich zu gestalten. Einige von ihnen besitzen spezielle Zertifikate für die Betreuung von Patienten mit chronischen Darmerkrankungen und mit Leberentzündungen. Wir alle möchten Sie so sorgfältig und einfühlsam wie möglich dabei unterstützen, Ihre Gesundheit solange es geht zu erhalten.Sollte bereits eine Erkrankung bestehen, ist es unser Bestreben, mit Ihnen gemeinsam ein Behandlungskonzept zu entwickeln, damit Sie Ihr Leben ungestört, elanvoll und glücklich verbringen können. Wir würden uns sehr freuen, Sie bei uns persönlich kennenzulernen.
            </p>
            <p>
             Dr. med. Corinna Lachmann und Dr. med. Kai Rösler
