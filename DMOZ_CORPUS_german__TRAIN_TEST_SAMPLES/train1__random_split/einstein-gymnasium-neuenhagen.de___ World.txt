 						function createCookie(name,value,days) {
 							if (days) {
 								var date = new Date();
 								date.setTime(date.getTime()+(days*60*1000));
 								var expires = '; expires='+date.toGMTString();
 							} else var expires = '';
 							document.cookie = name+'='+value+expires+'; path=/';
 						}
 
 						function readCookie(name) {
 							var nameEQ = name + '=';
 							var ca = document.cookie.split(';');
 							for(var i=0;i < ca.length;i++) {
 								var c = ca[i];
 								while (c.charAt(0)==' ') c = c.substring(1,c.length);
 								if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
 							}
 							return null;
 						}

          <p class="tiny_p" style="text-align:center">
           <span style="font-size:36px">
            <strong>
             <span style="color:#0000ff">
              Landeswettbewerb Jugend forscht 2018
             </span>
            </strong>
           </span>
          </p>
          <p class="CxSpFirst tiny_p">
           <span style="font-size:18px">
            Vom Einstein-Gymnasium sind 4 Projekte aus den Bereichen Chemie und Technik gestartet.
           </span>
          </p>
          <p class="CxSpMiddle tiny_p">
          </p>
          <p class="CxSpMiddle tiny_p">
           <span style="font-size:18px">
            <strong>
             Bereich: Chemie
            </strong>
           </span>
          </p>
          <p class="CxSpMiddle tiny_p">
           <img alt="arneniklasgino" height="300" src="https://fotos.verwaltungsportal.de/seitengenerator/gross/arneginoniclas.png" title="arneniklasgino" width="450"/>

          <p class="CxSpMiddle tiny_p">
           <span style="font-size:18px">
            <em>
             Projekt 1:
            </em>
           </span>
          </p>
          <p class="CxSpMiddle tiny_p">
           <span style="font-size:18px">
            Mobiles chemisches Coolpack mit Kaliumiodid
           </span>
          </p>
          <p class="CxSpMiddle tiny_p">
           "Nach unserer Teilnahme am Wettbewerb "Schüler experimentieren" im Jahr 2017, für den wir in der Chemie-AG ein Brausekühlgerät entwickelt haben, haben wir nach weiteren Kühlmitteln gesucht und mit dem Stoff Kaliumiodid unsere ersten Versuche erfolgreich abgeschlossen. Dabei haben wir bemerkt, dass sich dieser Stoff sehr gut zur Herstellung von regenerierbaren Coolpacks eignet. Unser Ziel ist es, den Kühleffekt weiter zu untersuchen und ein solches Coolpack zu bauen."
          </p>
