    $('.flagitem').hover(
                 function(){
                     var $this = $(this);
                     expand($this);
                 },
                 function(){
                     var $this = $(this);
                     collapse($this);
                 }
             );     
               
             function expand($elem){                
 
                 var $widthTest = $elem.find('.flagitem_content'); 
                 var innerH = $widthTest.html();
                 var lengthX = innerH.length;
                 
                 if(lengthX > 15){
                    lengthX = 15;
                 }
                 var pixelPerChar = 8.9;
                 
                 $widthTest =  lengthX * pixelPerChar + 90;
                 $elem.animate({width: parseInt($widthTest)}, 400);
                 
             } 
                
             
             function collapse($elem){
                 $elem.stop().animate({width:'65px'}, 400);
             }
