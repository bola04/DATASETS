 		function decryptCharcode(n,start,end,offset) {
 			n = n + offset;
 			if (offset > 0 && n > end) {
 				n = start + (n - end - 1);
 			} else if (offset < 0 && n < start) {
 				n = end - (start - n - 1);
 			}
 			return String.fromCharCode(n);
 		}
 			// decrypt string
 		function decryptString(enc,offset) {
 			var dec = "";
 			var len = enc.length;
 			for(var i=0; i < len; i++) {
 				var n = enc.charCodeAt(i);
 				if (n >= 0x2B && n <= 0x3A) {
 					dec += decryptCharcode(n,0x2B,0x3A,offset);	// 0-9 . , - + / :

 				} else {
 					dec += enc.charAt(i);
 				}
 			}
 			return dec;
 		}
