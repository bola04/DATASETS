   <script type="text/javascript">
    var flyoverName = "flyover";
 
       var flyoverId;
 
       function posX(obj)
       {
           var curleft = 0;
           if(obj.offsetParent)
               while(1)
           {
               curleft += obj.offsetLeft;
               if(!obj.offsetParent)
                   break;
               obj = obj.offsetParent;
           }
           else if(obj.x)
               curleft += obj.x;
           return curleft;
       }
 
       function posY(obj)
       {
           var curtop = 0;
           if(obj.offsetParent)
               while(1)
           {
               curtop += obj.offsetTop;
               if(!obj.offsetParent)
                   break;
               obj = obj.offsetParent;
           }
           else if(obj.y)
               curtop += obj.y;
           return curtop;
       }
 
       function showFlyover(who,dx,dy,content)
       {
           var tipO = who;
           flyoverId = document.getElementById(flyoverName);
           var x = posX(who);
           var y = posY(who);
           flyoverId.style.left = String(parseInt(x + dx) + 'px');
           flyoverId.style.top = String(parseInt(y + dy) + 'px');
           flyoverId.innerHTML = content;
           flyoverId.style.display = "block";
           tipO.onmouseout = hideFlyover;
       } 
 
       function hideFlyover() { flyoverId.style.display = "none"; }
   </script>
