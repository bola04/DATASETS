     <script>
      var objTabs;
 
 var TabElement = function() {
 	this.activePanel = 0;
 	this.nextPanel = 0;
 	this.inProgress = false;
 	
 	this.constructor = function () {
 		$( window ).resize(function() {
 			objTabs.onResize();
 		});
 		$.each($('.tab-control'), function (key,val) {
 			$(val).click(function(e){
 				e.stopPropagation();
 				if(!objTabs.inProgress){
 					objTabs.inProgress = true;
 					objTabs.nextPanel = $(this).data('tabnr');
 					objTabs.activatePanel();
 				}
 			});
 		});
 		$.each($('.panelheader'), function (key,val) {
 			$(val).click(function(e){
 				e.stopPropagation();
 				if(!objTabs.inProgress){
 					objTabs.inProgress = true;
 					objTabs.nextPanel = $(this).data('tabnr');
 					objTabs.activatePanel();
 				}
 			});
 		});
 		this.onResize();
 		$.each($('.tabs-panel'), function (key,val) {
 			if(key>0){
 				$(val).css('display','none');
 				$('#TabArrow'+key).removeClass('tab-arrow-up');
 				$('#TabArrow'+key).addClass('tab-arrow-down');
 			}
 		});
 	}
 	this.onResize = function() {
 		if($( window ).width() < 768) {
 			$('#ReiterStartseite').css('display','none');
 			$.each($('.panelheader'), function (key,val) {
 				$(val).css('display','block');
 			});
 		}else{
 			$('#ReiterStartseite').css('display','block');
 			$.each($('.panelheader'), function (key,val) {
 				$(val).css('display','none');
 			});
 		}
 	}
 	
 	this.activatePanel = function () {
 		
 		$.each($('.tab-control-active'),function(key,val) {
 			$(val).removeClass('tab-control-active');
 			$(val).addClass('tab-control-inactive');	
 		});
 		$('#TabCon' + this.nextPanel).removeClass('tab-control-inactive');
 		$('#TabCon' + this.nextPanel).addClass('tab-control-active');
 		$.each($('.tab-arrow'),function(key,val) {
 			$(val).removeClass('tab-arrow-up');
 			$(val).addClass('tab-arrow-down');
 		});
 		
 		if($('#ReiterStartseite').css('display')=='none'){
 			$.each($('.tab-panel-row-subheader'),function(key,val){$(val).css('position','relative')});
 		}	
 
 		$.each($('.tabs-panel'),function(key,val) {
 			if($('#ReiterStartseite').css('display')=='none'){
 				if(key==objTabs.activePanel && $(val).css('display')!='none'){
 					$('#TabArrow' + key).removeClass('tab-arrow-up');
 					$('#TabArrow' + key).addClass('tab-arrow-down');
 					$(val).slideUp({duration:300, complete: function(){
 						$(val).css('display','none');
 						$.each($('.tab-panel-row-subheader'),function(key2,val2){$(val2).css('position','absolute')});
 					}});
 				}
 				if(key==objTabs.nextPanel && $(val).css('display')!='block'){
 					$('#TabArrow' + key).removeClass('tab-arrow-down');
 					$('#TabArrow' + key).addClass('tab-arrow-up');
 					$('#TabContainer' + key).slideDown({duration:300,complete:function(){
 						$('#TabContainer' + key).css('display','block');

 					}});
 				}
 			}else{
 				if(key==objTabs.activePanel){
 
 					$('#TabArrow' + key).removeClass('tab-arrow-up');
 					$('#TabArrow' + key).addClass('tab-arrow-down');
 					$(val).css('display','none');

 				if(key==objTabs.nextPanel){
 					$('#TabArrow' + key).removeClass('tab-arrow-down');
 					$('#TabArrow' + key).addClass('tab-arrow-up');
 					$(val).css('display','block');
 				}
 			}
 		});
 
 		this.inProgress = false;
 		this.activePanel = objTabs.nextPanel;
 	}
 	
 	this.constructor();
 }
