   <script type="text/javascript">
    $(".form2mail .form-submit").click(function (e) {
         e.preventDefault();
 
 
         if ($("#message").val() == "") {
             $(".form-empty").show()
         } else {
             var $form = $(e.target).closest("form");
 
             $form.find(".form-loading").show();
             $form.find(".form-submit").hide();
 
             var data = $form.serialize();
             var target = $form.attr("action");
 
             if ($form.data("debug")) {
                 alert("Mail Text:\n" + data);
             }
 
             $.ajax({
                 type: "POST",
                 url: target,
                 data: data,
                 success: function (data) {
                     if (data.success) {
                         $form.find(".form-loading").hide();
                         $form.find(".form-success").show();
                         $form.find(".form-empty").hide();
                         $form.trigger("reset");
                     } else {
                         $form.find(".form-loading").hide();
                         $form.find(".form-empty").hide();
                         $form.find(".form-submit").show();
                         $form.find(".form-error").show();
                     }
 
                 },
                 error: function () {
                     $form.find(".form-loading").hide();
                     $form.find(".form-submit").show();
                     $form.find(".form-error").show();
                     $form.find(".form-empty").hide();
                 }
             });
         }
     });
   </script>
