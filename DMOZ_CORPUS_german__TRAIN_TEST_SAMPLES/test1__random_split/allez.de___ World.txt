     function getChildren($row) {
         var children = [];
         while($row.next().hasClass('table-row-content')) {
              children.push($row.next());
              $row = $row.next();
         }            
         return children;
     }        
     $(".table-row-list").click(function() {
 		$( this ).toggleClass( "table-row-open");
         var children = getChildren($(this));
         $.each(children, function() {
             $(this).toggleClass( "table-row-open");
         })
     });
