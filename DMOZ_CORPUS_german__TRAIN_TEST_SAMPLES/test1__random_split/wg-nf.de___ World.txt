     function is_touch_device() {
         return !!('ontouchstart' in window);

     var timer;
     var touchduration = 700; //length of time we want the user to touch before we do something
 
     function touchstart(elem) {
         if(!$(elem).parent().hasClass('dropdown-submenu')){
             clearSubMenus(elem);
         }
         if($(elem).parent().hasClass('open')) {
             $(elem).parent().removeClass('open');
             $(elem).parent().parent().find('ul').hide();
         } else {
             $(elem).parent().addClass('open');
             $(elem).parent().parent().find('ul').show();
         }
         timer = setTimeout(function(){onlongtouch(elem);}, touchduration);
     }
 
     function touchend() {
 
         if (timer)
             clearTimeout(timer);
     }
 
     function onlongtouch(elem) {
         var url = $(elem).attr('href');
         if(!url || url == '')
             return;
         window.location.replace(url);
     }

        <h1>
         WG-NF / Die Unabhängigen
        </h1>
        <p>
         Herzlich willkommen auf der Homepage der Wählergemeinschaft Nordfriesland - Die Unabhängigen -
        </p>
        <p>
         Wir freuen uns, dass Sie den Weg zu uns gefunden haben. Hier haben Sie die Möglichkeit, sich mit der Arbeit der Wählergemeinschaft Nordfriesland - Die Unabhängigen und unseren Zielen vertraut zu machen.
        </p>
        <p>
         Außerdem können Sie hier die Menschen kennenlernen, die hinter unserer Arbeit stehen - und diejenigen, die sie ausführen.
        </p>
        <p>
         Schon seit Gründung der WG-NF / Die Unabhängigen am 16.12.1989 agieren die Mitglieder frei und unabhängig von etwaigen parteipolitischen Ideologien.
        </p>
        <p>
         Hierbei spielt der Grundsatz der Unabhängigkeit die zentrale Rolle in der WG-NF. Wir verstehen unsere politische Aufgabe ausschließlich auf Orts- und Kreisebene.
        </p>
        <p>
         Dabei sind wir kein Kreisverband der örtlichen Wählergemeinschaften, sondern verstehen unsere Hauptaufgabe in den kommunalen Belangen des Kreises. Wir meinen, dass wir gerade durch diese Entscheidung den Anspruch der kommunalen Selbstverwaltung, insbesondere auf Ortsebene, stärken. Denn die kommunale Kreispolitik muss zugänglich und beeinflussbar für jeden sein.
        </p>
        <p>
         Das ist der zentrale Punkt der Demokratie!  In der WG-NF / Die Unabhängigen ist jeder herzlich willkommen, der gemeinschaftlich zum Wohle der Bürgerinnen und Bürger des Kreises Nordfriesland mit beitragen möchte.
        </p>
        <p>
         Deshalb hoffen wir, dass sich auch zukünftig viele Menschen bereit erklären, uns bei dieser ehrenamtlichen Arbeit zu unterstützen.
        </p>
        <p>
         <br/>
         Michael Lorenzen             Jürgen Jungclaus
         <br/>
         1. Vorsitzender                  Fraktionsvorsitzender
        </p>
