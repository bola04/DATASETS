 				});

 		mouseleave: function(){
 			$(this).find(".pop_content").stop().animate({

 			});

 				$(this).stop().addClass("hover").find("ul").fadeIn();
 			},
 			mouseleave: function(){
 				$(this).removeClass("hover").find("ul").hide();
 			}
 		});
 	});
 	
 	// POP CONTAINERS

 		var href = $(this).find("a[href]").attr("href");

 			$(this).find(".pop_content").bind("click", function(){
 				location.href = href;
 			})
 			.css({
 				cursor: "pointer"
 			});
 		}
 	});
 	
 	// RESIZE
