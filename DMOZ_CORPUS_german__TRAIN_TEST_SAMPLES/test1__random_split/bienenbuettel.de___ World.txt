    (function($) {
   window.addEvent('domready',function(){
     var fix = $('myNavigation'); // ID im Modul "nav_main"
     var fold = fix.getCoordinates().top;
     
     window.addEvent('scroll',function(){
       var scroll = window.getScroll();
 
       if(scroll.y > fold){
         fix.setStyles({
           position: 'fixed',
           width: '90%',
           top: -50,
           left: 130
         });
         fix.addClass('fixed');
         
       } else if(scroll.y <= fold){
         fix.setStyle('position', 'absolute');
         fix.setStyle('width', '100%');
         fix.setStyle('left', '0');
         fix.setStyle('top', '2%');
         fix.removeClass('fixed');
       }
       
       
     });
   });
 })(document.id);
