 function randommaker(range) {

 function initsnow() {

 		snow[i].style.top=snow[i].posy;
 	}
 	movesnow();
 }
 
 function movesnow() {

 			snow[i].posy=0;
 		}

 	var timer=setTimeout("movesnow()",50);
