 {        
         jQuery.ajax({
                     type : 'POST',
                     dataType : 'json',
                     url : "http://www.harburg-aktuell.de/index.php?option=com_jevents&ttoption=com_jevents&typeaheadtask=gwejson&file=fetchlatestevents&path=module&folder=mod_jevents_latest&token=060e067e9108a5cc05599c7e94f2390d",
                     data : {'json':JSON.stringify({'modid':modid, 'direction':direction})},
                     contentType: "application/x-www-form-urlencoded; charset=utf-8",
                     scriptCharset: "utf-8"
             })                        
                 .done(function( data ){                    
                     jQuery("#mod_events_latest_"+modid+"_data").replaceWith(data.html);
                     try {
                         document.getElementById("mod_events_latest_"+modid+"_data").parentNode.scrollIntoView({block: "start", behavior: "smooth"});
                     }
                     catch (e) {
                     }
                 })
                 .fail(function(x) {
         alert('fail '+x);
                 });
 }
