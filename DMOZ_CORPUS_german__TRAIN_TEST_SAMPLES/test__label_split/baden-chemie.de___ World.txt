       var captchaCode = '';
 function fillCaptchaCode() {
     if (captchaCode == '') {
 		var i = 0;
 		var j = 6;
 		var randomNumber = 0;
 		var randomMessage = '';
 		while(i<j)
 		{
 			randomNumber = (Math.floor((Math.random() * 100)) % 94) + 33;
 			  if ((randomNumber >=33) && (randomNumber <=47)) { continue; }
 			  if ((randomNumber >=58) && (randomNumber <=64)) { continue; }
 			  if ((randomNumber >=91) && (randomNumber <=96)) { continue; }
 			  if ((randomNumber >=123) && (randomNumber <=126)) { continue; }
 			randomMessage += String.fromCharCode(randomNumber);
 			i++;
 		}
         captchaCode = randomMessage;
         document.getElementById('captcha').innerHTML = captchaCode;
         checkCaptchaCode();
     }
 }
 function checkCaptchaCode()
 {
     var temp = document.getElementById('capthca_val').value;
     if (captchaCode == temp) {document.getElementById('sendbutton').removeAttribute("disabled");}
     else {document.getElementById('sendbutton').setAttribute("disabled", "disabled");}
 }
 
 function fcsArrf(obj,value){
 	if(obj.value==value){
 		obj.value='';
 	}
 }
 function blrArrf(obj,value){
 	if(obj.value==''){
 		obj.value=value;
 	}
 	fillCaptchaCode();
 }
 var mandatory_k_form=new Array();
 var inivals_k_form=new Array();
 function ValidateKForm(){
 	var errs=0; ;
 	for ( var key in inivals_k_form )	{
 		if(mandatory_k_form[key]){
 			var inputname="fld_"+key+"_";

 				if(document.getElementById(inputname).value){
 					var fieldvalue=document.getElementById(inputname).value;
 				}else{
 					var fieldvalue='';
 				}
 				
 				switch(mandatory_k_form[key])
 				{
 				case 'checkbox':
 				  if(document.getElementById("fld"+key).checked){
 					  document.getElementById(inputname).className="normalInput";
 				  }else{
 						document.getElementById(inputname).className="redInput";
 						errs += 1; 
 				  }
 				  break;
 				case 'email':
 				  	if((fieldvalue.lastIndexOf(".") > 2) && (fieldvalue.indexOf("@") > 0) && (fieldvalue.lastIndexOf(".") > (fieldvalue.indexOf("@")+1)) ){
 						document.getElementById(inputname).className="normalInput";
 					}else{
 //						document.getElementById(inputname).className="redInput";
 						document.getElementById(inputname+"x").className="form-group has-error";
 						errs += 1; 
 					}
 				  break;
 				default:
 					
 					if(!fieldvalue || (inivals_k_form[key]==fieldvalue)){	
 						document.getElementById(inputname+"x").className="form-group has-error";
 						errs += 1; 
 					}else{
 						document.getElementById(inputname+"x").className="form-group";
 					}
 				}
 				
 			}
 		}
 	}
 	return (errs==0);
