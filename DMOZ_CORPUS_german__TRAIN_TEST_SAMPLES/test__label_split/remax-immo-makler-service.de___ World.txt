     function cleanUrlAndReload() {
 
         var uri = window.location.toString();
         if (uri.indexOf("?") > 0 && (uri.indexOf("acAction") > 0 || uri.indexOf("sociallogin") > 0)) {
             var clean_uri = removeURLParameter(uri, "u");
             clean_uri = removeURLParameter(clean_uri, "acAction");
             clean_uri = removeURLParameter(clean_uri, "sociallogin");
             location.href = clean_uri;
         } else {
             window.location.reload(true);
         }
 
     }
 
     function AgentConnectGreetingRedirect(acAction, UserID) {
         
         if (acAction == "1") {
             MyProfileSignInService.instance()
                 .getAgentConnectGetConnectedScreen(UserID)
                 .done(function (html) {
                     $("#agentConnectGreeting").html(html.d);
                     $('#getConnected').on("hidden.bs.modal", function () {
                         cleanUrlAndReload();
                     }).modal('show');
                 })
                 .fail(function (xhr) {
                     HideWaitImg();
                 });
         }
         else if (acAction == "2") {
             MyProfileSignInService.instance()
                 .getAgentConnectConnectedScreen(UserID)
                 .done(function (html) {
                     $("#agentConnectGreeting").html(html.d);
                     $('#areConnected').on("hidden.bs.modal", function () {
                         cleanUrlAndReload();
                     }).modal('show');
                 })
                 .fail(function (xhr) {
                     HideWaitImg();
                 });
         }        
         else {
             var msg = '';
 
             switch (acAction) {
             case "INVALID_INVITATION_KEY":
                 msg = "Ungültige Einladung";
                 break;
             case "INVALID_AGENT":
                 msg = "Sorry, the Agent you are trying to connect with is not active. Please use our great Agent Search to find a new agent.";
                 break;
             case "INVALID_CONTACT":
                 msg = "Ungültiger Kontakt";
                 break;
             case "INVALID_INVITATION_ALREADY_ACCEPTED":
                 msg = "Looks like this invitation has expired. Please reach out to your Agent if you have any questions or sign into My RE/MAX to review your Agent Connect.";
                 break;
             }
             
             if (msg != '') {
                 var modalId = "modal-agentConnect-notification";
                 $("#modal-agentconnect-notification-title").html("Agent Connect Notification");
                 $("#modal-agentconnect-notification-text").html(msg);
 
                 BootstrapShowModalNotification(modalId, true, 3);
                 //alert(msg);
             }
         }
 
         if (acAction)
             location.hash = "";
     }
