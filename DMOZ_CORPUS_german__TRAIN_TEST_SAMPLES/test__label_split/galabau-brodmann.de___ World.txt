 				}

 				}
 			});		
 			
 			
 			
 			$(touch).on('click', function(e) {
 		        e.preventDefault();
 		      
 				if (menu.is(':visible')) {
 					menu.slideUp();
 					if($( window ).width() > 600) {
 						$('.logo img').css({'height': 140, 'box-shadow': '0px 8px 6px -6px #888' });

 					if($( window ).width() <= 600) {
 						$('.logo img').css({'height': 130, 'box-shadow': '0px 8px 6px -6px #888' }); 
 					}
 				} else {
 					menu.slideDown();
 					$('.logo img').css({'height': 100, 'box-shadow': 'none'}); 
 
 				}
 			});
 	
 			
 		});
 		
 	
 	
 	$(window).resize(function() {
 			
 			if( $( window ).width() == 1366 && $( window ).height() == 1024)  {
 				
 				$('.ipad').css({'display': 'block'});
 				$('.bot img').css({'display': 'none' });
 				
 			}
 			else{
 				$('.ipad').css({'display': 'none'});
 				$('.bot img').css({'display': 'block' });
 			}
 	
 			if($( window ).width() <= 1190) {
 					$('#navigator').css({'position': 'fixed', 'top': 0,  'background': 'white'});
 					$('#touch-menu').css({'color': 'black'  });
 					
 			}
 			if($( window ).width() > 1190) {
 					$('#navigator').css({'position': 'absolute', 'top': 0,  'background': 'white'});
 					$('#touch-menu').css({'color': 'black'  });
 					
 			}
 		});
