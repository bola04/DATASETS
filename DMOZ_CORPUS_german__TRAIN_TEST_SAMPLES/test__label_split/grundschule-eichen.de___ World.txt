        <script language="JavaScript">
         $(document).ready(function(){
 
 var newsPos = 0;
 var newsMax = 2;
 var autoNewsPlaying = false;
 var runNews;
 var currentNewsId = "";
 var currentNewsId2 = "";
 
 for( var i=0; i< newsMax; i++ )
 {
     currentNewsId = ".teaser_news_item_" + i;
 	$(currentNewsId).click(function(){
 		window.location.href = "/aktuelles/nachrichten/";
 	});	
 }
 	
 
 
 	$( "#news-next" ).click(function() {
 		if (newsPos < newsMax)
 		{
 			newsPos++;
 			fadeAllNews(newsPos);
 		}
 		else
 		{
 			fadeAllNews(0);
 			newsPos = 0;
 		}
 		stopAutoNews();
 	});
 
 	$( "#news-prev" ).click(function() {
 		if (newsPos > 0)
 		{
 			newsPos--;
 			fadeAllNews(newsPos);
 		}
 		else
 		{
 			fadeAllNews(newsMax);
 			newsPos = newsMax;
 		}
 		stopAutoNews();
 	});
 
 	function fadeAllNews(newsNumber){
 		$( "#newsContainer .teaser_news_items" ).each(function(){
 			$(this).fadeOut(700);
 		});
 
 		currentNewsId2 = "#teaser_news_item_" + newsNumber;
 		$( currentNewsId2 ).fadeIn(700);
 	 
 		newsPos = newsNumber;
 	}
 
 	  
 	function startAutoNews(){
 		autoNewsPlaying = true;
 		runNews = window.setInterval(function(){
 			if (newsPos < newsMax)
 			{
 				newsPos++;
 				fadeAllNews(newsPos);
 			}
 			else
 			{
 				fadeAllNews(0);
 				newsPos = 0;
 			}
 		}, 5000);
 	}
   
 	function stopAutoNews(){
 		window.clearInterval(runNews);
 		autoNewsPlaying = false;
 	}
 
 	startAutoNews();
 	});
        </script>
