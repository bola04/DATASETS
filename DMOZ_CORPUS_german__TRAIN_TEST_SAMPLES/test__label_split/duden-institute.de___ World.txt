        var fvTeaser = null;
             var itemMargin = 5;
             var teaserArea = jQuery('.teaserSliderArea');
             var sliderContainer = '#teaserSlider';
             var sliderItem = '.teaserSliderItem';
             var slideIndex = 1;
             var maxSlide = 0;
             jQuery(document).ready(function() {
 				var teaserWidth = teaserArea.width();
 				var teaserHeight = teaserArea.height();
 	
 				setCarouselIndicator();
 	
 				fvTeaser = new FlipView({
 					containerId: sliderContainer,
 				    itemClass: sliderItem,
 				    spaceBetweenItems: itemMargin,
 				    itemWidth: teaserWidth,
 				    itemHeight: teaserHeight,
 				    onSlideFinished: null,
 					initialIndex: null,
 					animationType: 'linear',
 					dragDisable: true,
 					setAutoFlip: true,
 					autoFlipTimeout: 8000,
 					autoFlipCallback: function() {
 						jQuery('.cIndicator').css("border-color","#FFF");
 				        jQuery(".cIndicator[data-index='"+parseInt(fvTeaser.getCurrentItemIndex()+1)+"']").css("border-color","#FDCF00");
 					}
 				});
 	
 				jQuery('.teaserSliderItem').width(teaserWidth);
 				
 				window.onresize = function() {
 					clearTimeout(fvTeaser.autoFlipTimeout);
 					fvTeaser = null;
 					
 					teaserWidth = teaserArea.width();
 					teaserHeight = teaserArea.height();
 					var slideFullWidth = teaserWidth + itemMargin;
 					var itemPos = slideFullWidth - (slideIndex*slideFullWidth);
 
 					fvTeaser = new FlipView({
 						containerId: sliderContainer,
 					    itemClass: sliderItem,
 					    spaceBetweenItems: itemMargin,
 					    itemWidth: teaserWidth,
 					    itemHeight: teaserHeight,
 						initialIndex: null,
 						animationType: 'linear',
 						dragDisable: true,
 						setAutoFlip: true,
 						autoFlipTimeout: 8000,
 						autoFlipCallback: function() {
 							jQuery('.cIndicator').css("border-color","#FFF");
 					        jQuery(".cIndicator[data-index='"+parseInt(fvTeaser.getCurrentItemIndex()+1)+"']").css("border-color","#FDCF00");
 						}
 					});
 
 					jQuery(sliderItem).width(teaserWidth);
 					jQuery(sliderContainer).css('left',itemPos+"px");
 				}
 	
 				jQuery('#teaserSlideLeft').on('click', function() {
 					clearTimeout(fvTeaser.autoFlipTimeout);
                     if(fvTeaser.getCurrentItemIndex() == 0) {
                         slideIndex = maxSlide;
                         fvTeaser.slideTo(maxSlide-1);
                     }else {
                         fvTeaser.slideToLeft();
                         slideIndex = fvTeaser.getCurrentItemIndex() + 1;
                     }
 	          		setCarouselIndicator();
 	      		});
 	
 				jQuery('#teaserSlideRight').on('click', function() {
 					clearTimeout(fvTeaser.autoFlipTimeout);
                     if(fvTeaser.getCurrentItemIndex() + 1 == maxSlide){
                         slideIndex = 1;
                         fvTeaser.slideTo(0);
                     }else {
                         fvTeaser.slideToRight();
                         slideIndex = fvTeaser.getCurrentItemIndex() + 1;
                     }
 					setCarouselIndicator();
                     
 	      		});
 	        });

                 jQuery('.cIndicator').css("border-color","#FFF");
                 jQuery(".cIndicator[data-index='"+slideIndex+"']").css("border-color","#FDCF00");

 					Die Hilfe erfolgt nach einer individuellen Beratung und Diagnostik.
