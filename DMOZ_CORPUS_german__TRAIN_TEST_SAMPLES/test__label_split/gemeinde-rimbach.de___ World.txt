          function setInitialDate()
 {
 	g = new Date();
 	tag = g.getDate();
 	vontag = ((tag < 10) ? "0" + tag : tag);

 	vonmonat = ((monat < 10) ? "0" + monat : monat);

 		tag-=14;
 		monat+=1;
 	}
 	if ((tag < 23) & (monat!=2))
 		tag+=7;

 	{
 		tag-=16;
 		monat+=1;
 	}
 	if (monat==13)

          <h3>
           <span style="font-weight: bold;">
            Rimbach digital erkunden -  QR-Codes auf Granitstelen und Tafeln für Rimbacher Sehenswürdigkeiten und Wanderwege
           </span>
          </h3>
          <p>
           Seit Oktober stehen nun bei diversen Sehenswürdigkeiten der Gemeinde Rimbach neu aufgestellte Granitsäulen und Tafeln mit einem QR-Code und dienen seitdem als "Fremdenführer fürs Handy".
          </p>
          <p>
           Bürgermeister Ludwig Fischer dankte den Initiatoren Amberger Matthias und Niedermayer Heinz jun., und Diana Graßl-Pongratz von der Tourist-Info Rimbach für die Umsetzung dieses Projektes.
          </p>
          <p>
           Jeweils eine Granitstele wurde im
           <span style="font-style: italic;">
            Mehrgenerationenpark
           </span>
           und bei der
           <span style="font-style: italic;">
            Thenrieder Kapelle
           </span>
           aufgestellt. Bei weiteren vier Sehenswürdigkeiten wurden jeweils Aludibond-Tafeln an geeigneter, gut sichtbarer Stelle angebracht. Viele interessante Informationen erhält ab sofort der interessierte Natur-und Wanderfreund über Rimbach und im speziellen über den
           <span style="font-style: italic;">
            Mehrgenerationenpark, die Pfarrkirche St. Michael, über die Burgruine Lichtenegg, über die Thenrieder Kapelle und dem Holmeierhaus.
           </span>
           Zudem sind weitere Informationen, wie etwa die Homepage der Gemeinde Rimbach, nur einen Klick weit entfernt.
          </p>
          <p>
           Auch entlang des
           <a class="external-link-new-window" href="http://www.gemeinde-rimbach.de/171.html#/de/landkreis-cham/default/detail/Tour/164ED6281C8BCC9CCF9962AA0EBBADCD/ri09-rimbacher-kapellenweg" target="_top" title="Öffnet internen Link im aktuellen Fenster">

            </span>
           </a>

Auch hier erfährt man interessante Hintergrundinformationen, z.B. wann oder von wem die jeweilige Kapelle errichtet wurde, oder vom Brauchtum der Aufstellung der Totenbretter etc..
          </p>
          <p>
           Zur Komplementierung wurde auch noch der
           <a class="external-link-new-window" href="http://www.gemeinde-rimbach.de/171.html#/de/landkreis-cham/default/detail/Tour/ADC99D28B8F4F923B21C7AECADC4735B/ri05-rimbacher-panoramaweg" target="_top" title="Öffnet internen Link im aktuellen Fenster">
