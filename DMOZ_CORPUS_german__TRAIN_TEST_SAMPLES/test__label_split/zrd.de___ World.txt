 		function slideImages(divOfSlideShow, caption){
 			var $speed = 2000;
 			var $div = "#"+divOfSlideShow;
 			var $active = $($div + ' .active');
 			var $next = ($active.next().length > 0) ? $active.next() : $($div + ' img:first');
 			$next.css('z-index',52);//move the next image up the pile
 			$("#imgcaption").fadeOut($speed/2, function(){
 				$("#imgcaption").html(caption);
 				$("#imgcaption").fadeIn($speed/2);	
 			});
 			$active.fadeOut($speed, function(){//fade out the top image
 				$active.css('z-index',51).show().removeClass('active');//reset the z-index and unhide the image
 				$next.css('z-index',53).addClass('active');//make the next image the top one
 			});
 		}
 		
 		var slideIdx = 0;
 		var slideMax = 4;
 		var slideCaption = [];
         slideCaption[0] = "MRT ohne Platzangst";
         slideCaption[1] = "MRT, CT, Röntgen, Nuklearmedizin und Mehr...";
 		slideCaption[2] = "Offene Kernspintomographie (MRT)";
         slideCaption[3] = "ZRD - Radiologie in Mannheim und Ludwigshafen";
         
 		//slideCaption[4] = "Gro&szlig;er Leistungskatalog";
         //slideCaption[2] = "Computertomographie (CT)";
 		
 		function slide_init(){
 			$("#imgcaption").html(slideCaption[slideIdx]);
 		}
 		
 		function slide(){
 			slideIdx++;
 			if (slideIdx == slideMax) slideIdx = 0;
 			slideImages('imgwrapper', slideCaption[slideIdx]);
 		}
 		
 		slide_init();
 		$(document).ready(function(){
 			setInterval("slide()", 6500);
 		});
