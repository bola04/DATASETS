     function getBgURL(elem) {
         var bg_url = elem.css('background-image');
         // ^ Either "none" or url("...urlhere..")

         return bg_url;

           var createOutletMapComponent = function(nodeID, data){

                     var map = new google.maps.Map(document.getElementById('outlet-map-'+ nodeID), {
                         zoom: 12,
                         center: new google.maps.LatLng(0, 0),
                     });
 
                     for(var i = 0; i < data.markers.length; i++){
                         createMarker(data.markers[i]);
                     }
                     function createMarker(item) {
                         var position = new google.maps.LatLng(item.lat, item.lng);
                         var marker = new google.maps.Marker({
                             position: position,
                             icon: item.icon,
                             title: item.name,
                             map: map,
                             scrollwheel: false,
                             animation: google.maps.Animation.DROP,
                         });
                         bounds.extend(marker.position);
                         //POPIN
                         var infowindow = new google.maps.InfoWindow({
                             content: item.content
                         });
                         marker.addListener('click', function() {
                             infowindow.open(map, this);
                         });
                     }
                     google.maps.event.addListenerOnce(map, 'idle', function() {
                         google.maps.event.trigger(map, 'resize');
                     });
                     map.setCenter(bounds.getCenter());
                     if(data.markers.length > 1) map.fitBounds(bounds);
                     map.setOptions({styles: data.style});
                 };

         /**
          * Actually submits a form.
          *
          * @param form jQuery form element
          */
         function submitForm(form) {
             var id_captcha = form.find('.id-captcha').val();
             if (grecaptcha.getResponse(id_captcha) !== '') {
                 $.ajax({
                     method: "POST",
                     url: form.attr('action'),
                     data: form.serializeArray()
                 }).done(function (response) {
                     grecaptcha.reset(id_captcha);
                     if (response !== 'error') {
                         form.parent().removeClass('border-alert')
                             .find('.callback_response').html(response).fadeIn('slow', function () {
                             $(this).delay(10000).fadeOut('slow');
                         });
                     } else {
                         form.parent().addClass('border-alert');
                     }
                 });
             }
         }
 
         /**
          * Tests multiple inputs at the same time using the testInput function.
          * Returns all errors at once instead of stopping at the first.
          *
          * @param {array} inputs Inputs to test
          * @return {boolean} True if every validation succeeds
          */
         function testInputs(inputs){
             var success = true;
             for (var i = 0; i < inputs.length; ++i){
                 success = testInput(inputs[i].input, inputs[i].type) && success
             }
             return success;
         }
 
         /**
          * Interceptor to the actual input validation.
          * Calls the validation and handles its result.
          *
          * @param input jQuery input element
          * @param {string} type Input type
          * @return {boolean} True if the validation succeeds
          */
         function testInput(input, type){
             var success = validateContentForm(input, type);
 
             if (success) {
                 input
                     .parent().removeClass('has-danger').addClass('has-success')
                     .find('.form-control-feedback').hide();
             } else {
                 input
                     .parent().removeClass('has-success').addClass('has-danger')
                     .find('.form-control-feedback').html(errors[type]).show();
             }
 
             return success;
         }
 
         /**
          * Actual input validation.
          *
          * @param input jQuery input element
          * @param {string} type Input type
          * @return {boolean} True if the validation succeeds
          */
         function validateContentForm(input, type) {
             if (input.size() > 0) {
                 switch (type) {
                     case "phone" :
                         var needToStrip = new RegExp('[0\(\)\.\+ ]', 'gi');
                         var phoneCheck = input.val().replace(needToStrip, '');
                         return !(phoneCheck != parseInt(phoneCheck));
 
                     case "email":
                         var pattern = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
                         return pattern.test(input.val());
 
                     case 'checkbox':
                         return input.prop('checked');
 
                     case 'input':
                     default:
                         return !!input.val();
                 }
             } else {
                 return false
             }
         }
