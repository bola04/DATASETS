     $('#upload-pin-btn').fileupload({
         url: '/pin/index/upload_image',
         dataType: 'json',
         done: function(e, data) {
             if(data.result.error == 1) {
                 $('#pin-upload-progress').removeClass('progress-success').addClass('progress-danger');
             }
             else if(data.result.files) {
                 $.each(data.result.files, function(index, file) {
                     $('#pin-upload-progress').removeClass('progress-danger').addClass('progress-success');
                     $('#pin-upload-progress').fadeOut();
                     $('.modal').modal('hide');
                     MPIN.loadPinForm(file.url, file.url);
                 });
             }
         },
         progressall: function(e, data) {
             $('#pin-upload-progress').show();
             var progress = parseInt(data.loaded / data.total * 100, 10);
             $('#pin-upload-progress .bar').css(
                     'width',
