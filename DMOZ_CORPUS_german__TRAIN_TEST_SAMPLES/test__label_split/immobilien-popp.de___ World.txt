         filterObjektart.onchange = function() {
             var objektart_value = $('#select-objektart').val();
 
             switch(objektart_value) {
                 case 'wohnung':
                 case 'haus':
                     showAllKaufmieteOptions();
                     var kaufmiete_value = $('#select-vermarktungsart').val();

                         case 'kauf':
                             showZimmerWohnflaecheKaufpreis();

                         case 'mietepacht':
                             showZimmerWohnflaecheKaltmiete();

                 case 'grundstueck':
                 case 'bueropraxen':
                 case 'gastgewerbe':
                 case 'einzelhandel':
                 case 'hallenlagerprod':
                     showAllKaufmieteOptions();
                     var kaufmiete_value = $('#select-vermarktungsart').val();
                     switch(kaufmiete_value) {
                         case 'kauf':
                             showFlaecheKaufpreis();
                             break;

                             showFlaecheKaltmiete();
                             break;

                     $("#select-vermarktungsart option[value='']").remove();
                     $("#select-vermarktungsart option[value='mietepacht']").remove();
                     showFlaecheKaufpreis();
                     break;
                 default:
                     showAllKaufmieteOptions();
                     showFlaeche();
                     break;
             }
         }
 
         function showAllKaufmieteOptions() {
             $("#select-vermarktungsart option[value='']").remove();
             $("#select-vermarktungsart option[value='kauf']").remove();
             $("#select-vermarktungsart option[value='mietepacht']").remove();
 
             $('#select-vermarktungsart').append('<option value="">Alle anzeigen</option>');
             $('#select-vermarktungsart').append('<option value="kauf">Kauf</option>');
             $('#select-vermarktungsart').append('<option value="mietepacht">Miete</option>');
 
             $("#select-vermarktungsart option[value='kauf']").attr('selected',true);
 
             var kaufmiete_value = $('#select-vermarktungsart').val();
             var objektart_value = $('#select-objektart').val();
             filterCondition(objektart_value,kaufmiete_value);
         }
         function filterCondition(objektart_value,kaufmiete_value) {
             switch(kaufmiete_value) {
                 case 'mietepacht':
                     switch(objektart_value) {
                         case 'wohnung':
                         case 'haus':
                             showZimmerWohnflaecheKaltmiete();
                             break;
                         case 'grundstueck':
                         case 'bueropraxen':
                         case 'gastgewerbe':
                         case 'einzelhandel':
                         case 'hallenlagerprod':
                             showFlaecheKaltmiete();
                             break;
                         default:
                             showFlaeche();
                             break;
                     }
                     break;
                 case 'kauf':
                     switch(objektart_value) {
                         case 'wohnung':
                         case 'haus':
                             showZimmerWohnflaecheKaufpreis();
                             break;
                         case 'grundstueck':
                         case 'bueropraxen':
                         case 'gastgewerbe':
                         case 'einzelhandel':
                         case 'hallenlagerprod':
                             showFlaecheKaufpreis();
                             break;
                         case 'filter_anlage':
                             $("#select-vermarktungsart option[value='filter_alle']").remove();
                             $("#select-vermarktungsart option[value='filter_miete']").remove();
                             showFlaecheKaufpreis();
                             break;
                     }
                     break;
                 default:
                     switch(objektart_value) {
                         case 'wohnung':
                         case 'haus':
                             showZimmerWohnflaeche();
                             break;
                         default:
                             showFlaeche();
                             break;
                     }
                     break;
             }
         }

       function setSliderTips(slider,property) {
                     var tipHandles = slider.getElementsByClassName('noUi-handle'),
                             tooltips = [];
                     // Add divs to the slider handles.
                     for ( var i = 0; i < tipHandles.length; i++ ){
                         tooltips[i] = document.createElement('div');
                         tipHandles[i].parentElement.parentElement.appendChild(tooltips[i]);
                         tooltips[i].className += 'noUi-handle-tooltip fixedTips';
                     }
 
                     // When the slider changes, write the value to the tooltips.
                     slider.noUiSlider.on('update', function( values, handle ){
                         tooltips[handle].innerHTML = values[handle];
 
                         handleparent = $(tipHandles[handle]).parent();
                         if (handleparent.attr('style') == 'left: 0%;' || handleparent.attr('style') == 'left: 100%;') {
                             handleparent.addClass('noUi-edge');
                         }else{
                             handleparent.removeClass('noUi-edge');
                         }
 
                         // &nbsp;m² und &nbsp;€ entfernen
                         for(var i = 0; i <= 1; i++) {
                             values[i] = values[i].replace('&nbsp;m²','');
                             values[i] = values[i].replace('&nbsp;€','');
                             values[i] = values[i].replace('.','');
                         }
 
                         $('input.'+property+'-min').val(values[0]);
                         $('input.'+property+'-max').val(values[1]);
                     });
                 }
