 function renderCards(cards, sep) {
         var str = '';
         if (! jQuery.isArray(cards))
                 return str;
 
         var old = '♠';
         for (var i in cards) {
                 var card = cards[i];
                 var suit = card.substr(-1, 1);
 
                 card = card.replace(/^(\S+)♠$/i, "<span class=\"black\">$1&spades;</span>");
                 card = card.replace(/^(\S+)♣$/i, "<span class=\"black\">$1&clubs;</span>");
                 card = card.replace(/^(\S+)♦$/i, "<span class=\"red\">$1&diams;</span>");
                 card = card.replace(/^(\S+)♥$/i, "<span class=\"red\">$1&hearts;</span>");
 
                 if (suit != old) {
                         str += sep;
                         old = suit;
                 }
 
                 str += card;
         }
 
         return str;
 }
 
 function renderGame(aData) {
         if (!aData)
                 return 'Последняя игра не найдена';
 
         //console.dir(aData);
 
         var cards   = aData['cards'];
         var trix    = cards['TRIX'];                        
         var winner  = cards['WINNER'];

         var discard = cards['DISCARD'];                        
         var ids     = aData['id'];
         var rid     = aData['rid'];
         var pos     = [ '(' + (1 + aData['pos'][0]) + ')',

         var player  = [ '<img src="' + aData['avatar'][0] + '" height="45"><br>' + pos[0] + aData['name'][0],

         var run = '';

                 for (var i = 0; i < trix.length && i < winner.length; i++) {
