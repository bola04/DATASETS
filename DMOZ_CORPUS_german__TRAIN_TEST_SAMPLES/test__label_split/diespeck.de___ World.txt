   function calculatetopvalue(sliderWidth, sliderHeight, imageWidth, imageHeight) {
     var topvalue = 1/2*(sliderHeight-sliderWidth/imageWidth*imageHeight);
     topvalue = topvalue | 0;
       //Bitwise OR to compute integer...
       //Siehe: http://stackoverflow.com/questions/596467/how-do-i-convert-a-number-to-an-integer-in-javascript
     return topvalue;
   }
 
   //
   // Floatende Elemente positionieren
   //
   function elementspositioning() {
     var sliderWidth = jQuery(window).width();
     var sliderHeight = jQuery('#slider-wrapper').height();
 
     jQuery('.bx-wrapper img').each(function(index) {
       var currImg = jQuery(this);
       
       var contentWidth = 960;
 
       var imageWidth = currImg.attr('width');
       var imageHeight = currImg.attr('height');
 
       //
       // Horizontal positioning of elements
       //
       var navElementIndent = 25;
       var navPagerIndent = 173;
       if (imageWidth < sliderWidth) {
         jQuery('.bx-wrapper .bx-prev').css('left', (imageWidth-contentWidth)/2+navElementIndent);
         jQuery('.bx-wrapper .bx-next').css('right', (imageWidth-contentWidth)/2+navElementIndent);
         jQuery('.bx-wrapper .bx-pager').css('left', (imageWidth-contentWidth)/2+navPagerIndent);
       } else if (sliderWidth < contentWidth) {
         jQuery('.bx-wrapper .bx-prev').css('left', navElementIndent);
         jQuery('.bx-wrapper .bx-next').css('right', navElementIndent);
         jQuery('.bx-wrapper .bx-pager').css('left', navPagerIndent);
       } else {
         jQuery('.bx-wrapper .bx-prev').css('left', (sliderWidth-contentWidth)/2+navElementIndent);
         jQuery('.bx-wrapper .bx-next').css('right', (sliderWidth-contentWidth)/2+navElementIndent);
         jQuery('.bx-wrapper .bx-pager').css('left', (sliderWidth-contentWidth)/2+navPagerIndent);
       }
 
       //
       // Vertical positioning of images
       //
       var topvalue = 0;
       if (imageWidth < sliderWidth) {
         topvalue = calculatetopvalue(imageWidth, sliderHeight, imageWidth, imageHeight);
 
         // Variante 1:
         //jQuery("#slider-wrapper").css('width', imageWidth);
         //jQuery("#slider-wrapper").css('margin', '0px auto');
 
         // Variante 2:
         jQuery("#slider-wrapper").css('width', sliderWidth);
         jQuery("#slider-wrapper ul li img").each(function(index) {
           var thisone = jQuery(this);
           thisone.attr('width', sliderWidth);
           thisone.attr('height', imageHeight/imageWidth*sliderWidth);
           //jQuery("#headerpagetitle").html('<font size="2">Sizing: ' +
           //                                 sliderWidth + " | " +
           //                                 imageHeight/imageWidth*sliderWidth + " | " +
           //                                 '</font>');
         });
 
       } else if (imageHeight/imageWidth*sliderWidth < sliderHeight) {
         topvalue = sliderHeight/2-imageHeight/imageWidth*sliderWidth/2;
         jQuery("#slider-wrapper").css('width', '100%');
         jQuery("#slider-wrapper").css('margin', '0px 0px 0px 0px');
       } else {
         topvalue = calculatetopvalue(sliderWidth, sliderHeight, imageWidth, imageHeight);
         jQuery("#slider-wrapper").css('width', '100%');
         jQuery("#slider-wrapper").css('margin', '0px 0px 0px 0px');
       }
       currImg.css('top', topvalue);
       
       //alert(sliderWidth + " | " + sliderHeight + " | " + imageWidth + " | " + imageHeight + " | " + topvalue);    
       //jQuery("#headerpagetitle").html('<font size="2">Centering: ' +
       //                                 currImg.attr('src') + " | " +
       //                                 sliderWidth + " | " +
       //                                 sliderHeight + " | " +
       //                                 imageWidth + " | " +
       //                                 imageHeight + " | " +
       //                                 topvalue + '</font>');
     });
 
     jQuery('.bx-wrapper .bx-controls-direction a').each(function(index) {
       var thisone = jQuery(this);
       thisone.css('top', -sliderHeight/2);
     });
 
   }
