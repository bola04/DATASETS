 		//sliderBoxes.bind('touchstart', function (evt) { // Verhindert swipe mit dem finger
 		sliderBoxes.on('click', function (evt) {
 			if (evt.target.nodeName != 'A') {
 
 				var $this = $(this);
 
 				if ($this.hasClass('active')) {
 					$this.removeClass('active');
 				} else {
 					sliderBoxes.filter('.active').removeClass('active');
 					$this.addClass('active');
 				}
 				return false;
 			}
 		});
 
 		/*
 		 * Blog-Teaser-Ansicht
 		 */
 		$('div.blogview').BlogTeaser();
 
 		var dot2Options = {
 			height: 110,
 			callback: function( isTruncated ) {
 				var $parent = $(this).parent();
 
 				if (!isTruncated) {
 					$parent.find('a.togglelink').hide();
 				}
 			}
 		};
 
 		// Da es sonst immer einen Hinweis vom dotdotdot-Plugin gibt, dass es den span nicht gibt
 		var $seoBoxSpan = $('div.infoblock.seobox span')
 		if ($seoBoxSpan.length > 0) $seoBoxSpan.dotdotdot(dot2Options);
 
 		$('div.infoblock.seobox').find('a.togglelink').click(function() {
 			var $this = $(this),
 					$dot3Container = $this.parent().find('span');
 
 			if ($this.hasClass('active')) {
 				$dot3Container.dotdotdot(dot2Options);
 
 				// Hide less link
 				$this.hide();
 
 				// Show more link
 				$this.parent().find('a.togglelink').not('.active').show();
 			} else {
 				var originalContent = $dot3Container.triggerHandler('originalContent');
 
 				$dot3Container
 						.trigger('destroy')
 						.html(originalContent);
 
 				// Hide show more link
 				$this.hide();
 
 				// Show less link
 				$this.parent().find('.active').show();
 			}
 
 			return false;
 		});
 
 	});
