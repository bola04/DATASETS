    function fade(fadingType, mouseAxn, objId, opacity) 
 {
 	obj = document.getElementById(objId);
 		
 	if (obj != null)
 	{
 		if (mouseAxn == 'down')
 		{
 			//document.images[objId].src = "assets/images/"+objId+"Klick.jpg";
 			//setOpacityFade('show', obj, 100);
 			return;
 		}
 					
 		if (fadingType == 'out')
 		{
 	    	   if (opacity >= 0) 
 		   {
 		     	setOpacityFade(fadingType, obj, opacity);
     		  	opacity = opacity - 10;
     			window.setTimeout("fade('out', '"+mouseAxn+"', '"+objId+"',"+opacity+")", 20);
 		   }
 		   else
 		   {
 		   	if (mouseAxn == 'over')
 		   	{
 		   		document.images[objId].src = "assets/images/"+objId+"On.jpg";
 				window.setTimeout("fade('in', '"+mouseAxn+"', '"+objId+"',"+opacity+")", 20);
 			}
 			else
 		    	{
 		    		document.images[objId].src = "assets/images/"+objId+".jpg";
 		    		window.setTimeout("fade('in', '"+mouseAxn+"', '"+objId+"',"+opacity+")", 20);
 	    	      }
 	         }
 	      }
 	      else if (fadingType == 'in')
 	      {
 		   if (opacity <= 100) 
 		   {
 	     	      setOpacityFade(fadingType, obj, opacity);
     			opacity = opacity + 10;
 			window.setTimeout("fade('in', '"+mouseAxn+"', '"+objId+"',"+opacity+")", 20);		    	
 		   }
 	      }
          }
 	}
 	
 	function setOpacityFade(fadingType, obj, opacity) 
 	{
 		if (fadingType == 'out')
 		{
 		  	opacity = (opacity == 0)?0:opacity;
 		}
 		else if (fadingType == 'in')
 		{
 			opacity = (opacity == 100)?100:opacity;
 		}
 		else
 		{
 			opacity = 100;
 		}
 	
   	      // IE/Win
 		obj.style.filter = "alpha(opacity:"+opacity+")";
   
 	  	// Safari<1.2, Konqueror
 	  	obj.style.KHTMLOpacity = opacity/100;
   
 	  	// Older Mozilla and Firefox
 	  	obj.style.MozOpacity = opacity/100;
   
 	  	// Safari 1.2, newer Firefox and Mozilla, CSS3
 	  	obj.style.opacity = opacity/100;
 	}
