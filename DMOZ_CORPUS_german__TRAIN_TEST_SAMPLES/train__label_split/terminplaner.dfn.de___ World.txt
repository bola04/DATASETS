 function addBefore(text) {
 	$("div.fcol").eq(0).clone().prependTo($("div.fcols")).find("input").attr('value', '').focus(function () {
 		fillfields();
 	});	
 	$("div.fcol").eq(0).find("input.fcoli").attr('value', text)
 }
 
 function addAfter(text) {
 
 	if (text != '' && $("div.fcol:last").find("input.fcoli").attr('value') == '') {
 		$("div.fcol:last").find("input.fcoli").attr('value', text).focus();
 	}
 	else {
 		$("div.fcol:last").clone().appendTo($("div.fcol").parent())
 		$("div.fcol:last").find("input.fcoli").attr('value', text).focus();
 	}
 
 	$("div.fcol:last").find("div.subcolcontainer input").each(function(i){
 		$("div.fcol:last").find("div.subcolcontainer input").eq(i).attr('value', '');
 	});
 
 	$("div.fcol:first").find("p.deletebox").removeClass('hidden');
 	$("div.fcol:last").find("p.deletebox").removeClass('hidden');
 }
 
 function getDefinitionString() {
 	var defs = Array();
 	
 	$("div.fcol").each(function(i){
 		if (!$("div.fcol").eq(i).hasClass('notinuse')) {
 			var tdef = Array();
 			$("div.fcol").eq(i).find("div.subcolcontainer input[@value != '']").each(function(ii){
 				tdef.push( $("div.fcol").eq(i).find("div.subcolcontainer input[@value != '']").eq(ii).attr('value').replace(/,/, ";") );
 			});
 			if (tdef.length > 0) {
 				defs.push( $("div.fcol").eq(i).find('input.fcoli').attr('value').replace(/,/, ";") + '(' + tdef.join(',') + ')' );
 			} else {
 				defs.push( $("div.fcol").eq(i).find('input.fcoli').attr('value').replace(/,/, ";") );
 			}
 		}
 	});
 	var defstr = defs.join('|');
 	return defstr;
 }
 
 function fillfields() {
 	var columncount = 0;
 	$("div.fcol").each(function(i){
 		columncount++;
 		$("div.fcol p.deletebox").eq(i).removeClass('hidden');
 	});
 	if (columncount == 1) {
 		$("div.fcol p.deletebox").addClass('hidden');
 	}
 }

 function submit_foodle_form() {
 	 /* Required form fields:
 	 	- name
 	 	- at least one column with some string
 	 	- deadline/expire (expiration date)
 	 	Don't submit if one of those isn't filled usefully ...
 	 */
 
 	var submit_ok = true;
 	var err_msg = '';
 	var tab_to_be_corrected = $("#foodletabs > ul").tabs('length'); //Tab number, count starts from left to right with 0
 	var tab_nr; //Tab, the wrong field is in, count starts from left to right with 0
 
 	/* name field */
 	tap_nr = 0;
 	if ($("input[@name=name]").attr('value') == '' ) {
 		submit_ok = false;
 		err_msg += '- Bitte vergeben Sie einen Namen.\n';
 		$("input[@name=name]").addClass('badfield');
 		if (tab_to_be_corrected >= 0 && tap_nr < tab_to_be_corrected) {
 			tab_to_be_corrected = 0;
 		}
 	}
 	else
 	{
 		$("input[@name=name]").removeClass('badfield');
 	}
 
 
 	/* column names */
 	tap_nr = 1;
 	var defstr = getDefinitionString();
 	if (defstr == '' || defstr.replace(/|/g, "") == '') {
 		submit_ok = false;
 		err_msg += '- Bitte legen Sie mindestens eine Spalte mit einen Spaltennamen an.\n';
 		$("div.fcol:first").parent().addClass('badfield');
 		if (tab_to_be_corrected >= 0 && tap_nr < tab_to_be_corrected) {
 			tab_to_be_corrected = 1;
 		}
 	}
 	else
 	{
 		$("div.fcol:first").parent().removeClass('badfield');
 	}
 
 	/* deadline/expire */
 	tap_nr = 0;
 	var tmp_date = $("input[@name=expire]").attr('value');
 	if (tmp_date == '' || isNaN(Date.parse(tmp_date.replace(/-/g, '/'))) === true ) {
 		submit_ok = false;
 		err_msg += '- Bitte geben Sie ein gültigesAblaufdatum an.\n';
 		$("input[@name=expire]").addClass('badfield');
 		if (tab_to_be_corrected >= 0 && tap_nr < tab_to_be_corrected) {
 			tab_to_be_corrected = 0;
 		}
 	}
 	else
 	{
 		$("input[@name=expire]").removeClass('badfield');
 	}
 
 	if (err_msg != '') {
 		alert('\nSie haben nicht alle erforderlichen Felder korrekt ausgefüllt:\n\n'+err_msg+'\n\n');
 		if (tab_to_be_corrected >= 0) {
 			$("#foodletabs > ul").tabs('select', tab_to_be_corrected);
 		}
 		return submit_ok;
 	}
 	else
 	{
 		/* O'reidy, let's go! */
 		return submit_ok;
 	}
 }

      <h1>
       Drei Versionen verfügbar
      </h1>
      <p>
      </p>
      <h3>
       <a href="https://terminplaner.dfn.de">
        DFNTerminplaner (Diese Version)
       </a>
      </h3>
      <p>
       Der DFNTerminplaner ist die einfachste von drei Versionen, sie ist auf die wichtigsten Funktionalitäten beschränkt und vollständig anonym nutzbar.
      </p>
      <h3>
       <a href="https://terminplaner2.dfn.de">
        DFNTerminplaner+
       </a>
      </h3>
      <p>
       Der DFNTerminplaner+ bietet einen größeren Funktionsumfang u.a. eine Benachrichtigungsfunktion für den Ersteller der Abstimmung und eine zusätzliche Antwortmöglichkeit in Form einer „Vielleicht “-Option. Um eine Abstimmung zu erstellen, muss ein Nutzerkonto eingerichtet werden.
      </p>
      <h3>
       <a href="https://abstimmung.dfn.de">
        DFNTerminplaner 3.0
       </a>
      </h3>
      <p>
       Der DFNTerminplaner 3.0 verfügt über ein aktualisiertes Nutzerinterface und bietet eine erweiterte Auswahl von Antwortmöglichkeiten (Ja/Nein oder Ja/Nein/Vielleicht). Außerdem erleichtert er das Einladen von Teilnehmern durch die Möglichkeit Gruppen anzulegen. Ersteller einer Abstimmung können sich im Terminplaner 3.0 über die DFN-AAI anmelden.
      </p>
