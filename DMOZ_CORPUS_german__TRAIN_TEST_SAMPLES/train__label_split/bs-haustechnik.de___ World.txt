     var iframe = $('.slidein-maps-wrapper iframe');

       var src = $(this).attr('src');
       $(this).removeAttr('src');
       return src;
     });
     iframe.fadeOut(250);
     $('.slidein-icon').click(function () {
       var iframe = $('.slidein-maps-wrapper iframe');
       if ($(this).parent().hasClass('direktanruf') && $(this).parent().hasClass('active')) {
         setTimeout(function () {
           iframe.attr('data-src', function () {
             var src = $(this).attr('src');
             $(this).removeAttr('src');
             return src;
           });
           iframe.fadeOut(0);
         }, 250);
       }
       else if ($(this).parent().hasClass('direktanruf')) {
         setTimeout(function () {
           iframe.attr('src', function () {
             return $(this).data('src');
           });
           iframe.fadeIn(250);
         }, 350);
       };
       if ($(this).parent().hasClass('active')) {
         $(this).parent().removeClass('active');
         $(this).css({
           'top': '0'
         });
         if (!$(this).closest('.slidein-bar').hasClass('align-right')) {
           $(this).parent().css({
             'left': ''
           });
           $(this).css({
             'left': ''
           });
         }
         else {
           $(this).parent().css({
             'right': ''
           });
           $(this).css({
             'left': '',
             'right': ''
           });
         }
         $(this).parent().siblings('li').removeClass('widget-hidden');
         $(this).closest('.slidein-bar').removeClass('widget-hidden');
       }

    function setCookie(cname,cvalue,exdays) { 
   var d = new Date(); 
   d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000)); 
   var expires = "expires=" + d.toGMTString(); 
   document.cookie = cname + "=" + cvalue + ";"+ expires + ";path=/" 
 } 
 function getCookie(cname) { 
   var name = cname; 
   var decodedCookie = decodeURIComponent(document.cookie); 
   var ca = decodedCookie.split(';'); 
   for(var i = 0; i < ca.length; i++) { 
     var c = ca[i]; 
     while (c.charAt(0) == ' ') { 
       c = c.substring(1); 
     } 
     if (c.indexOf(name) == 0) { 
       return c.substring(name.length, c.length); 
     } 
   } 
   return  ""; 
 }
