   <script>
    var impressions;
     $(document).ready(function() 
     {  
       if ( $(window).width() > 767)
       { 
         loadScript("../scripts/jquery.mim.impressions.js", function(){
           impressions = $('#impressions').mimImpressions({
             selectorToHide: '.hideForImpressions',
             onImageShow: function(showIndex) {
               showItem(showIndex);
             },
             onActivateOnce: function() {
               $('a.previewToggle').on('click', function() {
                 $('#impressions .top').fadeToggle();
                 $('#impressionThumbnails').fadeToggle();
                 if($('a.previewToggle').text() == 'Fotovorschau und Logo ausblenden')
                   $('a.previewToggle').text('Fotovorschau und Logo einblenden')
                 else
                   $('a.previewToggle').text('Fotovorschau und Logo ausblenden')
 
                 return false
               });
               $('#impressionBigBack').on('click', function() {
                 impressions.imageBack();
               });
               $('#impressionBigNext').on('click', function() {
                 impressions.imageNext();
               });
             },
             onActivate: function() {
               $('.changeImpressionButton').parent().addClass('selected');
               $('body').css('overflow-y','scroll');
             },
             onDeactivate: function() {
               $('.changeImpressionButton').parent().removeClass('selected');
               $('body').css('overflow-y','auto');
               if($('a.previewToggle').text() == 'Fotovorschau und Logo einblenden'){
                 $('a.previewToggle').trigger('click')
               }
             }
           });
           $('.changeImpressionButton').bind('click', function() {
             impressions.toggle();
             return false
           });
         });
       }
       else
       {
         $('#impressions').hide();
         $('#impressionButton').hide();
       }
     });  
     
     function showItem(showIndex)
     {
       var displayElement = $('ul#entries li').eq(showIndex);
       iText1 = displayElement.find('.impText1').text();
       iText2 = displayElement.find('.impText2').text();
       iImage = displayElement.find('.impImageBig').data('scrbig');
   /*------------------------------------------------------------------*/
       backStretch = backstretch.show(iImage);
   /*------------------------------------------------------------------*/     
     }
