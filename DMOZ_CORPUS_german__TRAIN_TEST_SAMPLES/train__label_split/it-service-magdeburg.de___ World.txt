         if (WordCloud.isSupported) {
             var list  = [];
             var items = $('#wordcloud-344 .wordcloud-items');
 
             items.hide();
 
             items.children('li').each(function () {
                 var item = $(this);
                 var link = item.children('a');

                     weight: item.data('weight'),
                     link: link.length ? link.attr('href') : '',
                     attributes: { 'class': link.length ? 'with-link' : ''}
                 })

                 document.getElementById('wordcloud-344-canvas'),

                     list: list,
                     gridSize: Math.round(18 * 960/1024),
                     weightFactor: 1,
                     rotateRatio: 0.5,
                     rotationSteps: 2,
                     drawOutOfBound: false,
                     shape: 'circle',
                     click: function (item, dimension, event) {
