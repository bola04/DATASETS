             function onPlayerPercent(event) {
                 //console.log("onPlayerPercent");
                 if (event.target.j["playerState"] == YT.PlayerState.PLAYING) {
 
                     // Set the played duration to every tenth because we'll need to also capture 90% played.
                     var t = event.target.j["duration"] - event.target.j["currentTime"] <= 1.5 ? 1 : (Math.floor(event.target.j["currentTime"] / event.target.j["duration"] * 10) / 10).toFixed(2);
 
                     if (parseFloat(t) < 0.25) {
                         t = 0.00;
                     }
                     else if (parseFloat(t) < 0.5) {
                         t = 0.25;
                     }
                     else if (parseFloat(t) < 0.75) {
                         t = 0.50;
                     }
                     else if (parseFloat(t) < 0.9) {
                         t = 0.75;
                     }
                     else if (parseFloat(t) < 1) {
                         t = 0.90;
                     }
                     // duration t needs to be fixed to 2 decimal places
                     t = t.toFixed(2);
 
                     //console.log(t);
 
                     if (!event["lastP"] || t != event["lastP"]) {
                         label = vid + ':' + title;
                         event["lastP"] = t;
                         dataLayer.push({
                             event: "youtube",
                             eventCategory: 'youtube',
                             eventAction: t * 100 + "%",
                             eventLabel: label
                         });

                     }
                     setTimeout(onPlayerPercent, 2000, event);
