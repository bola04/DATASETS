           $('#gallery').next('iframe').remove();

                   var gallery = $('#gallery .kgallerie');
 
           $('.kgallerie-play').click(function (e) {
             e.preventDefault();
 
             if(gallery.kGallerie('is_playing')) {
               gallery.kGallerie('stop');
             } else {
               gallery.kGallerie('play');
             }
           });
 
           $('a.prev').click(function(){gallery.kGallerie('stop').kGallerie('prev'); return false});
           $('a.next').click(function(){gallery.kGallerie('stop').kGallerie('next'); return false});

             var files       = $(this).attr('data-files').split(','),
                 folder      = $(this).attr('data-folder'),
                 background  = $(this).attr('data-background'),
                 count       = files.length,
                 i           = 0;
 
             for(; i < count; i++) {
               files[i] = '/' + folder + '/' + files[i];
             }
 
             $('#gallery').show().next('iframe').remove();
             gallery.kGallerie('reload', files);
 
             $('#submenu li').removeClass('selected');
             $(this).parent().addClass('selected');
 
             // Category background
             $('body').css('background-image', background !== '' ? 'url(\'/' + folder + '/' + background + '\')' : 'none');
 
             return false;
           });
 
           gallery.kGallerie({
             'onchange' : function (index, length, filename) {
               $('.kgallerie-counter').html('<div>' + (length === 0 ? 0 : index + 1) + '/' + length + '</div>');
               $('#gallery .filename span').text(filename);
             },
             'onthumbnails' : function (visible) {
               if(visible) {
                 $('a.prev, a.next').hide();
               } else {
                 $('a.prev, a.next').show();
               }
             },
             'onreset' : function () {
