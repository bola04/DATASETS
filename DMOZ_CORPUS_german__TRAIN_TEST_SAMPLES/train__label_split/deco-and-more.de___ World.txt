        <script type="text/javascript">
         var right = 0;
     
     $(function() {
         var logoandmenuwidth = $('.menuonly .mainmenu').width();
         var logoandmenumenupointwidth = 0;
         
         $( "#mainheader .mainmenupoint" ).each(function( index ) {
             
             logoandmenumenupointwidth = logoandmenumenupointwidth + $(this).width() + 15;
         });
         
         if($(document).width() >= 1230)
         {
             if(logoandmenuwidth < logoandmenumenupointwidth)
             {
                 $('#mainheader .menuonly').addClass('menutobig');
             }
         }
         
         if($('.menuonly .mainmenu').css('paddingRight') == '1px' && !$('#mainheader .menuonly').hasClass('menutobig'))
         {
             $('.menuonly .mainmenu').width(logoandmenumenupointwidth + 5);
         }
     });
 
     function openMainmenu()
     {
         right = $( ".mainmenu" ).css('right').replace('-', '');
         right = right.replace('px', '');
         
         $( ".mainmenu" ).css('opacity', 0);
         $( ".mainmenu" ).css('display', 'block');
         $( ".mainmenu" ).prepend('<span class="close" onclick="return closeMainmenu();">X</span>');
         
         $( ".mainmenu" ).animate({
             right: '+=' + right,
             opacity:1
         }, 300);
         
         return false;
     }
     
     function closeMainmenu()
     {
         $( ".mainmenu" ).animate({
             right: '-' + right,
             opacity:0
         }, 200, function(){
             $( ".mainmenu" ).css('display', 'none');
             $( ".mainmenu .close" ).remove();
         });
     }
