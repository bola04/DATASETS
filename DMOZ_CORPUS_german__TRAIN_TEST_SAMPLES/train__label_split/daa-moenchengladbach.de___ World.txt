     var typeNum = '476';

     function highlight( data, search )
     {
         //data = data.replace(/([\\\.\+\*\?\[\^\]\$\(\)\{\}\=\!\<\>\|\:])/g, "\\$1");
         search = search.trim().split(" ");
         for (searchPart in search) {
             data = data.replace( new RegExp( "(" + search[searchPart] + ")" , 'gi' ), "<strong>$1</strong>" );
         }
         return data;
     }
     function fireSearch(sword,SelectorAddition,location,typeNum,maxItems){
         var SelectorAddition = SelectorAddition || '';
         var location = location || 0;
         var typeNum = typeNum || 476;
         var maxItems = maxItems || 10;
         sword = sword.replace(/[|&;$%@"<>()+,]/g, "").replace(/\s{2,}/g, ' ');
 
         var locationAddition = '';
         if(location!=0){
             locationAddition = "&tx_daabase_pi1[location]="+location;
         }
         $.ajax({
             url: encodeURI('/index.php?type='+typeNum+'&tx_daabase_pi1[action]=liveSearch&tx_daabase_pi1[sword]='+sword+locationAddition),
             success: function(result) {
                 var results = JSON.parse(result);
                 var i = 0;
                 var addition = '<ul id="livesearch'+SelectorAddition+'"  class="list-unstyled livesearch-list">';
                 for (object in results) {
                     addition += '<li><a href="/index.php?id='+theDetailPage+'&tx_daabase_pi1%5Bcourse%5D='+results[i]['uid']+'&tx_daabase_pi1%5Baction%5D=show&tx_daabase_pi1%5Bcontroller%5D=Course&cHash='+results[i]['cHash']+'">'+highlight(results[i]['long_description'], sword)+' ('+results[i]['location_name']+')</a></li>';

                 $('#liveResults'+SelectorAddition).html(addition);
             },
         });
     }
     function debounce(funct, duration) {
         var timer;
         return function(){
             clearTimeout(timer);
             timer = setTimeout(funct, duration);
         }

      <h2>
       <br/>
       Herzlich willkommen
      </h2>
      <p>
       <br/>
       bei der Deutschen Angestellten-Akademie (DAA) in Mönchengladbach.
       <br/>
       Die Deutsche Angestellten-Akademie (DAA) ist eines der größten Weiterbildungsunternehmen der Bundesrepublik Deutschland. Unsere kompetenten Mitarbeiterinnen und Mitarbeiter an mehr als 300 Standorten haben alle ein gemeinsames Ziel: Sie wollen mit bedarfsgerechten Bildungsangeboten dazu beitragen, die beruflichen Ein- und Aufstiegschancen ihrer Kunden nachhaltig zu verbessern.
      </p>
      <p>
       Innerhalb unserer mehr als 60-jährigen Geschichte haben sich über 4 Millionen Menschen in den Einrichtungen der DAA qualifiziert und auf die jeweilig veränderten Bedingungen der Arbeitswelt vorbereitet.
      </p>
      <p>
       Wir freuen uns über Ihr Interesse und laden Sie ein, unser traditionsreiches und innovatives Unternehmen kennen zu lernen.
      </p>
