    var priorities =
   [
     [ "belegwuensche_1_1", "belegwuensche_2_1", "belegwuensche_3_1", "belegwuensche_4_1" ],
     [ "belegwuensche_1_2", "belegwuensche_2_2", "belegwuensche_3_2", "belegwuensche_4_2" ],
     [ "belegwuensche_1_3", "belegwuensche_2_3", "belegwuensche_3_3", "belegwuensche_4_3" ],
     [ "belegwuensche_1_4", "belegwuensche_2_4", "belegwuensche_3_4", "belegwuensche_4_4" ]
   ];
 
 var checkCrowdedColumn = () =>
   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const count = choices.reduce(isChecked, 0);
 
         console.log(`column ${choices} crowded: ${count}`);
 
         return count > 1;
       });
   };
 
 var checkEmptyColumn = () =>
   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const isEmpty = (c) => {
           const e = document.getElementById(c);
 
           return !e.checked;
         }
 
         const empty = choices.every(isEmpty);
 
         console.log(`column ${choices} empty: ${empty}`);
 
         return empty;
       });
   };
 
 document.addEventListener("DOMContentLoaded", function(_) {
   var div = document.getElementById("archetypes-fieldname-belegwuensche");
 
   div.addEventListener("click", function(event) {
       if (checkCrowdedColumn()) {
         console.log("at least one column contains more than one selection");
       } else {
         console.log("no column contains more than one selection");
       }
 
       if (checkEmptyColumn()) {
         console.log("at least one column contains no selection");
       } else {
         console.log("no column contains an empty selection");
       }
     });
 
   var form = document.getElementById("fg-base-edit");
 
   form.addEventListener("submit", function(e) {
       if (checkCrowdedColumn()) {
         alert("Mindestens eine Spalte enthält mehr als eine Auswahl. Bitte ändern Sie Ihre Belegwünsche.");
       } else {
         //
       }
 
       if (checkEmptyColumn()) {
         alert("Mindestens eine Spalte enthält keine Auswahl. Bitte ändern Sie Ihre Belegwünsche.");
       } else {
         //
       }
 
       e.preventDefault();
       e.stopPropagation();
     });
   });

   ];
 
 var checkCrowdedColumn = () =>
   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const count = choices.reduce(isChecked, 0);
 
         console.log(`column ${choices} crowded: ${count}`);
 
         return count > 1;
       });
   };
 
 var checkEmptyColumn = () =>
   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const isEmpty = (c) => {
           const e = document.getElementById(c);
 
           return !e.checked;
         }
 
         const empty = choices.every(isEmpty);
 
         console.log(`column ${choices} empty: ${empty}`);
 
         return empty;
       });
   };
 
 document.addEventListener("DOMContentLoaded", function(_) {
   var div = document.getElementById("archetypes-fieldname-belegwuensche");
 
   div.addEventListener("click", function(event) {
       if (checkCrowdedColumn()) {
         console.log("at least one column contains more than one selection");
       } else {
         console.log("no column contains more than one selection");
       }
 
       if (checkEmptyColumn()) {
         console.log("at least one column contains no selection");
       } else {
         console.log("no column contains an empty selection");
       }
     });
 
   var form = document.getElementById("fg-base-edit");
 
   form.addEventListener("submit", function(e) {
       if (checkCrowdedColumn()) {
         alert("Mindestens eine Spalte enthält mehr als eine Auswahl. Bitte ändern Sie Ihre Belegwünsche.");

         //
       }
 
       if (checkEmptyColumn()) {

       } else {
         //
       }

       e.stopPropagation();
     });
   });

   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const count = choices.reduce(isChecked, 0);
 
         console.log(`column ${choices} crowded: ${count}`);
 
         return count > 1;
       });
   };
 
 var checkEmptyColumn = () =>
   {
     return priorities.some((choices) =>
       {
         const isChecked = (acc, c) => {
           const e = document.getElementById(c);
 
           if (e.checked) { return acc + 1; } else { return acc; }
         };
 
         const isEmpty = (c) => {
           const e = document.getElementById(c);
 
           return !e.checked;
         }
 
         const empty = choices.every(isEmpty);
 
         console.log(`column ${choices} empty: ${empty}`);
 
         return empty;
       });
   };
