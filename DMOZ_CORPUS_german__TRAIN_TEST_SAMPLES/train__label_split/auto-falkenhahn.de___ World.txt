    function updateUrlSearchForModel() {
        var modelLbl = $('#ctl00_ctl08_s_Model option:selected').html();
        formatedBody = "";
        if (modelLbl == ddlModelLabel) {

    function setUrlWebStoreSearchEngine() {
        var advSearch = $('#ctl00_ctl08_hl_advSearch');
        var vehicleSearch = $('#ctl00_ctl08_hl_vehicleSearch');
       
        var price = $("#ctl00_ctl08_hdMaxPrice").val();

        var formatedModelUrl = "";
        if (formatedModel != "")
            formatedModelUrl = formatedModel + "/";

        var formatedBodyUrl = "";
        if (formatedBody != "")
            formatedBodyUrl = formatedBody + "/"

        var paramUrlAdvSearch = "";
        if (price != "")
            paramUrlAdvSearch = "?price=" + price + "&advSearch=true"+ "&CurrentLangMultilangue=" + "de-DE";
        else
            paramUrlAdvSearch = "?advSearch=true" + "&CurrentLangMultilangue=" + "de-DE"; ;

        var paramUrlSearch = "";
        if (price != "") {
            paramUrlSearch = "?price=" + price + "&CurrentLangMultilangue=" + "de-DE"; ;
        }

        var adresseWebService = urlWebstoreSearchEngine;

        if (isWebstoreVO == true) {
        
            var formatedMarqueVO = $('#ctl00_ctl08_Recherche_Marque_VO').val();
            if ($('#ctl00_ctl08_Recherche_Marque_VO option:first').val() == formatedMarqueVO || $('#ctl00_ctl08_Recherche_Marque_VO option:first').val() == undefined)
                formatedMarqueVO = "";
            if (formatedMarqueVO != "")
                formatedMarqueVO = formatedMarqueVO + "/";

            var formatedModelVO = $('#ctl00_ctl08_Recherche_Model_VO').val() ;
            if ($('#ctl00_ctl08_Recherche_Model_VO option:first').val() == formatedModelVO || $('#ctl00_ctl08_Recherche_Model_VO option:first').val() == undefined)
                formatedModelVO = "";
            if (formatedModelVO != "")
                formatedModelVO = formatedModelVO + "/";
                
           
            
            advSearch.attr("href", urlWebstoreVOSearchEngine + formatedMarqueVO + formatedModelVO + paramUrlAdvSearch);
            vehicleSearch.attr("href", urlWebstoreVOSearchEngine + formatedMarqueVO + formatedModelVO + paramUrlSearch);
        }
        else {
            advSearch.attr("href", adresseWebService + formatedModelUrl + formatedBodyUrl + paramUrlAdvSearch);
            vehicleSearch.attr("href", adresseWebService + formatedModelUrl + formatedBodyUrl + paramUrlSearch);
        }


    }
