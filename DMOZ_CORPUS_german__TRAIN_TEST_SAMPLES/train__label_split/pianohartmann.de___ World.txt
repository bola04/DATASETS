   <script type="text/javascript">
    $(function() {
 	var navigationTargets = [{
 			title: "Willkommen",
 			hash: "#start",
 			link: "html/start.htm"
 		},{
 			title: "Handwerk",
 			hash: "#service",
 			link: "html/service.htm"
 		},{
 			title: "Ratgeber",
 			hash: "#ratgeber",
 			link: "html/ratgeber.htm"
 		},{
 			title: "Sachverständiger",
 			hash: "#sachverstaendiger",
 			link: "html/sachverstaendiger.htm"
 		},{
 			title: "Über uns",
 			hash: "#ueberuns",
 			link: "html/ueberuns.htm"
 		},{
 			title: "Indien",
 			hash: "#indien",
 			link: "html/Indien.htm"
 		},{
 			title: "Impressum",
 			hash: "#impressum",
 			link: "html/impressum.htm"
 		}
 	];
 	
 	function navigate(navigateToHash) {
 		var link = navigationTargets[0].link;
 		var title = navigationTargets[0].title;
 		var hash = navigationTargets[0].hash;
 		var titlePrefix = "Pianohaus Hartmann OHG - Klaviere und Flügel aus Bonn - ";
 		
 		for (var index=0;index<navigationTargets.length;index++) {
 			var navigationTarget = navigationTargets[index];
 			if (navigationTarget.hash === navigateToHash) {
 				link = navigationTarget.link;
 				title = navigationTarget.title;
 				hash = navigationTarget.hash;
 			}
 		}
 		
 		// load content html with ajax and put it in the container
 		$(".contentContainer").load(link);
 		
 		// set title
 		document.title = titlePrefix + title;
 		
 		// highlight menu item by hash;
 		$("#mainNav li").removeClass("selected");
 		$('#mainNav li a[href="'+hash+'"]').parent().addClass("selected");
 	}
 	
 	function createMenu() {
 		var navContent = "<ul>";
 		for (var index=0;index<navigationTargets.length;index++) {
 			var navigationTarget = navigationTargets[index];
 			navContent += "<li><a href='"+navigationTarget.hash+"'>"+navigationTarget.title+"</a></li>"
 		}
 		navContent += "</ul>";
 		$("#mainNav").append(navContent);
 	}
 	
 	$(window).hashchange( function(){
 		navigate( location.hash );
 	})
 	
 	$(document).ready(function() {
 		// add navigation menu dynamically
 		createMenu();
 	
 		// first navigation to initial hash
 		navigate(location.hash);
 	});
 });
