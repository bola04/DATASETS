         /**
          * method will show the including content block and hide all other
          * @param element
          */
         function showContent(element)
         {
             hideAll();
             if (element.hasChildNodes()) {
                 var children = element.childNodes;
                 for (var i = 0; i <= children.length; i++) {
                     if(typeof children[i] == 'object') {
                         var className = children[i].className;
                         if (typeof className === 'string' ) {
                             if ( className.search('/content.+/')) {
                                 children[i].setAttribute('class', className + ' visible' );
                             }
                         }
                     }
                 }
             }
 
         }
 
         /**
          * will hide all elments in the div.padd with class .content

         function hideAll()
         {
             $("div.padd div.content").each(function(){
                 removeVisibility(this);
             });
         }
 
         /**
          * method to clear a single item, caused by the context problems
          * @param element
          */
         function removeVisibility(element)
         {
             var className = element.className;
             className = className.replace('visible','');
             element.setAttribute('class',className)
         }

         function fireEvent(element,event){
             //console.log("in Fire Event");
             if (document.createEventObject){
                 // dispatch for IE
                 //console.log("in IE FireEvent");
                 var evt = document.createEventObject();
                 return element.fireEvent('on'+event,evt)
             }
             else{
                 // dispatch for firefox + others
                 //console.log("In HTML5 dispatchEvent");
                 var evt = document.createEvent("HTMLEvents");
                 evt.initEvent(event, true, true ); // event type,bubbling,cancelable
                 return !element.dispatchEvent(evt);
             }
         }

         /**
          * method will show the including content block and hide all other
          * @param element
          */
         function showContent(element)
         {
             hideAll();
             if (element.hasChildNodes()) {
                 var children = element.childNodes;
                 for (var i = 0; i <= children.length; i++) {
                     if(typeof children[i] == 'object') {
                         var className = children[i].className;
                         if (typeof className === 'string' ) {
                             if ( className.search('/content.+/')) {
                                 children[i].setAttribute('class', className + ' visible' );
                             }
                         }
                     }
                 }
             }
 
         }
 
         /**
          * will hide all elments in the div.padd with class .content
          */
         function hideAll()
         {
             $("div.padd div.content").each(function(){
                 removeVisibility(this);
             });
         }
 
         /**
          * method to clear a single item, caused by the context problems
          * @param element
          */
         function removeVisibility(element)
         {
             var className = element.className;
             className = className.replace('visible','');
             element.setAttribute('class',className)
         }

         function fireEvent(element,event){
             //console.log("in Fire Event");
             if (document.createEventObject){
                 // dispatch for IE
                 //console.log("in IE FireEvent");
                 var evt = document.createEventObject();
                 return element.fireEvent('on'+event,evt)
             }
             else{
                 // dispatch for firefox + others
                 //console.log("In HTML5 dispatchEvent");
                 var evt = document.createEvent("HTMLEvents");
                 evt.initEvent(event, true, true ); // event type,bubbling,cancelable
                 return !element.dispatchEvent(evt);
             }
