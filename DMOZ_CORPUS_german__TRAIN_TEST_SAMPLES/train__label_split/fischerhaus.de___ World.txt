    var didScroll;
    var lastScrollTop = 0;
    var delta = 300;
 
    $(window).scroll(function() {
    	didScroll = true;
    });
 
    setInterval(function() {
    	if (didScroll) {
    		hasScrolled();
    		didScroll = false;
    	}
    }, 250);
 
    function hasScrolled() {
    	var st = $(this).scrollTop();
    	
    	if (Math.abs(lastScrollTop - st) <= delta)
    		return;
    	
    	if (st > lastScrollTop) {
    		$('.layout-default .sidebar').addClass('out');
    		$('.layout-default #cmsAsidemenu_nav').fadeOut('fast');
    	} else {
    		if(st + $(window).height() < $(document).height()) {
 			   $('.layout-default .sidebar').removeClass('out');
 			   $('.layout-default #cmsAsidemenu_nav').fadeIn('fast');
    		}
    	}
    	
    	lastScrollTop = st;
    };
