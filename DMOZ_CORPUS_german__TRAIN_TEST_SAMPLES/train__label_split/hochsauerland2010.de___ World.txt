     function DatumFilter() {
 	
 	if ($('#abreiseB').val() == '') {
         
         $('#booking-msg').html('<h3 class="text-white">Bitte geben Sie ein Abreisedatum an!</h3>' );
         $('#abreiseB').css('color','#cc0000');
 		return false;
 	}
 	else {
 		var DatumAnreise = $('#anreiseB').val();
 		var DatumAnreise = DatumAnreise.split('.');
 		
 		var AnreiseDatumTag = DatumAnreise[0];
 		var AnreiseDatumMonat = DatumAnreise[1];
 		var AnreiseDatumJahr = DatumAnreise[2];
 		
 		var DatumAbreise = $('#abreiseB').val();
 		var DatumAbreise = DatumAbreise.split('.');
 		var AbreiseDatumTag = DatumAbreise[0];
 		var AbreiseDatumMonat = DatumAbreise[1];
 		var AbreiseDatumJahr = DatumAbreise[2];
 		
 		var Anfangsdatum = new Date(AnreiseDatumJahr,AnreiseDatumMonat-1,AnreiseDatumTag);
 		var Enddatum = new Date(AbreiseDatumJahr,AbreiseDatumMonat-1,AbreiseDatumTag);
 		
 		Anfangsdatum = new Date(Anfangsdatum.getFullYear(),Anfangsdatum.getMonth(),Anfangsdatum.getDate());
 		Enddatum = new Date(Enddatum.getFullYear(),Enddatum.getMonth(),Enddatum.getDate());
 		
 		var Tage =  Math.floor(( Date.parse(Enddatum) - Date.parse(Anfangsdatum) ) / 86400000);
 			
 			Tage= parseInt(Tage);
 			
 		if (Tage <= 0 ) {
 			alert('Bitte kontrollieren Sie Ihr Anreise/-Abreisedatum!');
 			return false;
 		}
         
         $('#ankjj').val(AnreiseDatumJahr);
         $('#anktt').val(AnreiseDatumTag);
         $('#ankmm').val(AnreiseDatumMonat);
         $('#tage').val(Tage);
 		
 	}
     
     if ($('#anzahl_kind1').val() > 0 && $('#kind_alter1').val() == 'x' ) {
 		//alert('Bitte geben Sie das Alter des Kindes 1 ein!');
         $('#booking-msg').html('<h3 class="text-white">Bitte geben Sie das Alter des Kindes 1 ein!</h3>' );
         $('#kind_alter1').css('color','#cc0000');
         
 		return false;
 	}
     
     if ($('#anzahl_kind2').val() > 0 && $('#kind_alter2').val() == 'x' ) {
 		$('#booking-msg').html('<h3 class="text-white">Bitte geben Sie das Alter des Kindes 2 ein!</h3>' );
         $('#kind_alter2').css('color','#cc0000');
 		return false;
 	}
     
     
 }
    </script>
