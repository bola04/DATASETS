    $( document ).ready(function() {
                 $("#form1 input:text").first().focus();
 
                 $("#form1-passwortvergessen").click(function() {
                     $("#form1").hide();
                     $("#form1 .form1--alert").hide();
                     $("#form2 .form2--alert").hide();
                     $("#form2").show();
                     $("#form2 .form2--fields1").show();
                     $("#form2 input:text").first().focus();
                 });
 
                 $("#form1button1").click(function() {
                     var adr_nr_val = $("#adr_nr").val();
                     var adr_pass_val = $("#adr_pass").val();
                     $("#adr_nr_alt").val(adr_nr_val);
                     $("#adr_pass_alt").val(adr_pass_val);
 
                     var values = {};
                     var mykdnr = adr_nr_val;
                     var mypass = adr_pass_val;
                     var persitent = $("#rememberme").attr('checked')?1:0;
                     values['action'] = "login";
                     values['kdnr'] = mykdnr;
                     values['pass'] = mypass;
                     values['persistent'] = persitent;
                     if ($('#nxloginpage').length > 0) {
                         values['path'] = 'loginpage';
                     }else{
                         values['path'] = '#cart/step2';
                     }
                     $.ajax({
                         type:'POST',
                         url: "ajaxaction.php",
                         data:values,
                         success:logincallback,
                         dataType:"json"
                     });
                     return false;
                 });
 
                 function logincallback(data){
                     if (data.error) {
                         var alerts = "Anmeldung nicht erfolgreich!##Zugangsdaten unbekannt, gesperrt oder Passwort fehlerhaft. Bitte überprüfen Sie Ihre Eingaben und versuchen Sie es erneut.".split("##");
                         nxAlert(alerts[0],alerts[1],"error","OK","shopbutton danger","","",function(callback){
                             if (callback=="nxalertButton1") { $("#form1 input:text").first().focus(); }
                         },false,true)
                     } else {
                         window.location = ".";
                     }
                 }
 
                 $("#form2button1").click(function() {
                     $("#form2 .form2--fields1").hide();
                     $("#form2 .form2--fields2").show();
                 });
 
                 $("#form2button2").click(function() {
                     $("#form2 .form2--fields1").hide();
                     $("#form1").show();
                     $("#form1 input:text").first().focus();
                 });
             });
