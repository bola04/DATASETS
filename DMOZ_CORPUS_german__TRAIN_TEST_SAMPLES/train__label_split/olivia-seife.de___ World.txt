     var OFFSET = 200;
 	var TIMEOUT = 200;
 	var SIDEBAR = $("#SideBar");
 
 	function makeIt() {
 
 		var pageYOffset = window.pageYOffset;
 
 		if (pageYOffset - OFFSET <= 0) {
 			$("#SideBar").css("top", 0);
 			$("#SideBarTopEdge").hide();
 			return;
 		}
 
 		SIDEBAR.css("top", pageYOffset - OFFSET + 20);
 		$("#SideBarTopEdge").show();
 
 	}
 
 	var timeoutId;
 	var previousPageYOffset = 0;
 
 	function push() {
 
 		var pageYOffset = window.pageYOffset;
 
 		if (previousPageYOffset == pageYOffset) {
 			return;
 		}
 
 		previousPageYOffset = pageYOffset;
 
 		window.clearTimeout(timeoutId);
 		timeoutId = window.setTimeout(makeIt, TIMEOUT);
 
 	}
 
 	document.onmouseover = push;
 	document.onmousemove = push;

      <h1>
       Olivia's handgemachte Naturseife
      </h1>
      <img align="right" src="http://www.olivia-seife.de/home-Dateien/tadema.jpg" style="padding: 8px 0 8px 8px; "/>
      <p>
       02.01.2018
      </p>
      <p>
       Liebe Seifenfreunde!
      </p>
      <p>
       Im neuen Jahr möchten wir uns dem Wohlbefinden etwas umfassender widmen und beginnen mit dem Schlaf. Die Neuheiten finden Sie auf unserer Specials-Seite sowie in den entsprechenden Kategorien.
      </p>
      <p>
      </p>
      <p>
       Unsere Haut ist nicht nur unser bester Schutz, und unser größtes Sinnesorgan, sondern auch unser allererster Sympathieträger. Deshalb reinigen und pflegen wir sie.
      </p>
      <p>
       Meine Seifen lassen auf Ihrer Haut nur Eines zurück,
       <br/>
       und das ist Wohlbefinden.
      </p>
      <p>
       Für die Herstellung meiner Seifen verwende ich allerbeste Zutaten, große Sorgfalt und Erfahrung seit dem Jahr 2000
       <br/>
       Die vielfältigen Rezepte garantieren eine auf individuelle Bedürfnisse abgestimmte Pflege. Die Haut gewinnt die natürliche Ausgeglichenheit und Schönheit zurück, die durch Umwelbelastungen, modernen Lebensstil oder Alterserscheinungen ständig angegriffen wird.
      </p>
      <p>
      </p>
      <p>
       Die positive Resonanz aus allen Altersgruppen - von Jugendlichen bis hin zu pflegebedürftigen Senioren - ist eine unglaublche Motivation, Ihnen nur die besten Dinge herzustellen. Und natürlich daß wir mit den natürlichen Düften  zu angenehmen Freunden für unsere sensiblen Haustiere geworden sind.
      </p>
      <p>
      </p>
