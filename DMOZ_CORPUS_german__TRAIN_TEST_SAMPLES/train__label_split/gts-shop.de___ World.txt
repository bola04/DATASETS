     function getModels(manuId) { 
         var optionModels = "";
         optionModels += "<option>- Modell -</option>";
         $("#nrtd_models").removeAttr('disabled');
         if(!manuId) {
             $("#nrtd_models").html(optionModels);
             $("#nrtd_models").attr('disabled', 'disabled');
             return;
         }
         var data = callAjax("nrtd_getmodels", manuId);
         var currentCat = "";
         if(!data) return;       
         $.each(data, function(i) {
             var fromYear = ""; 
             var fromMon = ""; 
             var toYear = ""; 
             var toMon = ""; 
             if(data[i].yearOfConstrFrom){
             fromYear = data[i].yearOfConstrFrom.toString().substring(0,4);
             fromMon = data[i].yearOfConstrFrom.toString().substring(4,6);
             }
             if(data[i].yearOfConstrTo){
             toYear = data[i].yearOfConstrTo.toString().substring(0,4);
             toMon = data[i].yearOfConstrTo.toString().substring(4,6);
             }
 
             words = data[i].modelname.split(" ");
             if(currentCat != words[0]){
                 if(currentCat != "") optionModels += "</optgroup>";
                 currentCat = words[0];
                 optionModels += "<optgroup label="+currentCat+">";
             }
             optionModels += "<option value='"+data[i].modelId+"'>"+data[i].modelname+" ["+fromMon+"/"+fromYear+" - "+toMon+"/"+toYear+" ]</option>"; 
         });
         optionModels += "</optgroup>";
         $("#nrtd_models").html(optionModels);
         $("#nrtd_motors").attr('disabled', 'disabled');
     }
 
     function getMotors(manuId, modId) {     
         var optionMotors = "";
         optionMotors += "<option>- Motor -</option>";
         $("#nrtd_motors").removeAttr('disabled');
         if(!manuId && !modId) {
             $("#nrtd_motors").html(optionMotors);
             $("#nrtd_motors").attr('disabled', 'disabled');
             return;
         }
         var data = callAjax("nrtd_getmotors", manuId, modId);
         var currentCat = "";
         var items = {};       
         if(!data) return;        
         $.each(data, function(i) {
             var item = data[i];
             var powerKwFrom = ""; 
             var powerKwTo = ""; 
             var powerHpFrom = ""; 
             var powerHpTo = ""; 
             if(item.vehicleDetails.powerKwFrom) powerKwFrom = item.vehicleDetails.powerKwFrom;
             if(item.vehicleDetails.powerKwTo) powerKwTo = item.vehicleDetails.powerKwTo;
             if(item.vehicleDetails.powerHpFrom) powerHpFrom = item.vehicleDetails.powerHpFrom;
             if(item.vehicleDetails.powerHpTo) powerHpTo = item.vehicleDetails.powerHpTo;
             console.log(item.vehicleDetails2);
             if(item.vehicleDetails2 == undefined){
                 words = item.vehicleDetails.typeName.split(" ");
             }else{
                 words = item.vehicleDetails2.typeName.split(" ");
             }
 
             if(currentCat != words[0]){
                 if(currentCat != "") optionMotors += "</optgroup>";
                 currentCat = words[0];
                 optionMotors += "<optgroup label="+currentCat+">";
             }
             if(item.vehicleDetails2 == undefined){
                 optionMotors += "<option value='"+item.vehicleDetails.carId+"' >"+item.vehicleDetails.typeName+" ("+powerKwFrom+" KW / "+powerHpFrom+" PS)</option>";
             }else{
                 optionMotors += "<option value='"+item.vehicleDetails.carId+"' >"+item.vehicleDetails2.typeName+" ("+powerKwFrom+" KW / "+powerHpFrom+" PS)</option>";
             }
         });
         optionMotors += "</optgroup></select>";
 
         $("#nrtd_motors").html(optionMotors);
         $("#nrtd_motors").removeAttr('disabled');
     }
 
 
     function getGroupsTree(carId) {
         var out = "<ul>";

     nrMyCounter = 0;
     function loopTree(data, carId) {
         var out = "<ul id='catul"+nrMyCounter+"'>";          
         $.each(data, function(i) {           
             var name = i.replace(/\//g," / ");
             nrMyCounter++;
             out += "<li class='leerspan'>";
             if (isArray(data[i])){                
                out += "<span>"+name+"</span>";
                out += loopTree(data[i], carId);
             }
             else {
                var actClass = "";
                          
                out += "<a class='"+actClass+"' href='https://www.gts-shop.de/index.php?cl=alist&agid="+data[i]+"&carid="+carId+"&cnid=oxmore'>"+name+"</a>";
             }
             out += "</li>";            
         });        
         out += "</ul>";
         $("#nrtd_groupsTree").addClass("nowactiv");                    
         $("#left .tree").hide();
         $("#left .information").css({"border-top":"none","margin-top":"0"});                   
         return out;
     }
 
     function isArray(data) {
         var tmp = "";
         $.each(data, function(i) {
             tmp = 1;
             return false; // breaks the array
         });
         if(tmp == "") return false;
         return true;
     }

     function callAjax(fnc, param1, param2) {    
         var array = "";
         $.ajax({             
             url: 'https://www.gts-shop.de/?cl=nrtecdoc_ajax&fnc='+fnc+'&param1='+param1+'&param2='+param2,
             async: false,
             type: 'GET',
             dataType: 'json',                        
              beforeSend: function(data){
                nrtd_beforesend(data)
             },                       
             success: function (data){
                array = data;
                nrtd_success(data)
             }
         })
         return array;
     }
