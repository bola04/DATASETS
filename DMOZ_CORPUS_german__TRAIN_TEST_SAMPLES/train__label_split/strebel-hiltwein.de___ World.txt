             function ScrollScene_Update_triggerHooks() {
                 //Normalfall: Parallax-ScrollScenes befinden sind unterhalb des ViewPorts
                 //   => Beim Reinscrollen der Scene am unteren ViewPort-Rand soll das Parallax-Scrolling getriggert werden
                 //   => TriggerHook muss am unteren ViewPort-Rand liegen, also bei 100% => TriggerHook=1
                 //Aber falls die Parallax-ScrollScene schon beim Laden der WebSite im ViewPort liegen, darf TriggerHook nicht gleich 100% sein.
                 //   => Sonst würde diese ScrollScene bereits ohne das Scrollen des Users "vorgespult", weil die Scene ja schon am TriggerHook vorbei ist
                 //   => Hier muss der TriggerHook auf die Höhe des triggernden DIVs gesetzt werden.
                 var WindowHeight=$(window).height();
                 for (Key in MyScenes) {
                     var TriggerID=MyScenes[Key].triggerElement();
                     var SequenceNo = TriggerID.replace("#trigger", "");
                     var TriggerTop=$(TriggerID).offset().top;
                     if (TriggerTop<WindowHeight) {
                         MyScenes[Key].triggerHook(TriggerTop/WindowHeight);
                     } else {
                         MyScenes[Key].triggerHook(1);
                     }
 
 
                     
                     //Zusätzlich: Beim Window-Resize ändert sich ja auch die Anzeigehöhe, die durchgescrollt werden kann.
                     //=> 1.) ScrollDistanz anpassen (sog. "duration"), damit das gesamte Parallax-Bild im Anzeigebereich durchgescrollt wird
                         var SceneHeight = 0;
                         if (window["ParallaxRelativeHeight"+SequenceNo]) { //Falls der Benutzer eine dynamische Höhe angegeben hat, wird diese zunächst ausgewertet bevor die "duration" gesetzt werden kann.
                             SceneHeight = Math.round(window["ParallaxRelativeHeight"+SequenceNo]*WindowHeight);
                             $(TriggerID).height(SceneHeight);
                         } else {
                             SceneHeight = $(TriggerID).height();
                         }
                         window["duration" +SequenceNo] = WindowHeight+SceneHeight;
                         window["ImgOffset"+SequenceNo] = (window["ImgVerticalOffset"+SequenceNo] + (window["ImgScrollRatio"+SequenceNo]*window["duration" +SequenceNo])) + "px";
                         MyScenes[Key].duration(window["duration"+SequenceNo]);
                         MyScenes[Key].removeTween(true);
                         MyScenes[Key].setTween(TweenMax.to("#parallax"+SequenceNo, 1, {top: window["ImgOffset"+SequenceNo], ease: Linear.easeNone}));
 
                     //=> 2.) Mindesthöhe anpassen, damit das Parallax-Img den Anzeigebereich immer komplett ausfüllt.
                     //Da die Parallex-Bild-Höhe mit der Browser-Fensterbreite skaliert (weil die Parallax-Bilder nämlich immer 100% Browser-Fensterbreite ausfüllen),
                     //muss die min-width angepasst werden, damit die erforderliche Mindesthöhe nie unterschritten wird.
                         var DistToScroll= (-1* window["ImgScrollRatio"+SequenceNo] * (WindowHeight+SceneHeight) ) + SceneHeight - window["ImgVerticalOffset"+SequenceNo];
                         var ImgAspectRatio=window["ImgAspectRatio"+SequenceNo];
                         var MinWidth = Math.ceil(ImgAspectRatio*DistToScroll) + "px";
                         $("#parallax"+SequenceNo).css("min-width", MinWidth);
 
                 }
             }
 
             ScrollScene_Update_triggerHooks();
             $(window).bind("resizeEnd", ScrollScene_Update_triggerHooks);
         });
