     window.setDayFormat = function (date, widgetId, appointmentInformation, layoutId) {
         var style = "status_empty_" + layoutId;
         var dayInformation = window.findDayInformation(date.getFullYear(), date.getMonth(),
                 date.getDate(), appointmentInformation);
         if (dayInformation !== undefined) {
             var appointments = dayInformation.appointments;
             if (appointments.length > 0) {
                 // check if the whole day is blocked with appointments of the same type
                 var mixed = false;
                 var endSlot = 0;
                 var status = appointments[0].status;
                 style = "status_" + status;
 
                 for (var i = 0; i < appointments.length; i++) {
                     var appointment = appointments[i];
                     if (appointment.startSlot <= endSlot && appointment.status == status) {
                         endSlot = Math.max(endSlot, appointment.endSlot);
                     } else {
                         mixed = true;
                         break;
                     }
                 }
                 if (mixed || endSlot < 95) { // 95 == 23:45 the last selectable time
                     style = "status_mixed";
                 }
             }
         }
         return [true, style];
     };
