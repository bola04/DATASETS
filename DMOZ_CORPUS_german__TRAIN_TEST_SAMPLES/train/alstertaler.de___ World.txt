        function checkform() {

             //Fehlervariablen
             var noName       = false; // Kein Name
             //var noCompany  = false;
             var noPhone      = false; // Keine Telefonnummer
             var noEmail      = false; // Keine EMailadresse
             var invalidEmail = false; // Fehlerhafte EMailadresse
             var noMessage    = false; // Keine Nachricht

             //Ist das Feld "Name" gefüllt?
             if(document.getElementById('name').value.length == 0) {
               noName    = true;
             }

             //Ist das Feld "Firma" gefüllt?
             /*
             if(document.getElementById('firma').value.length == 0) {
               var noCompany = true;
             }
             */

             //Ist das Feld "Phone" gefüllt?
             if(document.getElementById('phone').value.length == 0) {
               noPhone = true;
             }

             //Ist das Feld "EMail" gefüllt und die Adresse valide?
             if(document.getElementById('email').value.length == 0) {
               noEmail = true;
             } else {
               if(validEmail(document.getElementById('email').value) == false) {
                 invalidEmail = true;
               }
             }

             //Ist das Feld "Nachricht" gefüllt?
             if(document.getElementById('message').value.length == 0) {
               noMessage = true;
             }


             /*
             * Prüfen der Fehlervariablen
             */
             if(noName == false && noPhone == false && noEmail == false && invalidEmail == false && noMessage == false) {         // Wenn alle Fehlervariablen false sind...
               return true;                                                                                                       // ... ist alles gut und es kann losgehen
             }
             if(noName == true) {                                                                                                 // Wenn nicht, stimmt der Name nicht?
               alert("Bitte geben Sie Ihren Namen an");
               document.getElementById('name').style.backgroundColor = '#FFCCCC';
               return false;
             }
             if(noEmail == true && noPhone == true) {                                                                             // EMail UND Telefon nicht vorhanden?
               alert("Bitte geben Sie mindestens eine Kontaktmöglichkeit (Email oder Telefon) an.");
               document.getElementById('phone').style.backgroundColor = '#FFCCCC';
               document.getElementById('email').style.backgroundColor = '#FFCCCC';
               return false;
             }
             if(noEmail == false && invalidEmail == true) {                                                                       // EMail falsch?
               alert("Die EMailadresse scheint fehlerhaft zu sein.\nBitte prüfen Sie Ihre Eingabe");
               document.getElementById('email').style.backgroundColor = '#FFCCCC';
               return false;
             }
             if(noMessage == true) {                                                                                              // EMail falsch?
               alert("Sie haben keine Nachricht eingegeben. Bitte prüfen.");
               document.getElementById('message').style.backgroundColor = '#FFCCCC';
               return false;
             }

        }
