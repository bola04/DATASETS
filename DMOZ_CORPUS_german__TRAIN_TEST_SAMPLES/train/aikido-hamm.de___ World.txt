   function marqueetext1(a, b) {
			var width = b.width();
			var start_pos = a.width();
			var end_pos = -width;
			function scroll() {
				if (b.position().left <= -width) {
					b.css('left', start_pos);
					scroll();
				}
				else {
					time = (parseInt(b.position().left, 10) - end_pos) *
						(60000 / (start_pos - end_pos) / 2);
					b.animate({'left': -width}, time, 'linear', function() {
						scroll();
					});
				}
			}
				if('start' == 'start') {		
					b.css({'width': width,'left': start_pos});
					scroll(a, b);				
					$('#text-text1').toggle(function(){
							b.stop();
							b.clearQueue();					
						},
						function(){
							scroll(a, b);
						});
				} else {
					b.css({'width': width,'left': start_pos});
					scroll(a, b);
				}		
			}
