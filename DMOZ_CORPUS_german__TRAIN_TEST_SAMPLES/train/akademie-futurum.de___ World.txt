             <div class="page-header">
              <h3 itemprop="name">
               Informationen zur Akademie
              </h3>
             </div>
             <p>
              Die Akademie Futurum organisiert in loser Abfolge Vorträge und Veranstaltungen zu pädagogischen, sozialen und kulturellen Themen sowie Konzerte.
             </p>
             <p>
              Außerdem bietet sie pädagogischen Einrichtungen und Trägern Fortbildungen an.
             </p>
             <h4>
              Veranstaltungen
             </h4>
             <p>
              Zur Zeit sind keine Veranstaltungen geplant, aber wir arbeiten dran.
             </p>
             <p>
              Unsere letzten Gäste im Ratssaal der Burg Mylau waren:
             </p>
             <table>
              <tbody>
               <tr>
                <td style="text-align: left; vertical-align: top;">
                 am 21. 01. 2015
                </td>
                <td style="text-align: left; vertical-align: top;">
                 Dieter Leicht
                 <br/>
                 „Ich muss alles zehnmal sagen - oder von der Konsequenz in der Erziehung“
                </td>
               </tr>
               <tr>
                <td style="text-align: left; vertical-align: top;">
                 am 18. 08. 2015
                </td>
                <td style="text-align: left; vertical-align: top;">
                 Gee Vero
                 <br/>
                 "Autismus - (M)Eine Andere Wahrnehmung"
                </td>
               </tr>
              </tbody>
             </table>
             <h4>
              Konzerte
             </h4>
             <p>
              Mittwoch, 30.12.2015 16.00 bis 18.00 Uhr:
              <br/>
              Konzert zwischen den Jahren - Musikschüler der Musikschule Vogtland musizieren
             </p>
             <p>
              Leider verpasst oder in guter Erinnerung:
              <br/>
              16. Mai 2015: Klavier-Recital mit Prof. Anahit Nersesyan (Armenien) - in Kooperation mit Calliope Duo Reichenbach
             </p>
             <h4>
              Kurse
             </h4>
             <p>
              Der Fortbildungskurs "Systemische Pädagogik" in Zusammenarbeit mit dem Sächsisches Institut für Systemische Beratung und Therapie/ Familientherapie (SIS) ist für dieses Schulhalbjahr ausgebucht. Bei Interesse wenden Sie sich bitte an uns, wir erstellen angepasst an Ihre Bedürfnisse ein Angebot dieses Kurses, der in drei Modulen (je 2 oder 3 Tage) insbesondere Lehrer sozialpädagogisch fit macht.
             </p>
             <p>
              In Vorbereitung: Yoga für Pädagogen im Elementar- und Primarbereich.
             </p>
             <p>
              Wir freuen uns über Ihr Interesse
              <br/>
              <em>
               Ihr Akademie-Futurum-Team
              </em>
             </p>

function autoSliderHeight() {
	
	
	
	
	if(jQuery('#slider').length) {
		var image_url = jQuery('#slider').css('background-image');
		var image;
		var slider_width = jQuery('#slider').width();
		var padding_left = parseInt( jQuery('#slider').css('padding-left').replace('px', '') );
		var padding_right = parseInt( jQuery('#slider').css('padding-right').replace('px', '') );
		// Remove url() or in case of Chrome url("")
		image_url = image_url.match(/^url\("?(.+?)"?\)$/);
		
		if(image_url[1]) {
			image_url = image_url[1];
			image = new Image();
			// just in case it is not already loaded
			jQuery(image).load(function () {
				slider_width = slider_width + padding_left + padding_right;
				var w = parseInt(100 * slider_width / image.width);
				var h = parseInt(w * image.height / 100);
				jQuery('#slider').height(h);
				jQuery('.slider_content, .slider_content > div').height(h);
				jQuery('#slider .owl-buttons div').css('top', h / 3 + 'px')
			});
			image.src = image_url;
		}
	}
}

              <div class="page-header">
               <h3 itemprop="name">
                Informationen zur Akademie
               </h3>
              </div>
              <p>
               Die Akademie Futurum organisiert in loser Abfolge Vorträge und Veranstaltungen zu pädagogischen, sozialen und kulturellen Themen sowie Konzerte.
              </p>
              <p>
               Außerdem bietet sie pädagogischen Einrichtungen und Trägern Fortbildungen an.
              </p>
              <h4>
               Veranstaltungen
              </h4>
              <p>
               Zur Zeit sind keine Veranstaltungen geplant, aber wir arbeiten dran.
              </p>
              <p>
               Unsere letzten Gäste im Ratssaal der Burg Mylau waren:
              </p>
              <table>
               <tbody>
                <tr>
                 <td style="text-align: left; vertical-align: top;">
                  am 21. 01. 2015
                 </td>
                 <td style="text-align: left; vertical-align: top;">
                  Dieter Leicht
                  <br/>
                  „Ich muss alles zehnmal sagen - oder von der Konsequenz in der Erziehung“
                 </td>
                </tr>
                <tr>
                 <td style="text-align: left; vertical-align: top;">
                  am 18. 08. 2015
                 </td>
                 <td style="text-align: left; vertical-align: top;">
                  Gee Vero
                  <br/>
                  "Autismus - (M)Eine Andere Wahrnehmung"
                 </td>
                </tr>
               </tbody>
              </table>
              <h4>
               Konzerte
              </h4>
              <p>
               Mittwoch, 30.12.2015 16.00 bis 18.00 Uhr:
               <br/>
               Konzert zwischen den Jahren - Musikschüler der Musikschule Vogtland musizieren
              </p>
              <p>
               Leider verpasst oder in guter Erinnerung:
               <br/>
               16. Mai 2015: Klavier-Recital mit Prof. Anahit Nersesyan (Armenien) - in Kooperation mit Calliope Duo Reichenbach
              </p>
              <h4>
               Kurse
              </h4>
              <p>
               Der Fortbildungskurs "Systemische Pädagogik" in Zusammenarbeit mit dem Sächsisches Institut für Systemische Beratung und Therapie/ Familientherapie (SIS) ist für dieses Schulhalbjahr ausgebucht. Bei Interesse wenden Sie sich bitte an uns, wir erstellen angepasst an Ihre Bedürfnisse ein Angebot dieses Kurses, der in drei Modulen (je 2 oder 3 Tage) insbesondere Lehrer sozialpädagogisch fit macht.
              </p>
              <p>
               In Vorbereitung: Yoga für Pädagogen im Elementar- und Primarbereich.
              </p>
              <p>
               Wir freuen uns über Ihr Interesse
               <br/>
               <em>
                Ihr Akademie-Futurum-Team
               </em>
              </p>

 function autoSliderHeight() {
 	
 	
 	
 	
 	if(jQuery('#slider').length) {
 		var image_url = jQuery('#slider').css('background-image');
 		var image;
 		var slider_width = jQuery('#slider').width();
 		var padding_left = parseInt( jQuery('#slider').css('padding-left').replace('px', '') );
 		var padding_right = parseInt( jQuery('#slider').css('padding-right').replace('px', '') );
 		// Remove url() or in case of Chrome url("")
 		image_url = image_url.match(/^url\("?(.+?)"?\)$/);
 		
 		if(image_url[1]) {
 			image_url = image_url[1];
 			image = new Image();
 			// just in case it is not already loaded
 			jQuery(image).load(function () {
 				slider_width = slider_width + padding_left + padding_right;
 				var w = parseInt(100 * slider_width / image.width);
 				var h = parseInt(w * image.height / 100);
 				jQuery('#slider').height(h);
 				jQuery('.slider_content, .slider_content > div').height(h);
 				jQuery('#slider .owl-buttons div').css('top', h / 3 + 'px')
 			});
 			image.src = image_url;
 		}
 	}
 }
 
 
 jQuery(window).resize(function() {
 	autoSliderHeight()
