 /**
  * Erstellt ein Javascript-Menu
  */
 function CreateMenu(Parent,MyOptions)
 {
 	var $Parent = $(Parent);
 	if ($Parent.length < 1) return false;
 	// Optionen
 	var Default = {
 		AnimateIn: 'show()',
 		AnimateOut: 'hide()',
 		HoverOffTime: 400,
 		HoverClass: 'HoverClass',
 		TreeClass: 'TreeClass'
 	};
     var CurOptions = $.extend({}, Default, MyOptions);
 	
 	// alle Ausblenden
 	$Parent.find('ul').hide();
 	// Hover
 	var MakeHover = function (Elem) {
 		Elem.find('>li').each(function(){
 			$this = $(this);
 			// Tiefer?
 			var Lower = $this.find('>ul').length;
 			$this.hoverIntent({
 				timeout: CurOptions.HoverOffTime,
 				over:function(){
 					$this = $(this);
 					//$this.append($Parent.find('li:not(.'+CurOptions.TreeClass+')>ul').length);
 					// Andere Ausblenden :not(.'+CurOptions.TreeClass+')
 					$Parent.find('li:not(.'+CurOptions.TreeClass+')>ul').hide();
 					$this.parent().find('ul').hide();
 					// Markieren
 					$this.find('>*:first').addClass(CurOptions.HoverClass);
 					if (Lower)
 					{
 						$this.addClass(CurOptions.TreeClass);
 						eval("$this.find('>ul')."+CurOptions.AnimateIn+";");
 					}
 				},
 				out: function(){
 					$this = $(this);
 					// Markierung entfernen
 					$this.removeClass(CurOptions.TreeClass)
 						.find('>*:first').removeClass(CurOptions.HoverClass);
 					if (Lower)
 					{
 						window.setTimeout(function(){
 								if ($this.find('.'+CurOptions.HoverClass).length != 0)
 									return;
 								eval("$this.find('>ul')."+CurOptions.AnimateOut+";");
 							},CurOptions.HoverOffTime);
 					}
 				}
 			}).css('cursor','pointer');
 			// Tiefer?
 			if (Lower) MakeHover($this.find('>ul'));
 		});
 	};
 	MakeHover($Parent);
 }
