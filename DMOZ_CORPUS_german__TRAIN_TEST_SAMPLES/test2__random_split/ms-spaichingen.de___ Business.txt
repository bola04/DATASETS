 		function check_textblock(){
 
 			if (!$('body').hasClass('tmpl_5'))
 			{
 				size_textblock();
 			}
 
 			if($('DIV.textblock DIV.textblockNews DIV.newsEintrag').length > 1){
 				var maxBlk = $('DIV.textblock DIV.textblockNews DIV.newsEintrag').length;
 				var aktBlk = 0;
 
 
 				$('div.textblockNews a.txtBlkNewMove').css('display','block');
 
 				$('div.textblockNews a.newsPrev').click(function(){
 					$akt = $('DIV.textblock DIV.textblockNews DIV.newsEintrag').eq(aktBlk);
 
 					aktBlk--;
 					if(aktBlk < 0) aktBlk = maxBlk-1;
 
 					$next = $('DIV.textblock DIV.textblockNews DIV.newsEintrag').eq(aktBlk);
 
 					var posAkt = '-101%';
 					var posNext = '101%';
 
 					$akt.animate({'left':posNext},550,function(){
 						$akt.removeClass('aktNews')
 							.css('left','101%');
 					});
 
 					$next.css('left',posAkt);
 					$next.animate({'left':0},550,function(){
 						$next.addClass('aktNews')
 							 .css('left','0');
 					});
 				});
 
 				$('div.textblockNews a.newsNext').click(function(){
 					$akt = $('DIV.textblock DIV.textblockNews DIV.newsEintrag').eq(aktBlk);
 
 					aktBlk++;
 					if(aktBlk >= maxBlk) aktBlk = 0;
 
 					$next = $('DIV.textblock DIV.textblockNews DIV.newsEintrag').eq(aktBlk);
 
 					var posAkt = '101%';
 					var posNext = '-101%';
 
 					$akt.animate({'left':posNext},550,function(){
 						$akt.removeClass('aktNews')
 							.css('left','101%');
 					});
 
 					$next.css('left',posAkt);
 					$next.animate({'left':0},550,function(){
 						$next.addClass('aktNews')
 							 .css('left','0');
 					});
 				});
 			}
 		}

    function checkTxtPos(){
 		var orgImgH = 450;
 
 		$('DIV#img_image DIV.imgEintrag').each(function(){
 			var $cont = $(this);
 
 			var aktH = $cont.height() - 20;
 
 			$imgTxt = $cont.find('div.imgText');
 			if($imgTxt.length){
 				if(aktH < orgImgH){
 					var aktPTop = Number($imgTxt.css('top').replace('px',''));
 
 					var perc = 100 / orgImgH * aktH;
 
 					var newMarg = aktPTop * (100 - perc) / 100 *(-1);
 
 					$imgTxt.css('margin-top',newMarg + 'px');
 				}
 				else{
 					$imgTxt.css('margin-top','');
 				};
 			}
 
 		});
 
 
 
 	}
 
 
 	$(document).ready(function(){
 		if (!$('body').hasClass('tmpl_5'))
