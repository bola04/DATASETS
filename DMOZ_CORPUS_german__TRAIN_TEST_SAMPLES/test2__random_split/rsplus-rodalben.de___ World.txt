            var mess="Aktuelles"         // Dein Text
 
 var neonz="#003399"                        //      Grundfarbe der Schrift
 var neony="#00CCFF"                         //      Wechselfarbe 1
 var neonx="#99CCFF"                           //      Wechselfarbe 2
 var tempo=180				     //      Geschwindigkeit des Farbwechsels; kleinere Zahl = schneller
 var letters1=2			                 //      Anzahl der Buchstaben in der Wechselfarbe 1

 //    Ab hier nichts mehr ändern !!! ////////////////////////////////////////////////////////////////////////////////////////////////////////
 
 var n=0
 if (document.all||document.getElementById){
 document.write('<font color="'+neonz+'">')
 for (m=0;m<mess.length;m++)
 document.write('<span id="neonlight'+m+'">'+mess.charAt(m)+'</span>')
 document.write('</font>')
 }
 else
 document.write(mess)
 
 function crossref(number){
 var crossobj=document.all? eval("document.all.neonlight"+number) : document.getElementById("neonlight"+number)
 return crossobj
 }
 
 function neon(){
 
 //Change all letters to base color
 if (n==0){
 for (m=0;m<mess.length;m++)
 crossref(m).style.color=neonz
 }
 
 //cycle through and change individual letters to neon color
 crossref(n).style.color=neony
 
 if (n>letters1-1) crossref(n-letters1).style.color=neonx 
 if (n>(letters1+letters2)-1) crossref(n-letters1-letters2).style.color=neonz
 
 if (n<mess.length-1)
 n++
 else{
 n=0
 clearInterval(flashing)
 setTimeout("start()",pause)
 return
 }
 }
 function start(){
 if (document.all||document.getElementById)
 flashing=setInterval("neon()",tempo)
 }
 start()
