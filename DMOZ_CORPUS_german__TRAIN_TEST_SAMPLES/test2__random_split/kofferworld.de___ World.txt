 			 * Load super attsibutes

 			 var SUPER = [];
 
 			 
 			jQuery.each(attributes, function(index, attribute)
 			{
 				if (jQuery(attribute).is('select'))
 				{
 					var name = jQuery(attribute).attr('name'), id = name.substring(name.indexOf('[') + 1, name.lastIndexOf(']'));
 
 					var option = jQuery(attribute).find('option:selected');
 
 					if (0 < parseInt(option.val()))
 					{
 						variants.push(
 						{
 							id: 	id,
 							text: 	option.text()
 						});
 					}
 				}
 			});
 
 			if (attributes.length == variants.length)
 			{
 				for (i = 0, l = variants.length; i < l; i++)
 				{
 					for (a = 0, b = SUPER.length; a < b; a++)
 					{
 						if (SUPER[a].id == variants[i].id)
 						{
 							variant.push(
 							[
 								SUPER[a].label,
 								variants[i].text
 							].join(':'));
 						}
 					}
 				}
 			}
 
 			if (!variant.length)
 			{
 				/**
 				 * Invoke original click event(s)
 				 */

 					 * Track time

 				}

 				{
 					document.location = element.parent().attr('href');
 				}
 				else 
 				{
 					return true;
 				}
 	     	}
 		};
 
 		/**
 		 * Track event
 		 */

 		 * Track time 
 		 */
 		AEC.Time.track(dataLayer, AEC.Const.TIMING_CATEGORY_ADD_TO_CART, element.data('name'), element.data('category'));

 		 * Facebook Pixel Tracking

 	/**
 	 * User timing

 	AEC.Time = (function()
 	{
 		var T = 
 		{
 			event: 			'trackTime',
 			timingCategory:	'',
 			timingVar:		'',
 			timingValue:	-1,
 			timingLabel:	''
 		};
 
 		var time = new Date().getTime();
 		
 		return {
 			track: function(dataLayer, category, variable, label)
 			{
 				T.timingValue = (new Date().getTime()) - time;
 				
 				if (category)
 				{
 					T.timingCategory = category;
 				}
 
 				if (variable)
 				{
 					T.timingVar = variable;
 				}
 
 				if (label)
 				{
 					T.timingLabel = label;
 				}
 				
 				/**
 				 * Track time
 				 */
 				dataLayer.push(T);
 			},
 			trackContinue: function(dataLayer, category, variable, label)
 			{
 				this.track(dataLayer, category, variable, label);
 
 				/**
 				 * Reset time
 				 */
 				time = new Date().getTime();
 			}
 		}
 	})();

         <h2>
          Welcher Koffer passt zu mir?
          <a href="file:///C:/Users/anna.schoonhofen/Downloads/Startseitentext_Kofferworld_CCO_v3.docx#_msocom_2" id="_anchor_2" name="_msoanchor_2">
           <br/>
          </a>
         </h2>
         <p>
          Die Entscheidung für das passende Gepäck ist von vielen Faktoren abhängig. Zunächst einmal sollten Sie wissen,
          <strong>
           wofür Sie einen neuen Koffer oder eine neue Tasche benötigen
          </strong>
          .
         </p>
         <ul>
          <li>
           <strong>
            Freizeit / Reisen:
           </strong>
           Ein
           <a href="https://www.kofferworld.de/reisegepack/koffer.html">
            Reisekoffer
           </a>
           sollte groß genug sein, um alles Wichtige sicher verstauen zu können. Gleichzeitig darf er nicht zu schwer oder unhandlich sein, da er häufig angehoben wird. Gerade bei großen Gepäckstücken
           <strong>
            sorgen vier statt zwei Rollen für zusätzlichen Komfort.
           </strong>
           Außerdem muss das Kofferset stabil sein, um auch bei widrigen Bedingungen zu bestehen.
          </li>
          <li>
           <strong>
            Business:
           </strong>
           Im Gegensatz dazu, sollte Businessgepäck in erster Linie robust, handlich und übersichtlich sein. Zudem sollten kleinere Gepäckstücke von allen Fluggesellschaften auch als Handgepäck akzeptiert werden. Zum Businessgepäck gehören aber natürlich vor allem auch
           <a href="https://www.kofferworld.de/business/aktentaschen.html">
            Aktentaschen
           </a>
           , Aktenkoffer und Messenger-Bags. Hier finden wichtige Dokumente, Laptops und Co. nicht nur Platz, sie werden gleichzeitig sicher verstaut und sind direkt griffbereit. Zumal die hochwertigen Taschen auch optisch einen guten Eindruck beim Geschäftspartner hinterlassen.
          </li>
         </ul>
         <p>
          Natürlich spielt nicht zuletzt das Budget und damit verbunden das bevorzugte Material der Koffer eine Rolle. Produkte aus echtem Leder, Aluminium oder Polycarbonat sind bei Kofferworld ebenso erhältlich wie preisgünstige Taschen aus Textil. Weiterhin gibt es noch Koffer aus Curv, Nylon, Polyester und Polypropylen. So bietet jedes Material seine eigenen Vorzüge:
         </p>
         <ul>
          <li>
           Leder: edle Optik, Unikat-Faktor, langlebig
          </li>
          <li>
           Textil: Stabil, langlebig, leicht zu reinigen
          </li>
          <li>
           Alu: Sehr stabil, sehr bruchfest, massives Material, steifes Gehäuse
          </li>
          <li>
           Polycarbonat: Sehr stabil &amp; extrem leicht, witterungsbeständig
          </li>
          <li>
           Curv: Extrem leicht &amp; schlagfest, witterungsbeständig, (nahezu) unkaputtbar
          </li>
          <li>
           Nylon: Extrem leicht, nahezu reißfester Stoff, strapazierfähig, wasserabweisend
          </li>
          <li>
           Polyester: Geringes Gewicht, für hohe Temperaturen geeignet, hohe Schlagfestigkeit
          </li>
          <li>
           Polypropylen: Geringes Gewicht, sehr schlagfest, sehr bruchsicher
          </li>
         </ul>
         <p>
          Anbei noch einmal einige grundlegende Fragen, die Ihnen beim Kauf des passenden Gepäcks die Entscheidung erleichtern:
         </p>
         <ul>
          <li>
           Was ist der Verwendungszweck des Gepäcks? Beruf? Freizeit? Reise?
          </li>
          <li>
           Welche Materialien bevorzugen Sie? Elegantes Leder? Leichtes Textil? Unverwüstliche Hartschale?
          </li>
          <li>
           Welche Größe ist erforderlich?
          </li>
          <li>
           Soll das Gepäckstück Rollen haben? Wenn ja, wie viele Rollen?
          </li>
          <li>
           Welches Design spricht Sie an? Elegant? Schlicht? Oder modern?
          </li>
          <li>
           Wie viel Geld möchten Sie für das Gepäckstück ausgeben?
          </li>
         </ul>

         <h3>
          Was unterscheidet die Koffer voneinander?
          <a href="file:///C:/Users/anna.schoonhofen/Downloads/Startseitentext_Kofferworld_CCO_v3.docx#_msocom_5" id="_anchor_5" name="_msoanchor_5">
          </a>
         </h3>
         <p>
          Wenn Sie eine Reise planen, finden Sie bei Kofferworld bestimmt das passende Produkt:
         </p>
         <ul>
          <li>
           <strong>
            Wochenendtrip:
           </strong>
           Leichte Trolleys und größere Rucksäcke sind ideal für einen kurzen Trip mit zwei bis vier Übernachtungen geeignet. Wenn Sie
           <a href="https://www.kofferworld.de/reisegepack/handgepack.html">
            Handgepäck
           </a>
           mit in den Flieger nehmen wollen, muss das Gepäckstück bestimmte Maße einhalten. Laut
           <a href="https://www.kofferworld.de/iata/">
            IATA
           </a>
           (International Air Transport Association) darf Handgepäck nur 55cm x 40cm x 20cm messen. Allerdings müssen hier immer die individuellen Richtlinien der Fluggesellschaft beachtet werden!
          </li>
          <li>
           <strong>
            Wochentrip:
           </strong>
           Die handlichen Trolleys in den Handgepäckmaßen stoßen ab etwa fünf bis sieben Tagen Urlaubszeit an ihre Grenzen. Für mehr Platz sorgen Koffer in „mittlerer“ Größe, was einem Fassungsvermögen von rund 20 Kilogramm und einer Höhe von 60 bis 69 Zentimeter entspricht. Hier finden bequem zwei Paar Schuhe, Kleidung für ein bis zwei Wochen und weiteres Gepäck wie etwa die Kulturtasche Platz.
           <strong>
           </strong>
          </li>
          <li>
           <strong>
