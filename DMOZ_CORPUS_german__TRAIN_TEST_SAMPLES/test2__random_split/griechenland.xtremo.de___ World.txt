     Vue.component('alert', {
         props: {
             type: {
                 type: String,
                 default: '',
                 required: true
             },
             dismissable: {
                 type: String,
                 default: '',
                 required: false
             },
             autoclose: {
                 type: String,
                 default: '',
                 required: false
             },
             placement: {
                 type: String,
                 default: '',
                 required: false
             }
         },
         data: function() {
             return {
                 visible: true
             };
         },
         computed: {
             compDuration: function () {
                 if (this.autoclose == '') {
                     return 8000;
                 }
                 return Number(this.autoclose);
             },
             compDismissable: function () {
                 if (this.dismissable == '') {
                     return true;
                 }
                 return false
             }
         },
         mounted: function () {
             this.$nextTick(function () {
                 if (this._timeout) clearTimeout(this._timeout)
                 if (this.compDuration > 0) {
                     var that = this;
                     this._timeout = setTimeout(function() {
                         that.visible = false;
                     }, this.compDuration)
                 }
             })
         },
         template: '#alert-template'
     })
 
     Vue.component('searchandcontactbox', {
         data: function () {
             return {
                 searchstr: null,
                 searching: false,
                 error: null,
                 reisebueros: [],
                 currentPage: 0,
                 itemsPerPage: 10,
                 resultCount: 0,
                 contact: '',
                 firstname: null,
                 lastname: null,
                 email: null,
                 message: null,
                 messageerror: null,
                 messagesuccess: null,
                 standort: 0,
                 lat: null,
                 lon: null
             }
         },
         computed: {
             totalPages: function () {
                 return Math.ceil(this.resultCount / this.itemsPerPage);
             },
             filteredReisebueros: function () {
                 this.resultCount = this.reisebueros.length;
                 if (this.currentPage >= this.totalPages) {
                     this.currentPage = Math.max(0, this.totalPages - 1);
                 }
                 var index = this.currentPage * this.itemsPerPage;
                 return this.reisebueros.slice(index, index + this.itemsPerPage);
             }
         },
         methods: {
             'doSearchbox': function () {
                 this.error = null;
             },
             'doContactbox': function () {
                 this.messageerror = null;
                 this.messagesuccess = null;
 
                 $('#contactform').parsley().reset();
             },
             'searchReisebueros': function (geolocation) {
                 this.error = null;
 
                 this.reisebueros = [];
                 var self = this;
 
                 this.searching = true;
 
                 if (geolocation == true) {
                     navigator.geolocation.getCurrentPosition(function(pos){
                         self.lat = pos.coords.latitude;
                         self.lon = pos.coords.longitude;
                         self.standort = 1;
 
                         axios.get('/api/_findreisebueros?standort=' + self.standort + '&lat=' + self.lat + '&lon=' + self.lon)
                             .then(function (response) {
                                 self.searching = false;
                                 self.reisebueros = response.data.reisebueros;
                                 self.error = response.data.error;
                             });
                     }, function () {
                         self.searching = false;
                         self.error = ' Es konnte kein Standort ermittelt werden.';
                     })
                 } else {
                     this.searching = true;
                     axios.get('/api/_findreisebueros?q=' + encodeURIComponent(this.searchstr))
                         .then(function (response) {
                             self.searching = false;
                             self.reisebueros = response.data.reisebueros;
                             self.error = response.data.error;
                         });
                 }
             },
             'sendMessage': function () {
                 if ( $('#contactform').parsley().validate() ) {
                     var self = this;
                     axios.post('/api/_message', {
                         'firstname': this.firstname,
                         'lastname': this.lastname,
                         'email': this.email,
                         'nachricht': this.message,
                         'contact': this.contact
                     }).then(function (response) {
                         self.messageerror = response.data.error;
                         self.messagesuccess = response.data.success;
 
                         if (response.data.success) {
                             self.message = null;
                             self.contact = '';
                         }
                     });
                 }
             },
             setPage: function (pageNumber) {
                 this.currentPage = pageNumber;
             }
         }
     })
