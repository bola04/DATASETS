 		function suggest() {
 			var pars = $("#txtSearch").val();
 			var oscs = $("#searchinput input[name=osCsid]").val();
 	
 			if(pars.length >= 3) {
 	
 				$.ajax({ url: 'https://schneidermusik.de/home/searchsuggest.php',
 					type: 'get',
 					data: ({ 'search': pars,
 							 'osCsid': oscs }),
 					success: function(code) {
 						var ss = $("#searchSuggest");
 	
 						ss.html = '';
 						var str = code.split("\n");
 	
 						for(i=0; i < str.length - 1; i++) {
 							var suggest  = '<div onmouseover="javascript:suggestOver(this);" ';
 								suggest += 'onmouseout="javascript:suggestOut(this);" ';
 								suggest += 'onclick="javascript:setSearch(this.innerHTML);" ';
 								suggest += 'class="suggest_link">' + str[i] + '<\/div>';
 								ss.html += suggest;
 						}
 						if (i==0 || str.length==1 ) {
 							$("#searchSuggest").fadeOut('500');
 						} else if (str.length >1) {
 							$("#searchSuggest").fadeIn('500');
 	
 							ss.html += '<div id="searchSuggestClose"><a onclick="javascript:suggestClose(this);">x<\/a><\/div>';
 							$("#searchSuggest").html(ss.html);
 						}
 					 },
 					 error: function(){ /* alert('Something went wrong...') */ }
 				});
 			} else {
 				$("#searchSuggest").fadeOut('500');
 			}
 		}
 	
 		function suggestClose (div_value) {
 			$("#searchSuggest").fadeOut('500');
 		}
 	
 		function suggestOver(div_value) {
 			div_value.className = 'suggest_link_over';
