             var files= e.target.files;
             var output = [];
             if(files && files.length){
                 f = files[0];
 
                 if (Math.round(escape(f.size)/1000).toFixed(0)>5000)

                      return false;
                 }
                 else
                 {
 
                   $('#request_form_popup form.custom .attach-file').removeClass("not-attached");
                    return true;
                 }
             }
 
         },
         onSubmit: function(file, ext){
             $("#request_form_popup form.custom .attach-file-name").html("");

               if (! (ext && /^(doc|docx|txt|rtf|pdf|zip|rar)$/.test(ext))){
                 $("#request_form_popup .attach-file-name").html("<span style='color:red'>Erlaubte Dateiformate - .doc, .docx, .txt, .rtf, .pdf, .zip, .rar</span>");
                 $('#request_form_popup #filesize').attr("datanewfile","");
                 return false;
             }
 
             $('#request_form_popup .attach-file').removeClass("not-attached");
             $("#request_form_popup .preloder-holder").show();
         },

             $(".custom-sm-submit button").prop("disabled",false);

                 $("#request_form_popup .attach-file-name").html("<span style='color:red'>Dateigröße sollte 5 Mb nicht überschreiten</span>");

             else
             {
                 $('#request_form_popup #filesize').attr("datanewfile",response);
             }
         }
     });
