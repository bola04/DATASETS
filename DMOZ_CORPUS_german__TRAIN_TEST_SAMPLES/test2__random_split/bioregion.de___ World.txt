   // debounce so filtering doesn't happen every millisecond
 function debounce( fn, threshold ) {
   var timeout;
   return function debounced() {
     if ( timeout ) {
       clearTimeout( timeout );

     function delayed() {
       fn();
       timeout = null;
     }
     timeout = setTimeout( delayed, threshold || 100 );
   }
