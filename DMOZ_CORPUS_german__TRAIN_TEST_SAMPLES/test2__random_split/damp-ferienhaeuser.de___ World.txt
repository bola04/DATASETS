 		function MOZfindVerticalOffset(id,height,zoomed_height) {
 		    var node = document.getElementById(id);     
 		    var curtop = 0;
 		    var curtopscroll = 0;
 			var scroll_top = MOZscrollTop();
 			var scroll_height = MOZclientHeight();
 			
 			
 			
 			
 		    if (node.offsetParent) {
 		        do {
 		            curtop += node.offsetTop;
 		            curtopscroll =0;//+= node.offsetParent ? node.offsetParent.scrollTop : 0;
 		        } while (node = node.offsetParent);
 
 				var imaged_y=(curtop - curtopscroll)-scroll_top;
 				var zoomed_y=imaged_y-((zoomed_height/2)-height/2);
 			
 				var VerticalOffset=0;
 				
 				if(zoomed_height>scroll_height)
 				{
 					VerticalOffset=-zoomed_y-(zoomed_height-scroll_height)/2;
 					
 				}
 				else
 				{
 					if(zoomed_y<0)
 						VerticalOffset=-zoomed_y;
 					
 					if(zoomed_y+zoomed_height>scroll_height)
 						VerticalOffset=(scroll_height)-(zoomed_y+zoomed_height);
 				}
 
 		    }
 			return VerticalOffset;
 		}
 		
 		function MOZfindHorizontalOffset(id,width,zoomed_width) {
 		    var node = document.getElementById(id);     
 		    var curleft = 0;
 		    var curleftscroll = 0;

 				var HorizontalOffset=0;
 				
 				if(zoomed_width>scroll_width)

          <h1 style="text-align: left;">
           <span style="font-size: x-large; color: #006699;">
            Ferienhäuser im Ostsee Resort Damp
            <span style="color: #ffffff;">
             Ferienhaus
            </span>
            <br/>
           </span>
          </h1>
          <h2 style="text-align: center;">
           <span style="font-size: x-large; color: #ffffff;">
            <span style="font-size: medium;">
             <span style="color: #006699;">
              Freizeit, Sport, Erholung, Gesundheit
              <span style="color: #ffffff;">
               Ferienhaus Ostsee Ferienhäuser Damp
              </span>
              <br/>
             </span>
            </span>
           </span>
          </h2>
          <h3 style="text-align: center;">
           <span style="font-size: x-large; color: #ffffff;">
            <span style="font-size: medium;">
             <span style="color: #006699;">
              <span style="color: #ffffff;">
               Ferienhaus  Ferienhäuser Damp Ostsee
              </span>
              <br/>
             </span>
            </span>
           </span>
          </h3>
          <p>
           <strong>
            Zeit zum Erholen, zum Erleben, zum Auftanken.
           </strong>
          </p>
          <p>
           <br/>
           Unsere Ferienhäuser von privater Hand sind individuell ausgestattet. Damit verfügen Gäste über einen wunderbaren Ausgangspunkt, um die schönste Zeit des Jahres entspannt zu verbringen - und das in unmittelbarer Nähe zum Strand (ohne Kurtaxe!) und zu den Einrichtungen des
           <em>
            <a href="/in-damp">
             Ostsee Resort Damp
            </a>
           </em>
           !
          </p>
          <p>
           Die Lage der Häuser, direkt in Damp, in unmittelbarer Nähe zum Strand und
           <a href="/in-der-umgebung">
            Mitten in Schwansen
           </a>
           , bietet Ihnen den idealen Ausgangspunkt für einen erlebnisreichen Urlaub. Durch das große Angebot an Ausflugsmöglichkeiten hier in Schwansen und Umgebung sowie den Sport-, Wellness- und Gesundheitseinrichtungen des Ostsee-Resorts, können Sie Ihren Urlaub abwechslungsreich, spannend und erholsam gestalten.
          </p>
          <p>
           Ein ausgedehntes Rad- Wanderwegnetz läd zum naturnahen Erkunden der Küste und des Ostseefjord Schlei ein. Natur Pur, ist hier die Devise. Und wer das Meer in vollen Zügen genießen möchte, der nutzt den kurtaxfreien, gepflegten
           <a href="/in-damp">
            Sandstrand
           </a>
           und läßt sich ein laues Lüftchen um die Ohren wehen.
          </p>
          <p>
           Die Promenade rund um den Yachthafen läd zum flanieren und verweilen ein. Die verschiedenen Gastronomieangebote lassen keine Wünsche offen, es ist für jeden Geschmack etwas dabei.
          </p>
          <p>
           Hier finden Sie die Zeit zum Erholen, zum Erleben, zum Auftanken.
          </p>
          <p>
           <a href="/profil">
            Unser Büro
           </a>
           ist täglich für Sie geöffnet. Es befindet sich ganz zentral in der "Ostsee-Residenz". So haben Sie stets einen Ansprechpartner vor Ort und können Ihren Urlaub wirklich genießen!
          </p>
