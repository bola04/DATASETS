        $(document).ready(function() {
 		step = 2
 		dur = 9
 		$(".calleft").hover(function() {
 			timerId = setInterval(function() {
 				pos = parseInt($('.calslide').css('left')) + step;
 				if (pos < step) {
 					$('.calslide').css('left', pos + 'px');
 				}
 			}, dur);
 
 		}, function() {
 			clearInterval(timerId);
 		});
 		$(".calright").hover(function() {
 			timerId = setInterval(function() {
 				pos = parseInt($('.calslide').css('left')) - step;
 				$('.calslide').css('width');
 				$('.calslide').css('left', pos + 'px');
 			}, dur);
 
 		}, function() {
 			clearInterval(timerId);
 		});
 
 		$('.date').mouseover(function() {
 			var uid = $(this).attr('id').replace('d', 'f');
 			$('.calfly').hide();
 			left =$(this).position().left;
 			left = left+parseInt($('.calslide').css('left'));
 			$('.' + uid).css('left',left+'px');
 			$('.' + uid).show();
 			$('.' + uid).delay(10000).hide(500); // fade out after 10 seconds
 		})
 		
 		$('.calflywrap').mouseleave(function() {
             $('.calfly').hide();
         })
 		
 	})
