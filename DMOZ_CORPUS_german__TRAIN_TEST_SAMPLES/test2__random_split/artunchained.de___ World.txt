 function switchTabs(clickedTab, fadeIt=true) {
 	var balken = ["beitrag", "kontakt", "weitere-beitraege", "kommentare", "hauptmenue" ];
 	var currentBarClass = clickedTab.attr("class");

 		$j(".header-balken img").attr("src", clickedTab.find(".balken-button-wide").attr("src")).removeAttr('style');
 		var leftPos = parseInt($j(".header-balken img").css("left").replace("px", ""))-clickedTab.find(".balken-button-wide").width()+35;

 				left: leftPos
 
 		});

 	}

 	/* Move bars to the left or right */
 	// Drop zone is a helper to sort things right
 	$j(".balken-rechts").find(".drop-zone").prependTo($j(".balken-rechts"));

 		$j(".balken-links").find("."+balken[index]).show();

 var timer, playing = false;
 function fadeGallery(direction="play") {
 	var balken = {"header-balken": "beitrag-wrap", 
 				  "header-balken-pos-0": "beitrag-wrap", 
 				  "header-balken-pos-1": "kontakt", 
 				  "header-balken-pos-2": "weitere-beitraege", 
 				  "header-balken-pos-3": "kommentare", 
 				  "header-balken-pos-4": "hauptmenue" };
 				  
 	var currentClass = balken[$j(".header-balken").attr('class').split(' ').pop()];
 	
 	//No images on current tab? Fallback to post images
 	layerClassFormated = " ." + currentClass + "-bilder ";
 	var letterBox = $j(".hintergrund-bilder"+layerClassFormated+".hg-letterbox");
 	if (letterBox.length <= 0) {
 		letterBox = $j(".hintergrund-bilder .beitrag-wrap-bilder .hg-letterbox");
 		currentClass = "beitrag-wrap";
 	}
 	
 	
 	switch(direction) {
 		case "play":
 			nextBGImage(currentClass);
 			playBGImage(currentClass);
 			break;
 			
 		case "stop":
 			clearTimeout(timer);
 			$j(".play").removeClass("play-active");
 			playing = false;
 			break;
 			
 		case "previous":
 			fadeGallery("stop");
 			previousBGImage(currentClass);
 			break;
 			
 		case "next":
 			fadeGallery("stop");
 			nextBGImage(currentClass);
 			break;
 	}
 }
 
 function greyedOutButtons(targetLayerClass=".beitrag-wrap-bilder") {
 	// Exchange Buttons
 	if ($j(targetLayerClass + " img").length > 1) {
 		$j(".play, .next, .previous").removeClass("greyed-out");
 	}
 	else {
 		$j(".play, .next, .previous").addClass("greyed-out");
 	}
 	//console.log(targetLayerClass + " img", $j(targetLayerClass + " img").length);
 }
 
 var bgImagesInProgress = false;
 function fadeToLayer(targetLayerClass) {
 	if (bgImagesInProgress) {
 			setTimeout(function() {
 				fadeToLayer(targetLayerClass);
 			}, 500);
 			return false;
 	}
 
 	var sourceLayerClass = " ." + $j(".hintergrund-bilder div:visible").attr("class");
 	if (targetLayerClass == "beitrag") targetLayerClass += "-wrap";
 	targetLayerClass = " ." + targetLayerClass + "-bilder";
 	
 	//If there're no images in the target layer, use posts layer as default
 	if ($j(targetLayerClass + " img").length <= 0) targetLayerClass = " .beitrag-wrap-bilder";
 	
 	if (sourceLayerClass != targetLayerClass) {
 		bgImagesInProgress = true;
 		fadeGallery("stop");
 		greyedOutButtons(targetLayerClass);
 		
 		$j(sourceLayerClass).addClass("top-hg-bild");
 		$j(targetLayerClass).addClass("fade-ziel-hg-bild").show();
 		$j(targetLayerClass).find(".hg-letterbox").first().addClass("top-hg-bild");
 		$j(sourceLayerClass).fadeOut(1000, function() {
 			$j(sourceLayerClass).removeClass("top-hg-bild");
 			$j(sourceLayerClass).hide().find(".top-hg-bild").removeClass("top-hg-bild").find("img").removeClass("zoom-bild");
 			$j(targetLayerClass).removeClass("fade-ziel-hg-bild");
 			$j(sourceLayerClass + " .hg-letterbox").each(function() {
 				$j(this).removeAttr("style");
 			});
 			bgImagesInProgress = false;
 		});
 	}
 	else {
 		bgImagesInProgress = false;
 	}
 }
 
 function playBGImage(layerClass="beitrag-wrap") {
 	layerClassFormated = " ." + layerClass + "-bilder ";
 	var letterBox = $j(".hintergrund-bilder"+layerClassFormated+".hg-letterbox");
 
 	if (letterBox.length > 1) {
 		timer = setTimeout(function() {
 			nextBGImage(layerClass);
 			playBGImage(layerClass);
 		}, 8000);
 		playing = true;
 		$j(".play").attr("class", "play play-active");
 	}
 }
 
 function nextBGImage(layerClass="beitrag-wrap") {
 	if (bgImagesInProgress) {
 			//console.log("next prevented");
 			return false;
 	}
 	layerClass = " ." + layerClass + "-bilder ";
 	var letterBox = $j(".hintergrund-bilder"+layerClass+".hg-letterbox"),	
 		topHGBild = $j(".hintergrund-bilder"+layerClass+".top-hg-bild");
 	if (letterBox.length > 1) {
 		bgImagesInProgress = true;	
 			
 		letterBox.each(function() {
 			$j(this).removeAttr("style");
 		});
 
 		if (topHGBild.next().length) {
 			topHGBild.next().addClass("fade-ziel-hg-bild");	
 		}
 		else {
 			letterBox.first().addClass("fade-ziel-hg-bild");	
 		}
 		var fadeZielHGBild = $j(".hintergrund-bilder"+layerClass+".fade-ziel-hg-bild");
 		topHGBild.fadeOut(1000, function() {
 			topHGBild.removeClass("top-hg-bild").find("img").removeClass("zoom-bild");
 			fadeZielHGBild.addClass("top-hg-bild").removeClass("fade-ziel-hg-bild").find("img").addClass(
 			"zoom-bild");	
 			bgImagesInProgress = false;
 		});
 	}
 	else bgImagesInProgress = false;
 }
 
 function previousBGImage(layerClass="beitrag-wrap") {
 	if (bgImagesInProgress) {
 			//console.log("next prevented");
 			return false;
 	}
 	var letterBox = $j(".hintergrund-bilder"+layerClass+".hg-letterbox "),	
 		topHGBild = $j(".hintergrund-bilder"+layerClass+".top-hg-bild");
 	if (letterBox.length > 1) {
 		bgImagesInProgress = true;
 		layerClass = " ." + layerClass + "-bilder";
 		
 		letterBox.each(function() {
 			$j(this).removeAttr("style");
 		});
 		if (topHGBild.prev().length) {
 			topHGBild.prev().addClass("fade-ziel-hg-bild");	
 		}
 		else {
 			letterBox.last().addClass("fade-ziel-hg-bild");	
 		}

 			topHGBild.removeClass("top-hg-bild").show().find("img").removeClass("zoom-bild");

 			bgImagesInProgress = false;			
 		});
 	}	
 	else bgImagesInProgress = false;
 }
 
 function toogleFullscreen() {
 	if ($j(".fixed-header").hasClass("fullscreen-mode")) {
 		fullscreenOff();

 	else {
 		 fullscreenOn();
 	}	
 }
 
 function fullscreenOff() {
 	if (!fullscreenInProgress) {
 		if ($j(".fixed-header").hasClass("fullscreen-mode")) {
 			$j(".hintergrund-bilder").removeClass("hg-fullscreen");
 			fullscreenInProgress = true;
 			$j(".fixed-header").css({
 				backgroundColor: "#000"
 			}).removeClass("fullscreen-mode");
 			$j(".balken-wrap").animate({
 				height: "100%"
 			});
 			$j(".weitere-beitraege .balken-button").css({
 				height: "auto"
 			});
 			$j(".beitrag-wrap").fadeIn(1000, function(){
 				fullscreenInProgress = false;
 			});
 		}
 	}
 }
 
 var fullscreenInProgress = false;
 function fullscreenOn() {
 	if (!fullscreenInProgress) {
 		// Switch to first tab on  Mobile
 		if ($j(".responsive-helper").css("display") == "block") {
 			switchTabs($j(".beitrag"), false);
 		}
 		if (!$j(".fixed-header").hasClass("fullscreen-mode")) {
 			fullscreenInProgress = true;
 			$j(".hintergrund-bilder").addClass("hg-fullscreen");
 			$j(".beitrag-wrap").fadeOut(500, function() {
 				$j(".fixed-header").css({
 					backgroundColor: "transparent"
 				}).addClass("fullscreen-mode");
 			});
 			
 			$j(".balken-wrap").animate({
 				height: 45
 			});
 			
 			$j(".weitere-beitraege .balken-button").animate({
 				height: 43
 			});
 			
 			// Mobile
 			if ($j(".responsive-helper-3").css("display") == "block") {
 				$j(".header-balken").css("width", "730px");
 				$j(".header-balken").animate({
 					width: 225
 				}, 500, function() {
 					$j(".header-balken").removeAttr("style");
 					// Strange delay of the above
 					setTimeout(function() {
 						fullscreenInProgress = false
 					}, 1000);
 				});
 			}
 			else {
 				fullscreenInProgress = false
 			}
 
 		}		
 	}
 }

 /* Trigger hover on mobile */
 function mobileScrollHover(scrollElement, hoverClass) {
 	var hT = scrollElement.offset().top,
 		hH = scrollElement.outerHeight(),
 		wH = $j(window).height(),
 		wS = $j(this).scrollTop();
 	if (wS > (hT+hH-wH)){
 		scrollElement.addClass(hoverClass);
 	}
 	else {
 		scrollElement.removeClass(hoverClass);
 	}
 }

 	var forms = document.getElementsByTagName( 'form' );
 	var pattern = /(^|\s)g-recaptcha(\s|$)/;
 
 	for ( var i = 0; i < forms.length; i++ ) {
 		var divs = forms[ i ].getElementsByTagName( 'div' );
 
 		for ( var j = 0; j < divs.length; j++ ) {
 			var sitekey = divs[ j ].getAttribute( 'data-sitekey' );
 
 			if ( divs[ j ].className && divs[ j ].className.match( pattern ) && sitekey ) {
 				var params = {
 					'sitekey': sitekey,
 					'type': divs[ j ].getAttribute( 'data-type' ),
 					'size': divs[ j ].getAttribute( 'data-size' ),
 					'theme': divs[ j ].getAttribute( 'data-theme' ),
 					'badge': divs[ j ].getAttribute( 'data-badge' ),
 					'tabindex': divs[ j ].getAttribute( 'data-tabindex' )
 				};
 
 				var callback = divs[ j ].getAttribute( 'data-callback' );
 
 				if ( callback && 'function' == typeof window[ callback ] ) {
 					params[ 'callback' ] = window[ callback ];
 				}
 
 				var expired_callback = divs[ j ].getAttribute( 'data-expired-callback' );
 
 				if ( expired_callback && 'function' == typeof window[ expired_callback ] ) {
 					params[ 'expired-callback' ] = window[ expired_callback ];
 				}
 
 				var widget_id = grecaptcha.render( divs[ j ], params );
 				recaptchaWidgets.push( widget_id );
 				break;
 			}
 		}
 	}
 };
 
 document.addEventListener( 'wpcf7submit', function( event ) {
 	switch ( event.detail.status ) {
 		case 'spam':
 		case 'mail_sent':
 		case 'mail_failed':
 			for ( var i = 0; i < recaptchaWidgets.length; i++ ) {
 				grecaptcha.reset( recaptchaWidgets[ i ] );
 			}
 	}
 }, false );
    </script>
   </div>
  </body>
