         //global
 	var fader1 = null;
 	
 	//Fader class
 	function Fader () {
 	
 		var inst = this;
 	
 		//wraps the main fading image
 		this.wrap;
 		this.index;
 		this.autoplay = true;
 		this.interval;
 		this.numberOfImages;
 		this.pause;
 	
 		this.init = function () {
 			this.inst = this;
 			this.pause = false;
 			this.interval = 1000;
 			this.index = 0;
 						
 			var imgCount = $(".fadeItems").children().size();
 			this.numberOfImages = imgCount;
 		}
 		
 		this.show = function (index) {
 			this.index = index;
 			
 			//fade out
 			this.wrap.fadeTo("fast", 0.5, function() {
 				// fade out Animation complete
 				var nextImg = $(".fadeItems img").get(index);
 				inst.load(nextImg.title, index);
 			});			   
 		}
 		
 		this.load = function (img_src, index) { 
 			//load image					
 			var url = img_src;			
 			var img = new Image();
 			img.onload = function() {
 				inst.wrap.fadeTo("fast", 1);
 				$("#image_wrap_frame").attr("src", url);
 			};
 			
 			img.src = url;
 				
 			this.highlightTabs(index);	
 			this.setHeadlines(index);
 		}
 		
 		this.setHeadlines = function (index) {
 			var faderListEntryTitle = $("#faderList .fader_title").get(index);
 			var faderListEntrySubtitle = $("#faderList .fader_subtitle").get(index);
 			var faderListEntryHREF = $("#faderList .fader_href").get(index);
 		
 			var faderH1 = $("#faderH1");
 			var faderH2 = $("#faderH2");
 			
 			//set headlines
 			
 			var related = $(faderListEntryTitle).attr("title");
 			$('#faderRelatedAlbum').css('display', 'none');
 			$('#faderRelatedVoting').css('display', 'none');
 			$('#faderRelatedAV').css('display', 'none');
 			if (related != null && related.length > 0) {
 				if (related.indexOf('album') > 0) {
 					$('#faderRelatedAlbum').css('display', 'block');
 				}
 				if (related.indexOf('voting') > 0) {
 					$('#faderRelatedVoting').css('display', 'block');
 				}
 				if (related.indexOf('av') > 0) {
 					$('#faderRelatedAV').css('display', 'block');
 				}
 			}
 			$(faderH1).text($(faderListEntryTitle).text() );  
 			$(faderH2).text($(faderListEntrySubtitle).text() ); 
 			
 			//set links
 			var link = $(faderListEntryHREF).text();
 			$(faderH1).attr("href", link);  
 			$(faderH2).attr("href", link); 
 			$("#lnk_image_wrap_frame").attr("href", link);
 		}
 		
 		this.highlightTabs = function (index) {
 			//for ie6 we do apply  inline style on current link in tab 
 			var currFadeLnk = $(".fade_lnk2").get(index);
 				//$("." + this.sliderTablLinkClass).css("background-color","transparent");
 				$(".fade_lnk2").removeClass("highlight");
 				//$(currFadeLnk).css("background-color", "#ffffff");	
 				$(currFadeLnk).addClass("highlight");	
 		}
 		
 		this.play = function () {
 			//autoplay
 			window.setInterval (function() {
 				if (inst.pause==true || inst.autoplay==false) return false;
 				
 				if (inst.index < inst.numberOfImages - 1){
 					inst.index = inst.index + 1;
 					 inst.show(inst.index );
 					
 				} else {
 					inst.index = 0;
 					inst.show(inst.index);
 				}
 			}, inst.interval);
 		}
 		
 		this.pause = function (){
 			this.pause = true;
 		}
 	}
 	
 	/*
 	* Set Events
 	*/
 
     $(document).ready(function() {
 		fader1 = new Fader();
 		fader1.wrap = $("#image_wrap");
