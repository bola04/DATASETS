  var slideshow = setInterval("changeAufmacher()", 5000); // der variablen slideshow wird ein 5 sekunden intervall mit der funktion "changeAufmacher" zugewiesen

  var html = ''; // der variable html wird ein leerer string wert zugewiesen
  for(var i = 0; i < $('div.opener div.opener_img img').length; i++)// solange i kleiner als die anzahl der bilder ist
  {
  html += i == 0 ? '<span class="icon_'+i+' open"></span>' : '<span class="icon_'+i+' closed"></span>'; /* wenn html = html + i gleich 0 ist dann beokmmt das span mit der class icon_open nach dem underscore die variable i eingesetzt, andernfalls bekommt das span mit der class icon_closed nach dem underscore die variable i eingesetzt */
  }
  $('div.gallery_switch').html(html); //die anzahl der icons wird durch die durchläufe der for-schleife festgesetzt
 
  $('div.gallery_switch span').click(function() { // wenn einer der kleinen böbbel angeklickt wird dann wird 
  clearInterval(slideshow); //slideshow wird unterbrochen
  var icon = $(this).attr('class'); // in der variable icon wird die aktuelle klasse von div.gallery_switch span gespeichert
  var arIcon = icon.split(' '); //in der variable aricon wird die klasse die in der icon variable gespeichert wurde am leerzeichen gesplitet: icon_ und open bzw closed
  var arIconClass = arIcon[0].split('_'); //in der variable ariconClass wird die erste hälfte von der zuvor geteilten klasse am underscore geteilt: icon und _ 
  $('div.opener div.opener_img img').css({opacity: 0.0}); // alle bilder werden durchsichtig gemacht
  $('div.opener div.opener_img img:nth-child('+(Number(arIconClass[1]) + 1) +')').css({opacity: 1.0}); //das ausgwählte bild wird gefunden und sichtbar gemacht
  $('div.gallery_switch span.open').removeClass('open').addClass('closed');//allen span class open entfernt man die klasse open und fügt die klasse closed hinzu
  $(this).removeClass('closed').addClass('open'); //allen span wird die klasse closed entfernt und die klasse open hinzugefügt
  });
  }
  });
 
  function changeAufmacher() { //die funktion changeAufmacher
  //images
  var $alt = $('div.opener div.opener_img img.obersteebene'); //der variablen $alt wird das sichtbare bild zugewiesen
 
  if($alt.next().length) // wenn es ein nächstes bild gibt
  {
  var $neu = $alt.next(); //der variablen $neu wird das darauffolgende bild zugewiesen
  }
  else
  {
  var $neu = $('div.opener div.opener_img img:first'); //der variablen $neu wird das erste bild zugewiesen
  }
 
  $alt.addClass('mittlereebene'); //dem sichtbaren bild wird die klasse mittlereebene hinzugefügt
  $alt.removeClass('obersteebene'); //dem sichtbaren bild wird die klasse obersteebene entfernt
  $neu.css({opacity: 0.0}); //sichtbarkeit vom neuen bild wird auf 0 gesetzt
  $neu.addClass('obersteebene'); //dem neuen bild wird die klasse obersteebene hinzugefügt
 
  $neu.animate({opacity: 1.0}, 2000, function() { //das neue bild wird animiert eingeblendet
  $alt.removeClass('mittlereebene'); //dem alten bild wird nun noch die klasse mittlereebene entfernt damit es ganz ausgeblendet ist
  });
 
  //icons
  var $icon_neu = $('div.gallery_switch span.open').next().length ? $('div.gallery_switch span.open').next() : $('div.gallery_switch span:first'); /*der variable $icon_neu wird zugewiesen: wenn es einen nächsten böbbel gibt dann wird dieser einfach ausgewählt. wenn nicht wird wieder zum ersten gesprungen*/
  $('div.gallery_switch span.open').removeClass('open').addClass('closed'); //dem alten icon wird die klasse open entfernt und die klasse closed hinzugefügt
  $icon_neu.removeClass('closed').addClass('open'); // dem neuen icon wird die klasse closed entfernt und die klasse open hinzugefügt
  }
