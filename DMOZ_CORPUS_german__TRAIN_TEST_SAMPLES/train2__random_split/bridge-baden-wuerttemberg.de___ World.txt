 function gentab(p,tabid,year) {
    jQuery.ajaxSetup({ cache: false });
    jQuery.ajax({
       url: "dir.php",
       dataType: "json",
       data: { "path": p.base },
       type: "GET",
       pass: { "p":p, "tabid":tabid, "year":year },
       success: function(json) { gentabs2(this.pass.p,this.pass.tabid,this.pass.year,json) },
      error: function() { gentabs2(this.pass.p,this.pass.tabid,this.pass.year,{}) }
    });
 
    function gentabs2(p,tabid,year,dirlst) {
       var head = [ [0,""], [0,""], [1,"Verteilungen"], [1,"Boardzettel"],
          [1,"Teamscores"] ]
       // p.imgsrc="wp-content/uploads/2017/05/icon-layout.gif"
       p.imgsrc="upload/icon-layout.gif"
 
       for ( var i=0; i<p.head2.length; i++ ) { head[i][1] = p.head2[i] }
       var div = jQuery("#"+tabid).html("").html(
          "<div class=gt_tab_pn style='display:none;'/>" +
          "<table class=gt_tab_p>" +
          "  <tr><td><a class=gt_tab_lnkr href=#><div class=gt_divb><table class=gt_tab_r/></div></a></td></tr>" +
         "  <tr><td><div class=gt_divb><table class=gt_tab_mb0 style='width:auto;'><tr><td><table class='gt_tab1 gt_tab_mb0' /></td></tr></table></div></td></tr>" +
          "  <tr><td><table class=gt_tab2/></td></tr>" +
          "</table>"
       )
       if ( div.length == 0 ) {
          alert("id="+tabid+" ist nicht definiert")
          return(false)
       }
 
       // Ergebnis abholen und bearbeiten
       var file = fcheck(dirlst,"Endergebnis","Zwischenstand","Zwischenergebnis","Ergebnis")
       if ( file ) {
          jQuery.ajax({
             url: p.base + "/" + file[1],
             dataType: "html",
             pass: { "p":p, "url":p.base + "/" + file[1] },
             type: "GET",
             beforeSend: function( xhr ) { xhr.overrideMimeType( "text/html; charset=iso8859-1" ) },
             success: function(data) {
                var self = this;
                //jQuery("#"+tabid).html( jQuery("#"+tabid+" .team").first().closest("table").html() )
                jQuery("#"+tabid+" .gt_tab_pn").html(data);
                jQuery("#"+tabid+" .gt_tab_r").html( jQuery("#"+tabid+" .gt_tab_pn table table").first().html());
                jQuery("#"+tabid+" .gt_tab_pn").remove()
                jQuery("#"+tabid+" .gt_tab_r .besetzung").parent().remove();
                if ( jQuery("#"+tabid+" .gt_tab_r .MsoNormal").closest("TR").length > 0 ) {
                   jQuery("#"+tabid+" .gt_tab_r tr").each( function (idx) {
                  if ( idx > 2 && ( idx%2 == 1 )) { jQuery(this).remove() }
                   })
                }
                jQuery("#"+tabid+" .gt_tab_r TH,#"+tabid+" .team").css("font-weight","bold");
                jQuery("#"+tabid+" .gt_tab_r *").attr("class","");
                jQuery("#"+tabid+" .gt_tab_lnkr").attr({href:self.pass.url , target:self.pass.url});
                jQuery("#"+tabid+" .gt_tab_r TH, #"+tabid+" .gt_tab_r TD").attr("align", function() {
                   jQuery(this).css("text-align",jQuery(this).attr("align"))
                });
 
             }
          });
       } else {
          jQuery(".gt_tab_r").closest("tr").remove()
       }
 
       //-----> Generiere die Links von head1
       var tab = jQuery(div).find(".gt_tab1")
       for ( var i=0; i<p.head1.length; i++ ) {
          var id = p.head1[i]
          if ( id.toLowerCase() == "ergebnis" ) {
             file = fcheck(dirlst, "Endergebnis","Zwischenstand","Zwischenergebnis","Ergebnis");
          } else {
             file = fcheck(dirlst, id);
          }
          if ( file ) {
             var link =p.base+"/"+file[1]
             jQuery(tab).append("<tr/>")
             jQuery(tab).find("tr:last-child").append(
                "<td class=gt_head width=30px>" +
                "<a title="+file[0]+" target="+link+" href="+link+">" +
                "<img class=gt_icon src="+p.imgsrc+" /></td>"+
                "<td class=gt_head><a target="+link+" href="+link+">"+file[0]+"</a></td>")
          }
       }
       if ( jQuery(div).find(".gt_tab1 tr").length == 0 ) {
          jQuery(div).find("table.gt_tab_p>tbody>tr").first().remove()
       }
 
       //-----> Generiere die Links von head2 ( einzelne Runden )
       tab = jQuery(div).find(".gt_tab2")
       jQuery(tab).append("<tr/>")
       var crow = jQuery(tab).find("tr:last-child")
       for ( var i=0; i<head.length; i++ ) {
          var mclass = ( head[i][0] == 1 ) ? "gt_head gt_h_rotate" : "gt_head  gt_h_normal"
          crow.append("<td class='"+mclass+"'><div>"+head[i][1]+"</div></td>")
       }
       var rnd =0
       for ( var i=0; i<p.table.length; i++ ) {
          for ( var j=0; j<p.table[i][1].length; j++ ) {
             rnd++;
             jQuery(tab).append("<tr/>")
             var crow = jQuery(tab).find("tr:last-child")
             if ( j==0 ) {
                jQuery(crow).append("<td rowspan="+p.table[i][1].length+" class=gt_col1>"+p.table[i][0]+"</td>")
             }

             for ( var k=2; k<head.length; k++ ) {
                jQuery(crow).append("<td class=gt_data gt_data_miss></td>")
                var id = head[k][1]

                if ( file ) {

                      "title": id[0].toUpperCase() + id.substring(1) + " Runde " + rnd,

                         "<div width=100% height=100%>" +
                         "<img src="+p.imgsrc+" class=gt_icon></img></div></a>")
                }
             }
          }
       }
 
       //*-----> Filecheck Routine
       function fcheck() {
          var result = [];
          var args = Array.prototype.slice.call(arguments);
          //console.log("req: " + JSON.stringify(args) );
          for (var i=1;i<args.length;i++) {
             for (var j=0;j<args[0].length;j++) {
                if ( args[0][j][0].toLowerCase().indexOf(args[i].toLowerCase()) >= 0 ) {
                   result.push([args[i],args[0][j][0]])
                   break
                }
             }
 
          }
          //console.log(JSON.stringify(result[0]))
          return(result[0])
       }
    }
