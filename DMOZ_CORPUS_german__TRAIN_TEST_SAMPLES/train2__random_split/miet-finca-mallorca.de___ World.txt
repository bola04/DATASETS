     function createSuchTags() {
         var anztags = 0;
 
         if ($('input:hidden[name=startDate]').val() !== '') {
             anztags++;
             $("#zeitraumtag").show();
 
             var start = moment($("input:hidden[name=startDate]").val());
             var end = moment($("input:hidden[name=endDate]").val());
             var label = start.format('DD.MM.YYYY') + ' - ' + end.format('DD.MM.YYYY');
 
             var dateart = "";
 
             if ($("input:hidden[name=suchart]").val() == 1) {
                 dateart = 'Exakt: ';
             } else {
                 dateart = 'Variabel: ';
             }
 
             $("#zeitraumtag > span").text(dateart + label);
         } else {
             $("#zeitraumtag").hide();
         }
 
         if ($('#frm-filterForm-personen').val() !== 'a') {
             anztags++;
             $("#personentag").show();
             $("#personentag > span").text($('#frm-filterForm-personen option:selected').text());
         } else {
             $("#personentag").hide();
         }
         if ($('#frm-filterForm-preis').val() !== 'a') {
             anztags++;
             $("#preistag").show();
             $("#preistag > span").text($('#frm-filterForm-preis option:selected').text());
         } else {
             $("#preistag").hide();
         }
         if ($('#frm-filterForm-objekt_typ').val() > 0) {
             anztags++;
             $("#typtag").show();
             $("#typtag > span").text($('#frm-filterForm-objekt_typ option:selected').text());
         } else {
             $("#typtag").hide();
         }
         if ($('#frm-filterForm-region').val() > 0) {
             anztags++;
             $("#regiontag").show();
             $("#regiontag > span").text($('#frm-filterForm-region option:selected').text());
         } else {
             $("#regiontag").hide();
         }
         if ($('#frm-filterForm-ort').val() > 0) {
             anztags++;
             $("#orttag").show();
             $("#orttag > span").text($('#frm-filterForm-ort option:selected').text());
         } else {
             $("#orttag").hide();
         }
 
         if (anztags > 0) {
             $("#tagfeld").show();
         } else {
             $("#tagfeld").hide();
         }
     }
