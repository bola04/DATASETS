 function Clock_formatTime () {
   var time = new Date();
   time.setTime(time.getTime() + this.offsetSec * 1000);
   var Wday = hours * 3600 < this.offsetSec ? time.getDay()-1 : time.getDay();
   var hours = time.getHours();
   var minutes = time.getMinutes();
   var seconds = time.getSeconds();
   var day = hours * 3600 < this.offsetSec ? time.getDate()-1 : time.getDate();
   var month = hours * 3600 < this.offsetSec ? time.getMonth()-1 : time.getMonth();
