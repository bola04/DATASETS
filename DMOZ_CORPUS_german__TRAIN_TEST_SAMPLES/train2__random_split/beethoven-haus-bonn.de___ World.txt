 function getQueryStrings() {
   var assoc  = {};
   var decode = function (s) { return decodeURIComponent(s.replace(/\+/g, " ")); };
   var queryString = location.search.substring(1);
   var keyValues = queryString.split('&');
 
   for(var i in keyValues) {
     var key = keyValues[i].split('=');
     if (key.length > 1) {
       assoc[decode(key[0])] = decode(key[1]);
     }
   }
 
   return assoc;
 }

   function getBrowserFlashDetection(){
     var hasFlash = false;
     try {
       hasFlash = Boolean(new ActiveXObject('ShockwaveFlash.ShockwaveFlash'));
     } catch(exception) {
       hasFlash = ('undefined' != typeof navigator.mimeTypes['application/x-shockwave-flash']);
     }
     var isOpera = (!!window.opr && !!opr.addons) || !!window.opera || navigator.userAgent.indexOf(' OPR/') >= 0;
     
     if (isOpera==true){
       checkFlashPlugin("Opera",hasFlash);
     }
       // Firefox 1.0+
       var isFirefox = typeof InstallTrigger !== 'undefined';
     
     if (isFirefox==true){
       checkFlashPlugin("FireFox",hasFlash);
     }
     
     // Safari 3.0+ "[object HTMLElementConstructor]" 
     var isSafari = /constructor/i.test(window.HTMLElement) || (function (p) { return p.toString() === "[object SafariRemoteNotification]"; })(!window['safari'] || safari.pushNotification);
     
     if(isSafari==true){
       checkFlashPlugin("Safari",hasFlash);
     }
     
     // Internet Explorer 6-11
     var isIE = /*@cc_on!@*/false || !!document.documentMode;
     
     if(isIE==true){
       checkFlashPlugin("IE",hasFlash);
     }
     
     // Edge 20+
     var isEdge = !isIE && !!window.StyleMedia;
     
     if(isEdge==true){
       checkFlashPlugin("Edge",hasFlash);
     }
     
     var isChrome = !!window.chrome && !!window.chrome.webstore;
 
     if(isChrome==true){
       checkFlashPlugin("Chrome","true");
     }
   }

   // ===============================================================
   // functions for Diashow on Homepage (Diashow Startseite)
   // ===============================================================
 
   var diashowTimerCalls = 0;
   var diashowTimerIsActive = false;
   var diashowTimer;
 
   // ---------------------------------------------------------------
   function diashowStartTimer() {
   // ---------------------------------------------------------------
       // console.log("diashowStartTimer: slidetime=" + $(".posmitte").attr('data-slidetiming'));
       var timestamp_wait_begin = new Date().getTime();
       diashowTimerCalls += 1;
       diashowTimerIsActive = true;
       diashowTimer = setTimeout(function () {
       var timestamp_wait_end = new Date().getTime();
       // console.log("diashowStartTimer elapsed: " + (timestamp_wait_end - timestamp_wait_begin ));
       diashowNext();
       diashowStartTimer();
       }, parseInt($(".posmitte").attr('data-slidetiming')));
   };
 
   // ---------------------------------------------------------------
   function diashowStopTimer() {
   // ---------------------------------------------------------------
       if (diashowTimerIsActive){
         clearTimeout(diashowTimer);
         diashowTimerIsActive = false;
       }
   };
 
   // ---------------------------------------------------------------
   function diashowNext() {
   // ---------------------------------------------------------------
     var slides = $(".slidescontainer").children();
     var numberofslides = slides.length;
     var activeslidenumber = -1;
     for (var i = 0; i < numberofslides; i++) {
         var slide = slides[i]
         if ($(slide).hasClass("posmitte")) {
             activeslidenumber = i;
         }
     }
     // console.log("diashowNext: previously active slidenumber=" + activeslidenumber);
     if (activeslidenumber == -1) {
         // console.log("diashowNext: No Element found with Class: posmitte");
         return
     }
     if (activeslidenumber + 1 < numberofslides) {
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posleft");
         $(slides[activeslidenumber + 1]).removeClass("posright");
         $(slides[activeslidenumber + 1]).addClass("posmitte");
         if (activeslidenumber + 2 >= numberofslides) {
             $(slides[0]).removeClass("posleft");
             $(slides[0]).addClass("posright");
         } else {
             $(slides[activeslidenumber + 2]).removeClass("posleft");
             $(slides[activeslidenumber + 2]).addClass("posright");
         }
     } else {
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posleft");
         $(slides[0]).removeClass("posright");
         $(slides[0]).addClass("posmitte");
         $(slides[1]).removeClass("posleft");
         $(slides[1]).addClass("posright");
     }
   }
 
   $(document).ready(function () {
     if (!diashowTimerIsActive){
        // console.log("document ready: diashow gets started now")
        diashowStartTimer();
     }
 
   // ---------------------------------------------------------------
     $(".nextslidebutton").on("click", function () {
   // ---------------------------------------------------------------
       if (diashowTimerIsActive){
          diashowStopTimer();
       }
       var slides = $(this).prev(".slidescontainer").children();
       var numberofslides = slides.length;
       var activeslidenumber = -1;
       for (var i = 0; i < numberofslides; i++) {
         var slide = slides[i]
         if ($(slide).hasClass("posmitte")) {
           activeslidenumber = i;
         }
       }
       // console.log("nextslidebtn: numberofslides=" + numberofslides);
       // console.log("nextslidebtn: previously active slidenumber=" + activeslidenumber);
       if (activeslidenumber == -1) {
         // console.log("nextslidebtn: No Element found with Class: posmitte");
         return
       }
       if (activeslidenumber + 1 < numberofslides) {
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posleft");
         $(slides[activeslidenumber + 1]).removeClass("posright");
         $(slides[activeslidenumber + 1]).addClass("posmitte");
         if (activeslidenumber + 2 >= numberofslides) {
           $(slides[0]).removeClass("posleft");
           $(slides[0]).addClass("posright");
         } else {
           $(slides[activeslidenumber + 2]).removeClass("posleft");
           $(slides[activeslidenumber + 2]).addClass("posright");
         }
       } else {
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posleft");
         $(slides[0]).removeClass("posright");
         $(slides[0]).addClass("posmitte");
         $(slides[1]).removeClass("posleft");
         $(slides[1]).addClass("posright");
       }
 
       diashowStartTimer();
     })
 
   // ---------------------------------------------------------------
     $(".prevslidebutton").on("click", function () {
   // ---------------------------------------------------------------
       if (diashowTimerIsActive){
          diashowStopTimer();
       }
       var slides = $(this).prev().prev(".slidescontainer").children();
       var numberofslides = slides.length;
       if (numberofslides < 2) {
         return
       }
       var activeslidenumber = -1;
       for (var i = 0; i < numberofslides; i++) {
         var slide = slides[i]
         if ($(slide).hasClass("posmitte")) {
           activeslidenumber = i;
         }
       }
       // console.log("prevslidebutton: numberofslides=" + numberofslides);
       // console.log("prevslidebutton: previously active slidenumber=" + activeslidenumber);
       if (activeslidenumber == -1) {
         //console.log("No Element found with Class: posmitte");
         return
       }
       if (activeslidenumber > 0) {
 
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posright");
         $(slides[activeslidenumber - 1]).removeClass("posleft");
         $(slides[activeslidenumber - 1]).addClass("posmitte");
         if (activeslidenumber - 1 > 0) {
           $(slides[activeslidenumber - 2]).removeClass("posright");
           $(slides[activeslidenumber - 2]).addClass("posleft");
         } else {
           $(slides[numberofslides - 1]).removeClass("posright");
           $(slides[numberofslides - 1]).addClass("posleft");
         }
       } else {
         $(slides[activeslidenumber]).removeClass("posmitte");
         $(slides[activeslidenumber]).addClass("posright");
         $(slides[numberofslides - 1]).removeClass("posleft");
         $(slides[numberofslides - 1]).addClass("posmitte");
 
         $(slides[numberofslides - 2]).removeClass("posright");
         $(slides[numberofslides - 2]).addClass("posleft");
       }
       diashowStartTimer();
     })
   });
