    $('#load_login').ready(function () {
 	function submitter(){
 		$('#col').submit(function() {

 				checker(true);
 			});
 			return false;
 		});
 	}
 	function checker(from_login){
 		$.getJSON('/de/sec/am/i/authenticated', null, function(data){
 			if (data != undefined && typeof data === 'object' && data['isauth'] === true){
 				$.get('/de/sec/i/am/authenticated', null, function(data){
 					$('#load_login_f').hide('fade', null, 200, function(){
 						$('#load_login').html(data);
 					});
 					$('#load_login_f').show('fade', null, 200, function(){
 							submitter();
 					});
 				});
 			}else{
 				$.get('/de/sec/login', null, function(data){
 					$('#load_login_f').hide('fade', null, 200, function(){
 						$('#load_login').html(data);
 						if(from_login) $("#jserr").text("Fehlerhafte Anmeldedaten. Bitte versuchen Sie es erneut.");
 					});
 					$('#load_login_f').show('fade', null, 200, function(){
 						submitter();
 					});
 				});
 			}
 		});
 	}
 	checker();
 });
