         var suggest_image_dir = '90x90/';

             function display_quicksearch(data){
                 $('#suggest_popup').remove();
 
                 //keine Daten - Abbruch
                 if(typeof data == 'undefined' || data == '' || data == '----'){
                     return;
                 }
 
                 $('#quicksearch').append("<div id='suggest_popup'></div>");
 
                 var products_manufacturers = data.split('----');
 
                 var content = '';
                 content += '<div id="suggest_products">';
                 content += '<h1 id="suggest_title_products">Produkte</h1>';
 
                 if(typeof products_manufacturers[0] != 'undefined' && products_manufacturers[0] != ''){
                     var products = products_manufacturers[0].split("\n");
                     content += '<ul id="suggest_list_products">';
                     for(i = 0; i < products.length; i++){
                         var product = products[i].split(';');
                         var productname = '';
                         //Name bei Bedarf in Spans aufteilen
                         if(product[1].indexOf('|') != -1){
                             productnameArr = product[1].split('|');
                             for(iname = 0; iname < productnameArr.length; iname++){
                                 productname += '<span class="suggest_name_' + iname + '">' + productnameArr[iname] + '</span>';
                             }
                         }
                         else{
                             productname = product[1];
                         }
 
                         //FĂźr alternative Bilderstruktur, bei den die Artnr der Ordnername ist, wie bei BP
                         if(typeof suggest_folder_image_filename == 'undefined'){
                             suggest_folder_image_filename = ''; //e.g. /212x159_06
                         }
 
                         if ( typeof suggest_image_dir === "undefined" ) {
                             productimage = '<img src="/img/products/' + product[0] + suggest_folder_image_filename +'.jpg" class="suggest_productimage" alt="' + product[1] + '">';
                         }
                         else {
                             productimage = '<img src="/img/products/'+ suggest_image_dir + product[0] + suggest_folder_image_filename +'.jpg" class="suggest_productimage" alt="' + product[1] + '">';
                         }
                         content += '<li><a href="/details.php?p_id=' + product[0] + '">' + productimage + productname + '</a></li>';
                     }
                     content += '<li id="quickSearchMore"><a href="/search.php?search_str=' + $("#header_search_box").val() + '">Mehr...</a></li>';
                     content += '</ul>';
                 }
                 content += '</div>';
 
                 content += '<div id="suggest_manufacturers">';
                 content += '<h1 id="suggest_title_manufacturers">Marken</h1>';
 
                 if(typeof products_manufacturers[1] != 'undefined' && products_manufacturers[1] != ''){
                     var manufacturers = products_manufacturers[1].split("\n");
                     content += '<ul id="suggest_list_manufacturers">';
                     for(i = 0; i < manufacturers.length; i++){
                         var manufacturer = manufacturers[i].split(';');
                         if(suggest_manuf_textonly != true){
                             manuf_image = '<img src="/img/manufacturers_logos_suggest/' + manufacturer[0].toLowerCase() + '.jpg" class="suggest_manuf_image" alt="' + manufacturer[1] + '">';
                         }
                         else{
                             manuf_image = manufacturer[1];
                         }
                         content += '<li><a href="/search.php?manufacturer=' + manufacturer[0] + '">' + manuf_image + '</a></li>';
                     }
                     content += '</ul>';
                 }
                 content += '</div>';
 
                 $('#suggest_popup').html(content);
                 return;
             }
 
             var suggestActive;
             var suggestCounter = 0;
             var resultsTimer;
 
             $( document ).ready(function() {
                 //Klick ausserhalb Popup schlieĂt selbiges**
                 //Ipad kennt click nicht, touchstart verwenden
                 var ua = navigator.userAgent,
                 event = (ua.match(/iPad/i)) ? "touchstart" : "click";
 
                 $( "body" ).on(event, function (ev) {
                     $('#suggest_popup').remove();
                 });
 
                 //initialen Wert lĂśschen, wir verlassen uns darauf das "Suchbegriff" darin vorkommt
                 $("#header_search_box").focus(function() {
                     var search_value = $("#header_search_box").val();
                     if( search_value.indexOf('Suchbegriff') != -1 ){
                         $("#header_search_box").val('');
                     }
                 });
 
                 $( "#header_search_box" ).keyup(function() {
                     if (suggestActive){
                         suggestActive.abort();
                     }
 
                     var search_value = encodeURIComponent($("#header_search_box").val());
                     if($.trim(search_value) == ''){
                         return true;
                     }
                     var token = ++suggestCounter;
 
                     clearTimeout(resultsTimer);
                     resultsTimer = setTimeout(function(){
                         suggestActive = $.ajax({
                             type: "GET",
                             url: "/search_suggest.php?search_str=" + search_value + "&cat_id=0&lmt=5&manuf=true&pp=true"
                         }).done(function( return_data ) {
                             suggestActive = null;
                             if (token != suggestCounter){
                                 return true;
                             }
                             display_quicksearch(return_data);
 
                             //** fĂźr Popup Vererbung verhindern
                             $('#suggest_popup').on(event, function (ev) {
                               ev.stopPropagation();
                             });
 
                             return true;
                         });
                     },500);
                 });
 
             });

 function updatecartbox(){
     $.ajax({
         type: "GET",
         url: "/static.php?static=ajax_cartbox_hinweis&show_header=true"
     }).done(function( return_data ) {
         var cartbox = document.getElementById('hinweis_cartbox');

             cartbox.innerHTML = return_data;
         }
     });
            
     $.ajax({
         type: "GET",
         url: "/static.php?static=ajax_cartbox_default&show_header=true"
     }).done(function( return_data ) {

         if(typeof cartbox != "undefined" && cartbox != null){
             cartbox.innerHTML = return_data;
         }
     });
     var header_cartbox_count_element = document.getElementById('header_cartbox_count');
     if(typeof header_cartbox_count_element != 'undefined' && header_cartbox_count_element != null){
             $.ajax({
                  type: "GET",
                  url: "/static.php?static=ajax_cartbox_count&show_header=true"
             }).done(function( return_data ) {
                      header_cartbox_count_element.innerHTML = return_data;
             });

 function addItemToCart() 
         {   
         $.ajax({
             type: "POST",
             url: "/ajax_backend.php?mode=getProducts",
             data: 'pid='+productid+'&quantity='+amount
         }).done(function( return_data_json ) {
             jsonObj = jQuery.parseJSON(return_data_json);
             if(jsonObj.error == false){
                 var productName = jsonObj.productname;
                 productName = productName.replace(/\|/g," "); 
                 var productNameAll = "<span>";
                    
 
                 productNameAll += "<a href='/details.php?p_id=" + productid + "' title='" + productName + "'>" + productName + "</a> ";
                 
                 if ($('#hinweis_img').length){ 
                     var imagePfad = $("#hinweis_img").attr("src");
                     imagePfad = imagePfad.substr(0,imagePfad.lastIndexOf("/"));
                     var imgsrc = imagePfad + "/" + productid + ".jpg";
                     var img = new Image();
                     img.onload = function (evt){
                         $("#hinweis_img").attr("src", imgsrc);             
                     }
                     img.src = imgsrc;
                 }
             }
             else {
                 var productNameAll = "Es ist ein Fehler aufgetreten.";
             }
 
             $("#hinweis").hide(0);
             $("#hinweis_content").html(productNameAll);
            
             
             $("#hinweis_img").css({display: 'block'});
             $("#hinweis_cartbox").css({display: 'block'});
              $("#hinweis").css({top: ((windowHeight/2) - (height/2))+'px'});
             $("#hinweis").delay(500).show(0); 
             var height = $("#hinweis").height();
             var topx = $("#hinweis").position().top;
             var windowHeight = $(window).height();
             if(windowHeight-30 <= (height + topx))
             {
                 $("#hinweis").css({ top: '0px' });
                 if(windowHeight <= height)
                 {
                     $("#hinweis_img").css({display: 'none'});
                     $("#hinweis_cartbox").css({display: 'none'});
                 }  
             }
             else
             {
                 $("#hinweis_img").css({display: 'block'});
                 $("#hinweis_cartbox").css({display: 'block'});
                 $("#hinweis").css({top: ((windowHeight/2) - (height/2))+'px'});
             }
                
             if (doupdatecartbox == "true"){
                 updatecartbox();	
             }
         });
     }

 function cartCoords()
 {
     var data = new Object();

 }
