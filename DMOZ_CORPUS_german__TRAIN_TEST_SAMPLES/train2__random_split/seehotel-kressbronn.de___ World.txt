        <h2>
         Herzlich willkommen
        </h2>
        <p>
         Das
         <b>
          Seehotel Kressbronn am Bodensee
         </b>
         führt raus aus der Hektik des Alltags, hinein in die faszinierende Natur des Voralpenlandes. Das Hotel bietet Raum zum Entspannen in einem einmaligen Ambiente – Leben im Standby- Modus. Die perfekte Umgebung, um abzuschalten, innerlich zur Ruhe zu kommen und wieder neue Impulse und Ideen zu sammeln.
        </p>
       </div>
       <div class="introbox num2">
        <p>
         <a href="/unterkunft-bodensee/zimmer" target="_self" title="Unterkunft am Bodensee">
          In 25 Zimmern
         </a>
         – nur Nichtraucher – dominieren ungekünsteltes Design, offene Leichtigkeit und herrliche Ruhe, fast immer mit Blick auf den See. Große Fenster, ein schnörkelloser Innenausbau, leichte und frische Farben in allen Zimmern sorgen fürs Wohlbefinden.
        </p>
        <p class=" text-right">
         <b>
          Wir freuen uns auf Ihren Besuch!
         </b>
        </p>

       var check = false;
       var regExp = /^\d{1,2}\.\d{1,2}\.\d{4}$/;
       if ( regExp.test(value) ) {
         var formattedDateString;
         var dateString = value.split('.');
         var dd = parseInt(dateString[0],10);
         var mm = parseInt(dateString[1],10);
         var yyyy = parseInt(dateString[2],10);
         var dateObject = new Date(yyyy,mm-1,dd);
         if ( ( dateObject.getFullYear() == yyyy ) && ( dateObject.getMonth () == mm - 1 ) && ( dateObject.getDate() == dd ) ) {

           $(element).val( formattedDateString );
           $(element).change();
         } else {
           check = false;
         }
       } else {
         check = false;
       }
       return this.optional(element) || check;
