 var fade_time = 800;  // wie lange soll ueberblendet werden? [ms]
 var fade_steps = 20;  // Frames der Ueberblendung. Je mehr, desto glatter die Ueberblendung. 
 var img_time = 8000;  // wie lange soll ein Bild gezeigt werden? [ms]
 var img_length = 11;  // wie viele Bilder gibt es?
 var img_path = '/slideshow';  // relativer oder absoluter Pfad zum Verzeichnis in dem die Bilder liegen
 var img_suffix = 'Korfu-';  // Praefix des Dateinamens der Bilder
 var img_format = 'jpg';  // Format der Bilder
 var randomize = true;  // Abspielreihenfolge zufaellig?
 
 
 var img=new Array();
 var number=new Array();
 var random1, random2, temp;
 var i, k;
 var f=fade_steps;
 var fade_frame, opacity;
 var img1dia, img2dia;
 var fadein, fadeout;
 var init=false;
 for(k=img_length; k--;) {
     number[k]=k;  // Schreibe Array mit Zahlen 0,1,2,3, ... ,img_length-1
 }
 if(randomize===true) {
     for(k=(img_length*3); k--;) {
         random1=Math.floor(Math.random()*img_length);
         random2=Math.floor(Math.random()*img_length);
         temp=number[random1];
         number[random1]=number[random2];  // permutiere diese Zahlen zufaellig fuer eien zufaellige Abspielreihenfolge
         number[random2]=temp;
     }
 }
 
 function init_function() {
     img[0]=new Image();
     img[0].src=img_path + "/" + img_suffix + number[0] + "." + img_format;  // lade das erste Bild
     img[1]=new Image();
     img[1].src=img_path + "/" + img_suffix + number[1] + "." + img_format;  // lade das zweite Bild
     k=0;
     img1dia=document.images["img1dia"];

     fadein=img1dia;
     fadeout=img2dia;
     fade_frame=fade_time/fade_steps;
     opacity_frame=100/fade_steps;
 }
 
 function setAlpha(element, value) {
     // Browserabhaengige Transparenz fuer Ueberblendung
     var opacity=value/100;
     if(img1dia.style.filter!==null) {element.style.filter="Alpha(opacity="+value+")";}  // IE
     if(img1dia.style.opacity!==null) {element.style.opacity=opacity;}                   // Opera, Chrome, Firefox, Safari

     if(img1dia.style.MozOpacity!==null) {element.style.MozOpacity=opacity;}             // alte Firefox

 animation = function() {

              <td>
               <div class="utext3">
                <b>
                 <font color="#666666">
                  Ionische Insel Korfu - erste 
                             Eindrücke. Ein kurzer Korfu-Überblick:
                 </font>
                </b>
                <br/>
                <br/>
                Korfu (griechisch:
                <i>
                 Kerkyra
                </i>
                ) wurde erstmals in 
                           Homers
                <i>
                 Odyssee
                </i>
                erwähnt. Neben Venedig, England 
                           wurde die Insel in der jüngeren Geschichte auch 
                           von Frankreich nachhaltig geprägt. Die Hauptstadt 
                           Korfu-Stadt (Kerkyra) gilt als die schönste Stadt 
                           Griechenlands, wurde daher in das
                <b>
                 UNESCO Welterbe
                </b>
                aufgenommen und erinnert architektonisch stark an die 
                           Lagunenstadt
                <i>
                 Venedig
                </i>
                in Italien.
                <br/>
                <br/>
                Die griechische Insel Korfu ist perfekt für einen 
                           abwechslungsreichen und aktiven Urlaub. Zahlreiche
                <b>
                 Aktivitäten
                </b>
                warten auf Sie, z.B. Schwimmen, Schnorcheln,
                <i>
                 Scuba-Diving
                </i>
                , 
                           Paragliding (
                <i>
                 Gleitschirm
                </i>
                fliegen),
                <i>
                 Kite-Surfen
                </i>
                ,
                <i>
                 Windsurfen
                </i>
                , Katamaran, Motorboot- oder
                <i>
                 Bootsausflüge
                </i>
                auf die Nachbarinseln ider auf das albanische (Saranda 
                           und Butrint) und griechische Festland (z.B. Blaue Lagune, 
                           Vikos-Schlucht oder Sivota).
                <br/>
                <br/>
                Für kleine Wasserratten ist ein Besuch im
                <b>
                 Aqualand 
                           Korfu
                </b>
                ein tolles Erlebnis (Ropa-Tal). Ein gigantisches 
                           Erlebnis-Spaß-Bad. Einen weiteren, etwas kleineren 
                           Wasserpark finden Sie auch im Norden Korfus bei Almiros 
                           und Acharavi, das
                <i>
                 Hydropolis
                </i>
                .
                <br/>
                <br/>
                Corfu ist ein wahres Paradies zum
                <b>
                 Wandern
                </b>
                . Der
                <i>
                 Corfu-Trail
                </i>
                , ein
                <b>
                 Wanderweg
                </b>
                der die Insel 
                           von Süden nach Norden durchzieht ist ein Geheimtipp 
                           unter Wanderfreunden. Wer es etwas zügiger mag, 
                           dem bieten sich zahlreiche
                <b>
                 Mountainbike
                </b>
                -Touren 
                           an. Ob zu Fuß oder auf zwei Rädern: Die Landschaften, 
                           Bergdörfer, Flora &amp; Fauna Korfus sind so abwechslungsreich 
                           wie beeindruckend.
                <br/>
                <br/>
                Und es gibt sie noch, die gute
                <b>
                 griechische Küche
                </b>
                . 
                           Besonders in den Dörfern abseits der Touristenorte 
                           können Sie noch in der Küche in die Töpfe 
                           schauen. Besonders Fischgerichte wie Pulpo (Oktopus), 
                           Kalamares, Dorade aber auch typisch korfiotische Speisen 
                           wie Sofrito oder Stifado sind besonders zu empfehlen. 
                           Natürlich kommt auch leckeres Gyros, Souvlaki und 
                           Moussaka auf den Tisch. Oftmals wird selbst angebautes 
                           Obst und Gemüse verarbeitet.
               </div>
              </td>

              <td>
               <div class="utext3">
                <b>
                 <font color="#666666">
                  Korfu: Insel 
                           zwischen Kultur, Sandstrand, Party und Sport
                 </font>
                </b>
                <br/>
                <br/>
                Für kulturell Interessierte warten auf Korfu Unmengen 
                           von Kirchen und Museen auf Ihren Besuch. Ein Muss ist 
                           die
                <b>
                 alte Festung
                </b>
                beim französisch geprägten
                <i>
                 Esplanade-Platz
                </i>
                (Liston) mit seinen Arkaden. 
                           Auch die
                <b>
                 neue Festung
                </b>
                (venezianisch) lohnt einen 
                           Ausflug. Insgesamt sollten Sie für Korfu-Stadt 
                           mindestens einen Tag einplanen. Auch sehr zu empfehlen 
                           ist Korfu-Stadt bei Nacht. Genießen Sie einen 
                           Frappe oder griechischen Kaffee und beobachten im Liston 
                           das Geschehen.
                <br/>
                <br/>
                Das Schloss der österreichischen Kaiserin Elisabeth 
                           (
                <b>
                 Sissi-Schloss
                </b>
                ) nennt sich
                <a href="/Korfu/Achilleion">
                 Achilleion bzw. Achillion
                </a>
                und befindet sich in der Nähe von
                <a href="/Korfu/Benitses">
                 Benitses
                </a>
                . 
                           Gerade in den Sommermonaten ist hier mit vielen Besuchern 
                           zu rechnen, wobei die parkähnliche Gartenanlage 
                           mit einem Traumpanorama die eigentliche Hauptattraktion 
                           darstellt.
                <br/>
                <br/>
                <b>
                 Sandstrände
                </b>
                finden Sie auf Korfu vor allem 
                           im Südwesten (
                <a href="/Korfu/Agios-Georgios-Argirades">
                 Agios 
                           Georgios
                </a>
                bei
                <a href="/Korfu/Argirades">
                 Argirades
                </a>
                ,
                <a href="/Korfu/Vitalades">
                 Vitalades
                </a>
                ,
                <a href="/Korfu/Marathias">
                 Marathias
                </a>
                ) 
                           und Norden der Insel (
                <a href="/Korfu/Almiros">
                 Almiros
                </a>
                ,
                <a href="/Korfu/Acharavi">
                 Acharavi
                </a>
                ) 
                           , im Nordwesten ist es eher felsig. Der Südosten 
                           Korfus (
                <a href="/Korfu/Boukari">
                 Kouspades
                </a>
                ,
                <a href="/Korfu/Boukari">
                 Boukari
                </a>
                ,
                <a href="/Korfu/Petriti">
                 Petriti
                </a>
                ,
                <a href="/Korfu/Benitses">
                 Benitses
                </a>
                ) 
                           erinnert ein wenig an die Gardasee-Küste. Hier 
                           finden Sie feinkiesige Strände mit Schatten spendenden 
                           Bäumen bis zum Ufer und blühenden Oleander 
                           bis in den Spätsommer - eine wahre Farbenpracht.
                <br/>
                <br/>
                Zahlreiche Bootsstege führen in das kristallklare 
                           Wasser und animieren zum Träumen, Schnorcheln oder 
                           einfach zum Faulenzen.
                <br/>
               </div>
              </td>
