 			if (classes[j].indexOf("rheight") == 0) {
 				var rheight = classes[j].substring(7)/100;
 			} else if (classes[j].indexOf("ropacity") == 0) {
 				var ropacity = classes[j].substring(8)/100;
 			}
 		}
 		
 		Reflection.add(rimages[i], { height: rheight, opacity : ropacity});
 	}
 }
