 			var ikissCookie = {
 				set: function(name, value, days) {
 					if (!days) { days = 30; }
 					var date = new Date();
 					date.setTime(date.getTime()+(days*24*60*60*1000));
 					var expires = "; expires="+date.toGMTString();
 					document.cookie = name+"="+value+expires+"; path=/";
 				},
 				
 				get: function(name) {
 					var n = name+"=";
 					var cs = document.cookie.split(';');
 					for(var i=0;i < cs.length;i++) {
 						var c = cs[i];
 						while (c.charAt(0)==' ') c = c.substring(1, c.length);
 						if (c.indexOf(n) == 0) return c.substring(n.length, c.length);
 					}
 					return null;
 				},
 						
 				del: function(name) {
 					this.set(name,"",-1);
 				}
 			};
 			
 			
 			$(document).ready(function () {

 						return intx;

      <h3>
       Kommunale Internetauftritte mit dem CMS iKISS
      </h3>
      <div class="einleitung">
       <div style="display:inline">
        <p>
         Wir bieten Ihnen unser eigenes Content-Management-System iKISS, mit dem Sie Ihre Webseite schnell, einfach und umfassend pflegen können. Darüber hinaus erhalten Sie bei uns alle nötigen Dienstleistungen für Ihren kommunalen Internetauftritt aus einer Hand.
        </p>
       </div>
      </div>

        Zertifiziertes CMS
       </h4>
       <p>
        <a class="csslink_extern_img" href="https://www.teleport.de/effizient-digitales-wissen-nutzen/infodienste-linie6plus/zertifizierte-schnittstellen/" target="_blank" title="Externer Link: Zertifizierte bidirektionale TSA-Schnittstelle">

        </a>
       </p>
       <p>
        Das CMS iKISS bietet eine zertifizierte Schnittstelle zu den Zuständigkeitsfindern der TSA Teleport GmbH.
       </p>
