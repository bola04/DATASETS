   <script type="text/javascript">
    /* Light YouTube Embeds by @labnol */
 /* Web: http://labnol.org/?p=27941 */
 
 document.addEventListener("DOMContentLoaded",
 	function() {
 		var div, n,
 			v = document.getElementsByClassName("youtube-player");
 		for (n = 0; n < v.length; n++) {
 			div = document.createElement("div");
 			div.setAttribute("data-id", v[n].dataset.id);
 			div.innerHTML = labnolThumb(v[n].dataset.id);
 			div.onclick = labnolIframe;
 			v[n].appendChild(div);
 		}
 	});
 
 function labnolThumb(id) {
 	var thumb = '<img src="templates/rolladen_neu/img/layout/hqdefault.jpg" />',
 		play = '<div class="play"></div>';
 	return thumb.replace("ID", id) + play;
 }
 
 function labnolIframe() {
 	var iframe = document.createElement("iframe");
 	var embed = "https://www.youtube.com/embed/ID?autoplay=1";
 	iframe.setAttribute("src", embed.replace("ID", this.dataset.id));
 	iframe.setAttribute("frameborder", "0");
 	iframe.setAttribute("allowfullscreen", "1");
 	this.parentNode.replaceChild(iframe, this);
 }

             <h2>
              Rolläden bieten Beschattung, Sonnenschutz und Sichtschutz für Fenster und Türen
             </h2>
             <p>
              Rolläden sind für Ihre Wohnung oder Ihr Haus wertsteigernd. Geschlossene Rolläden erhöhen die Temperatur der dahinterliegenden Räume und tragen damit zur Energieeinsparung bei. Zwischen Fenster oder Tür und geschlossenem Rolladen bildet sich ein stark isolierendes Luftpolster. Das gewährt einen optimalen Wärmeschutz. Der Dämmwert eines Fensters kann bis zu 20% verbessert werden.
             </p>
             <h3 class="pad-top-20">
              Rolläden von Rolloscout - für mehr Sicherheit zu Hause
             </h3>
             <p>
              Leider steigt die Zahl der Einbrüche in Deutschland. Statistiken belegen, daß Einbruchsversuche nach spätestens 5 Minuten abgebrochen werden - diese Zeit gilt es zu überstehen.
             </p>
             <p>
              Beim Kunststoffrollladen sind die Stäbe einfach aus der Führung zu reißen, deshalb ist der Schutz unzureichend. Anders dagegen beim Alurolladen. Ein Rolladen aus Aluminiumlamellen hält den Bemühungen der Einbrecher länger stand und bietet erhöhten Einbruchschutz.
             </p>
             <p>
              Neben der Stabilität spricht auch die Hitze- und Lichtbeständigkeit für einen Alurolladen. Starke Sonneneinstrahlung erweicht die Oberfläche der PVC Rolläden und führt zum Ausbleichen der Farben und kann den Rolladen verformen. Für einen Alurolladen sind hohe Temperaturen kein Problem, seine Funktion und Stabilität bleibt zuverlässig erhalten.
             </p>
