 var searchReq = getXmlHttpRequestObject();
 
 
 function searchSuggest() {
 	if (searchReq.readyState == 4 || searchReq.readyState == 0) {
 		var str = escape(document.getElementById('txtSearch').value);
 		searchReq.open("GET", 'searchSuggest.php?search=' + str, true);
 		searchReq.onreadystatechange = handleSearchSuggest; 
 		searchReq.send(null);
 	}		
 }
 
 
 function handleSearchSuggest() {
     if (searchReq.readyState == 4) {
         var ss = document.getElementById('search_suggest')
         ss.innerHTML = '<div class="suggest_header">Vorschläge zu Ihrem Suchbegriff</div>';
         var str = searchReq.responseText.split("\n");
         for(i=0; i < str.length - 1; i++) {
        
             var suggest = '<div onmouseover="javascript:suggestOver(this);" ';
             suggest += 'onmouseout="javascript:suggestOut(this);" ';
             suggest += 'onclick="javascript:setSearch(this.innerHTML);" ';
             suggest += 'class="suggest_link">' + str[i] + '</div>';
             ss.innerHTML += suggest;
         }
         if (i==0) {
             ss.style.visibility  = "hidden";
         } else {
             ss.style.visibility  = "visible";
         }
     
         ss.innerHTML += '<div class=suggest_close><a onmouseover="javascript:suggestOver(this);" onmouseout="javascript:suggestOut(this);" onClick="javascript:gotoSearch(this);"><b>...mehr Resultate<b /><a /><br /><a onmouseover="javascript:suggestOver(this);" onmouseout="javascript:suggestOut(this);" onClick="javascript:suggestClose(this);"><b>Fenster schließen<b /><a /><div />';
     }
 }
 
 
 function gotoSearch(div_value) {
     var strText = escape(document.getElementById('txtSearch').value);
     top.location.href = "../../../shop/advanced_search_result.php?keywords=" + strText; 
 }
 
 
 function suggestClose (div_value) {
 	document.getElementById('search_suggest').innerHTML = '';
 	document.getElementById('search_suggest').style.visibility  = "hidden";
 }
 
 
 function suggestOver(div_value) {
 	div_value.className = 'suggest_link_over';
 }
 
 function suggestOut(div_value) {
 	div_value.className = 'suggest_link';
 }
