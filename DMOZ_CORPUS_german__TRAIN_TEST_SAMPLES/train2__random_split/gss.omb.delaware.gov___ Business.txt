 	$("body").addClass("enhanced");

 		}

    'use strict';
         var App = angular.module('RSSFeedApp', []);
 
         App.controller("FeedCtrl", ['$scope', '$timeout', 'FeedService', function($scope, $timeout, Feed) {
             var feedSrc = 'http://rss.blackboardconnect.com/215074/DEClosings/feed.xml?nocache=';
             getContext();
 
             function getContext() {
                 feedSrc += (new Date).getTime();
                 Feed.parseFeed(feedSrc).then(function(res) {
                     if (res !== undefined && res.status === 200) {
                         $scope.feeds = res.data.responseData.feed.entries;
                         if ($scope.feeds.length > 0) {
                             toastr.error($scope.feeds[0].content);
                         }
                     }
                 });
             }
         }]);
 
         App.factory('FeedService', ['$http', function($http) {
             return {
                 parseFeed: function(url) {
                     return $http.jsonp('//ajax.googleapis.com/ajax/services/feed/load?v=1.0&num=50&callback=JSON_CALLBACK&q=' + encodeURIComponent(url));
                 }
             }
         }]);
   </script>
