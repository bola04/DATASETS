 						function createCookie(name,value,days) {
 							if (days) {
 								var date = new Date();
 								date.setTime(date.getTime()+(days*60*1000));
 								var expires = '; expires='+date.toGMTString();
 							} else var expires = '';
 							document.cookie = name+'='+value+expires+'; path=/';
 						}
 
 						function readCookie(name) {
 							var nameEQ = name + '=';
 							var ca = document.cookie.split(';');
 							for(var i=0;i < ca.length;i++) {
 								var c = ca[i];
 								while (c.charAt(0)==' ') c = c.substring(1,c.length);
 								if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
 							}
 							return null;
 						}

        <h1>
         Willkommen im Oberpfälzer Seenland!
        </h1>
        <p class="tiny_p">
         Das Oberpfälzer Seenland ist die perfekte Urlaubsregion für jeden, der sich gerne ins nasse Vergnügen stürzt:
         <strong>
          Baden, Bootswandern, Wasserski fahren, Segeln, Surfen, Tauchen
         </strong>
         – die
         <strong>
          größten Seen Ostbayerns
         </strong>
         , unzählige Teiche und idyllische Flüsse machen all‘ dies möglich.
        </p>
        <p class="tiny_p">
        </p>
        <p class="tiny_p">
         Dazu noch
         <strong>
          gruselige Geisterburgen, beeindruckende Festspiele und diverse Museen
         </strong>
         . Auf unserer Visitenkarte stehen die große
         <strong>
          Kreisstadt Schwandorf,
         </strong>
         die
         <strong>
          Städte Neunburg vorm Wald und Nittenau
         </strong>
         , die
         <strong>
          Märkte Bruck in der Oberpfalz und Schwarzenfeld
         </strong>
         sowie die
         <strong>
          Gemeinden Bodenwöhr, Steinberg am See und Wackersdorf
         </strong>
         . Zusammen mit unseren Partnern, dem
         <strong>
          Bezirk Oberpfalz
         </strong>
         und dem
         <strong>
          Landkreis Schwandorf
         </strong>
         bilden sie das Oberpfälzer Seenland.
        </p>
        <p class="tiny_p">
        </p>
