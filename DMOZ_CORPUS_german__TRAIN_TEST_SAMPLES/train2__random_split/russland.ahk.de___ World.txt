 		function decryptCharcode(n,start,end,offset) {
 			n = n + offset;
 			if (offset > 0 && n > end) {
 				n = start + (n - end - 1);
 			} else if (offset < 0 && n < start) {
 				n = end - (start - n - 1);
 			}
 			return String.fromCharCode(n);

 			var dec = "";
 			var len = enc.length;
 			for(var i=0; i < len; i++) {

 				} else {
 					dec += enc.charAt(i);
 				}
 			}
 			return dec;
 		}
