 function handleResponsiveTable() {
 	var resplayout;
 	
 	$("table").each(function(){
 		/* darf unter umstaenden nicht auf alle Tabellen angewendet werden, daher an resplayout gehaengt*/
 		if ($(this).attr("resplayout")){
 			resplayout = $(this).attr("resplayout");
 			
 			if(resplayout === undefined){
 				resplayout = "scroll";
 			}
 			
 			if(resplayout == "break"){
 				if($(window).width()< 768){
 					$("td", this).each(function(){
 						$(this).data("mywidth", $(this).attr("width"))
 					});
 					
 					$("td", this).each(function(){
 						$(this).removeAttr("width")
 						$(this).css("width", "")
 						/* Hoehe nicht leer lassen. Dann kann man auch die Zellengroesse für responsive Tabellen anpassen */
 						/* $(this).css("height", "") */
 					});
 					 
 					$(this).addClass("dante-responsive-tables")				 
 				}
 				else{
 					$("td").each(function(){
 						$(this).attr("width", $(this).data("mywidth"))
 						$(this).css("width", $(this).data("mywidth"))
 					});
 					
 					$(this).removeClass("dante-responsive-tables")
 				}
 			} else {
 				$(this).parent().addClass("table-responsive");
 			}
 		}
 	});
 }
