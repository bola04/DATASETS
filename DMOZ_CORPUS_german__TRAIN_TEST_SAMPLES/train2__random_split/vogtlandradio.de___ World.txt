 function seite_berechnen(hoehe_window){
 	
 	//Verhaeltniss format berechnen
 	var format =  hoehe_window / 590 ;
 	
 	$('#aktuelles_format_site_format').val(format);
 	
 	$('.read_size').each(function(e){
 		
 		//daten aus element lesen
 		var data_size = $(this).attr('data_size');
 		var obj_data_size = jQuery.parseJSON(data_size);
 		
 		//berechnen höhe
 		if(obj_data_size.height){
 		   var height = obj_data_size.height * format ;
 	       $(this).height(r(height));
 		}
 		
 		//berechnen breite
 		if(obj_data_size.width){
 		   var width = obj_data_size.width * format ;
 		   $(this).width(r(width));
 		}
 		//berechnen left
 		if(obj_data_size.left){
 		   var left = obj_data_size.left * format ;
 	       $(this).css('left' , r(left) );
 		}
 		
 		//berechnen right
 		if(obj_data_size.right){
 		   var right = obj_data_size.right * format ;
 	       $(this).css('right' , r(right) );
 		}
 		
 		
 		//berechnen background_size
 		if(obj_data_size.background_size){
 		   var background_size = obj_data_size.background_size * format ;
 	       $(this).css('background-size' , r(background_size));
 		}
 		
         //berechnen TOP
 		if(obj_data_size.top){
 		   var top = obj_data_size.top * format ;
 	       $(this).css('top', r(top) );
 		}
 		
 		//margin
 		 if(obj_data_size.margin){
 		    var margin = obj_data_size.margin * format ;
 		    $(this).css('margin', r(margin) );	
 		}
 		//margin top
 		if(obj_data_size.margin_top){
 		    var margin_top = obj_data_size.margin_top * format ;
 		   $(this).css('margin-top', r(margin_top) );
 		 }
 		 
 		//margin left
 		 if(obj_data_size.margin_left){
 		    var margin_left = obj_data_size.margin_left * format ;
 		    $(this).css('margin-left', r(margin_left) );	
 		}
 		//margin right
 		 if(obj_data_size.margin_right){
 		    var margin_right = obj_data_size.margin_right * format ;
 		    $(this).css('margin-right', r(margin_right) );	
 		}
 		//margin bottom
 		 if(obj_data_size.margin_bottom){
 		    var margin_bottom = obj_data_size.margin_bottom * format ;
 		    $(this).css('margin-bottom', r(margin_bottom) );	
 		}
 		
 		//padding right
 		 if(obj_data_size.padding){
 		    var padding = obj_data_size.padding * format ;
 		    $(this).css('padding', r(padding) );	
 		}
 		//font-size
 		if(obj_data_size.font_size){
 		   var font_size = obj_data_size.font_size * format ;
 		    $(this).css('font-size', r(font_size) );	
 		}
 		//font-size
 		if(obj_data_size.min_width){
 		   var min_width = obj_data_size.min_width * format ;
 		    $(this).css('min-width', r(min_width) );	
 		}
 		//max_width
 		if(obj_data_size.max_width){
 		   var max_width = obj_data_size.max_width * format ;
 		    $(this).css('max-width', r(max_width) );	
 		}
 		//max_height
 		if(obj_data_size.max_height){
 		   var max_height = obj_data_size.max_height * format ;
 		    $(this).css('max-height', r(max_height) );	
 		}
 		
 		//table_padding_td
 		if(obj_data_size.table_padding_td){
 		   var table_padding_td = obj_data_size.table_padding_td * format ;
 		    $(this).find('td').css('padding-top', table_padding_td).css('padding-bottom', table_padding_td);	
 		}
 	});
 }

 function create_all_players(){ 
          
 		 
 		 $('.generatenewplayer').each(function(index, element ){
 			   //daten auslesen
 			   var daten = $(this).data('playerid');
 			   var aktuell = $(this); 
 			   $.ajax({  //type: "POST",  
 						 url: "https://www.vradio.info/EXTERN/player/ajax_create_player.php?id="+daten ,

 						 dataType : 'html',
 						 //data: {id:daten},

 												//init Player

 															  $('#player_'+playerdaten.id).find('.jp-play').click(function(e) {
 																  if($(this).attr('loaded') != 'true'){

 																   //set loaded
 																   $(this).attr('loaded' , 'true');
 																  }
 															  });
 															  },
 												  swfPath: playerdaten.swf ,
 									  cssSelectorAncestor: '#player_'+playerdaten.id ,
 												 supplied: "mp3",
 													wmode: "window"
 															   }).bind($.jPlayer.event.play, function(){ 
 																		 $(this).jPlayer("pauseOthers");
 																});
 						});
 					  }
 				});
 		});
 }

 function setCookie(name,value,days) {
     var expires = "";
     if (days) {
         var date = new Date();
         date.setTime(date.getTime() + (days*24*60*60*1000));
         expires = "; expires=" + date.toUTCString();
     }
     document.cookie = name + "=" + (value || "")  + expires + "; path=/";
 }
 function getCookie(name) {
     var nameEQ = name + "=";
     var ca = document.cookie.split(';');
     for(var i=0;i < ca.length;i++) {
         var c = ca[i];
         while (c.charAt(0)==' ') c = c.substring(1,c.length);
         if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
     }
     return null;
 }
 function eraseCookie(name) {   
     document.cookie = name+'=; Max-Age=-99999999;';  
 }
