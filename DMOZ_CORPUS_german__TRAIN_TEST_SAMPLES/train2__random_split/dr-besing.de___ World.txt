 * hoverIntent is similar to jQuery's built-in "hover" function except that
 * instead of firing the onMouseOver event immediately, hoverIntent checks
 * to see if the user's mouse has slowed down (beneath the sensitivity
 * threshold) before firing the onMouseOver event.
 *
 * hoverIntent r5 // 2007.03.27 // jQuery 1.1.2+
 * <http://cherne.net/brian/resources/jquery.hoverIntent.html>
 *
 * hoverIntent is currently available for use in all personal or commercial

 *
 * // basic usage (just like .hover) receives onMouseOver and onMouseOut functions
 * $("ul li").hoverIntent( showNav , hideNav );
 *
 * // advanced usage receives configuration object only
 * $("ul li").hoverIntent({
 *	sensitivity: 7, // number = sensitivity threshold (must be 1 or higher)
 *	interval: 100,   // number = milliseconds of polling interval
 *	over: showNav,  // function = onMouseOver callback (required)
 *	timeout: 0,   // number = milliseconds delay before onMouseOut function call
 *	out: hideNav    // function = onMouseOut callback (required)
 * });
 *
 * @param  f  onMouseOver function || An object with configuration options
 * @param  g  onMouseOut function  || Nothing (use configuration options object)
 * @author    Brian Cherne <brian@cherne.net>
 */
 (function($) {
 	$.fn.hoverIntent = function(f,g) {
 		// default configuration options
 		var cfg = {
 			sensitivity: 1,
 			interval: 0,
 			timeout: 0
 		};
 		// override configuration options with user supplied object
 		cfg = $.extend(cfg, g ? { over: f, out: g } : f );
 
 		// instantiate variables
 		// cX, cY = current X and Y position of mouse, updated by mousemove event
 		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
 		var cX, cY, pX, pY;
 
 		// A private function for getting mouse position
 		var track = function(ev) {
 			cX = ev.pageX;
 			cY = ev.pageY;
 		};
 
 		// A private function for comparing current and previous mouse position
 		var compare = function(ev,ob) {
 			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
 			// compare mouse positions to see if they've crossed the threshold
 			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) < cfg.sensitivity ) {
 				$(ob).unbind("mousemove",track);
 				// set hoverIntent state to true (so mouseOut can be called)
 				ob.hoverIntent_s = 1;
 				return cfg.over.apply(ob,[ev]);
 			} else {
 				// set previous coordinates for next time
 				pX = cX; pY = cY;
 				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
 				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
 			}

 			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);

 			return cfg.out.apply(ob,[ev]);
 		};

         <h1>
          Ihr Dermatologie- und Laserzentrum zwischen Starnberg und München im Würmtal
         </h1>
         <p>
          In unserer modern ausgestatteten Praxis bieten wir ein umfassendes Behandlungsspektrum nach dem neuesten Stand des medizinischen Wissens zu allen Themen rund um Ihre Hautgesundheit.
         </p>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Haut, Haare, Nägel
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Sie leiden unter Akne, Ekzemen, Haarausfall, Pilzerkrankungen oder Psoriasis? Durch lokale und systemische Therapien, optimierte Hautpflege, den Einsatz orthomolekularer Medizin oder neue Therapieformen wie platelet-rich Plasma (PRP) bei Haarausfall lässt sich die Lebensqualität verbessern. Medizinische Kosmetik bei Akne, Laserbehandlung bei Nagelpilz und medizinische Fußpflege ergänzen die kompetenten Leistungen der klassischen und operativen Dermatologie hervorragend. Nutzen Sie die weitreichenden Behandlungsmöglichkeiten unserer Praxis in Gauting, um sich möglichst schnell von lästigen Hauterkrankungen zu befreien.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Digitale Hautkrebsvorsorge
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Immer mehr Menschen in Deutschland erkranken an Melanom und hellem Hautkrebs. Dermatologen empfehlen dringend ein jährliches Screening der Haut. Um erkranktes Gewebe frühzeitig zu entdecken, nutzen wir neueste digitale Verfahren der Hautkrebsvorsorge.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Digitale Hautkrebsvorsorge
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Bei Hautkrebs-Vorstufen (aktinische Keratosen) und einigen Hautkarzinomen lassen sich auch große lichtgeschädigte Areale narbenfrei ohne Operation mit der Photodynamischen Therapie behandeln. Mit der neuen Methode der Tageslicht-PDT steht eine fast schmerzfreie Variante zur Verfügung, die Sie mit unserer professionellen Unterstützung auch bequem zu Hause durchführen können.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Innovative Lasertherapie
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Hochwertige Premium-Lasersysteme sind aus der modernen Dermatologie nicht mehr wegzudenken: Altersflecken, störende Hautveränderungen im Gesicht und am Körper, Besenreiser, Äderchen oder Blutschwämmchen, Narben oder Dehnungsstreifen beseitigen Laser schonend und effektiv.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Operative Dermatologie
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Die Dermatochirurgie gehört zu unseren Schwerpunkten. Ein eingespieltes Team führt alle Operationen in unserer Gautinger Praxis ambulant durch. Langjährige Erfahrung und moderne Technologie sorgen für überzeugende Ergebnisse.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Hautkrebstherapie: Dermato-Onkologie und Nachsorge
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Bei Melanom und Karzinomen der Haut (Basalzellkarzinom, Spinalzellkarzinom, Morbus Bowen etc.) therapieren wir nach den neuesten Leitlinien und schaffen Sicherheit durch regelmäßige Nachsorge.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Ästhetische Medizin
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Jugendliche Haut bis ins hohe Alter – wer wünscht sich das nicht? Wir stimmen ästhetische Behandlungskonzepte selbstverständlich auf Ihre Wünsche und die Bedürfnisse Ihrer Haut ab. Wir beraten Sie zu den vielfältigen Möglichkeiten der Faltentherapie mit Botulinumtoxin, Hyaluronsäure und modernem Fadenlift, oder zu Volumen- und Konturverbesserung mit Fillern und PDO-Fäden. Zur Verjüngung der Hautstruktur setzen wir den fraktionalen Laser, Microneedling, Peelings und Mesotherapie im Bereich von Gesicht, Hals, Dekolleté und Händen ein. Unsere Hautarztpraxis im Würmtal zwischen Starnberg und München ist ein Kompetenzzentrum für ästhetische Medizin. Das umfangreiche Behandlungsspektrum wird durch das assoziierte fachärztlich geleitete Institut DermaCult mit Beratung zu Hautpflege und Behandlungen in den Bereichen Kosmetik und AntiAging optimal ergänzt.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Allergologie
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Heuschnupfen, Lebensmittelallergie oder Hand-Ekzeme: Sie vermuten, eine Allergie entwickelt zu haben? Wir nehmen alle notwendigen Tests vor, die Ihnen Gewissheit verschaffen. Oft lassen sich die Symptome schon durch gezielte Verhaltensänderungen im Alltag lindern.
           </p>
          </div>
         </div>
         <div class="su-spoiler su-spoiler-style-default su-spoiler-icon-plus-square-1 su-spoiler-closed">
          <div class="su-spoiler-title">
           <span class="su-spoiler-icon">
           </span>
           Kinderdermatologie
          </div>
          <div class="su-spoiler-content su-clearfix">
           <p>
            Kinderdermatologie erfordert besonders viel Fingerspitzengefühl. Dr. Gudrun Besing nutzt ihre langjährige klinische Erfahrung zum Wohl ihrer kleinen Patienten. Sie leitet alle notwendigen diagnostischen Schritte ein, berät zu individuellen Therapie-Alternativen und zur optimalen Pflege der empfindlichen Kinderhaut.
           </p>
          </div>
         </div>
         <p>
          Professionelles Qualitätsmanagement und kontinuierliche Fortbildung des gesamten Teams sichern Ihnen dabei beste Behandlungserfolge. Dr. Gudrun Besing hat Zertifikate in Dermatologischer Lasertherapie und Medizinisch-Dermatologischer Kosmetologie erworben und ist Laserschutzbeauftragte. Gerne können Sie sich persönlich von unserem Know-how überzeugen, wir freuen uns auf Sie.
         </p>
         <p>
          Bei Fragen zu den Behandlungen oder zur Terminvereinbarung stehen wir Ihnen gerne telefonisch zur Verfügung. Außerhalb unserer Praxisöffnungszeiten können Sie
          <a href="http://www.dr-besing.de/termine_online.html" target="_blank">
           Termine online
          </a>
          vereinbaren.
         </p>
