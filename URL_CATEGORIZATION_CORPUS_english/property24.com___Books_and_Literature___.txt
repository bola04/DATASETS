    var loginFormValidation = $("#loginFormModal").validate({
                 rules: {
                     loginUsername: {
                         required: true,
                         email: true
                     },
                     loginPassword: {
                         required: true
                     }
                 },
                 messages: {
                     loginUsername: {
                         required: "Please enter your email address.",
                         email: "Please enter a valid email address.",
                     },
                     loginPassword: "Please enter your password.",
                 },
                 submitHandler: function () {
                     $("#loginSpinner").show();
                     $("#loginSubmit").on("click", false);
                     var userName = $("#loginUsername").val();
                     var passWord = $("#loginPassword").val();
                     var rememberMe = $("#loginRememberMe").is(":checked");
 
                     Property24.LoginAndRegisterPage.Login('https://www.property24.com/User/LoginModal', userName, passWord, rememberMe);
                     $("#loginSpinner").hide();
                 }
             });
 
             var verificationCodeFormValidation = $("#verificationCodeFormModal").validate({
                 rules: {
                     loginVerificationCode: {
                         required: true
                     }
                 },
                 messages: {
                     loginVerificationCode: {
                         required: "Verification code required."
                     }
                 },
                 submitHandler: function () {
                     $("#verificationCodeSpinner").show();
                     $("#verificationCodeSubmit").on("click", false);
                     var verificationCode = $("#loginVerificationCode").val();
 
                     Property24.LoginAndRegisterPage.ValidateVerificationCode('https://www.property24.com/User/ValidateVerificationCodeModal', verificationCode);
                     $("#verificationCodeSpinner").hide();
                 }
             });
 
             var regFormValidation = $("#regFormModal").validate({
                 rules: {
                     regUsername: {
                         required: true,
                         email: true
                     },
                     regPassword: {
                         required: true
                     },
                     regTermsAndConditions: {
                         required: true
                     },
                 },
                 messages: {
                     regUsername: {
                         required: "Please enter your email address.",
                         email: "Please enter a valid email address.",
                     },
                     regPassword: "Please enter your password.",
                     regTermsAndConditions: "Please agree to our terms and conditions.",
                 },
                 // custom error placement for T&C checkbox
                 errorPlacement: function (error, element) {
                     if (element.attr("type") == "checkbox") {
                         error.insertAfter($(element).parents('label'));
                     }
                     else {
                         error.insertAfter(element); // standard behaviour
                     }
                 },
                 submitHandler: function () {
                     $("#regSpinner").show();
                     $("#regSubmit").on("click", false);
                     var userName = $("#regUsername").val();
                     var passWord = $("#regPassword").val();
                     var termsAndConditions = $("#regTermsAndConditions").is(":checked");
 
                     Property24.LoginAndRegisterPage.Register('https://www.property24.com/User/SignUpModal', userName, passWord, termsAndConditions);
 
                     $("#regSpinner").hide();
                 }
             });
 
             $("#regUsername, #regPassword").on('keyup', function () {
                 var $errorElement = $("#regModalInnerError");
                 if ($errorElement.text() != "") {
                     $errorElement.text("")
                 }
             });
 
             var RegisterLogin = { Login: 0, Register: 1 };
             var regLogin = RegisterLogin.Login;
             var redirect = undefined;
 
             $(".open-loginModal").on("click", function () {
                 regLogin = $(this).data("id") == 1 ? RegisterLogin.Register : RegisterLogin.Login;
 
                 var redirectUrl = $(this).data("redirecturl");
                 if (redirectUrl != undefined) {
                     redirect = redirectUrl;
                 }
             });
 
             $("#loginUsername,#regUsername").on("blur", function () {
                 $(".js_emailSuggestions").remove();
                 $(this).mailcheck({
                     suggested: function (element, suggestion) {
                         element.parent().append('<div class="error js_emailSuggestions">Did you mean: ' + suggestion.full + ' ?</div>');
                         $(".js_emailSuggestions").click(function () {
                             element.val(suggestion.full);
                             $(".js_emailSuggestions").remove();
                         });
                     },
                     empty: function (element) {
                         $(".js_emailSuggestions").remove();
                     }
                 });
             });
 
             $("#loginModal").on("shown.bs.modal", function () {
                
                 $(this).find(".nav-tabs li").each(function () { $(this).removeClass("active"); });
                 $(this).find(".nav-tabs li").eq(regLogin).addClass("active");
 
                 if (regLogin == RegisterLogin.Login) {
                     $("#loginTab").addClass("active");
                     $("#registerTab").removeClass("active");
                     $("#loginUsername").focus();
                 }
                 else if (regLogin == RegisterLogin.Register) {
                     $("#loginTab").removeClass("active");
                     $("#registerTab").addClass("active");
                     $("#regUsername").focus();
                 }
             });
 
             $("#loginModal").on("hidden.bs.modal", function () {
                 $("#loginShowError").hide();
                 $("#loginSpinner").hide();
                 $("#loginSubmit").off("click", false);
                 loginFormValidation.resetForm();
 
                 $("#regShowError").hide();
                 $("#regSpinner").hide();
                 $("#regSubmit").off("click", false);
                 $("#registerSuccessTab").hide();
                 regFormValidation.resetForm();
 
 
                 Property24.LoginAndRegisterPage.HideVerificationCodeForm();
                 $("#verificationCodeSubmit").off("click", false);
                 verificationCodeFormValidation.resetForm();
                 
             });
 
             function handleTwoStepVerificationResponse(response)
             {
                 if (response.IsSuccess)
                 {
                     $("#verificationCodesuccessful").removeClass("hide");
                 }
                 else if (response.NoSession)
                 {
                     Property24.LoginAndRegisterPage.HideVerificationCodeForm();
                     Property24.LoginAndRegisterPage.ShowLoginPanel();
                     $("#loginSubmit").off("click", false);
                 }
             }
 
             $("#resendVerificationCode").on("click", function ()
             {
                 $.get('/User/ResendVerificationCode')
                     .done(function (response)
                     {
                         handleTwoStepVerificationResponse(response);
                     });
                 return false;
             });
