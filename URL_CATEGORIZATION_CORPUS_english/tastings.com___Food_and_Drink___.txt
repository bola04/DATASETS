         function SetResizeHeight() {
             $('#spaImageHeight').html($("#divSwiperTopLeft img:nth-child(2)").height());
             $("#divSwiperTopRight").height($("#divSwiperTopLeft").height());
             //$(".m-home-top-banner-right-bottom-left").css('top', $('#text').outerHeight() + 'px');
 
             nContainerWidth = oContainer.width();
             if (nContainerWidth >= 600 && nContainerWidth <= 900) {
                 var nFontPct = (nContainerWidth - 600) / 300;
 
                 var nFontSize = parseInt(36 * nFontPct);
                 if (nFontSize < 22) {
                     nFontSize = 22
                 }
 
                 var nPaddingTop = nFontSize;
                 if (nFontSize > 28) {
                     nPaddingTop = 28
                 }
 
                 var nPaddingBottom = parseInt(18 * nFontPct);
                 oHomeTopBannerRightTitle.css("font-size", nFontSize + 'px');
                 oHomeTopBannerRightTitle.css("line-height", nFontSize + 'px');
                 oHomeTopBannerRightTitle.css("padding-bottom", nPaddingBottom + 'px');
 
                 nFontSize = parseInt(18 * nFontPct);
                 if (nFontSize < 12) {
                     nFontSize = 12
                 }
                 var nLineHeight = parseInt(28 * nFontPct);
                 if (nLineHeight < 20) {
                     nLineHeight = 20
                 }
                 oHomeTopBannerRightCaption.css("font-size", nFontSize + 'px');
                 oHomeTopBannerRightCaption.css("line-height", nLineHeight + 'px');
                 oHomeTopBannerRightTop.css("padding-top", nPaddingTop + 'px')
             }

         function GetHomeSlideIncrement() {
             var nSlideIncrement = 2;
             var innerWidth = window.innerWidth;
             if (innerWidth >= 634) { nSlideIncrement = 2 }
             if (innerWidth >= 748) { nSlideIncrement = 3 }
             if (innerWidth >= 863) { nSlideIncrement = 4 }
             if (innerWidth >= 976) { nSlideIncrement = 5 }
 
             return nSlideIncrement;
         }

         function SwiperMenuItemClick(pThis, pKindWeb) {
             $("#divSwiperMenu" + pKindWeb + " .m-tabs-swiper-menu-link").removeClass("m-tabs-swiper-menu-link-active");
             $(pThis).addClass("m-tabs-swiper-menu-link-active");

         function SwiperRefresh(pKindWeb, pMenuTitle, pDataSwiperButton) {
 
             var sQS = "&KindCode=" + encodeURIComponent(pKindWeb);
             sQS += "&Title=" + encodeURIComponent(pMenuTitle);
             sQS += "&DisplayYear=" + encodeURIComponent($("#inpSwiper" + pKindWeb + "DisplayYear").val());
             sQS += "&DisplayMonth=" + encodeURIComponent($("#inpSwiper" + pKindWeb + "DisplayMonth").val());
             sQS += "&DisplayButton=" + encodeURIComponent(pDataSwiperButton);
             if (pDataSwiperButton == "TOP-PICKS") {
                 sQS += "&CarouselFeatureTopPicks=" + encodeURIComponent($("#inpSwiper" + pKindWeb + "CarouselFeatureTopPicks").val());
             }
             if (pDataSwiperButton == "NEW-PRODUCTS") {
                 sQS += "&CarouselFeatureNewProducts=" + encodeURIComponent($("#inpSwiper" + pKindWeb + "CarouselFeatureNewProducts").val());
             }
             if (pDataSwiperButton == "BEST-BUYS") {
                 sQS += "&CarouselFeatureBestBuys=" + encodeURIComponent($("#inpSwiper" + pKindWeb + "CarouselFeatureBestBuys").val());
             }
 
             ProcessRequest.GetCarouselTopPicksListService(sQS, callbackSwiperRefresh);
 
             if (pDataSwiperButton == "TOP-PICKS") {
                 $("#spaSwiperTitle" + pKindWeb).html(pMenuTitle);
                 $("#divSwiperMenu" + pKindWeb).show();
                 $("#btnSwiperMenuPrev" + pKindWeb).show();
                 $("#btnSwiperMenuNext" + pKindWeb).show();
                 $("#divSwiperMenuBlank" + pKindWeb).hide();
             }
             if (pDataSwiperButton == "NEW-PRODUCTS") {
                 $("#spaSwiperTitle" + pKindWeb).html("New Products");
                 $("#divSwiperMenu" + pKindWeb).hide();
                 $("#btnSwiperMenuPrev" + pKindWeb).hide();
                 $("#btnSwiperMenuNext" + pKindWeb).hide();
                 $("#divSwiperMenuBlank" + pKindWeb).show();
             }
             if (pDataSwiperButton == "BEST-BUYS") {
                 $("#spaSwiperTitle" + pKindWeb).html("Best Buys");
                 $("#divSwiperMenu" + pKindWeb).hide();
                 $("#btnSwiperMenuPrev" + pKindWeb).hide();
                 $("#btnSwiperMenuNext" + pKindWeb).hide();
                 $("#divSwiperMenuBlank" + pKindWeb).show();
             }
 
         }
 
         function callbackSwiperRefresh(pResult) {
 
             var oResult = jQuery.parseJSON(pResult);
             var sCarouselItemCount = oResult.RecordCount
             var sCarouselMenuComment = oResult.MenuComment;
             var sCarouselHtmlM = oResult.HtmlListM;
             var sMoreLink = oResult.MoreLink;
             var sKindWeb = oResult.KindWeb;
 
             switch (sKindWeb) {
                 case "Spirits":
                     oSwiperSpirits.destroy(true, true);
                     $('#divSwiperWrapperSpirits').html(sCarouselHtmlM);
                     oSwiperSpirits = Swiper('#divSwiperSpirits', oSwiperSpiritsSettings);
                     break;
                 case "Wine":
                     oSwiperWine.destroy(true, true);
                     $('#divSwiperWrapperWine').html(sCarouselHtmlM);
                     oSwiperWine = Swiper('#divSwiperWine', oSwiperWineSettings);
                     break;
                 case "Beer":
                     oSwiperBeer.destroy(true, true);
                     $('#divSwiperWrapperBeer').html(sCarouselHtmlM);
                     oSwiperBeer = Swiper('#divSwiperBeer', oSwiperBeerSettings);
                     break;
             }
             $("#inpSwiper" + sKindWeb + "ItemCount").val = oResult.RecordCount;
             $("#lnkMoreReviews" + sKindWeb).attr("href", oResult.MoreLink);
             $("#lnkMoreReviews" + sKindWeb + "Side").attr("href", oResult.MoreLink);
             $("#spaHomeComment" + sKindWeb).html(oResult.MenuComment);
 
             if (parseInt(oResult.RecordCount) > 5) {
                 $('#btnSwiperPrev' + sKindWeb).show();
                 $('#btnSwiperNext' + sKindWeb).show();
             } else {
                 $('#btnSwiperPrev' + sKindWeb).hide();
                 $('#btnSwiperNext' + sKindWeb).hide();
             }
 
             switch (sKindWeb) {
                 case "Spirits":
                     oSwiperSpirits.update();
                     break;
                 case "Wine":
                     oSwiperWine.update();
                     break;
                 case "Beer":
                     oSwiperBeer.update();
                     break;
             }
 
             SetReviewSlideHover();
             SetResizeHeight();
 
         }
