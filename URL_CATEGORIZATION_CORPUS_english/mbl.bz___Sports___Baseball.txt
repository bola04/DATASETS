          var countdownApp = new Vue({
 		el: '#countdown-app',
 		data: {
 			title: null,
 			endDate: null,
 			countdown: 0,
 			admin: false
 		},
 		computed: {
 			seconds: function() {
 				return Math.floor((this.countdown % (1000 * 60)) / 1000);
 			},
 			minutes: function() {
 				return Math.floor((this.countdown % (1000 * 60 * 60)) / (1000 * 60));
 			},
 			hours: function() {
 				return Math.floor((this.countdown % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
 			},
 			days: function() {
 				return Math.floor(this.countdown / (1000 * 60 * 60 * 24));
 			}
 		},
 		created: function () {
 			this.fetchCountdown = function () {
 				console.debug('Countdown: fetching countdown');
 
 				this.$http.get('/api/countdown').then(function (response) {
 					return response.json();
 				})
 				.then(function (data) {
 					countdownApp.title = data.title;
 					countdownApp.date = data.date;
 					console.debug('Countdown: fetching countdown finished');
 
 					countdownApp.startCountdown();
 				})
 				.catch(function (response) {
 					if (response.status === 404) {
 						console.debug('Countdown: No countdown set, hiding widget');
 					} else {
 						console.error('Countdown: Unable to fetch countdown', response);
 					}
 				});
 
 				console.debug('Countdown: created');
 			};
 
 			this.startCountdown = function () {
 				var countDownDate = new Date(this.date).getTime();
 				var now = (new Date()).getTime();
 				this.countdown = countDownDate - now;
 
 				var timer = setInterval(function() {
 					var now = (new Date()).getTime();
 					countdownApp.countdown = countDownDate - now;
 
 					// stop timer if finished
 					if (countdownApp.countdown <= 0) {
 						clearInterval(timer);
 						countdownApp.countdown = 0;
 						console.debug("Countdown: Ended, hiding widget");
 					}
 				}, 1000);
 
 				Vue.nextTick(function () {
 					console.debug('Countdown: updating grid');
 					updateGrid();
 				});
 
 				console.debug('Countdown: Started');
 			};
 
 			console.debug('Countdown: created');
 		},
 		mounted: function () {
 			this.$nextTick(function() {
 				this.fetchCountdown();
 			});
 
 			console.debug('Countdown: mounted');
 		}
 	});

 					var geocoder = new google.maps.Geocoder();
 
 					console.debug('Weather: fetching location date from google maps');
 
 					geocoder.geocode({'latLng': latLng}, function (results, status) {
 						if (status !== google.maps.GeocoderStatus.OK) {
 							console.error("Weather: Geocoder failed due to: " + status);
 							reject();
 							return;
 						}
 
 						if (!results[1]) {
 							console.error("Weather: No Geocoder results found");
 							reject();
 							return;
 						}
 
 						var location = {};
 						for (var i = 0; i < results[0].address_components.length; i++) {
 							for (var b = 0; b < results[0].address_components[i].types.length; b++) {
 								if (results[0].address_components[i].types[b] === "administrative_area_level_1") {
 									location.state = results[0].address_components[i].short_name;
 								} else if (results[0].address_components[i].types[b] === "locality") {
 									location.city = results[0].address_components[i].long_name;
 								}
 							}
 						}
 
 						console.debug('Weather: fetching location date from google maps finished');
 
 						resolve(location);
 					});
