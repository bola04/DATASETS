 		var curSelObj = $('.warBody li.current');
 		var nextSelObj = $('.warBody li.current + li');
 		$('.warBody li').removeClass('current');
 
 		window.setTimeout(function(){
 			var childPos = $(nextSelObj).offset();
 			var parentPos = $(nextSelObj).parent().offset();
 			var childOffset = {
 			    top: childPos.top - parentPos.top,
 			    left: childPos.left - parentPos.left
 			}
 
 			$('.warBodyCon').animate({scrollLeft: childOffset.left - 75}, 500);
 			
 			$(nextSelObj).addClass('current');
 			$('.warHeader li').removeClass('active');
 			var curYear = $(nextSelObj).find('.expand').attr('ref');
 			console.log(curYear);
 			$('#war-' + curYear).addClass('active');
 		}, 550);
 
 	});
 
 	$('.warHeader li').each(function(){
 		var startDate = 1754;
 		var endDiff = 29;
 		cur_date = $(this).attr('ref');
 		cur_date = cur_date - startDate;
 		cur_diff = (cur_date / endDiff) * 100;
 		//console.log(cur_diff);
 		$(this).css('left',cur_diff + '%');
 
 	});
 
 	$('.warHeader li').click(function(){
 		var targetID = $(this).attr('id').replace('war-','');
 		$('.warHeader li').removeClass('active');
 		$(this).addClass('active');
 		closeContent();
 		openContent(targetID);
 
 	});
 
 	$('.warBody .expand').click(function(){
 		closeContent();
 
 		var tarObj = $(this);
 		var parentID = $(this).attr('ref');
 		console.log(parentID);
 		$('.warHeader li').removeClass('active');
 		$('#war-' + parentID).addClass('active');
 
 		window.setTimeout(function(){
 			var childPos = $(tarObj).parent().parent().offset();
 			var parentPos = $(tarObj).parent().parent().parent().offset();
 			var childOffset = {
 			    top: childPos.top - parentPos.top,
 			    left: childPos.left - parentPos.left
 			}
 
 			$('.warBodyCon').animate({scrollLeft: childOffset.left - 75}, 500);
 			$(tarObj).parent().parent().addClass('current');
 		}, 600);
 
 	});
 
 	$('.warBody .close').click(function(){
 		closeContent();
 	});
 
 	function closeContent(){
 		$('.warBody li').removeClass('current');

At the start of the war, the French North American colonies had a population of roughly 60,000 European settlers, compared with 2 million in the British North American colonies.               <p>
               President General Joseph W. Dooley 2013-2014 (and Genealogist General 2008-2011) has published an article, "Who Were The Parents of Elizabeth (Terhune) Earle of Bergen County, New Jersey?" in The NEHG Register, the Spring 2018 Volume 172 issue.
              </p>
              <p>
               Elizabeth Terhune (1756–1847) was married to the American Revolutionary War soldier, Morris Earle (1757–1833).[1] Since at least 1924 there has been some question about the identity of her parents, and multiple candidates have been put forward. This article will show that the parents of Elizabeth Terhune were Albert Terhune and Jannetje Vanderhoof.
               <br/>
               In History and Genealogy of the Earles of Secaucus, Rev. Isaac Newton Earle states that Elizabeth’s parents were Albert Terhune and Betsy Van der Linde.[2] In Descendants of Albert Albertse Terhune of 1654, Herbert S. Ackerman shows...
              </p>
