     function alertmsg(title, msg, autohide) {
         $("#alertpop div.title").html(title);
         $("#alertpop div.msg").html(msg);
         //center div in window. doing this with css brurs the text in I.E.
         $(window).off('resize');
         $(window).on('resize', function () {
             $("#alertpop").css("left", Math.round(($(window).width() - $("#alertpop").width()) / 2));
             $("#alertpop").css("top", Math.round(($(window).height() - $("#alertpop").height()) / 2));
         }).resize();
         $("#alertblanket").show();
         $("#alertpop").show();
         if (autohide) {
             alertmsgtimer = window.setTimeout("hidealertmsg()", 1000 * 5);
         }
         $("#alertpop div.close").off("click");
         $("#alertpop div.close").click(function () {
             hidealertmsg();
         });
         $("#alertblanket").off("click");
         $("#alertblanket").click(function () {
             hidealertmsg();
         });
     }
     function hidealertmsg() {
         $("#alertblanket").hide();
         $("#alertpop").hide();
     }
