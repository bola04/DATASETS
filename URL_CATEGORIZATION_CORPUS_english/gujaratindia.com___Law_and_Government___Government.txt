         grp = 0;

         if (pwd.match(/[A-Z]/)) grp++;

     function GetEncrypt(salt) {
         var pwd = document.getElementById('txtPassword');
 
         if (pwd != null && pwd.value != '' && salt != '') {
             //                var enc1 = hex_md5(pwd.value);
             //                var enc = hex_md5(salt + enc1);

             document.getElementById('txtPassword').value = enc;
 
         }
