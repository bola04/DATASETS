 function decompte_noel()
 {
 		var days_translation = 'days';
 		var hours_translation = 'hours';
 		var minutes_translation = 'minutes';
 		var seconds_translation = 'seconds';
 		if (langue == "fr") {
 			days_translation = 'jours';
 			hours_translation = 'heures';
 			minutes_translation = 'minutes';
 			seconds_translation = 'secondes';
 		}
 
 		var today = new Date();
 		var jourdenoel = new Date("December  25, 2018");
 		var nbsecondes = (jourdenoel - today) / 1000;
 		if (nbsecondes<0)
 		{
 			jourdenoel = new Date("December  25, 2019");
 			nbsecondes = (jourdenoel - today) / 1000;
 		}
 		var joursrestants = Math.floor (nbsecondes / (24 * 3600));
 		var heuresrestantes = Math.floor ((nbsecondes - (joursrestants * 24 * 3600)) / 3600);
 		var minutesrestantes = Math.floor ((nbsecondes - ((joursrestants * (24 * 3600) + heuresrestantes * 3600))) / 60);
 		var secondesrestantes = Math.floor (nbsecondes - ((joursrestants * (24 * 3600) + heuresrestantes * 3600 + minutesrestantes * 60)));
 		if (heuresrestantes<10) heuresrestantes = '0'+heuresrestantes;
 		if (minutesrestantes<10) minutesrestantes = '0'+minutesrestantes;
 		if (secondesrestantes<10) secondesrestantes = '0'+secondesrestantes;
 		$('#j_noel').text(joursrestants+' '+days_translation);
 		$('#h_noel').text(heuresrestantes+' '+hours_translation);
 		$('#m_noel').text(minutesrestantes+' '+minutes_translation);
 		$('#s_noel').text(secondesrestantes+' '+seconds_translation);
 		setTimeout("decompte_noel();",1000);
 }
