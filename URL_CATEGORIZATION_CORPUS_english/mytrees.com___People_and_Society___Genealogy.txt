 function checkAAIForm()
 {
 	var len = 0;
 	var good = true;
 	if (window.matchMedia("(min-width: 801px)").matches)
 		len = document.queryFormBig.Surname.value.length;
 	else if (window.matchMedia("(min-width: 550px) and (max-width: 800px)").matches)
 		len = document.queryFormMedium.Surname.value.length;
 	else if (window.matchMedia("(max-width: 549px)").matches)
 		len = document.queryFormSmall.Surname.value.length;
 	if (len == 0)
 	{
 		good = false
 		alert('You must include a "last name" for your search.')
 	}
 	if (window.matchMedia("(min-width: 801px)").matches)
 	{
 		len = document.queryFormBig.Year.value.length;
 		if (len == 4 && document.queryFormBig.Year.value.indexOf('Year') >= 0)
 		{
 			len = 0;
 		}
 	}
 	else if (window.matchMedia("(min-width: 550px) and (max-width: 800px)").matches)
 	{
 		len = document.queryFormMedium.Year.value.length;
 		if (len == 4 && document.queryFormMedium.Year.value.indexOf('Year') >= 0)
 		{
 			len = 0;
 		}
 	}
 	else if (window.matchMedia("(max-width: 549px)").matches)
 	{
 		len = document.queryFormSmall.Year.value.length;
 		if (len == 4 && document.queryFormSmall.Year.value.indexOf('Year') >= 0)
 		{
 			len = 0;
 		}
 	}
 	if (len == 0)
 	{
 		good = false
 		alert('You must include an estimated event year.')
 	}
 	if (good)
 	{
 		if (window.matchMedia("(min-width: 801px)").matches)
 			document.queryFormBig.submit()
 		else if (window.matchMedia("(min-width: 550px) and (max-width: 800px)").matches)
 			document.queryFormMedium.submit()
 		else if (window.matchMedia("(max-width: 549px)").matches)
 			document.queryFormSmall.submit()
 	}
 }

    /* Loop through all dropdown buttons to toggle between hiding and showing its dropdown content - This allows the user to have multiple dropdowns without any conflict */

 var i;
 for (i = 0; i < dropdown.length; i++) {
   dropdown[i].addEventListener("click", function() {
     this.classList.toggle("active");
     var dropdownContent = this.nextElementSibling;
     if (dropdownContent.style.display === "block") {
       dropdownContent.style.display = "none";
     } else {
       dropdownContent.style.display = "block";
     }
   });
