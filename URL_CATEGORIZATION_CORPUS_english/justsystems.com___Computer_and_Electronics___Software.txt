    //match height
   $(function() {
     $('#section02 .about_block > div figure ').matchHeight();
     $('#section03 .products li .logo .holder ').matchHeight();
     $('#section03 .products li p ').matchHeight();
     $('#section03 .products li h3 ').matchHeight();
   });
   // main-copy-effect(animation)
   $(window).load(function () {
     $(function(){
       $(window).scroll(function (){
           $('#mainvisual .holder').on().each(function(){
               var elemPos = $(this).offset().top;
               var scroll = $(window).scrollTop();
               var windowHeight = $(window).height();
               if (scroll > elemPos - windowHeight + windowHeight/5){
                   $(this).addClass('animate');
               }
           });
       });
   });
   });
   $(function() {
       $('#mainvisual .holder').on('inview', function(event, isInView, visiblePartX, visiblePartY) {
           if(isInView){
               $(this).stop().addClass('animate');
           }
       });
   });
   // header with scroll
   $(window).scroll(function() {
   var position = $("#mainvisual").height();
     
   if ($(this).scrollTop() > position){
       $('#header').addClass("fixed");
     }
     else{
       $('#header').removeClass("fixed");
     }
   });
   // slider
   $(document).ready(function() {
     $('#mainvisual').sliderPro({
       fade: true,
       fadeDuration: 3000,
       fadeOutPreviousSlide:4000,
       width: "100%",
       buttons: false,
       autoplayOnHover:'none',
       height: 660,
       responsive: false,
       touchSwipe:false,
       breakpoints: {
         640: {
           height: 498,
         }
       }
     });
   });
   $(function(){
      // Click at #hash-Link
      $(document).on('click', 'a[href^=#]', function() {
         // speed
         var speed = 500; // in msec
         // get #hash
         var href= $(this).attr("href");
         // get Element
         var target = $(href == "#" || href == "" ? 'html' : href);
         // get Position
         var position = target.offset().top;
          // calc Position
       var header = $(".global-nav .global-nav-in").outerHeight();
         // scroll to go
         $('body,html').animate({scrollTop:position - header}, speed, 'swing');
         return false;
      });
   });
