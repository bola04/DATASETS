     function fixBar() {
       var $div = $("#section-bar");
       if ($(window).scrollTop() > $div.data("top") - 78) { 
           $('#section-bar-pad').show();
           $('#section-bar').addClass('fixed');
           //$('#promo').addClass('scroll-padding');
       }
       else {
           $('#section-bar-pad').hide();
           $('#section-bar').removeClass('fixed');
           //$('#promo').removeClass('scroll-padding');
       }
       $scrollButtons = [];
       $('a[id*="-button"]').each(function() {
         $scrollButtons.push($(this).attr('id'));
       });
       $('a[id*="-button"]').parent('li').removeClass('active');
       for(i = 0; i < $scrollButtons.length; i++){
         if (($(window).scrollTop() > $('div[id="' + $scrollButtons[i].replace('-button','') + '"]').offset().top - 140) && ($(window).scrollTop() < $('div[id="' + $scrollButtons[i].replace('-button','') + '"]').offset().top + $('div[id="' + $scrollButtons[i].replace('-button','') + '"]').outerHeight() - 140)) {
           $('a[id="' + $scrollButtons[i] + '"]').parent('li').addClass('active');
         }
       }
     }

         $promosTL.play();
