             $(function() {
   var carouselApi = $('#j-carousel').getPRESTO_Carousel()
   var sliderApi = $('#thumb-slider .slider').scrollable({ circular: false, next: '#thumb-slider > .next', prev: '#thumb-slider > .prev', keyboard: false }).navigator({ navi: '#thumb-slider > .nav' }).data('scrollable');
   //var controls = $('#thumb-slider .control');
   var controls = $('#thumb-slider .item').not('.cloned').children('.control');
   var columnLimit = 4;
   var slideCount = 3;
 
   $('#j-carousel').bind('afterSlide', function(e, activeIndex, lastIndex) {
 
     if (lastIndex >= 0) {
       controls.eq(lastIndex).removeClass('active');
     }
     controls.eq(activeIndex).addClass('active');
 
     var activeControlSlideIndex = Math.floor( (activeIndex + columnLimit) / columnLimit ) - 1;
     var activeControlPositionInSlide = controls.eq(activeIndex).index();
     var currentSlideIndex = sliderApi.getIndex();
 
     if (lastIndex >= 0 && activeControlPositionInSlide == 0 && activeControlSlideIndex != currentSlideIndex) {
       sliderApi.seekTo(activeControlSlideIndex);
     }
   });
 
   controls.click(function(e) {
     if (!$('#j-carousel').hasClass('loading') && !$(this).hasClass('active')) {
       var i = controls.index(this);
       carouselApi.run(i);
     }
   });
 
 });
