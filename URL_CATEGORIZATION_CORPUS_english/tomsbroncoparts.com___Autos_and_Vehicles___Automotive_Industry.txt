     function linkSlide(item) {   
       var slideAmount = 0;
       if (item != currentSlide) {
         if (item < currentSlide) {
           slideAmount = (currentSlide - item)*(1024);
         }
         else {
           slideAmount = (item - currentSlide)*(-1024);
         }
         slideIt(slideAmount);
       }

     function slideIt(amt) {
       // Disable the buttons
       if (!isSliding) {
         isSliding=true;
         timer.stop();
         var slide = false;
         if (amt < 0 && currentSlide == itemCount) {
           // Go back to slide one
           amt = ((itemCount - 1) * 1024);
           dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box.png";
           currentSlide = 1;
           dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box-current.png";
           var slideArgs = {
             node: "slideProductHolder",
             top: (dojo.coords("slideProductHolder").t).toString(),
             left: (dojo.coords("slideProductHolder").l + amt).toString(),
             duration: 750,
             unit: "px"
           };
           var anim = dojo.fx.slideTo(slideArgs);
           dojo.connect(anim, "onEnd", function(){
             timer.start();
             isSliding=false;
           });
           anim.play();
         }
         // check for sliding to 5 from 1
         else if (amt > 0 && currentSlide == 1) {
           // Go back to slide one
           amt = -((itemCount - 1) * 1024);
           dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box.png";
           currentSlide = 5;
           dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box-current.png";
           var slideArgs = {
             node: "slideProductHolder",
             top: (dojo.coords("slideProductHolder").t).toString(),
             left: (dojo.coords("slideProductHolder").l + amt).toString(),
             duration: 750,
             unit: "px"
           };
           var anim = dojo.fx.slideTo(slideArgs);
           dojo.connect(anim, "onEnd", function(){
             timer.start();
             isSliding=false;
           });
           anim.play();
         }
 
         else if (amt < 0 && currentSlide != itemCount) {
             slide = true;
             dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box.png";
             currentSlide += Math.abs(amt/1024);
             dojo.byId("sliderItem"+currentSlide).src = "/images/BTN-slider-box-current.png";
         }
         else if (amt > 0 && currentSlide != 1) {
