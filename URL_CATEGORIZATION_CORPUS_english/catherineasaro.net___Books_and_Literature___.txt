         //fallback to make sure the photo transitions in even if the previous transition end doesn't fire

         },1500);
     $('#openmenu').on('click',function(e) {
        e.preventDefault();
        $('.mainnav li').toggleClass('navitemvisible');
        var openmenu=$('#openmenu').find('.navtext');
        if ($(openmenu).text()=="OPEN MENU") {
            $(openmenu).text("CLOSE MENU");
            } else {
            $(openmenu).text("OPEN MENU");
            }
        });
 });
