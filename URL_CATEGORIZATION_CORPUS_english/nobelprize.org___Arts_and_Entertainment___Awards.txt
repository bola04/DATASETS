   <script>
    $(document).ready(function() {
 	
 	var el = $('.yes-filler');
 	var originalHeight = el.height();
 	
 	var properties = {
 	   height : originalHeight * 0.9
 	}
 	
 	el.pulse(properties, {duration : 1000, pulses : -1});
 	
 	
 	// Voting
 	$('.ein-poll-alt').change(function() {
 		
 		poll_stats = [];
 		poll_stats_percent = [];
 		alt_ids = [];
 		total_votes = 0;
 		num_alts = 0;
 		alt_counter = 0;
 		
 		ein_poll_id = $(this).data('pollid');
 		ein_alt_id = $(this).data('altid');
 		
 		
 		
 		el.pulse('destroy');
 		
 		$('.poll-'+ein_poll_id+'-alt').each(function() {
 			
 			$(this).attr('disabled',true);
 			alt_ids[alt_counter] = $(this).data('altid');
 			
 			alt_counter++;
 			num_alts++;
 		});
 	
 		// Lets save da vote yao
 		$.ajax({
 			type: "POST",
 			url: "//www.nobelprize.org/prog/php/poll/api.php",
 			data: {poll_id: ein_poll_id,
 					alt_id: ein_alt_id},
 			dataType: "json",
 			beforeSend:function(){
 				//alert("sending");
 			},
 			error: function(jqXHR, textStatus, errorThrown){
 				//alert("Error: "+errorThrown);
 			},
 			success: function(data){
 				
 				$.each(data, function(index, element) {
 					
 					//self.questions[i]['id'] = element.question_id;
 					poll_stats[element.id] = [];
 					poll_stats[element.id]['votes'] = parseInt(element.votes);
 					total_votes += parseInt(element.votes);
 				});
 	
 				//# Calculate percent % stats
 				var percent_per_vote = 100/total_votes;
 				
 				var tot_percent = 0;
 				
 				for(var per_counter = 0; per_counter < num_alts; per_counter++) {
 					
 					var curr_alt_id = alt_ids[per_counter];
 					poll_stats[curr_alt_id]['percent'] = parseInt(poll_stats[curr_alt_id]['votes'] * percent_per_vote);
 					
 					tot_percent += poll_stats[curr_alt_id]['percent'];
 				}
 				
 				$('#poll-num-votes-'+ein_poll_id).html('<p style="padding: 0; margin: 0;">Answers: <b>'+total_votes+'</b></p>');
 				
 				setTimeout(function() {
 				for(var insta = 0; insta < num_alts; insta++) {
 
 					var curraltid = alt_ids[insta];
 					var curraltpercent = poll_stats[curraltid]['percent'];
 					curraltpercentminus = 100-curraltpercent;
 						
 					$('#filler-'+curraltid).css("height",curraltpercentminus+"px");
 					$('#green-filler-'+curraltid).append('<center><span style="color: #fff; font-size: 12px; font-weight: bold;">'+curraltpercent+'%</span></span>');
 				}
 				},1100);
 			}
 		});
 	});
 });
