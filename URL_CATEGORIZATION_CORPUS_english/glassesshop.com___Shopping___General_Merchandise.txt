    function set_query_string(key, value, op) {
         var current_controller = 'index';
         var url_params = [], _query_params = [], _exist_key = false;
         var _search = decodeURIComponent(window.location.search).replace('+', ' ');
         if (_search.length > 0) {
             _query_params = _search.slice(1).split('&');
         }
         if (_query_params.length > 0) {
             for (var i = 0; i < _query_params.length; i++) {
                 var urlparam = _query_params[i].split('=');
                 if (urlparam[0] == key) {
                     if (key == 'price') {
                         continue;
                     }
                     if (key == 'sort') {
                         continue;
                     }
                     _exist_key = true;
                     var url_param_value = urlparam[1].split(',');
                     var url_param_value_new = [];
                     url_param_value.push(value);
                     for (var j = 0; j < url_param_value.length; j++) {
                         if (op == 'remove' && url_param_value[j] == value) {
                             continue;
                         }
                         url_param_value_new.push(url_param_value[j]);
                     }
                     if (url_param_value_new.length > 0) {
                         url_params.push(urlparam[0] + '=' + url_param_value_new.join(','));
                     }
                 }
                 else {
                     url_params.push(_query_params[i]);
                 }
             }
         }
         if (!_exist_key) {
             url_params.push(key + '=' + value);
         }
         if (current_controller != 'category' && current_controller != 'promotion') {
             if (url_params.length > 0) {
                 location.href = '/search/filter?' + url_params.join('&');
             }
             else {
                 location.href = '/';
             }
         }
         else {
             if (url_params.length > 0) {
                 window.location.search = url_params.join('&');
             }
             else {
                 window.location.search = '';
             }
         }
     }
