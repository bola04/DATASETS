    //form validation
         function fnSendSupportRequest() {
 
             $("#ValName").text('');
             $("#ValEmailAddress").text('');
             $("#ValMessage").text('');
             var flag = 1;
 
             if ($("#Name").val() == "") {
                 flag = 0;
                 $("#ValName").text('Name is required.');
             }
 
             if ($("#EmailAddress").val() == "") {
                 flag = 0;
                 $("#ValEmailAddress").text('Email is required.');
             }
             else if (!isEmail($("#EmailAddress").val())) {
                 flag = 0;
                 $("#ValEmailAddress").text('Please enter valid email address.');
             }
             if ($("#Message").val() == "") {
                 flag = 0;
                 $("#ValMessage").text('Message is required.');
             }
 
             if (flag == 0)
                 return;
 
             var supportrequest = new Object();
             supportrequest.Name = $("#Name").val();
             supportrequest.EmailAddress = $("#EmailAddress").val();
             supportrequest.Subject = $("#Subject").val();
             supportrequest.Message = $("#Message").val();
 
             $.ajax({
                 url: '/Home/SupportRequest',
                 type: 'POST',
                 contentType: "application/json",
                 data: JSON.stringify(supportrequest),
                 success: function (response) {
                     $("#msgDiv").html(response);
                     if ($("#ResponseResult").val() == "SUCCESS") {
                         $("#Name").val('');
                         $("#EmailAddress").val('');
                         $("#Message").val('');
                     }
                 }
             })
         }
 
         function fnShowPassKeyPopup(th) {
              var StatusID = $(th).attr("data-StatusID");
 
             $("#PassKey").val('');
             $(".passkey-detail").hide();
 
             if (StatusID == '1')
                 $("#trNotOpen").show();
             else if (StatusID == '3')
                 $("#trFull").show();
             else if (StatusID == '4')
                 $("#trClosed").show();
             else
                 $(".trEnter").show();
 
             $("#EnterPassKey").attr("onClick", "ManageInvitationalButton('#"+$(th).attr("id")+"')");
             $(".tourament-invitational").modal('show');
         }
         function ManageInvitationalButton(th) {
             if ($("#PassKey").val() == "") {
                 notifyDanger("Please enter passkey first.");
                 return;
             }
             var encryptedTournamentID = $(th).attr("data-encid");
 
             var objTournament = new Object();
             objTournament.Tournament = GetTournamentObject(th);
             objTournament.PassKey = $("#PassKey").val();
 
             $.ajax({
                 url: '/Api/Tournament/ValidatePassKeyAndSelectTournament',
                 contentType: "application/json",
                 async: false,
                 method: "POST",
                 data: JSON.stringify(objTournament),
                 success: function (response) {
                     if (response.Result == 1) {
                         notifySuccess(response.Message);
                         location.href = '/Tournament/BookTournament/' + encryptedTournamentID;
                     }
                     else
                         notifyDanger(response.Message);
                 }
             })
         }
         function GetTournamentObject(th) {
             var objTournament = new Object();
             objTournament.TournamentID = $(th).attr("data-id");
             objTournament.Date = $(th).attr("data-Date");
             objTournament.Name = $(th).attr("data-name");
             objTournament.Rating = $(th).attr("data-Division");
             objTournament.Gender = $(th).attr("data-Gender");
             objTournament.Location = $(th).attr("data-Location");
             objTournament.EntryFee = $(th).attr("data-EntryFee");
             objTournament.TournamentDirector = $(th).attr("data-TournamentDirector");
             objTournament.MaxPlayers = $(th).attr("data-MaxPlayers");
             objTournament.MinPlayers = $(th).attr("data-MinPlayers");
             objTournament.RequiredFemalePlayers = $(th).attr("data-RequiredFemalePlayers");
             objTournament.RequiredMalePlayers = $(th).attr("data-RequiredMalePlayers");
             return objTournament;
         }
         function fnAddTournament(th) {
             var objTournament = GetTournamentObject(th);
             var encryptedTournamentID = $(th).attr("data-encid");
 
             $.ajax({
                 url: '/Api/Tournament/SelectTournament',
                 contentType: "application/json",
                 async: false,
                 method: "POST",
                 data: JSON.stringify(objTournament),
                 success: function (response) {
                     if (response.Result) {
                         notifySuccess(response.Message);
                         location.href = '/Tournament/BookTournament/' + encryptedTournamentID;
                     }
                     else
                         notifyDanger(response.Message);
                 }
             })
         }
