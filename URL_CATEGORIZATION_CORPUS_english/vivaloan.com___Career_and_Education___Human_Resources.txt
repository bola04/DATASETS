   var submitForm = false;
   $("#returning-modal-form").on('submit', function(e) {
     // If our variable is false, stop the default action.
     // The first time 'submit' is triggered, we should prevent the default action
     if (!submitForm) {
       e.preventDefault();
     }
     var form = $(this);
     form.parsley().validate();
     // If the form is valid
     if (form.parsley().isValid()) {
       // Set the variable to true, so that when the 'submit' is triggered again, it doesn't
       // prevent the default action
       submitForm = true;
 
       $("#loading-progress-over").addClass("spin-on");
       $("#returning-modal-form #submitbutton").addClass("submit-progress-loading");
       setTimeout(function() {
         // Trigger form submit
       $("#returning-modal-form #submitbutton").addClass("submit-progress-complete");
     }, 4750);
       setTimeout(function() {
         // Trigger form submit
         form.submit();
       }, 6300);
     }
     // else {
     // There could be times when the form is valid and then becames invalid. In these cases,
