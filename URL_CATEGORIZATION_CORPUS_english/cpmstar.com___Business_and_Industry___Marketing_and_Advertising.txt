         var docViewTop = $(window).scrollTop();
         var docViewBottom = docViewTop + $(window).height();

    function showSnackbar() {
             var x = document.getElementById("snackbar");
             x.className = "show";
 
             var y = document.getElementById("loginarrowwrapper");
             if(y) y.style.bottom = "50px";
 
 
             localStorage.setItem("snackbarClosedDate", "1970/01/01 00:00 AM");
             
         }
         function closeSnackbar() {
             var x = document.getElementById("snackbar");
             setTimeout(function(){ x.className = x.className.replace("show", "hide"); }, 0);
             var y = document.getElementById("loginarrowwrapper");
             if(y) y.stylebottom = "0px";
 
             var nowStr = new Date().toString();
             localStorage.setItem("snackbarClosedDate", nowStr);
         }
 
         var snackbarUpdateDate = new Date(Date.parse("2018/05/24 00:00 AM"));
         var snackbarClosedDateStr = localStorage.getItem("snackbarClosedDate");        
         var snackbarClosedDate = new Date(Date.parse("1970/01/01 00:00 AM"));
         if(snackbarClosedDateStr) {
             try { 
                 snackbarClosedDate = new Date(Date.parse(snackbarClosedDateStr));
             } catch(e) {
             }
         }        
         if(snackbarClosedDate == null || snackbarClosedDate < snackbarUpdateDate) {
             showSnackbar();
         }
