           <script type="text/javascript">
            $(document).ready(function () {
         if ($('.MoreShowsBtn').length > 0)
         {
             $('.MoreShowsBtn').on('click', function (e) {
                 if ($(this).hasClass("unclickable")) {
                     e.preventDefault();
                 } else {
                     $(this).addClass("unclickable");
                     var direction = $(this).find('i').data('direction');
                     scrollShows($(this).find('i'));
                 }
             });
         }
 
         $('.Items:not(:first)').hide();
 
         //Handling for UTR show with many performances
         $('.tonightBtn:contains("International")').parent().parent().find(".showTime").addClass("smallFont").css("width","25%");
 
     });
 
     
     function scrollShows($this)
     {
         var CurrentSection = $('#ShowContainer').find('.Items:visible');
 
         if ($this.data('direction') == 'down') {
             if (CurrentSection.next().length > 0)
             {
                 CurrentSection.slideUp();
                 CurrentSection.next().slideDown(400, 'swing', function () {
                     $('.MoreShowsBtn.unclickable').removeClass("unclickable");
                 });
 
                 if(CurrentSection.next().next().length == 0)
                 {
                    $('.MoreShowsBtn i[data-direction="down"]').css('color', '#ccc')
                         .css('cursor', 'default');
                 }
                 else
                 {
                     $('.MoreShowsBtn i[data-direction="down"], .MoreShowsBtn i[data-direction="up"]')
                         .css('cursor', 'pointer')
                         .css('color', 'black');
                 }
             }            
         }
         else
         {
            if(CurrentSection.prev().length > 0)
             {
                 CurrentSection.slideUp();
                 CurrentSection.prev().slideDown(400, 'swing', function () {
                     $('.MoreShowsBtn.unclickable').removeClass("unclickable");
                 });
 
                 if (CurrentSection.prev().prev().length == 0) {
                     $('.MoreShowsBtn i[data-direction="up"]').css('color', '#ccc')
                         .css('cursor', 'default');
                 }
                 else
                 {
                     $('.MoreShowsBtn i[data-direction="down"], .MoreShowsBtn i[data-direction="up"]')
                         .css('cursor', 'pointer')
                         .css('color', 'black');
                 }
             }
        }
         $('#TonightPerformance div.tonightBtn').fitText(.9, { minFontSize: '12px', maxFontSize: '16px' });
     }
