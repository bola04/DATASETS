 function trim(str) {
 	str = str.replace(/^\s+/, '');
 	for (var i = str.length - 1; i >= 0; i--) {
 		if (/\S/.test(str.charAt(i))) {
 			str = str.substring(0, i + 1);
 			break;
 		}
 	}
 	return str;
 }
 
 
 function fitStringToWidth(str,width,className) {
 	  
 	  var span = document.createElement("span");
 	  if (className) span.className=className;
 	  span.style.display='inline';
 	  span.style.visibility = 'hidden';
 	  span.style.padding = '0px';
 	  document.body.appendChild(span);
 
 	  var result = str;
 	  span.innerHTML = result;
 
 	  if (span.offsetWidth > width) {
 		var posStart = 0, posMid, posEnd = str.length, posLength;
 		while (posLength = (posEnd - posStart) >> 1) {
 		  posMid = posStart + posLength;
 		  span.innerHTML = str.substring(0,posMid) + '&hellip;';
 
 		  if ( span.offsetWidth > width ) posEnd = posMid; else posStart=posMid;
 		}
 
 		var maxDisplay = str.substring(0,posStart);
 		var indexCut = maxDisplay.lastIndexOf(" ");
 		var finalDisplay = str.substring(0,indexCut);
 		if(str.length>width){
 				result = finalDisplay;
 		} else {
 				result = finalDisplay + '&hellip;';
 		}
 	  }
 	  document.body.removeChild(span);
 	  document.write(result);
 }

   var param_value = false;
   var params = loc.split("&");
   for (i=0; i<params.length;i++) {

           param_value = params[i].substring(params[i].indexOf('=')+1)

 function uBillboard(canvasId,sec){
 
 	this.canvas = document.getElementById(canvasId);
 
 	this.curPos = 0;
 
 	this.curNode = null;
 
 	this.list = new Array();
 
 	this.sec = sec * 1000;
 
 	this.event = null;
 
 	
 
 	this.load = function (nodeList){
 
 		var len = this.list.length;
 
 		nodeList.forEach(function(node, index, arr){
 
 			var anchor = node.getElementsByTagName("a")[0];
 
 			if (anchor){
 
 				var content = new uBillboardContent(node,len,node.getAttribute("cImg"),node.getAttribute("cUrl"));
 
 				ntuc_bd.list[len] = content;
 
 				// set the anchor's attrs
 
 				dojo.connect(anchor, "onclick", function(){ntuc_bd.set(content);}); 
 
 				if (len==0){
 
 					ntuc_bd.display(content);
 
 				}
 
 				len++;				
 
 			}
 
 		});
 
 	}
 
 	
 
 	this.display = function(content){
 
 		// de-reference the prev node
 
 		if (this.curNode) {this.curNode.className="";}
 
 		// set reference to the next node
 
 		this.curNode = content.node;
 
 		this.curNode.className = "current";
 
 		this.canvas.innerHTML = "<a href='"+content.url+"' onmouseover='ntuc_bd.pause();' onmouseout='ntuc_bd.start();'><img src='"+content.img+"' width='616' height='330' /></a>";
 
 	}
 
 	
 
 	this.rotate = function(){
 
 		this.curPos++;
 
 		if (this.curPos == this.list.length){
 
 			this.curPos = 0;
 
 		}
 
 		this.display(this.list[this.curPos]);
 
 	}
 
 	
 
 	this.start = function(){
 
 		if (!this.event){
 
 			this.event = setInterval("ntuc_bd.rotate();",this.sec);
 
 		}
 
 	}
 
 	
 
 	this.pause = function(){
 
 		clearInterval(this.event);
 
 		this.event = null;
 
 	}
 
 	
 
 	this.set = function(content){
 
 		this.pause(); // pause the app
 
 		this.curPos = content.pos;
 
 		this.display(content);
 
 		this.start(); // resume the app
 
 	}
 
 	
 
 	
 
 }
 
 
 
 function uBillboardContent(node,pos,img,url){
 
 	this.node=node;
 
 	this.pos=pos;
 
 	this.img=img;
 
 	this.url=url;
 
 }

 function validateSignUpInput(IdEntity,EntityName,EntityContact,CdDay,CdMonth,CdYear){
 	// check if all the input are entered
 	if(IdEntity==null || EntityName==null || EntityContact==null || CdDay==null||CdMonth==null|| CdYear==null

 		return null;
 	}
 }

 function validateDate(CdDay,CdMonth,CdYear){
 	// before passed to this function, these parameters have been checked in validateSignUpInput or validateRenewInput
 	var objDate;
 	var mSeconds;
 	mSeconds = (new Date(CdYear, CdMonth, CdDay)).getTime();
 	objDate = new Date();
 	objDate.setTime(mSeconds);
 	
 	if ( (objDate.getFullYear()==null) || (objDate.getMonth()==null) || (objDate.getDate()==null)
 		||	(objDate.getFullYear()!=CdYear) || (objDate.getMonth()!=CdMonth) || (objDate.getDate()!=CdDay))
 	{
 		return false;
 	}else {
 		return true;		
 	}
 	
 }
 
 //Trang
 function validateAge(CdDay,CdMonth,CdYear){
 	// before passed to this function, these parameters have been checked in validateSignUpInput or validateRenewInput
 	// check if applicant is at least 16
 	var objDate;
 	var mSeconds;
 	mSeconds = (new Date(CdYear, CdMonth, CdDay)).getTime();
 	objDate = new Date();
 	objDate.setTime(mSeconds);
 	
 	var today = new Date();
 	if((objDate.getFullYear() != null) && (today.getFullYear() != null)){
 		var age = today.getFullYear() - objDate.getFullYear();
 		if (age >= 16){
 			return true;
 		}else {
 			return false;
 		}
 	}else {
 		return false;
 	}
 }

 function validateNRICFIN(inputVal) {
     
 	inputVal = inputVal.toUpperCase();
     var regex = /^[S,T,F,G][0-9]{7}[a-zA-Z]$/;
 
     if (!regex.test(inputVal)) {
         return false;
     }
 
     var firstChar = inputVal.substring(0, 1);
     var midDigits = inputVal.substring(1, 8).split("");
     var lastChar = inputVal.substring(8, 9);
 
     var totalValue = 0;
     var idx = 0;
     totalValue += midDigits[idx++] * 2;
     totalValue += midDigits[idx++] * 7;
     totalValue += midDigits[idx++] * 6;
     totalValue += midDigits[idx++] * 5;
     totalValue += midDigits[idx++] * 4;
     totalValue += midDigits[idx++] * 3;
     totalValue += midDigits[idx++] * 2;
     
     if ((firstChar === 'G') || (firstChar === 'T')) {
         totalValue += 4; //FIN number need to add 4 in total value
     }
 
     var modValue = totalValue % 11;
     
     var nricEndCharArr = {'0':'J', '1':'Z', '2':'I', '3':'H', '4':'G', '5':'F', '6':'E', '7':'D', '8':'C', '9':'B', '10':'A' }
     var finEncCharArr = {'0':'X', '1':'W', '2':'U', '3':'T', '4':'R', '5':'Q', '6':'P', '7':'N', '8':'M', '9':'L', '10':'K'};
     
     if ((firstChar === 'S') || (firstChar === 'T')) {
         //NRIC
         if (lastChar !== nricEndCharArr[modValue]) {
             return false;
         }
     }
     else if ((firstChar === 'G') || (firstChar === 'F')) {
         //FIN
         if (lastChar !== finEncCharArr[modValue]) {
             return false;
         }
     }
 
     return true;
 }
