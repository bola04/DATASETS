 		var captcha_code = $("#captcha_code").val();
 		
 		if(captcha_code == "")
 		{
 			alert("Pelase enter captcha code");
 			$("#captcha_code").focus();
 			return false;
 		}
 		else
 		{
 			var objValue=false;
 			//var dataString = 'vpb_captcha_code='+ vpb_captcha_code;
 			var dataString = $("#frmquestion").serialize();
 			$.ajax({
 				type: "POST",
 				url: "https://www.2fixyourtrafficticket.com/sendcontactmail",
 				data: dataString,
 				cache: false,
 				beforeSend: function() 
 				{
 					$("#captchaResponse").html('<div style="padding-left:100px;margin-bottom:30px;"><font style="font-family:Verdana, Geneva, sans-serif; font-size:12px; color:black;">Please wait</font> <img src="https://www.2fixyourtrafficticket.com/assets/images/loading.gif" align="absmiddle" /></div>');				
 				},
 				success: function(response)
 				{
 					//$("#captchaResponse").html(response);return false;
 					if(response=="failure"){
 						refresh_aptcha();
 						$("#captchaResponse").hide().fadeIn('slow').html("<div style='color:red;'>The captcha is invalid.</span></div>");
 						objValue=false;
 					}else{
 					 //Refresh the Captcha Image on success or on wrong security submitted
 						//window.location="https://www.2fixyourtrafficticket.com/thank-you.html";					
 						$("#captchaResponse").html('');
 						CloseContactForm("Your inquiry has been sent successfully. We will contact you at the earliest.");
 					}
 				}
 			});
 			return objValue;
 		}
 	}
