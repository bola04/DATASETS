 							function animateEgg(egg,total_eggs,to){
 
 								var area = (to/total_eggs) * 23256,
 									height = Math.sqrt(area/.795);
 									width = height * .795;
 
 									if (height < 4) {
 										height = 4;
 
 									}
 
 
 								// 23256 = base height of eggs (171) * base width (136)
 
 								egg.animate({'height': height + 'px','width':width + 'px','marginLeft':(138-width)/2 + 'px'});
 
 							}
 
 
 							if (thisCount > highAttempts) {
 								$banner.text('We Broke the Record!');
 								animateEgg($blue, 100,100);
 								$white.hide();
 								$record.show();
 								$previously.show();
 							} else {
 								$banner.text('Help Us Break Our Record!');
 								animateEgg($blue,highAttempts,thisCount);
 								$white.show();
 								$record.hide();
 								$previously.hide();
 							}
