     function submitLogin(e) {
         e.preventDefault();
 
         
             var refererValid = false;
             $.ajax({
                 type: 'GET',
                 async: false,
                 url:  'login-refererCheck.jsp',
                 success: function(data) {
                     refererValid = (data.indexOf("SUCCESS") != -1);
                 }
             });
 
             if (!refererValid) {
                 alert("Your Referer HTTP header is not correct. Please contact support.");
                 return;
             }
 
             // Don't do $(this).bind('click', false), it would block referLink.click() because this can be the Window object
             $('#loginform').bind('click', false);
 
             //IE8 workaround (missing Referer)
             function redirect(url) {
                 if (/MSIE (\d+\.\d+);/.test(navigator.userAgent)){
                     var referLink = document.createElement('a');
                     referLink.href = url;
                     document.body.appendChild(referLink);
                     referLink.click();
                 } else {
                     location.href = url;
                 }
             }
 
             var URL = '/login.jsp?loginView=login&modalOnly=true';
             $.defaultBlockUI();
             $.ajax({
                 type: 'POST',
                 url: URL,
                 data: $("#loginform").serialize(),
                 success: function (data) {
                     if (data) {
                         data = $.trim(data);
                         if (data.indexOf("redirect:") == 0) {
                             var loc = data.substring(9);
                             redirect(loc);
                             //window.location = loc; IE8 bug
                         } else {
                             $.unblockUI();
                             $('#modal_div_content').html(data);
                         }
                     } else {
                         alert('No Data');
                     }
                 },
                 dataType: "html"
             });
             return false;
         
     }
