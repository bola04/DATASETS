     function refresh_select(source, fld, src, trg, trgtab) {
         
         r_source[last] = source;
         r_fld[last] = fld;
         r_src[last] = src;
         r_trg[last] = trg;
         r_trgtab[last] = trgtab;
         last++;
         if ((last - 1) == 0) {refresh_select_process();}
         
     }
 
     function refresh_select_process() {
         source = r_source[current];
         fld = r_fld[current];
         src = r_src[current];
         trg = r_trg[current];
         trgtab = r_trgtab[current];
         current++;
         
         if (source.length > 0) {
             refresh_select_process_server(source, fld, src, trg);
         } else {
             refresh_select_process_local(src, trg, trgtab);
         }
     }
 
     function end_process() {
         if (last > current) {
             refresh_select_process();
         } else {
             //if (navigator.appName == "Netscape") {
             //    history.go(0);
             //}
             if (server) {top.select.location.href = "blank.htm"; }
             last = 0; current = 0; server = false;
         }
     }
 
 
     function refresh_select_process_local(src, trg, trgtab) {
         
         source = eval("document.Search1." + src);
         target = eval("document.Search1." + trg);
 
        
 
         l = target.length;
         for(rnum=0;rnum<l;rnum++) {target.options[0]=null;}
         value = source.options[source.selectedIndex].value;
         //alert("src=" + src + ", trg=" + trg + ", value=" + value);
         array_value = eval(trgtab + "_val");
         array_label = eval(trgtab + "_lbl");
         array_dep = eval(trgtab + "_dep");
         i = 0;
         newOption = new Option('', '');
         target.options[i++] = newOption;
         for (rnum=0;rnum<array_value.length;rnum++) {
             if ((array_dep[rnum] == 0) || (array_dep[rnum] == value)) {
                 newOption = new Option(array_label[rnum], array_value[rnum]);
                 target.options[i++] = newOption;
             }
         }
         if (target.length > 0) {
             target.options[0].selected = true;
             if (typeof(target.onchange) == "function") {
                 target.onchange();
                
             }
         }
         end_process();
     }

     function MultiLine2String(MLSelect)
     {
         var sReturn="";
         var bNotFirstOne=false;
         for (i = 0;  i < MLSelect.options.length;  i++)
         {
             if (MLSelect.options[i].selected == true)
             {
                 if (bNotFirstOne)
                 {
                     sReturn=sReturn + ",";
                 }
                 bNotFirstOne=true;
                 sReturn = sReturn + MLSelect.options[i].value;
             }
         }
         return (sReturn);
     }
