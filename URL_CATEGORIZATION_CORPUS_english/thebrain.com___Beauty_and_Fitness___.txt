    <script>
     var isSearchHidden = true;
 var aniTime = 150;
 
 function showSearchBar(imageElementId) {
   if($('body').hasClass('platform-ios') || $('body').hasClass('platform-android')) {
     location = '/search';
     return;
   }
 
   isSearchHidden = false;
   var height = $('.navbar').height();
   $('.search-bar').css('display', 'block');
   $('.search-bar').css('height', height+'px');
   $('.search-bar').fadeTo(aniTime, 1, function() {
     $('.search-bar').css('display', 'block'); // set again since it sometimes doesn't work
   });
   
   var searchInput = $('#navSearch');
   searchInput.css('left', $('#nav-container').offset().left+'px');
   searchInput.css('width', ($('#nav-container').width()-40)+'px');
   searchInput.focus();
 
   var searchButton = $('#cancelSearch');
   searchButton.css('left', $(imageElementId).offset().left+'px');
   searchButton.css('top', ($(imageElementId).offset().top - $(window).scrollTop())+'px');
   searchButton.css('position', 'fixed');
   searchButton.css('width', $(imageElementId).width()+'px');
   searchButton.css('height', $(imageElementId).height()+'px');
 }
 
 function hideSearchBar() {
   isSearchHidden = true;
   $('.search-bar').fadeTo(aniTime, 0, function() {
     $('.search-bar').css('display', 'none');
   });
 }
 
 $('#navSearch').on('keydown', function(e) {
   if(e.which == 13 || e.keyCode == 13) {
     var query = $('#navSearch').val();
     location.href = '/search?q='+encodeURIComponent(query);
     e.preventDefault;
   }
   else if(e.which == 27 || e.keyCode == 27) {
     hideSearchBar();
   }
 });
 
 $(window).on('resize', hideSearchBar);
 $(window).on('scroll', hideSearchBar);
    </script>
