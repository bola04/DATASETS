     function bindInputs () {
     var inputContainers = document.querySelectorAll('.tk-input-container');
     var inputs = document.querySelectorAll('.tk-input-container input, .tk-input-container textarea');
     var selects = document.querySelectorAll('.tk-input-container select');
 
     [].forEach.call(inputs, function(i) {
         bindInput(i);
     });
 
     [].forEach.call(selects, function(i) {
         i.addEventListener('change', handleSelects);
 
         handleSelects.call(i);
     });
 }
 
 function bindInput(i) {
     i.addEventListener('keyup', handleInputs);
 
     handleInputs.call(i);
 }
 
 function handleInputs() {
     if (this.value != '') {
         this.parentNode.classList.remove('input__empty');
     } else {
         this.parentNode.classList.add('input__empty')
     }
 }
 
 function handleSelects() {
     if (this.selectedIndex != 0) {
         this.parentNode.classList.remove('input__empty');
     } else {
         this.parentNode.classList.add('input__empty');
     }
 }
 
 
 function initMDAAJAXForms(wrapper){
     if(!wrapper){
         var wrapper = jQuery('body');
     }
     
     var mdaAJAXFormsWrappers = wrapper.find('.js-ajax-form[data-url]');
     
     mdaAJAXFormsWrappers
         .each(function(){
             var ajaxFormWrapper = jQuery(this);
             
             ajaxFormWrapper
                 .css({
                     height: ajaxFormWrapper.height(),
                     position: 'relative'
                 })
                 .html('<p class="ajax-form-loading">loading...</p>');
                 
             jQuery.ajax({
                 url: ajaxFormWrapper.attr('data-url'),
                 success: function(HTML){
                         ajaxFormWrapper
                             .empty()
                             .html(HTML)
                             .removeAttr('style');
 
                         ajaxFormWrapper.find('.tk-input-container input').each(function (i) {
                             bindInput(this);
                         });
 
                         var overlayContentOnly = ajaxFormWrapper.find('div.plus-overlay-musclewalk');
                         
                         if (overlayContentOnly.length > 0) {
                             ajaxFormWrapper.empty().html('<div class="plus-overlay-musclewalk">' + overlayContentOnly.html() + "</div>");
                         }               
                         
                         var ajaxForm = ajaxFormWrapper.find('form');
 
                         if(ajaxForm.length > 0){
                         
                             if(ajaxForm.is('.join-form')){
                                 joinForm();
                             }
                             if (ajaxForm.parent().hasClass('plus-overlay-musclewalk')) {
                                 ajaxForm.parent().parent().css("padding", "0px");
                                 
                                 initPlusBounce();
                             } else {
                                 initMDAAJAXFormsSubmit(ajaxFormWrapper);
                             }
                         }
                     }
             });
                 
         });
 }
 
 
 function initMDAAJAXFormsSubmit(wrapper){
     if(!wrapper){
         var wrapper = jQuery('body');
     }
     
     var mdaAJAXForms = wrapper.is('.js-ajax-form')?wrapper.find('form:first'):wrapper.find('.js-ajax-form form');
     mdaAJAXForms
         .submit(function(e){
             e.preventDefault();
             
             var ajaxForm = jQuery(this);
             var ajaxFormWrapper = ajaxForm.parents('.js-ajax-form:first');
             
             jQuery.ajax({
                 url: ajaxForm.attr('action'),
                 type: ajaxForm.attr('method'),
                 data: ajaxForm.serialize(),

                         ajaxFormWrapper
