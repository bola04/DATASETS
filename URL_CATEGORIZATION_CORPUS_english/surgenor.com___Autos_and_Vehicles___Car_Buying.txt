     function removeAccents(vehicleName){
         var accents = ['à','â','ä','è','ê','é','ë','ï','î','ô','ö','ù','û','ü','ÿ','ç','À','Â','Ä','Ç','É','È','Ê','Ë','Î','Ï','Ô','Ö','Ù','Û','Ü','Ÿ'];
