 					moveBanner((position_animated_banner+1)%total_animated_banner);
 					
 					return false;
 				}	
 			);
 
 			//Previous banner
 			jQuery('.animated_banner a.previous').click(
 				function() {
 					var list_animated_banner = jQuery('.animated_banner .banner');
 					var element_animated_banner = jQuery('.animated_banner .current_banner');
 					var position_animated_banner = jQuery(list_animated_banner).index(element_animated_banner);
 					var total_animated_banner = list_animated_banner.length;
 					
 					if((position_animated_banner-1) < 0) {
 						position_animated_banner = total_animated_banner;
 					}
 					moveBanner(position_animated_banner-1);
 					
 					return false;
 				}	
 			);
 			
 			//Pager
 			jQuery('.animated_banner .banner .pager li a').click(
 				function() {
 					var list_pager = jQuery('.animated_banner .current_banner .pager li a');
 					var element_pager = jQuery(this);
 					var position_pager = jQuery(list_pager).index(element_pager);
 					
 					moveBanner(position_pager);
 					
 					return false;
 				}	
 			);
 
 			//Pause animation on hover
 			jQuery('.animated_banner').hover(
 				//Mouse enter -> Pause animation
 				function() {
 					autoAnimationPause = true;
 				},
 				//Mouse over -> Animation
 				function() {
 					autoAnimationPause = false;
 				}
 			);
 		}
 	});
 
 	var autoAnimationPause = false;
 	
 	function autoAnimation() {
 		if(autoAnimationPause) {
 			setTimeout('autoAnimation()', '1000');
 		} else {
 			var list_animated_banner = jQuery('.animated_banner .banner');
 			var element_animated_banner = jQuery('.animated_banner .current_banner');
 			var position_animated_banner = jQuery(list_animated_banner).index(element_animated_banner);
 			var total_animated_banner = list_animated_banner.length;
 			
 			//call the function to move the banner
 			moveBanner((position_animated_banner+1)%total_animated_banner);
 	
 			//Call the function to animate banner in 3 seconds
 			setTimeout('autoAnimation()', 5000);
 		}
 	}
 
 	function moveBanner(position) {
 		//alert(position);
 		jQuery('.animated_banner .banner').removeClass('current_banner').hide();
 		jQuery('.animated_banner .banner:eq('+position+')').addClass('current_banner').show();
 
 		jQuery('.animated_banner .banner:eq('+position+') .pager li').removeClass('active');
 		jQuery('.animated_banner .banner:eq('+position+') .pager li:eq('+position+')').addClass('active');
 	}
