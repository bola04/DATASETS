 function revizelogin()
 {
 	var loginSpan = document.getElementById('revize-login');
 	//var loginSpan = document.getElementById('powered-by-revize-link');
 	if (!loginSpan) return;
 
 	var color,styleElement;
 	var selector1 = "#revize-login a";
 	var rule1 = "text-decoration:none !important;color:";
 	var selector2 = "#revize-login a:hover";
 	var rule2 = "text-decoration:underline !important";
 	if (document.all)
 	{
 		color = loginSpan.currentStyle.color;
 		styleElement = document.createStyleSheet();
 		styleElement.addRule(selector1 , rule1+color+' !important')
 		styleElement.addRule(selector2 , rule2)
 	}
 	else
 	{
 		color =  document.defaultView.getComputedStyle(loginSpan,'').getPropertyValue('color');
 		styleElement = document.createElement("style");
 		document.getElementsByTagName("head")[0].appendChild(styleElement);
 		styleElement = document.styleSheets[document.styleSheets.length-1];
 		styleElement.insertRule(selector1 + ' {' + rule1+color + ' !important}',0);
 		styleElement.insertRule(selector2 + ' {' + rule2 + '}',0);
 	}
 }
 if (typeof window.addEventListener != "undefined")
 	window.addEventListener("load", revizelogin, false);
 else if (typeof window.attachEvent != "undefined")
 	window.attachEvent("onload", revizelogin);

    function isOffsite(e)
 {
 	var el = null;
 	if (e.srcElement)		//IE
 		el = e.srcElement;
 	else if (e.target)		//Standard
 		el = e.target;
 
 	var bases = document.getElementsByTagName('base');
 	var baseHref = "";
 	if (bases.length > 0) {
 	    baseHref = bases[0].href;
 	    if( baseHref != "" && baseHref != undefined) {
 		    var end = baseHref.lastIndexOf("/")+1;
 		    baseHref = baseHref.substring(0,end);
 	    }
 	}
 	
 	while (el)
 	{
 		if (el.tagName && el.tagName.toUpperCase() == 'A')
 		{
 			if (!el.href) break;	//should not happen
 			
 			var url = 'external.php#' + el.href;
 			var isIEw3c = navigator.appVersion.indexOf('MSIE') == -1
 					   && navigator.userAgent.indexOf('Trident') != -1
 					   && navigator.appCodeName.indexOf('Mozilla') != -1;
 			if (navigator.appVersion.indexOf('MSIE') != -1 
 				|| navigator.appVersion.indexOf('Edge') != -1
 				|| isIEw3c)
 				url = baseHref + url;
 			location.href = url;
 			
 			if (e.preventDefault)
 				e.preventDefault(); // DOM style
 			return false;			// IE 
 		}
 		// if not <a...> tag, move up to parent
 		if (!el.parentNode) break;	//should not happen
 		el = el.parentNode;
 	}
 	return true;
 }
 function RZcatchlinks()
 {
 	var links = document.getElementsByTagName('a');
 	var whitelist = [];
 	//collect whitelist
 	if( typeof RZ.webspace_config.redirect_whitelist !== 'undefined' 
 	&& RZ.webspace_config.redirect_whitelist !== "") 
 	{
 		whitelist = RZ.webspace_config.redirect_whitelist.split("|");
 	}
 	for (i=0;i<links.length;i++)
 	{
 		var el = links[i];
 		var shouldCatch = false;
 		if (el.href.search(/https?:/i) == 0 
 		&& el.href.search(/https?:\/\/.*?\/revize\//i) == -1						   
 		&& el.href.indexOf(location.host) == -1
 		&& el.id != 'revize_leave_site')
 		{
 			shouldCatch = true;
 			// check against whitelist
 			for(var j=0;j<whitelist.length;j++){
 				if( whitelist[j] !== "" && el.href.indexOf(whitelist[j]) !== -1 ){
 					shouldCatch = false;
 					break;
 				}
 			}
 		}
 
 		if ( shouldCatch === true )
 		{
 			if (el.addEventListener) el.addEventListener('click', isOffsite, false);
 			else if (el.attachEvent) el.attachEvent("onclick", isOffsite);
 		}
 	}
 }
 if (window.addEventListener) window.addEventListener('load', RZcatchlinks, false);
 else if (window.attachEvent) window.attachEvent("onload", RZcatchlinks);
