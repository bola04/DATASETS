         function ageCompare() {
             if ($(".form-cont").valid() === true) {
 
                 var day = $("#day").val();
                 var month = $("#month").val();
                 var year = $("#year").val();
                 var minAge = $(".age-gate").attr("data-minage");
                 var dob = new Date(year, (month - 1), day, 0, 0, 0, 0);
                 var serviceUrl = $(".age-gate").attr('data-serviceuri');
 
                 var remember = false;
                 if ($(".remember").is(':checked')) {
                     remember = true;
                 }
 
                 var valid = validateDate(dob, year, month, day) && (getAge(dob) >= minAge);
                 if (valid) {
                     setAgeGateCookie();
                     location.reload();
                 }
                 else {
                     $(".message-warning").css("display", "block");
                 }
             }
             return false;
         }
 
         function setAgeGateCookie() {
             var d = new Date();
             d.setTime(d.getTime() + (365 * 24 * 60 * 60 * 1000));
             var expires = "expires=" + d.toUTCString();
             document.cookie = "SamAdamsAgeGate=true;" + expires + ";path=/";
         }
 
         function getAge(birthDate) {
             var birthDate = new Date(birthDate)
             var today = new Date();
             var age = today.getFullYear() - birthDate.getFullYear();
             var m = today.getMonth() - birthDate.getMonth();
             if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
                 age--;
             }
             return age;
         }
 
         function validateDate(dob, year, month, day) {
             var testDate = month + "/" + day + "/" + year;
             // First check for the pattern
             if (!/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(testDate))
                 return false;
 
             if (Object.prototype.toString.call(dob) === "[object Date]") {
                 // it is a date
                 if (isNaN(dob.getTime())) {  // d.valueOf() could also work
                     return false;
                 }
                 else {
                     return true;
                 }
             }
             else {
                 return false;
             }
         }
