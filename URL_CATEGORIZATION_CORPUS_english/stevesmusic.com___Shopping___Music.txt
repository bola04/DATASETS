     function initHoverOut(id)

         jQuery("#"+id+'-hover').hover(function(){

         if (length>0)
             {  
             jQuery(".product-grid-hover").each(function(){
                 var element = jQuery(this);
                 var hovered = checkHover(element);
                 if (hovered==false)
                     {
                     jQuery(this).remove();
                     }
                 });
             }
     }, 200);
 
     function checkHover(element)
         {
         var posX = jQuery("#posX").val();
         var posY = jQuery("#posY").val();
         var offset = element.offset();
         var top = offset.top;
         var left = offset.left;
         var width = element.css('width');
         width = parseInt(width);
         var height = element.css('height');
         height = parseInt(height);
         var posX1 = parseInt(left);
         var posX2 = parseInt(left) + parseInt(width);
         var posY1 = parseInt(top);
         var posY2 = parseInt(top) + parseInt(height);
         var hovered = false;
         if ((posX >= posX1) && (posX <= posX2))
             {
             if ((posY >= posY1) && (posY <= posY2))
                 {
                 hovered = true; 
                 }        
             }
         return hovered;
         }

 function addToNewsletter()
     {
     var email = jQuery("#foot_mailing_list_input").val();
     var resub = jQuery("#resub").val();
     if (validateEmail(email)==false)
         {
         swal(
             'Error',
             'You must provide a valid email address.',
             'error'
             );    
         jQuery("foot_mailing_list_input").focus();
         }
     else
         {
         if (resub=='0')
             {
             var request = jQuery.ajax({
                 url: "/ajax-mailinglist.php",
                 method: "POST",
                 data: { email : email },
                 dataType: "html"
                 });
             }
         else
             {
             var request = jQuery.ajax({
                 url: "/ajax-mailinglist.php",
                 method: "POST",
                 data: { email : email, resub: 1, },
                 dataType: "html"
                 });
             }
         
         request.done(function( msg ) {
             var obj = JSON.parse(msg);
             if (obj.success==true)
                 {
                 swal(
                     'Success!',
                     'You have been added to our mailing list.',
                     'success'
                     );        
                 }
             else
                 {
                 if (obj.error=='exists')
                     {
                     swal(
                         'Success',
                         'You are already subscribed.',
                         'success'
                         );  
                     }
                 if (obj.error=='unsubscribed')
                     {
                     swal({
                       title: 'Your email is currently unsubscribed',
                       text: "Would you like to re-subscribe?",
                       type: 'warning',
