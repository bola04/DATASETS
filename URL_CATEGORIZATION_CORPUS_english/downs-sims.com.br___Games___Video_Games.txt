  * Simple class for preloading images with support for progress reporting
  * Copyright 2007 Tomocchino.
  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
 
 var Preloader = new Class({
   
   Implements: [Events, Options],
 
   options: {
     root        : '',
     period      : 100
   },
   
   initialize: function(options){
     this.setOptions(options);
   },
   
   load: function(sources) {
     this.index = 0;
     this.images = [];
     this.sources = this.temps = sources;
     this.total = this. sources.length;
     
     this.fireEvent('onStart', [this.index, this.total]);
     this.timer = this.progress.periodical(this.options.period, this);
     
     this.sources.each(function(source, index){
       this.images[index] = new Asset.image(this.options.root + source, {
         'onload'  : function(){ this.index++; if(this.images[index]) this.fireEvent('onLoad', [this.images[index], index, source]); }.bind(this),
         'onerror' : function(){ this.index++; this.fireEvent('onError', [this.images.splice(index, 1), index, source]); }.bind(this),
         'onabort' : function(){ this.index++; this.fireEvent('onError', [this.images.splice(index, 1), index, source]); }.bind(this)
       });
     }, this);
   },
   
   progress: function() {
     this.fireEvent('onProgress', [Math.min(this.index, this.total), this.total]);
     if(this.index >= this.total) this.complete();
   },
   
   complete: function(){
     $clear(this.timer);
     this.fireEvent('onComplete', [this.images]);
   },
   
   cancel: function(){
     $clear(this.timer);
   }
   
 });
 
 /* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
  * Follows: formatString (function)
  * Original name: Yahoo.Tools.printf
  * Copyright Yahoo.
  * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
 
 function formatString() {
 	var num = arguments.length;
 	var oStr = arguments[0];
 	for (var i = 1; i < num; i++) {
 		var pattern = "\\{" + (i-1) + "\\}"; 
 		var re = new RegExp(pattern, "g");
 		oStr = oStr.replace(re, arguments[i]);
 	}
 	return oStr; 
 }

             <a href="http://www.downs-sims.com.br/2017/01/lista-de-informacoes-sobre-os-vampiros.html">
              Lista de informações sobre os Vampiros!
             </a>
            </h3>
            <div class="post-header-line-1">
            </div>
            <div class="post-body entry-content">
             <div class="separator" style="clear: both; text-align: center;">
              <a href="https://3.bp.blogspot.com/-Im-BQHJDOA0/WIBajyUpYXI/AAAAAAAAAcY/HwzJg1p-uekEj7gFTosnC9uLni8Hua9NwCEw/s1600/the%2Bsims%2B4%2Bvampiros.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
               <img border="0" src="https://3.bp.blogspot.com/-Im-BQHJDOA0/WIBajyUpYXI/AAAAAAAAAcY/HwzJg1p-uekEj7gFTosnC9uLni8Hua9NwCEw/s1600/the%2Bsims%2B4%2Bvampiros.jpg"/>
              </a>
             </div>
             <br/>
             Com o anúncio do novo pacote de jogo, The Sims 4 Vampiros, vários simmers entraram no twitter para fazer perguntas a alguns SimGurus desta área, o site
             <a href="http://www.alalasims.com/noticias/2017/01/lista-de-informacoes-sobre-os-vampiros/" target="_blank">
              Alala Sims
             </a>
             reuniu as melhores informações, confira abaixo:
             <br/>
             <br/>
             - No Modo Criar um Sim o seu Sim vampiro terá duas opções, dentre elas ele pode ser um vampiro normal ou um vampiro mais medonho.
             <br/>
             - O mundo de Forgotten Hollow será um mundo residencial.
             <br/>
             - Em Forgotten Hollow teremos uma mansão sinistra e também 5 lotes jogáveis.
             <br/>
             - Teremos novos traços para os lotes.
             <br/>
             - No Criar um Sim, não poderemos transformar um Sim e Alien em vampiro, o Sim normal só pode virar vampiro ao ser mordido. Ainda no Criar um Sim, teremos uma opção como a dos Aliens de criar um sim vampiro.
             <br/>
             - Aliens e Vampiros podem se reproduzir, mas o filho terá a chance de nascer Alien ou Vampiro. Mas os poderes só serão liberados ao bebê vampiro quando se tornar Adolescente.
             <br/>
             - O tempo em que o vampiro pode ficar no sol depende de suas características, poderes e fraquezas.
             <br/>
             - Vampiros não refletem no espelho.
             <br/>
             - Vampiros não podem sugar sangue de seus companheiros vampiros, sendo apenas possível nos Sims normais.
