 		var width = window.innerWidth * devicePixelRatio;
 		if (width > 2047 || devicePixelRatio > 1) { // check ratio
 					return "x-high";
 		} else if (width > 991) {
 					return "high";
 		} else if (width > 767) {
 					return "medium";
 		} else {
 			return "small"; // default version

 	}
