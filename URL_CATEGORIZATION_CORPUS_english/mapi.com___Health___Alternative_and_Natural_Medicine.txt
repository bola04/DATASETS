 			for (var ind in sortedGroups) {
 				if (data['suggests'].hasOwnProperty(sortedGroups[ind]) != null) {
 					for (var ind1 in data['suggests'][sortedGroups[ind]]) {
 						var searchResult = data['suggests'][sortedGroups[ind]][ind1];
 						var searchResultNode = '<div class="ss360-search-row" p="' + (parseInt(index / itemsPerPage) + 1) + '">';
 
 						// content and image
 						if (searchResult['image'] != undefined) {
 							searchResultNode += '<img src="' + searchResult['image'] + '" style="float:left;max-width:100px" />';
 						}
 						
 						searchResultNode += '<div class="right-search-block" style="margin-left:110px">';	
 						
 						if (searchResult['name'] != null) {
 							searchResult['name'] = searchResult['name'].replace(new RegExp(data.query, 'ig'), '<span style="font-weight:bold;color:#152edd">' + data.query + '</span>');
 						}
 
 						searchResultNode += '<h4 style="margin-bottom:0px;color:#333;background-color:transparent;padding-left: 0px;">' + searchResult['name'] + '</h4>';
 						
 						if (searchResult['dataPoints'] != null && searchResult['dataPoints'][0] != null && searchResult['dataPoints'][0]['key'] == 'author') {
 							searchResultNode += '<p><i>' + searchResult['dataPoints'][0]['value'] + '</i></p>';
 						} else if (searchResult['dataPoints'] != null && searchResult['dataPoints'][0] != null && searchResult['dataPoints'][0]['author'] != null) {
 							searchResultNode += '<p><i>' + searchResult['dataPoints'][0]['author'] + '</i></p>';
 						} else if (searchResult['dataPoints'] != null && searchResult['dataPoints'][0] != null && searchResult['dataPoints'][0]['key'] == 'Rating') {
 							var rating = parseFloat(searchResult['dataPoints'][0]['value']).toFixed(1);
 							var reviewCount = searchResult['dataPoints'][1]['value'];							
 							if (rating > 4) {
 								var ratingStars = '';
 								for (var i = 0; i < parseInt(rating); i++) {
 									ratingStars += '<img src="/home/images/review/fullstar.jpg" style="max-width:15px;margin-top:-3px;display:inline-block">';
 								}
 								if (rating - parseInt(rating) > 0 && rating - parseInt(rating) < 0.5) {
 									ratingStars += '<img src="/home/images/review/halfstar.jpg" style="max-width:15px;margin-top:-3px;display:inline-block">';
 								} else if (rating - parseInt(rating) >= 0.5) {
 									ratingStars += '<img src="/home/images/review/fullstar.jpg" style="max-width:15px;margin-top:-3px;display:inline-block">';
 								}
 								
 								searchResultNode += '<p>' + ratingStars + ' ' + reviewCount + ' review' + (reviewCount > 1 ? 's' : '') + '</p>';
 							}
 						}
 						
 						
 						// link
 						if (searchResult['link'] != undefined) {
 							
 							searchResultNode += '<a href="' + searchResult['link'] + '" style="color:#337ab7">' + searchResult['link'] + '</a>';
 						}
 
 						// content
 						if (searchResult['content'] != undefined) {
 							searchResult['content'] = searchResult['content'].replace(new RegExp(data.query, 'ig'), '<span style="font-weight:bold;color:#152edd">' + data.query + '</span>');
 							searchResultNode += '<p>' + searchResult['content'] + '</p>';
 						}
 
 						searchResultNode += '</div>';
 
 						searchResultNode += '<div class="unibox-ca"></div></div>';
 
 						resultNode.append(jQuery(searchResultNode));
 						
 						index++;
 					}
 				}
 			}	
 			
 			var totalPages = parseInt(index / itemsPerPage) + (index % itemsPerPage > 0 ? 1 : 0);
 			if (totalPages > 1) {
 				var pagingHtml = '<div class="paging text-center">';
 				for (var i = 1; i <= totalPages; i++) {
 					pagingHtml += '<a href="javascript:void(0)" onclick="search360GoToPage(' + i + ')" class="page" p="' + i + '">' + i + '</a>';
 				}
 				pagingHtml += '</div>';
 				resultNode.append(jQuery(pagingHtml));
 				search360GoToPage(1);
 			}
 		}
 		if (data.totalResults == 0) {
 			resultNode.html('<h3>Search results for "' + data.query + '"' + '</h3>');
 			resultNode.append(jQuery('<div class="paging not-found text-center">Sorry, we have <b>not found any matches</b> for your query.</div>'));
 		}
     }
 
 };

 	jQuery('.ss360-search-row[p=' + index + ']').show();
 	jQuery('a.page').removeClass('active');
 	jQuery('a.page[p=' + index + ']').addClass('active');
 }
