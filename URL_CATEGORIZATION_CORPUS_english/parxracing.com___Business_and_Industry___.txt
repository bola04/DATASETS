   $(document).ready(function(){
 
 initTweet();
 
 function initTweet(){
 			
 	$.getJSON('/social/access.php?function=get_tweets',
 		function(data){
 			
 			$.each(data, function(i,item){
 	   
 				user = item.user.screen_name;
 				imagePic = item.user.profile_image_url.replace('http','https');
 				text = item.text.replace(/\s(#)(\w+)/g,' #$2');
 				time_value = item.created_at;
 				
 				var tweet_date = new Date(time_value);
 				var today = new Date();
 				var tweet_time = tweet_date.getTime();
 				var today_time = today.getTime();
 				
 				var minDiff = Math.ceil((today_time - tweet_time)/(60*1000));
 				var hourDiff = Math.ceil((today_time - tweet_time)/(60*60*1000))-1;
 				var dayDiff = Math.ceil((today_time - tweet_time)/(24*60*60*1000));
 				
 				var m_names = new Array("January", "February", "March", 
 				"April", "May", "June", "July", "August", "September", 
 				"October", "November", "December");
 				
 				var d = new Date(time_value);
 				var curr_date = d.getDate();
 				var curr_month = d.getMonth();
 				var curr_year = d.getFullYear();
 				
 				
 				var post_time;
 				if(minDiff < 1){
 				post_time=('<p class="twit_time">posted about '+minDiff+' minutes ago</p>');
 				}
 				else if (minDiff < 59){
 				post_time=('<p class="twit_time">posted about '+minDiff+' minutes ago</p>');
 				}
 				else if(hourDiff < 2){
 				post_time=('<p class="twit_time">posted about '+hourDiff+' hour ago</p>');
 				}
 				else if(hourDiff < 24){
 				post_time=('<p class="twit_time">posted about '+hourDiff+' hours ago</p>');
 				}
 				else if(dayDiff <= 5){
 				post_time=('<p class="twit_time">posted about '+dayDiff+' days ago</p>');
 				}
 				else if(dayDiff > 5){
 				post_time=('<p class="twit_time">posted '+m_names[curr_month]+' '+curr_date+', '+curr_year+'</p>');
 				};
 				
 				$("#tweet_contain>ul").append('<li><div class="bottom_fade"></div><a href="http://twitter.com/parxracing" target="_blank"><img src="https://pbs.twimg.com/profile_images/466296153177214976/z0z6BWFG_200x200.png" width="48";/><p><span class="twit_user">@'+user+'</span></br><span class="twit_copy">'+text+'</span></p><h3>'+post_time+'</h3></a></li>');
 				
 		});
 		
 		var count = 1
 		var time = 7000
 		
 		$("#tweet_contain>ul>li").hide();
 		$("#tweet_contain>ul>li:first-child").show();
 	
 		setTimeout(function(){advanceEvent()},time);
 	
 		function advanceEvent(){
 			setTimeout(function() {fadeThemOut()},250);
 		 	setTimeout(function() {animationFade()},500);
 		 	setTimeout(function() {fadeThemIn()},1000);
 		};
 	
 	
 		function fadeThemOut(){
 			$("#tweet_contain>ul").fadeOut(250);
 		};
 	
 		function fadeThemIn(){
 			$("#tweet_contain>ul").fadeIn(500);
 			setTimeout(function(){advanceEvent()},time);
 		};
 	
 	
 		function animationFade(){
 		$("#tweet_contain>ul>li:nth-child("+count+")").hide();
 			count++;
 	        if (count == 5) {
 				$("#tweet_contain>ul>li").hide();
 				count = 1;
 			};
 			$("#tweet_contain>ul>li:nth-child("+count+")").show();
 			return false;
 		};
 		
 	});
 };
 
 $('.calendar_trigger').toggle(function(){
 	if(!($('.calendar_modal').is(':animated'))){
 		$('.calendar_modal').fadeIn();
 	}
 },function(){
 	if(!($('.calendar_modal').is(':animated'))){
 		$('.calendar_modal').fadeOut();
 	}
 });
 
 
 });
