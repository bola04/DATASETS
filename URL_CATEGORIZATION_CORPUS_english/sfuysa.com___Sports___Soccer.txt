    <script type="text/javascript">
     var pivotSize = 768;
     var pivotScroll = 200;
     var idPageHeader = "PPROPageHeaderGeneric";
 
     $(document).ready(function () {
         calculateMainDistance();
     });
 
     $('#' + idPageHeader).find('img').on('load', function () {
         calculateMainDistance();
     });
 
     $(window).scroll(function (event) {
         var scroll = $(window).scrollTop();
         var width = $(window).width();
         if (width > pivotSize) {
             var show = scroll < pivotScroll;
             dinamicMenu(show);
         }
     });
 
     $(window).resize(function (event) {
         var scroll = $(window).scrollTop();
         var show = $(window).width() <= pivotSize ? true : scroll < pivotScroll;
         dinamicMenu(show);
     });
 
     function calculateMainDistance() {
         var objHeader = $("#" + idPageHeader);
         var height = objHeader.height();
         var isFixed = objHeader.css("position") == 'fixed';
 
         if (isFixed) {
             $("#main").css("margin-top", height + 20);
         } else {
             $("#main").css("margin-top", "2%");
         }
     }
