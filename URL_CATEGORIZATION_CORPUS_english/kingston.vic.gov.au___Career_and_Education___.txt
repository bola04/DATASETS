          <script>
           var sizes = ["0.55","0.65","0.75","0.85","0.95"];
 var sizeUnit = 'em'
 var cookieName = 'textSize';
 var daysToExpire = 365;
 var currentSizeIndex = 2;
 var currentFontSize = 0.75;
 
 if (currentFontSize == null) 
 {
 	currentFontSize = 0.75;
 }
 
 function decreaseTextSize()
 {
 	if (currentSizeIndex > 0)
 	{
 		currentSizeIndex--;
 		currentFontSize = sizes[currentSizeIndex];
 		return true;
 	}
 	return false;
 }
 
 function increaseTextSize()
 {
 	if (currentSizeIndex < sizes.length - 1)
 	{
 		currentSizeIndex++;
 		currentFontSize = sizes[currentSizeIndex];
 		return true;
 	}
 	return false;
 }
 
 function writeCookie()
 {
 	$.cookie(cookieName, currentFontSize, { expires: daysToExpire});
 }
 
 function setFontSize()
 {
 	$('body').css('font-size', currentFontSize + sizeUnit);
 }
 
 $(document).ready(function ()
 {
 	$(".btnTextDecrease").click(function (e) 
 	{
 		e.preventDefault();
 		var changed = decreaseTextSize();
 		if(changed)
 		{
 			writeCookie();
 			setFontSize();
 		}
 	});
 
 	$(".btnTextIncrease").click(function (e) 
 	{
 		e.preventDefault();
 		var changed = increaseTextSize();
 		if(changed)
 		{
 			writeCookie();
 			setFontSize();
 		}
 	});
 });
          </script>
