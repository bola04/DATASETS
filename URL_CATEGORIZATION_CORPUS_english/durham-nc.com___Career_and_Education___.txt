         function googleTranslateElementInit() {
 		new google.translate.TranslateElement({

 		}, 'google_translate_element');

 	$(function() {
 		var readCookie = function(name) {
 			var nameEQ = name + "=";
 			var ca = document.cookie.split(';');
 			for(var i=0;i < ca.length;i++) {
 				var c = ca[i];
 				while (c.charAt(0)==' ') c = c.substring(1,c.length);
 				if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
 			}
 			return null;
 		}
 
 		/* In some cases the subdomain cookie won't get deleted so google translate will retain the last
 		   translation when you navigate to a new page. This function will delete the sub-domain cookie as well. */
 		var deleteCookie = function(name) {
 			var delCookie = name + '=;expires=Thu, 01 Jan 1970 00:00:01 GMT; path=/';
 			document.cookie = delCookie;
 			document.cookie = delCookie + '; domain=.'+window.location.host;
 			document.cookie = delCookie + '; domain=.'+window.location.host.split('.').slice(-2).join('.');
 		}
 
 		/* Preselects the current option. In some cases the hash can disappear so we can also check for 
 		   the google translate cookie if there is no hash language code */
 		var gthash = window.location.hash.match(/\#googtrans\([A-Za-z_0-9\-]+\|([A-Za-z_0-9\-]+)\)/);
 		var gtcookie = String(readCookie('googtrans')).match(/\/[A-Za-z_0-9\-]+\/([A-Za-z_0-9\-]+)/);
 		var lang =  gthash !== null ? gthash[1] : gtcookie !== null ? gtcookie[1] : "";
 		$(".googleTranslateDropdown").val(lang);
 
 		var gtTranslate = function(lang) {
 			if (lang === "") return;
 			if (lang === 'en') deleteCookie('googtrans');
 			window.location.hash = lang === 'en' ? '' : "#googtrans(en|" + lang + ")";
 			window.location.reload(true);
 		}
 
 		$(document).ready(function() {
 			$(".googleTranslateDropdown").change(function() {
 				gtTranslate($(this).val());
 				$(".googleTranslate").addClass("active");
 			});
 		});
 	});
        </script>

            function googleTranslateElementInit() {
 		new google.translate.TranslateElement({
 			
 				gaTrack : true,
 				gaId : "UA-350639-2",
 			
 			pageLanguage: 'en'
 		}, 'google_translate_element');

 	$(function() {
 		var readCookie = function(name) {
 			var nameEQ = name + "=";
 			var ca = document.cookie.split(';');
 			for(var i=0;i < ca.length;i++) {
 				var c = ca[i];
 				while (c.charAt(0)==' ') c = c.substring(1,c.length);
 				if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
 			}
 			return null;
 		}
 
 		/* In some cases the subdomain cookie won't get deleted so google translate will retain the last
 		   translation when you navigate to a new page. This function will delete the sub-domain cookie as well. */
 		var deleteCookie = function(name) {
 			var delCookie = name + '=;expires=Thu, 01 Jan 1970 00:00:01 GMT; path=/';
 			document.cookie = delCookie;
 			document.cookie = delCookie + '; domain=.'+window.location.host;
 			document.cookie = delCookie + '; domain=.'+window.location.host.split('.').slice(-2).join('.');
 		}
 
 		/* Preselects the current option. In some cases the hash can disappear so we can also check for 
 		   the google translate cookie if there is no hash language code */
 		var gthash = window.location.hash.match(/\#googtrans\([A-Za-z_0-9\-]+\|([A-Za-z_0-9\-]+)\)/);
 		var gtcookie = String(readCookie('googtrans')).match(/\/[A-Za-z_0-9\-]+\/([A-Za-z_0-9\-]+)/);
 		var lang =  gthash !== null ? gthash[1] : gtcookie !== null ? gtcookie[1] : "";
 		$(".googleTranslateDropdown").val(lang);
 
 		var gtTranslate = function(lang) {
 			if (lang === "") return;
 			if (lang === 'en') deleteCookie('googtrans');
 			window.location.hash = lang === 'en' ? '' : "#googtrans(en|" + lang + ")";
 			window.location.reload(true);
 		}
 
 		$(document).ready(function() {
 			$(".googleTranslateDropdown").change(function() {
 				gtTranslate($(this).val());
 				$(".googleTranslate").addClass("active");
 			});
 		});
 	});
