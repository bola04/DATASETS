    <script type="text/javascript">
     function scrollTopFn(divToShow)
     {
         if (divToShow !== '') {
             divToShow = divToShow.replace('&','\\&');
             divToShow = divToShow.replace('#','');
             if ($("a[id='" + divToShow + "']").length > 0) {
                 var offSetVal = ($("header").length > 0) ? 65 : 5;
                 $('html, body').animate({scrollTop: $("a[id='" + divToShow + "']").offset().top - offSetVal}, "medium");
             }
         }
     }
     $(document).ready(function() {
         $('a[href*="#"]').click(function(e) {
             e.preventDefault();
             var divToShow = $(this).attr("href");
             scrollTopFn(divToShow);
         });
