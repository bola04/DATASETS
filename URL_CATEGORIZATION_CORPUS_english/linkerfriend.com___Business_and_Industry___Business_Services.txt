    $(".shopifyForm").submit(function(e){
         var redirect_shop = $(this).find("input[name=shop]");
         
         
         if(replaceAll(" ","",$(redirect_shop).val()).length == 0 ){
             e.preventDefault();
             return false;
         }
  
         var shop_url= replaceAll("http://","",$(redirect_shop).val());
         shop_url= replaceAll("https://","",shop_url);
         $(redirect_shop).val(shop_url)
         
     });
 
     function replaceAll(find, replace, str) {
         return str.replace(new RegExp(find, 'g'), replace);
     }
