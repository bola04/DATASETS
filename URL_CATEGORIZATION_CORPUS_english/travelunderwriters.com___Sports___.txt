     function getPartners(e) {
         e.preventDefault();
         var me = $(this);
         var parent = me.parents('.partnerFinder');
         var selectedProvince = me.find('.provinceSelect').val();
         var provinceInfo = provinces[selectedProvince];
         var j;
         var title = parent.find('.panel-heading.results h4');
 
         title.html(title.html().split(' ')[0] + ' ' + provinceInfo.name);
 
         parent.find('.panel-body.results ul').empty();
 
         if (provinceInfo.partners.length > 0) {
             for (j = 0; j < provinceInfo.partners.length; j += 1) {
                 parent.find('.panel-body.results ul').append('<li class="partner"><a href="' + provinceInfo.partners[j].url + '" target="_blank">' + provinceInfo.partners[j].name + '</a></li>')
             }
         } else {
             parent.find('.panel-body.results ul').append("No partners found");
         }
 
         parent.find('.form').fadeOut(function () {
             parent.find('.results').fadeIn();
         });
 
         $('.panel-body.results ul a').click(function() {
             ga('create', 'UA-7998416-19', 'auto');
             ga('send', 'event', 'Partner Finder', 'CTA Click', $(this).attr('href'));
         });
 
         return false;
     }
 
     function resetForm() {
         var me = $(this);
         var parent = me.parents('.partnerFinder');
 
         parent.find('.results').fadeOut(function () {
             parent.find('.form').fadeIn('');
         });
     }

     function populateBlogSlide(postData) {

     function digestWPData(data) {
         var i;
         var tmp;
         data.sort(function (a, b) {
             return a.menu_order - b.menu_order;
         });
         for (i = 0; i < data.length; i += 1) {
             var imgUrl = data[i].featured_image.source;
             tmp = $($('.posts a.post-link')[i]);
             tmp.attr('href', data[i].link);
             if (imgUrl) {
                 tmp.find('.post').css('background-image', 'url("' + imgUrl + '")');
             }
             tmp.find('.post-title').html(data[i].title);
         }
 
         populateBlogSlide(data[0]);
     }
 
     $.get('https://blog.tugo.com/' + lang + '/wp-json/posts?filter[category_name]=featured-post', digestWPData)
