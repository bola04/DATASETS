          if (filter == '') {
              filter += $(this).val();
          } else {
              filter += ',' + $(this).val();
          }
         }
      });
      
      if (filter == '') {
         for (var i=0;i<locationData.num;i++) {
            if (gmarkers[i]) {
               gmarkers[i].setVisible(false);
            }
         }

        //console.log(filters);

            for(var j=0;j<filters.length;j++) {

                 showMarker = true;
              }
            }
          
           if (showMarker == true) {
            gmarkers[i].setVisible(true);
           }
         }
       }
      }
     }
   });
