 	function QuickSearch(sender, itemText, itemValue, params)
 	{
 		var type;
 		var value;
 
 		if(itemValue)
 		{
 			type = itemValue.split("-")[0];	
 			value = itemValue.split("-")[1];
 		}
 		
 		var query;
 		switch (type) {
 			case "Provider": 
 				query = "ProviderName=" + encodeURIComponent(itemText); 
 				break;
 			case "Specialty": 
 				query = "SpecialtyId=" + encodeURIComponent(value); 
 				query += "&SpecialtyName=" + encodeURIComponent(itemText); 
 				break;
 			case "Insurance": 
 				query = "InsuranceId=" + encodeURIComponent(value);  
 				query += "&InsuranceName=" + encodeURIComponent(itemText); 
 				break;
 			case "Language": 
 				query = "LanguageId="  + encodeURIComponent(value);  
 				query += "&LanguageName=" + encodeURIComponent(itemText); 
 				break;
 			case "ClinicalInterest": 
 				query = "ClinicalInterestId=" + encodeURIComponent(value);  
 				query += "&ClinicalInterestName=" + encodeURIComponent(itemText); 
 				break;
 			case "Location": 
 				query = "OUId=" + encodeURIComponent(value);  
 				query += "&OUName=" + encodeURIComponent(itemText); 
 				break;
 			case "Affiliation": 
 				query = "AffiliationId=" + encodeURIComponent(value); 
 				break;
 			default: 
 				if (sender._text.indexOf("View All") != -1) {
 					//text the user typed into the quick search
 					query = "ProviderName=" + encodeURIComponent(sender._filterText); 
 				} else {
 					//text the user selected in the quick search
 					query = "ProviderName=" + encodeURIComponent(sender._text); 
 				}
 				
 				break;
 		}
 
 		query += "&QuickSearchText=" + encodeURIComponent(itemText) + "&";
 		query += "Random=" + encodeURIComponent($('#ctl00_PlaceHolderMain_ctl04_RandomizeHiddenField').val()) + "&";
 
 		//Add querystring values for visible controls
 		for (var key in params) {
 			if (key) {
 				var control = $("#" + params[key])[0];
 				var controlObj = $(control);
 
 				//If Control is hidden, do not insert its value into the query
 				if(!controlObj.is(":visible")) {
 					continue;
 				} else if(key == "Postal" && control.parentNode.parentNode.style.display == "none") {
 					continue;
 				} else {
 
 					var prefixLength='28';
 					query += control.id.substring(prefixLength) + "=1&";
 				}
 			}
 		}
 		query += "mode=" + $("#ctl00_PlaceHolderMain_ctl04_SearchTypeHyperLink").html();
 		query += "&PrimaryOrgUnitId=" + 2;
 
 		var providerGroups = $("#ctl00_PlaceHolderMain_ctl04_ProviderGroupsHiddenField")[0].value;
 		if(providerGroups != "")
 			query += "&ProviderGroups=" + $("#ctl00_PlaceHolderMain_ctl04_ProviderGroupsHiddenField")[0].value;
 
 		
 		var searchResultPage ='/Pages/PNRS/ProvidersSearchResult.aspx';
 		window.location =  searchResultPage + "?" + query + "&DescendantOption=" + 1 + "&LinkedOption=" + 0;
 	}
 
 	function getParameterByName(name) {
 		name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
 		var regexS = "[\\?&]" + name + "=([^&#]*)";
 		var regex = new RegExp(regexS);
 		var results = regex.exec(window.location.search);
 		if(results == null)
 			return "";
 		else
 			return decodeURIComponent(results[1].replace(/\+/g, " "));
 	}

 	function initInputValidation() {
 		if (typeof(Page_Validators) === "undefined" || typeof(Page_ValidationSummaries) === "undefined")  
 			return;
 			
 		// go through and disable all client side validators so that they don't validator onblur
 		var $validators = $(Page_Validators);
 		$validators.each(function() {
 			var $validator = $(this);
 			$validator.css({"display":"none"});
 			ValidatorEnable($validator[0], false);
 		});	
 	}
 
     function handleInputValidation(group) {
 		// ignore this method if no validators are on the page
 		if (typeof(Page_Validators) === "undefined" || typeof(Page_ValidationSummaries) === "undefined")  
 			return;
 
 		// get a reference to our expressions and jquery objects
 		var $validationSummaries = $(Page_ValidationSummaries);
 		var $validators = $(Page_Validators);
 		
 		// go through and re-enable all client side validators
 		$validators.each(function() {
 			ValidatorEnable($(this)[0], true);
 		});		
 		
 		// determine if client side validation has passed or failed
 		var isValidInput = Page_ClientValidate(group);
 
 		// initially remove any error classes on our form-groups
 		$validationSummaries.siblings('.form').find('.form-group.has-error').each(function() {$(this).removeClass("has-error");});
 		
 		// remove the hard-coded styles added to the summaries section
 		$validationSummaries.removeAttr("style");
 		
 		// do all these things when validation fails
 		if (!isValidInput) {
 			$validators.each(function() {
 				var $validator = $(this);
 
 				// if validation fails, loop through the validators and add the error class to their form-group container
 				if(!$validator[0].isvalid)
 					$validator.parents('.form-group').addClass("has-error");
 
 				// clear any inline styles that asp.net validation controls add
 				$validator.removeAttr("style");
 				
 				// attach input field onblur events to add and remove error class when necessary.
 				$validator.siblings("input, select, textarea").on("blur", function() {
 					ValidatorEnable($validator[0], false);
 					if($validator[0].isvalid)
 						$validator.parents('.form-group').removeClass("has-error");
 					else
 						$validator.parents('.form-group').addClass("has-error");
 				});
 			});
 			
 			// if we're dealing with pre-submitted form validations, remove the unordered-list for a less cluttered validation summary
 			$('form').on('submit', function() {
 				// add nice "pre-message" to show up 
 				if($validationSummaries.find('ul>li.help-block').length === 0)
 				{
 					// handle formatting of the validation summary
 					var $failed = $validationSummaries.find("ul"); 
 					$validationSummaries.find("ul").remove(); 
 					var headingText = $validationSummaries.text(); 
 					$validationSummaries.html("").prepend("<h4>" + headingText + "</h4>");
 					$validationSummaries.append($failed);
 					$validationSummaries.find('ul').prepend('<li>Please check your entries for errors and try again.</li>');
 				}
 
 				if(!Page_IsValid) {
 					// clear any inline styles that asp.net validation summaries add
 					$validationSummaries.removeAttr("style");
 					$validationSummaries.find('ul>li:not(:first-child)').remove();
 				}
 			});
 		}				
 	}
