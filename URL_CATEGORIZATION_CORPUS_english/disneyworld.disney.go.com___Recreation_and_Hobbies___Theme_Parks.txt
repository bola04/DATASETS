    (function () {
         function isMobileDevice() {
             return (typeof window.orientation !== 'undefined') || (navigator.userAgent.indexOf('IEMobile') !== -1);
         }
         function prerender() {
             try {
                 var prerenderHtml = '';
                 var defaultMobile = 'https://secure.cdn1.wdpromedia.com/resize/mwImage/1/200/200/75/' +
                     'wdpromedia.disney.go.com/media/wdpro-assets/home/hpr/' + window.globalSiteId +
                     '-default-mobile.jpg';
                 var defaultDesktop = 'https://secure.cdn1.wdpromedia.com/resize/mwImage/1/400/200/75/' +
                     'wdpromedia.disney.go.com/media/wdpro-assets/home/hpr/' + window.globalSiteId +
                     '-default-desktop.jpg';
                 if (!isMobileDevice()) {
                     if (localStorage.heroImageDesktop) {
                         prerenderHtml += '<img id="heroDesktop" alt="preloading hero image" src="' +
                             localStorage.heroImageDesktop + '">';
                     } else {
                         prerenderHtml += '<img id="heroDesktop" alt="preloading hero image" src="' +
                             defaultDesktop + '">';
                     }
                 }
 
                 if (localStorage.heroImageMobile) {
                     prerenderHtml += '<img id="heroMobile" alt="preloading hero image" src="' +
                         localStorage.heroImageMobile + '">';
                 } else {
                     prerenderHtml += '<img id="heroMobile" alt="preloading hero image" src="' +
                         defaultMobile + '">';
                 }
 
                 document.getElementById('pre-render-content').innerHTML = prerenderHtml;
             } catch (e) {}
         }
         if (window.globalSiteId !== 'dcl') {
             prerender();
         }
     })();
   </script>
