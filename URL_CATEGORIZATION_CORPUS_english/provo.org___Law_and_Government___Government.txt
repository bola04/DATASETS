        /**********************************************************

     /*javascript to delay the disappearance of the menu when the user mouses out.  Uses jquery library.*/
     $(function () {
         /*make the menus work again when going from mobile back to desktop*/
         jRes.addFunc({
             breakpoint: 'small',
             enter: function () { },
             exit: function () {
                 $('.mainnav > ul  li').bind('mouseenter', mainnav_hover());
             }
         });
         //Calculate Document Width
         var docWidth = $(document).width();
         var resizeTimer;
         $(window).resize(function () {
             clearTimeout(resizeTimer);
             resizeTimer = setTimeout(function () {
                 docWidth = $(document).width();
 
                 $(".mainnav > ul.dropdownmenu ul[caled]")
                     .removeAttr("caled")
                     .css("right", "");
             }, 200);
         });
 
         //flyout menu animation
         $(">ul", 'li.dropdownitemhaschild').addClass("slide_me");
         $(".dropdownmenu > li > ul").css("margin-top", "-10px").css("opacity", "0");
         $(".dropdownmenu ul.slide_me").css("margin-left", "140px");
         $('.dropdownmenu ul.slide_me').each(function () {
             this.style.setProperty('left', '0', 'important');
         });
 
         var timer = null;
         var delay = 800; //change this number for shorter or longer delay
 
         //using regular dropdowns
         var dropdowncontainer = '>ul';
         var activePage = $(".mainnav .dropdownmenu > li.active");
         var lastRootLevelMenuInMouseOut = null;
         mainnav_hover();
         function mainnav_hover() {
             $(".mainnav > .dropdownmenu li").hoverIntent(
                 //what happens on hover over
                  function (event) {
                      if ($(this).not(event.currentTarget).length > 0) {
                          return;
                      }
 
                      //end timer and stop hiding if they are mousing back in.
                      if (timer != null) {
                          clearTimeout(timer);
                          timer = null;
                      }
                      // Find the hovered menu's sub-menu
                      var $menu = $(this);
                      var $submenu = $(this).find(dropdowncontainer);
 
                      var rootLevelMenu = $menu;
                      var firstLevelSubMenu = $submenu;
                      while (!rootLevelMenu.closest("ul").hasClass("dropdownmenu")) {
                          firstLevelSubMenu = rootLevelMenu.closest("ul");
                          rootLevelMenu = firstLevelSubMenu.closest("li");
                      }
 
 
                      // hide any other submenus that are open in the first level
                      $('.mainnav > .dropdownmenu>li' + dropdowncontainer).not(firstLevelSubMenu).stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                      $('.mainnav > .dropdownmenu>li.active').not(rootLevelMenu).not(activePage).removeClass('active');
 
                      // show current menu in the first level
                      firstLevelSubMenu.not('.slide_me').stop(true, true).animate({ marginTop: "0px", opacity: "1" }).css('display', 'block');
                      rootLevelMenu.addClass('active');
                      if (!firstLevelSubMenu.attr("caled") && !firstLevelSubMenu.hasClass("slide_me")) {
                          var subMenuWidth = firstLevelSubMenu.outerWidth(true);
                          var subMenuLeft = firstLevelSubMenu.offset().left;
                          if (subMenuWidth > 0 && subMenuLeft + subMenuWidth > docWidth && subMenuWidth < docWidth) {
                              firstLevelSubMenu
                                  .attr("caled", "caled")
                                  .stop(true, true)
                                  .animate({ right: "0px" });
                          }
 
                      }
 
                      $(".mainnav > .dropdownmenu li._current").not($menu).removeClass('_current');
                      if (lastRootLevelMenuInMouseOut) {
                          // switch a root menu
                          if (lastRootLevelMenuInMouseOut.not(rootLevelMenu).length > 0) {
                              lastRootLevelMenuInMouseOut.find("li" + dropdowncontainer).not('.slide_me').stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                          } else {
                              var sameLevelOtherMenus = $menu.closest("ul").find(">li" + dropdowncontainer).not($submenu);
                              sameLevelOtherMenus.not('.slide_me').stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                              sameLevelOtherMenus.find("ul.slide_me").css('display', 'none').css("margin-left", "140px");
                              $($menu).find("li li " + dropdowncontainer).css('display', 'none').css("margin-left", "140px");
                          }
                      }
                      lastRootLevelMenuInMouseOut = null;
 
                      $menu.addClass("_current");
                      if ($submenu.hasClass('slide_me')) {
 
                          if ($submenu.length > 0 && !$submenu.attr("caled")) {
                              var subMenuWidth = $submenu.outerWidth();
                              var subMenuLeft = $menu.offset().left + 180;
 
                              if (subMenuWidth > 0 && subMenuLeft + subMenuWidth > docWidth && subMenuWidth) {
                                  if ($submenu.css("margin-left") == "140px") {
                                      $submenu.css("margin-left", "-160px");
                                  }
                                  $submenu.css('display', 'block').stop(true, true).animate({ marginLeft: "-200px", opacity: "1" });
                              } else {
                                  $submenu.css('display', 'block').stop(true, true).animate({ marginLeft: "180px", opacity: "1" });
                              }
                          }
                      }
 
                      //???
                      var sameLevelOtherMenus = $menu.closest("ul").find(">li" + dropdowncontainer).not($submenu);
                      if (sameLevelOtherMenus.hasClass('slide_me')) {
                          sameLevelOtherMenus.css('display', 'none').css("margin-left", "140px");
                      } else {
                          sameLevelOtherMenus.stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                      }
 
                      sameLevelOtherMenus.find("ul.slide_me").css('display', 'none').css("margin-left", "140px");
                  },
 
                 //what happens on mouse out
                 function (event) {
                     if ($(this).not(event.currentTarget).length > 0) {
                         return;
                     }
 
                     var $submenu = $(this).find(dropdowncontainer);
                     var $menu = $(this);
 
                     // get current root level and first level menu items
                     var rootLevelMenu = $menu;
                     var firstLevelSubMenu = $submenu;
                     while (!rootLevelMenu.closest("ul").hasClass("dropdownmenu")) {
                         firstLevelSubMenu = rootLevelMenu.closest("ul");
                         rootLevelMenu = firstLevelSubMenu.closest("li");
                     }
                     lastRootLevelMenuInMouseOut = rootLevelMenu;
 
                     // delay disappearance
                     clearTimeout(timer);
                     timer = setTimeout(function () {
                         // If parent menu item is hover, set its status to "_current"
                         if (!$menu.closest("ul").hasClass("dropdownmenu")) {
                             var parentMenu = $menu.closest("ul").closest("li");
                             if (parentMenu.is(":hover")) {
                                 parentMenu.addClass("_current");
                             }
                         }
 
                         $menu.removeClass("_current");
                         rootLevelMenu.not(activePage).removeClass('active');
 
                         var currentPage = $(".mainnav .dropdownmenu li._current");
                         if (currentPage.length == 0) {// no menu item is hover.
                             firstLevelSubMenu.not('.slide_me').stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                             rootLevelMenu.find("ul.slide_me").css('display', 'none').css("margin-left", "140px");
                             rootLevelMenu.not(activePage).removeClass('active');
                         } else {
                             $submenu.not('.slide_me').stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                             $submenu.find("li>ul.slide_me").css('display', 'none').css("margin-left", "140px");
                             var currentRootLevelMenu = currentPage;
                             while (!currentRootLevelMenu.closest("ul").hasClass("dropdownmenu")) {
                                 currentRootLevelMenu = currentRootLevelMenu.closest("ul").closest("li");
                             }
                             // if the root level menu item has been changed
                             if (currentRootLevelMenu.not(rootLevelMenu).length > 0) {
                                 firstLevelSubMenu.not('.slide_me').stop(true, true).animate({ marginTop: "-10px", opacity: "0" }, 400, function () { $(this).css('display', 'none') });
                                 rootLevelMenu.find("ul.slide_me").css('display', 'none').css("margin-left", "140px");
                                 rootLevelMenu.not(activePage).removeClass('active');
                             }
                         }
 
                         clearTimeout(timer);
                         timer = null;
                     }, delay);
                 }
             );
         }
 
     });
