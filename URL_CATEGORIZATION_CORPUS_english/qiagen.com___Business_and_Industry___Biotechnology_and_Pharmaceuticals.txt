         /* Declares the Header View that has header-related logic. It does not include the main navigation bar, which is handled by Q.NextQ.Views.MainNavView
         *
         */
         //
         Q.NextQ.Views.HeaderView = function () {
             //Enforce that every time a constructor function is called, this function is invoked properly using the new operator
             if (false === (this instanceof Q.NextQ.Views.HeaderView)) {
                 return new Q.NextQ.Views.HeaderView();
             }
 
             var public = this;
             var view = public.view = this;
             var _private = public._private = {};
             var globalHandlerInitialized = this.globalHandlerInitialized = { value: false };
 
             //holds the timer for hovering the logout menu
             var hoverTimeout = null;
 
             //Constants
             var HOVER_TIMEOUT = 800;
 
             ////////////////////////////////////////////////////////////////////////////////////////
             /* public functions*/
 
             public.initializeOnExecute = function () {
                 var instanceType = $('#hdnInstanceType').val();
 
                 if (instanceType == "punchout") {
 
                     $(document).on('click', '.account .contactusIcon', function (e) {
                         e.preventDefault();
 
                         $('#contact-info-wrap').fadeToggle('fast', 'linear');
 
                     });
 
                     $(document).on('click', '#contact-info-inquire .close', function (e) {
                         e.preventDefault();
 
                         $('#contact-info-wrap').fadeToggle('fast', 'linear');
                     });
 
 
                     $(document).on('click', "a.inquire", function (e) {
                         if ($("a.inquire").length) {
                             $("a.inquire").fancybox();
                         }
                     });
                 }
 
                 _private.registerHandlers();
                 if (instanceType == "b2c" || instanceType == "punchout") {
                     _private.initializeCartCount();
                 } else {
                     _private.initializeWishlistCount();
                 }
                 $('.hideNoJS').removeClass('hideNoJS');
             }
 
             public.initializeOnLoad = function () {
                 if (typeof (Sys) != 'undefined' && typeof (Sys.WebForms) != 'undefined') {
                     var prm = Sys.WebForms.PageRequestManager.getInstance();
                     prm.add_pageLoaded(view.initializeOnReady);
                 }
             }
 
             public.initializeOnReady = function () {
                 if (!globalHandlerInitialized.value) {
                     globalHandlerInitialized.value = true;
                 }
             };
 
             ////////////////////////////////////////////////////////////////////////////////////////
             /* private functions*/
 
             _private.registerHandlers = function () {
 
                 //handler for the header search field
 
                 //IE does not recognize keypress, therefore we need to use keydown.
                 var keyEvent = "keydown";
 
                 $(".header .searchValue").on(keyEvent + ".header-keycodehandler", function (event) {
                     return searchManager.searchKeyPressed(event, '.searchValue');
                 });
 
                 //******************************************* Commented by Kanica Gupta **********************************
                 //************************************************* For QEA-4679 ****************************************
                 //handler for the logout button
                 $("a[id$='lnkLogout']").on("click.header-logout", function ConfirmLogout() {
                     // required logic to avoid the browser scenarios when cartCount.textContent returns a NaN.
                     var count = _private.parseValidInt(jQueryQgn('#cartCount').text());
                     if (count > 0) {
                         var answer = confirm("This Cart contains items which will be lost on Logout. You should save the cart if not done already. Are you sure you wish to Logout?");
                         if (answer)
                             window.location.href = "/rendering/layouts/nextq/logout.aspx?Confirm=true";
                     }
                     else {
                         window.location.href = "/rendering/layouts/nextq/logout.aspx";
                     }
 
                     return false;
                 });
 
                 //************************************************* For QEA-4679 ****************************************
 
                 //handlers for the logout menu
                 $(".qlogout-menu").on("click.header-logout-menu", function (e) {
                     var logoutMenu = j$(e.target);
                     var logoutParent = j$(this).parent();
                     //if IE is 8 or less, we use a fallback menu without shadows
                     if (j$.browser.msie && j$.browser.version < "9") {
                         logoutParent.find(".qsubmenu-top-pointer").addClass("ie");
                     }
 
                     //events within the qsubmenu are ignored
                     if (logoutMenu.closest(".qsubmenu").size() === 0) {
                         var sm = logoutParent.find(".qsubmenu");
                         //20 is a magic number, due to paddings/margins
                         sm.css("min-width", logoutMenu.width() + 20);
                         var smpointer = logoutParent.find(".qsubmenu-top-pointer");
                         if (sm.is(":visible")) {
                             sm.slideUp({
                                 complete: function () {
                                     smpointer.slideUp();
                                 }
                             });
 
                         } else {
                             smpointer.toggle();
                             sm.slideToggle();
                         }
 
 
                         e.preventDefault();
                         return false;
                     }
                 });
 
                 var logoutContainer = $(".qlogout-menu").parent();
 
                 logoutContainer.on("mouseenter.header-logout-menu", function (e) {
                     clearTimeout(hoverTimeout);
                 });
 
                 logoutContainer.on("mouseleave.header-logout-menu", function (e) {
                     var logoutMenu = j$(this).find(".qlogout-menu");
                     var logoutParent = logoutMenu.parent();
                     var sm = logoutParent.find(".qsubmenu");
                     clearTimeout(hoverTimeout);
                     if (sm.is(":visible")) {
                         hoverTimeout = setTimeout(function () {
                             var smpointer = logoutParent.find(".qsubmenu-top-pointer");
                             sm.slideUp({
                                 complete: function () {
                                     smpointer.slideUp();
                                 }
                             });
                             clearTimeout(hoverTimeout);
                         }, HOVER_TIMEOUT);
                     }
                 });
             };

                 }

             };
 
             _private.initializeWishlistCount = function () {

                     url: "/rendering/controls/WishlistService.asmx/GetTotalWishListItemsCount",

                     contentType: "application/json; charset=utf-8",
                     dataType: "json",
                     success: function (msg) {
                         var counterSpan = $('#wishListCount');
                         counterSpan.html(msg.d);
                         if (msg.d != '0')
                             counterSpan.removeClass('empty');
 
                     }
                 });
             };
 
             _private.readCookie = function (name) {
                 var nameEQ = name + "=";
                 var ca = document.cookie.split(';');
                 for (var i = 0; i < ca.length; i++) {
                     var c = ca[i];
                     while (c.charAt(0) == ' ') c = c.substring(1, c.length);
                     if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
                 }
                 return "us";
             }

     var selectorClick = function () {
         $(window)
             .on('click',
                 function (e) {
                     if (!$(e.target).parents('.selector').length) {
                         $('#languageCountrySelector .selector').removeClass('active');

                         selector = $(this).parents('.selector'),
                         className = 'active';
 
                     if (!selector.hasClass(className)) {
                         selectors.removeClass(className);
                         selector.addClass(className);
                     } else {
                         selector.removeClass(className);
                     }
                 });
     };
 
     var countrySelectorHandler = function () {
         $('#countrySelector .countrySelectionLink')
             .on('click',
                 function (e) {
                     e.preventDefault();
 
                     var href = $(this).attr('href');
                     var btnId = 'md_f_main_footer_0_countryChangedButton';
                     var inputId = 'md_f_main_footer_0_hdnSelectedCountryUrl';
 
                     setHiddenInputValueAndClickButton(inputId, href, btnId);
                 });
 
         if ($('#countrySelector').length) {
             $('#countrySelector').show();
         }
     };
 
     var languageSelectorHandler = function () {
         $('#languageSelector .languageSelectionLink').on('click', function (e) {

             var href = $(this).data('lang-code');
             var btnId = 'md_f_main_footer_0_languageChangedButton';
             var inputId = 'md_f_main_footer_0_hdnSelectedLanguageCode';
 
             setHiddenInputValueAndClickButton(inputId, href, btnId);
         });
 
         if ($('#languageSelector').length) {
             $('#languageSelector').show();
         }
     };
 
     function setHiddenInputValueAndClickButton(inputId, value, buttonId) {
         $('#' + inputId).val(value);
         $('#' + buttonId).click();
