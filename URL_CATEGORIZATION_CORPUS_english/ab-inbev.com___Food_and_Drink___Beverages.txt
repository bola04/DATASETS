                     var stickyMenu = function(disableAdjust) {

                          * window scroll event callback for sticky menu bar

                         function onScrollEvent () {

                                     mainHeaderHeight = $mainHeader.height();

                                     $mainHeader.addClass('sticky-main-header');
                                     if (!disableStickyAdjust) {
                                         adjustPaddinBody(mainHeaderHeight);
                                     }
                                 } else {
                                     $mainHeader.removeClass('sticky-main-header');
                                     if (!disableStickyAdjust) {
                                         adjustPaddinBody();
                                     }
                                 }
 
                                 if($inputs && $('.sticky-main-header') && scrollPos != 0) {
                                     $('.sticky-main-header').width('100%');
                                 } else {
                                     $mainHeader.width('100%');
                                 }
                             }
                         }

                          * function to adjust sticky menu bar on resize event.
                          * @param  {string} padding paddinTop value
