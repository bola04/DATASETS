     function InitSlider(ageFrom, ageTo) {
         var minAge = 18;
         var maxAge = 78;
         var oldValue = "old-value";
         $("input#ageFrom").val(ageFrom);
         $("input#ageTo").val(ageTo);
         $("input#ageFrom, input#ageTo").focus(function() {
             this.select();
             $(this).attr(oldValue, $(this).val());
         });
         $("input#ageFrom").change(function() {
             var value = parseInt($(this).val());
             var max = parseInt($("input#ageTo").val());
             if (max && value && value <= max && value >= minAge)
                 $("#slider-range").slider('values', 0, value);
             else
                 $(this).val($(this).attr(oldValue));
         });
         $("input#ageTo").change(function() {
             var value = parseInt($(this).val());
             var min = parseInt($("input#ageFrom").val());
             if (min && value && value >= min && value <= maxAge) {
                 $("#slider-range").slider('values', 1, value);
                 if (value == maxAge)
                     $("#maxAgePlus").show();
                 else
                     $("#maxAgePlus").hide();
             } else
                 $(this).val($(this).attr(oldValue));
         });

                 $("input#ageFrom").val(ui.values[0]);
                 $("input#ageTo").val(ui.values[1]);
                 if (ui.values[1] == maxAge)
                     $("#maxAgePlus").show();
                 else
                     $("#maxAgePlus").hide();
             }
         });
