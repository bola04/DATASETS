     HomePgAdjustments = function () {
         if ($(window).width() >= 992) {
             $("#homeIntroImgRow").hide();
             $("#homeIntroSection").addClass("home-intro");
             $("#homeSolutionSection").addClass("section-grey-background");
             $("#homeSolutionImgRowMobile").hide();
             $("#homeSolutionBBB").show();
             $("#homeSolutionBBBMobile").hide();
             $("#homeProtect").append($("#homeProtectContentSection"))
         }
         else {
             $("#homeIntroImgRow").show();
             $("#homeIntroSection").removeClass("home-intro");
             $("#homeSolutionSection").removeClass("section-grey-background");
             $("#homeSolutionImgRowMobile").show();
             $("#homeSolutionBBB").hide();
             $("#homeSolutionBBBMobile").show();
             $("#homeProtectColMobile").append($("#homeProtectContentSection"));
         }
         
         if ($(window).width() >= 750) {
             alignRowLink(1);
             alignRowLink(2);
         }
 
         if ($("#homeIntroCol1Text").height() > $("#homeIntroCol2Text").height())
         {
             adjustTopMargin(1, 2);
         }
         else if ($("#homeIntroCol1Text").height() < $("#homeIntroCol2Text").height())
         {
             adjustTopMargin(2, 1, ($("#homeIntroCol2Text").height() - $("#homeIntroCol1Text").height()));
         }
     }
 
     function adjustTopMargin(taller, shorter)
     {
         var heightDif = $("#homeIntroCol"+taller+"Text").height() - $("#homeIntroCol"+shorter+"Text").height();
         var margin = parseInt($("#btnHomeIntroCol" + taller).css("margin-top"));
         $("#btnHomeIntroCol" + taller).css("margin-top", 30);
         $("#btnHomeIntroCol" + shorter).css("margin-top", 30 + heightDif);
     }
 
     function alignRowLink() {
         $("[id^=homeTaxResourcesRow]").each(function () {
             var maxHeight = $(this).height();
             // For each col in row
             $(this).find("[id^=homeTaxResourceCell]").each(function () {
                 var adjust = maxHeight - $(this).height();
                 $(this).find("[id^=homeTaxResourceLink]").each(function () {
                     var currentMargin = parseInt($(this).css("margin-top"));
                     if (currentMargin > 30) currentMargin = 30;
                     $(this).css("margin-top", (currentMargin + adjust));
                 });
             });
         });
     }
