         var geocoder = new google.maps.Geocoder();
 
     function searchProvider()
     {
         var address = jQuery('input#searchAddress').val();
         var zip = jQuery('input#searchZip').val();
 
         if(address === jQuery('input#searchAddress').attr('placeholder'))
         {
             address = '';
         }
 
         if(zip === jQuery('input#searchZip').attr('placeholder'))
         {
             zip  = '';
         }
 
         if (address+zip == ''){
             alert('Oops... Please enter your address and/or Zip code!');
             return false;
         }
         if (zip && (address == '')) {
             address = zip;
         }
 
         geocoder.geocode({ 'address': address}, function (results, status) {
             if(status == google.maps.GeocoderStatus.OK)
             {
                 var lat = results[0].geometry.location.lat();
                 var lng = results[0].geometry.location.lng();
 
                 location.href = "/providers/index/search/?data_categories=&data_billing=&address=" + address + "&zip=" + zip + "&lat=" + lat + "&lng=" + lng + "&button=Search";
             }
             else
             {
                 alert("Sorry we can't find your address");
             }
         });
 
         return false;
     }
