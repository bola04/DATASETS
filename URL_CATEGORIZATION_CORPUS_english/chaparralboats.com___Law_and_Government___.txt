 	jQuery(document).ready(function(){
 
     if (navigator.geolocation) {
         navigator.geolocation.getCurrentPosition(setCoords, forceCoords);
     } 
     else {
         //HTML5 Geo location Not supported use fallback
         forceCoords();
     }

     function setCoords(position){
     	if (position.coords.latitude == "")
     	{
     		forceCoords();
     	}
     	else
     	{
     		jQuery("#latitude").val(position.coords.latitude);
         	jQuery("#longitude").val(position.coords.longitude);
         }
     }
     
     // Fallback function - uses provider/tower location info rather than exact customer location
     function forceCoords(latitude, longitude){
     	jQuery.ajax( { 
         	url: '//freegeoip.net/json/', 
   			type: 'POST', 
   			dataType: 'jsonp',
   			success: function(location) {
     		var longitude = location.longitude;
     		var latitude = location.latitude;
     		jQuery("#latitude").val(latitude);
     		jQuery("#longitude").val(longitude);
     		}
 		});
 
     }
