 function countrychg(form) { //country selection changed
  var csi,found
  found=false
  csi=form.country.selectedIndex
  for (var j=cdlinkidx; j<cd[csi].length; j++)
   if ((csi>0)&&(cd[csi][j]==""))
    form.item.options[j-cdnameidx].text=""
   else {
    form.item.options[j-cdnameidx].text=cd[0][j]
    found=true
   }
  if (found) form.item.options[0].text=cd[0][cdnameidx]
  else form.item.options[0].text="No data available"
  form.item.selectedIndex=0
  return
 }
 
 
 function itemchg(form) { //item selection changed
  var csi,isi,isi1
  csi=form.country.selectedIndex
  isi=form.item.selectedIndex
  isi1=isi+cdlinkidx-1
  if ((csi>0)&&(isi>0)) {
   if(cd[csi][isi1]!="") window.location.href=cd[csi][isi1]
   else form.item.selectedIndex=0
  }
  return
 }
 
 
 function dalert(as,add) { // date alert: insert resp. mark, if date not passed
 // as = alert Date string in MEZ, add = days to add to as
 // write output in document
  document.write(dalerts(as,add))
 }
 
 
 function dalerts(as,add) { // date alert: insert resp. mark, if date not passed
 // as = alert Date string in MEZ, add = days to add to as
 // output is string
  var d1 = 10; // days before alert date for 1st alarm
  var d2 = 5; // days before alert date for 2nd alarm
  var r = (Date.parse(as) - nowtime)/mspday + add + 1; // remainder - days
  var x=""
  if (r > 0) {
   x+='<IMG SRC="ico/at.gif" width="12" height="10" title="Alert"> '
   if (r <= d2) {
    d=d2
    x+='<IMG SRC="ico/at.gif" width="12" height="10" title="Alert"> '
   }
   if (r <= d1) {
    x+='<IMG SRC="ico/at.gif" width="12" height="10" title="Alert"> <FONT color="#BF0000">(less than '+Math.ceil(r)+' day'
    if (r>1) x+='s'
    x+=' left for submissions):</FONT><BR>'
   }
  }
  return x
 } //dalerts
 
 
 
 
 function elfind(el) { //find element texts in rss feed
 //el=element object with public tags:  title, link, pubDate, and namespace tag:  wup:alertDate
  var ad, pd, x
     x='<li>'
     pd=el.find("pubDate").text()
     ad=el.find("wup\\:alertDate, alertDate").text() //escape colon of namespace tag
     if (ad!=null)
      x+='<font size="-1">'+dalerts(ad,0)+'</font>'
     //eliminate base addr. for offline use
     x+='<b><font size="-1"><a href="'+el.find("link").text().split("/")[3]+'">';
     x+=el.find("title").text()+'</a></font></b><br>';
     x+='<font size="-1">'+pd+'</font><br>';
     x+="<br>";
  return x
 } //elfind
