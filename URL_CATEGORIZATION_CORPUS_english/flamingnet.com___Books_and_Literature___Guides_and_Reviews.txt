     let captchaTemplate = '<ul class="captcha"><li class="numbers">[Number1] + [Number2]</li>'
                         + '<li><input name = "captcha" placeholder="Enter result" > '
                         + '<input type="hidden" name = "captchaResult" value = "[CaptchaResult]" >'
                         + '<span class = "error captcha" style = "display: none;" >Incorrect Answer. Try Again.</span>'
                         + '</li>';
     function captcha( targetForm, errorMessage ) {
     if ( typeof errorMessage === "undefined" ) {
         let errorMessage    = "Incorrect Answer. Try Again.";
     }
     let number1     = Math.floor( Math.random() * 10 );
     let number2     = Math.floor( Math.random() * 10 );
     let result      = parseInt( number2 ) + parseInt( number1 );
     let captchaTpl  = captchaTemplate.replace( "[Number2]", number2 );
     captchaTpl      = captchaTpl.replace( "[Number1]", number1 );
     captchaTpl      = captchaTpl.replace( "[CaptchaResult]", result );
     captchaTpl      = captchaTpl.replace( "[CaptchaError]", errorMessage );
     let formObj = $( targetForm );
     formObj.find( ".captcha-temp" ).prepend( captchaTpl);
 formObj.submit( function( e ) {
         let myForm  = $( e.target );
         let result  = myForm.find( "input[name=captchaResult]" ).val();
         let humanRes = myForm.find( "input[name=captcha]" ).val();
         if ( result == humanRes ) {
             myForm.submit();
         } else {
             e.preventDefault();
             myForm.find( "span.captcha" ).slideDown();
             myForm.find( ".captcha input" ).css('border-color', '#d60f0a');
         }
 
     });
 }
 
 $(document).ready( function() {
     captcha( "#contactForm" );
     $('input[name="captcha"]').click(function() {
         $('.error.captcha').hide();
         $( ".captcha input" ).css('border-color', '#ccc');
     });
 
 });
    </script>
