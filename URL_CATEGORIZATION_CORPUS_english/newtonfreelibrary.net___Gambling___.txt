     //Below controls Slideshow auto rotate and button controls//
 //Auto Rotating Slide Code//
 var currentIndex = 0,                                                              //Sets current index to 0 so autoSlide can properly cycle through list//
     items = $('#reads_div div.slide_div'),                                         //Creates list of slide divs//
     headers = $('#reads_div h3'),                                                  //Creates list of slide headers//
     itemAmt = items.length;                                                        //Determines number of slide divs//
 
 function cycleItems() {                                                            //Controls the slide div fade and dot button active states
     var item = $('#reads_div div.slide_div').eq(currentIndex),                     //Sets item variable to whichever slide div corresponds to the currentIndex variable//
         dot = $('#slide_dots li').eq(currentIndex),                                //Sets dot variable to whichever dot button corresponds to the currentIndex variable//
         header = $('#reads_div h3').eq(currentIndex);                              //Sets header variable to whichever slide header corresponds to the currentIndex variable//
     items.hide();                                                                  //Hides all slide divs//
     headers.hide();                                                                //Hides all slide headers//
     header.fadeIn();                                                               //Fade in proper slide header based on currentIndex//
     item.fadeIn();                                                                 //Fade in proper slide div based on currentIndex//
     if (dot.attr('class') === 'active') {                                          //If dot button has active class during this cycle remove active class//
         $('#slide_dots li').attr('class', '');
     }
     else {                                                                         //If dot button does not have active class during this cycle add active class//
         $('#slide_dots li').attr('class', '')
         dot.addClass('active');
     }
 } 
 function autoSlide() {                                          
     currentIndex += 1;                                                             //Advance currentIndex by one every cycle//
     if (currentIndex > itemAmt - 1) {                                              //If currentIndex is ever greater than itemAmt minus one reset currentIndex to zero//
         currentIndex = 0;
     }
     cycleItems();                                                                 //Call to cycleItems which controls the display of divs//
 }
 
 var slideshowTimer = setInterval(autoSlide, 5000);                                 //Timer for automatic slideshow, takes the autoSlide function and a time in milliseconds//
 
 //Dot Button Control Code//
 $('#slide_dots li').click(function (e) {                                           //Dot button click event handler for dot controls, selects slide_dots//
     $('#slide_dots li').attr('class', '');                                         //Clears dot button of any class//
     var id_dot = $(this).attr('name');                                             //Sets id_dot equal to the name attribute of the clicked dot button//
     currentIndex = $('#slide_dots li').index(this);                                //Sets currentIndex of slide cycle script to the position of clicked dot button within slide_dots array//
     if ($("#" + $(this).attr('name')) === id_dot) {                                //If clicked dot button is active remain active//
         $(this).attr('class','active');				
     }
     else {
         $(this).attr('class', 'active');                                           //If clicked dot button is not active make active//
         $('#reads_div div.slide_div').hide();                                      //Hide all slide divs//
         $('#reads_div h3').hide();                                                 //Hide all slide headers//
         $('#' + $(this).attr('name')).fadeIn();                                    //Fade in slide div that has ID attribute equal to name attribute of clicked control dot// 
         $('#head' + $(this).attr('name')).fadeIn();                                //Fade in slide header that has ID attribute equal to '#head' + name of clicked control dot Ex. '#headdiv1'//
         clearInterval(slideshowTimer);                                             //Stops slideshowTimer upon clicking inactive dot//
         slideshowTimer = setInterval(autoSlide, 5000);                             //Reactivates and initializes slideshowTimer after clicking inactive dot
     }			
 });
    </script>
