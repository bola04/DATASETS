 	// str				The user-entered search string.
 	// include_trusts	Whether trusts (series O) funds should be included in the results.
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.filter_by_string = function( str, include_trusts )
 	{
 		if( !fc.m_funds )
 			return false;
 		
 		var filter_sets = fc.parse_filter_string( str );
 		if( !filter_sets )
 			return false;
 		
 		return fc.filter( filter_sets, include_trusts );
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// filter
 	//
 	// filter_sets		An array of groups of search parameters. Each group represents a search
 	//					to be conducted (e.g., Name contains "balanced" and Series = "B" or "F").
 	//					See fc.parse_filter_string for structure.
 	//
 	// include_trusts	Should these results include trust funds (series O).
 	// 
 	// Returns			
 	//					The results of all searches are merged into a structure like so:
 	//					{
 	//						axis_codes: { axis_code: [...], ... },
 	//						fund_codes: { fund_code: [...], ... },
 	//					}
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.filter = function( filter_sets, include_trusts )
 	{
 		if( filter_sets === false )						// If passed one target set, turn it into an array.
 			filter_sets = new Object();
 		
 		if( typeOf( filter_sets ) === 'object' )		// If passed one target set, turn it into an array.
 			filter_sets = [ filter_sets ];
 		
 		// There are two data structures to search: the fund-code structure (containing the 
 		// fund code, series, and load), and the axis-code structure, with the name and currency.
 		var fc_recs = fc.m_funds['fund_codes'];		// fc's: Data stored by fund-code (number, series, load)
 		var ac_recs  = fc.m_funds['axis_codes'];	// ac's: Data stored by axis-code (name, currency)
 		var top_funds = fc.m_funds['top_funds'];	// weekly top 5 selling funds category: array of axis codes
 		var merged_results = new Object( { 'axis_codes': new Object(), 'fund_codes': new Object() } );
 		
 		$.each( filter_sets, function ( i, filter_set )
 		{
 			var matched_fc_recs = new Object();
 			var matched_ac_recs = new Object();
 			
 			// PHASE 1
 			// Start with searching for matches with fund code records since these are short/fast comparisons.
 			var filter_fc_recs = filter_set['fund_code'] || filter_set['series'] || filter_set['load'] || filter_set['category'];
 			if( filter_fc_recs )
 			{
 				$.each( fc_recs, function ( fund_code, fc_rec )
 				{
 					// Fund code is a string padded with zeros for less than 3 digits
 					//fund_code = parseInt( fund_code );	// convert from string.
 					fund_code = fc.padZeros( fund_code, FUND_CODE_ZERO_PAD_WIDTH );
 					
 					if( filter_set['fund_code'] && filter_set['fund_code'].indexOf( fund_code ) === -1 )
 						return;
 					
 					if( filter_set['series'] && (filter_set['series'].indexOf( fc_rec[ i_SERIES ] ) === -1) )
 						return;
 					
 					if( filter_set['load'] && (filter_set['load'].indexOf( fc_rec[ i_LOAD ] ) === -1) )
 						return;
 					
 					// If watchlist category is selected, make sure this fund code is in there
 					if( filter_set['category'] && filter_set['category'].length 
 							&& $.inArray( 'watchlist', filter_set['category']) !== -1 )
 					{
 						if ( typeof( watch ) === 'undefined' || ! watch.hasFund( fund_code ) )
 							return;
 					}
 					
 					// It should be impossible to get to this point if we haven't found a match for each target.
 					
 					if( fc_rec[ i_SERIES ] !== 'O' || include_trusts )	// Are we including trusts in this search?
 					{
 						matched_fc_recs[ fund_code ] = fc_rec;	// Store this is a matched fund code w/ record.
 						
 						// We only need to search the axis_codes that relate to found fund code.
 						var axis_code = fc_rec[ i_AXIS_CODE ];
 						matched_ac_recs[ axis_code ] = ac_recs[ axis_code ]		// Below, we will search matched_ac_recs for name and currency
 					}
 				} );
 			}
 			else	// If we didn't do any filtering above, then fill out the "matched_ac_recs" with all records.
 			{
 				matched_fc_recs = $.extend( {}, fc_recs );
 				matched_ac_recs = $.extend( {}, ac_recs );
 			}
 			
 			// PHASE 2
 			// Search the matched_ac_recs and remove any that don't match the name and currency filter_set.
 			var filter_ac_recs = filter_set['name'] || filter_set['currency'] || filter_set['category'];
 			if( filter_ac_recs )
 			{	// OUTER EACH LOOP:
 				$.each( matched_ac_recs, function ( axis_code, ac_rec )
 				{	
 					
 					
 					// If this is not the currency we want, remove it from matches.
 					if( filter_set['currency'] && !fc.compareCurrency( filter_set['currency'], ac_rec[ i_CURRENCY ] ) )
 					{
 						fc.remove_matched_fc_recs( ac_rec[ i_REL_FUND_CODES ], matched_fc_recs );
 						delete matched_ac_recs[ axis_code ];
 						return;		// "continue" with next axis code (this one has now been removed).
 					}
 
 					// If this is not the category we want, remove it from matches.
 					if( filter_set['category'] && filter_set['category'].length )
 					{
 						var found = false;
 						
 						$.each( filter_set['category'], function ( i, category_cd ) 
 						{
 							// Special category: the check already happened above with the fund code. If   
 							// we've gotten here it must have passed the first check so just mark it found.
 							if (category_cd === 'watchlist') {
 								found = true;
 							}
 							
 							// Special category: top 5 funds
 							else if ( category_cd === 'top_5' && top_funds && 
 									$.inArray( axis_code, top_funds ) !== -1 ) {
 								found = true;
 							}
 							
 							// OR: The fund must match at least one of the selected fund categories
 							else if ( ac_rec[ i_REL_CATS ] && ac_rec[ i_REL_CATS ].length ) {
 								$.each( ac_rec[ i_REL_CATS ], function ( i, ac_category_arr ) 
 								{
 									var ac_category_cd = ac_category_arr[0];
 									if ( category_cd == ac_category_cd )
 										found = true;
 								} );
 							}
 						} );
 						
 						if ( !found ) {
 							fc.remove_matched_fc_recs( ac_rec[ i_REL_FUND_CODES ], matched_fc_recs );
 							delete matched_ac_recs[ axis_code ];
 							return;		// "continue" with next axis code (this one has now been removed).
 						}
 					}
 					
 					if( filter_set['name'] )
 					{
 						var found = 0;
 						var fund_name = ac_rec[ i_NAME ].toUpperCase();	// i_NAME index is adjusted for current page language.
 						if( ac_rec[ i_KEYWORDS ] && ac_rec[ i_KEYWORDS ].length )
 							fund_name += " " + ac_rec[ i_KEYWORDS ];
 						
 						// MIDDLE EACH LOOP: We must match all keywords in the 'name'
 						$.each( filter_set['name'], function ( i, name_target )
 						{	
 							var re = new RegExp( '\\b' + name_target );
 							if( fund_name.search( re ) < 0 )
 								return false;
 							
 							found++;	// Record that a name_target was found.
 						} );
 						
 						if( found !== filter_set['name'].length )
 						{
 							fc.remove_matched_fc_recs( ac_rec[ i_REL_FUND_CODES ], matched_fc_recs );
 							delete matched_ac_recs[ axis_code ];
 							return;		// "continue" with next axis code (this one has now been removed).
 						}
 					}
 				} );
 			}
 			
 			$.extend( merged_results['axis_codes'], matched_ac_recs );
 			$.extend( merged_results['fund_codes'], matched_fc_recs );
 		} );
 
 		if( !include_trusts )
 			fc.filter_out_trusts( merged_results );
 		
 		// If our search filters didn't match anything...
 		if( !Object.keys( merged_results['axis_codes'] ).length )	
 			return false;
 		
 		return merged_results;
 	};
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// filter_out_trusts
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.filter_out_trusts = function( results )
 	{
 		if( !results )
 			return;
 		
 		fc.remove_matched_fc_recs( fc.m_funds['trust_fund_codes'], results['fund_codes'] );
 		
 		$.each( fc.m_funds['trust_single_fund_axis_codes'], function ( i, axis_code )
 		{
 			if( results['axis_codes'][ axis_code ] )
 				delete results['axis_codes'][ axis_code ];
 		} );
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// remove_matched_fc_recs
 	//
 	// This utility function is passed an array of fund codes to remove from matched_fc_recs.
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.remove_matched_fc_recs = function( fund_codes, matched_fc_recs )
 	{
 		var n = fund_codes.length;
 		
 		for( var i = 0; i < n; i++ )
 			delete matched_fc_recs[ fund_codes[i] ];
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// parse_filter_string
 	// 
 	// Creates an array of structured search objects from the user-entered string, to be used by
 	// fc.filter(). Most of the time there is only one interpretation for a string, and so
 	// only one search object in the array. When a search term is ambiguous, there may be multiple
 	// search objects in the array. All found records are merged.
 	//
 	// e.g.,: "2010" needs to do a search for fund_code = 2010 but also name-contains 2010.
 	// 
 	// Returned search objects will vary but, at most, can contain all the elements shown in the 
 	// filter_set variable initialized below.
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.parse_filter_string = function( filter_string )
 	{
 		filter_string = fc.apply_prefilters( filter_string );
 		
 		var tokens = filter_string.trim().split(' ');
 		
 		if( !tokens.length || (tokens.length === 1 && tokens[0] === '') )
 			return false;
 		
 		var filter_sets = [];
 		var filter_set = new Object(
 		{
 			'fund_code': [],
 			'currency':  false, // Only 1 currency assignable
 			'load':      [],
 			'name':      [],
 			'series':    [],
 		} );
 		
 		
 		$.each( tokens, function ( i, token )
 		{
 			var upper_token = token.trim().toUpperCase();
 			
 			if( fc.m_funds['loads'].indexOf( upper_token ) > -1 )
 			{
 				if( filter_set['load'].indexOf( upper_token ) === -1 )
 					filter_set['load'].push( upper_token );
 			}
 			else if( fc.m_funds['series'].indexOf( upper_token ) > -1 )
 			{
 				if( filter_set['series'].indexOf( upper_token ) === -1 )
 					filter_set['series'].push( upper_token );
 			}
 			else if( fc.m_funds['currencies'].indexOf( upper_token ) > -1 )
 			{
 				
 				var fsetCurr = upper_token;
 				if ('CAD' == fsetCurr) {
 					fsetCurr = 'CDN';
 				} else if ('USD' == fsetCurr) {
 					fsetCurr = 'US';
 				}
 
 				filter_set['currency'] = fsetCurr;
 			}
 			else
 			{	// Is this a fund code?
 				var num = parseInt( upper_token );
 				
 				if( !isNaN( num ) && num > 0 && num < MAX_FUND_CODE && yearClearPath.indexOf( num ) === -1 ) 
 				{
 					// Fund code is a minimum 3 digit string padded with zeros
 					var num_str = fc.padZeros( upper_token, FUND_CODE_ZERO_PAD_WIDTH ); 
 					if( filter_set['fund_code'].indexOf( num_str ) === -1 )
 						filter_set['fund_code'].push( num_str );
 				}
 				else
 				{	// Strings less than 3 chars aren't searchable.
 					if( upper_token.length < 3 )
 						return;
 					
 					if( filter_set['name'].indexOf( upper_token ) === -1 )
 						filter_set['name'].push( upper_token );
 				}
 			}
 		} );
 		
 		fc.apply_postfilters( filter_set );
 		
 		filter_sets.push( filter_set );
 		fc.create_alternate_searches( filter_sets );
 		
 		// Clear out the empty search items.
 		$.each( filter_sets, function ( i, filter_set )
 		{
 			if( filter_set['fund_code'].length === 0 ) delete( filter_set['fund_code'] );
 			if( !filter_set['currency'] )              delete( filter_set['currency'] );
 			if( filter_set['load'].length === 0 )      delete( filter_set['load'] );
 			if( filter_set['name'].length === 0 )      delete( filter_set['name'] );
 			if( filter_set['series'].length === 0 )    delete( filter_set['series'] );
 		} );
 		
 		if( !Object.keys( filter_sets[0] ).length )
 			return false;
 		
 		return filter_sets;
 	};		
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// apply_prefilters
 	// 
 	// These filters look for known patterns like /fid\d+/ and translate that into just a fund code.
 	// These filters are defined in the m_funds.
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.apply_prefilters = function( filter_string )
 	{
 		if( !fc.m_funds['search_prefilters'] )
 			return filter_string;
 		
 		if( typeof( filter_string ) !== 'string' )
 			return '';
 		
 		$.each( fc.m_funds['search_prefilters'], function ( i, filter )
 		{
 			var re = new RegExp( filter[0], 'gi' );
 			filter_string = filter_string.replace( re, filter[1] );
 		} );
 		
 		return filter_string;
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// apply_postfilters
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.apply_postfilters = function( filter_set )
 	{
 		if( !fc.m_funds['search_postfilters'] )
 			return filter_set;
 		
 		$.each( fc.m_funds['search_postfilters'], function ( i, filter )
 		{
 			if( !filter_set['name'] )
 				return;
 			
 			$.each( filter_set['name'], function ( i, name )
 			{
 				var re = new RegExp( filter[0], 'gi' );
 				filter_set['name'][i] = filter_set['name'][i].replace( re, filter[1] );
 			} );
 		} );
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// create_alternate_searches
 	//
 	// Not all search strings have only one interpretation. At the moment, the only alternative
 	// is that fund codes like "2010" could be a fund code, but could also be part of a name.
 	// This must be done as an entirely separate search object because "equity 2010"
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.create_alternate_searches = function( filter_sets )
 	{
 		var new_filter_set;
 		var new_filter_set_created = false;
 		
 		// =========================
 		// Special handling for use of "US" string, which was pre-filtered to "XUSX".
 		// We want to change the original search to be for "U.S." but add an alternate search for "US$" or "$ US".
 		
 		var i = filter_sets[0]['name'].indexOf('XUSX');			// flag string created by prefilters out of "US".
 		if( i >= 0 )
 		{
 			filter_sets[0]['name'][i] = 'U\\.S\\.';								// Backslash for regexp
 			new_filter_set = $.extend( true, {}, filter_sets[0] );				// Clone current target.
 			
 			new_filter_set['name'][i] = (LANG === 'fr') ? '\\$ US' : 'US\\$';	// Backslash for regexp.
 			filter_sets.push( new_filter_set );
 		}
 		
 		// =========================
 		// Check for the scenario where a fund code has been specified that could also be a year in a name.
 		// This only creates the addition search for years that are modulo 5 (2005, 2010, 2015...).
 		new_filter_set = $.extend( true, {}, filter_sets[0] );	// Clone current target.
 		new_filter_set_created = false;
 		
 		$.each( new_filter_set['fund_code'], function ( i, fund_code )
 		{
 			// Fund code is a 3 digit string padded with zeros
 			var num_code = parseInt( fund_code.replace(/^[0]+/g,"") );
 			if( num_code >= 2005 && num_code <= 2100 && !(num_code % 5) )
 			{
 				new_filter_set_created = true;
 				new_filter_set['name'].push( fund_code );
 				new_filter_set['fund_code'].splice(i, 1);
 			}
 		} );
 		
 		if( new_filter_set_created )
 			filter_sets.push( new_filter_set );
 		
 		// =========================
 	}
 
 
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// store_funds
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.store_funds = function( searchFiltersData )
 	{
 		if( typeOf( searchFiltersData ) !== 'object' || !searchFiltersData['search_prefilters'] )
 		{
 			console.log( "Fund Cache: Bad data returned from server." );
 			return;
 		}
 
 		var fundDatasObj = fc.convertFundDataForSearch();
 		var axisCodeDataObj = fundDatasObj['axis_codes'];
 		var fundCodeDataObj = fundDatasObj['fund_codes'];
 		var subcatsObj = fundDatasObj['subcats'];
 
 		fundDatasObj['search_prefilters'] = searchFiltersData['search_prefilters'];
 		fundDatasObj['search_postfilters'] = searchFiltersData['search_postfilters'];
 
 		fc.build_subcat_to_axis_lookup( subcatsObj, axisCodeDataObj );
 		fc.build_axis_code_lookups( axisCodeDataObj, fundCodeDataObj );
 		fc.build_trusts_list( fundDatasObj, axisCodeDataObj, fundCodeDataObj );
 
 		fc.m_funds = fundDatasObj;
 		fc.init_dependents();
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// build_subcat_to_axis_lookup
 	// 
 	// We need to add a structure to the sub-categories to find their related default axis_codes.
 	// Both parameters are an object so is pass-by-reference.
 	// 
 	// "CDNE":
 	//  [
 	//		name,
 	//		name_fr,
 	//      [A, AB, ABC,...]  <-- associated axis codes. This is what this function adds.
 	//	]
 	//
 	// A fund_record looks like:
 	//    "1":["A","LL2","UGD"]
 	// or
 	//    fund_code: [ series, load, axis_code ]
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.build_subcat_to_axis_lookup = function( subcats, axis_codes )
 	{
 		var first_subcat_id;	// We just need something to use to get the new object length.
 		
 		$.each( subcats, function( i, subcat_record )
 		{
 			if( !first_subcat_id )
 				first_subcat_id = i;
 			
 			subcats[i].push( [] );	// Initialize
 		} );
 		
 		// Get the index of the empty array we just created above.
 		var i_related_axis_codes = subcats[ first_subcat_id ].length - 1;
 		
 		$.each( axis_codes, function( axis_code, axis_code_record )
 		{
 			var def_subcat = axis_code_record[ i_DEF_SUBCAT ];
 			
 			if (def_subcat == null || subcats[ def_subcat ] == null) {
 				console.log("ERROR: Invalid subcategory " + def_subcat + " for fund " + axis_code + ". Skipping.");
 				return;
 			}
 			
 			// Add this axis_code to the array added to the axis_code record, initialized above.
 			subcats[ def_subcat ][ i_related_axis_codes ].push( axis_code );
 		} );
 	}
 	
 	
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	// build_axis_code_lookups
 	// 
 	// We need to add a structure to every axis_code record so we can easily connect it to its
 	// associated funds and series.
 	//
 	// "AD":
 	//   [
 	//		"Fidelity American Disciplined Equity Fund",
 	//		"Fonds Fidelity Discipline Actions(MD) Amerique",
 	//		"",         <-- english keywords
 	//		"Amerique",	<-- french keywords
 	//		"CAD",
 	//		"ABC",	          <-- default sub-category
 	//		["ABC","DEF"],    <-- sub-categories
 	//      [6, 83, 206,...]  <-- related fund codes.
 	//      ['A', 'B', 'O']   <-- related series.
 	//	 }
 	//
 	// A fund_record looks like:
 	//    "1":["UGD","A","LL2"]
 	// or
 	//    fund_code: [ axis_code, series, load ]
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.build_axis_code_lookups = function( axis_codes, fund_codes )
 	{
 		$.each( axis_codes, function( axis_code, ac_rec )
 		{
 			ac_rec[ i_REL_FUND_CODES ] = [];	// Initialize
 			ac_rec[ i_REL_SERIES ] = [];
 		} );
 		
 		$.each( fund_codes, function( fund_code, fc_rec )
 		{
 			var axis_code = fc_rec[ i_AXIS_CODE ];
 			
 			// Add this fund_code to the array added to the axis_code record, initialized above.
 			axis_codes[ axis_code ][ i_REL_FUND_CODES ].push( fund_code );	
 
 			var ac_rec = axis_codes[ axis_code ];
 			var series = fc_rec[ i_SERIES ];
 			
 			if( ac_rec[ i_REL_SERIES ].indexOf( series ) < 0 )
 				ac_rec[ i_REL_SERIES ].push( series );
 		} );
 		
 		// Sort the found series.
 		$.each( axis_codes, function ( axis_code, ac_rec )
 		{
 			ac_rec[ i_REL_SERIES ].sort();
 		} );
 	}

 	// build_trusts_list
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.build_trusts_list = function( data, axisCodeData, fundCodeData )
 	{
 		var trusts_list = [];
 		
 		$.each( fundCodeData, function ( fund_code, fc_rec )
 		{
 			if( fc_rec[ i_SERIES ] === 'O' )
 				trusts_list.push( fund_code );
 		} );
 		
 		data['trust_fund_codes'] = trusts_list;
 
 		
 		// Search axis codes that only have 1 fund that's series "O".
 		
 		var single_fund_axis_codes = [];
 		
 		$.each( axisCodeData, function ( axis_code, ac_rec )
 		{
 			var rel_series = ac_rec[ i_REL_SERIES ];
 
 			if( rel_series.length === 1 && rel_series[0] === 'O' )
 				single_fund_axis_codes.push( axis_code );
 		} );
 		
 		data['trust_single_fund_axis_codes'] = single_fund_axis_codes;
 	}

 	// get
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.get = function( type, index )
 	{
 		if( !fc.m_funds )
 			return false;
 		
 		var response = fc.m_funds;
 		
 		if( typeof( type ) !== 'undefined' )
 		{
 			if( response[ type ] )
 				response = response[ type ];
 			else
 				return false;
 		}
 		
 		if( typeof( index ) !== 'undefined' )
 		{
 			if( response[ index ] )
 				response = response[ index ];
 			else
 				return false;
 		}
 		
 		return response;
 	}

 	// is_available
 	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//
 	
 	fc.is_available = function()
 	{
 		return !!fc.m_funds;
 	}

 	fc.convertFundDataForSearch = function() {
 
 		var fundInfoBySeries = ficDataLayer.fundInfoBySeries;
 		var fundReference = ficDataLayer.fundReference;
 		var fundDatasObj = {};
 
 		var axisCodeDataObj = {};
 		var fundCodeDataObj = {};
 		$.each( fundInfoBySeries, function( i, fund ) {
 			var catgryArray = [];
 			$.each( fund.catgries, function( i, category ) {
 				catgryArray.push([category.catgry, category.subCatgry]);
 			});
 			var subCat = (fund.catgries != null && fund.catgries.length > 0) ? fund.catgries[0].subCatgry : null;
 			axisCodeDataObj[fund.axisCd] = [null, null, null, null, fund.curr, subCat, catgryArray];
 			axisCodeDataObj[fund.axisCd][i_NAME] = fund.name;
 			axisCodeDataObj[fund.axisCd][i_KEYWORDS] = fund.oldNm.toUpperCase();
 			// For French searching with non-accented term, append name to keywords and unaccent
 			if (isFrench) {
 				axisCodeDataObj[fund.axisCd][i_KEYWORDS] = fc.unaccent(axisCodeDataObj[fund.axisCd][i_KEYWORDS] + " " + fund.name.toUpperCase());
 			}
 			fundCodeDataObj[fund.code] =  [fund.series, fund.chrgOptn, fund.axisCd, fund.url];
 		});
 		fundDatasObj["axis_codes"] = axisCodeDataObj;
 		fundDatasObj["fund_codes"] = fundCodeDataObj;
 
 		var seriesArr = [];
 		var loadSet = {};
 		$.each( fundReference.fundSeriesLoad, function( i, seriesLoad ) {
 			seriesArr.push(seriesLoad.series);
 			// Load may be given as comma-seperated list or single value
 			if (seriesLoad.load) {
 				var loadArray = seriesLoad.load.split(",");
 				$.each(loadArray, function(i, load) {
 					loadSet[load] = true;
 				}); 
 			}
 		});
 		fundDatasObj["series"] = seriesArr;
 		var loadArr = Object.keys(loadSet);
 		fundDatasObj["loads"] = loadArr;
 
 		var currencyArr = [];
 		$.each( fundReference.fundCurrencies, function( i, currency ) {
 			currencyArr.push(currency.curr);		
 		});
 		fundDatasObj["currencies"] = currencyArr;
 
 		var catsObj = {};
 		$.each( fundReference.fundCategories, function( i, cats ) {
 			catsObj[cats.catgry] = [null, null];
 			catsObj[cats.catgry][i_NAME] = cats.name
 		});
 		fundDatasObj["cats"] = catsObj;
 
 		var subcatsObj = {};
 		$.each( fundReference.fundSubCategories, function( i, subCats ) {
 			subcatsObj[subCats.catgry] = [null, null];
 			subcatsObj[subCats.catgry][i_NAME] = subCats.name;
 		});
 		fundDatasObj["subcats"] = subcatsObj;
 
 		var topFundsObj = [];
 		$.each( fundReference.topFunds, function( i, topFund ) {
 			topFundsObj.push(topFund.axisCd);
 		});
 		fundDatasObj["top_funds"] = topFundsObj;
 		
 		return fundDatasObj;
 	}
 
 	fc.unaccent = function(thisString) {
 		return thisString.replace(/[^A-Za-z\[\] ]/g, function(thisChar) { 
 			return diacriticsMap[thisChar] || thisChar;
 		});
 	}
 
 	fc.padZeros = function(str, width) {
 		if (!str || str.length >= width)
 			return str;
 		while (str.length < width) 
 			str = '0' + str;
 		return str;
 	}
 	
 	fc.currency_cad = ['CAD', 'CDN', 'CA'];
 	fc.currency_usd = ['USD', 'US'];
 	
 	// The global search has multiple different codes for the same 
 	// currency. We compare all the possibilities and return a match.
 	fc.compareCurrency = function(currency1, currency2) {
 		if (!currency1 && !currency2)
 			return false;
 		return ($.inArray(currency1, fc.currency_cad) !== -1 && $.inArray(currency2, fc.currency_cad) !== -1) 
 			|| ($.inArray(currency1, fc.currency_usd) !== -1 && $.inArray(currency2, fc.currency_usd) !== -1);
 	}
 
 	return fc.setup();
 }
