     function CalculateClosestBranch(address, updateField) {
         $("#loader").css("display", "");
 
         var geo = new google.maps.Geocoder;
         var addressLatLong = null;
 
         geo.geocode({ 'address': address },
             function (results, status) {
                 if (status == google.maps.GeocoderStatus.OK) {
                     addressLatLong = results[0].geometry.location;
 
                     $.ajax({
                         url: '/Branch/GetDeliveryCollectionBranchesByCountryID?countryID=1',
                         async: false,
                         success: function (json) {
                             if (json != null) {
                                 var closestBranch = null;
 
                                 for (var index = 0; index < json.length; index++) {
                                     var distance = google.maps.geometry.spherical.computeDistanceBetween(addressLatLong, new google.maps.LatLng(json[index].Latitude, json[index].Longitude));
 
                                     if ((closestBranch == null) || ((closestBranch != null) && (distance < closestBranch.DistanceFromAddress))) {
                                         closestBranch = json[index];
                                         closestBranch.DistanceFromAddress = distance;
                                     }
                                 }
 
                                 $(updateField).val(closestBranch.ID);
 
                                 if (($(updateField).attr("id").toLowerCase() == "deliverybranchid") && ($('form#formReservationWizard').find("#CollectVehicle").attr("checked")))
                                     CalculateClosestBranch($('form#formReservationWizard').find("#CollectionAddress_Address").val(), $('form#formReservationWizard').find("#CollectionBranchID"));
                                 else
                                     $('form#formReservationWizard').submit();
                             }
                         },
                         datatype: "json"
                     });
                 }
                 else {
                     $("#loader").css("display", "none");
                     $("#status-msg-text").html("Geolocation failed.");
                     $("#status-msg").fadeIn("fast");
                 }
             });
     }
