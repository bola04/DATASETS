          <script>
           (function() {
     var MarketingServices = window.MarketingServices || {},
         OneKOID = window.OneKOID || {};
 
     MarketingServices.LoginPopUp = (function() {
         'use strict';
 
         var $loginPopUpElement,
             initPopUp,
             popUpTimeout;
 
         function setLoginDeniedCookie(value) {
             MarketingServices.CookieManager.setItem('login-popup-denied', value);
         }
 
         function checkUserAuthentication() {
             return OneKOID.OAuthSSO.getAllCookies().then(function(cookies) {
                 return OneKOID.AuthControl.isYellowZone(cookies);
             });
         }
 
         function checkValidPopUp() {
             return checkUserAuthentication().then(function(isAuthenticated) {
                 return $loginPopUpElement.length &&
                     !isAuthenticated &&
                     !MarketingServices.CookieManager.getItem('login-popup-denied') &&
                     window.navigator.cookieEnabled;
             }, function() {
                 return false;
             });
         }
 
         function showPopUp() {
             var $signInButton;
 
             checkValidPopUp().then(function(isValidPopUp) {
                 // If login modal is visible, don't do anything
                 if (!$('#janrainModal:visible').length && isValidPopUp) {
                     $loginPopUpElement.modal('show');
                     $loginPopUpElement.on('hidden.bs.modal', function() {
                         setLoginDeniedCookie(true);
                         $loginPopUpElement.off('hidden.bs.modal');
                     });
 
                     $signInButton = $('#sign-in-button');
                     if ($signInButton) {
                         $signInButton.on('click', function() {
                             setLoginDeniedCookie(true);
                             $signInButton.off('click');
                         });
                     }
                 }
             });
         }
 
         function startPopUpTimer(time) {
             var interval = time * 1000;
             popUpTimeout = setTimeout(function() {
                 showPopUp();
             }, interval);
         }
 
         function detectLoginModalOpenEvent() {
             var $modal = $('.capture_modal_open');
             if ($modal.length) {
                 $modal.on('click', function() {
                     $(window).scrollTop(0);
                     $loginPopUpElement.modal('hide');
                     clearTimeout(popUpTimeout);
                 });
             }
         }
 
         function detectLoginSignInEvent() {
             MarketingServices.JanrainHelper.addSessionCreatedHandler(function() {
                 setLoginDeniedCookie(true);
             });
         }
 
         function detectLoginCloseModalEvent() {
             if (MarketingServices.JanrainHelper.getCloseModalHandlers().length === 0) {
                 MarketingServices.JanrainHelper.addCloseModalHandler(function() {
                     initPopUp();
                 });
             }
         }
 
         initPopUp = function() {
             $loginPopUpElement = $('#login-popup');
 
             checkValidPopUp().then(function(isValidPopUp) {
                 if (isValidPopUp) {
                     var timer = $loginPopUpElement.attr('data-timer');
                     startPopUpTimer(timer);
                     detectLoginModalOpenEvent();
                     detectLoginCloseModalEvent();
                     detectLoginSignInEvent();
                 }
             });
         };
 
         return {
             initPopUp: initPopUp
         };
     }());
 }());
          </script>
