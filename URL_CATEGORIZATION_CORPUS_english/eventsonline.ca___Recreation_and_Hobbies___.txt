    var ajaxUrl = 'https://events.com/wp-admin/admin-ajax.php';
 
     jQuery(document).ready( function() {
 
         jQuery("#talkToSpecialistForm").validate({
             rules: {
                 firstname: "required",
                 lastname: "required",
                 eventname: "required",
                 number_of_participants: "required",
                 email: {
                     required: true,
                     email: true
                 }
             },
             /*messages: {
                 firstname: "Please complete this mandatory field.",
                 lastname: "Please complete this mandatory field.",
                 email: "Please complete this mandatory field."
             },*/
             submitHandler: function(form) {
                 jQuery('.mktFormModalSubmitButton').prop('disabled', true);
                 var data = {
                     action: 'talk_to_specialist_submit',
                     firstname: jQuery('#firstname').val(),
                     lastname: jQuery('#lastname').val(),
                     email: jQuery('#email').val(),
                     eventname: jQuery('#eventname').val(),
                     participants: jQuery('#number_of_participants').val(),
                     nonce: jQuery('#nonce').val()
                 };
 
                 jQuery.ajax({
                     type : "post",
                     dataType : "json",
                     url : ajaxUrl,
                     data : data,
                     success: function(response) {
                         if(response.type == "success") {
                             showSuccess();
                         } else {
                             showError(response.message);
                         }
                     },
                     error: function() {
                         showError('Unexpected error. Please try again later.');
                     }
                 })
 
                 return false;
             }
         });
 
         $('#talkToSpecialistModal').on('hidden.bs.modal', function () {
             cleanTalkToSpecialistForm();
         });
     })
 
     function cleanTalkToSpecialistForm()
     {
         jQuery('#firstname').val('');
         jQuery('#lastname').val('');
         jQuery('#email').val('');
         jQuery('#eventname').val('');
         jQuery('#number_of_participants').val('');
         jQuery('.mktFormModalSubmitButton').prop('disabled', false);
         jQuery('.mktFormModalSuccessWrapper').hide();
         jQuery('.mktFormModalWrapper').show();
         jQuery('.error-panel').hide();
         jQuery('#talkToSpecialistForm').validate().resetForm();
     }
