 window.add_search_suggestions = function(element_id){
     $.get(
         "ajax_targets/solr.php",
         {query: "synonyms:spellcheck", term: ""},
         function(data){
             function add_custom_row(values,type){
                 var values_lengh = values.length;
                 if(values_lengh > 0){
                     var links = [];
                     $.each(values, function(index,value){
                         var search_href = "index/page/search?FullText=" + value;
                         var link = $("<a style='font-size: 16px;'></a>").addClass("term").attr("href",search_href).append(value);
                         links.push(link);
                         if(index < values_lengh - 1){
                             links.push(", ");
                         }
                     });
     
                     var row_text = "";
     
                     switch(type){
                         case "spellcheck":
                             row_text = "Did you mean? ";
                         break;
                         case "synonyms":
                             row_text = "Try searching: ";
                         break;
                     }
     
                     var type_text = $("<span></span>").addClass("text").append(row_text);
                     var type_terms = $("<span></span>").addClass("terms").append(links);
     
     
                     $("#" + element_id).append(
                         $("<div style='font-size: 16px; font-weight: bold;'></div><br><br>").addClass("custom_row").append(type_text,type_terms)
                     );
                 }
             }
     
             $("#" + element_id).empty();
             add_custom_row(data.spellcheck, "spellcheck");
             add_custom_row(data.synonyms, "synonyms");      
         },
         "json"
     );
 }

 function ProductReviewModule() {}
 ProductReviewModule.prototype.openWriteReview = function() {
     var previousFade = Shadowbox.options.animateFade;
     Shadowbox.options.animateFade = false;
     Shadowbox.close();
     
     Shadowbox.open({
         content:        '#form-content',
         height:         580,
         width:          1024,
         player:         'inline',
         options: {
             animateFade: previousFade
         }
     });
 }
 
 ProductReviewModule.prototype.submitReview = function(form) {
     var serialized_data = form.serializeArray();
     var any_missing = false;
     $.each(serialized_data, function(key, field) {
       if(field.value === "") {
         any_missing = true;
         alert(field.name + " is required");
         return;
       }
     })
     if(any_missing) {
         return;
     }
     else if($(".radio:checked").size() === 0 ) {
         alert('Rating value is required');
         return;
     }
     else {
         $.post('https://www.hockeyworld.com/ajax_targets/product_reviews.php', $(form).serialize())
             .done(function(res) {
                 res = JSON.parse(res);
                 if (res.success) {
                     alert('Review added successfully!');
                     Shadowbox.close();
                 }
                 else {
                     alert('There was an issue adding your review. Review your information and try again.');
                 }
             })
             .fail(function(res) {
                 alert("Server Error: " + res.status + " " + res.statusText);
             }
         );
     }
 }
 
 ProductReviewModule.prototype.updatePagination = function() {
     var row_template = "product_review_row";
     var pagination_template = "product_reviews_pagination_controls";
     var product_id = "";
     
     $.ajax({
         url: "ajax_targets/product_review_pagination.php",
     	action: "update_pagination",
     	beforeSend: function() {
             $("#pagination_loading").show();
         },
         data: $("#pagination_controls_form").serialize() + 
             "&product_id=" + product_id +
             "&row_template=" + row_template +
             "&pagination_template=" + pagination_template ,
         method: "post"
     })
     .done(function(response) {
     	$(".reviews_prds").empty();
     	$(".reviews_prds").append(response);
     })
     .always(function(event) {
         $("#pagination_loading").hide();
     })
 }
