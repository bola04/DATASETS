    // Custom HawkSearch AutoSuggest Display Functions:
     function showSamaritanContent(suggestDiv, results, title, autoSuggest) {
         //suggestDiv.empty();
         suggestDivNode = suggestDiv[0];
 
         var providers = [];
         var locations = [];
         var documents = [];
         var content = [];
 
         for (var i = 0; i < results.length; i++) {
             var result = results[i];
             var url = result.Url;
             if (!url)
                 continue;
 
             var type = result.CustomFields['type'];
             switch (type) {
                 case 'Providers':
                     result.Html = result.Html.replace('hawk-sqItemImage-thumb', 'hawk-sqItemImage-thumb circle-crop');
                     providers.push(result);
                     break;
                 case 'Locations':
                     locations.push(result);
                     break;
                 case 'Documents':
                     documents.push(result);
                     break;
                 case 'Videos':
                 case 'Testimonials':
                     //skip
                     break;
                 default:
                     content.push(result)
             }
         }
 
         if (providers.length) {
             autoSuggest.showSection(suggestDivNode, providers, "Providers", true);
         }
         if (locations.length) {
             autoSuggest.showSection(suggestDivNode, locations, "Locations", true);
         }
         if (documents.length) {
             autoSuggest.showSection(suggestDivNode, documents, "Documents", true);
         }
         if (content.length) {
             autoSuggest.showSection(suggestDivNode, content, 'Suggestions', true);
         }
     }

         //appending link and count to container
         footerContainer.appendChild(footerLink);
         // footerContainer.appendChild(footerCount);
 
         //appending container to suggestDiv
         suggestDiv.append(footerContainer);
 
     }
 
     (function (HawkSearch, HawkSearchLoader, undefined) {
 
         var baseUrl = '/hawksearch/proxy/samhealth';
         var hostUrl = '/hawksearch/host';
         var apiUrl = hostUrl + '/sites/samhealth';
         if ('https:' == document.location.protocol) {
             apiUrl = apiUrl.replace('http://', 'https://');

         if (!HawkSearch.TrackingUrl) HawkSearch.TrackingUrl = apiUrl;
         if (!HawkSearch.HawkUrl) HawkSearch.HawkUrl = apiUrl;

         HawkSearchLoader.loadjQuery = false;
         HawkSearchLoader.debugMode = false;
 
         HawkSearchLoader.IncludesFolder = '/Includes/HawkSearch/';

         HawkSearch.initAutoSuggest = function () {
             /*
             ----------------------------
             Auto Suggest Configurations:

 			3. Optional settings:

                 showProducts: customProductDisplayFunc,
                 showContent: customContentDisplayFunc,
