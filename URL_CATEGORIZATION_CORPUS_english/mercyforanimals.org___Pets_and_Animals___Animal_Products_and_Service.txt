    function setLightboxContent( content)
 {
 	if (content &&
 		content.indexOf &&
 		(content.lastIndexOf('.png') === content.length - 4 ||
 		content.lastIndexOf('.jpg') === content.length - 4))
 	{
 		content = '<img src="' + htmlspecialchars(content) + '" border=0 onclick="hideLightbox()" style="display:block">';
 	}
 
 	var element = document.getElementById("lightboxContent");
 	if (element)
 	{
 		if (content.nodeType)
 		{
 			element.innerHTML = "";			// clear all children
 			element.appendChild(content);
 		}
 		else
 			element.innerHTML = content;
 		return true;
 	}
 	return false;
 }
 function setLightboxTemplate( templateId, values)
 {
 	var template = buildFromTemplate(templateId, values);
 	if (!is_null(template))
 		return setLightboxContent(template);
 	return false;
 }
 function hideLightbox()
 {
 	document.body.style.overflow = "auto";
 	document.getElementsByTagName('html')[0].style.overflow = "auto";
 
 	var element = document.getElementById("lightboxContainer");
 	if (element)
 	{
 		addElementClass(element,'invisible');
 	}
 
 	// remove content in case it's a video (otherwise the sound keeps playing)
 	element = document.getElementById("lightboxContent");
 	if (element)
 		element.innerHTML = "";
 }
 function showLightbox( content /* optional */, disableClose /* optional */)
 {
 
 	document.body.style.overflow = "hidden";
 	document.getElementsByTagName('html')[0].style.overflow = "hidden";
 
 	if (content)
 	{
 		if (!setLightboxContent( content))
 			return;
 	}
 
 	var element = document.getElementById("lightboxContent");
 	if (element)
 	{
 		element.onclick = function (event)
 		{
 			if (event)
 			{
 				if (event.stopPropagation)
 					event.stopPropagation();
 				else
 					event.cancelBubble = true;
 			}
 		};
 	}
 
 	element = document.getElementById("lightboxContainer");
 	if (element)
 	{
 		element.onclick = function ()
 		{
 			if (!disableClose) hideLightbox();
 		};
 		removeElementClass(element,'invisible');
 	}
 
 	return;
 }
 function showLightboxTemplate( templateId, values, disableClose /* optional */)
 {
 	if (!setLightboxTemplate(templateId, values)) return false;
 	return showLightbox(null,disableClose);
 }

But behind the closed doors of modern farms they endure brutal cruelty. Most spend their entire lives in dark, crowded, waste-filled sheds. Some are locked in cages so small they can barely turn around. Because many animal cruelty laws do not protect them, farmed animals are often beaten, mutilated, and painfully slaughtered.
             </p>
             <p class="text-gray text-left">
              Together, we can expose and end this abuse. Join us in helping protect farmed animals by inspiring compassionate food choices and policies.
             </p>
