           Handlebars.registerHelper('eachByPosition', function(context, options){
         var output = '';
         var contextArray = [];
         //Convert json object to array
         for (var i in context) {
             if (context.hasOwnProperty(i)) {
                 contextArray.push(context[i]);
             }
         }
         //sort by position
         var contextSorted = contextArray.sort( function(a, b) { return a.position - b.position } );
         for(var i=0, j=contextSorted.length; i<j; i++) {
             output += options.fn(contextSorted[i]);
         }
         return output;
     });

     /**
      * Clear the input values in the subscribe modal

     function clearSubscribeInputs() {

     /**
      * Display a message to the user within the sign up modal
      *
      * @param {string} message The message to display
      * @param {string} type    The type of message. Likely success or fail
      *
      * @returns null
      */
     function displaySignUpModalMessage(message, type) {
         var messageDiv = document.querySelector(".signup-status");
         var messageClass = (type === 'success') ? "success" : "validation-advice";
         messageDiv.setAttribute("class", 'signup-status ' + messageClass);
         messageDiv.innerHTML = message;
     }
 
     /**
      * Clear any previous signup status messages
      */
     function resetStatusMessage() {
         var messageDiv = document.querySelector(".signup-status");
         messageDiv.setAttribute("class", 'signup-status');
         messageDiv.innerHTML = "";
     }
 
     /**
      * Toggle the visibility of the loading indicator on the Sign Up modal button.
      * The button text will be hidden while the loading graphic is showing
      *
      * @param {boolean} visible Indicate if the graphic should be showing or not
      */
     function toggleLoadingIndicator(visible) {
         var signUpButton = document.querySelector(".subscribe-container .actions button");
         if (visible) {
             signUpButton.querySelector("span").hide();
             signUpButton.querySelector("div").show();
         } else {
             signUpButton.querySelector("span").show();
             signUpButton.querySelector("div").hide();
         }
 
     }
 
     /**
      * Get the action of the form
      * This will return a URL with either HTTPS or HTTP, depending on the current page URL protocol
      *
      * @param {Element} form The form DOM node
      *
      * @returns {string}
      */
     function getFormAction(form) {
         var currentAction = form.getAttribute('action');
 
         //If the page URL protocol and the protocol for the form URL match, then don't change anything
         if (window.location.protocol === 'https:' && currentAction.search("https://") >= 0
             || window.location.protocol === 'https:' && currentAction.search("http://") >= 0) {
             return currentAction;
         }
 
         //Otherwise, update the URL to use the same protocol as the current page
         return currentAction.replace(/http.:/, window.location.protocol);
     }
