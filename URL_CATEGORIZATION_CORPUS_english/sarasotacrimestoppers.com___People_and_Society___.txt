     function toggle_password(target) {
         var d = document;
         var tag = d.getElementById('txt' + target);
         if ($('#txt' + target).length == 0) {
             tag = d.getElementById('con' + target);
         }
         var tag2 = d.getElementById('tgl' + target);
 
         if (tag2.value == 'Show') {
             tag.setAttribute('type', 'text');
             tag2.value = 'Hide';
         } else {
             tag.setAttribute('type', 'password');
             tag2.value = 'Show';
