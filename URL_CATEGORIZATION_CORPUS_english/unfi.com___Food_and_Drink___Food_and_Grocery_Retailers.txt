 	return {
 		block_info: [
 			{id:"block-one", map:"auto1map", tabId:"tabone"},
 			{id:"block-two", map:"auto2map", tabId:"tabtwo"},
 			{id:"block-three", map:"auto3map", tabId:"tabthree"},
 			{id:"block-four", map:"auto4map", tabId:"tabfour"},
 			{id:"block-five", map:"auto5map", tabId:"tabfive"},
 			{id:"block-six", map:"auto6map", tabId:"tabsix"}
 		],
 		init: function(){
 			jQuery("li.map-tab").click(function(event) {
 				// get reference to element that was clicked
 				$target = jQuery(event.target);
 				tabId = $target.attr("id");
 				
 				// remove class active and set display to none
 				jQuery("li.map-tab").removeClass("active");
 				
 				// add active and display to item clicked
 				$target.addClass("active");
 				
 				mapLen = UNFI.locationTabs.block_info.length;
 				tmp = null;
 				thisObj = {};
 				for(x = 0; x < mapLen; x++){
 					tmp = UNFI.locationTabs.block_info[x];
 					//console.log(tmp.id);
 					jQuery("#" + tmp.id).css("display", "none");
 					if(tabId == tmp.tabId) {
 						console.log('activate ' + tmp.id);
 						thisObj = tmp;
 					}
 				}
 				
 				jQuery("#" + thisObj.id).css("display", "block");
 				
 				mapObj = null;
 				mapObj = window.Drupal.gmap.getMap(thisObj.map);
 				currCenter = mapObj.map.getCenter();
 				window.google.maps.event.trigger(mapObj.map,'resize');  
 				mapObj.map.setCenter(currCenter);
 				
 			});
 		}
 	};   // end return
 }();
