         $(moduleId + ' .responsive-table *').removeAttr('style width height');

       $(window).load(function(e) {

           var tab  = $(moduleId).find('.tab-link[data-section="' + hash + '"]');
 
           if(tab.length == 1) {
 
             $('html, body').stop().animate({
 
               'scrollTop': tab.offset().top - $('#header').height()

           }
 
         }

         if(! mediaQueries.mm_md_up.matches) {
 
           $(moduleId).hasClass('expanded') ?
             $(moduleId).removeClass('expanded'):
             $(moduleId).addClass('expanded');
         }
 
         $(moduleId).find('.tab-link')
           .removeClass('tab-link-active')
           .addClass('tab-link-inactive');
 
         $(moduleId).find('.tab-content')
           .removeClass('tab-content-active')
           .addClass('tab-content-inactive');
 
         $(moduleId).find('.tab-link:nth-child(' + nthChild +')')
           .removeClass('tab-link-inactive')
           .addClass('tab-link-active');
 
         $(moduleId).find('.tab-content:nth-child(' + nthChild +')')
           .removeClass('tab-content-inactive')
           .addClass('tab-content-active');
