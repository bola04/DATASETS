 	function getCandidate(electoratesData, s){
 		if(electoratesData.length == 1){
 			var electorate = electoratesData[0].name.toLowerCase();
 
 			var matches = [];
 			for (var i = 0; i < candidatesData.length; i++) {
 				if(candidatesData[i].electorate.toLowerCase() == electorate){
 					matches.push(candidatesData[i]);
 				}
 			}
 
 			if(matches.length == 1){
 				console.log(matches);
 				console.log("https://www.alp.org.au/"+matches[0].nbslug);
 				window.location.href = "https://www.alp.org.au/"+matches[0].nbslug;
 			} else if(matches.length > 1){
 				buildResultsList(matches, electorate);
 			} else {
 				throwError("Sorry, we couldn't find any candidates for your postcode.");
 			}
 		} else if(electoratesData.length > 1) {
 			var matches = [];
 			for (var i = 0; i < electoratesData.length; i++) {
 				var electorate = electoratesData[i].name.toLowerCase();
 				for (var j = 0; j < candidatesData.length; j++) {
 					if(candidatesData[j].electorate.toLowerCase() == electorate){
 						matches.push(candidatesData[j]);
 					}
 				}
 			}
 			if(matches.length == 1){
 				window.location.href = "https://www.alp.org.au/"+matches[0].nbslug;
 			} else if(matches.length > 1){
 				buildResultsList(matches, s);
 			} else {
 				throwError("Sorry, we couldn't find any candidates for your postcode.");
 			}
 
 		} else {
 			throwError("Sorry, we couldn't find an electorate to match that postcode.");
 		}
 	}
