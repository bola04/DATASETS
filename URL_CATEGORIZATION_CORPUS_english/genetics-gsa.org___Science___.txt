         function checkOutOfOrderRespond(page){

     function load(){
         var $articleList = $('#article-list');
         var page =  $articleList.data('nextPage');
         var url = 'http://genestogenomes.org/iframeforgsahomepage/';
         if (page == 1){
             //do nothing, other wise wordpress will send redirect to http://genestogenomes.org/iframeforgsahomepage/ without cors header

             //something, the trailing / is for removeing the redirect
             url += 'page/'+page+'/';

             if(checkOutOfOrderRespond(page)){
                 return;
             }
             $articleList.append(data);
             $articleList.data('nextPage', page + 1);
         }).error(function(){
