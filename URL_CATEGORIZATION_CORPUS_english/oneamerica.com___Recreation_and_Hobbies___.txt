 function trackToutLinks(){
 $('.tout-block a').each(function() {
   var oldURI = $(this).attr('href');
   if (oldURI.indexOf('http') != 0)
   {
 	  var addURI = 'toutURI=' + encodeURIComponent(window.location.pathname + '#' + this.title) + '&toutTitle=' + encodeURIComponent($(this).text());

 	  var addSep = '&';
 	  if (oldURI.indexOf('?') == -1)
 	  {
 		addSep = '?';
 	  }
 	  $(this).attr('href', oldURI + addSep + addURI);
   }
 });
