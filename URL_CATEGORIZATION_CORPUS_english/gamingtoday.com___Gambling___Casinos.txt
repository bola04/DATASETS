 		var adminDefaultLeague = 5;
 		var defaultTimezone = sessionStorage.getItem('defaultTimezone');
 
 		if (leagueVal == 5) {
 			$('.mlb').removeClass('hidden');
 		} else {
 			$('.others').removeClass('hidden');
 		}
 
 		// Show first column of feed data
 		var showFirstColumn = function() {
 			$('.wagering-table th.casino').first().show();
 			$('.wagering-table tbody tr').each(function(index) {
 				$(this).find('.casino').first().show();
 			});
 		}
 
 		showFirstColumn();
 
 		$('.datepicker').datetimepicker({
 			format: 'YYYY-MM-DD',
 			defaultDate: new Date(dateSelected),
 			useCurrent: false,
 			minDate: moment().subtract(1, 'years').format('YYYY-MM-DD'),
 			widgetPositioning: {
 				'vertical': 'bottom'
 			}
 		});
 
 		$('.datepicker').data('DateTimePicker').date(dateSelected);
 
 		// If a league session default is set, show that on page load. Otherwise
 		// show the admin league default.
 		var updateDefault = true;
 		if (!defaultLeague) {
 			defaultLeague = adminDefaultLeague;
 			updateDefault = false;
 		}
 		leagueVal = defaultLeague;
 		$('.leagueselect').val(leagueVal);
 		updateDefaultGameType(leagueVal);
 		updateWageringBoard(dateSelected, autoRefresh, showFirstColumn, updateDefault);
 
 		// If a timezone session default is set, show that on page load
 		if (defaultTimezone) {
 			$('.timezone').val(defaultTimezone);
 			updateWageringBoard(dateSelected, autoRefresh, showFirstColumn, false, true);
 		}
 
 		$('.datepicker').on('dp.change', function(e){
 			dateSelected = moment(e.date).format('YYYY-MM-DD');
 			var oldDate = moment(e.oldDate).format('YYYY-MM-DD');
 			if (e.oldDate && (dateSelected != oldDate)) {
 				// toggle autoRefresh to use the new date in it's current state
 				onOffRefresh(dateSelected, showFirstColumn);
 				onOffRefresh(dateSelected, showFirstColumn);
 
 				updateWageringBoard(dateSelected, autoRefresh, showFirstColumn);
 			}
 		});
 
 		// Show previous column of feed data and hide current column on arrow back
 		$(document).on('click', '.glyphicon-chevron-left', function(e){
 			var current = $('th.casino:visible');
 			var prev = current.prev('.casino');
 			if (prev.length == 0) {
 				$('th.casino').last().show();
 			} else {
 				prev.show();
 			}
 
 			current.hide();
 
 			$('tbody tr').each(function() {
 				var current = $(this).find('td.casino:visible');
 				var prev = current.prev('.casino');
 				if (prev.length == 0) {
 					$(this).find('td.casino').last().show();
 				} else {
 					prev.show();
 				}
 
 				current.hide();
 			});
 
 			e.stopImmediatePropagation();
 		});
 
 		// Show next column of feed data and hide current column on arrow forward
 		$(document).on('click', '.glyphicon-chevron-right', function(e){
 			var current = $('th.casino:visible');
 			var next = current.next('.casino');
 			if (next.length == 0) {
 				$('th.casino').first().show();
 			} else {
 				next.show();
 			}
 
 			current.hide();
 
 			$('tbody tr').each(function() {
 				var current = $(this).find('td.casino:visible');
 				var next = current.next('.casino');
 				if (next.length == 0) {
 					$(this).find('td.casino').first().show();
 				} else {
 					next.show();
 				}
 
 				current.hide();
 			});
 
 			e.stopImmediatePropagation();
 		});
 
 		// Set both gametype filters to the same value.
 		$('.gametype').on('change', function() {
 			var val = $(this).val();
 			$('.gametype').val(val);
 		});
 
 		// Process the League, Game Type and Time Zone selects and update the wagering board based on values.
 		$('.select-filter').change(function() {
 			var id = $(this).attr('id');
 			var updateLeague = id === 'OddLeague' ? true : false;
 			var updateTimezone = id === 'OddTimeZone' ? true : false;
 			updateWageringBoard(dateSelected, autoRefresh, showFirstColumn, updateLeague, updateTimezone);
 		});
 
 		// Previous and Next links for date.
 		$('.previous-link, .next-link').on('click', function(e){
 			e.preventDefault();
 			dateSelected = $(this).data('date');
 			autoRefresh = $(this).data('autorefresh');
 			updateWageringBoard(dateSelected, autoRefresh, showFirstColumn);
 		});
 
 		// Auto refresh
 		$('.auto-refresh .on-off-btn').click(function(){
 			onOffRefresh(dateSelected, showFirstColumn);
 		});
 
 		if (autoRefresh == 1) {
 			onOffRefresh(dateSelected, showFirstColumn);
 		}
 	});
 
 	var timerRefresh = 0;
 
 	function onOffRefresh(dateSelected, showFirstColumn) {
 		if ($('.on-off-btn').html()=='Off') {
 			$('.on-off-btn').html('On').css("background","green");
 			timerRefresh = setInterval(function(){
 				updateWageringBoard(dateSelected, 1, showFirstColumn);
 			},1000*90);
 		} else {
 			$('.on-off-btn').html('Off').css("background","#C60000");;
 			clearInterval(timerRefresh);
 			updateWageringBoard(dateSelected, 0, showFirstColumn);
 		}
 	}
 
 	function updateDefaultGameType(leagueVal) {
 		var defaultGameType = 'spread_totals';
 		// for MLB (5) and NHL (7) show "Money Lines" by default
 		if (leagueVal == 5 || leagueVal == 7) {
 			defaultGameType = 'money_lines';
 		}
 		$('.gametype').val(defaultGameType);
 		return defaultGameType;
 	}
 
 	function updateWageringBoard(dateSelected, refresh, showFirstColumn, updateLeague, updateTimezone) {
 		var leagueVal= $('.leagueselect option:selected').val();
 		var typeVal;
 		var expand = false;
 		var timezoneVal = $('.timezone option:selected').val();
 
 		if (leagueVal == 5) {
 			$('.mlb').removeClass('hidden');
 			$('.others').addClass('hidden');
 			typeVal = $('.mlb option:selected').val();
 		} else {
 			$('.mlb').addClass('hidden');
 			$('.others').removeClass('hidden');
 			typeVal = $('.others option:selected').val();
 		}
 
 		if (updateLeague === true) {
 			typeVal = updateDefaultGameType(leagueVal);
 
 			// set the league selection as session default
 			sessionStorage.setItem('defaultLeague', leagueVal);
 		}
 
 		// set the timezone selection as session default
 		if (updateTimezone === true) {
 			sessionStorage.setItem('defaultTimezone', timezoneVal);
 		}
 
 		$('.datepicker').data('DateTimePicker').date(dateSelected);
 
 		if ($('.expand-board').length > 0) {
 			var paramsObj = {
 				'league' : leagueVal,
 				'type' : typeVal,
 				'timezone' : timezoneVal,
 				'date' : dateSelected,
 				'autorefresh' : refresh
 			};
 
 			var params = $.param(paramsObj);
 			var href = $('.expand-board').attr('href');
 			var index = href.indexOf('?')
 			if (index >= 0) {
 				href = href.substr(0, index)
 			}
 
 			href += '?' + params;
 			$('.expand-board').attr('href', href);
 		}
 
 		$.ajax({
 			url: '/odd/index',
 			type: 'get',
 			data: {
 				'league': leagueVal,
 				'type': typeVal,
 				'date': dateSelected,
 				'timezone': timezoneVal,
 				'autorefresh': refresh,
 				'expand': expand
 			},
 			dataType: 'html',
 			success: function(html) {
 				$('.wagering-data').empty().html(html);
 				if (!expand) {
 					$('.wagering-score').remove();
 				}
 				showFirstColumn();
 				$('.date-select span:not(.glyphicon)').empty().text(moment(dateSelected).format('ddd, MMM D, YYYY'));
 				$('.previous-link').attr('data-autorefresh', refresh);
 				$('.previous-link').data('autorefresh', refresh);
 				$('.previous-link').attr('data-date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
 				$('.previous-link').data('date', moment(dateSelected).subtract(1, 'd').format('YYYY-MM-DD'));
 				$('.next-link').attr('data-autorefresh', refresh);
 				$('.next-link').data('autorefresh', refresh);
 				$('.next-link').attr('data-date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
 				$('.next-link').data('date', moment(dateSelected).add(1, 'd').format('YYYY-MM-DD'));
 			}
 		});
 	}
