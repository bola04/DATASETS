 			function loadKeywords(myField, KWStartfrom, KWLimit, KWList, count){
 
 				// do the dots
 				var dots = '';
 				for (var i = 0; i < count % 4; i++ ){
 					dots+='.';
 				}
 				$('#dots' + myField).html(dots);
 
 				$.ajax({
 					data		: {

 						'field'		: myField,
 						'format'		: 'json',
 						'limit'		: KWLimit,
 						'location_id'	: '1',
 						'startfrom'	: KWStartfrom
 					},
 					dataType	: 'json',
 					error	: function (jqXHR, textStatus, errorThrown){
 						//alert('ERROR:\n' + textStatus + '\n' + errorThrown);
 					},
 					success	: function(data, textStatus, jqXHR){
 						//	data['hits']		the number of hits, so the next KWStartfrom should be one greater
 						//	data['returns']	the array of keywords
 
 						//	combine these results with any previous
 						for ( var KWIndex in data['returns'] ){
 							KWList[ data['returns'][KWIndex] ] = '<li><a href="/?location_id=' + '1&amp;adlibmultifilter[fields][' + myField + ']=1&amp;adlibmultifilter[term]=' + encodeURIComponent(data['returns'][KWIndex]) + '">' + data['returns'][KWIndex] + '</a></li>';
 						}
 
 						//	alphabetise
 						//KWList.sort();
 						KWKeys = [];
 						for( var KWIndex2 in KWList ){
 							KWKeys.push(KWIndex2);
 							//alert( KWIndex2 );
 						}
 						//KWKeys.sort();
 						KWKeys.sort(function(a, b) {
 							if (typeof a === 'string' && typeof b === 'string') {
 								return a.toLowerCase().localeCompare(b.toLowerCase());
 							}
 						});
 
 
 
 						//var KWString = KWList.join('');
 						var KWString = '';
 						for( var KWIndex3 in KWKeys ){
 							//alert( KWIndex3 );
 							KWString += KWList[KWKeys[KWIndex3]];
 						}
 
 						//	write the array into the list
 						$('ul#kf_' + myField).html( KWString );
 
 						//	run again? if there are any hits
 						if ( parseInt(data['hits']) > 0 ){
 						//	If so,
 							//	calculate the new start point
 							//loadKeywords(myField, parseInt(KWStartfrom) + parseInt(data['hits']), KWLimit, KWList, parseInt(count)+1);
 							loadKeywords(myField, parseInt(KWStartfrom) + KWLimit, KWLimit, KWList, parseInt(count)+1);
 							//	run the function
 						} else {
 						//	otherwise
 							//	get rid of the loading message
 							$('#loadingmessage' + myField).remove();
 						}
 
 					},
 					type		: 'post',
 					url		: '/formhandlers/adlibscan.php'
 				});
