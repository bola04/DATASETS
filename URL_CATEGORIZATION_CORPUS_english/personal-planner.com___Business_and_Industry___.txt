 	function redirectCountryCheck(countryIso){
 		countryContinent = CountrySelect.getCountryByIso(countryIso).continent;
 		
 		if( "SE" == countryIso ) redirectCountry("personligalmanacka.se","personligalmanacka.se","SE","");
 		else if( "NO" == countryIso ) redirectCountry("personligalmanakk.com",   "personligalmanakk.com","NO","");
 		else if( "DE" == countryIso ) redirectCountry("persoenlicherkalender.de","persoenlicherkalender.de","DE","");
 		else if( "AT" == countryIso ) redirectCountry("personalplanner.com/EU", "personalplanner.com","EU",countryIso);
 		else if( "FI" == countryIso ) redirectCountry("personalplanner.com/FI", "personalplanner.com","FI","");
 		else if( "GB" == countryIso ) redirectCountry("personalplanner.com/EN", "personalplanner.com","EN","");
 		else if( "US" == countryIso ) redirectCountry("personalplanner.com/US", "personalplanner.com","US","");
 		else if( "AU" == countryIso ) redirectCountry("personalplanner.com/AU", "personalplanner.com","AU","");
 		else if( "DK" == countryIso ) redirectCountry("personalplanner.com/DK", "personalplanner.com","DK","");
 		else if( "AX" == countryIso ) redirectCountry("personalplanner.com/US", "personalplanner.com","US",countryIso);
 		else if( "Europe" == countryContinent ) redirectCountry("personalplanner.com/EU","personalplanner.com","EU",countryIso);
 		else if( "Europe" != countryContinent ) redirectCountry("personalplanner.com/US","personalplanner.com","US",countryIso);
 		
 	}
 	function redirectCountry(topDomain,newDomain,newISO,selectedISO){
 		var currentUrl = window.location.href;
 		if("/".length == 5){
 			//dev
 			if(selectedISO.length > 0){			
 				window.location.href = "https://"+ window.location.hostname + "/" + newISO + "/?iso=" + selectedISO;
 			}else{
 				window.location.href = "https://"+ window.location.hostname + "/" + newISO + "/";
 			}
 		}else{
 			//live
 			if( currentUrl.indexOf(topDomain) == -1 ){
 				if(selectedISO.length > 0){			
 					window.location.href = "https://www." + newDomain + "/" + newISO + "/?iso=" + selectedISO;
 				}else{
 					window.location.href = "https://www." + newDomain + "/" + newISO + "/";
 				}
 			}
 		}
 	}

        function updateCartList()
 		{
 			getCartList();
 			$('#dropdownMenuCart').delay(200).css({opacity:0}).animate({opacity:1},200).delay(100).animate({opacity:0},100).delay(30).animate({opacity:1},200);
 		}
 		function getCartList()
 		{
 			//get user cart list
 			$.ajax({
 				type: 'POST',
 				url: '/US/common/php/getUserCartList.php',
 				dataType: 'json',
 				success: cartListResponse,
 				error: function (xhr,textStatus,errorThrown) { alert(textStatus+':'+errorThrown); }
 			});
 		}
 		function cartListResponse( data )
 		{
 			var html = "";
 			if( data.length == 0 ){
 				//empty cart
 				html += '<br>';
 				html += '<span class="startnav__dropdown-item_nolink startnav__dropdown-item_nolink--center">Your cart is empty</span>';
 				html += '<div class="startnav__horisontalLine"></div>';
 				html += '<a href="/US/planner_editor/" class="startnav__dropdown-btn_starthere">Start here</a>';
 			}else{
 				html += '<span class="startnav__dropdown-item_nolink startnav__dropdown-item_incart">In your cart</span>';
 				html += '<div class="startnav__horisontalLine"></div>';
 				//list cart items
 				var totalItems = 0;
 				for(var c=0; c<data.length; c++){
 					html += '<a class="dropdown-item startnav__dropdown-item__link" href="/US/cart/">'+ data[c]['Total'] +'pc(s) <b>'+ data[c]['Name'] +'</b></a>';
 					totalItems += parseInt(data[c]['Total']);
 				}
 				html += '<div class="startnav__horisontalLine"></div>';
 				html += '<a href="/US/cart/" class="startnav__dropdown-btn_starthere">Go to checkout </a>';
 			}
 			html += '<br>';
 			$('#dropdown-menu__cart').html(html);
 			$('#startnav__cart__list__totalItems').html(totalItems);
 		}
 		function cartListReset()
 		{
 			var html = "";
 			//empty cart
 				html += '<br>';
 				html += '<span class="startnav__dropdown-item_nolink startnav__dropdown-item_nolink--center">Not signed in</span>';
 				html += '<div class="startnav__horisontalLine"></div>';
 				html += '<a href="#" class="startnav__dropdown-btn_starthere" onclick="openLoginWindow(false);">Sign in</a>';
 				var totalItems = 0;
 				$('#dropdown-menu__cart').html(html);
 				$('#startnav__cart__list__totalItems').html(totalItems);
 		}
 		function loginSuccessAppend()

 			$('#startnav__inloggad').removeClass('pointer');

 			$('#startnav__user__inloggad').html('<div class="startnav__loggin__useremail">Signed in: '+ userAccount.email +'</div>');
 			$('#startnav__user__myaccount').removeClass("startnav__textmuted");
 			$('#startnav__user__orders').removeClass("startnav__textmuted");
 			//document.getElementById('startnav__inloggad').onclick = "";
 			
 			if($('#login_span_user'))
 				$('#login_span_user').addClass('hidden');
 			if($('#logout_span_user'))
 				$('#logout_span_user').removeClass('hidden');
 			
 			//get cart list
 			getCartList();

 		}
