                 $(".articles-link").removeClass("selected");
                 $(this).addClass("selected");
                 $("#lp-articles-div").html("");
                 $("#lp-articles-div").hide();
                 $("#articles-loading").show();
                 $.ajax({
                     type: "GET",
                     url: '/Home/GetArticles',
                     dataType: "text",
                     data: { Type: t },
                     cache: false,
                     success: function (html) {
                         $("#articles-loading").hide();
                         $("#lp-articles-div").html(html);
                         $("#lp-articles-div").show();
                     },
                     error: function() {
                         $("#articles-loading").hide();
                         $("#lp-articles-div").html("<p>Could not load articles</p>");
                     }
                 });
                 return false;
             });

                 });
