 	function validateSignInForm() {
 		// added this line of  code for lazy loading
 	   $('body').prepend('<div class="ajaxLoading-bg"></div><div class="ajaxLoading-img"></div>');
 		var favouriteURL = $('#favourite_link').val();
 		var vURL = "/login/headerlogin";
 		$
 				.ajax({
 					type : 'POST',
 					url : vURL,
 					data : $("#signInForm").serialize(),
 					success : function(result){
 
 						//First, remove the spinner
 						$(".ajaxLoading-bg, .ajaxLoading-img").remove();
 
 						if (result[0].name == "Success") {
 							if(favouriteURL != '') {
 								$.post(favouriteURL, function(result) {
 									window.location.href = window.location.href;
 								});
 							} else {
 								window.location.href = window.location.href.replace(window.location.hash, "");
 							}
 						} else if(result[0].name == "SuccesswithCartMerge") {
 							showCartMerge(window.location.href);
 						} else if(result[0].name == "/cart/checkout"){
 							window.location.href = result[0].name;
 						} else if(result[0].name == "/cart/checkoutwithCartMerge"){
 							showCartMerge("/cart/checkout");
 						} else if(result[0].name =="/checkout/single/quotesummary") {
 							window.location.href = result[0].name;
 						} else if(result[0].name == "/checkout/single/quotesummarywithCartMerge"){
 							showCartMerge("/checkout/single/quotesummary");
 						} else if(result[0].name == "redirect:/welcome") {
 								window.location.href = "/welcome";
 						} else if (result[0].name == "maxpasswordCount") {
 							$('#j_username').next('span').remove();
 							$('#j_username').after('<span id="span" class="help-block form-error">Having trouble logging in? Use the forgot your password link below to recover your login details.</span>');
 						} else if(result[0].name == "email not found") {
 							$('#j_username').next('span').remove();
 							$('#j_username').after('<span id="span" class="help-block form-error">Entered email address or password is wrong</span>');
 
 						} else if(result[0].name == "Bad Password") {
 								$('#j_username').next('span').remove();
 								$('#j_username').after('<span id="span" class="help-block form-error">Entered email address or password is wrong</span>');

 							$('#check_availability_modal').html(result[0].value)

 						} else {
 							$('#j_username').next('span').remove();
