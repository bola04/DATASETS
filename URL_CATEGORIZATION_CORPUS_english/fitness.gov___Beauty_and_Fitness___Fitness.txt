 	        var win = $(window);
 
 	        var viewport = {
 	            top : win.scrollTop(),
 	            left : win.scrollLeft()
 	        };
 	        viewport.right = viewport.left + win.width();
 	        viewport.bottom = viewport.top + win.height();
 
 	        var height = this.outerHeight();
 	        var width = this.outerWidth();
 
 	        if(!width || !height){
 	            return false;
 	        }
 
 	        var bounds = this.offset();
 	        bounds.right = bounds.left + width;
 	        bounds.bottom = bounds.top + height;
 
 	        var visible = (!(viewport.right < bounds.left || viewport.left > bounds.right || viewport.bottom < bounds.top || viewport.top > bounds.bottom));
 
 	        if(!visible){
 	            return false;
 	        }
 
 	        var deltas = {
 	            top : Math.min( 1, ( bounds.bottom - viewport.top ) / height),
 	            bottom : Math.min(1, ( viewport.bottom - bounds.top ) / height),
 	            left : Math.min(1, ( bounds.right - viewport.left ) / width),
 	            right : Math.min(1, ( viewport.right - bounds.left ) / width)
 	        };
 
 	        return (deltas.left * deltas.right) >= x && (deltas.top * deltas.bottom) >= y;
 
 	    };
 
 		// Checks if at least 10% of the footer is visible
 		var isFooterVisible = $('#hhs-global-footer').isOnScreen(0.1,0.1);
 
 	    // Variable for base path issues with subsites.  Value dependent on ThemeKey module.
 		var $iia_subsite_name = '';
 
 	    if (Drupal.settings.basePath === '/') {
 	        $iia_subsite_name = '';
 	    } else if (typeof Drupal.settings.basePath === 'undefined' || Drupal.settings.basePath === '') {
 		    $iia_subsite_name = '';
 	    } else {
 	        $iia_subsite_name = Drupal.settings.basePath;
 	    }
 
 	    //	variable used to acccount for the fact that $(window).scroll is called on load
 	    var scrolled = false;
 	    var prevScrollTop = 0;
 	    var topContainerHeight = parseInt($('.top-container').css("height"));
 
 		$(window).scroll(function() {
 
 			if (scrolled)
 			{
 				var isHomePage = $('.html').hasClass('front');
 				var isMobile = $(window).width() < 768;
 
 				if (isMobile)
 				{
 					//	clear input text from search box on mobile
 					$('input.global-search').get(0).value = "";
 
 					//	remove the suggestion box
 					if ($("#auto-suggest-popup").length > 0)
 						$("#auto-suggest-popup").remove();
 				}
 
 				if(!isFooterVisible || isMobile)
 				{
 					if ($(this).scrollTop() > prevScrollTop)	//	if scrolling down
 					{
 						$('.top-container').removeClass('navbar-fixed-top');
 						$('.top-container').removeClass('search-mode');
 						$('.hhs-global-breadcrumbs').removeClass('navbar-fixed-top-shown');
 
 					}
 					else if ($(this).scrollTop() < prevScrollTop)	//	if scrolling up
 					{
 						if ($(this).scrollTop() <= 0)
 						{
 							$('.top-container').removeClass('navbar-fixed-top');
 							$('.top-container').removeClass('search-mode');
 							$('.hhs-global-breadcrumbs').removeClass('navbar-fixed-top-shown');
 
 							if (!isFooterVisible && !isMobile)
 							{
 								hideStickyNavSearch();
 							}
 						}
 						else
 						{
 							$('.top-container').addClass('navbar-fixed-top');
 							$('.hhs-global-breadcrumbs').addClass('navbar-fixed-top-shown');
 
 							if (!isFooterVisible && !isMobile && $('.hhs-blue-header .inner .sticky-nav-search').length == 0)
 							{
 								showStickyNavSearch();
 							}
 						}
 					}
 				}
 
 				//	Back to Top
 				if ($(window).scrollTop() > 0)
 				{
 					$('.back-to-top').fadeIn('slow');
 				}
 				else
 				{
 					$('.back-to-top').fadeOut('slow');
 				}
 
 				$('.back-to-top').css("left", getLeftForBackToTop() + "px");
 			}
 
 			scrolled = true;
 			prevScrollTop = $(this).scrollTop();
 		})
 
 
 		function showStickyNavSearch()
 		{
 			$('.hhs-blue-header .inner').append('<span class="sticky-nav-search hidden" aria-hidden="false"></span>');
 			$('.sticky-nav-search').append('<a href="#" role="button"></a>');
 			addSearchIcon();
 			$(".sticky-nav-search").removeClass("hidden");
 
 			$(".sticky-nav-search").click(function(){
 
 				if ($(".navbar-fixed-top").hasClass("search-mode"))
 				{
 					//	sticky nav search close icon clicked
 					$(".top-container").removeClass("search-mode");
 					$('.hhs-global-search').attr('aria-hidden', true);
 					$(".sticky-nav-search a #search-close-icon").remove();
 					addSearchIcon();
 					setTabIndexForSearchElements(-1);
 				}
 				else
 				{
 					//	sticky nav search icon clicked
 					$(".top-container").addClass("search-mode");
 					$('.hhs-global-search').attr('aria-hidden', false);
 					//$(".navbar-fixed-top .hhs-global-search .global-search").focus();
 					$(".sticky-nav-search a #search-icon").remove();
 					addSearchCloseIcon();
 					setTabIndexForSearchElements(0);
 				}
 
 				return false;
 			});
 
 			$('.hhs-global-search').attr('aria-hidden', true);
 			setTabIndexForSearchElements(-1);
 		}
 
 		function hideStickyNavSearch()
 		{
 			$('.sticky-nav-search').addClass('hidden');
 			$('.hhs-blue-header .inner .sticky-nav-search').remove();
 			$('.hhs-global-search').attr('aria-hidden', false);
 			setTabIndexForSearchElements(0);
 		}
 
 		function setTabIndexForSearchElements(tabIndex)
 		{
 			$('.hhs-global-search .search-form .global-search').attr('tabindex', tabIndex);
 			$('.hhs-global-search .search-form .global-search-btn').attr('tabindex', tabIndex);
 			$('.hhs-global-search .a-z a').attr('tabindex', tabIndex);
 		}
 
 		function addSearchIcon() {
 			if ($iia_subsite_name !== "") {
 		        $(".sticky-nav-search a").append('<img id="search-icon" src="' + $iia_subsite_name + 'sites/all/themes/project_h/css/images/persistent-header-search-icon.png" alt="search" />');
 		    } else {
 		        $(".sticky-nav-search a").append('<img id="search-icon" src="/sites/all/themes/project_h/css/images/persistent-header-search-icon.png" alt="search" />');
 		    }
 		}
 
 		function addSearchCloseIcon() {	
 			if ($iia_subsite_name !== "") {
 		        $(".sticky-nav-search a").append('<img id="search-close-icon" src="' + $iia_subsite_name + 'sites/all/themes/project_h/css/images/persistent-header-close-button.png" alt="close" />');
 		    } else {
 		        $(".sticky-nav-search a").append('<img id="search-close-icon" src="/sites/all/themes/project_h/css/images/persistent-header-close-button.png" alt="close" />');
 		    }
 		}
 
 		function getLeftForBackToTop()
 		{
 			if ($(window).width() < 751)
 			{
 				return $(window).width()/2 - $('.back-to-top').width()/2;
 			}
 			else if ($(window).width() < 1150)
 			{
 				return $(window).width() - $('.back-to-top').width() - 20;
 			}
 			else
 			{
 				return ($(window).width() - $('.main-section').width())/2 + $('.main-section').width() + 20;
 			}
 		}
 
 
 		//	Back to Top
 		$(window).resize(function() {
 			$('.back-to-top').css("left", getLeftForBackToTop() + "px");
 
 			// Checks if at least 10% of the footer is visible
 			// when user resizes the window
 			isFooterVisible = $('#hhs-global-footer').isOnScreen(0.1,0.1);
 		})
 
 		$('.back-to-top').click(function() {
 			$('html, body').animate({scrollTop: 0}, 700);
 			$(this).blur();
 			return false;
 		});
 
 		$('.back-to-top').keypress(function(e) {  // Attach the form handler to the keypress event
 			if (e.keyCode == 13 || e.keyCode == 32) { // If the the enter key or space was pressed
 				$(this).click();
 				$(this).blur();
 				return e.preventDefault(); // Prevent the form submit
 			}
 		});
 
 		$('.back-to-top').focus(function() {
 			$(this).addClass('focused');
 		});
 
 		$('.back-to-top').blur(function() {
 			$(this).removeClass('focused');
 		});
 	})(jQuery);
