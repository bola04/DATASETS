       featured_showing = 0;
 	featured_width = 639;
 	featured_items = document.getElementsByClassName("featured_box_item");
 	
 	function show_feature(i) {
 		// this is only needed for a show/hide setup:
 		//featured_items[featured_showing].className += " hide"; // use this if no other classes are present: = "hide";
 		//featured_items[i].className = featured_items[i].className.replace(/ hide/g, ""); // use this if no other classes are present: = ""
 		var amount = (i - featured_showing) * featured_width;
 		var direction = (amount > 0) ? "right" : "left" ;
 		scroll_inner_content("featured_box_slider", "featured_box_slider_inner", direction, Math.abs(amount), true);
 		featured_showing = i;
 		
 		// copy the link href in the featured text to the link href around the border image, since we can't click through the border image
 		// this saves us having to slice the border image into four pieces
 		var links = featured_items[i].getElementsByTagName("a");
 		var href = links[0].href;
 		document.getElementById("featured_box_link").href = href;
 	}
 	
 	function show_previous_feature(stop) {
 		previous_feature = featured_showing - 1;
 		if (previous_feature < 0) { previous_feature = featured_items.length - 1; }
 		show_feature(previous_feature);
 		if (stop) {
 			clearInterval(rotate_timer);
 		}
 	}
 	
 	function show_next_feature(stop) {
 		next_feature = featured_showing + 1;
 		if (next_feature > (featured_items.length - 1)) { next_feature = 0; }
 		show_feature(next_feature);
 		if (stop) {
 			clearInterval(rotate_timer);
 		}
 	}
 	
 	show_feature(0); // this is only needed to set the initial link
