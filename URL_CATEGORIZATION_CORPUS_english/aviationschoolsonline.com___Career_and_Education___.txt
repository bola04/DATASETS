          <script type="text/javascript">
           //-- Hide Form --//

         jQuery(document).ready(function($) {
             /*------------------------------------
              
              All Functions done on page load
              
              -------------------------------------*/
             if ($('#newLeadForm select[name="country"]').val() != 'United States') {
                 $('#newLeadForm select[name="state"]').parent('label').parent('td').parent('tr').hide();
                 $('#newLeadForm input[name="zip"]').parent('label').parent('td').parent('tr').hide();
              }
             //-- Hide Labels on page load if error --//
             $('#newLeadForm input.lead-text, #newLeadForm textarea').each(function() {
                 if ($(this).val()) {
                     $(this).siblings('span').css({'visibility': 'hidden'});
                 }
             });
             $('#newLeadForm select').each(function() {
                 if ($(this).val()) {
                     $(this).css({'color': '#000000'});
                 } else {
                     $(this).css({'color': '#8d8d8d'});
                 }
             });
             //-- Select color change --//
             $('#newLeadForm select').change(function(event) {
                 if ($(this).val()) {
                     $(this).css({'color': '#000000'});
                 } else {
                     $(this).css({'color': '#8d8d8d'});
                 }
             });
             //-- Show Form --//
             $('#newLeadForm').show('fast');
             /*---  end page load  ---*/
 
             //-- Fill Zip Code --//
             $('#newLeadForm select[name="country"]').change(function(event) {
                 if ($(this).val() != 'United States') {
                     $('#newLeadForm select[name="state"]').parent('label').parent('td').parent('tr').hide();
                     $('#newLeadForm input[name="zip"]').parent('label').parent('td').parent('tr').hide();
                     $('#newLeadForm select[name="state"]').val(0);
                     $('#newLeadForm input[name="zip"]').val('');
                 } else {
                     $('#newLeadForm select[name="state"]').parent('label').parent('td').parent('tr').slideDown('fast');
                     $('#newLeadForm input[name="zip"]').parent('label').parent('td').parent('tr').slideDown('fast');
                 }
                 //-- RESET State Field Label when change country --//
                 if (!$('#newLeadForm select[name="state"]').val()) {
                     $('#newLeadForm select[name="state"]').css({'color': '#8d8d8d'});
                     //$('#newLeadForm select[name="state"]').siblings('span').css({'visibility': 'visible', 'color': '#8d8d8d'});
                 } else {
                     $('#newLeadForm select[name="state"]').css({'color': '#000000'});
                     $('#newLeadForm select[name="state"]').siblings('span').css({'visibility': 'hidden'});
                 }
                 if (!$('#newLeadForm input[name="zip"]').val()) {
                     $('#newLeadForm input[name="zip"]').css({'color': '#8d8d8d'});
                     //$('#newLeadForm select[name="state"]').siblings('span').css({'visibility': 'visible', 'color': '#8d8d8d'});
                 } else {
                     $('#newLeadForm input[name="zip"]').css({'color': '#000000'});
                     $('#newLeadForm input[name="zip"]').siblings('span').css({'visibility': 'hidden'});
                 }
             });
             $('#requestInfoForm').submit(function() {
                 var zipVal = $('#newLeadForm input.lead-text[name="zip"]').val().replace(/\s{2,}/g, '');
                 var countryVal = $('#newLeadForm select[name="country"]').val();
                 var rx_charValid = /[!$%^&*()_+|~=`{}\[\]:";'<>?,.\/]/;
                 if (countryVal != 'United States' && countryVal != 'Canada' && countryVal != '') {
                     if (zipVal == '' || !zipVal || rx_charValid.test(zipVal) || zipVal.length < 2) {
                         $('#newLeadForm input.lead-text[name="zip"]').val('');
                     }
                 }
                 //alert('Form is submitting... zip is : "'+ zipVal +'" -- It was changed to this: '+ $('#newLeadForm input.lead-text[name="zip"]').val());
                 return true;
             });
 
             //-- Dealing with Labels && Visibility --//
             $('#newLeadForm input.lead-text, #newLeadForm select, #newLeadForm textarea').focusin(function(event) {
                 if (!$(this).val()) {
                     $(this).siblings('span').css({'color': '#c4c4c4'});
                 }
             });
             $('#newLeadForm input.lead-text, #newLeadForm textarea').keyup(function(event) {
                 if ($(this).val()) {
                     $(this).siblings('span').css({'visibility': 'hidden'});
                 } else {
                     $(this).siblings('span').css({'visibility': 'visible'});
                 }
             });
             $('#newLeadForm select').change(function(event) {
                 if ($(this).val()) {
                     $(this).siblings('span').css({'visibility': 'hidden'});
                 } else {
                     $(this).siblings('span').css({'visibility': 'visible'});
                 }
             });
             $('#newLeadForm input.lead-text, #newLeadForm select, #newLeadForm textarea').focusout(function(event) {
                 if (!$(this).val()) {
                     $(this).siblings('span').css({'color': '#8d8d8d'});
                 }
             });
         });
