                <h2>
                 <span>
                  About Edgar Cayce
                 </span>
                </h2>
                <p>
                 <span>
                  Edgar Cayce (1877-1945) has been called the "sleeping prophet," the "father of holistic medicine," and the most documented psychic of the 20th century. Cayce was born on a farm in Hopkinsville, Kentucky, in 1877, and his psychic abilities began to appear as early as his childhood. As an adult, Cayce would put himself into a state of meditation, connecting with the universal consciousness and from this state, came his "readings". From holistic health and the treatment of illness to dream interpretation and reincarnation, Cayce's readings and insights offer practical help and advice to individuals from all walks of life, even today.
                 </span>

    (function ($) {
         function postFile() {
             var contentId = $("#contentId").val();
 
             return $.ajax({
                 url: "/Handlers/PublishContentHandler.ashx",
                 type: "GET",
                 contentType: "application/json; charset=utf-8",
                 data: { id: contentId }
             });
         }
 
         function progressQuery() {
             return $.ajax({
                 url: "/Handlers/PublishContentHandler.ashx",
                 type: "GET",
                 contentType: "application/json; charset=utf-8",
                 data: { method: 'GetUploadStatus' }
             });
         }
 
         function getErrors() {
             return $.ajax({
                 url: "/Handlers/PublishContentHandler.ashx",
                 type: "GET",
                 contentType: "application/json; charset=utf-8",
                 data: { method: 'GetErrors' }
             });
         }
 
         var progressInterval;
 
         function getProgress() {
             progressInterval = setInterval(function () {
                 var query = progressQuery();
                 query.done(function (data) {
                     var jsonData = $.parseJSON(data);
                     console.log(jsonData);
                     $('.upload-results').html(jsonData.ResultsMessages);
                     $('#ProgressDiv').css('width', jsonData.Progress + '%');
 
                     if (jsonData.IsComplete) {
                         clearInterval(progressInterval);
                     }
                 });
             },
                 200);
         }
 
         function resetUpload() {
             return $.ajax({
                 url: "/Handlers/PublishContentHandler.ashx",
                 type: "GET",
                 contentType: "application/json; charset=utf-8",
                 data: { method: 'ResetUpload' }
             });
         }
 
         $(document)
             .ready(function () {
                 $('#UploadBtn')
                     .on('click',
                         function (e) {
                             e.preventDefault();
                             var upload = postFile();
                             upload.done(function (data) {
                                 console.log('upload complete');
                                 //populateErrors();
                             });
                             getProgress();
                     });
 
                 $("#print-link").on("click",
                     function() {
                         window.print();
                     });
 
                 $(".checkoutBtn").each(function() {
                     var item = $(this);
 
                     item.on("click",
                         function() {
                             $('.loading-container').removeClass('hidden');
                         });
                 });
             });
     })(jQuery);
