     function stripExt(fileName){
         // if fileName has extension, returns string stripped of extension; otherwise returns string as-is
         dotIndex = fileName.indexOf(".");

             return fileName.substring(0, dotIndex);
         } else{
             return fileName;
         }

     function getGoal(){
         //  Will be updated to handle reservations, etc. Currently set up just for general Web Interaction
         //  returns "consume content" unless url is reserve.venetian.com
         var host = window.location.hostname;
         var splitHost = host.split(".");
         if (splitHost[0] != "reserve"){
             return "consume content";
         }
     }
 
     function getSubGoal(){
         // if URL contains fragment and is a restaurants or entertainment page : view filtered content
         // if page is a pdf file : view file
         // if URL contains 1 (2 if "resort") directory or file : view general content
         // if URL contains 2 (3 if "resort") or more directories : view specific content
         // else: general
         var path = window.location.pathname;
         var offset = path.indexOf("resort") == -1 ? 0 : 1;
         var hash = window.location.hash;
         var pathParts = path.split("/");
         var pageName = path.substring(1, path.indexOf("."));
         if (hash && (pageName == "restaurants" || pageName == "entertainment")){
             return "view filtered content";
         } else if (path.split('.').pop() == "pdf") {
             return "view file";
         } else if (pathParts.length > (2 + offset)) {
             return "view specific content";
         } else {
             return "view general content";
         }
     }
 
     function getCategory(){
         // if page is a pdf file : directory after "venetian" or "palazzo"
         // otherwise, if url has sub-directories: top-level directory or file
         var path = window.location.pathname;
         var pathParts = path.split("/");
         var lastIndex = pathParts.length - 1;
         if (path.split('.').pop() == "pdf") {
             siteName = getHostPart(1);
             catIndex = Math.min(lastIndex, pathParts.indexOf(siteName) + 1);
             return pathParts[catIndex];
         } else if(pathParts.length > 0){
             fileOrDir = pathParts[1];
             return stripExt(fileOrDir);
         } else {
             return "general";
         }
     }
 
     function getSubCategory(){
         // if page is a pdf file : second directory after "venetian" or "palazzo"
         // if URL contains fragment and is a restaurants or entertainment page : string after last "/"
         // default: file or directory directly after top-level directory
         var path = window.location.pathname;
         var pathParts = path.split("/");
         if(pathParts.length == 1){
             return 'general';
         }
         var hash = window.location.hash;
         var pageName = path.substr(1, path.indexOf(".") - 1);
         var lastIndex = pathParts.length - 1;
         if (path.split('.').pop() == "pdf") {
             var siteName = getHostPart(1);
             var subCatIndex = Math.min(lastIndex, pathParts.indexOf(siteName) + 2);
             var fileOrDir = pathParts[subCatIndex];
             return stripExt(fileOrDir);
         } else if (hash && (pageName == "restaurants" || pageName == "entertainment")) {
             return pathParts[lastIndex];
         } else {
             var subCatIndex = Math.min(lastIndex, 2);
             var fileOrDir = pathParts[subCatIndex];
             return stripExt(fileOrDir);
         }
     }

     function addListener(element, eventName, handler) {
       if (element.addEventListener) {
         element.addEventListener(eventName, handler, false);
       }
       else if (element.attachEvent) {
         element.attachEvent('on' + eventName, handler);
       }
       else {
         element['on' + eventName] = handler;
       }
     }
 
     function removeListener(element, eventName, handler) {
       if (element.addEventListener) {
         element.removeEventListener(eventName, handler, false);
       }
       else if (element.detachEvent) {
         element.detachEvent('on' + eventName, handler);
       }
       else {
         element['on' + eventName] = null;
       }
     }
