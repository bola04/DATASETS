 		function getCountryCode(){

 				type: 'GET',
 				url: 'https://api.teletext.io/api/v1/geo-ip',

 					if (xhr && xhr.overrideMimeType) {
 						xhr.overrideMimeType('application/json;charset=utf-8');
 					}
 				},

 					var countryCode = data.alpha2.toLowerCase();
 					if(countryCode == 'us'){
