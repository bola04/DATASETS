 // adds a user tag 'latest_print_date' after a print session
 function track_print() {
 	
 	 if (OneSignal)
 	 {
 		try
 		{
 			OneSignal.isPushNotificationsSupported();
 		}
 		catch (err)
 		{
 			console.log(err);
 			return false;
 		}
 	 var isPushSupported = OneSignal.isPushNotificationsSupported();
 	  if (isPushSupported) {
 		// Push notifications are supported
 	  } else {
 		return false;
 	  }
 	
 	OneSignal.isPushNotificationsEnabled(function(isPushEnabled) {
 	
     // checks to see if push notifications are enabled
 	if (isPushEnabled) {
 		var today = new Date();
 		var dd = today.getDate();
 		var mm = today.getMonth()+1; //January is 0!
 		var yyyy = today.getFullYear();
 
 		if(dd<10) {
 			dd='0'+dd
 		} 
 
 		if(mm<10) {
 			mm='0'+mm
 		} 
 
 		today = yyyy+mm+dd;
 			
 			OneSignal.sendTag("latest_print_date", today);
 		}
 	});
 	 }
 }
