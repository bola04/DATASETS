     $(document).ready(function () {
         $('[data-scroll-to]').on('click', handleScrollClick);
 
         function handleScrollClick(evt) {
             var $this = $(this),
 				$scrollTarget = $($this.attr('data-scroll-to') + ':visible'),
 				headerHeight = $('header').height(),
 				scrollTargetOffset = calculateScrollOffset($this, $scrollTarget, headerHeight);
 
             // animate scroll to target
             $('html, body').animate({
                 scrollTop: scrollTargetOffset
             });
         };
 
         function calculateScrollOffset($this, $scrollTarget, headerHeight) {
             if (!$scrollTarget || !$scrollTarget.length) return headerHeight;
 
             return $scrollTarget.offset().top - headerHeight;
         };
     });
