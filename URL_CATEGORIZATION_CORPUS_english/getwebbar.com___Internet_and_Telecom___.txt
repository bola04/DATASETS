      // validate contact form
 $(function() {
     $('#contact').validate({
         rules: {
             name: {
                 required: true,
                 minlength: 2
             },
             email: {
                 required: true,
                 email: true
             },
             message: {
                 required: true
             },
             captcha: {
                 required: true,
                 answercheck: true
             }
         },
         messages: {
             name: {
                 required: "This field is required",
                 minlength: ""
             },
             email: {
                 required: "This field is required"
             },
             message: {
                 required: "This field is required",
                 minlength: "This field is required"
             },
             captcha: {
                 required: "This field is required"
             }
         },
         submitHandler: function(form) {
             $(form).ajaxSubmit({
                 type:"POST",
                 data: $(form).serialize(),
                 url:"process.php",
                 success: function() {
                     $('#contact').fadeTo( "slow", 0.15, function() {
                         $('#success').fadeIn();
                     });
                 },
                 error: function() {
                     $('#contact').fadeTo( "slow", 0.15, function() {
                         $('#error').fadeIn();
                     });
                 }
             });
         }
     });
 });
