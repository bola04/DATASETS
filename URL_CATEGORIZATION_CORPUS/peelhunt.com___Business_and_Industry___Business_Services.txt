      // onRegisterSuccess()
     $("#register-form").on("submit", function(e) {

      $.ajax({
        type: "POST",
        url : $(this).attr("action"),
        data: $(this).serialize(),
        success:function() {
          onRegisterSuccess();
        },
        error: function(data, a, b) {
          alert(data.responseText)
          $("#register-form input[type='text'], #register-form input[type='email'], #register-form input[type='tel']").val("")
        }
      })

      return false;
    })
