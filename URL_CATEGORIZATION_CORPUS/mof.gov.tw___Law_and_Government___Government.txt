        function calendar_act(Nowdate) {
            postURL = '/Home/NewActivityContent';
            $.ajax({
                method: "POST",
                url: postURL,
                dataType: "text",
                async: true,
                data: { NowDate: Nowdate },
                cache: false,
                success: function (data) {
                    if (data.split('@@@')[0] != "") {
                        var BoxTime = (data.split('@@@')[1].length > 30) ? data.split('@@@')[1].substr(0, 30) + "..." : data.split('@@@')[1];
                        var Boxheadline = "";
                        var CheckString = "";
                        var PagePath = "";
                        var day = "";
                        var week = "";
                        for (var i = 0; i <= data.split('@@@')[0].length; i++) {
                            if (Boxheadline.length < 70) {

                                if (data.split('@@@')[0].charAt(i) == '<') {
                                    CheckString = "<";
                                }

                                if (CheckString == "<" && data.split('@@@')[0].charAt(i - 1) == ">") {
                                    CheckString = "";
                                }

                                if (CheckString == "" && data.split('@@@')[0].charAt(i) != "<") {

                                    Boxheadline += data.split('@@@')[0].charAt(i);
                                    if (Boxheadline.length % 25 == 0) {
                                        Boxheadline += "\n";
                                    }

                                    if (Boxheadline.length + 1 > 70) {
                                        Boxheadline += "....";
                                    }
                                    CheckString = '';
                                }
                                else {
                                    CheckString = "<";
                                }
                            }
                            else {
                                break;
                            }
                        }

                        PagePath = "/Detail/Index?nodeid=" + data.split('@@@')[2] + "&pid=" + data.split('@@@')[3] + "";
                        Boxheadline = "<a href=\"" + PagePath + "\">" + Boxheadline + "</a>";
                        day = data.split('@@@')[5];
                        week = data.split('@@@')[4];
                        $('#calendar_day').html(day);
                        $('#calendar_week').html(week);
                        $('#calendar_content').html(Boxheadline);
                        //$("..indexStrokeBox_more a").attr("href", PagePath);
                        //$(".indexScheduleBox_time").html(BoxTime);
                        //$(".indexScheduleBox_headline").html(Boxheadline.replace("/[\r\n\t]/g", "").replace("/<[^>]+>/g", "|"));
                    }
                    else {
                        var day = "";
                        var week = "";
                        day = data.split('@@@')[5];
                        week = data.split('@@@')[4];
                        $('#calendar_day').html(day);
                        $('#calendar_week').html(week);
                        $('#calendar_content').html('<p>本日無活動公告</p>');
                        //$(".indexScheduleBox_headline").html("本日無活動公告");
                        //$(".indexStrokeBox_more a").attr("href", "./Pages/Detail.aspx?nodeid=42&pid=64831");
                        //$(".indexScheduleBox_time").html("");
                    }
                },
                error: function () {
                    //closeLoading();
                }
            });
        }
