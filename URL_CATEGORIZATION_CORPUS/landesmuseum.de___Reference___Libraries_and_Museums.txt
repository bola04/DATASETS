      <script>
       seriesScrollTimeout = null;
seriesScrollTimeoutTime = 4500;
seriesScrollAnimationTime = 1500;
seriesScrollProducWidth = 189;
seriesAutoscroll = true;

function ID(id) { return document.getElementById(id); }

function cloneSeriesElements()
{
	clones = new Array();
	for(c=0; c<ID('product_series_list_ul').childNodes.length; c++)
	{
		clones[clones.length] = ID('product_series_list_ul').childNodes[c].cloneNode(true);
	}
	for(c in clones)
	{
		ID('product_series_list_ul').appendChild(clones[c]);
	}
}
function seriesScroll()
{
	if(parseInt($('#product_series_box').scrollLeft()) >= ((parseInt(ID('product_series_list').style.width) / 2 )))
	{
		// scroll to 0
		$('#product_series_box').animate({ scrollLeft: 0 }, 0);
	}
	// else
	{
		$('#product_series_box').animate({ scrollLeft: '+='+seriesScrollProducWidth }, seriesScrollAnimationTime);
	}
	serieScrollStart();
}
function seriesScrollBack()
{
	if(parseInt($('#product_series_box').scrollLeft()) <= 0)
	{
		// scroll to 0
		$('#product_series_box').animate({ scrollLeft: ((parseInt(ID('product_series_list').style.width) / 2 )) }, 0);
	}
	// else
	{
		$('#product_series_box').animate({ scrollLeft: '-='+seriesScrollProducWidth }, seriesScrollAnimationTime);
	}
}
function serieScrollStop()
{
	window.clearTimeout(seriesScrollTimeout);
}
function serieScrollStart()
{
	if(seriesAutoscroll == true) seriesScrollTimeout = window.setTimeout("seriesScroll();", seriesScrollTimeoutTime);
}
function serieScrollInit()
{
	// $('#product_series_box').css('overflow', 'hidden');
	ID('product_series_box').style.overflow = 'hidden';
	ID('product_series_list').style.width = (((ID('product_series_list').getElementsByTagName('li').length) * seriesScrollProducWidth) ) + "px";
	if(parseInt(ID('product_series_list').style.width) > parseInt($('#product_series_box').width()))
	{
		cloneSeriesElements();
		ID('product_series_list').style.width = (((ID('product_series_list').getElementsByTagName('li').length) * seriesScrollProducWidth) ) + "px";
		serieScrollStart();
		ID('product_serie_prev').style.display = '';
		ID('product_serie_next').style.display = '';
	}
	else
	{
		ID('product_serie_prev').style.display = 'none';
		ID('product_serie_next').style.display = 'none';
	}
	
}
function seriesScrollNext()
{
	serieScrollStop();
	seriesScroll();
	serieScrollStop();
}
function seriesScrollPrev()
{
	serieScrollStop();
	seriesScrollBack();
	serieScrollStop();
}
