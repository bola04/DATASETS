       $(function(){

	function initAuth(e){
		e.preventDefault();
		$('#loginformboxwrap').fadeToggle("fast");
		validateLogin();
	}

	function validateLogin()
	{   
	    jQuery('#email').focus();
	    jQuery('.loginbox form').validate({
	        submitHandler: submitLogin,
	        errorPlacement: function(error, element) {
	                jQuery('.err', jQuery(element).parent().parent()).html(error);
	        },
	        rules: {
	        	 email: {
	                required: true,
	                email: true
	            },
	            password: {
	              required: true
	            }
	        },
	        
	        messages:{ 
	        	password: 'Šis lauks ir aizpildāms obligāti.', 
	        	email: { 
	        		required: 'Šis lauks ir aizpildāms obligāti.',
	        		email: 'Nepareizs e-pasts'
	        	 }
	        }
	    });
	}
	function submitLogin()
	{   
	    jQuery('.loginbox .error_message').hide();
	    form = $('.loginbox form');
	    form.ajaxSubmit({'dataType':'json', 'success': loginResponse, 'url': form.attr('action')+'?ref=a'});
	    return false;
	}
	function loginResponse(response, statusText, xhr, $form) 
	{        
		if(response.msg != 'ok'){
			jQuery('.loginbox .error_message').show();
			return false;
		}
    else{
			window.location.reload();
    }
	}
	$(document).keyup(function(e) {
		if (e.keyCode == 27) { 
			jQuery('#loginformboxwrap').fadeOut('fast'); 
		}
	});
