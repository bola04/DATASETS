   $(document).ready(function () {
                //Add target blank to every external link
                $('a[href^="http://"], a[href^="https://"]').attr('target', '_blank');

                //Remove Height and with on all images
                $('img').each(function () {
                    $(this).removeAttr('width');
                    $(this).removeAttr('height');
                });
            });


            /*Add tracking on all tel: and mailto:*/
            window.onload = function () {
                var anchors = document.getElementsByTagName('a');
                for (var i = 0; i < anchors.length; i++) {
                    //telephone
                    if (anchors[i].toString().indexOf('tel:') > -1) {
                        anchors[i].setAttribute('onClick', 'ga(\'send\', \'event\', \'telefonnummer\', \'klikk\', \'' + anchors[i].innerHTML.replace(' ', '').replace(' ', '').replace(' ', '').replace(' ', '') + '\')');
                    }
                    //email
                    if (anchors[i].toString().indexOf('mailto:') > -1) {
                        anchors[i].setAttribute('onClick', 'ga(\'send\', \'event\', \'mail\', \'klikk\')');
                    }
                }
            };
  </script>
