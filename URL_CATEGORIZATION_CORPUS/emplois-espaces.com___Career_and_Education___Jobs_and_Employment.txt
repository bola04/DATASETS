function reset_moteur_marches (){
  suppr_pre_sel("purge","");
  $('#francemap').vectorMap('set', 'colors',couleurs);
  $('#emplois_search_departement').val("");
  $('#emplois_search_secteur').val("");
  $('#emplois_search_type').val("");
  $('#emplois_search_type_contrat').val("");
}


function suppr_pre_sel(bloc,valeur){
  $.ajax({
    type: 'POST', 
    url: '/services/emplois_pre_sel_search.php',
    dataType: 'html', 
    data: "selection=supprimer&bloc=" + bloc + "&valeur=" + valeur ,
    success: function(data) {
      affiche_sel();
      pre_resultat();
      maj_carte();
    }
  })
}
function affiche_sel(){
  $.ajax({
    type: 'POST', 
    url: '/services/emplois_pre_sel_search.php',
    dataType: 'html', 
    data: "selection=afficher",
    success: function(data) {
      $('#box_selection_type').html(data);
      pre_resultat();
    }
  })
}

function maj_carte(){
  $.getJSON('/services/emplois_pre_sel_search.php?selection=couleurdepartement', function(data){
    $('#francemap').vectorMap('set', 'colors',couleurs);
    $('#francemap').vectorMap('set', 'colors',data);
  })
}

function pre_resultat(){
  $.ajax({
    type: 'POST', 
    url: '/services/emplois_pre_search.php',
    dataType: 'html', 
    data: "" ,
    success: function(data) {
      $('#emplois_search_bouton').val(data);
    }
  })
}
