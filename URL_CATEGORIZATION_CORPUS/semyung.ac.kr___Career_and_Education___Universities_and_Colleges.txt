        function site_search(t){
                    var site_search = {};
                    var keyword = $("#site_keyword").val();
                    site_search.btn = $("#site_submit");
                    site_search.obj = $("#site_search");
                    site_search.data = false;
        			
					
					//console.log("t.qt.value : "+t.qt.value);
                    if(t.qt.value === "" && site_search.obj.attr("class").match("open")){
                        t.qt.value = "";
                        site_search.obj.removeClass("open");

                        site_search.data = false;
						//console.log("111111111");
        
                    }else if(t.qt.value === "" && !site_search.obj.attr("class").match("open")){
                        site_search.obj.addClass("open");  
                        site_search.data = true;
  
                        //console.log("222222222");
        
                    }else{
                        $("#pQuery").val(keyword);
                        $("#pPastQuery").val(keyword);
   
                        t.submit();
						//console.log("3333333333");
        
                    }
        
                }

                    function _in(event){
                        if(!$("#site_search").attr("class").match("open")){

                    function _out(event){
                        $("#site_search").removeClass("open");
                    }
        
                    $("#site_keyword").on("focusin",_in);
                    $("#header #logo span .ir-logo, #sitelink .sitemap").on("focusin",_out);
