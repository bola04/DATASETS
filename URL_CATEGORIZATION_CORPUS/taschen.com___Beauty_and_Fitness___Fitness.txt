  <script>
   var win_width = 0;
var has_mouseevents = 0;

 	function get_window_width()
 	{
	return window.innerWidth; //$(window).width();
	}
 
 
 	function add_animate_events()
 	{
	$( "body" ).on( "mouseenter", ".press_highlight_teaser", press_highlight_teaser_mouseenter);
	$( "body" ).on( "mouseleave", ".press_highlight_teaser", press_highlight_teaser_mouseleave);

	has_mouseevents = 1;
	}
 

 	function remove_animate_events()
 	{
	$( "body" ).off( "mouseenter", ".press_highlight_teaser", press_highlight_teaser_mouseenter);
	$( "body" ).off( "mouseleave", ".press_highlight_teaser", press_highlight_teaser_mouseleave);
				
	has_mouseevents = 0;
	}
 
 
	function press_highlight_teaser_mouseenter()
 	{
	presshighlight_hover_func(this, 1, 100, 1.1, 0.05, 20, '255,255,255', '#DEDEDE');	
 	}


 	function press_highlight_teaser_mouseleave()
 	{
	presshighlight_hover_func(this, 0, 100, 1, 0, 0, '245,245,245', '#F5F5F5');
 	}
 
 
 	function remove_animate_styles()
 	{
	$('.press-highlight-grid .grid-inner').removeAttr('style');
	$('.grid-inner .content-block-img').removeAttr('style');
	$('.grid-inner .content-block-text').removeAttr('style');
	}

 
 
 

 
 
	$(document).ready(function()
	{	
	win_width = get_window_width(); //$(window).width();
		
		// animate styles entfernen bei resize, ggf. Eventlistener entfernen oder dazu
		$(window).bind('resize', function(e)
		{
		remove_animate_styles();
		
		win_width = get_window_width();		
			
				if((win_width < 768) && (has_mouseevents == 1))
				{
				remove_animate_events();
				}
			
				if((win_width >= 768) && (has_mouseevents == 0))
				{	
				add_animate_events();
				}

		});


		if(win_width >= 768) 
		{
		add_animate_events();
		}

	});

		{

	// Slash durch Underscore ersetzen
	tmp_search_string = tmp_search_string.replace(/\//g, '_'); // global replacement
	
	// klein
	tmp_search_string = tmp_search_string.toLowerCase();
	
	// Leerzeichen durch + oder - ersetzen
	tmp_search_string = tmp_search_string.replace(/ /g, tmp_delimiter); // global replacement
