	for( i=0; i<count; i++ )
	{
		var tta = ta[i].split(/\:/);
		if( tta.length != 2 ) { continue; }
		tta[0] = tta[0].toLowerCase();
		if( tta[0].match(/\-/) )

			var taa = tta[0].split(/\-/);
			taa[1] = taa[1].charAt(0).toUpperCase() + taa[1].slice(1);
			tta[0] = taa.join("");
		}
		eval('document.getElementById("cmf_button-field").style.'+tta[0]+'="'+tta[1]+'"');
	}
}
