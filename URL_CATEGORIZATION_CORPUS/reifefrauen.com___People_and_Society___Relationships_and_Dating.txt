                if (undefined !== jQuery(this).prop('src')) {
                    var src = jQuery(this).prop('src');
                    if (0 < src.indexOf('/gtag/') && 0 < src.indexOf('id=')) {
                        window.gaSettings.found = true;
                        window.gaSettings.disableStr += src.substr(src.indexOf('id=') + 3);
                        
                        document.getElementById('gaOptOut').addEventListener('click', function () {
                            gaOptout(window.gaSettings);
                            gaOptOutFeedback(window.gaSettings);
                        });
                        if (document.cookie.indexOf(window.gaSettings.disableStr + '=true') > -1) {
                            window[window.gaSettings.disableStr] = true;
                            gaOptOutFeedback(window.gaSettings);
                        }
                    }
                }
            });
        };
        if (null !== document.getElementById('gaOptOut')) {
            jQuery(document).ready(function () {
                gaOptOutInit();
                window.setTimeout(function () {
                    if (false === window.gaSettings.found) {
                        gaOptOutInit();
                    }
                }, 1000);
            });
        }
