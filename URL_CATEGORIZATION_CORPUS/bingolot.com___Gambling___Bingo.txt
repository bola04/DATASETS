   currentSelection = null;
                function setBackground(div, suffix) {
                    if (div == null) {
                        return;
                    }
                    var s;
                    if (suffix == null || suffix == "") {
                        s = "";
                    } else {
                        s = "-" + suffix;
                    }
                    if (div.style) {
                        div.style.backgroundImage = "url(i/menu-bg" + s + ".png)";
                        if (div.childNodes[0] != null) {
                            div.childNodes[0].style.backgroundImage = "url(i/menu-bg" + s + "-l.png)";
                            div.childNodes[0].childNodes[0].style.backgroundImage = "url(i/menu-bg" + s + "-r.png)";
                        }
                    }

                }
                function mmOver(o) {
                    if (currentSelection != null) {
                        setBackground(currentSelection, null);
                    }
                    if (o.childNodes != null) {
                        setBackground(o, "s");
                        currentSelection = o;
                    }
                }
            
                function mmOut(e) {
                    if (e == currentSelection) {
                        currentSelection = null;
                        setBackground(e, null);
                    }
                }
