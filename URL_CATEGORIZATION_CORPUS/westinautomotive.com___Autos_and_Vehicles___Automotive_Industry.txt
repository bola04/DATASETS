function collectWeights()
{
	var totalWeight = 0;
	$('.answerRadio:checked').each(
	function()
	{
		totalWeight += parseFloat(weightArray[($(this).attr('id'))]);
	})
	$('.selectbox').each(
	function()
	{
		id = $(this).val();
		if(id)
		{
			totalWeight += parseFloat(weightArray['answer_'+id]);
		}
		
	});
	if(totalWeight == 0 || isNaN(totalWeight))
	{
		$('#surveyWeight').hide();
	}
	else
	{
		$('#surveyWeight').html(totalWeight);
		$('#surveyWeight').show();
	}
}

			});

		{
			sendResponse(response_status,'');
		}

		//$('#surveyReview').hide();
		sendResponse(response_status,1);
	}

	{
		// $('.emailOnly[value=""]').addClass('validate[required,custom[email]]');
	}
}


function sendResponse(response_status,end)
{
	eSBM.response_status = response_status;
	eSBM.end = end;
	if($('.fileUpload').length)
	{
		$('#surveyFrame').on('load',function()
		{
			$('.formError').remove();
			var test = $('#surveyFrame').contents().find('div').length;
			if(test)
			{
				$('#surveyFrame').contents().find('div').each(
				function()
				{
					var content = $(this).html();
					var contentArray = content.split(":");
					var answer_id = contentArray[0];
					var error = contentArray[1];
					$('#'+answer_id).validationEngine('showPrompt',error,'','',1);
				});
			}
			else
			{
				finishResponse();
			}


		});
		$("#command").val('esbmUploadResponse');
		$("#surveyForm").submit();
	}
	else
	{
		$('#surveyFrame').off('load');
		finishResponse();
	}
}

function finishResponse()
{
	$('#surveyForm').hide();
	response_status = eSBM.response_status;
	end = eSBM.end;
	$("#command").val('esbmSurveyResponse');
	var params = $('#surveyForm').serialize();
	if(params)
	{
		esbmGetPage('&command=esbmSurveyResponse&response_status='+response_status+'&'+params,'surveyContent',
		function()
		{
			if(end==1)
			{
				$('.surveyReview').hide();
				$('.surveySave').hide();
				$('.surveySubmit').hide();
			}
		});
	}
}

function esbmGetPage(params,div,func,async)
{
	if(async != 'false')
	{
		async = 'true';
	}
	params += '&ajax=1&ajdt=1';
	if(div == 'NEW')
	{
		window.open(url+params);
	}
	else
	{
		$.ajax
		({
			type: type,
			url: url,
			data: params,
			async: true,
			success: function(data)
			{
				resetData = data;
				if(div == 'COLORBOX')
				{
					$.fn.colorbox({html:data});
				}
				else if(div == 'DIALOG')
				{
					$('#dialogList').append(data);
				}
				else if(div != '')
				{
					$('#'+div).html(data);
				}

				if(typeof func !== 'undefined' && func != '')
				{
					if(typeof func === 'function')
					{
						func(data);
					}
					else if(typeof window[func] === 'function')
					{
						window[func](data);
					}
				}

			},
			complete: function(xhr,statusText)
			{
				if(xhr.status == 0 || xhr.status == '')
				{
					if(count < 5)
					{
						count++;
						esbmGetPage(params,div,func,async,count);
					}
				}
			}
		});
	}
}
