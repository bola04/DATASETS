   $(document).ready(function() {

        $( ".search_button" ).click(function() {
            var keyword = $(this).parents('.autocomplete_search').children('.search_box').val();

            if(keyword == "") {
                return false;
            }

            var q = $.parseQuery($(this).attr('data-meta'));
            var document_type_name = q['document_type_name'];

            url = "/" + document_type_name + "/keyword/" + keyword;
            url = normalize_url(url);

            if(!document_type_name) {
                console.log("Error: No document type specified.");
                return false;
            }

            window.location = (url);

        });

        $(".search_box").live("focus", function() {

        //var search_box_title = $(".search_box").attr("title");

        var q = $.parseQuery($(this).attr('data-meta'));
        var document_type_name = q['document_type_name'] || 'quotes';
        var document_type_name_singular = document_type_name.substr(0, document_type_name.length-1);

    	$( this )
			.bind( "keydown", function( event ) {
                q = $.parseQuery($(this).attr('data-meta'));
                document_type_name = q['document_type_name'] || 'quotes';
                document_type_name_singular = document_type_name.substr(0, document_type_name.length-1);
				if ( event.keyCode === $.ui.keyCode.TAB &&
						$( this ).data( "autocomplete" ).menu.active ) {
					event.preventDefault();
				}
			})     
        .autocomplete({
			minLength: 2,
			source: function( request, response ) {
					$.getJSON( "/json/advanced_autocomplete?type=" + q['type'] + "&document_type_name=" + q['document_type_name'], {
						//term: extractLast( request.term )
                        term: function() {
                            if(q['list'] == "true") {
                                return extractLast( request.term );
                            }
                            else {
                                return request.term;
                            }
                        }
					}, response );
				},
			search: function() {
				// custom minLength
				var term = extractLast( this.value );
				if ( term.length < 2 ) {
					return false;
				}
			},
			focus: function() {
				return false;
			},
			select: function( event, ui ) {

                if(q['override_select'] != "" && q['override_select'] != null) {
                    eval(q['override_select'] + "(event, ui, this, q['document_id'])");
                    return false;
                }

                if(q['type'] == "topic") {
                    var terms = split( this.value );
                    terms.pop();
                    terms.push( ui.item.name );
                    terms.push( "" );
                    this.value = terms.join( ", " );
                    return false;
                }
                else if(q['type'] == "author") {
                    
                    if(q['list']) {
                        var terms = split( this.value );
                        terms.pop();
                        terms.push( ui.item.name );
                        terms.push( "" );
                        this.value = terms.join( ", " );
                    }
                    else {
                        this.value = ui.item.name;
                    }
                    return false;
                }
                var string = ui.item.name;
                var url = "";

                if(ui.item.type == "author") {
                    url = "/" + document_type_name + "/author/" + ui.item.first_name + "-SPLIT-" + ui.item.last_name;
                    string = ui.item.name;
                }
                else if(ui.item.type == "user") {
                    url = "/" + document_type_name + "/author/" + ui.item.uid;
                    string = ui.item.uid;
                }
                else if(ui.item.type == "document") {
                    url = "/" + document_type_name_singular + "/" + ui.item.id;
                    string = ui.item.title || ui.item.body;
                }
                else {
                    url = "/" + document_type_name + "/" + ui.item.type + "/" + ui.item.name;
                }

                $(this).val( string );

                url = normalize_url(url);

                window.location = (url);

                return false;
			}
		})
        .data( "autocomplete" )._renderItem = function( ul, item ) {
            var description = "";
            var movie_year = "";
            var css_class = item.type;
            var name = item.name || "";
            var doc = item.title || item.body || "";
            if(item.description != null && item.description != "") {
                description = '<br /><small><span style="color:gray;">' + item.description + "</span></small>";
            }
            if(item.type == "movie") {
                if(item.year != null && item.year != "0000") {
                    movie_year = " <small>" + item.year + "</small>";
                }
            }
            else if(item.type == "author") {
                if(item.birth && item.birth != "0000-00-00") {
                    name = name + " <small>" + item.birth + "</small>";
                }
                if(item.death && item.death != "0000-00-00") {
                    name = name + " <small> - " + item.death + "</small>";
                }
            }

            var image = "";
            
            if(item.has_image == 1) {
                var image_folder = "authors";
                if(item.type == "user") {
                    image_folder = "users";
                }
                image = '<img src="/static/images/' + image_folder +'/' + item.id + '-small.jpg" alt="" class="left" style="margin-right:8px;height:33px;" />'
            }
            
            var doc_count = "";
            //if(item.doc_count == 0) {
                //console.log("Skip - has no documents.");
                // do nothing
                //return;
            //}
            if(item.type != 'keyword' && item.doc_count != null && q['show_count'] != 'false') {
                doc_count =
                    '<span style="color:gray;font-size:10px;"> - ' 
                    + item.doc_count
                    + ' Documents Found</span>';
            }

            if(item.no_results == 1) {
                return $( '<li></li>' )
                    .data( "item.autocomplete", item )
                    .append( "No Results Found" )
                    .addClass(css_class + "_auto")
                    .appendTo( ul );
            }
            else {
                return $( '<li></li>' )
                    .data( "item.autocomplete", item )
                    .append( 
                        '<a>' 
                        + image
                        + name 
                        + doc
                        + movie_year 
                        + description 
                        + doc_count
                        + "</a>" )
                    .addClass(css_class + "_auto")
                    .appendTo( ul );
            }
		};

        function split( val ) {
			return val.split( /,\s*/ );
		}
        function extractLast( term ) {
			return split( term ).pop();
		}

});
	});

function normalize_url(url) {
    url = url.replace(/-SPLIT-/g, "/");
    return url.replace(/ /g, "+");
}

      A Girl and her Troubles
     </h2>
     <img alt="Alicia Haynes" src="http://www.1-love-quotes.com/static/managed_images/11/11AA9ADA-76D3-11E2-A86C-86F9D2894A9A.jpg" style="vertical-align:middle; float:left; margin-right:12px; width:250px"/>

      Mother's Day this Sunday...
      <br/>
      <br/>
      Louisa May Alcott, the beloved writer of Little Women, which was basically a model of her own life with her sisters and her mother, asked this question, "What do girls do who haven't any mothers to help them through their troubles?" This question is very plaguing, because when we look around us at the world, at the girls that we grow up with in school, the ones that we see growing up in the media spot light, and the women that we grow to love as close friends through out our lives. Well, it becomes clear to us that not every girl has been able to grow up with the
      <a href="http://www.1-love-quotes.com/quotes/category/Love">
       love
      </a>
      of a
      <a href="http://www.1-love-quotes.com/quotes/category/Mothers">
       mother
      </a>
      .
      <br/>
      <br/>
      We see all of the ways that these types of situations can have on a girl. She might grow up to be unsure of herself. She might not receive all of the important life lessons that a mother can give her daughter. This might not be because of an abandonment, it might be from the passing of the mother at an early age, but no matter what the reason is for the absence of a mother in a little girl's life, it really can affect them and mold them through out the rest of their life.
      <br/>
      <br/>
      We see the girls who have grown up with no mother to guide them in the gentle, and sometimes quite frank ways that a mother does. There are situations that come up in a girls life, especially a young teenager when it is just hard to talk about a man, your father, about it. These are things like your menstruation, the period of time that you are essentially, becoming a woman. It is an awkward, strange time filled with raging hormones and a nervous anxiety that goes along with it. A man can't put himself in a girls shoes, because he has not been in her shoes before. A woman though, she is well aware of all of the horribly awkward steps to becoming a woman.
      <br/>
      <br/>
      There are the times when a girl is going to feel unsure of herself, she is going to need that boost in confidence that only a mother or a very close girl friend can give. There are things such as cosmetics, hair products, feminine products, and things that are just considered for girls, that a man will not get, they will not understand. There are so many troubles that a girl will live through in even just one day of her teenage life, so with no mothers influence to help her set things straight, and help her to stay positive, even through the most embarrassing moments in her life so far.
      <br/>
      <br/>
      <a href="http://www.1-love-quotes.com/quotes/author/Louisa+May/Alcott">
       Louisa May Alcott
      </a>
      grew up with a bunch of sisters and a devoted mother, so although she probably had enough of her own embarrassing moments in life she did have a big support group. There are a lot of girls who will never be able to experience this first hand, but they can always through her books about the March sisters
