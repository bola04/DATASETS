function filterResultes(){
	var d = new Date();
	var dateStart = ((''+(d.getMonth()+1)).length<2 ? '0' : '') + (d.getMonth()+1) + '/' + ((''+d.getDate()).length<2 ? '0' : '') + d.getDate() + '/' + d.getFullYear();
	var categoryID = $('.catSelect select').val();
	
	if($('input.dateStart').val() != 'Select a start date' && $('input.dateStart').val() != ''){
		dateStart = $('input.dateStart').val();
	}
	
	if($('.catSelect select').val() == 'Select an event category'){
		categoryID = '';
	}
	
	var dateSpanAmount = parseInt($('.dateSpan').val());
	var dateSpanType = $('.dateSpan').val().substr(1);
	
	var loadingBackground = $('<div/>').css({
		'background':'#000000',
		'opacity':'0.6',
		'filter':'alpha(opacity=60)',
		'width':'100%',
		'height':'100%',
		'position':'absolute',
		'top':'0',
		'left':'0',
		'z-index':'100'
	});
	var overlayImg = $('<div/>').css({
		'width':'24px',
		'height':'24px',
		'position':'absolute',
		'top':'50%',
		'left':'50%',
		'margin-left':'-64px',
		'margin-top':'-7px',
		'z-index':'101'
	}).html('<img src="/flashcab/directory/apps/calendarOfEvents/v1/ajax-loader.gif">');
	
	
	$('.eventCal > .jspContainer > .jspPane').css('left','0');
	api.destroy();
	
	$('.eventCal').append(loadingBackground).append(overlayImg).load('/flashcab/directory/apps/calendarOfEvents/v1/loadedPage.cfm?siteID=SITE_131001_10093522_E1EHH&dateSpanAmount='+dateSpanAmount+'&dateSpanType='+dateSpanType+'&dateStart='+dateStart+'&categoryID='+categoryID,function(){
		$('.eventPreview').click(function(){
			eventBinding(this);
			$('.eventCal > .jspContainer > .jspPane').css('left','0');
		});
	});
}

function eventBinding(foo){
	thisEvent = $(foo).parents('.event');
	if(thisEvent.width() == thisEvent.find('.eventContainer').width()){
		
		if(eventJSP != ''){
			eventJSP.destroy();
			eventJSP = '';
		}
		
		thisEvent.animate({
			width:'155px'
		},400,function(){
			if(openSlide){
				$('.eventCalCont').width($('.eventCalCont').width() - (thisEvent.find('.eventContainer').width() - 155));
			}
			openSlide = false;
		});
	}else{
		if(!openSlide){
			$('.eventCalCont').width($('.eventCalCont').width() + (thisEvent.find('.eventContainer').width() - 155));
		}
		openSlide = true;
		
		if(eventJSP != ''){
			eventJSP.destroy();
		}
		
		$('.event#'+currentSlide).animate({
			width:'155px'
		},400);
		
		thisEvent.animate({
			width:thisEvent.find('.eventContainer').width() + 'px'
		},400,function(){
			currentSlide = thisEvent.attr('id');
			thisEvent.find('.eventDetails').jScrollPane();
			eventJSP = thisEvent.find('.eventDetails').data('jsp');
		});
	}
}
