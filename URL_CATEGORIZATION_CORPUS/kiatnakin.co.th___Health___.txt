        function Subscribe() {
            if ($('#lgFormGroupInput').val() != "") {
                if (validateEmail($('#lgFormGroupInput').val())) {

                    $('.cms-btn-receive').prop('disabled', true);
                    $('#lgFormGroupInput').prop('readonly', true);
                    $('.cms-btn-receive').prepend('<i class="fa fa-spinner fa-pulse fa-fw mr-2"></i>');

                    $.ajax({
                        url: '/Utility/Subscribe',
                        type: 'GET',
                        data: {
                            Email: $('#lgFormGroupInput').val()
                        },
                        success: function (response) {
                            var datares = JSON.parse(response);
                            var txtMsg = '';
                            if (datares.Success != true) {
                                if (datares.ErrorCode === '99895') {
                                    txtMsg = 'อีเมลนี้ถูกใช้สมัครรับข่าวสารแล้ว';
                                } else {
                                    txtMsg = 'สมัครรับข่าวสารไม่สำเร็จ';
                                }
                            } else {
                                txtMsg = 'สมัครรับข่าวสารเรียบร้อย';
                            }
                            SubscribeValidateMsg(txtMsg, datares.Success);
                            $('.cms-btn-receive .fa-spinner').remove();
                            $('#lgFormGroupInput').prop('readonly', false);
                            $('.cms-btn-receive').prop('disabled', false);
                        }
                    });
                }
                if (!validateEmail($('#lgFormGroupInput').val())) {
                    lgFormGroupInput.subscribeInvalid('รูปแบบอีเมลไม่ถูกต้อง');
                }
            }
        }
        function SubscribeValidateMsg(txtMsg, status) {
            if (status === true) {
                lgFormGroupInput.subscribeSuccess(txtMsg);
            }
            if (status != true) {
                lgFormGroupInput.subscribeInvalid(txtMsg);
            }
        }
        function validateEmail(email) {

            return re.test(email);
        }
