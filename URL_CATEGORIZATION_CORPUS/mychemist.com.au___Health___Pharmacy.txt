        function GetTimestampForShortlive() {
            var d = new Date();
            var day = d.getDate();
            var hour = d.getHours();
            var minute = d.getMinutes();
            if (minute >= 0 && minute <= 15)
                minute = 1;
            else if (minute >= 16 && minute <= 30)
                minute = 2;
            else if (minute >= 31 && minute <= 45)
                minute = 3;
            else
                minute = 4;

            return day + hour + minute;
        }

            //test for existance
