	function validateWords(input, page) {
		var words = input.value.split(" ");
		var found = false;
		var inputText = input.value + " ";
		var invalidCharNode = document.getElementById("invalidCharHeader");
		var invalidWordNode = document.getElementById("invalidWordHeader");
