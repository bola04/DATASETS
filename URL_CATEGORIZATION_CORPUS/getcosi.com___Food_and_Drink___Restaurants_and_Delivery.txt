    jQuery(function($){
	/*
	 * Google Places Autocomplete
	 */
	var autocomplete;
	var pointsOfInterest = ["street_number", "street_address", "premise", "subpremise", "point_of_interest", "park", "train_station"];

	var boundsPaddingLookup = {
		"default" : { "lat": .01, "lng": .01 },
		"locality" : { "lat": .1, "lng": .1 }/* ,
		"postal_code" : { "lat": .01, "lng": .01 },
		"ward" : { "lat": .001, "lng": .001 },
		"political" : { "lat": .001, "lng": .001 } ,
		"sublocality" : { "lat": .0006, "lng": .0006 },
		"neighborhood" : { "lat": .0006, "lng": .0006 }
		*/
	};

	function initializeAutocomplete(searchElement) {
		// Create the autocomplete object, restricting the search to geographical location types.
		autocomplete = new google.maps.places.Autocomplete( (searchElement), { types: ['geocode'] });

		// When the user selects an address from the dropdown, populate the geography fields in the form.
		google.maps.event.addListener(autocomplete, 'place_changed', fillInAddress);
	}

	function padBounds(bounds, type) {
		var padding = boundsPaddingLookup["default"];

		if (boundsPaddingLookup.hasOwnProperty(type)) {
			padding = boundsPaddingLookup[type];
		}

		var lat1, long1, lat2, long2 = null;

		lat1 = bounds.getSouthWest().lat() - padding.lat;
		if (lat1 < -90) {
			lat1 = -90 + (-1 * lat1 - 90);
		}
		long1 = bounds.getSouthWest().lng() - padding.lng;
		if (long1 <= -180) {
			long1 = 180 - (-1 * long1 - 180);
		}

		lat2 = bounds.getNorthEast().lat() + padding.lat;
		if (lat2 > 90) {
			lat2 = 90 - (lat2 - 90);
		}
		long2 = bounds.getNorthEast().lng() + padding.lng;
		if (long2 > 180) {
			lat2 = -180 + (long2 - 180);
		}

		return new google.maps.LatLngBounds(
			new google.maps.LatLng(lat1, long1),
			new google.maps.LatLng(lat2, long2)
		);
	}

	function fillInAddress(place) {
		if ( place == undefined ) {
			place = autocomplete.getPlace();
		}

		var useBounds = true;
		var zoomLevel = "";
		var placeType = "";

        // Set Zoom Level if address is specific
        if ( place.address_components && place.address_components.length && place.address_components[0].types && place.address_components[0].types.length ) {
            placeType = place.address_components[0].types[0];
            if ( pointsOfInterest.indexOf(placeType) >= 0 ) {
            	useBounds = false;
            	zoomLevel = 15;
            }
        }

		// Set Lat/Lng or Bounding Box for maps
		if ( place.geometry ) {
			if ( place.geometry.viewport && useBounds) {
				var bounds = place.geometry.viewport,
					location = place.geometry.location;

				// Pad the bounds to show a little more depending on the type found
				bounds = padBounds(bounds, placeType);

				// Update the form fields with the new bounds
				document.getElementById("boundLat1").value = bounds.getSouthWest().lat();
				document.getElementById("boundLong1").value = bounds.getSouthWest().lng();
				document.getElementById("boundLat2").value = bounds.getNorthEast().lat();
				document.getElementById("boundLong2").value = bounds.getNorthEast().lng();

				if ( place.geometry.location ) {
					var location = place.geometry.location;

					document.getElementById("latitude").value = location.lat();
					document.getElementById("longitude").value = location.lng();
					$("#radius").value = 50;
				}
				
				document.getElementById("zoomLevel").value = zoomLevel;

				jQuery("#location").parents("form:first").submit();
			} else if ( place.geometry.location ) {
				var location = place.geometry.location;

				document.getElementById("latitude").value = location.lat();
				document.getElementById("longitude").value = location.lng();
				document.getElementById("zoomLevel").value = zoomLevel;
				document.getElementById("boundLat1").value = "";
				document.getElementById("boundLong1").value = "";
				document.getElementById("boundLat2").value = "";
				document.getElementById("boundLong2").value = "";
				$("#radius").value = 50;

				jQuery("#location").parents("form:first").submit();
			}
		}
	}
	google.maps.event.addDomListener(window, 'load', initializeAutocomplete(document.getElementById('location')));

	(function pacSelectFirst(input) {
		// store the original event binding function
		var _addEventListener = (input.addEventListener) ? input.addEventListener : input.attachEvent;

		function addEventListenerWrapper(type, listener) {
			// Simulate a 'down arrow' keypress on hitting 'return' when no pac suggestion is selected,
			// and then trigger the original listener.

			if (type == "keydown" || type == "blur") {
				var orig_listener = listener;
				listener = function (event) {
					var suggestion_selected = $(".pac-item-selected").length > 0;

					if (event.which == 13 && !suggestion_selected) {
						var simulated_downarrow = $.Event("keydown", {keyCode:40, which:40})
						orig_listener.apply(input, [simulated_downarrow]);
					}

					orig_listener.apply(input, [event]);
				};
			}

			// add the modified listener
			_addEventListener.apply(input, [type, listener]);
		}

		if (input.addEventListener){
			input.addEventListener = addEventListenerWrapper;
		} else if (input.attachEvent) {
			input.attachEvent = addEventListenerWrapper;
		}

	})(document.getElementById('location'));


	$("#location").siblings("button[type='submit']").on({
		click: function(event) {
			var form = $(this).parents("form"),
						firstResult = $('.pac-item:first').children();

			if ( firstResult.length ) {
				event.preventDefault();

				var	geocoder = new google.maps.Geocoder(),
					placeName = firstResult[1].textContent,
					placeAddress = firstResult[2].textContent,
					address = placeName,
					tempAddress = placeAddress.split(',');

				// Remove the Country for lookup
				if ( tempAddress.length >= 3 ) {
					for(var i=0; i<tempAddress.length-1; i++) {
						address += ", " + tempAddress[i];
					}
				} else {
					address += ", " + placeAddress;
				}

				geocoder.geocode({"address": address }, function(results, status) {

					if (status == google.maps.GeocoderStatus.OK) {
						form.find("input[name=location]").val(address);
						fillInAddress(results[0], false);
					}

					form.submit();
				});
			} else {
				form.submit();
			}
		}
	});
});
