   function nuevoAjax()
    { 
        var xmlhttp=false; 
        try 
        { 
            xmlhttp=new ActiveXObject("Msxml2.XMLHTTP"); 
        }
        catch(e)
        { 
            try
            { 
                xmlhttp=new ActiveXObject("Microsoft.XMLHTTP"); 
            } 
            catch(E) { xmlhttp=false; }
        }
        if (!xmlhttp && typeof XMLHttpRequest!='undefined') { xmlhttp=new XMLHttpRequest(); } 

        return xmlhttp; 
    }
    function selectAll(){
        var checkboxes = document.getElementsByTagName('input');
        for(var i=0; i<checkboxes.length; i++){
            var obj = checkboxes[i];
            if(obj){
                if(obj.name.indexOf('amigo_')!=-1)
                    obj.checked = true;
            }
        }
    }

    function enviar_quote(){
        quote      = document.getElementById('quote').value;
        checkboxes = document.getElementsByTagName('input');
        var par = "";
        for(var i=0; i<checkboxes.length; i++){
            if(checkboxes[i].name.indexOf('amigo_')!=-1) {
                if(checkboxes[i].checked == true) {
                    par += "amigo[" + checkboxes[i].value + "]=true&";
                }
            }
        }

        nom_1 = document.getElementById('nom_1').value;
        nom_2 = document.getElementById('nom_2').value;
        ema_1 = document.getElementById('ema_1').value;
        ema_2 = document.getElementById('ema_2').value;

        par+="noms=" + nom_1 + "|" + nom_2+"&";
        par+="emails=" + ema_1 + "|" + ema_2;

        ajax=nuevoAjax();
        ajax.open("POST", "/amigos_mail.php?lang=es",true);
        ajax.onreadystatechange=function() {
            if (ajax.readyState==4) {
                document.getElementById('enviaramigos').innerHTML = ajax.responseText
            }
        }
        ajax.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
        ajax.send("&quote="+quote + "&" + par );
    }

    function autoComplete(){
        var i = 0;
  
        for(var node; node = document.getElementsByTagName('input')[i]; i++){
            var type = node.getAttribute('type').toLowerCase();
            if(type == 'text'){
                node.setAttribute('autocomplete', 'off');
