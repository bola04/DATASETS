$(function() {
  $('[rel*=popup]').
    mouseover(function() {
    	var offset = $('#top').offset();
    	var padding = 0;
    	var addoffsetleft = 0;
    	if ($.browser.mozilla && $.browser.version > '1.9.0.0' && $.browser.mozilla && $.browser.version < '1.9.0.9')	addoffsetleft = 2;

    	var offset1 = $('#top-menu').offset();
      var menutop = offset1.top+$('#menu-0').height();
			var offset2 = $('#'+this.id).offset();
      var menuleft = offset2.left+addoffsetleft;

      if ($('#d'+this.id).width() + menuleft > offset.left+$('#top').width()) {
    	  menuleft = menuleft - $('#d'+this.id).width() + $('#'+this.id).width() - padding;  
      }
      
      $('#d'+this.id).css({top:menutop+"px", left:menuleft+"px"}).show();
      $('#'+this.id).addClass("amenuover");
            $('#d'+this.id).show();    
    }).
    mouseout(function() {
    	$('#'+this.id).removeClass("amenuover");

    });
