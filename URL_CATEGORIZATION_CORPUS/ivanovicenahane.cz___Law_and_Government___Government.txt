        function kaPrevMonth()
{
	var m = kaGetSelectedMonth.apply(this);
	var y = kaGetSelectedYear.apply(this);
	var minY = Math.min(getSelectOptionValues(this.find('#year')));

	if(m === 1){
		m = 12;
		y -= (y === minY ? 0 : 1);
	}else{
		m -= 1;
	}

	kaRedrawCalendar.apply(this, [y, m]);
}

function kaNextMonth()
{
	var m = kaGetSelectedMonth.apply(this);
	var y = kaGetSelectedYear.apply(this);
	var maxY = Math.max(getSelectOptionValues(this.find('#year')));

	if(m === 12){
		m = 1;
		y += (y === maxY ? 0 : 1);
	}else{
		m += 1;
	}

	kaRedrawCalendar.apply(this, [y, m]);
}

function getSelectOptionValues($select){
	var values = [];
	$select.find('opiton').each(function(){
		values.push(this.value);
	});

	return values;
}

var sShow=false;
var bShowDetail=false;
function ka_showdetail(id)
{
	oEl=document.getElementById(id);
	if (oEl)
	{
		if (sShow!=false)
		{
			cls.add(document.getElementById(sShow),'hide');
		}
		sShow=id;
		cls.remove(oEl,'hide');
		oEl.style.zIndex=9999;
	}
	else
		sShow=false;
}

function ka_hidedetail(id)
{
	oEl=document.getElementById(id);
	if (oEl && !bShowDetail)
	{
		cls.add(oEl,'hide');
		sShow=false;
		bShowDetail=false;
	}
}

function ka_hidedetails()
{
	if (sShow!=false)
	{
		oEl=document.getElementById(sShow);
		if (oEl)
		{
			cls.add(oEl,'hide');
			sShow=false;
			bShowDetail=false;
		}
	}
}

function kaRedrawCalendar(year, month){
	var $form = this;
	if(year !== undefined){
		$form.find('#year option[value="' + year + '"]').prop('selected', true);
	}
	if(month !== undefined){
		$form.find('#month option[value="' + month + '"]').prop('selected', true);
	}

	jQuery.ajax({
		url: 'modules/kalendar/ax_kalendar.php'
		, data: $form.serialize()
		, type: 'POST'
		, beforeSend: function(xhr, settings){}
		, success: function(data){
			if(data.length)
				$form.find('#kal_table table').replaceWith(data);
		}
		, complete: function(){}
	});
}

function kaGetSelectedMonth(){
	return parseInt(this.find('#month option:selected').val(), 10);
}
function kaGetSelectedYear(){
	return parseInt(this.find('#year option:selected').val(), 10);
}

function getParentForm(el) {
	return jQuery(el).closest('form');
}

jQuery(document).ready(function(){
	jQuery('.kalendar-4').live('submit', function(e){
		e.preventDefault();
		return false;
	});

	jQuery('.kalendar-4 .arrows input[type=image]:eq(0)').live('click', function(e){
		e.preventDefault();
		kaPrevMonth.apply(getParentForm(this));
	});

	jQuery('.kalendar-4 .arrows input[type=image]:eq(1)').live('click', function(e){
		e.preventDefault();
		kaNextMonth.apply(getParentForm(this));
	});

	jQuery('.kalendar-4 #month, .kalendar-4 #year').live('change', function(e){
		e.preventDefault();
		kaRedrawCalendar.apply(getParentForm(this));
	});
});

        function csPrevMonth()
	{
		var m = csGetSelectedMonth();
		var y = csGetSelectedYear();
		var minY = csGetMin('#communal_form #clndr_year');

		if(m === 1){
			m = 12;
			y -= (y === minY ? 0 : 1);
		}else{
			m -= 1;
		}

		csRedrawCalendar(y, m);
	}

	function csNextMonth()
	{
		var m = csGetSelectedMonth();
		var y = csGetSelectedYear();
		var maxY = csGetMax('#communal_form #clndr_year');

		if(m === 12){
			m = 1;
			y += (y === maxY ? 0 : 1);
		}else{
			m += 1;
		}

		csRedrawCalendar(y, m);
	}

	function csGetMin(select){
		return Math.min.apply(this, csGetSelectOptionValues(select));
	}
	function csGetMax(select){
		return Math.max.apply(this, csGetSelectOptionValues(select));
	}

	function csGetSelectOptionValues(select){
		var values = [];
		jQuery("option", jQuery(select)).each(function(){
			values.push(this.value);
		});

		return values;
	}

	function csRedrawCalendar(year, month){
		if(year !== undefined){
			jQuery('#communal_form #clndr_year option[value="' + year + '"]').prop('selected', true);
		}
		if(month !== undefined){
			jQuery('#communal_form #clndr_month option[value="' + month + '"]').prop('selected', true);
		}

		var formData = jQuery('#communal_form').serialize();

		jQuery.ajax({
			url: 'modules/communal_services/ax_calendar.php'
			, data: formData
			, type: 'GET'
			, beforeSend: function(xhr, settings){}
			, success: function(data){
				if(data.length)
					jQuery('#communal_form .cal_table table').replaceWith(data);
			}
			, complete: function(){}
		});
	}

	function csGetSelectedMonth(){
		return parseInt(jQuery('#clndr_month option:selected').val(), 10);
	}
	function csGetSelectedYear(){
		return parseInt(jQuery('#clndr_year option:selected').val(), 10);
	}

	jQuery(document).ready(function(){
		jQuery('#communal_form').live('submit', function(e){
			e.preventDefault();
			return false;
		});

		jQuery('#communal_form .arrows input[type=image]:eq(0)').live('click', function(e){
			e.preventDefault();
			csPrevMonth();
		});

		jQuery('#communal_form .arrows input[type=image]:eq(1)').live('click', function(e){
			e.preventDefault();
			csNextMonth();
		});

		jQuery('#clndr_month, #clndr_year, #clndr_place, #clndr_waste', jQuery('#communal_form')).live('change', function(e){
			e.preventDefault();
			csRedrawCalendar();
		});
	});
