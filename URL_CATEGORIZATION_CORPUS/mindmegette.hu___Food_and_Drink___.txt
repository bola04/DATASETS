 * @param int slaveId Zóna azonosítója

 * @returns void
 */
ZONES.setHasAd = function( slaveId, hasAd ) {
	for( var idx=0; idx < this.Blocks.length; idx++ ) {
		if( this.Blocks[idx].slaveId === slaveId ) {
			this.Blocks[idx].hasAd = hasAd;
			break;
		}
	}
	
	this.setBlockFetched( slaveId );

ZONES.setBlockFetched = function( slaveId ) {
	var numBlocks = this.Blocks.length;
	var fetchedBlocks = 0;
	
	for( var idx=0; idx < numBlocks; idx++ ) {
		if( this.Blocks[idx].slaveId === slaveId ) {
			this.Blocks[idx].fetched = true;
			fetchedBlocks++;
		} else {
			if ( this.Blocks[idx].fetched === true ) {fetchedBlocks++;}
		}
	}
	if (fetchedBlocks === numBlocks) {
		this.allBlockFetched();
	}
}

/**
 * Szerverről letölti a zónákat
 * 
 * @returns {void}
 */
ZONES.fetchBlocks = function() {
	
	var me = this;
	
	for( var idx=0; idx < this.Blocks.length; idx++ ) {
		
		var currentBlock = this.Blocks[ idx ];
		
		// AdOcean - SEND SLAVE ID
		ado.slave( currentBlock.slaveId , {
			myMaster: currentBlock.masterId, 
			onServerEmission:  function(){
				me.setHasAd( this.orgId, this._hasAd );
				me.showBlock( this.orgId );
			},
			onServerEmissionEmpty:  function(){
				me.setHasAd( this.orgId, this._hasAd );
			}
		});
	}
}

		});

    });
}
