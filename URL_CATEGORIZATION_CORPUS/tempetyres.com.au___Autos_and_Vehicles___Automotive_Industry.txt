		function setCollectionTime() {
			var valDate = $("#collectiondate").find("option:selected").text();
			var valDay = valDate.split(',')[0];
			
			$("#collectiontime").find('option').remove().end();
			
			var ddlT = document.getElementById('collectiontime');                      
			
			// Weekdays
			var minHour = 8;
			var maxHour = 17;			
			
			if (valDay == 'Saturday') {			// Saturday
				minHour = 8;
				maxHour = 15;			
			} else if (valDay == 'Sunday') {	// Sunday
				minHour = 9;
				maxHour = 13;			
			} 
							
			// Time Loop depends on the day
			for(var c=minHour;c<=maxHour;c++)
			{              
				var option = document.createElement('option');
				option.value = c+":00";
				option.text  = c+":00";
