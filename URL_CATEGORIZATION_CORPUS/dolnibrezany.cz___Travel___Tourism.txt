function ActualTime(params){
  var date=null;
  this.serverTime=null;
  this.clientTime=null;
  this.differentTimeInMilliseconds=0;
  this.srcElement=null;
  this.mainInterval=null;
            
  this.ActualTime=function(params){  
    if(params.srcElement)this.srcElement=document.getElementById(params.srcElement);    
        
    if(this.srcElement){
      this.setBeginTimesDifferent();  // COUNT DIFFERENT BETWEEN SERVER TIME AND CLIENT TIME
      this.startTimer();
    }
  }
  this.setBeginTimesDifferent=function(){
    var serverParsedTime=this.srcElement.innerHTML.replace(/ /g,"").split(":");
    if(serverParsedTime.length>=2){ // AT LEAST HOURS AND MINUTES MUST BE SET
      try{
        this.serverTime=new Date();
        this.serverTime.setHours(serverParsedTime[0]);
        this.serverTime.setMinutes(serverParsedTime[1]);
        if(serverParsedTime.length==3)this.serverTime.setSeconds(serverParsedTime[2]);

        this.clientTime=new Date();
            
        this.differentTimeInMilliseconds=this.serverTime.getTime()-this.clientTime.getTime();
      }
      catch(err){}
    }
  }
  this.startTimer=function(){
    var actualTime=this;
    var hours="00";
    var minutes="00";
    var seconds="00";
    this.mainInterval=setInterval(
      function(){
        date=new Date();
        date.setMilliseconds(date.getMilliseconds()+actualTime.differentTimeInMilliseconds);
        
        hours=date.getHours();        
        minutes=date.getMinutes();
        seconds=date.getSeconds();
        
        hours=(hours<10 ? "0"+hours : hours);        
        minutes=":"+(minutes<10 ? "0"+minutes : minutes);
        seconds=":"+(seconds<10 ? "0"+seconds : seconds);
        collectedDate=hours+minutes+seconds;
        actualTime.srcElement.innerHTML=collectedDate;
      }
    ,400);
  }
  this.ActualTime(params);

function getFromUrlByParam_codeFromWejlib(url,paramName){
  paramName=paramName.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
  var regexPattern="[\\?&]"+paramName+"=([^&#]*)";
  var regex=new RegExp( regexPattern );
  var results=regex.exec( (!url?window.location.href:url) );
  if(results==null)
    return "";
  else
    return results[1];
}

  pmc.make = function( inputId, options ) {
    var opts = options || {};
    var input = document.getElementById(inputId);

    if (input) {
      var isInputElement = input.tagName.toLowerCase() == "input";
      var cdStyle = window.getComputedStyle(input, null);
      var fontSize = parseFloat(cdStyle.getPropertyValue('font-size'));
      fontSize = Math.round(fontSize);
     // var fontStyle = cdStyle.getPropertyValue('font-style');
      var fontFamily = cdStyle.getPropertyValue('font-family');
      var fontWeight = cdStyle.getPropertyValue('font-weight');
      var textAlign = cdStyle.getPropertyValue('text-align');
      textAlign = textAlign == "center" ? "middle" : textAlign;
      var lineHeight = parseInt(cdStyle.getPropertyValue('line-height')) || fontSize*1.2142;
      var leftPadding = parseInt(cdStyle.getPropertyValue('padding-left')) || 0;
      var label = document.querySelector('label[for="' + input.id + '"]');
      var color, bgImagePath, textStr;

      var bTop = parseInt(cdStyle.getPropertyValue('border-top-width')) || 0;
      var bBottom = parseInt(cdStyle.getPropertyValue('border-bottom-width')) || 0;

      textStr = opts.text || label.textContent;

      textStr = pmc.lowercase ? ( pmc.lowercase === false ? textStr : textStr.toLowerCase() ) : textStr;
      fontStyle = pmc.italic ? ( pmc.italic === true ? "italic" : "normal" ) : "normal";
      color = pmc.color !== null ? pmc.color : ( opts.color || cdStyle.getPropertyValue('color') );
      color = color.replace("#","%23");
      textStr = encodeURI(textStr);

      if (fontFamily.indexOf(",") > -1) {
        fontFamily = fontFamily.split(",");
        fontFamily = fontFamily[0];
      }

      var yPos = 0;

      if (input.tagName.toLowerCase() == "textarea") {
        yPos = fontSize + parseInt(cdStyle.getPropertyValue('padding-top'));
      } else if (input.tagName.toLowerCase() == "input") {
        yPos = ((input.offsetHeight/2)+(fontSize/2)-((lineHeight-fontSize)/2)) - (bTop + bBottom);
      }

      yPos = Math.floor(yPos);

      svgStr = "data:image/svg+xml;charset=utf8,%3Csvg xmlns='http://www.w3.org/2000/svg' ";
      svgStr = svgStr + " width='" + input.offsetWidth + "' height='" + input.offsetHeight + "' text-anchor='" + textAlign + "' viewBox='0 0 " + input.offsetWidth + " " + input.offsetHeight + "'%3E%3Ctext x='" + ( textAlign == "middle" ? input.offsetWidth/2 : leftPadding ) + "' y='" + yPos + "' fill='" + color + "' font-family='" + fontFamily + "' font-weight='" + fontWeight + "' font-style='" + fontStyle + "' font-size='" + fontSize + "'%3E" + textStr + "%3C/text%3E%3C/svg%3E";

      bgImagePath = 'url("' + svgStr + '")';

      if ((isInputElement && input.value.length == 0) || (!isInputElement && (input.value.length==0 || input.innerHTML.length==0 || input.innerHTML == "&nbsp;"))) {
        var paramFromUrl=getFromUrlByParam_codeFromWejlib(null,"q");

          if (isInputElement) {
            input.value = paramFromUrl;
          } else {
            input.innerHTML = paramFromUrl;
          }

          input.style.backgroundImage = "url()";
