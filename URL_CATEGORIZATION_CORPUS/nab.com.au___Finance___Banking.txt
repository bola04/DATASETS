        determineValueType = function(currentElement, numberElement, isFeeValue, isPercentageFeeValue) {
            numberElement = currentElement.find('.number');
            var numberValue = numberElement.text();
            var description = currentElement.find('.description').text();
			var overrideSign = currentElement.find('.overrideSign').text();
            //Check for the word fee in the description. If it exists, we are assuming we'll place a $ symbol in front of the number, otherwise it will be %p.a. The only exception being reload fee for the traveller card
            
            //Override for new Sings provided now.
            if(overrideSign !==undefined && overrideSign.trim().length>0){
            	if(overrideSign ==='none'){
            		numberElement.html(numberValue);
            	}else if(overrideSign ==='$'){
            		numberElement.html('$'+ numberValue);
            	}else if(overrideSign ==='c'){
            		numberElement.html(numberValue+ 'c');
            	}else if(overrideSign ==='%'){
            		numberElement.html(numberValue + '<span>%</span>');
            	}else{
            		numberElement.html(numberValue+'<span>%<br/>p.a.</span>');
            	} 
            }else{
            
	            if ((description.search('fee') > -1) || (description.search('charge') > -1)) {
	                if ((description.search('reload fee') == -1)) {

	            }
	            else {
	                percentageSymbolHTML = '';
	            } 
	            if(numberElement.text().indexOf('%')===-1 && numberElement.text().indexOf('$')===-1){
	                numberElement.html(((isFeeValue)?'$':'') + numberValue + percentageSymbolHTML);
	            }
            
            }
