            items: [],
            update: function(url) {
                url = baseUrl + (url?url:'');
                $.get(

                    {},
                    function (data) {
                        ret.content = data.content;
                        ret.items = data.items;
                        ret.cantidad = data.cantidad;
                        ret.subtotal = data.subtotal;
                        ret.descuento = data.descuento;
                        ret.envio = data.envio;
                        ret.total = data.total;

                        $(document).trigger('cc:update');
                    }
                );
            },
            add: function (item, count) {
                var url = 'add/'+item;
                if (count) {
                    url += '/'+count;
                }
                this.update(url);
            },
            inc: function (item) {
                return this.add(item);
            },
            remove: function (item, cantidad) {
                var url = 'remove/'+item;
                if (cantidad) {
                    url += '/'+cantidad;
                }
                this.update(url);
            },
            dec: function (item) {
                return this.remove(item, 1);
            },
            clear: function () {
                this.update('clear');
            },
            cupon: function (cupon) {
                this.update();
            },
            applyCP: function(cp, cb) {
                if (cp.length < 4) {
                    return false;
                }
                url = baseUrl + '/cp/'+cp;
                $.get(

                    function (data) {

                        if (cb) {
                            cb(data.localidades);
                        }
                    }
