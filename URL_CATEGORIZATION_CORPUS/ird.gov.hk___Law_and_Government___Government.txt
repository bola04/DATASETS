  <script>
   function GetCookie(cookieName) {
        thisCookie = document.cookie.split("; ")
        for (i = 0; i < thisCookie.length; i++) {
          if (cookieName == thisCookie[i].split("=")[0]) {
            return thisCookie[i].split("=")[1];
          }
        }
        return "";
      }
      
      function SetCookie (name, value) {
        var argv = SetCookie.arguments;
        var argc = SetCookie.arguments.length;
        var expires = (2 < argc) ? argv[2] : null;
        var path = (3 < argc) ? argv[3] : null;
        var domain = (4 < argc) ? argv[4] : null;
        var secure = (5 < argc) ? argv[5] : false;
        document.cookie = name + "=" + escape (value) +
        ((expires == null) ? "" : ("; expires=" + expires.toGMTString())) +
        ((path == null) ? "" : ("; path=" + path)) +
        ((domain == null) ? "" : ("; domain=" + domain)) +
        ((secure == true) ? "; secure" : "");
      }
  </script>
  <script>
   function changeLanguage() {
        var langVersion = GetCookie('language');
        // check null
        // if null, go to /eng/welcome.htm
        if (langVersion==""){
          window.location.href = '/eng/welcome.htm';
          } else {
          if(langVersion == 'eng') {
            // eng
            // go to /eng/welcome.htm
            window.location.href = '/eng/welcome.htm';
          } else if (langVersion == 'chi'){
            // chi
            // go to /chi/welcome.htm
            window.location.href = '/chi/welcome.htm';
          } else {
            // chs
            // go to /chs/welcome.htm
            window.location.href = '/chs/welcome.htm';
          }
        }
      }
      changeLanguage();
  </script>
