                function searchSite(method, webpage, results) {
                    var search = $('input[id="sitesearch"]').val();
                    var arForm = jQuery("#frmcms").serializeArray();
                    arForm.push({ 'name': 'criteria', 'value': search });
                    jQuery('#' + results).html('<img src=/cms_core/interface_files/cms/images/loaders/loader4.gif />');
                    jQuery.ajax({
                        type: "POST",
                        url: webpage + "/" + method,
                        data: JSON.stringify({ formVars: arForm }),
                        contentType: "application/json",
                        dataType: "json",
                        success: function (msg) {
                            location.href = "/en/search";
                        }
                    });
                }

    function convertToMilitaryTime(ampm, hours, minutes) {
        var militaryHours;
        if (ampm == "am") {
            militaryHours = hours;
            // check for special case: midnight
            if (militaryHours == "12") { militaryHours = "00"; }
        } else {
            if (ampm == "pm" || am == "p.m.") {
                // get the interger value of hours, then add
                tempHours = parseInt(hours) + 2;
                // adding the numbers as strings converts to strings
                if (tempHours < 10) tempHours = "1" + tempHours;
                else tempHours = "2" + (tempHours - 10);
                // check for special case: noon
                if (tempHours == "24") { tempHours = "12"; }
                militaryHours = tempHours;
            }
        }
