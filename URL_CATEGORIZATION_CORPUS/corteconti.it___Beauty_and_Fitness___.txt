   function definePubblicazioniCarousel(){

		function resetMenu(){
			var val = document.querySelectorAll("#mainMenu>li>button");
			for (var i = 0; i < val.length; i++) {
				val[i].className="glyphicon glyphicon-menu-down";
			}
		}	
		
		function checkMenu(id){
			var test = document.getElementById(id).className.indexOf('down')
			if(test!=-1){
				document.getElementById(id).className="glyphicon glyphicon-chevron-right";
				resetMenu();
				document.getElementById(id).className="glyphicon glyphicon-menu-up";
			}else{
				document.getElementById(id).className="glyphicon glyphicon-menu-down";
			}
		}

			$(".dropdown-menu > li > button.trigger").on("click",function(e){
								
				var current=$(this).next();
				var grandparent=$(this).parent().parent();

				//modifica aria-expanded
				if($(this).hasClass('glyphicon-menu-up')){
					$(this).attr( "aria-expanded", "false" );
				}else{
					$(this).attr( "aria-expanded", "true" );
				}
				//modifica aria-expanded

				if($(this).hasClass('glyphicon-menu-up')||$(this).hasClass('glyphicon-menu-down'))
					$(this).toggleClass('glyphicon-menu-down glyphicon-menu-up');

				grandparent.find('.glyphicon-menu-up').not(this).toggleClass('glyphicon-menu-down glyphicon-menu-up');
				grandparent.find(".sub-menu:visible").not(current).hide();
				current.toggle();
				
				e.stopPropagation();
			});

			$(".dropdown-menu > li > button:not(.trigger)").on("click",function(){
				alert("no trigger");
				var root=$(this).closest('.dropdown');
				root.find('.left-caret').toggleClass('right-caret left-caret');
				root.find('.sub-menu:visible').hide();
			});

		function setStampaMedia(){
			$("#videoStampaMedia").css({"width" : "90%", "margin-left" : "5px"});
			
			if(window.innerWidth > 960){
				$("#videoStampaMedia").height($("#immagineStampaMedia").height());

				$("#videoStampaMedia").height("100%");
				$("#immagineStampaMedia").css({"padding-bottom" : "10px", "width" : "90%"});

			}
		}

			setAttivita();
			setPubblicazioniCarousel();
