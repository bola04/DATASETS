    formValidation: function(event, form)
    {
        var newmodel        = this;
        var form_id         = '#' + jQuery(form).attr('id');
        var data            = window.location.search.substring(1) + '&' + jQuery(form).serialize();
        var data_arr        = data.split('&');
        var unique_data_arr = this.getUniqueValues(data_arr);
        data                = unique_data_arr.join('&');

        newmodel.clearAllFormErrors(form_id);
        newmodel.showFormErrors([{'field': 'form', 'message': 'Processing...'}], form_id);

        jQuery.ajax({
            async:    false,
            cache:    false,
            url:      jQuery(form).attr('action'),
            type:     'GET',
            data:     data,
            dataType: 'json',
            success: function (result) {
                if (result && result.errors.length == 0)
                {
                    newmodel.clearAllFormErrors(form_id);
                    newmodel.processFormRedirect(result);
                }
                else
                {
                    AutoNewModel.showFormErrors(result.errors, form_id);
                }
            },
            error: function (XHR, textStatus) {
                AutoNewModel.showFormErrors([{'field': 'form', 'message': 'An error has occured please try again.'}], form_id);
                newmodel.sendAlert(XHR, textStatus, data, form_id);
            }
        });
    },

    "leave_behind"        - opens NM lead form in PopUp and lbrates as leavebehind
    "nm_leave_behind"     - opens NM lead form as leave behind and lbrates in PopUp
    "quoteoptions_newtab" - opens NM lead form in new Tab and lbrates as leavebehind

    processFormRedirect: function(data)
    {
        if (data.popup)
        {
            this.popupWindow(data.popup, 'newwindow');
        }
        if (data.newtab)
        {
            window.open(data.newtab, 'newtab');
        }
        if (data.redirect)
        {
            this.redirectUrl(data.redirect);
        }
    },

    popupWindow: function(url, name)
    {
        var ua = jQuery.getBrowser();
        var x  = (window.screenX || window.screenLeft || 0);
        var y  = (window.screenY || window.screenTop || 0);
        var j  = screen.width;
        var k  = screen.height;
        if ( ua.msie && ua.version.slice(0,3) == '6.0' )
        {
            window.open(url, name, "top=" + y + ", left=" + x + ", width=" + (j != null ? j : 1) + ",height=" + (k != null ? k : 1) + ",resizable=1,scrollbars=1,toolbar=1,menubar=1,location=1", false);
        }
        else
        {
            window.open(url, name, "top=" + y + ", left=" + x + ", width=" + (j != null ? j : 1) + ",height=" + (k != null ? k : 1) + ",resizable=1,scrollbars=1,toolbar=1,menubar=1,location=1", false).focus();
        }
    },

    redirectUrl: function(url)
    {
        var ua = jQuery.getBrowser();
        if ( ua.msie && ua.version.slice(0,3) == '6.0' )
        {
            window.location.href(url);
        }
        else
        {
            window.location.href = url;
        }
    }

            return;
        }

        /* Find Privacy Policy link in reverse order and copy its href. */

        var linkFound    = document.getElementById("privacyLinkFound");
        var linkNotFound = document.getElementById("privacyLinkNotFound");
        var privacyLinks = ['privacy', 'privacy policy', 'privacy & security', 'privacy statement'];

            var linktext = (links[i].innerText || links[i].textContent).toLowerCase().trim();
