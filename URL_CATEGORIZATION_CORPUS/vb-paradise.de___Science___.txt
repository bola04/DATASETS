	function formatDate(date) {
		// Heute ab 00:00:00.000 (Mitternacht)
		var today = new Date();
		today.setMilliseconds(0);
		today.setSeconds(0);
		today.setMinutes(0);
		today.setHours(0);

		var today_end = new Date();
		today_end.setMilliseconds(999);
		today_end.setSeconds(59);
		today_end.setMinutes(59);
		today_end.setHours(23);

		var tomorrow_end = new Date();
		tomorrow_end.setDate(tomorrow_end.getDate() + 1);
		tomorrow_end.setMilliseconds(999);
		tomorrow_end.setSeconds(59);
		tomorrow_end.setMinutes(59);
		tomorrow_end.setHours(23);


		// Gestern: Heute - 1 Tag
		var yesterday = new Date(today.getTime());
		yesterday.setDate(yesterday.getDate() - 1);

		var minutes = new String(date.getMinutes());
		if(minutes.length == 1)
			minutes = "0" + minutes;

		var hours = new String(date.getHours());
		if(hours.length == 1)
			hours = "0" + hours;

		// Wenn der Eintrag morgen erstellt wird: Morgen, Stunden:Minuten
		// (Erstelldatum größer als heutiges Tagesende aber kleiner als morgiges Tagesende)
		if (date > today_end && date < tomorrow_end) {
			var pattern = "Morgen, (0):(1)";
			return pattern.format(hours, minutes);
		}

		// Wenn der Eintrag heute erstellt wurde: <b>Heute</b>, Stunden:Minuten
		// (Erstelldatum größer als heutiger Tagesbeginn aber kleiner als Tagesende)
		if (date > today && date < today_end) {
			var pattern = "<b>Heute</b>, (0):(1)";
			return pattern.format(hours, minutes);
		}

		// Wenn der Eintrag gestern erstellt wurde: Gestern, Stunden:Minuten
		// (Erstelldatum größer als gestriger Tagesbeginn aber kleiner als der heutige)
		else if (date > yesterday && date < today) {
			var pattern = "Gestern, (0):(1)";
			return pattern.format(hours, minutes);
		}

		// Das Erstelldatum liegt weiter zurück: Wochentagsname, Monatstag. Monatsname Jahr, Stunden:Minuten
		else {
			var weekday = new Array("Sonntag", "Montag", "Dienstag", "Mittwoch", "Donnerstag", "Freitag", "Samstag");
			var month = new Array("Januar", "Februar", "März", "April", "Mai", "Juni", "Juli", "August", "September", "Oktober", "November", "Dezember");
			var pattern = "(0), (1). (2) (3), (4):(5)";

			return pattern.format(weekday[date.getDay()], date.getDate(), month[date.getMonth()], date.getFullYear(), hours, minutes);
		}
	}
