   var donationBtn = document.getElementsByClassName('donation-amount');
   var amountEl = document.getElementsByClassName('amount-input')[0];

   function btnListener(e){
      var formAmount = document.getElementById('form-amount');
      formAmount.value = e.target.getAttribute('data-amount');
      amountEl.parentNode.classList.remove('filled');
      amountEl.value = "";
   }
   function amountElHandler(e){
      var _this = e.target;
      if (_this.value < parseInt(_this.min) && _this.value !== "") {
         _this.classList.add('invalid');
         _this.nextElementSibling.classList.add('invalid');
         _this.parentNode.classList.remove('filled');
      } else if (_this.value === "") {
         _this.classList.remove('invalid');
         _this.parentNode.classList.remove('filled');
      } else {
         //if all is good...
         var donationBtn = document.getElementsByClassName('donation-amount');
         for (var i = 0; i < donationBtn.length; i++) {donationBtn[i].classList.remove('active');}
         _this.classList.remove('invalid');
         _this.nextElementSibling.classList.remove('invalid');
         _this.parentNode.classList.add('filled');
         document.getElementById('form-amount').value = _this.value;
      }
   }
   function amountElClick(e){
      var _this = e.target;
      _this.classList.remove('invalid');
      _this.nextElementSibling.classList.remove('invalid');
   }
   //Attach handlers
   for (var i = 0; i < donationBtn.length; i++) {
      donationBtn[i].addEventListener('click', btnListener);
   }
   amountEl.addEventListener('blur', amountElHandler);
   amountEl.addEventListener('keyup', amountElHandler);
   amountEl.addEventListener('click', amountElClick);
  </script>
