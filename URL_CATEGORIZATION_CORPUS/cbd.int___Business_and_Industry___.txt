    (function(window, document) { "use strict";

	var tokenLoaded;
	var authorizationFrame;
	var autoReload = true;
	var pendingOperation = false;

	if(window.location.pathname.lastIndexOf('/')>0) { // Fix cookie with invalid path
		document.cookie = ".ASPXAUTH_BSCMS=; expires=Thu, 01 Jan 1970 00:00:00 UTC; Path=" + window.location.pathname.substr(0, window.location.pathname.lastIndexOf('/'));
	}


	//==============================
	//
	//
	//==============================
	function setCookie(cname, cvalue, exdays) {

		if(!cvalue)
			exdays = -365; 

		var expires="" // session
		
		if(exdays!==undefined && exdays!==null) {
			var d = new Date();
			d.setTime(d.getTime() + (exdays*24*60*60*1000));
			expires = "expires="+d.toUTCString() + "; ";
		}

		document.cookie =  cname + "=" + cvalue + "; Path=/; " + expires;
	}

	//==============================
	//
	//
	//==============================
	function getCookie(cname) {

		var name = cname + "=";
		var ca = document.cookie.split(';');
		for(var i=0; i<ca.length; i++) {
			var c = ca[i];
			while (c.charAt(0)==' ') c = c.substring(1);
			if (c.indexOf(name) == 0) return c.substring(name.length,c.length);
		}
		return "";
	}

	//==============================
	//
	//
	//==============================
	function getQueryVariable(variable) {
		var query = window.location.search.substring(1);
		var vars = query.split('&');
		for (var i = 0; i < vars.length; i++) {
			var pair = vars[i].split('=');
			if (decodeURIComponent(pair[0]) == variable) {
				return decodeURIComponent(pair[1]);
			}
		}
	}

	//==============================
	//
	//
	//==============================
	function authenticate() {

		autoReload = false;
		pendingOperation = true;

		tokenReady(function() {

			var redirectUrl = getQueryVariable("returnUrl") || getQueryVariable("returnurl") || '/';

			try
			{
				window.history.replaceState({}, "The Convention on Biological Diversity", redirectUrl); //delete this entry from the history
				window.location.reload(true);
			}
			catch(e)
			{
				window.location = redirectUrl; //backward compatibility
			}
		}, true);
	}

	//==============================
	//
	//
	//==============================
	function signIn(defaultReturnUrl) {

		autoReload = false;
		pendingOperation = true;

		tokenReady(function(){

			if(defaultReturnUrl===true) //user current url;
				defaultReturnUrl = window.location.href.replace(/^\w+\:\/\/[a-zA-Z0-9@:\.]+/, '');

			var isAuthenticated = !!getCookie(".ASPXAUTH_BSCMS");
			var redirectUrl     = getQueryVariable("returnUrl") || getQueryVariable("returnurl") || defaultReturnUrl;

			if(!isAuthenticated) {

				var returnUrl = window.location.protocol +"//"+ window.location.hostname + "/user/authenticate?returnUrl="+encodeURIComponent(redirectUrl);

				redirectUrl = "https://accounts.cbd.int/signin?returnUrl=" + encodeURIComponent(returnUrl);
			}

			window.location = redirectUrl;

		}, true);
	}

	//==============================
	//
	//
	//==============================
	function signOut() {

		autoReload = false;
		pendingOperation = true;

		tokenReady(function(){

			setCookie('.ASPXAUTH_BSCMS', null);

			authorizationFrame.contentWindow.postMessage(JSON.stringify({ type : 'setAuthenticationToken', authenticationToken : null }), 'https://accounts.cbd.int')

			setTimeout(function(){ window.location = '/'; }, 50);

		}, true);
	}

	//==============================
	//
	//
	//==============================
	window.addEventListener('message', function(event) {

		if(event.origin!='https://accounts.cbd.int')
			return;

		var message = JSON.parse(event.data);

		if(message.type!='authenticationToken')
			return;

		var oldToken = getCookie(".ASPXAUTH_BSCMS") || '';
		var newToken = message.authenticationToken  || '';

		if(oldToken!=newToken) {
			setCookie('.ASPXAUTH_BSCMS', newToken)
	
			if(autoReload) {
				window.location.reload(true);
				return;
			}
		}

		tokenLoaded = true;

	});

	//==============================
	//
	//
	//==============================
	function initialize(frame){

		if(!authorizationFrame)
			authorizationFrame = frame;

		authorizationFrame.contentWindow.postMessage(JSON.stringify({ type : 'getAuthenticationToken' }), 'https://accounts.cbd.int');
	}

	//==============================
	//
	//
	//==============================
	function tokenReady(callback, ignorePendinOperation){

		if(/DesignMode=true/.test(getCookie("CmsVolatileSettings")||'') && /\?dev=1/.test(window.location.href))
			return;
		
		if(!tokenLoaded) { // frame
			setTimeout(function() { tokenReady(callback, ignorePendinOperation); }, 50);
			return;
		}

		if(!pendingOperation || ignorePendinOperation)
			callback();
	}


	//==============================
	//
	// EXPORTS
	//
	//==============================
	window._account = {
		initialize : initialize,
		authenticate : authenticate,
		signIn : signIn,
		signOut : signOut,
		tokenReady : tokenReady
	}


})(window, document);
