            var link = $(this).attr('href');
            if (typeof (link) != 'undefined') {
                if (link.match(pdfRegEx)) {
                    var pageURL = window.location.pathname;
                    ga('send', 'event', 'PDF Download', pageURL, link);
                }
                if (link.match(externalRegEx)) {
                    var pageURL = window.location.pathname;
                    ga('send', 'event', 'External Link', pageURL, link);
                }
            }

            var link = $(this).context.innerText;

                    actionText = "Collapsed";
                } else {
                    actionText = "Expanded";
                }
                var linkText = $(this).text().trim();
                var pageURL = window.location.pathname;
