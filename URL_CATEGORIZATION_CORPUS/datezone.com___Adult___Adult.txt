        // Ukrywanie błędu
        function ukryjBlad(nazwa_etykiety, tresc_bledu, wyswietlBlad) {
            var wyswietl_blad = true;

            // Jeśli nie podano nazwy etykiety (ukrywanie wszystkich błedów)
            if (typeof nazwa_etykiety == "undefined" || nazwa_etykiety == null) {
                var visible_label = $('[data-form-label]:visible');

                if (visible_label.length >= 1) {
                    visible_label.slideUp(100, function () {
                        $('.form-group').removeClass('has-danger');
                        $(this).text('');

                        if ($.isFunction(wyswietlBlad)) {
                            wyswietlBlad();
                        }
                    });
                } else {
                    if ($.isFunction(wyswietlBlad)) {
                        wyswietlBlad();
                    }
                }
            } else if (typeof nazwa_etykiety != "undefined" && nazwa_etykiety != null) {
                var visible_label = null;

                $.each($('[data-form-label]'), function () {
                    if ($(this).is(':visible')) {
                        visible_label = $(this);
                        if (visible_label.text() == tresc_bledu) {
                            wyswietl_blad = false;
                        }
                        return false;
                    }
                });

                if (visible_label == null) {
                    if (wyswietl_blad && $.isFunction(wyswietlBlad)) {
                        wyswietlBlad();
                    }
                } else {
                    if (wyswietl_blad) {
                        visible_label.slideUp(100, function () {
                            $('.form-group').removeClass('has-danger');
                            $(this).text('');

                            if (wyswietl_blad && $.isFunction(wyswietlBlad)) {
                                wyswietlBlad();
                            }
                        });
                    }
                }
            }
        }

                wyswietlEtykieteBledu(input_name, 'Nie wybrano płci');
            } else {
                ukryjBlad(input_name);
            }

            return no_error;
        }

        // Sprawdzanie daty urodzenia partnera
        function sprawdzDatyParner() {
            var no_error = true;
            var input_name = 'day_para';
            var inputs_data_partner = $('#form-main-register #data_partner select');

            var input_day = $('[name="day_para"]').val();
            var input_month = $('[name="month_para"]').val();
            var input_year = $('[name="year_para"]').val();

            if (inputs_data_partner.is(':visible')) {
                inputs_data_partner.each(function () {
                    input_name = $(this).attr('name');
                    if ($(this).val() == null) {
                        wyswietlEtykieteBledu(input_name, 'Nie wybrano daty urodzenia');
                        no_error = false;
                    }
                });

                if (no_error) {
                    if (!checkData(input_day, input_month - 1, input_year)) {
                        wyswietlEtykieteBledu(input_name, 'Nieprawidłowa data urodzenia');
                        no_error = false;
                    }
                }

            }



            if (no_error && input_name != null) {
                ukryjBlad(input_name);
            }

            return no_error;
        }

        // Ogólne sprawdzanie daty i płci w przypadku płci konta "para"
        function sprwadzDatyPlcie() {
            var no_error = false;

            if (!sprawdzPlecPara()) {
                return false;
            } else if (!sprawdzDaty()) {
                return false;
            } else if (!sprawdzPlecParaPartner()) {
                return false;
            } else if (!sprawdzDatyParner()) {
                return false;
            } else {
                return true;
            }

            return no_error;
        }

        // Sprawdzanie miejscowości

            var no_error = true;
            var input_name = null;
            var inputs_miejscowosc = $('#form-main-register input[name="adress_show"]');
            var inputs_miejscowosc_select = $('#form-main-register select[name="kraj"], #form-main-register select[name="wojewodztwo"], #form-main-register [name="miasto"]');

            if (inputs_miejscowosc.is(':visible')) {

                }
            } else {
                inputs_miejscowosc_select.each(function () {

                    if ($(this).is(':visible') && ($(this).val() == null || $(this).val() == '')) {
                        wyswietlEtykieteBledu(input_name, 'Nie wybrano miejscowości');
                        no_error = false;
                        return false;
                    }
                });
            }

            if (no_error && input_name != null) {
                ukryjBlad(input_name);
            }

            return no_error;
        }

        function walidacja(prev, active) {
            switch (prev) {
                case 'login_or_email':
                    sprawdzPrzypomnienieHasla();
                    break;
                case 'login':
                    sprawdzLoginLogowanie();
                    break;
                case 'plec':
                    sprawdzCzyWybranoPlec();
                    break;
                case 'registration_login':
                    sprawdzLogin();
                    break;
                case 'password':
                    $('#form-login input[name="password"]').is(':visible') ? sprawdzHasloLogowanie() : sprawdzHaslo();
                    break;
                case 'email':
                    sprwadzMail();
                    break;
                case 'plec_para':
                    sprawdzPlecPara();
                    break;
                case 'day':
                case 'month':
                case 'year':
                    var select_date = (['day', 'month', 'year']);
                    if (!($.inArray(active, select_date) != -1 && $.inArray(prev, select_date) != -1)) {
                        sprawdzDaty();
                    }
                    break;
                case 'plec_para_partner':
                    sprawdzPlecParaPartner();
                    break;
                case 'day_para':
                case 'month_para':
                case 'year_para':
                    var select_date = (['day_para', 'month_para', 'year_para']);
                    if (!($.inArray(active, select_date) != -1 && $.inArray(prev, select_date) != -1)) {
                        sprawdzDatyParner();
                    }
                    break;
                case 'adress':
                case 'adress_show':
                case 'kraj':
                case 'wojewodztwo':
                case 'miasto':
                    var select_date = (['adress', 'adress_show', 'kraj', 'wojewodztwo', 'miasto']);
                    if (!($.inArray(active, select_date) != -1 && $.inArray(prev, select_date) != -1)) {
                        sprwadzMiejscowosc();
                    }
                    break;
            }
        }

        // Przechwytywanie kliknięć dla walidacji
        var active_selected_input = null;
        var prev_selected_input = null;
        $(document).on('click', function (event) {

            if ($(event.target).is(':not([type="submit"])')) {
                if (active_selected_input != null) {
                    prev_selected_input = active_selected_input;
                }
                active_selected_input = $(event.target).attr('name') || null;
                if (active_selected_input == null) {
                    active_selected_input = $(event.target).find('input').attr('name') || null;
                }

                if ($(event.target).is('option')) {
                    active_selected_input = $(event.target).parents('select').attr('name') || null;
                }

                if (prev_selected_input != null && prev_selected_input != active_selected_input) {
                    walidacja(prev_selected_input, active_selected_input);
                }
            }
        });


    }

    function zmiana_kraju() {

        $('[name="wojewodztwo"]').html('');
        $('[name="wojewodztwo"]').append('<option data-sortable="0" selected disabled>Region</option>');
        $('[name="wojewodztwo"]').prop('selectedIndex', 0);

        $('[name="miasto"]').changeElementType('select');
        $('[name="miasto"]').removeAttr('type');
        $('[name="miasto"]').removeAttr('placeholder');
        $('[name="miasto"]').html('');
        $('[name="miasto"]').append('<option data-sortable="0" selected disabled>Miejscowość</option>');
        $('[name="miasto"]').prop('selectedIndex', 0);
        $('[name="miasto"]').prop('disabled', true);

        var kraj = $('[name="kraj"]').val();
        var request = $.get('https://www.datezone.com/index.php?action=rejestracja_ajax&mode=wczytaj_wojewodztwa_dla_kraju&kraj=' + kraj);
        request.done(function (response) {
            try {
                response = JSON.parse(response);
            } catch (e) {
                response = null;
            }
            if (response != null) {
                $('[name="wojewodztwo"]').show();
                $('[name="wojewodztwo"]').prop('disabled', false);
                $('[name="wojewodztwo"]').html('');
                $('[name="wojewodztwo"]').append('<option data-sortable="0" selected disabled>Region</option>');
                $.each(response, function (index, wojewodztwo) {
                    $('[name="wojewodztwo"]').append('<option data-sortable="1" value=\"' + index + '\">' + wojewodztwo + '</option>');
                });

                $('[name="wojewodztwo"] option').each(function () {
                    if ($(this).text() == "") {
                        $(this).attr("selected", true);
                        zmiana_wojewodztwa();
                        return true;
                    }
                });

                $('[name="miasto"]').changeElementType('select');
                $('[name="miasto"]').removeAttr('type');
                $('[name="miasto"]').removeAttr('placeholder');
            } else {
                $('[name="wojewodztwo"]').hide();
                $('[name="miasto"]').changeElementType('input');
                $('[name="miasto"]').attr('type', 'text');
                $('[name="miasto"]').attr('placeholder', 'Miejscowość');
                $('[name="miasto"]').removeClass('placeholder');
                $('[name="miasto"]').prop('disabled', false);

            }
            self();
        });
    }


    // Event: zmiana kraju
    $('[name="kraj"]').on('change', function () {
        zmiana_kraju();
    });


    function zmiana_wojewodztwa() {
        functions_placeholder();
        var wojewodztwo = $('[name="wojewodztwo"]').val();
        var request = $.get('https://www.datezone.com/index.php?action=rejestracja_ajax&mode=wczytaj_miasta_dla_wojewodztwa&wojewodztwo=' + wojewodztwo);
        request.done(function (response) {
            response = JSON.parse(response);
            $('[name="miasto"]').prop('disabled', false);
            $('[name="miasto"]').html('');
            $('[name="miasto"]').append('<option data-sortable="0" selected disabled>Miejscowość</option>');
            $.each(response, function (index, miejscowosc) {
                $('[name="miasto"]').append('<option data-sortable="1" value=\"' + miejscowosc.miasto + '\">' + miejscowosc.miasto + '</option>');
            });

            $('[name="miasto"] option').each(function () {
                if ($(this).text() == "") {
                    $(this).attr("selected", true);
                    return true;
                }
            });

            $('[name="miasto"]').prop('disabled', false);
            self();
        });
    }

    // Event: zmiana województwa

        zmiana_wojewodztwa();
    });
