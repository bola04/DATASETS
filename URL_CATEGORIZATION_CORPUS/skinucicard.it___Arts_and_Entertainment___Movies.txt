   function symfonyRegisterDatepickerType() {
      if (Modernizr.inputtypes.date && Modernizr.touchevents) {
        $(".js_datepicker").parent().addClass('hidden');
        $('.native_datepicker').each(function() {
          $(this).attr('type','date');
          $(this).attr('required','required');
        });

        //set date for each child, get fallback date from bootstrap-datepicker and fill related native-datepicker
        $('.native_datepicker').each(function() {
          var attrName = $(this).attr('name').replace(/native_datepicker/g, 'js_datepicker');
          var fallbackDt = $('input[name="' + attrName + '"]');
          var date = moment(fallbackDt.val(), "DD/MM/YYYY");
          $(this).val(date.format("YYYY-MM-DD"));
        });
      }
      else {
        //hide native-datepicker
        $(".native_datepicker").addClass('hidden');

        //after page is rendered update all native datepicker with date picked up from js datepicker
        $('.js_datepicker').each(function() {
          var attrName = $(this).attr('name').replace(/js_datepicker/g, 'native_datepicker');
          var fallbackDt = $('input[name="' + attrName + '"]');
          var date = moment($(this).val(), "DD/MM/YYYY");
          $(fallbackDt).val(date.format("YYYY-MM-DD"));
        });

        //initialize bootstrap-datepicker
        $(document).on('focus', ".js_datepicker", function() {
          var defaultConfig = {
              language: "it",
              format: "dd/mm/yyyy",
              autoclose: true,
              startView: 2,
              disableTouchKeyboard: true
          };
          $(this).parent('.input-group-date').datepicker(Object.assign({}, defaultConfig, JSON.parse($(this).attr('data-config'))));
        });
      }
    };

    $(document).ready(function() {
