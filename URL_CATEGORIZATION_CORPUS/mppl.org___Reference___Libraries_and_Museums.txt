    init   : function()
    {
      this.cacheDom();
      this.bindEvents();
      this.data.dataArray.sort(this.sortData);
      this.castData();
      this.hideResults();
    },

    cacheDom : function()
    {
      this.input = document.getElementById("s");
      this.displayResults = document.getElementById("displayResults");
      this.catalogSelect = document.getElementById("catalogSelect");
    },

    bindEvents : function()
    {

      this.input.addEventListener("keyup",  this.searchData.bind(this));

    },

    castData : function()
    {

      for(var i = 0; i < this.data.dataArray.length; i++)
        {
          this.data.dataArray[i].SEARCH_TERM = this.data.dataArray[i].SEARCH_TERM.toString();

        }
    },

    clearData : function()
    {
      this.displayArray.length = 0;
    },


    hideResults : function()
    {
        window.onclick = function(){
            this.displayResults.style.display = "none";
        }
    },

    searchType : function()
    {
        if(this.catalogSelect.value == "website")
        {
            return true;
        }
    },

    searchData : function()
    {

      if(this.searchType())
      {
        return;
      }

      this.clearData();

      var inputVal = this.input.value.toLowerCase();

      for(var i = 0; i < this.data.dataArray.length; i++)
        {

          if(this.data.dataArray[i].SEARCH_TERM.indexOf(inputVal) === 0 && inputVal.length !== 0)
            {
              this.displayArray.push(this.data.dataArray[i].SEARCH_TERM);
            }
        }

      this.render();
    },

    sortData : function(a,b)
    {
      if (a.HITS > b.HITS)
      {
        return -1;
      }
      else if (a.HITS < b.HITS)
      {
        return 1;
      }
      else
      {
        return 0;
      }

    },


    render : function()
    {
      this.displayResults.innerHTML = "";

      if(this.displayArray.length === 0)
      {
        this.displayResults.style.display = "none";
      }
      else
      {
        this.displayResults.style.display = "block";
      }

      for(var i = 0; i < this.displayArray.length; i++)
        {
          var textNode = document.createTextNode(this.displayArray[i]);
          var li = document.createElement("li");
          var a = document.createElement("a");
          a.href = "http://mtpros.ent.sirsi.net/client/en_US/default/search/results?qu=" + this.displayArray[i];
          li.appendChild(textNode);
          a.appendChild(li);
          this.displayResults.appendChild(a);

          if(i === 9)
          {
            return;
          }
        }
    }
