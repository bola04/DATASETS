       jQuery(document).ready( function($){
          var i_now = 0;
          var i_max = -1
          var o_arr = [];

          var max_h = 20;
          var objs = $('.kanko_box');
         	// ランダム表示用に退避
         	$.each(objs,function(){
         	  o_arr.push($(this));
         	  if( max_h < $(this).height() ) max_h = $(this).height();
            i_max++;
          });
          // 親ボックスの高さを整える
          $('#slider').css('min-height',max_h+1);
          $('#slider').css('height',max_h+1);
        	o_arr.sort(function() {
        		return Math.random() - Math.random();
        	});
        	// 表示
        	$(o_arr[i_now]).show();
          
          $('#kankojohodiv .control-prev').click(function(){
            $(o_arr[i_now]).fadeOut('slow',function(){
              if( i_now == i_max ) i_now = -1;
              i_now++;
              $(o_arr[i_now]).fadeIn('slow');
            });
            return false;
          });
          $('#kankojohodiv .control-next').click(function(){
            $(o_arr[i_now]).fadeOut('slow',function(){
              if( i_now == 0 ) i_now = i_max + 1;
              i_now--;
              $(o_arr[i_now]).fadeIn('slow');
            });
            return false;
          });
        });
