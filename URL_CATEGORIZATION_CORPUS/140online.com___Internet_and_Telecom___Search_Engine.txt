                var active_color = '#000';
                // Colour of user provided text 
                var inactive_color = '#4d4f52';
                // Colour of default text/** * No need to modify anything below this line */
                $(document).ready(function () {
                    $("input.textbooks").css("color", inactive_color);
                    var default_values = new Array();
                    $("input.textbooks").focus(function () {
                        if (!default_values[this.id]) {
                            default_values[this.id] = "اسم شركة / اسم صنف / علامة تجارية ";
                        }
                        if (this.value == default_values[this.id]) {
                            this.value = '';
                            this.style.color = active_color;
                        }
                        $(this).blur(function () {
                            if (this.value == '') {
                                this.style.color = inactive_color;
                                this.value = default_values[this.id];
                            }
                        });
                    });
                });

     $(document).ready(function () {



                     $.widget("custom.threecolumnautocomplete", $.ui.autocomplete, {

                         //going to extend the AutoComplete widget by customizing renderMenu and renderItems
                         _renderMenu: function (ul, items) {
                             var self = this;



                             ul.append("<table class='customautocomplete' cellpadding='0'>\
                            <tbody  class='catstr'><tr><td>\
                                 <div class='border div44 divclas'><th4 class='th1'>تصنيفات</th4><ul class='clas'></ul></div>\

                            </td></tr></tbody>\
                        </table>");

                             $.each(items, function (index, item) {                                 
                                 $('ul.clas li:even').addClass('even');
                                 $('ul.prod li:even').addClass('even');
                                 $('ul.brnd li:even').addClass('even');
                                 $('ul.comp li:even').addClass('even');
                                 //$('li:odd').addClass('odd');
                                 self._renderItem(ul.find("table tbody"), item);
                             });
                         },

                         _renderItem: function (table, item) {

                             var lang = "";



                             if (item.l.toLowerCase() == 'e')
                                 lang = "En/";



                             if (item.cat.toLowerCase() == "comp") {
                                 return $("<li></li>")
                .data("item.autocomplete", item)
                .append("<a href=\"http://www.140online.com/company/" + lang + item.id + "/" + item.label + "\">" + item.nm + "</a>")
                .appendTo(table.find("ul.comp"));
                             }
                             if (item.cat.toLowerCase() == "prod") {
                                 return $("<li></li>")

                .appendTo(table.find("ul.prod"));
                             }

                             if (item.cat.toLowerCase() == "clas") {
                                 return $("<li></li>")
                .data("item.autocomplete", item)
                .append("<a href=\"http://www.140online.com/class/" + lang + item.id + "/" + item.label + "\">" + item.label + "</a>")
                .appendTo(table.find("ul.clas"));
                             }

                             if (item.cat.toLowerCase() == "brnd") {
                                 return $("<li></li>")

                .append("<a href=\"http://www.140online.com/brand/" + lang + item.id + "/" + item.label + "\">" + item.label + "</a>")
                .appendTo(table.find("ul.brnd"));
                             }





                             // default if a category was not matched, just append a row to the containing table
                             return $("<tr></tr>")
                .data("item.autocomplete", item)
                .append("<td colspan='4'>" + item.label + "</td>")
