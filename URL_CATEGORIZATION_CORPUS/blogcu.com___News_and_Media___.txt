                linux: [/(linux)/i],
                macos: [/(mac\sos\sx)\s?([\w\s\.]+\w)*/i, /(macintosh|mac(?=_powerpc)\s)/i],

                iphone: [/^((?!windows).)*\((ip[honed|\s\w*]+);.+(apple)/i],
                other: [/(windows\sce)(.*)(iemobile)/i],
                windowsphone: [/(windows\sphone(?:\sos)*|windows\smobile)[\s\/]?([ntce\d\.\s]+\w)/i]
            }
        };

        var matchPatterns = function (patterns, text) {
            var i, pattern;
            for ( pattern in patterns ) {
                if ( patterns.hasOwnProperty(pattern) ) {
                    for ( i = 0; i < patterns[pattern].length; i++ ) {
                        if ( patterns[pattern][i].test(text) ) {
                            return pattern;
                        }
                    }
                }
            }
            return false;
        };

        var ua = userAgent || ((window && window.navigator && window.navigator.userAgent) ? window.navigator.userAgent : '');

        this.getUserAgent = function () {
            return ua;
        };

        this.getOS = function () {
            var os = matchPatterns(REGEX_PATTERNS.mobile, ua) || matchPatterns(REGEX_PATTERNS.desktop, ua) || 'other';
            return os;
        };

        return this;
    };

    function setCookie(c_name,value,exmins)
    {
        var exdate=new Date();
        exdate.setMinutes(exdate.getMinutes() + exmins);
        var c_value=escape(value) + ((exmins==null) ? "" : "; expires="+exdate.toUTCString());

            y=ARRcookies[i].substr(ARRcookies[i].indexOf("=")+1);
            x=x.replace(/^\s+|\s+$/g,"");
            if (x==c_name)
            {
                return unescape(y);
            }
        }
    }

    function getScrollY() {
        scrOfY = 0;
        if( typeof( window.pageYOffset ) == "number" ) {
            scrOfY = window.pageYOffset;
        } else if( document.body && ( document.body.scrollLeft || document.body.scrollTop ) ) {
            scrOfY = document.body.scrollTop;
        } else if( document.documentElement && ( document.documentElement.scrollLeft || document.documentElement.scrollTop ) ) {
            scrOfY = document.documentElement.scrollTop;
        }
        return scrOfY;
    }
