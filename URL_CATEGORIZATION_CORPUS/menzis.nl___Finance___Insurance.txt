       (function ($) {

        var pushTopTaak = true;     
        var lastTopTaak = null;
        
        $(window).on("popstate", function (e) {            

            if(e.originalEvent.state === null || "False" === "False"){
                return;
            }

            try{
                var splitted = e.originalEvent.state.split("|");
                if(splitted.length !== 2){
                    return;
                }

                // zoeken vorige element.
                var prevTab = $("#" + splitted[0]).find("[aria-controls=" + splitted[1] + "][role='tab']");         

                pushTopTaak = false;   
            
                // tab/accordion bepalen.
                var itemIndex = $("#" + splitted[0]).closest(".toptaken").find(".resp-tabs-list").css("display") === "none" ? 1 : 0;                
                $(prevTab[itemIndex]).trigger("click");        
            }
            catch(err){ }

            pushTopTaak = true;
        });


        function getDefaultTab(){
            var tabIndex = 1;

            // backbutton, check history.
            if( history.state !== null && history.state.indexOf("_topTaken|") !== -1) {
                tabIndex = parseInt(history.state.substr(history.state.lastIndexOf("-") + 1)) + 1;
            }

            return tabIndex;
        }


        $(document).ready(function () {
                        
            tabsAccordeons($('.toptaken'), {
                type: 'default', //Types: default, vertical, accordion           
                width: 'auto', //auto or any width like 600px
                fit: true,   // 100% fit in a container
                closed:  false , // Start closed if in accordion view use: 'accordion'
                defaultActiveItemindex: getDefaultTab(),
                activate: function (event) { // Callback function if tab is switched
                    var $tab = $(this);
                    var $info = $('#tabInfo');
                    var $name = $('span', $info);
                    $name.text($tab.text());
                    $info.show();
                        
                    if( "False" === "True"){
                            if(pushTopTaak === true){
                                var idTaken = $tab.closest(".toptaken").attr('id');
                                var tabAria = $tab.attr('aria-controls');

                                if(tabAria !== lastTopTaak){
                                    history.pushState(idTaken + "|" + tabAria, null, null);
                                    lastTopTaak = tabAria;
                                }
                            }
                        }
                    }
            });
        });
    })(jQuery);
