        //カルーセル
        var total;
        var nowPos = 0;
        var interval = 7000;
        $(function() {
            total = $("#mainvisual .carousel li").length;
            $("#mainvisual .carousel li").each(function(i) {
                this.id = i;
                this.pos = i;
                if (this.pos >= (total - 2)) this.pos -= total;
                $(this).css({
                    left: this.pos * 100
                });
                if (this.pos == 0) {
                    $("img", this).stop(true).css({
                        opacity: 1
                    });
                    $(this).css("zIndex", "999");
                } else {
                    $("img", this).stop(true).css({
                        opacity: 0
                    });
                    $(this).css("zIndex", "auto");
                }
            })
            for (var i = 0; i < total; i++) {
                var html = '<li><a href="javascript:void(0)" class="opacityover">' + i + '</a></li>';
                var obj = $(html);
                obj[0].id = i;
                obj.bind("click", function() {
                    jump(this.id);
                });
                $("#mainvisual .currentarea .current").append(obj)
            }
            checkCurrent();
            $("#wrapper").stop(true).animate({
                opacity: 1
            }, interval, "linear", next);
            $("#mainvisual").each(function() {
                $(this).bind("mouseover", function() {
                    $("#wrapper").stop(true);
                });
                $(this).bind("mouseout", function() {
                    $("#wrapper").stop(true).animate({
                        opacity: 1
                    }, interval, "linear", next);
                });
            })
        });

        function next() {
            nowPos += 1;
            if (nowPos >= total) nowPos = 0;
            $("#mainvisual .carousel li").each(function(i) {
                if (this.id == nowPos) {
                    $(this).css("zIndex", "999");
                    $(this).stop(true).animate({
                        left: 100
                    }, 0);
                    $(this).stop(true).animate({
                        left: 0
                    }, 1500, "easeInOutQuart");
                    $("img", this).stop(true).animate({
                        opacity: 1
                    }, 1500, "easeInOutQuart");
                } else {
                    $(this).css("zIndex", "auto");
                    $(this).stop(true).animate({
                        left: -100
                    }, 1500, "easeInOutQuart");
                    $("img", this).stop(true).animate({
                        opacity: 0
                    }, 1500, "easeInOutQuart");
                }
            });
            checkCurrent();
            $("#wrapper").stop(true).animate({
                opacity: 1
            }, interval, "linear", next);
        }

        function prev() {
            nowPos -= 1;
            if (nowPos < 0) nowPos = total - 1;
            $("#mainvisual .carousel li").each(function(i) {
                if (this.id == nowPos) {
                    $(this).css("zIndex", "999");
                    $(this).stop(true).animate({
                        left: -100
                    }, 0);
                    $(this).stop(true).animate({
                        left: 0
                    }, 1500, "easeInOutQuart");
                    $("img", this).stop(true).animate({
                        opacity: 1
                    }, 1500, "easeInOutQuart");
                } else {
                    $(this).css("zIndex", "auto");

                        left: 100
                    }, 1500, "easeInOutQuart");
                    $("img", this).stop(true).animate({
                        opacity: 0
                    }, 1500, "easeInOutQuart");
                }
            });
            checkCurrent();
            $("#wrapper").stop(true).animate({
                opacity: 1
            }, interval, "linear", next);
        }

        function jump(num) {
            var n = num - nowPos;
            if (n > 0) {
                $("#mainvisual .carousel li").each(function(i) {
                    if (this.id == num) {
                        $(this).css("zIndex", "999");
                        $(this).stop(true).animate({
                            left: 100
                        }, 0);
                        $(this).stop(true).animate({
                            left: 0
                        }, 1500, "easeInOutQuart");
                        $("img", this).stop(true).animate({
                            opacity: 1
                        }, 1500, "easeInOutQuart");
                    } else {
                        $(this).css("zIndex", "auto");
                        $(this).stop(true).animate({
                            left: -100
                        }, 1500, "easeInOutQuart");
                        $("img", this).stop(true).animate({
                            opacity: 0
                        }, 1500, "easeInOutQuart");
                    }
                });
            } else if (n < 0) {
                $("#mainvisual .carousel li").each(function(i) {
                    if (this.id == num) {
                        $(this).css("zIndex", "999");
                        $(this).stop(true).animate({
                            left: -100
                        }, 0);
                        $(this).stop(true).animate({
                            left: 0
                        }, 1500, "easeInOutQuart");
                        $("img", this).stop(true).animate({
                            opacity: 1
                        }, 1500, "easeInOutQuart");
                    } else {
                        $(this).css("zIndex", "auto");
                        $(this).stop(true).animate({
                            left: 100
                        }, 1500, "easeInOutQuart");
                        $("img", this).stop(true).animate({
                            opacity: 0
                        }, 1500, "easeInOutQuart");
                    }
                });
            }
            nowPos = Number(num);
            checkCurrent();
            $("#wrapper").stop(true).animate({
                opacity: 1
            }, interval, "linear", next);
        }

        function checkCurrent() {
            $("#mainvisual .currentarea .current li").each(function() {
                $("a", this).removeClass("active");
                if (nowPos == this.id) $("a", this).addClass("active");
            })
        }
