      function refreshMap() {

        if (markerClusterer) {
//         	var markerarray = markerClusterer.getMarkers();
//         	for (var i = 0, len = markerarray.length; i < len; i++) {
//         		markerClusterer.removeMarker(markerarray[i]);
//         	}
            markerClusterer.clearMarkers();
        }

        var markers = [];

        // First, determine the map bounds
        var bounds = map.getBounds();

        // Then the points
        var swPoint = bounds.getSouthWest();
        var nePoint = bounds.getNorthEast();

        // Now, each individual coordinate
        var swLat = swPoint.lat();
        var swLng = swPoint.lng();
        var neLat = nePoint.lat();
        var neLng = nePoint.lng();        

        var radius = google.maps.geometry.spherical.computeDistanceBetween(swPoint, nePoint);
        //taking a radius to a diameter
        radius = radius * .5;
        //calculating miles from meters (1609.34 per mile)
        radius = radius / 1609.34;
        var center = map.getCenter();

		var data = JSON.parse(Get('includes/dataradius.php?lat=' + center.lat() + '&lng=' + center.lng() + '&radius=' + radius));
		var totalCount = data.count;

		var bounds1 = new google.maps.LatLngBounds();
            
	    for (var i = 0; i < totalCount; i++) {
		        
			var latLng = new google.maps.LatLng(data.locations[i].lat, data.locations[i].lng)
	          
	    	locname = data.locations[i].name;

	        //this is where we build our markerimage (pin)
	        var status = data.locations[i].status;
	        if(status == 'Brewery'){
	        	var markerImage = new google.maps.MarkerImage(breweryImageUrl,new google.maps.Size(16, 16));
	        }else if(status == 'Brewpub'){
	        	var markerImage = new google.maps.MarkerImage(brewpubImageUrl,new google.maps.Size(16, 16));
	        }else if(status == 'Beer Bar/Restaurant'){
	        	var markerImage = new google.maps.MarkerImage(beerbarImageUrl,new google.maps.Size(16, 16));
	        }else if(status == 'Beer Store'){
	        	var markerImage = new google.maps.MarkerImage(beerstoreImageUrl,new google.maps.Size(16, 16));
	        }else if(status == 'Homebrew'){
	        	var markerImage = new google.maps.MarkerImage(homebrewImageUrl,new google.maps.Size(16, 16));
	        }else{
	        	var markerImage = new google.maps.MarkerImage(imageUrl,new google.maps.Size(16, 16));
	        }
	          
	        var marker = new google.maps.Marker({
	            position: latLng,
	            draggable: false,
	            icon: markerImage,
	            title: locname,
	            infowindow: infowindow,
	            type: status
	        });  
	        if (!markerGroups[status]) markerGroups[status] = [];
	        markerGroups[status].push(marker);   
	        if(data.locations[i].total > 89){
				markerGroups['90s'].push(marker);
	        }else if(data.locations[i].total > 79){
	        	markerGroups['80s'].push(marker);
	        }else{
	        	markerGroups['70s'].push(marker);
	        }
	
	        //Attach click event to the marker.
	        (function (marker, data) {
	
				var contentString = '<div id="content" style="width:250px;min-height:40px">'+
		          '<div id="siteNotice">'+
		          '</div>'+
		          '<h1 id="firstHeading" class="firstHeading"><a href="https://beermapping.com/location/' + data.locations[i].id + '">' + data.locations[i].name + '</a></h1>'+
		          '<div id="bodyContent">'+
		          '<div class="address">' + 
		          data.locations[i].street + '<br />' +
		          data.locations[i].city + ', ' + data.locations[i].state + ' ' + data.locations[i].zip + '<br />' +
		          data.locations[i].country + '<br />' +
		          data.locations[i].phone + '<br /><a href="' +
		          data.locations[i].url + '">website</a><br />' + 
		          data.locations[i].status + '</div>';
	
				  if(data.locations[i].total == 0){
					  contentString += '<div class="no-score">No Score</div>';
				  }else{
					  contentString += '<div class="score"><a href="https://beermapping.com/location/' + data.locations[i].id + '">' + data.locations[i].total + '</a></div>';
				  } 
				  if(!!data.locations[i].gtotal){
					  if(data.locations[i].gtotal != '0'){
					  	contentString += '<div style="clear:both;"></div><div class="gscore"><strong>' + data.locations[i].gtotal + ' out of 5 </strong>(Google Reviews)</div>';
					  }
				  }  
	
		          contentString += '<div style="clear:both;"></div>'+
		          '</div>'+
		          '</div>';
		                   
	           google.maps.event.addListener(marker, "click", function (e) {
	                  //Wrap the content inside an HTML DIV in order to set height and width of InfoWindow.
	                  infowindow.setContent(contentString);
	                  infowindow.open(map, marker);
	              });
	         })(marker, data);          
	          
	          markers.push(marker);
        }

        //setting these to -1 to be the default
        var zoom = parseInt(11);
        var size = parseInt(-1);
        var style = parseInt(-1);
        zoom = zoom === -1 ? null : zoom;
        size = size === -1 ? null : size;
        style = style === -1 ? null: style;

        markerClusterer = new MarkerClusterer(map, markers, {
          maxZoom: zoom,
          gridSize: size,
          styles: styles[style],
          imagePath: 'mapincludes/images/m'
        });
      }

      function CenterControl(controlDiv, map) {

    	  // Set CSS for the control border.
    	  var controlUI = document.createElement('div');
    	  controlUI.style.backgroundColor = '#fff';
    	  controlUI.style.border = '2px solid #fff';
    	  controlUI.style.borderRadius = '3px';
    	  controlUI.style.boxShadow = '0 2px 6px rgba(0,0,0,.3)';
    	  controlUI.style.cursor = 'pointer';
    	  controlUI.style.marginBottom = '22px';
    	  controlUI.style.textAlign = 'center';
    	  controlUI.title = 'Click to center map on your position';
    	  controlDiv.appendChild(controlUI);

    	  // Set CSS for the control interior.
    	  var controlText = document.createElement('div');
    	  controlText.style.color = 'rgb(25,25,25)';
    	  controlText.style.fontFamily = 'Roboto,Arial,sans-serif';
    	  controlText.style.fontSize = '12px';
    	  controlText.style.lineHeight = '28px';
    	  controlText.style.paddingLeft = '5px';
    	  controlText.style.paddingRight = '5px';
    	  controlText.innerHTML = 'Center on Me';
    	  controlUI.appendChild(controlText);

    	  // Setup the click event listeners: simply set the map to Chicago.
    	  controlUI.addEventListener('click', function() {
    	    map.setCenter(myLocation);
