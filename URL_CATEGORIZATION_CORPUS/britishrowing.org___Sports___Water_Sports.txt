    // Address Lookup function
    function lookupAddress() {

        var serviceURL = 'https://ws.postcoder.com/pcw/'+searchKey+'/address/UK';

        var addressListDropdown = $('#addressListSelect');

        var lookupValue = encodeURIComponent($('#searchField').val());

        if (lookupValue == '') {
            return;
        }

        // Build the RESTful URL
        var ajaxURL = serviceURL + '/' + lookupValue + '?identifier=' + identifier + '&lines=' + numberOfAddressLines;

        // this example has a separate organsiation line, so exclude organisation from the address lines:
        ajaxURL += '&exclude=organisation';

        // 'include' works too, for example to include town and postcode in the address lines section, use:
        //ajaxURL += '&include=posttown,postcode';

        // Make the actual JSON request
        $.getJSON(ajaxURL, function (data) {

            addressData = data; // data is the return from the query supplied to the "success" function we're now inside.

            if (data.length > 0) { // If there's actually any results then show the result box
                $('#selectDropdownDiv').show();

                // Clear the current dropdown
                addressListDropdown.find('option').remove();
                addressListDropdown[0].options.add(new Option('Select an Address:','-1'));

                $.each(data, function (index, value) {
                    // Iterating through returned Addresses
                    addressListDropdown[0].options.add(new Option(value.summaryline,index));
                })
            }

        }).fail(function (jqxhr, textStatus, error){
            if (typeof(console) != "undefined"){
                console.error("getJSON failed, status: " + textStatus + ", error: "+error);
            }
        });
    }

    function CheckUserName(username) {
      // Get the value
      username = $('#txtUsername').val();

      // Make sure its longer than 4 characters
      if(username.length < 4) {
        return;
      } else {
        // We are now checking...
        $('.username-text').html('<p>Checking username availability...</p>');
      }

      //https://membership.test.britishrowing.org/IntegratedRegistration/UsernameAvailable?username=simon
      // Check the username availabilty against the membership server
      ajaxURL = 'https://membership.britishrowing.org/IntegratedRegistration/UsernameAvailable?username='+username;
      $.post(ajaxURL, function (data) {
        $.each(data, function (index, value) {
          // If we get a true result then it is available
          if(value == true) {
            $('.username-text').html('<p>This username is available for you to use</p>');
            $('#txtUsername').removeClass('notAvailable');
            $('#txtUsername').removeClass('error');
            $('.txtUsername_notAvailable').remove();
          // Otherwise it is not available
          } else {
            $('.username-text').html('<p>We\'re sorry, this username is not available, please choose another one</p>');
            $('#txtUsername').addClass('notAvailable');
          }
        })
      }).fail(function (jqxhr, textStatus, error){
          if (typeof(console) != "undefined"){
              console.error("post failed, status: " + textStatus + ", error: "+error);
          }
      });

    }
