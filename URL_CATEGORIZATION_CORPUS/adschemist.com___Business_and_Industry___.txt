  <script>
   $(function() {
    $("#formulario").submit(function() {
        var name = $.trim($(this).find(":input[name=name]").val());
        var email = $.trim($(this).find(":input[name=email]").val());
        var message = $.trim($(this).find(":input[name=message]").val());
        
        // Validar los datos de entrada:
        if(!name) {
            $.fancybox($("<div>Please enter your name!</div>"));
            return false;
        }
        if(!email) {
            $.fancybox($("<div>Please enter your email!</div>"));
            return false;
        }
        if(!validateEmail(email)) {
            $.fancybox($("<div>Please enter a valid email!</div>"));
            return false;
        }
        if(!message) {
            $.fancybox($("<div>Please enter a message!</div>"));
            return false;
        }
        
        $.fancybox({
            modal: true,
            showCloseButton: false,
            enableEscapeButton: false,
            content: $("<div>Your message is being sent<br><img src='images/spinner.gif'></div>").css("text-align", "center")
        });
        
        $.ajax({
            url: "ajax.php",
            dataType: "json",
            data: {
                action: "do_contact",
                name: name,
                email: email,
                message: message
            },
            type: "post",
            success: function(res) {
                $.fancybox.close();
                
                if(res.errores.length) {
                    var msg = "<div>The following errors occured:";
                    for(var i=0; i<res.errores.length; i++) {
                        msg += "<br>- " + res.errores[i];
                    }
                    msg += "</div>";
                    $.fancybox($(msg));
                    return;
                }
                $.fancybox($("<div>Your message has been sent correctly.</div>"));
                
                $("#formulario :input[name=name]").val("");
                $("#formulario :input[name=email]").val("");
                $("#formulario :input[name=message]").val("");
            },
            error: function() {
                $.fancybox.close();
                $.fancybox($("<div>There was an unknown Ajax error while sending your message.</div>"))
            }
        });
        
        return false;
    });
});

function validateEmail(email) { 
    var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    return re.test(email);
}
