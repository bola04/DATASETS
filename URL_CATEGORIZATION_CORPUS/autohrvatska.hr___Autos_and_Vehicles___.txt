    function resizeItems(){

        var offerItemHeight = $('.offer-item');

        var height = 0;
        if(offerItemHeight.length > 0){
            offerItemHeight.each(function(){
                if(height < $(this).find('.item-description').find('article').outerHeight())

            });

            $('.item-description').find('article').css('height', height +'px');
