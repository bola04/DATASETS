    $(function ()

        $("#switchimg").mouseleave(function () { $(this).attr("src", "https://i.euroinvestor.com/icons/switchon.png"); });

        ddlfrom = $("#ctl00_content_ctl00_ctl02_DDLCalculatorFromCurr2").children();

    function SwitchDDLsNew() {
        var curFrom = $("#cdd_1_selected").val();
        var curTo = $("#cdd_2_selected").val();

        curdd_selectitem(1, '' + curTo + '');
        curdd_selectitem(2, '' + curFrom + '');

    }

    function SwitchDDLs()
    {
        var selIndexFrom = ddlfrom.index(ddlfrom.filter(":selected"));
        var selIndexTo = ddlto.index(ddlto.filter(":selected"));

        ddlfrom.get(selIndexTo).selected = true;
        ddlto.get(selIndexFrom).selected = true;

        ConvertCalc(false);
    }

    function ConvertCalc(tofrom)
    {
       
        var amount = tofrom ? parseFloat(textto.val().replace(',', '.')) : parseFloat(textfrom.val().replace(',', '.'));
        if (isNaN(amount) || amount == undefined)
        {
            if (tofrom) textfrom.val("");
            else textto.val("");
            return;
        }

        var curFrom = parseFloat(ddlfrom.filter(":selected").val());
        var curTo = parseFloat(ddlto.filter(":selected").val());
        var res = (tofrom ? curTo / curFrom : curFrom / curTo) * amount;
        res = (Math.round(res * 100) / 100).toFixed(2);
        if (tofrom) textfrom.val(("" + res).replace('.', EI.Localization.decimalSeparator()));
        else textto.val(("" + res).replace('.', EI.Localization.decimalSeparator()));
    }

    function ConvertCalcNew(e) {        
        // don't process when user types decimal separator        
        if (e.keyCode == 188 || e.keyCode == 190 || e.keyCode == 110) return;

        var switchdir = typeof (e) != "object";
        if (!switchdir) lastActiveInputElement = $(this); // update last active element on user keypress
        //var fromelem = switchdir ? $("#cdd_" + e + "_inp") : $(this);
        var fromelem = lastActiveInputElement || $("#cdd_" + e + "_inp");
        var numberfrom = fromelem.attr("id").split('_')[1];
        var numberto = numberfrom == "1" ? "2" : "1";
        var toelem = $("#cdd_" + numberto + "_inp");

        //switch direction when dropdown is used
//        if (switchdir)
//        {
//            var temp = numberfrom;
//            numberfrom = numberto;
//            numberto = temp;

//            temp = fromelem;
//            fromelem = toelem;
//            toelem = temp;
//        }

        if (fromelem.val() == "")
        {
            toelem.val("");
            return;
        }

        var curFrom = parseFloat($("#cdd_" + numberfrom + "_value").val());
        var curTo = parseFloat($("#cdd_" + numberto + "_value").val());
        //var amount = parseFloat(fromelem.val().replace(',', '.'));
        var amount = EI.Localization.LocalizedStringToJSFloat(fromelem.val());
        var res = (curFrom / curTo) * amount;
        res = (Math.round(res * 100) / 100).toFixed(2);        
        //toelem.val(("" + res).replace('.', EI.Localization.decimalSeparator()));
        if (amount % 1 != 0) // dont delete trailing zeros as soon as user types them
            fromelem.val(EI.Localization.JSFloatToLocalizedString(amount));
        toelem.val(EI.Localization.JSFloatToLocalizedString(res));
    }

    function addToPageViewCount() {
        if (typeof pp_gemius_identifier !== "undefined" && $.isFunction(pp_gemius_hit)) {
            pp_gemius_nhit = 1;
            pp_gemius_hit(pp_gemius_identifier.replace("USED_", ""));              
        }
        _gaq.push(['_trackPageview']);
    }

    function resetInputFields() {
        $("#cdd_1_inp").val('');

    };

    function OpenDatePicker()
    {
        var dp = $(".ui-datepicker");
        if (dp.length > 0)
        {
            if (dp.is(":visible")) dp.hide();
            else dp.show();
        }
        else
        {
            EI.DelayLoad.loadScriptFile("https://ajax.googleapis.com/ajax/libs/jqueryui/1.8.13/jquery-ui.min.js", function () {
                EI.DelayLoad.loadStyleSheet("https://ajax.googleapis.com/ajax/libs/jqueryui/1.7.2/themes/base/jquery-ui.css", function () {
                    $.datepicker.setDefaults($.datepicker.regional[EI.Localization.siteCountry()]);

                    $("#dateselector").datepicker({
                        dateFormat: EI.Localization.shortDateFormat(),
                        maxDate: 0,
                        onSelect: function (dateText, inst) {
                            //dropdowns selected (number should not be hardcoded... oh well)
                            var sel1 = $("#cdd_1_selected").val();
                            var sel2 = $("#cdd_2_selected").val();
                            var cp = $("#divCP");

                            var header = cp.find("div.read_more span");
                            if (header.html().indexOf("(") > -1) {
                                var newHeader = header.html().substring(0, header.html().indexOf("(")) + "(" + dateText + ")";
                                header.html(newHeader);
                            }

                            EI.Global.getJson("GetConverterData", [["date", dateText]], function (json) {
                                var items = $("div[id^=cdditem]");
                                for (var i = 0; i < json.length; i++) {

                                    items.filter("div[id$=" + json[i].Currency + "]").each(function (index, elem) { $(elem).attr("val", json[i].Value); });
                                    if (json[i].Currency == sel1) $("#cdd_1_value").val(json[i].Value);
                                    if (json[i].Currency == sel2) $("#cdd_2_value").val(json[i].Value);
                                    cp.find("[id$=" + json[i].Currency + "]").html(json[i].Value);
                                }
                                $("#datedest").text(dateText);
                                $(".ui-datepicker").hide();
                                ConvertCalcNew("1");
                                addToPageViewCount();
                            });
                            //CP if date<dags dato, skjul red and green

                            if (dateText < new Date().format("d")) {
                                $('#divCP .percentChange span').hide();
                            }
                            else {
                                $('#divCP .percentChange span').show();
                            }                            
                        }
                    });

                    $(".ui-datepicker").css("position", "absolute").css("top", "5px");
                });
            });
        }
    }
