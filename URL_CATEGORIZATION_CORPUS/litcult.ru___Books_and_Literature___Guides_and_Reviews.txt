var catchSubmit = function catchSubmit(event) {
    if ((document.getElementById('addressi') == null) ||
      (document.getElementById('latitudei') == null) ||
      (document.getElementById('longitudei') == null)) {
    return true;
  }
  if (!isNaN(document.getElementById('latitudei').value.trim()) &&
      !isNaN(document.getElementById('longitudei').value.trim())) {
    var curLat = parseFloat(document.getElementById('latitudei').value.trim());
    var curLon = parseFloat(document.getElementById('longitudei').value.trim());
    if ((curLat > -90.0)&&(curLat < 90.0)&&(curLon > -90.0)&&(curLon < 90.0)) {
      return true;

  var myGeocoder = ymaps.geocode(
    		         // Строка с адресом, который нужно геокодировать
        		     addr, {
    		             // - искать только в этой области
    		             strictBounds: false,
    		             // - требуемое количество результатов
    		             results: 1
    		         }
    		       );
          myGeocoder.then(
            function(response) {
        	    var latitude = 88.00000;
        	    var longitude = 88.00000;
                if (response.geoObjects.getLength()) {
                    // point - первый элемент коллекции найденных объектов
                    var point = response.geoObjects.get(0);
                    latitude = point.geometry.getBounds()[0][0];
                    longitude = point.geometry.getBounds()[0][1];
                }
                document.getElementById('latitudei').set("value", latitude);
                document.getElementById('longitudei').set("value", longitude);
                document.getElementById('DD_Form').submit();
                            }
          );
    return false;
}
