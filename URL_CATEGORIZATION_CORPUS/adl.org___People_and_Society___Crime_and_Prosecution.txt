			document.getElementById('header-bottom').classList.remove('expand-search--mobile');
			document.documentElement.classList.remove('no-overflow');
		}
	}

	function toggleMenuItem(context,interactionType){
		clearHeaderBottom("DESKTOP");
		toggleDesktopSearch("CLOSE");
		if(interactionType === "CLICK" && activeDropdown === jQuery(context).data('menu-id')){
			activeDropdown = false;
			return;
		}
		activeDropdown = jQuery(context).data('menu-id');
		jQuery("#header-bottom .menu-item[data-menu-id='"+activeDropdown+"']").css('display','block').addClass('ani-in');
		jQuery(context).addClass('dropped-down');
	}

	function toggleDesktopSearch(forceState){
		if(desktopSearchOpen || forceState === "CLOSE"){
			jQuery('#google-search-form-container--desktop').removeClass('expand-search');
			desktopSearchOpen = false;
			jQuery('#search-expand-button--desktop').removeClass('icon-close');
			jQuery('#search-expand-button--desktop').addClass('icon-search');
		}else{
			clearHeaderBottom("DESKTOP");
			jQuery('#google-search-form-container--desktop').addClass('expand-search');
			jQuery('#google-search-form-container--desktop input#edit-keys--2').focus()
			desktopSearchOpen = true;
			jQuery('#search-expand-button--desktop').addClass('icon-close');
			jQuery('#search-expand-button--desktop').removeClass('icon-search');
		}
	}


	window.onload = function(){
		jQuery('#header .menu-level-0 .menu-item').hover(function(){
			if(window.innerWidth >= desktopBreakpoint){
				clearHeaderBottom("DESKTOP");
				toggleMenuItem(this);
			}
		})
		jQuery('#header-bottom .menu-level-0').mouseleave(function(){
			if(window.innerWidth >= desktopBreakpoint){
				clearHeaderBottom("DESKTOP");
			}
		})

		jQuery('#header .menu-level-0 .menu-item').click(function(e){
			if(window.innerWidth < desktopBreakpoint && window.innerWidth >= tabletBreakpoint){
				e.preventDefault();
				toggleMenuItem(this,"CLICK");
			}
		});

		jQuery('#search-expand-button--desktop').mousedown(function(e){
			e.preventDefault();
			toggleDesktopSearch();
		});
		jQuery('#search-expand-button--desktop').focus(function(e){
			console.log("focus",e);
			toggleDesktopSearch();
		});
	};

	function jumpToTop(){
		var jumpToEl = jQuery('.header-video-container');
	    if(window.mobileNavOpen) {
			jQuery('.header-bottom-menu').animate({
		        scrollTop: 0 ,
		    }, 400);
	    }else{
	    	jQuery('html, body').animate({
		        scrollTop: 0 ,
		    }, 400);
	    }

	}

	var resizeTimer;
	window.onresize = function(){
		clearTimeout(resizeTimer);
		resizeTimer = setTimeout(function() {
			if(window.innerWidth <= 768) {
				clearHeaderBottom("DESKTOP");
				toggleDesktopSearch("CLOSE");
			}else{
				clearHeaderBottom("MOBILE");
			}
		}, 150);
	}
     </script>
