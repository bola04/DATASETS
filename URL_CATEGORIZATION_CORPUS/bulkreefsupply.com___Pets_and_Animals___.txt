					jQuery(".ytid_" + videoData.id + " .video_details .viewcount").html( videoData.statistics.viewCount.toString().replace(/.(?=(?:.{3})+$)/g, '$&,') + " views" );
				});
			});
		});

		function parseDurationData(videoData) {
			var ytTimes = "";
			var lastIncrement = "";

			var ytTimesArr = videoData.contentDetails.duration.substring(2).split(/(\d{1,2}h)*(\d{1,2}m)*(\d{1,2}(\.\d+)*s)*/i).filter(Boolean);

			ytTimesArr.forEach(function(time, index) {
					var timeData = time.split(/(\d+)(\D+)/).filter(Boolean);
					if( index > 0 ) { ytTimes += ":"; }
					if( ytTimes.length > 0 && timeData[0].length == 1 && timeData[1].toLowerCase() != "h" ) { ytTimes += "0"; }
					ytTimes += timeData[0];
					lastIncrement = timeData[1];
				});
			if( lastIncrement.toLowerCase() == "m" ) { ytTimes += ":00"; }  // Minute only - No seconds
			if( lastIncrement.toLowerCase() == "h" ) { ytTimes += ":00:00"; }  // Hour only - No minutes or seconds

			// Add time data to video
			jQuery("." + ytClassPrefix + videoData.id + " .duration").html( ytTimes );
		}

		// Date helpers
		var monthsNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
		function nth(d) { if(d>3 && d<21) return 'th'; switch (d % 10) { case 1:  return "st"; case 2:  return "nd"; case 3:  return "rd"; default: return "th"; } }

		function parseTimeData(videoData) {
			// Parse time data
			var ytTimes = "";
			if( jQuery("#video_category_view_first_video").hasClass( ytClassPrefix + videoData.id ) ) {
				var pubTime = new Date( videoData.snippet.publishedAt );
				var ytTimes = monthsNames[pubTime.getMonth()] + " " + pubTime.getDate() + nth(pubTime.getDate()) +", " +pubTime.getFullYear();
			} else {
				var daysSincePublished = Math.floor( ( Math.abs( new Date() - new Date( videoData.snippet.publishedAt ) ) ) / 8.64e7 );
				if( daysSincePublished < 7 ) {
					var ytTimes = daysSincePublished + " day" + ( daysSincePublished > 1 ? "s" : "" ) +" ago";
				} else {
					var weeksSincePublished = Math.floor( daysSincePublished / 7 );
					if( weeksSincePublished < 4 ) {
						if( weeksSincePublished == 0 ) { weeksSincePublished = 1; }
						var ytTimes = weeksSincePublished + " week" + ( weeksSincePublished > 1 ? "s" : "" ) +" ago";
					} else {
						var monthsSincePublished = Math.floor( daysSincePublished / 30 );
						if( monthsSincePublished < 12 ) {
							if( monthsSincePublished == 0 ) { monthsSincePublished = 1; }
							var ytTimes = monthsSincePublished + " month" + ( monthsSincePublished > 1 ? "s" : "" ) +" ago";
						} else {
							var yearsSincePublished = Math.floor( daysSincePublished / 365 );
							if( yearsSincePublished == 0 ) { yearsSincePublished = 1; }
							var ytTimes = yearsSincePublished + " year" + ( yearsSincePublished > 1 ? "s" : "" ) +" ago";
						}
					}
				}
			}
			// Add time data to video
			jQuery("." + ytClassPrefix + videoData.id + " .date").html( ytTimes );
		}
