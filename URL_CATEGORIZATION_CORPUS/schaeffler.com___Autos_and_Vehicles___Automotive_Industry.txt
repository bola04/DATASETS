   ;(function (global, undefined) {
            var $ = global.jQuery,
                _ = global.Utils;

            var selectedURL = '';


            var URLSwitch = {
                onSubmit: function (event) {
                    var _evLabel = 'No selection';

                    event.preventDefault();
                    event.stopPropagation();

                    if (selectedURL) {
                        _evLabel = 'Website for ' + $('#countryform').find('[value="'+ selectedURL +'"]').html() + ' was chosen';
                    }

                    dataLayer.push({
                        'event': 'customEvent',
                        'eventCategory': 'Button',
                        'eventAction': 'Select country Website',
                        'eventLabel': '' + _evLabel
                    });

                    global.location.href = 'http://www.' + selectedURL;
                },

                onChange: function (data) {
                    selectedURL = data.value;

                    if (selectedURL === '') {
                        return;
                    }

                    $('#countryform').submit();
                }
            }


            var $$countrySelect = new Forms.Select({
                container: '#countries'
            });
            $$countrySelect.render().startUp();
            $$countrySelect.addObserver({ update: URLSwitch.onChange });

            $('#countryform').on(_.eventType.SUBMIT, URLSwitch.onSubmit);
            
        })(this, (void 0));
  </script>
