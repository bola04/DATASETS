    <script type="text/javascript">
     $(document).ready(function(){
		// refresh capcha	    
	    $('.refresh-popup').click(function(){
	    	refreshCaptchaPopup();
	    });
	});

	//enable login pressing Enter key
	$('#signin input[name="data[email]"], #signin input[name="data[password]"]').keyup(function(event){
	    if(event.keyCode == 13){
	        $("#btn-login").trigger('click');
	    }
	});
                			
	// normal login
	$('#btn-login').on( "click", function(e) {
	
		var email 	 	= $('#signin input[name="data[email]"]').val();
		var password 	= $('#signin input[name="data[password]"]').val();
		var remember_me = $('#signin input[name="data[remember_me]"]').is(':checked') ? 1 : 0;

		$.ajax({
            type: "POST",
            data : {email : email, password : password, remember_me : remember_me},
            dataType : 'json',
            cache : false,
            url: '/swift/login/',
            
            success: function(response){
				if (response.is_success){
					location.reload(); 
				}
				else {
					$('#signin #signin-error').html(response.message);
					$('#signin #signin-error').addClass('alert alert-danger');
				}
            },
            error: function (response, status, error) {
                alert(error);
            }
        })
	});

	// show facebook signup button only if member is natural
	$('input[name="data[type]"]').click(function(e){
		var member_type = $(this).val();
		if( member_type == 'Natural' ){
			$(".fb-fields").show();
		}
		else{
			$(".fb-fields").hide();
		}
    });

    //enable login pressing Enter key
    var registration_fields = '#registration input[name="data[email_signup]"], #registration input[name="data[password_signup]"], #registration input[name="data[password_repeat]"], #registration input[name="data[captcha]"]';
	$(registration_fields).keyup(function(event){
	    if(event.keyCode == 13){
	        $("#btn-signup").trigger('click');
	    }
	});

	// normal signup/register
	$('#btn-signup').on("click", function(e) {
		var member_type 	= $('#registration input[name="data[type]"]:checked').val();
		var email 	 		= $('#registration input[name="data[email_signup]"]').val();
		var password 		= $('#registration input[name="data[password_signup]"]').val();
		var password_repeat	= $('#registration input[name="data[password_repeat]"]').val();
		var accept_terms 	= ( $('#registration input[name="data[accept_terms]"]').is(':checked') ) ? 1 : 0;
		var captcha			= $('#registration input[name="data[captcha]"]').val();

		if(!areTermsAccepted(e)){ return; }

		$.ajax({
            type: "POST",
            data : {
                	member_type		:member_type, 
                	email			:email, 
                	password		:password, 
                	password_repeat	:password_repeat, 
                	accept_terms	:accept_terms, 
                	captcha			:captcha
            },
            dataType : 'json',
            cache : false,
            url: '/swift/signup/',
            
            success: function(response){
				if (response.is_success){
					$('#registration #signup-success').html(response.message);
					$('#registration #signup-success').addClass('alert alert-success');
					var inputs_to_clear = '#registration input[name="data[email_signup]"], #registration input[name="data[password_signup]"], #registration input[name="data[password_repeat]"], #registration input[name="data[captcha]"]';

					// clear fields after registration
					$(inputs_to_clear).val('');
					$('#registration input[name="data[accept_terms]"]').attr('checked', false);
					$('#registration input[name="data[type]"]').val(['Natural']);

					$('#registration #signup-error').html('');
					$('#registration #signup-error').removeClass('alert alert-danger');
					refreshCaptchaPopup();
				}
				else {
					$('#registration #signup-error').html(response.message);
					$('#registration #signup-error').addClass('alert alert-danger');
					refreshCaptchaPopup();
				}
            },
            error: function (response, status, error) {
                alert(error);
            }
        });
	});

	// facebook signup/registration
	$('#registration .face-login').click(function(e){		
		if(!areTermsAccepted(e)){ return; }
        e.preventDefault();
        
        //upisivanje vrednosti social mreze u logout
        var member_type = $('#registration input[name="data[type]"]:checked').val();        
        if( member_type == 'Natural' ){
	        var left = ($(window).width()/2)-(800/2),
	        top = ($(window).height()/2)-(500/2),
	        pop = window.open("/socialauth/login/Facebook", "popup", "width=800, height=500, top="+top+", left="+left);
        }
        else{
            alert('Facebook registracija je namenjena samo fizičkim licima.');
        }
	});

	// facebook signup/registration
	$('.facebook-login').click(function(e){		
        e.preventDefault();
        
	        var left = ($(window).width()/2)-(800/2),
	        top = ($(window).height()/2)-(500/2),
	        pop = window.open("/socialauth/login/Facebook", "popup", "width=800, height=500, top="+top+", left="+left);
        
	});

	// only if terms and conditions are accepted
	function areTermsAccepted(e){
		if(!$('input[name="data[accept_terms]"]:checked').length) {
            alert("Morate prihvatiti pravila i uslove korišćenja.");
            //stop the form from submitting
            e.preventDefault();
            return false;
        }
		return true;
	}

	function refreshCaptchaPopup(){
        $("#code").val("");
        $("#captcha").attr("src", $("#captcha").attr("src").trim() + "/" + Math.round(Math.random()*10000000000));
	}
    </script>
