        function subscribeUser() {
            var email = $('#subscriberEmail').val();
            if (email.length === 0) {
                $('#msg').fadeIn();
                $('#msg').html("Please enter your mail id!");
                $('#msg').delay(10000).fadeOut();
            } else {
                if (validateEmail(email)) {
                    $.ajax({
                        url: "https://royalenfield.com/de/ajax/add_subscriber.aspx",
                        type: "POST",
                        data: {
                            useremail: email
                        },
                    }).done(function (msg) {
                        //console.log( "Session Set: " + msg );

                            $('#subscriberEmail').empty();
                            $('#msg').fadeIn();

                            $('#msg').fadeIn();
                            $('#msg').html("Your mail id is already subscribed with us!");
                            $('#msg').delay(10000).fadeOut();
                        }
                    });
                } else {
                    $('#msg').fadeIn();
                    $('#msg').html('Not a valid email id!');
                    $('#msg').delay(10000).fadeOut();
                }
                return false;
            }
        }
