				}
				else if (screenwidth == "medium") {

					}
					else {
						$("#osw-socialwall-col2").append(getSocialWallPost(this, i));
					}
				}
				else {
					if (i%3 == 0) {
						$("#osw-socialwall-col1").append(getSocialWallPost(this, i));
					}
					else if (i%3 == 1) {
						$("#osw-socialwall-col2").append(getSocialWallPost(this, i));
					}

						$("#osw-socialwall-col3").append(getSocialWallPost(this, i));
					}

		});
