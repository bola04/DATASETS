     function resizeUpperHeaderItems() {
            var w = 0;
            $.each($('.js-upper-header-items li'), function () {
                $(this).removeAttr('style');
                w += $(this).innerWidth();
            });
            var n = $('.js-upper-header-items li').length;
            var padding = Math.abs(((($('.second-block').innerWidth()-20) - w ) / n) / 2);
            $('.js-upper-header-items li').css('padding-left', padding + 'px').css('padding-right', padding + 'px');
        }

        function initUpperHeader() {

            $('.js-quicklinks-wrap').on('mouseenter', function () {
                $(this).find('.quick-links-overlay').removeClass('quicklinks-hide');
                $(this).addClass('opened');
            }).on('mouseleave', function () {
                $(this).find('.quick-links-overlay').addClass('quicklinks-hide');
                $(this).removeClass('opened');
            });

            $('.js-btn-header-search .btn-header-search').on('click', function (e) {
                var wrapper = $(this).closest('.js-btn-header-search');
                wrapper.toggleClass('opened');
                wrapper.find('.input-search').val('');
            });
            //resizeUpperHeaderItems();

            var resizeTimer;
            $(window).on('resize', function (e) {
                clearTimeout(resizeTimer);
                resizeTimer = setTimeout(function () {
                    resizeUpperHeaderItems();
                }, 150);
            });
        }

        initUpperHeader();

        $(window).load(function () {
            resizeUpperHeaderItems();
        })
