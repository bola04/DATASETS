   function show_child(type) {
        $("#child_list p").each(function(){
            $(this).attr('style', 'display: none');
        });
        if (type == 'none') {
            $('#bread').attr('style', 'display: ');
        } else {
            $('#child_'+type).attr('style', 'display: ');
            $('#bread').attr('style', 'display: none');
        }
        return true;
    }

    function show_search_type(type) {
        if (type) {
            $("#secrch_list").attr('style', '');
        } else {
            $("#secrch_list").attr('style', 'display: none');
        }
        return true;
    }

    function change_search_type(type) {
        if (type == 'app') {
            $("#search_app").attr('style', '');
            $("#search_news").attr('style', 'display: none');
            $("#search_type").val('app');
        } else {
            $("#search_app").attr('style', 'display: none');
            $("#search_news").attr('style', '');
            $("#search_type").val('news');
        }
        $("#secrch_list").attr('style', 'display: none');
        return true;
    }

    function enterIn(evt){
        var evt = evt ? evt : (window.event?window.event:null);//兼容IE和FF
        if (evt.keyCode==13){
            submit_search();
        }
    }

    function submit_search() {
        var info = $("#search_info").val();
        var type = $("#search_type").val() == 'app' ? '' : 'news';
        var url  = '/search/'+ type +'?keyword='+encodeURI(info) +'&search_type=' + $("#search_type").val();
        location.href = url;
        return false;
    }
    change_search_type($("#search_type").val());
  </script>
