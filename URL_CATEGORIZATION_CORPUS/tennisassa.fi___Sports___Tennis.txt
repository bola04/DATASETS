  var bindClientValidation = function () {
    $('form').each(function () {
      $(this).removeData("validator").removeData("unobtrusiveValidation");

      if ($.validator) { 
        var originalDateValidator1 = $.validator.methods.date;
        var originalDateValidator2 = $.validator.methods.dateISO;

        $.validator.methods.date = function (value, element) {
          var isValidDate =
              originalDateValidator1.apply(this, arguments) ||
              originalDateValidator2.apply(this, arguments);
          return isValidDate;
        };
      }

      //Parse the form again (for dynamic forms)
      $.validator.unobtrusive.parse(this);

      var form = $.data(this, 'validator');
      if (form) {
        var settings = form.settings;
        settings.onfocusout = function (element) {
          if ($(element).hasClass('js-val-ignore_focusout')) return;
          this.element(element);
        };
        settings.onkeyup = function (element, event) {
          if ($(element).data('val-remote')) { return false; }//do not valide remote on keyup
          else { //validate default 
            var excludedKeys = [
				      16, 17, 18, 20, 35, 36, 37,
				      38, 39, 40, 45, 144, 225
            ];
            if (event.which === 9 && this.elementValue(element) === "" || $.inArray(event.keyCode, excludedKeys) !== -1) {
              return;
            } else if (element.name in this.submitted || element.name in this.invalid) {
              this.element(element);
            }
          }
        };
        var baseErrorPlacement = settings.errorPlacement;
        settings.errorPlacement = function (error, inputElement) {
          baseErrorPlacement(error, inputElement);
          //Set parent label class to form__error
          var label = inputElement.parents(".js-form-wrapper");//$("label[for='" + inputElement.prop('id') + "']");
          if (label) {
            if (error && error.text().length > 0) {
              label.addClass('form__error');
            } else {
              label.removeClass('form__error');
            }
          }
        };
        settings.ignore = ".js-ignore_val"
      }
    });
  }
