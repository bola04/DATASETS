    function scroll(href) {
      // スクロールの速度
      var speed = 500 // ミリ秒

      var formattedHref = href.match(".*\#(.*)$")[1];

      // 移動先を取得
      var target = $('#' + formattedHref)

      
      // if (formattedHref == 'focus') {
      //   location.href = href;
      //   return;
      // }

      // 移動先がないとき
      if (typeof target.offset() === "undefined") {
        location.href = href;
        return;
      }

      var position = target.offset().top - 110

      // スムーススクロール
      $('body,html').animate({scrollTop:position}, speed, 'swing')
    }

        $(this).find('.sub-menu-outer .sub-menu').css('padding-left', $(this).offset().left - m + 'px')
      }
    })

    if ((hash = location.hash) != '') {
      scroll(hash)
    }
