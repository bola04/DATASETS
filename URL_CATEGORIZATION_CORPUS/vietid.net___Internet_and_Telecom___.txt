  <script>
   //check validate form register
    $("#register_vietid").live('click', function(){
       var email = $('#email').val();
       var password = $("#password").val();
       var re_password = $('#re_password').val();
       var fullname = $("#fullname").val();
       //check empty email
       if(emptyForm(email) == false) {
           clearError();
           $('#mess_email').show();
           $("#mess_email").html('Bạn chưa nhập Email!');
           $("#email").focus();
           return false;
       }
       //check empty password
       if(emptyForm(password) == false) {
           clearError();
           $('#mess_password').show();
           $('#mess_password').html('Bạn chưa nhập Mật khẩu!');
           $("#password").focus();
           return false;
       }
       
       //check empty repassword
       if(emptyForm(re_password) == false) {
           clearError();
           $('#mess_repassword').show();
           $('#mess_repassword').html('Bạn chưa nhập lại Mật khẩu!');
           $("#re_password").focus();
           return false;
       }
       
       //check empty fullname
       if(emptyForm(fullname) == false) {
           clearError();
           $('#mess_fullname').show();
           $('#mess_fullname').html('Bạn chưa nhập tên đầy đủ!');
           $("#fullname").focus();
           return false;
       }
       if(password != re_password) {
           
            clearError();
            $('#mess_repassword').html('Xác nhận mật khẩu không chính xác!');
            $("#re_password").focus();
            return false;
       }
       if(validateForm(email) == false){
           clearError();
           $('.error_password').html('Email không chính xác!');
           $("#email_login").focus();
           return false;
       }
              
       $("#form_register").submit();
    })
    //validate email
    function validateForm(email)
    {
        var atpos=email.indexOf("@");
        var dotpos=email.lastIndexOf(".");
        if (atpos<1 || dotpos<atpos+2 || dotpos+2>=email.length)
        {
             return false;
        } else return true;
    }
    function emptyForm(text)
    {
        if(!text || 0 === text.length) {
           return false;
        }else return true;
    }
    function clearError(){
        $("#mess_fullname").hide();
        $("#mess_location").hide();
        $("#mess_sex").hide();
        $("#mess_password").hide();
        $("#mess_repassword").hide();
    }
  </script>
