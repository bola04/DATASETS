	function refreshSearchSuggestions(event) {
        if(event.keyCode == 40) {

    function checkEnterKey(event) {
        if(event.keyCode == 13) {
            try {
                event.cancelBubble = true;
                event.returnValue = false;
                var newLocation = $('.searchListItem.selected a')[0].href;
                document.location = newLocation;
                return false;
            }
            catch(err) {
                __doPostBack('ctl02$btnSearch','');
            }
        }
        return false;
    }

    function updateSearchResults() {
        selectedIndex = -1;
		if (window.XMLHttpRequest)
			httpRequest = new XMLHttpRequest();
		else if (window.ActiveXObject)
			httpRequest = new ActiveXObject("Microsoft.XMLHTTP");
		else
			return false;
		
		if(httpRequest) {
		    httpRequest.open("GET", requestUrl+"&dlv_"+searchListName+"=(keyword="+document.getElementById(txtSearchBox).value+")", true);
            httpRequest.setRequestHeader('Content-Type', 'text/html');
            document.getElementById("seacrhListBox").innerHTML = "<li class=\"searchListItem\" >Loading...</li>";
            document.getElementById("seacrhListBox_body").style.display="";
            httpRequest.onreadystatechange = httpRequestDoCallback;
            httpRequest.send('');
		}
    }
    
    function httpRequestDoCallback() {
		if (httpRequest.readyState == 4 && httpRequest.status == 200) {
			if ((httpRequest.responseText == null))
				return;
		}
		else {
		    return;
		}
		
		if(httpRequest.responseText == "") {
		    $("#seacrhListBox").html("<div style='padding:10px 10px 0px;'>No Results</div>");
		}
		else
            document.getElementById("seacrhListBox").innerHTML = httpRequest.responseText;
    }
    
    function exitSearch() {
        $("#seacrhListBox_body").fadeOut(100);
    }

    function changeSearchSelection(direction) {
        var listItems = $('.searchListItem');
        var selectedItem = $('.searchListItem.selected');
        
        if (selectedItem.length == 0) {
            listItems[0].className = 'searchListItem selected';
            return;
        }
        
        selectedItem[0].className = 'searchListItem';
        
        if (direction > 0) {
            if (selectedItem.next().length == 0)
                listItems[0].className = 'searchListItem selected';
            else
                selectedItem.next()[0].className = 'searchListItem selected';
            return;
        }
        else {
            if (selectedItem.prev().length == 0)
                listItems[listItems.length-1].className = 'searchListItem selected';
            else
                selectedItem.prev()[0].className = 'searchListItem selected';
            return;
        }
    }

    function changeSelectionTo(listElement) {
        try {
            $('.searchListItem.selected')[0].className = 'searchListItem';
        }
        catch (err) { var errS = err; }
        listElement.className = 'searchListItem selected';
    }
