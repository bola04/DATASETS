	function submenu(items, level) {
		function showSubmenu(el) {
			var submenu = $('.l' + level, el);
			el.addClass('hover');
			el.clearQueue();
			if (!submenu.is(':visible')) {
				el.delay(500).queue(function() {
					$('.l' + level, el.parent()).hide();
					submenu.fadeIn(100);
				});
			}
		}

		function hideSubmenu(el) {
			var submenu = $('.l' + level, el);
			el.removeClass('hover');
			el.clearQueue();
			if (submenu.is(':visible')) {
				el.delay(500).queue(function() {
					submenu.fadeOut(100);
				});
			}
		}

		items.on('mouseenter', function() {
			showSubmenu($(this));
		}).on('mouseleave', function() {
			hideSubmenu($(this));
		});
	}

	$('.menu').addClass('js-menu');
	submenu($('.menu .multi'), 1);
	submenu($('.menu ul.l1 li'), 2);
	submenu($('.menu ul.l2 li'), 3);
