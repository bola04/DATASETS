  <script>
   $(document).ready(function() {
		$('#pt-cn-wrap').fullpage({
			sectionSelector: '.pt-cn-section',
			css3:true,
			navigation:true,
			navigationPosition: 'right',
			navigationColor: '#fff',
			// scrollingSpeed: 500,
			afterLoad: function(anchorLink, index){
				if(index == 1){
					$('.pt-cn-arrow').off('click').on('click',function(){
						$.fn.fullpage.moveSectionDown();
					});
					$('#pt-cn-footer').addClass('in');
					$('#pt-cn-partner').addClass('in');
					$('#pt-cn-customer').addClass('in');
				}
			},
			onLeave: function(index, nextIndex, direction){
				if (nextIndex == 2){
					$('.pt-cn-box').removeClass('page3 page4 page5').addClass('page2').find('.pt-cn-font-wrap').addClass('active').css('background','#4a4a4a');
				}else if(nextIndex == 3){
					$('.pt-cn-box').removeClass('page2 page4 page5').addClass('page3').find('.pt-cn-font-wrap').addClass('active').css('background','#6c4cae');
				}else if(nextIndex == 4 ){
					$('.pt-cn-box').removeClass('page2 page3 page5').addClass('page4').find('.pt-cn-font-wrap').addClass('active').css('background','#8bc24a');
				}else if(nextIndex == 5){
					$('.pt-cn-box').removeClass('page2 page3 page4').addClass('page5').find('.pt-cn-font-wrap').addClass('active').css('background','#3d82f7');
				}else{
					$('.pt-cn-box').find('.pt-cn-font-wrap').removeClass('active');
				}

				// 修复通过右侧直接点击出现的字体叠加bug
				if( index==2 && direction == 'down'){
					$('.pt-cn-ptengine-font').removeClass('hidedown').addClass('hideup');
				}else if( index==2 && direction == 'up'){
					$('.pt-cn-ptengine-font').removeClass('hideup').addClass('hidedown');
				}else if( index==3 && direction == 'down'){
					$('.pt-cn-dd-font').removeClass('hidedown').addClass('hideup');
				}else if( index==3 && direction == 'up'){
					$('.pt-cn-dd-font').removeClass('hideup').addClass('hidedown');
				}else if( index==4 && direction == 'down'){
					$('.pt-cn-app-font').removeClass('hidedown').addClass('hideup');
				}else if( index==4 && direction == 'up'){
					$('.pt-cn-app-font').removeClass('hideup').addClass('hidedown');
				}else if( index==5 && direction == 'up'){
					$('.pt-cn-ths-font').removeClass('hideup').addClass('hidedown');
				}else if( index==5 && direction == 'down'){
					$('.pt-cn-ths-font').removeClass('hidedown').addClass('hideup');
				}else if(index==7 && direction== 'up'){
					var top=$("#pt-cn-partner").scrollTop();
					console.log(top);
					window.scrollTo(0,top);
				}

				if(nextIndex == 2){
					$('.pt-cn-font-animate').removeClass('show');
					$('.pt-cn-ptengine-font').addClass('show');
				}else if(nextIndex == 3){
					$('.pt-cn-font-animate').removeClass('show');
					$('.pt-cn-dd-font').addClass('show');
				}else if(nextIndex == 4){
					$('.pt-cn-font-animate').removeClass('show');
					$('.pt-cn-app-font').addClass('show');
				}else if(nextIndex == 5){
					$('.pt-cn-font-animate').removeClass('show');
					$('.pt-cn-ths-font').addClass('show');
				}else{
					$('.pt-cn-font-animate').removeClass('show');
				}
			}
		});
	});
