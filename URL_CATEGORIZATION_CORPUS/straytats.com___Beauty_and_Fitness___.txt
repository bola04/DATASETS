    var base_container = $("#base_container"),
        base_content = $("#base_content");

    function centerBaseContent() {
      if (base_content.is(":hidden")) {
        var first_run = 1;
      }

      difference = (base_container.height() - base_content.height()-50) / 2;
      if (difference > 150) {
        difference = 150;
      } else if (difference < 20) {
        difference = 20;
      }

      if (first_run) {
        //base_content.show(0);
        base_content.hide(0);
      }

      base_content.stop(true, false);
      base_content.animate({'padding-top': difference}, {'duration': delay_center});
      base_content.fadeIn(delay_fade);
    }

    // Handle centering of base_content
    $(document).ready(function() {
      //centerBaseContent();
      $(window).trigger('resize');
    });

    $(window).resize(function() {
      centerBaseContent();
    });
