    var BouncyBall = function (paper, x, y, diameter, ballImage, angle) {
        this.paper = paper;

        this.r = angle;

    BouncyBall.prototype.calculateBounce = function (xVelocity, yVelocity, finalX, finalY, gravity, bounce, decelleration, rotations, rolldecelleration, rVelocity) {
        this.vx = xVelocity;
        this.vy = yVelocity;
        this.g = gravity;
        this.fac = bounce;
        this.finalRotation = rotations * 360;
        this.finalX = finalX;
        this.finalY = finalY;
        this.xfac = decelleration;
        this.rfac = rolldecelleration;
        this.vr = rVelocity;

        this.animating = $.Deferred();
        this.loadBouncePath();
    };
    BouncyBall.prototype.loadBouncePath = function(){
        this.animationSteps = [];
        var notfinished = true;
        var currentStep = {
            vy:this.vy,
            y:this.y,
            x:this.x,
            vx:this.vx,
            vr:this.vr,
            r:this.r
        };
        while(notfinished){
            currentStep.vy += this.g;
            currentStep.y += currentStep.vy;
            if (currentStep.y > this.finalY) {
                currentStep.y = this.finalY;
                currentStep.vy *= -this.fac;
            }
            var currentX = currentStep.x;

            currentStep.vx = currentStep.vx - (currentStep.vx * this.xfac);
            if (currentStep.vx < .25 && currentStep.vx > -.25) {
                if (currentStep.vx > 0) {
                    currentStep.vx = .25;
                } else {
                    currentStep.vx = -.25;
                }
            }
            currentStep.x += currentStep.vx;
            if (Math.abs(currentStep.x - this.finalX) > Math.abs(currentX - this.finalX)) {
                if (!this.logged) {

                    this.logged = true;
                }
                currentStep.x = this.finalX;
            }

            currentStep.vr = currentStep.vr - (currentStep.vr * this.rfac);
            if (currentStep.vr < 3 && currentStep.vr > -3) {
                if (currentStep.vr > 0) {
                    currentStep.vr = 3;
                } else {
                    currentStep.vr = -3;
                }
            }
            currentStep.r += currentStep.vr;
            if (Math.abs(currentStep.r) > Math.abs(this.finalRotation)) {
                currentStep.r = this.finalRotation;
            }
            this.animationSteps.push({
                y: currentStep.y,
                x: currentStep.x,
                transform: "r " + currentStep.r
            });
            if (!(currentStep.x == this.finalX && currentStep.y == this.finalY && currentStep.r == this.finalRotation)) {
                notfinished = true;
            }else{
                notfinished = false;
            }

        }
    };
    BouncyBall.prototype.bounce = function (frame) {
        if (frame < this.animationSteps.length) {
            var animate = this.animationSteps[frame];
            this.ballInstance.attr(animate);
            return true;
        }else{
            var animate = this.animationSteps[this.animationSteps.length - 1];
            this.ballInstance.attr(animate);
            return false;
        }
    };

            } catch (err) {
                img.onload = noop; // IE7 :(
                img.onerror = noop; // IE7 :(
            }
            deferred.resolve();
        };
        img.src = src;
        return deferred;
    }
