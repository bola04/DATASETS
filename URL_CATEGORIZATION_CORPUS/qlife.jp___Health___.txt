   <div class="notice-badge-list">
      <p class="jBox_head">お知らせ</p>
        <span class="close">
            <img src="/common/img/pc/btn_badge_close.png" width="25" height="25">
        </span>
        <ul>
            {{for notices}}
                <li {{if unread != true}}class="read"{{/if}}><a href="{{:url}}" target="_blank" data-id="{{:id}}" >{{:message}}</a></li>
            {{/for}}
        </ul>
    </div>
  </script>
  <!-- javascript お知らせバッジ機能 (定義) -->
  <script>
   (function($){
        var def = {
            domain  : 'https://www.qlife.jp',
            service : 'unknown',
            count   : 'span',
            template: '#jq-tmpl-notice-badge',
            width   : 300,
            height  : 360
        };

        var NoticeBadge = function(target, option){
            var self = this;
            self.config = $.extend({}, def, option);
            self.target = target;
            self.number = $(self.config.count, self.target);
            $.when(
                self.requestList()
            ).done(function(data){
                self.setCount(data.unread)
                    .loadTooltip(data.list);
            });
        };
        $.extend(NoticeBadge.prototype, {
            loadTooltip: function(list){
                var self = this, icon = $(self.target);
                if(list.length > 0) {
                    $.when(
                        $(self.config.template).render({
                            notices: list
                        })
                    ).done(function(tmpl) {
                        var tooltip = $(tmpl),
                            jbox = new jBox('Tooltip', {
                                attach  : $(self.target),
                                trigger : 'click',
                                width   : self.config.width,
                                height  : self.config.height,
                                content : tooltip,
                                position: {
                                    x: 'center',
                                    y: 'bottom'
                                },
                                outside : 'y',
                                closeOnClick: 'body',
                                closeOnEsc  : false
                            });

                        // 閉じるボタン
                        $('span.close', tooltip).click(function(){
                            jbox.close();
                        });

                        // 各リンク
                        $('a', tooltip).click(function(){
                            var a = $(this), id = a.data('id'), li = a.parent();
                            self.requestRead(id, {
                                service   : self.config.service,
                                url       : location.href,
                                message   : a.text(),
                                link      : a.attr('href'),
                                is_unread : !a.hasClass('read')
                            });
                            if(!li.hasClass('read')){
                                li.addClass('read');
                                self.reloadCount();
                            }
                            return true;
                        });
                    });
                    icon.show();
                } else {
                    icon.hide();
                }
                return self;
            },
            setCount: function(unread){
                var self = this;
                self.number.text(unread);
                if(unread > 0){
                    self.number.show();
                }
                return self;
            },
            reloadCount: function(){
                var self = this,
                    cur = self.number.text(),
                    cnt = cur - 1;
                if (cnt >= 0) {
                    self.number.text(cnt);
                }
                if (cnt <= 0) {
                    self.number.hide();
                }
                return self;
            },
            requestList: function(){
                var defer = $.Deferred();
                $.ajax({
                    type    : 'GET',
                    url     : this.config.domain + '/members/webapi/notice.jsonp',
                    data    : {},
                    dataType: 'jsonp',
                    jsonp   : 'callback'
                }).done(function(res) {
                    defer.resolve(res);
                });
                return defer.promise(this);
            },
            requestRead: function(id, data){
                // TreasureData
                data.td_global_id = $.cookie ? ($.cookie('_td_global') || '') : '';

                var defer = $.Deferred();
                $.ajax({
                    type    : 'GET',
                    url     : this.config.domain + '/members/webapi/notice/' + id + '.jsonp',
                    data    : data,
                    dataType: 'jsonp',
                    jsonp   : 'callback'
                }).done(function(res) {
                    defer.resolve(res);
                });
                return defer.promise(this);
            }
        });
        $.fn.noticeBade = function(option) {
            return this.each(function(){
                new NoticeBadge(this, option);
            });
        };
    }(jQuery));
  </script>
