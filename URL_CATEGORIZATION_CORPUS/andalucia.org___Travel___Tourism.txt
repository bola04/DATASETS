         Buscar

        <input autocomplete="off" class="searchText" id="searchtextinput" name="q" placeholder="Buscar" type="text"/>

       <div id="autocomplete-results" style="display:none;">

      <script>
       (function($) {
    $(function() {
      var $autocomplete_results = $('#autocomplete-results');
      var $searchtextinput = $('#searchtextinput');

      $searchtextinput.on({
        keyup: function(e) {
          if (e.which == 40 || e.which == 38) {
            return false;
          }
          var last_query = '';
          var query = $(this).val();
          query = query.trim().replace(/\s+/g, ' ');
          if (query === '') {
            last_query = '';
            $autocomplete_results.empty().hide();
          } else if (query !== last_query && query.length >= 3) {
            last_query = query;
            $.ajax({
              dataType: 'json',
              url: '/ajax/autocomplete/queries/',
              data: {'q': query},
              success: function(data) {
                var results = data.results;
                if (results.length > 0) {
                  var items = [];
                  var re = new RegExp(query, 'gi');
                  for (var i = 0, l = results.length; i < l; i++) {
                    var result = results[i].replace(re, '<strong>' + query + '</strong>');
                    items.push('<li>' + result + '</li>');
                  }
                  var html_list = '<ul>' + items.join('') + '</ul>';
                  $autocomplete_results.html(html_list).show();
                }
              }
            });
          }
        },
        keydown: function(e) {
          if (e.which == 40 || e.which == 38) {
            var $list = $autocomplete_results.find('li');
            var list_length = $list.length;
            if (list_length > 0) {
              var index = $list.parent().find('.selected').index();
              var $selected;
              if (e.which == 40) {  // down arrow
                if (index === -1) {
                  $selected = $list.first();
                  $selected.addClass('selected');
                  $searchtextinput.val($selected.text());
                } else {
                  $list.parent().find('.selected').removeClass('selected');
                  index += 2;  // nth-child is 1-indexed
                  if (index <= list_length) {
                    $selected = $list.parent().find('li:nth-child(' + index + ')');
                    $selected.addClass('selected');
                    $searchtextinput.val($selected.text());
                  }
                }
              } else if (e.which == 38) { // up arrow
                if (index === -1) {
                  $selected = $list.last();
                  $selected.addClass('selected');
                  $searchtextinput.val($selected.text());
                } else {
                  $list.parent().find('.selected').removeClass('selected');
                  if (index >= 1) {  // nth-child is 1-indexed
                    $selected = $list.parent().find('li:nth-child(' + index + ')');
                    $selected.addClass('selected');
                    $searchtextinput.val($selected.text());
                  }
                }
              }
            }
          }
        }
      });

      $autocomplete_results.on({
        click: function() {
          $searchtextinput.val($(this).text());
          $('#portalsearcher').submit();
        },
        mouseenter: function() {
          $autocomplete_results.find('li').removeClass('selected');
          $(this).addClass('selected');
        },
        mouseleave: function() {
          $(this).removeClass('selected');
        }
      }, 'li');

      $('html').on({
        click: function() {
          $autocomplete_results.hide();
        }
      });

    });
  }(jQuery));
      </script>

         <script type="text/javascript">
          (function($) {
    $(document).ready(function() {
        geocoder_map = function(evt) {
            var map_main_location = $("#googlemap").data("map");
            var container = $(this).parents('form');
            var province = container.find("select[name='location__cities__province__exact']");
            var city = container.find("select[name='location__cities__exact']");
            var val_province = province.val();
            var val_city = city.val();
            if (val_province != null && val_province != "") {
                var geocoder = new google.maps.Geocoder();
                var is_localized = true;
                if (geocoder) {
                    var address = province.find("option[value=" + val_province + "]").html();
                    if (val_city != null && val_city != "") {
                        address = city.find("option[value=" + val_city + "]").html() + " " + address;
                    }
                    address += ", Andalusia Spain";
                    geocoder.geocode({'address': address}, function(results, status) {
                        if (status == google.maps.GeocoderStatus.OK) {
                                if (results.length > 0) {
                                    var lnglat = results[0].geometry.location;
                                    map_main_location.panTo(lnglat);
                                    if (val_city != null && val_city != "") {
                                        map_main_location.setZoom(11);
                                    } else {
                                        map_main_location.setZoom(9);
                                    }
                                }else {
                                    is_localized = false;
                                }
                        } else {
                            is_localized = false;
                        }
                        if (!is_localized) {
                            alert("no localizado");
                        }
                    });
                }
            }
        };

        $('#mapsearcher form').submit(function (evt){
            evt.preventDefault();
            var map_filter_value = $(this).parents('li').attr('data-id');
            var map_filter = $("input[value='" + map_filter_value + "']");
            if (map_filter.attr("checked") == "checked"){
                map_filter.attr("checked", "")
                map_filter.click();
            }
            map_filter.attr("checked", "checked");
            map_filter.click();
        });
        $("select[name='location__cities__exact']").change(geocoder_map);
        $("select[name='location__cities__province__exact']").change(geocoder_map);
    });
})(jQuery);
