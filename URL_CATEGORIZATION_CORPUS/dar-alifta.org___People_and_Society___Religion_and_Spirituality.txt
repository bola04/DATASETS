                        p.x = p.x + obj.offsetParent.offsetLeft;
                        p.y = p.y + obj.offsetParent.offsetTop;
                        if (obj == document.getElementsByTagName("body")[0]) {
                            break;
                        }
                        else {
                            obj = obj.offsetParent;
                        }
                    }
                    return p;
                }
            }
            else
            {
                document.getElementById("imgSplash").style.display = "none";
            }
        });
        function UpdateSplashScreen() {
            var index = Math.floor(Math.random() * imageURLs.length);
            $('#imgSplash').attr('src', imageURLs[index]);
        }
        function CancelSplashTimer() {
            clearInterval(timeoutid);
        }
        function ResetSplashTimer() {
            // Cancel old timer.
            CancelSplashTimer();

            // Create one new timer.
            timeoutid = setInterval('UpdateSplashScreen()', 30000);
        }
