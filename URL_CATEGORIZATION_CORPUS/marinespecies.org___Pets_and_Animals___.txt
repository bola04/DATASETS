        $(document).ready( function() {
                    $.ajax({
                        method: "POST",
                        dataType: "xml",
                        url: "aphia.php?p=rss"
                    }).done(function( xmlDoc ) {
                        $xml = $( xmlDoc );
                        //Handle global
                        var lastBuildDate = new Date($xml.find("lastBuildDate").text());
                        $('#rss_latest_taxa_last_update').append("<small><i>Updated: " + lastBuildDate.toISOString().slice(0, 16) + "</i></small>");

                        //Handle items
                        $xml.find("item").each(function(index) {
                           if (index >= 5) {
                               return;
                           }
                           var cur_item = $(this);
                           var title = cur_item.find('title').text();
                           var url = cur_item.find('link').text();
                           var author = cur_item.find('dc\\:creator').text();
                           //console.log(cur_item.find('dc\\:creator'));
                           var category = "";
                           cur_item.find('category').each(function() {
                               category += $(this).text() + " > ";
                            });
                            category = category.slice(0, category.length - 3);
                           var date = new Date(cur_item.find('pubDate').text());
                           var cur_item_id = url.match(/id=(\d+)/);

                           var html =
                               "<li class=\"list-group-item\">" +
                                    "<a href=\"" + url + "\" title=\"" + title + "\" target=\"_blank\">" + title + "</a>" +
                                    "<br />" +
                                    "<small><i>Added: " + date.toISOString().slice(0, 10) + "</i></small>" +
                                    "<div class=\"pull-right\">" +
                                        "<a href=\"http://www.marinespecies.org/imis.php?module=person&show=search&searchField=" + encodeURI(author) + "\" title=\"Author: " + author + "\" target=\"_blank\">" +
                                            "<span class=\"fa fa-user fa-1\" aria-hidden=\"true\"></span>" +
                                        "</a>&nbsp;" +
                                        "<a href=\"aphia.php?p=browser&id[]=" + cur_item_id[1] + "#focus\" title=\"" + category + "\" target=\"_blank\">" +
