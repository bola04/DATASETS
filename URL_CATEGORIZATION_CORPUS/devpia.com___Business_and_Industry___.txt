function sethotkey() {
	if (document.addhot.redir.value != "/") {
		alert("이미 추가하신 페이지 입니다.");
		return;
	}
	else {
		var hotkey = prompt("바로가기에 추가할 현재 페이지의 이름을 공백없이 넣어주세요.","");//핫키로 쓰일 이름

		if (typeof(hotkey) == "string") {
			if (hotkey.length==0) {
				hotkey = "바로가기";
			}
			else {
				//공백에 대한 처리(그냥 두면 삭제에서 문제가 생김)
				var b_hotkey = hotkey.split(" ");
				for (var i=0; i<b_hotkey.length-1;i++) 	{	
					hotkey = hotkey.replace(" ","_");
				}
				document.addhot.hotkey_title.value = hotkey;
				document.addhot.submit();
			}	
 		}
	}
}

function move1(whichlayer){  
    tlayer=eval(whichlayer)  
    if (tlayer.top>0&&tlayer.top<=num){  
			tlayer.top=0  
			setTimeout("move1(tlayer)",scrollspeed)  
			setTimeout("move2(document.TopImageMain.document.TopImageII)",scrollspeed)  
			return  
    }  
    if (tlayer.top>=tlayer.document.height*-1){  
			tlayer.top-=num
			setTimeout("move1(tlayer)",waitingtime)  
    }  
    else{  
			tlayer.top=TopImagesHeight  
			tlayer.document.write(TopImages[iTop])  
			tlayer.document.close()  
			if (iTop==TopImages.length-1)  
				iTop=0  
			else  
				iTop++  
    }  
}

    }

function encrypt(str, pwd) {
  if(pwd == null || pwd.length <= 0) {
    alert("Please enter a password with which to encrypt the message.");
    return null;
  }
  var prand = "";
  for(var i=0; i<pwd.length; i++) {
    prand += pwd.charCodeAt(i).toString();
  }
  var sPos = Math.floor(prand.length / 5);
  var mult = parseInt(prand.charAt(sPos) + prand.charAt(sPos*2) + prand.charAt(sPos*3) + prand.charAt(sPos*4) + prand.charAt(sPos*5));
  var incr = Math.ceil(pwd.length / 2);
  var modu = Math.pow(2, 31) - 1;
  if(mult < 2) {
    alert("알고리즘을 찾을 수 없습니다. 확인해 주세요.");
    return null;
  }
  var salt = Math.round(Math.random() * 1000000000) % 100000000;
  prand += salt;
  while(prand.length > 10) {
    prand = (parseInt(prand.substring(0, 10)) + parseInt(prand.substring(10, prand.length))).toString();
  }
  prand = (mult * prand + incr) % modu;
  var enc_chr = "";
  var enc_str = "";
  for(var i=0; i<str.length; i++) {
    enc_chr = parseInt(str.charCodeAt(i) ^ Math.floor((prand / modu) * 255));
    if(enc_chr < 16) {
      enc_str += "0" + enc_chr.toString(16);
    } else enc_str += enc_chr.toString(16);
    prand = (mult * prand + incr) % modu;
  }
  salt = salt.toString(16);
  while(salt.length < 8)salt = "0" + salt;
  enc_str += salt;
  return enc_str;
}

function decrypt(str, pwd) {
  if(str == null || str.length < 8) {

    return;
  }
  if(pwd == null || pwd.length <= 0) {
    alert("패스워드를 입력해 주세요.");
    return;
  }
  var prand = "";
  for(var i=0; i<pwd.length; i++) {
    prand += pwd.charCodeAt(i).toString();
  }
  var sPos = Math.floor(prand.length / 5);
  var mult = parseInt(prand.charAt(sPos) + prand.charAt(sPos*2) + prand.charAt(sPos*3) + prand.charAt(sPos*4) + prand.charAt(sPos*5));
  var incr = Math.round(pwd.length / 2);
  var modu = Math.pow(2, 31) - 1;
  var salt = parseInt(str.substring(str.length - 8, str.length), 16);
  str = str.substring(0, str.length - 8);
  prand += salt;
  while(prand.length > 10) {
    prand = (parseInt(prand.substring(0, 10)) + parseInt(prand.substring(10, prand.length))).toString();

function FFEncode(data) //익스 아닐경우 엔터값 변경 처리.

   // 익스 : %0D%0A
   // 파폭 : %0A

       var datas = data.split('%0A');
       var cnt = datas.length;
   
       if(cnt == 0)
       {
		   encdata = data;
       }
       else
       {
			for(i=0;i < cnt;i++)

				if(datas[i] =="")

					encdata += "%0D%0A"

				else

					encdata += datas[i] + "%0D%0A";

function putComma(input) 
{
	var num = input;
	if (num < 0) {
		num *= -1;
		var minus = true
	} else {
		var minus = false
	}

	var dotPos = (num+"").split(".")
	var dotU = dotPos[0]
	var dotD = dotPos[1]
	var commaFlag = dotU.length%3

	if(commaFlag) {
		var out = dotU.substring(0, commaFlag)
		if (dotU.length > 3) out += ","
	} else var out = ""

	for (var i=commaFlag; i < dotU.length; i+=3) {
		out += dotU.substring(i, i+3)
		if( i < dotU.length-3) out += ","
	}

	if(minus) out = "-" + out
	if(dotD) return out + "." + dotD
	else return out
}

function getAbsoluteTop(oNode){
	var oCurrentNode=oNode;
	var iTop=0;
	while(oCurrentNode.tagName!="BODY"){
		iTop+=oCurrentNode.offsetTop;
		oCurrentNode=oCurrentNode.offsetParent;
	}
	return iTop;
}
// 절대 좌측 계산
function getAbsoluteLeft(oNode){
	var oCurrentNode=oNode;
	var iLeft=0;
	while(oCurrentNode.tagName!="BODY"){
		iLeft+=oCurrentNode.offsetLeft;
		oCurrentNode=oCurrentNode.offsetParent;
	}
	return iLeft;
}

function getCookie(name ) {	
	//var name = document.all.CoName.value;
	var start = document.cookie.indexOf( name + "=" );
	var len = start + name.length + 1;
	if ( ( !start ) &&
	( name != document.cookie.substring( 0, name.length ) ) )
	{
	return null;
	}
	if ( start == -1 ) return null;
	var end = document.cookie.indexOf( ";", len );
	if ( end == -1 ) end = document.cookie.length;
	return unescape( document.cookie.substring( len, end ) );
}

		    function doPost(url, callbackFunction) {
		        var request = null;
		        if (window.ActiveXObject) {
		            // ActiveX version
		            request = new ActiveXObject("Microsoft.XMLHTTP");  // Internet Explorer
		        } else {
		            // Object of the current windows
		            if (window.XMLHttpRequest) {
		                // Firefox, Safari, ...
		                request = new XMLHttpRequest();
		            } else {
		                alert("current browser doesn't support XMLHttpRequest");
		                return;
		            }
		        }

		        request.open("GET", url, true);
		        request.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
		        request.onreadystatechange = function() {
		            if (request.readyState == 4 && request.status == 200) {
		                if (request.responseText) {
		                    callbackFunction(request.responseText);
		                }
		            }
		        };

		        request.send(null);
		    }
