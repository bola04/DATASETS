            function cases_init(){
 cases_content('303','1','10',true,'RECENT', 1,'ALL_CASES');
}
function cases_init_items(){
 cases_load_items();
}
function cases_content(contentId, pageNr, pageSize, showPageNavigation, sortType, loadType, casesShowType){
if (loadType == 1){
cases_load();
} else if (loadType == 2){
cases_load_items();
}
$.ajax({
url: '/cms/asynchronousrendering/ModuleCoCreatie/CmsShowCoCreatieCasesRenderControl.aspx?id=45196', data: 'contentId=' + contentId + '&pageNr=' + pageNr + '&pageSize=' + pageSize + '&showPageNavigation=' + showPageNavigation + '&sortType=' + sortType + '&casesShowType=' + casesShowType + '', type: 'get', cache: false, error: function(xhr, ajaxOptions, thrownError){cases_error(xhr, ajaxOptions, thrownError);}, success: function(data){
     var jsonObject = eval('(' + data + ')');
     $('#cases_order_by_most_recent_link').attr('href', 'javascript:cases_content('+contentId+','+pageNr+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',"RECENT", 2,"'+casesShowType+'");');
     $('#cases_order_by_highest_rating_link').attr('href', 'javascript:cases_content('+contentId+','+pageNr+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',"RATING", 2,"'+casesShowType+'");');
     $('#cases_order_by_most_votes_links').attr('href', 'javascript:cases_content('+contentId+','+pageNr+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',"VOTES", 2,"'+casesShowType+'");');
     $('#cases_order_by_name_link').attr('href', 'javascript:cases_content('+contentId+','+pageNr+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',"NAME", 2,"'+casesShowType+'");');
     $('#cases_order_by_author_link').attr('href', 'javascript:cases_content('+contentId+','+pageNr+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',"AUTHOR", 2,"'+casesShowType+'");');
     var casesHtml = '';
     if (jsonObject.items.length > 0){
         casesHtml += '<ul>';
         for (var itemIndex = 0; itemIndex < jsonObject.items.length; itemIndex++){
             var item = jsonObject.items[itemIndex];
             var bedrijfsnaam = (item.CreatorBedrijf == '') ? '<span class="bedrijf"></span>' : '<span class="bedrijf">, ' + unescape(item.CreatorBedrijf) + '</span>';
             var createdDate = new Date(parseInt(item.Created.substr(6)));
             var toevoegHtml = '';
             if (jsonObject.magCaseAanmaken)
             {
                 toevoegHtml = '<div class="project_toevoegen"><a href="/ledenpodium/artikel-toevoegen" title="Voeg jouw artikel toe">Voeg jouw artikel toe</a></div>';
             }
             var itemHtml = '<li><span class="case_titel"><a href="/cms/modules/ModuleCoCreatie/RedirectToCase.aspx?caseId=' + item.ID + '">' + unescape(item.Titel) + '</a></span><br/><div class="case"><div class="foto_en_likes"><div class="foto"><a href="/cms/modules/ModuleCoCreatie/RedirectToCase.aspx?caseId=' + item.ID + '"><img src="' + item.PictureUri + '" alt="' + unescape(item.Titel) + '" /></a></div><div class="likes_en_dislikes"><span id="like_' + item.ID + '_container" onclick="cases_likedislike_content(\'like_' + item.ID + '_container\',\'dislike_' + item.ID + '_container\',\'like\',' + item.ID + ');"  class="likes">' + item.Likes + 'x</span><span id="dislike_' + item.ID + '_container" onclick="cases_likedislike_content(\'like_' + item.ID + '_container\',\'dislike_' + item.ID + '_container\',\'dislike\',' + item.ID + ');" class="dislikes">' + item.Dislikes + 'x</span></div></div><div class="profiel_en_omschrijving"><strong class="profiel"><a href="' + item.CreatorProfileUri + '">' + unescape(item.CreatorNaam) + '</a>' + bedrijfsnaam + '</strong><span class="aanmaakdatum">' + createdDate.toLocaleString() + '</span><div class="introtekst">' + unescape(item.Introtekst) + '<a href="/cms/modules/ModuleCoCreatie/RedirectToCase.aspx?caseId='+item.ID + '" class="meerLink">Lees verder</a></div><div class="meer"><a href="/cms/modules/ModuleCoCreatie/RedirectToCase.aspx?caseId=' + item.ID + '">' + unescape(item.Titel) + '</a></div></div>' + toevoegHtml + '</div></li>';
             casesHtml += itemHtml;
         }
         casesHtml += '</ul>';
     } else {
         casesHtml += '<div class="tekst empty">Er zijn momenteel nog geen artikelen!</div>'
     }
     if (jsonObject.showPageNavigation && jsonObject.numberOfPages > 1){
         var pagesHtml = '';
         for (var pageIndex = 1; pageIndex <= jsonObject.numberOfPages; pageIndex++){
             if (pageIndex == jsonObject.pageNr){
                 pageHtml = '<span class="aktief">' + pageIndex + '</span>';
             } else {
                 pageHtml = '<a href="#" onclick="cases_content('+contentId+','+pageIndex+','+jsonObject.pageSize+','+jsonObject.showPageNavigation+',\''+jsonObject.sortType+'\', 2,\''+jsonObject.casesShowType+'\'); return false;">' + pageIndex + '</a>';
                 
             }
             pagesHtml += pageHtml;
         }
         $('#cases_paginering').html(pagesHtml);
     } else {
         $('#cases_paginering').html('');
     }
     $('#cases_rows').html(casesHtml);
     if (loadType == 1){
         cases_loaded();
     } else if (loadType == 2){
         cases_loaded_items();
     }
   }
 });
}
function cases_likedislike_content(like_clientId, dislike_clientId, action, caseId){
$.ajax({
global: false, url: '/cms/asynchronousrendering/ModuleCoCreatie/CmsShowCoCreatieCaseLikeDislikeRenderControl.aspx?id=45196', data: 'action=' + action + '&CoCreatieCaseId=' + caseId + '', type: 'get', cache: false, success: function(data){
     var jsonObject = eval('(' + data + ')');
     document.getElementById(like_clientId).innerHTML = jsonObject.likes + 'x';
     document.getElementById(dislike_clientId).innerHTML = jsonObject.dislikes + 'x';
}
});
}
function cases_load(){
 $('#cases_control').removeClass('loaded').addClass('loading');
 $('#cases_control .loadable_content').fadeOut(100, function(){
     $('#cases_control .loadable_content').css('filter','');
 });
 }
function cases_load_items(){
 $('#cases_control').removeClass('loaded_items').addClass('loading_items');
 $('#cases_control .loadable_content_items').fadeOut(100, function(){
     $('#cases_control .loadable_content_items').css('filter','');
 });
 }
function cases_loaded(){
 $('#cases_control').removeClass('loading').addClass('loaded loaded_items');
 $('#cases_control .loadable_content').fadeIn(500, function(){
     $('#cases_control .loadable_content').css('filter','');
 });
}
function cases_loaded_items(){
 $('#cases_control').removeClass('loading_items').addClass('loaded_items');
 $('#cases_control .loadable_content_items').fadeIn(500, function(){
     $('#cases_control .loadable_content_items').css('filter','');
 });
}
function cases_error(xhr, ajaxOptions, thrownError){
$('#cases_control').html('<div class="tekst" control="cases">Inhoud kon niet worden geladen</div>');cases_loaded();
}
function cases_error_items(xhr, ajaxOptions, thrownError){
$('#cases_control').html('<div class="tekst" control="cases">Inhoud kon niet worden geladen</div>');cases_loaded_items();
}
