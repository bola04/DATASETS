        function getAvatarNickname(nickname) {
            if ( ! nickname) {
                nickname = document.querySelector('.nav-popover-username');

                if ( ! nickname) {
                    return '';
                }

                nickname = nickname.title;
            }

            if ( ! nickname) {
                return '';
            }

            if (nickname.match(/[a-zA-Z]/g)) {
                nickname = nickname.match(/[a-zA-Z]/g).join('');
            }

            nickname = nickname.slice(0,2);
            nickname = nickname.charAt(0).toUpperCase() + nickname.charAt(1).toLowerCase();

            return nickname || '';

        function updateCount() {
            setTimeout(function() {
                var memCount = (users_element.getAttribute('data-count') || '199,981').replace(',', '').replace('.', '') * 1;

                if (Math.floor(Math.random() * 10) % 2 == 0) { // UP or DOWN count
                    --memCount;
                } else {
                    ++memCount;
                }

                memCount = memCount > 0 ? memCount : 0;

                users_element.setAttribute('data-count', memCount);
                users_element.innerHTML = (memCount * 1).toLocaleString('en-US');

                updateCount();
            }, -(1000 / 0.5) * Math.log(Math.random()));
        }

        updateCount();
        setPositionToContent();
        cloneNavigationToFixed();
        setParallaxToElements();

        // Add event to click on navigation slider
        [].slice.call(document.querySelectorAll('#support-dev-nav > div'))
            .forEach(function(item) {
                addEvent(item, 'click', function() {
                    var self = this;

                    var slide_element = document.getElementById(this.id.replace('support-icon-', 'support-slide-'));

                    if (this.hasClass('active') &&
                        slide_element.hasClass('active')) {
                        return;
                    }

                    var label = document.querySelector('#support-dev-nav > .active');
                    if (label) {
                        label.removeClass('visible').removeClass('active');
                    }

                    var slide = document.querySelector('#support-dev-slides > .active');
                    if (slide) {
                        slide.removeClass('visible').removeClass('active');
                    }

                    this.addClass('visible');
                    slide_element.addClass('visible');

                    setTimeout(function() {
                        self.addClass('active');
                        slide_element.addClass('active');
                    }, 0);
                });
            });

        setOsParamsIndexPage();

        setInterval(function() {
            var next = {
                img: document.querySelector('#testimonials-photo > img.active + img')
                , label: document.querySelector('#testimonials-content > .testimonials-content-lay.active + .testimonials-content-lay')
            };

            document.querySelector('#testimonials-photo > img.active').removeClass('active');
            document.querySelector('#testimonials-content > .testimonials-content-lay.active').removeClass('active');

            (next.img || document.querySelector('#testimonials-photo > img')).addClass('active');
            (next.label || document.querySelector('#testimonials-content > .testimonials-content-lay')).addClass('active');
        }, 4000);
    });

    // Default video sizes
    var vidH = 404;
    var vidW = 718;

    // Parallax transition length

    // Parallax transition speed

    /**
     * Set Position to all content (include parallax content)
     */
    function setPositionToContent() {
        var div_video = document.getElementById('video-background');
        var video_replacer = document.querySelector('#video-background > img');
        var tag_video = document.querySelector('#video-background > video');
        var video_overlay = document.getElementById('video-overlay');

        var calculate_styles = {
            add_class: '',
            margin_left: 0,
            margin_top: 0
        };

        var winH = window.innerHeight || document.documentElement.clientHeight || document.getElementsByTagName('body')[0].clientHeight;
        var winW = window.innerWidth || document.documentElement.clientWidth || document.getElementsByTagName('body')[0].clientWidth;

        var header_height = document.getElementById('header').clientHeight;

        var welcome_content_height = document.getElementById('welcome-content').clientHeight;

        var testimonials_block = 100;

        if (winW <= 800 || current_os == 'android') {
            min_height = 870;
            testimonials_block = 200;
        }

        if (winH < min_height) {
            winH = min_height;
        }

        if ( (vidH/vidW) > (winH/winW) ) {
            calculate_styles = {
                add_class: 'portrait',
                margin_left: 0,
                margin_top: ((winH - winW * vidH / vidW) / 2)
            };
        } else {
            calculate_styles = {
                add_class: 'landscape',
                margin_left: ((winW - winH * vidW / vidH) / 2),
                margin_top: 0
            };
        }

        div_video.style.width = winW + 'px';
        div_video.style.height = winH + 'px';
        div_video.style.marginTop = -header_height + 'px';

        // Wellcome section
        document.getElementById('welcome-section').style.height = winH - header_height + 'px';
        document.getElementById('welcome-content').style.top = (winH - welcome_content_height - (isBannedDevices() ? testimonials_block : (offset_static + testimonials_block))) / 2 + 'px';

        if ( ! video_overlay) {
            return;
        }

        video_overlay.style.height = winH + 'px';

        if (isBannedDevices()) {
            video_replacer
                .removeClass('portrait')
                .removeClass('landscape')
                .addClass(calculate_styles.add_class)
                .removeClass('hide');

            video_replacer.style.marginLeft = calculate_styles.margin_left + 'px';
            video_replacer.style.marginTop = calculate_styles.margin_top + 'px';

            tag_video.addClass("hide");
        } else {
            tag_video
                .removeClass('portrait')
                .removeClass('landscape')
                .addClass(calculate_styles.add_class)
                .removeClass('hide');

            tag_video.style.marginLeft = calculate_styles.margin_left + 'px';
            tag_video.style.marginTop = calculate_styles.margin_top + 'px';

            video_replacer.addClass("hide");
        }

        // Parallax section
        var parallax_img = document.getElementById('parallax-img');

        if (parallax_img) {
            parallax_img.style.marginTop = -header_height + 'px';
            parallax_img.style.paddingBottom = header_height + 'px';
        }
    }


    /**
     * Parallax elements on page
     */
    function setParallaxToElements() {
        var parallax_img = document.getElementById('parallax-img');
        var offset = window.pageYOffset;

        if (offset > offset_img * speed || isBannedDevices()) {
            parallax_img.style.marginTop = -offset_static + 'px';
            parallax_img.style.paddingBottom = '0';
        } else {
            parallax_img.style.marginTop = -offset_static - offset_img + Math.floor(offset / speed) + 'px';
            parallax_img.style.paddingBottom = offset_img - Math.floor(offset / speed) + 'px';
        }

        showFixedNavigation();
    }


    /**
     * SlideDown/SlideUp fixed menu
     */
    function showFixedNavigation() {
        if ( ! isBannedDevices()) {
            if (window.pageYOffset > min_fixed_length) {
                document.getElementById('nav-fixed').addClass('show');
            } else {
                document.getElementById('nav-fixed').removeClass('show');
            }
        }
    }

    function newHomePage(variant) {
        if (['classic', 'video', 'autoplay'].indexOf(variant) < 0) {
            return;
        }

        document.getElementById('wrapper').addClass(variant);
        if (variant == 'autoplay') {
            document.getElementById('wrapper').addClass('video');
        }
        document.getElementById('video-background').innerHTML = '';

        if (variant == 'classic') {
            return;
        }

        var autoplay = '';

        if ((/Chrome/.test(navigator.userAgent) && /Google Inc/.test(navigator.vendor)) ||
            (/Safari/.test(navigator.userAgent) && /Apple Computer/.test(navigator.vendor))) {

            autoplay = '';
        } else if (variant == 'autoplay') {

            autoplay = 'autoplay="autoplay"';
        }

        library_data.empty.div.innerHTML = document.getElementById('template-video').innerHTML
            .replacePHs('autoplay', autoplay);

        setTimeout(function() {
            document.getElementById('welcome-content').appendChild(library_data.empty.div.children[0]);
        }, 0);
    }
