           <script>
            var slideIndex = 0;
    // Change image every 3,5 seconds
    var carouselTimeout = 3500;
    carousel();

    function plusDivs(n)
    {
        setPosition(slideIndex + n);
        showDivs();
        // stop carousel;
        carouselTimeout = null;
    }

    function pickDiv(n)
    {
        setPosition(n);
        showDivs(n);
        // stop carousel;
        carouselTimeout = null;
    }

    function showDivs()
    {
        var i;
        var slides = document.getElementsByClassName("banner-slide");

        for (i = 0; i < slides.length; i++)
        {
            slides[i].style.display = "none";
        }
        slides[slideIndex-1].style.display = "block";

        var badges = document.getElementsByClassName("badge");
        for (i = 0; i < badges.length; i++)
        {
            badges[i].className = badges[i].className.replace(" active", "");
        }

        badges[slideIndex-1].className += " active";
    }

    function setPosition(n)
    {
        var slides = document.getElementsByClassName("banner-slide");
        if (n > slides.length)
        {
            slideIndex = 1;
        }
        else if (n < 1)
        {
            slideIndex = slides.length;
        }
        else
        {
            slideIndex = n;
        }
    }

    function carousel()
    {
        if (carouselTimeout)
        {
            setTimeout(carousel, carouselTimeout);
            setPosition(slideIndex + 1);
        }

        showDivs(slideIndex);
    }
           </script>
