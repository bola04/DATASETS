  <script>
   function data_display() {

          d3.select('svg').remove();
          var tip = null;

                  var lowwater = 0.55;
        
                  var medwater = 3.5;
        
          var addscale = 1;

          var margin = {top: 20, right: 20, bottom: 200, left: 40},
              ww = document.getElementById("waterlevel").clientWidth,
              width = ww - margin.left - margin.right,
              height = 500 + margin.top - margin.bottom;

          var x = d3.scale.ordinal()
              .rangeRoundBands([0, width], .1);

          var y = d3.scale.linear()
              .range([height, 0]);

          var xAxis = d3.svg.axis()
              .scale(x)
              .orient("bottom");

          var yAxis = d3.svg.axis()
              .scale(y)
              .orient("left")
              .ticks(5, "M");

          var svg = d3.select("#waterlevel").append("svg")
              .attr("width", width + margin.left + margin.right)
              .attr("height", height + margin.top + margin.bottom)
              .append("g")
              .attr("transform", "translate(" + margin.left + "," + margin.top + ")")

          var div = d3.select("#waterlevel").append("div")
              .attr("class", "tool-tip")
              .style("opacity", 0);

          var hoverOn = function () {
              var theClass = $(this).attr('class');
              $('.waterlevelkey li.' + theClass + ', rect.' + theClass).addClass('hover');
          };
          var hoverOff = function () {
              $(this).removeClass('hover');
              var theClass = $(this).attr('class');
              $('.waterlevelkey li.' + theClass + ', rect.' + theClass).removeClass('hover');
          };


          $('.waterlevelkey li').hover(hoverOn, hoverOff);


          d3.csv("http://www.riverdee.org.uk/river-station/park.csv", type, function (error, data) {
              if (error) throw error;
              x.domain(data.map(function (d) {
                  return d.letter;
              }));
              y.domain([0, d3.max(data, function (d) {
                  return d.levelVal + addscale;
              })]);

              // high level background
              svg.append("rect")
                  .attr("y", -20) // start at the top of the graph (it's offset 20px)
                  .attr("x", 0)
                  .attr("width", width)
                  .attr("height", height + 20) // height of the graph is fine, each layer overlays this and if it's off the top, nobody sees it anyway!
                  .attr("class", "highlevel")
                  .on("mouseover", hoverOn)
                  .on("mouseout", hoverOff)

              // medium level background
              svg.append("rect")
                  .attr("y", function (d) {
                      return y(medwater);
                  })
                  .attr("x", 0)
                  .attr("width", width)
                  .attr("height", function (d) {
                      return height - y(medwater);
                  })
                  .attr("class", "medlevel")
                  .on("mouseover", hoverOn)
                  .on("mouseout", hoverOff);

              // low level background
              svg.append("rect")
                  .attr("y", function (d) {
                      return y(lowwater);
                  })
                  .attr("x", 0)
                  .attr("width", width)
                  .attr("height", function (d) {
                      return height - y(lowwater);
                  })
                  .attr("class", "lowlevel")
                  .on("mouseover", hoverOn)
                  .on("mouseout", hoverOff);

              svg.append("g")
                  .attr("class", "x axis")
                  .attr("transform", "translate(0," + height + ")")
                  .call(xAxis)
                  .selectAll("text")
                  .attr("y", 0)
                  .attr("x", 15)
                  .attr("dy", ".35em")
                  .attr("transform", "rotate(90)")
                  .style("text-anchor", "start");

              svg.append("g")
                  .attr("class", "y axis")
                  .call(yAxis)
                  .append("text")
                  .attr("transform", "rotate(-90)")
                  .attr("y", -50)
                  .attr("x", -100)
                  .attr("dy", "2em")
                  .style("text-anchor", "end")
                  .text("River Level SG(m)");

              svg.selectAll(".bar")
                  .data(data)
                  .enter().append("rect")
                  .attr("class", "bar")
                  .attr("x", function (d) {
                      return x(d.letter);
                  })
                  .attr("width", x.rangeBand())
                  .attr("y", function (d) {
                      return y(d.levelVal);
                  })
                  .attr("height", function (d) {
                      return height - y(d.levelVal);
                  })
                  .on("mouseover", function (d) {
                      div.transition()
                          .duration(200)
                          .style("opacity", 1);
                      div.html(d.levelVal + "(m)")
                          .style("left", (x(d.letter) + 15 ) + "px")
                          .style("top", (y(d.levelVal) + 15 ) + "px");
                  })
                  .on("mouseout", function (d) {
                      div.transition()
                          .duration(500)
                          .style("opacity", 0);
                  });
          });
          function type(d) {
              d.levelVal = +d.levelVal;
              return d;
          }
      }

      data_display();
      d3.select(window).on('resize', resize);
      function resize() {
          data_display();
      }
  </script>
