function initPromoPopup() {
	/**
	 *	Promo Popup
	 */
	// conditions to make popup.
	if (jQuery.cookie) {
        var _promo_hide = jQuery.cookie('_ph');
        var _frontend = jQuery.cookie('frontend');
        var _promo_hide = parseInt(_promo_hide);
        _promo_hide = (isNaN(_promo_hide))? 0 : _promo_hide;

        // conditions that would make popup possible, thus disable fanplayr
        var badUrlConditions = (document.location.href.indexOf('customer/account') != -1 || document.location.href.indexOf('checkout') != -1 || document.location.href.indexOf('bronto') != -1);
        var loggedIn = !(jQuery('a[title="Log Out"]').length == 0 && jQuery('.welcome-msg a').length == 0);
        var mobile = (window.screen.width <= 640);

        if (!badUrlConditions && !loggedIn && _frontend == null && !mobile) {
            if (_promo_hide == 0 || isNaN(_promo_hide)) {
                setTimeout('canDisplayPromoPopup()', 5000); // 5 seconds
            } else if (_promo_hide < 0) {
                // display the slide down
                jQuery('#PromoPopup').appendTo('.header-container2');
                jQuery('#PromoPopupClose').click(closePromoPopup);
                jQuery('#PromoEmailField').click(clickPromoEmailField);
                jQuery('#PromoEmailField').blur(blurPromoEmailField);
                jQuery('#PromoFirstnameField').click(clickPromoFirstnameField);
                jQuery('#PromoFirstnameField').blur(blurPromoFirstnameField);
                jQuery('#PromoLastnameField').click(clickPromoLastnameField);
                jQuery('#PromoLastnameField').blur(blurPromoLastnameField);
                jQuery('#PromoEmailContainer span').click(promoSubmit);
                jQuery('.promo_field').keypress(function(e) {
                    if ( e.which == 13 ) {
                        e.preventDefault();
                        promoSubmit();
                    }
                });
                setTimeout('displayPromoPopup()', 1000); // 1 seconds
            }
        }
    }
}

    var ph = jQuery.cookie('_ph');
    if (ph == null) {

    }
}

function promoSubmit() {
	var d = new Object();
	var email = jQuery('#PromoEmailField').val();
	var firstname = jQuery('#PromoFirstnameField').val();
	var lastname = jQuery('#PromoLastnameField').val();
	if (email.length == 0 || email.indexOf('E-Mail') != -1) {
		jQuery('#PromoMessage').text('Invalid E-Mail: "' + email + '"').addClass('bad').show();
		alert('Invalid E-Mail: "' + email + '"');
		return;
	}
	firstname = firstname.replace('First name', '');
	lastname = lastname.replace('Last name', '');
	d.email = email;
	d.firstname = firstname;
	d.lastname = lastname;
	jQuery('#PromoBlock').show();
	jQuery.ajax({
	  type: "POST",
	  url: 'https://www.seattlecoffeegear.com/newsletter/promo/subscribe/',
	  data: d,
	  success: function(data) {
	  		jQuery('#PromoBlock').hide();
		  if (data.error) {
		  	jQuery('#PromoMessage').text(data.message).addClass('bad').show();
		  	jQuery.cookie('_ph', 3, { expires: 1, path: '/', domain: '.seattlecoffeegear.com' });
		  } else if (data.success) {
			jQuery('#PromoMessage').text(data.message).removeClass('bad').show();
			jQuery('.promo_container').hide();
			jQuery.cookie('_ph', 2, { expires: 360, path: '/', domain: '.seattlecoffeegear.com' });
			setTimeout('document.location.href = "/customer/account";', 3000);
		  } else {
		  	  jQuery('#PromoMessage').text(data.message).removeClass('bad').show();
			  showNameFields();
		  }
	  },
	  dataType: 'JSON'
	});
}

}
