   var first = "";

    	success: function(data) {
      	document.getElementById("crime").innerHTML = data;
        $("h1").slabText({});
	},

   function generate(){
	   if(first==""||second==""){
		   generateurl = 'http://theyfightcrime.net/generate.php';
	   } else {
			vars = "?first=" + first + "&second=" + second + "&g1=" + g1 + "&g2=" + g2;
			uri = encodeURI(vars);
			generateurl = 'http://theyfightcrime.net/generate.php'+uri;
	   }
	   document.getElementById("crime").style.opacity = 0;
	$.ajax({
   		url: generateurl,
     	success: function(data) {
      	document.getElementById("crime").innerHTML = data;
        $("h1").slabText({});
		document.getElementById("crime").style.opacity = 1;
	},
   });  
   }
   
   function generateNames(){
	   first = document.getElementById("firstname").value;
	   first = first.replace("?", "");
	   second = document.getElementById("secondname").value;
	   second = second.replace("?", "");
	   g1 = document.getElementById("g1").value;
	   g2 = document.getElementById("g2").value;
	   if(first==null||first==""||second==null||second==""){
		   document.getElementById("empty-error").style.display = "block";
	   } else {
		   
		   document.getElementById("addNames").style.display = "none";
		   document.getElementById("clear-names").style.display = "inline-block";
		   	document.getElementById("crime").style.opacity = 0;
		   	$('#nameModal').modal('toggle');
			document.getElementById("empty-error").style.display = "none";
			vars = "?first=" + first + "&second=" + second + "&g1=" + g1 + "&g2=" + g2;
			uri = encodeURI(vars);
			$.ajax({
				url: 'http://theyfightcrime.net/generate.php'+uri,
				success: function(data) {
				document.getElementById("crime").innerHTML = data;
				$("h1").slabText({});
				document.getElementById("crime").style.opacity = 1;
			},
		});
	}
   }
