         <div style="float: left;">

          </a>
         </div>
         <i>
          Jun 14, 2018.
         </i>
         <p>
          FlyBase is pleased to introduce our new “Pathway reports”.
    These form a subset of our Gene Group resource, which houses functional collections of genes such as complexes and gene families.
    Pathway reports list genes that have been experimentally shown to act within a pathway or to regulate a pathway.
    The reports are tailored to display information that may be useful to understanding the role of a particular pathway component or regulator.

    function setQualifierInputs( anchorObj, tab, visible ) {
      //  console.log('[setQualifierInputs] got visible='+visible);
        // iterate over '[tab]InputsWrapper' child nodes
        for( var n in document.getElementById(tab+'InputsWrapper').childNodes ) {
            var node = document.getElementById(tab+'InputsWrapper').childNodes[n];
            // for those nodes with a class attribute containing 'hidden_qualifier', alter style.display
            if( node.className && node.className.match('hidden_qualifier') ) {
                node.style.display = (visible=='true') ? 'block' : 'none';
              //  console.log('hidden_qualifier node found; setting display to '+node.style.display);
            }
        }
        // toggle the sign & message
        var imgSign = 'plus';
        var message = "refine search by adding qualifier terms";
        if( visible=='true' ) {
            imgSign = 'minus';
            message = "don't use qualifier terms";
            // need to also clear qualifier fields in this case - can I do this in loop above?
            // do this in toggle handler
        }
        anchorObj.innerHTML = '<img class="toggle_button" src="/images/reports/'+imgSign+'.png" style="margin: 0 1px"><span> '+message+'</span>';
    }

    function orthologTabInputSpeciesResponse( inputSpSelect ) {
        $.cookie("QS_orthoinput", inputSpSelect.value, { expires: 365 } );
        $('#orthodboutdiv').hide();
        $('#dioptoutdiv').show();
        var tText = '';
        if(inputSpSelect.value=='Dmel') {
            tText = "e.g. Cdk1, CG5363, FBgn0004106, 34411";
            $('#orthoOutputSwitchMsg').show();
        }
        else {
            $('#orthoOutputSwitchMsg').hide();
            $('#orthodboutToggle').prop('checked',false);
            tText = "symbol(s) or ID(s)";
            if( inputSpSelect.value=='Hsap' ) tText = "e.g. CDK1, HGNC:1722, OMIM_GENE:116940, 983";
            if( inputSpSelect.value=='Rnor' ) tText = "e.g. Cdk1, RGD:2319, 54237";
            if( inputSpSelect.value=='Mmus' ) tText = "e.g. Cdk1, MGI:88351, 12534";
            if( inputSpSelect.value=='Xtro' ) tText = "e.g. cdk1, XB-GENE-482750, 394503";
            if( inputSpSelect.value=='Drer' ) tText = "e.g. cdk1, ZDB-GENE-010320-1, 80973";
            if( inputSpSelect.value=='Cele' ) tText = "e.g. cdk-1, WBGene00000405, 176374";
            if( inputSpSelect.value=='Atha' ) tText = "e.g. CDC2, AT3G48750";
            if( inputSpSelect.value=='Scer' ) tText = "e.g. CDC28, S000000364, 852457";
            if( inputSpSelect.value=='Spom' ) tText = "e.g. cdc2, 2539869";
        }
        $('#ortholog').attr('placeholder',tText); $('#ortholog').attr('title',tText);
        // responses based on sp.
      // console.log('disabling '+inputSpSelect.value);
        $("#dioptoutdiv input").attr('disabled',false);  // clear disabling
        $("#dioptoutdiv label").removeAttr('style');     // clear greys
        $("#dioptoutdiv input[value='"+inputSpSelect.value+"']").attr('disabled',true);
      //  $("#dioptoutdiv input[value='"+inputSpSelect.value+"']").prop('checked',false);
        $("#dioptoutdiv label[for='dioptoutput_"+inputSpSelect.value+"']").css('color','grey');
      //  $("input[name='fld']").val(inputSpSelect.value+'-SYMBOL');
        inputSpSelect.form.fld.value = inputSpSelect.value+'-SYMBOL';
    };
