			if (date_str) {
				// assumes date_str has format dd/mm/yyyy
		
				var date_parts = date_str.split('/');
				
				var day = parseInt(date_parts[0]);
				var month = parseInt(date_parts[1]);

		function adjust_drop_off_date () {
		
			// Update drop off date?
			
			// check if drop off date is before pick up date
			
			var pick_up_date_str = $('#pu_date').val();
			var drop_off_date_str = $('#do_date').val();

				force_update = true;

		function pad2(number) {
			return (number < 10 ? '0' : '') + number
		}

		var default_forward_days = '3';

		function add_number_of_days_to_date( datestr, numDays) {
		
			//assuming datestr has the format d/m/y
			
			var pos_day_month_sep = datestr.indexOf("/",0);
			
			var day = parseInt(datestr.substring(0, pos_day_month_sep), 10);  // Day
			
			var pos_month_year_sep = datestr.indexOf("/",pos_day_month_sep+1);
			
			var month = parseInt(datestr.substring(pos_day_month_sep+1, pos_month_year_sep), 10); // Month
			
			var year = parseInt(datestr.substring(pos_month_year_sep+1), 10);  // Year
			
			var dt = new Date(year, month-1, day); 
						  
			dt.setDate(dt.getDate() + parseInt(numDays));
			
			var updated_day = dt.getDate();
			var updated_month = dt.getMonth()+1;
			var updated_year = dt.getFullYear();
			
			updated_day = pad2 ( updated_day );
			updated_month = pad2 ( updated_month );
			updated_year = dt.getFullYear();
			
			updated_date = updated_day + "/" + updated_month + "/" + updated_year;
							  
			var results = new Array ();
			results['updated_date'] = updated_date;
			results['year'] = updated_year;
			results['month'] = updated_month;
			results['day'] = updated_day;
			
			return results;
		}
