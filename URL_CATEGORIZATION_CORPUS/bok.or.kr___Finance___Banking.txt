           var fnSrchKeyEvt = function(str) {
										$.ajax({
											url : "/ark.do",
											type : "GET",
											data : {
												query : str,
												convert : "fw",
												target : "common",
												charset : "utf-8",
												datatype : "json"
											},
											async : false,
											dataType : "json",
											timeout : 3000,
											success : function(data) {
												$(".autoList").empty();
												if (data.result) {
													$.each(data.result, function(n,value1) {
														if (value1.items) {
															$.each(value1.items, function(n2,value2) {
																var $li = $("<li>");
																var $a = $("<a>").attr({"href" : "/portal/search/search.do?query=" + encodeURIComponent(value2.keyword)});
																$a.html(value2.hkeyword);
																$li.append($a);

																$(".autoList").append($li);
															});
														}
													});
												}
												if ($(".autoList").find("li").size() == 0) {
													var $li = $("<li>").text("해당 단어로 시작하는 검색어가 없습니다.");
													$(".autoList").append($li);
												}
											}
										});
										$(".autoList").show();
									}
									$("#queryTop").keyup(function(data) {
										$(".dqAuto").show();
										fnSrchKeyEvt($(this).val());
									});
									$("#topSrchClose").keydown(function(event) {
										//console.log(event.keyCode);
										if((!event.shiftKey && event.keyCode == 9)|| event.keyCode == 13){
											$(".dqAuto").hide();
											$(".keyListOpener").focus();
										}
										return false;
									});
									$("#topSrchClose").click(function() {
										$(".dqAuto").hide();
										$(".keyListOpener").focus();
										return false;
