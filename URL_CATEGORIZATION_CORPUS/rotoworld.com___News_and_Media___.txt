        <script type="text/javascript">
         function loadSportNews(elm, doad) {
        $('#sublinks').children("li").children(".active").removeClass();
        var sportname = "";
        switch (elm) {
            case "MLB": sportname = "baseball"; break;
            case "NFL": sportname = "football"; break;
            case "NBA": sportname = "basketball"; break;
            case "NHL": sportname = "hockey";  break;
            case "NAS": sportname = "nascar"; break;
            case "GOL": sportname = "golf"; break;
            case "CFB": sportname = "college-football"; break;
            case "BPL": sportname = "premier-league"; break;
            default: sportname = "football"; elm = "NFL";  break;
            
        }
        if (elm == "MLB") { $("#leagueAL").show(); $("#leagueNL").show() } else { $("#leagueAL").hide(); $("#leagueNL").hide() }
        if (elm == "NFL") { $("#skill").show(); } else { $("#skill").hide(); }

        $("#morenews").attr("href", "/playernews/" + elm.toLowerCase() + "/" + sportname + "/?ls=roto:" + elm.toLowerCase() + ":morenewstop");
        $("#morenews2").attr("href", "/playernews/" + elm.toLowerCase() + "/" + sportname + "/?ls=roto:" + elm.toLowerCase() + ":morenewsbottom");
        
        if ($("#" + elm + "newscont").html() == "") {
            $.ajax({
                type: "GET",
                url: "/serv" + "ices/news.as" + "mx/GetPlayerNews",
                data: "sport=" + elm + "&visibleCount=" + 15 ,
                dataType: "xml",
                success: function (msg) {
                    $("#" + elm + "newscont").html($(msg).find('string').text());
                    manageContainers(elm, doad);
                   
                }
             });
           
        
        } else {
            //hide all containers, show requested container.
            manageContainers(elm, doad); 
        }

        $("#hyppn" + elm).addClass('active');
      
        return false;
    }

    function manageContainers(cntrshow, doad) {
        var cont = [ 'MLB', 'NFL', 'NBA',  'NHL', 'CFB', 'GOL', 'NAS', 'BPL' ];
        for (i=0; i<cont.length; i++) {
            if (cont[i] != cntrshow) {
                $("#" + cont[i] + "newscont").hide();
            }
        } 
        $("#" + cntrshow + "newscont").show();

        if (doad) {
            //sport, page, showid (showid only for slideshows)
            manageAdStream(cntrshow, rw_pagename, rw_prop1, rw_prop4, rw_sponsor);
        }

    }
