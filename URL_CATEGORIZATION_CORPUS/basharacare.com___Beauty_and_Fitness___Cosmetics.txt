        function getCookie(name) {

                offset = cookie.indexOf(search);
                if (offset != -1) {
                    offset += search.length;
                    end = cookie.indexOf(";", offset);
                    if (end == -1) {
                        end = cookie.length;
                    }
                    setStr = decodeURI(cookie.substring(offset, end));
                }
            }
            return(setStr);
        }

            $('#accordion').find('.accordion-toggle').click(function(e) 
            {
                if (!$(this).hasClass('active')) 
                {
                    $('.accordion-toggle').removeClass('active');
                }
                $(this).next().slideToggle('fast');
                $(this).toggleClass('active');
                e.stopPropagation();
                $(".accordion-panel").not($(this).next()).slideUp('fast');
            });
            $("body").mouseup(function() {
                $('.options').mouseup(function(eve){ 
                    eve.stopPropagation();
                    return false;
                });
                
            });
            $(document).click(function(event){
                if ($(event.target).parents('.accordion-panel').size() < 1) {
                    $('.accordion-panel:visible').siblings('.accordion-toggle').trigger('click');
                }
            });
    
        });
    });
