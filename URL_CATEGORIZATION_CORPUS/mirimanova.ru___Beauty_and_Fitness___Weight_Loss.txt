        function setCookie(name, value, options) {
          options = options || {};

          var expires = options.expires;

          if (typeof expires == "number" && expires) {
            var d = new Date();
            d.setTime(d.getTime() + expires*1000);
            expires = options.expires = d;
          }
          if (expires && expires.toUTCString) {
            options.expires = expires.toUTCString();
          }

          value = encodeURIComponent(value);

          var updatedCookie = name + "=" + value;

          for(var propName in options) {
            updatedCookie += "; " + propName;
            var propValue = options[propName];
            if (propValue !== true) {
              updatedCookie += "=" + propValue;
             }
          }

          document.cookie = updatedCookie;
        }

                    $('.ozon_link').attr('href', data.ozon_link);
                    $('.litres_link').attr('href', data.litres_link);
                    $('.labirint_link').attr('href', data.labirint_link);
                    $('#index_book_pdf>a').attr('href', link + data.id);
                    $('.book-image>img').attr('src', data.picture);
                    if(data.color){
                        $('.books-preview').css('background-color', data.color);
                    }else{
                        $('.books-preview').css('background-color', 'green');

                    $(".next_book").attr('onClick', "get_book("+data.id+")");
