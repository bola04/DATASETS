            for (var i = 0; j = paths.length, i < j; i++) {

                    return true;

            var data = $search_vehicle_form.serialize();
            $.getq('VehicleCount', count_vehicles_url, data)
            .done(function(res) {
                if (res.count) {
                    var result_bit = res.count != 1 ? 'resultaten' : 'resultaat';
                    $('#btn-vehicle-search').text('Toon ' + format_number(res.count) + ' ' + result_bit);
                    $.getq('VehicleSlider', search_offers_url, data)
                    .done(function(res) {
                        $('#stock-carrousel-wrapper').fadeIn('slow', function() {
                            $('#stock-carrousel-wrapper').html(res.html);
                        });
                    });
                }
                else {
                    $('#btn-vehicle-search').text('0 resultaten');
