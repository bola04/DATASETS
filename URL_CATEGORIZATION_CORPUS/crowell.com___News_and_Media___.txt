         $(function(){
	$('#storiesNext, #storiesPrev').click(function() {
		$('.swiper-wrapper2').slick('slickSetOption', 'slidesToScroll', 3);
		$('.swiper-wrapper2').slick('slickPause');
	});

	$('.story-content').click(function () { 
			if (changingSlide) return;

			$('.swiper-wrapper2').slick('slickPause');

			storyContent = this;
			if ($(this).parent().children('.modal').length == 0)
				return true;
			originalContent = $(this).parent().children('.modal').first();
			modalContent = $(originalContent).clone();

			modalContent.css('background-image', 'url("'+modalContent.attr('data-bgimg')+'")');

			showModal = function() {
					$(modalContent).attr('id', modalContentID);

					$(modalContent).attr('data-origelemlen', $(storyContent).html().length);

					$('#pagebody_rdnpane_MainStories_ctl00_PageLoadRotatorPanel >div').append(modalContent);	
					closeButton = $('#closeButton').clone();
					$(closeButton).click(function(){
						$(closeButton).remove();
						$('#'+modalContentID).effect('transfer', {to : storyContent}, 250, function() {
							$('#'+modalContentID).remove();
						//}});
						});
					});
					$('#pagebody_rdnpane_MainStories_ctl00_PageLoadRotatorPanel >div').append(closeButton);	
					$(storyContent).effect('transfer', {to : '#pagebody_rdnpane_MainStories_ctl00_PageLoadRotatorPanel'}, 400, function() {
						$(modalContent).show();
						$(closeButton).show();
						window.scroll(modalContent.offsetX, modalContent.offsetY);
					});				
			};

			modalContentID = 'swiperModalContent';
			sameElement = ($('#'+modalContentID).attr('data-origelemlen')) == $(storyContent).html().length;
			if ($('#'+modalContentID).length) {
				$('#'+modalContentID).effect('transfer', {to : storyContent}, 250, function() {
					$(closeButton).remove();
					$('#'+modalContentID).remove();

					if (!sameElement) {
						showModal();
					}
				//}});
				});
			} else {
				showModal();
			}
		});
});
