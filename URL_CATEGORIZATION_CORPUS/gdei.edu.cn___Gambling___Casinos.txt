		function substr(str, len)

		{

			if(!str || !len) { return ''; }

			var a = 0;

			var i = 0;

			var temp = '';

		 

			for (i=0;i<str.length;i++)

			{

				if (str.charCodeAt(i)>255)

				{ a+=2;}

				else

				{a++;}

				if(a > len) { temp +="..."; return temp; }

				 temp += str.charAt(i);

			}

			return str;

		}
