   var current = 0;

function changeBox(direct) {
	max = jQuery("#slider .content-box").length;

	next = (direct == 1) ? current + 1 : current - 1;
	if (direct == 0 && current == 0) next = max - 1;
	if (direct == 1 && (current+1) == max ) next = 0;

	jQuery("#slider .content-box").fadeOut().eq(next).fadeIn();
	jQuery(".points img").removeClass("active").eq(next).addClass("active");

	current = next;
}
