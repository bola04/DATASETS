function add_remove_fixed_menu() {
    if ($(window).scrollTop() >= 93 && sticky_ability == 1) {
        $('.sticky_div').addClass('sticky_div_fix');
        $('.sticky_div_panel').addClass('sticky_div_panel_fix');
    } else {
        $('.sticky_div').removeClass('sticky_div_fix');
        $('.sticky_div_panel').removeClass('sticky_div_panel_fix');
    }
}

function check_window_width() {
    if ($(window).width() > 800) {
        sticky_ability = 1;
    } else {
        sticky_ability = 0;
    }
}
