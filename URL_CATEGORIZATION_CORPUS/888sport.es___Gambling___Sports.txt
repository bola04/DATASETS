    <script>
     /** makes the counter's time calculation 
     * @param goal = the game's date and time
     * @param serverTime = the server date and time
     * @param horusElm = the span element that holds the hours
     * @param minutesElm = the span element that holds the minutes
     * @param secsElm = the span element that holds the secs */
    function eoCount(goal, serverTime, hoursElm, minutesElm, secsElm) {
        goal = new Date(goal);
        goal = goal.getTime();

        now = serverTime;

        //find teh difference between the goal date and now
        var difference = goal - now;

        //calculate the secs, minutes and hours
        var sec = parseInt(difference / 1000) % 60;
        var minutes = parseInt(parseInt(difference / 1000) / 60) % 60;
        var hours = parseInt(parseInt(difference / 1000) / 3600);

        //append them to the relevant html elements
        $(secsElm).html(oddsAddZero(sec));
        $(minutesElm).html(oddsAddZero(minutes));
        $(hoursElm).html(oddsAddZero(hours));
    }

    /** adds the zero infront of single digits 
     * @param num = the number on which the change is made */
    function oddsAddZero(num) {
        num = num.toString();

        if (num.length < 2)
            num = "0" + num;

        return num;
    }

    function countAllSlides() {
        $(".eo-slide").not('.slick-cloned').each(function(index) {
            //gets the index of the goals array out of the 'count-index' attribute of the element
            var i = $(this).attr("count-index");
            var that = $(this);
            getServerRealTime(rlServerTime, index, function(serverTime) {
                eoCount(goals[parseInt(i)], serverTime, that.find(".hours"), that.find(".minutes"), that.find(".secs"));
            })
        });
    }

    /** creates and attaches the counter elements to the DOM*/
    function generateCounters() {

        $(".eo-slide").not('.slick-cloned').each(function(index) {
            $(this).attr("count-index", index);
        });

        //get's all of the buttons in the enchanced odds' slides, because the counter will be inserted after the button
        var buttons = $(".eo-btn-container");

        for (var i = 0; i < goals.length; i++) {
            //take the relevant goal date from the goals array						
            var goal = new Date(goals[i]);
            goal = goal.getTime();

            //create the counter wrappers
            var father = $('<div class="counter"></div>');
            var wrapper = $('<div class="counter-wrapper"></div>');

            //create the counter numbers' spans
            var secs = $('<span class="secs"></span>');
            var minutes = $('<span class="minutes"></span>');
            var hours = $('<span class="hours"></span>');

            //append the counter to the page, add it under the CTA
            $(wrapper).append(hours).append($('<span>:</span>')).append(minutes).append($('<span>:</span>')).append(secs);
            $(father).append(wrapper).insertAfter(buttons[i]);

        }
    }

    function getServerRealTime(rlServerTime, index, cb) {
        var serverTime;
        if (window.counter.firstTimeCycle.hasOwnProperty(index)) {
            serverTime = new Date(window.counter.ServerTimestamp[index]).getTime() + 1000;
        } else {
            serverTime = new Date(rlServerTime).getTime();
            window.counter.firstTimeCycle[index] = 1;
        }

        window.counter.ServerTimestamp[index] = serverTime

        cb(serverTime);
        return true;
    }

    function dateToISO(date) {
        date += 'Z';
        return date.replace(' ', 'T');

    }

    function addClassToOdd() {
       var count = 0;
            $('.eo-offer-odds').each(function(index) {

            if (count === 3) {
                $('.eo-offer-odds').addClass('three-digits');
            } else if (count === 4) {
                $('.eo-offer-odds').addClass('four-digits');
            } else if (count === 2) {
                $('.eo-offer-odds').addClass('two-digits');
            }
    }
