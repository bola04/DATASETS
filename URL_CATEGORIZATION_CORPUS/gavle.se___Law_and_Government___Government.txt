           jQuery(document).ready(function($) {

                if(is_ios() || is_android()) {
                    $('.scroll-x-content').css( {'overflow-x' : 'scroll' });
                }
                else {
                    
                    $('.scroll-x-content').css( {'overflow-x' : 'hidden' });
                    $('.scroll-x-content').kinetic();
                }

                $active = $('.scroll-x-content > .active:first');

                if($active.length > 0) {

                    el = $active.offset().left;
                    ew = $active.width();
                    cw = $('.scroll-x-content').width();

                    $('.scroll-x-content').scrollLeft(el - ((cw/2) - (ew/2)));
                }

                function is_ios() {
                    return navigator.userAgent.match(/(iPad|iPhone|iPod)/g);
                }

                function is_android() {
                    var ua = navigator.userAgent.toLowerCase();
                    return ua.indexOf("android") > -1;
                }

                function scrollInfo() {

                    var $this =  $('.scroll-x-content');
                    var total = 0;

                    $('.scroll-x-content .menu-item').each(function(index, element) {
                        total += $(this).width();
                    });

                    return {
                        left: $this.scrollLeft(),
                        width: $this.width(),
                        total: total
                    }
                }

                function scrollRefresh() {

                    // Always hide by default
                    $('.scroll-x-left').hide();
                    $('.scroll-x-right').hide();

                    // Get scroll info
                    var scroll = scrollInfo();

                    if(scroll.left > 0) {
                        $('.scroll-x-left').show();
                    }
                    
                    if((scroll.left+scroll.width) < (scroll.total-30)) {
                        $('.scroll-x-right').show();
                    }
                }

                // Default
                scrollRefresh();

                $(window).resize(function(){
                    scrollRefresh();
                });

                $('.scroll-x-content').scroll(function(){
                    scrollRefresh();
                });

                $('.scroll-x-left').click(function () {
                    var scroll = scrollInfo();
                    $('.scroll-x-content').animate({ 'scrollLeft': (scroll.left-scroll.width) }, 500);
                });

                $('.scroll-x-right').click(function () {
                    var scroll = scrollInfo();
                    $('.scroll-x-content').animate({ 'scrollLeft': (scroll.left+scroll.width) }, 500);
                });

            });
