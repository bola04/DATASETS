       jQuery(document).ready(function() {

		jQuery('#mod_rbg_310 .loggedout form').submit(function(event) {
			jQuery('#mod_rbg_310 input.login.button').hide();

		});

			doSSOn_mod_rbg_310();

		});

			// Try to login on 'enter/return' key
			if (event.keyCode == 13) {
				jQuery('#mod_rbg_310 form.mbLogin').submit();
				event.preventDefault();
			}
		});

		jQuery('#mod_rbg_310 .loggedout .rbg-tooltip').infoToolTip({
			offsetX: 5,
			offsetY: 15
		});

			doSSOff_mod_rbg_310();
			event.preventDefault();
		});

	});
		
	function doSSOn_mod_rbg_310() {
		try {
			jQuery('#mod_rbg_310 input.login.button').hide();
			jQuery('#mod_rbg_310 img.ajax-loading-login').show();
			jQuery.ajax({
				url: "https://www.bibliotheekalmelo.nl/index.php?option=com_rbg&format=json&layout=mod_rbg&module_id=310&action=bibsys_login",
				type: "POST",
				cache: false,
				async: true,
				dataType: "json",
				data: jQuery('#mod_rbg_310 .mbLogin input').serialize(),
				success: function(data, textStatus, XHR) {
					var message = data.mesg;
											switch(data.stat) {
							case "1":
							  message = "Lener onbekend of leeg wachtwoord";
							  break;
							case "2":
							  message = "Wachtwoord fout";
							  break;
							case "3":
							  message = "Aantal (meld_tel) overschreden";
							  break;
							case "5":
							  message = "Gebruik uw wachtwoord - niet uw PIN - om in te loggen";
							  break;
							case "6":
							  message = "Blokkade of abon verlopen";
							  break;
							case "7":
							  message = "Niet toegestaan in deze bibliotheek";
							  break;
							case "8":
							  message = "IP-nummer blokkade (aantal overschreden)";
							  break;
							case "9":
							  message = "Geen sessie/abon/instantie etc";
							  break;
							default:
							  message = "Onbekende fout";
						}
										if (data.stat == "0") {
						jQuery('#mod_rbg_310 .username').removeClass("error");
						jQuery('#mod_rbg_310 .password').removeClass("error");
						var scherm_meldingen = 0;
						if (typeof data.scherm_meldingen != "undefined") {
							jQuery.each(data.scherm_meldingen, function(key, melding) {	
								if (melding.id=='webcat.eerst_wachtwoord_aanmaken') {
									scherm_meldingen = 1;
									var redir_passw = jQuery('#mod_rbg_310 .redir_passw').attr('value');
									if (redir_passw != '' && redir_passw != "undefined") {
										document.location = redir_passw;
									} else {
										alert(melding.text);
									}
									return false; // stop each
								}
							});
						}
						if (scherm_meldingen == 0) {
							document.location = data.redir;
						}
					} else {
						jQuery('#mod_rbg_310 .password').val("");
						if (data.stat == "1") {
							jQuery('#mod_rbg_310 .username').addClass("error");
							jQuery('#mod_rbg_310 .password').addClass("error");
						} else if (data.stat == "2") {
							jQuery('#mod_rbg_310 .username').removeClass("error");
							jQuery('#mod_rbg_310 .password').addClass("error");
						} else {
							jQuery('#mod_rbg_310 .username').removeClass("error");
							jQuery('#mod_rbg_310 .password').removeClass("error");
						}
						jQuery('#mod_rbg_310 .error-popup').show().find("p").text(message);
						var t = setTimeout(function () {
							jQuery('#mod_rbg_310 .error-popup').hide()
						}, 5000);
						jQuery('#mod_rbg_310 input.login.button').show();
					}
					//RBG.log2console("Ajax succes; status: " + data.fullname);
				},
				error: function(XHR, textStatus, errorThrown) {
					RBG.log2console("Error with login: " + textStatus + "/" + errorThrown);
					RBG.log2console("Received XML: " + XHR.responseText + "\n\n" + "[" + XHR.responseXML + "]");
				},
				complete: function(XHR, textStatus) {
					//RBG.log2console("Ajax call complete; status: " + textStatus);
					jQuery('#mod_rbg_310 img.ajax-loading-login').hide();
				}
			});
		} catch (err) {
			RBG.log2console("Try/Catch Error: " + err);
		}
	};

	function doSSOff_mod_rbg_310() {
		try {
			jQuery('#mod_rbg_310 .logout.uitloggen').hide();
			jQuery('#mod_rbg_310 img.ajax-loading-logout').show();
			jQuery.ajax({
				url: "https://www.bibliotheekalmelo.nl/index.php?option=com_rbg&format=json&layout=mod_rbg&module_id=310&action=bibsys_logout",
				type: "POST",
				cache: false,
				async: true,
				dataType: "json",
				data: jQuery('#mod_rbg_310 .mbLogoff input').serialize(),
				success: function(data, textStatus, XHR) {
					window.location.href = data.redir;
				},
				error: function(XHR, textStatus, errorThrown) {
					RBG.log2console("Error with logout: " + textStatus + " / " + errorThrown);
					RBG.log2console("Received: " + XHR.responseText + " [" + XHR.responseXML + "]");
				},
				complete: function(XHR, textStatus) {
					//RBG.log2console("Ajax call complete; status: " + textStatus);
					jQuery('#mod_rbg_310 img.ajax-loading-logout').hide();
				}
			});
		} catch (err) {
			RBG.log2console("Try/Catch Error: " + err);
		}
	};

       <script type="text/javascript">
        (function($) {
	mod_rbg_activiteiten$Class = {
		selLink: '.link',
	    selContent: '.content',
		catsearch: '',
		currentMonth: '',
		currentYear: '',
		selectedCat: '',
	
	    constructor: function(module, id){
	        jQuery.extend(true, this, mod_rbg_activiteiten$Class);
			this.module = module;
			this.id = id;
			this.block = module + '_' + id;
			this.catsearch = $('#' + this.block + ' .activ-current input.catsearch').val();
			this.selectedCat = $('#' + this.block + ' .activ-current input.actfilter').val();
			this.currentMonth = $('#' + this.block + ' .activ-current input.month').val();
			if (typeof this.currentMonth == 'undefined') this.currentMonth = '0'; // maandfilter raadplegen?
			this.currentYear = $('#' + this.block + ' .activ-current input.year').val();
			if (typeof this.currentYear == 'undefined') this.currentYear = '0'; // maandfilter raadplegen?
	        //this.content = $('#' + this.block + ' ' + this.selContent);
			//this.link = $('#' + this.block + ' ' + this.selLink);
	        this.init();
	    },
	
	    init: function(){
	        var self = this;
			var block = this.block;
			
			var catsearch = this.catsearch;
			var currentYear = this.currentYear;
			var currentMonth = this.currentMonth;
			
			var selectedCat = this.selectedCat;

			self.getItems(currentYear, currentMonth, 0, "", selectedCat, 1, "init"); // initial state
			
			$('#' + block + ' a.activ-prev-month').live('click', function() {
				prevYear = $(this).children('input.prevYear').val();
				prevMonth = $(this).children('input.prevMonth').val();
				if (parseInt(prevYear) >= parseInt('2018')) {
					if (parseInt(prevMonth) >= parseInt('06') || parseInt(prevYear) > parseInt('2018')) {
						// specifieke maand
						self.getItems(prevYear, prevMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-prev-month");
					} else {
						// alle maanden
						self.getItems(0, 0, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-prev-month");
					}
				}
				
				return false;
			});
			$('#' + block + ' a.activ-next-month').live('click', function() {
				nextYear = $(this).children('input.nextYear').val();
				nextMonth = $(this).children('input.nextMonth').val();
				if ( (parseInt(nextYear) == parseInt('2018') && parseInt(nextMonth) >= parseInt('06')) || (parseInt(nextYear) > parseInt('2018') && parseInt(nextMonth) < parseInt('06'))) {
					self.getItems(nextYear, nextMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-next-month");
				}
				return false;
			});
			$('#' + block + ' .activ-select-box .cat_id').live('change', function() {
				if (typeof $('#' + block + ' .activ-select-box .cat_id').val() != "undefined") {
					self.selectedCat = $('#' + block + ' .activ-select-box .cat_id').val();
				} else {
					//self.selectedCat = '';
				}
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-select-cat");
				return false;
			});
			
			$('#' + block + ' .activ-submit').live('click', function() {
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-submit");
				return false;
			});
			$('#' + block + ' .activ-searchtext').keydown(function(event) {
				// Start the filter when the user hits the Enter key (DOM_VK_RETURN)
				if (event.keyCode == 13) {
					currentMonth = $('#' + block + ' .activ-current input.month').val();
					currentYear = $('#' + block + ' .activ-current input.year').val();
					self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-searchtext");
					event.preventDefault();
				}
			});
			$('#' + block + ' .activ-clear').live('click', function() {
				$('#' + block + ' .activ-searchtext').val("").focus();
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, "", "", 0, "activ-clear");
				return false;
			});
			$('#' + block + ' .pager.pagenav a').live('click', function() {
				if (typeof $(this).attr('href') != "undefined") {
					var navOffset = $(this).children('input.nav-offset').val();
					var navYear = $(this).children('input.nav-year').val();
					var navMonth = $(this).children('input.nav-month').val();
					self.getItems(navYear, navMonth, navOffset, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-pagenav");
				}
				return false;
			});
	    },
		
		getItems: function(year, month, startPage, searchText, searchCat, useSession, caller) {
			
			var self = this;
			var block = this.block;
			var id = this.id;
			var module = this.module;
			
			this.currentMonth = month;
			this.currentYear = year;
			
			var extradata = '';
			if (caller != 'init') {
				extradata += "&searchtext=" + String(searchText);
				extradata += "&searchcat=" + String(searchCat);
			} else {
				// init routine
				if (this.catsearch=="0" || searchCat!="") {
					extradata += "&searchcat=" + String(searchCat);
				}
			}
			
			$('#' + block + " .activ-body .items").html("");
			$('#' + block + " .activ-body .waitaminute").show();
			
			$.ajax({
				url: "https://www.bibliotheekalmelo.nl/index.php?option=com_rbgcontent&view=activiteiten&format=raw&layout=" + this.module + "&mid=" + this.id + "&action=content" +
					"&year=" + String(year) +
					"&month=" + String(month) +
					"&limitstart=" + String(startPage) +
					"&useSession=" + String(useSession) + 
					extradata,
				type: "GET",
				dataType: "html",
				//async: false,
				success: function(data, textStatus, XMLHttpRequest) {
					$('#' + block + " .activ-body .waitaminute").hide();
					$('#' + block + " .activ-body .items").html(data);
					$('#' + block + " .activ-body .activity-item-type").each(function () {
						$(this).infoToolTip(module);
					});
				},
				error: function(XMLHttpRequest, textStatus, errorThrown) {},
				complete: function(XMLHttpRequest, textStatus) {
					$('#' + block + " .cat_id").selectbox();
					if (typeof $('#' + block + ' .activ-select-box .cat_id').val() != "undefined") {
						self.selectedCat = $('#' + block + ' .activ-select-box .cat_id').val();
					} else {
						//self.selectedCat = '';
					}
				}
			});
			
			if (useSession==0) {
				var el = $('#' + block).closest('h2'); // werkt niet, uitzoeken
				if (el.length == 0) el = $('#' + block).parents('.rbg-widget').find('h2:first');
				if (el.offset() != null) $('html,body').animate({scrollTop: el.offset().top}, 'fast');
			}
			return false; // Prevent default click action
		},
	
	};
	mod_rbg_activiteiten = mod_rbg_activiteiten$Class.constructor;
})(jQuery);
