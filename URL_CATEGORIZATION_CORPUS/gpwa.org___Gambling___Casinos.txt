       function selected(val)
		{	
			$(val).removeClass("verticals-header-hover");

		{
			$(val).removeClass("verticals-header-hover");

			$(val).addClass("verticals-header");

		}
		
		function hovering(val)
		{
			$(val).removeClass("verticals-header-selected");

		}
	
		function slideToggle(module)
		{
				if ($(module).is(":hidden"))
				{
					$(".verticals-module").slideUp(); /* these two lines if commented out will allow for multiple verticals to be opened at the same time. */
					unselected($(".verticals-module").prev());
					selected($(module).prev());
				}
				else
				{
					unselected($(module).prev());
				}
				$(module).slideToggle('slow');
		}

							{
