function log (msg)
{
    setTimeout (
	function() {
	    throw new Error(msg);
	},
	0);
}

function el (x)
{
    return (typeof x == "string") ? document.getElementById(x) : x;
}

function hide (x)
{
    el(x).style.display = 'none';
}

function show (x)
{
    el(x).style.display = 'block';
}

function show_inline (x)
{
    el(x).style.display = 'inline';
}

function visible (x)
{
    return el(x).style.display == 'none' ? false : true;
}

function toggle (x)
{
    if (visible(x))
    {
	hide(x);
    }
    else
    {
	show(x);
    }
    return false;
}

function toggle_tips ()
{
    if (visible('div_tips'))
    {
	hide('div_tips');
	el('span_tips').innerHTML = 'Show tips';
    }
    else
    {
	show('div_tips');
	el('span_tips').innerHTML = 'Hide tips';
    }
    return false;
}

function getCookie (name)
{
    var match = document.cookie.match (new RegExp (name + '=([^;]+)'));
    if (match)
    {
	return match[1];
    }
    else
    {
	return '';
    }
}

function setCookie (name, value, expiry_days)
{
    document.cookie = name + '=; expires=Thu, 01 Jan 1970 00:00:01 GMT;';
    var date = new Date();
    date.setTime (date.getTime() + (expiry_days * 24 * 60 * 60 * 1000));
    var expires = "expires="+ date.toUTCString();
    var cookie = name + "=" + value + "; " + expires + "; path=/; domain=.crazyguyonabike.com";
    document.cookie = cookie;
}

var request = null;

function createRequest()
{
    request = null;
    try
    {
	request = new XMLHttpRequest();
    }
    catch (trymicrosoft)
    {
	try
	{
	    request = new ActiveXObject("Msxm12.XMLHTTP");
	}
	catch (othermicrosoft)
	{
	    try
	    {
		request = new ActiveXObject("Microsoft.XMLHTTP");
	    }
	    catch (failed)
	    {
		request = null;
	    }
	}
    }
}

 var context = '';
 var type = '';
 var id = 0;

 var action = '';
 var rating = '';
 var counter = 0;

 function doRatingsCallback()
 {
     if (request.readyState == 4)
     {
	 var result = request.responseText;
	 if (result.substring(0,5) == 'Error')
	 {
	     if (result.indexOf ('logged in') > -1)
	     {
		 
		 window.location = '/login/?ref=' + escape(window.location);
	     }
	     else
	     {
		 alert (result);
	     }
	 }
	 else if (result == '')
	 {
	     alert ('Request failed, could be due to spotty network connectivity or the webserver is down (hopefully temporarily). Please give it a minute, then reload this page and try again.');
	 }
	 else
	 {
	     document.getElementById ('divRatings_' + type + '_' + id + '_' + counter).innerHTML = result;
	     if (action == 'toggle')
	     {
		 document.getElementById('RatingCollapsedScore_' + type + '_' + id + '_' + counter).innerHTML = 'Score';
		 document.getElementById('divRatings_' + type + '_' + id + '_' + counter).style.display = 'block';
		 document.getElementById('divRatingsContainer_' + type + '_' + id + '_' + counter).style.zIndex = 100;
		 document.getElementById('divRatings_' + type + '_' + id + '_' + counter).style.zIndex = 50;
	     }
	 }
	 action = '';
     }
 }

 function closeRatingsMenu ()
 {
     if (id > 0 && action == '')
     {
	 document.getElementById('RatingCollapsedScore_' + type + '_' + id + '_' + counter).innerHTML = document.getElementById('RatingScore_' + type + '_' + id + '_' + counter).innerHTML;
	 document.getElementById('divRatings_' + type + '_' + id + '_' + counter).style.display = 'none';
	 document.getElementById('divRatingsContainer_' + type + '_' + id + '_' + counter).style.zIndex = 1;
	 document.getElementById('divRatings_' + type + '_' + id + '_' + counter).style.zIndex = 1;
	 id = 0;
     }
     var old_context = context;
     context = '';
     return (old_context == '') ? true : false;
 }

 function toggleRatings (the_type, the_id_field, the_id, version, format, filter, the_counter, topic_id)
 {
     createRequest();
     if (request == null)
     {
	 return true;
     }
     else
     {
	 context = 'toggleRatings';
	 action = '';
	 if (the_id == id)
	 {
	     closeRatingsMenu ();
	     context = 'toggleRatings';
	 }
	 else
	 {
	     closeRatingsMenu ();
	     context = 'toggleRatings';
	     type = the_type;
	     id = the_id;
	     id_field = the_id_field;
	     counter = the_counter;
	     action = 'toggle';
	     var url = "/ajax/doRatings.html?id_field=" + the_id_field + "&id=" + the_id + "&counter=" + counter + "&v=" + version + "&format=" + format + "&filter=" + filter + "&token=" + "&topic_id=" + topic_id;
	     request.open("GET", url, true);
	     request.onreadystatechange = doRatingsCallback;
	     var result = request.send(null);
	 }
	 return false;
     }
 }

 function doRating (the_type, the_id_field, the_id, the_counter, the_rating, the_delta, version, format, filter, do_confirm, topic_id)
 {
     context = 'doRating';
     type = the_type;
     id = the_id;
     id_field = the_id_field;
     rating = the_rating;
     counter = the_counter;
     createRequest();
     if (request == null)

	 return true;

     {
	 if ((do_confirm == 1 && (( the_rating != 'miscategorized' && the_rating != 'troll' && the_rating != 'abuse' && the_rating != 'spam' && the_rating != 'offtopic' ) || confirm ("Are you sure you want to use this rating? It should not be used lightly. If you click 'Ok' then the webmaster will be notified so further action can be taken.") )) || do_confirm == 0)
	 {
	     action = 'do_rating';
	     var url = "/ajax/doRatings.html?id_field=" + the_id_field + "&id=" + the_id + "&counter=" + counter + "&rating=" + the_rating + "&filter=" + filter + "&delta=" + the_delta + "&v=" + version + "&format=" + format + "&token=" + "&topic_id=" + topic_id;
	     request.open("GET", url, true);
	     request.onreadystatechange = doRatingsCallback;
	     var result = request.send(null);
	 }
	 return false;
     }
 }

            <a href="/doc/?o=1mr&amp;doc_id=15552&amp;v=Lz">
             Epistles from the Northern Tier
            </a>
           </h3>
           <b>
            Northern Tier
           </b>
           <br/>
           By
           <nobr>
            <a href="/directory/?o=1mr&amp;user=nickpatrone&amp;v=Q" title="nickpatrone joined February 8th 2015">
             Nicholas Patrone
            </a>
           </nobr>
           <br/>
           <small>
            <font color="gray">
             #830 - posted Sunday June 17, 2018 by Neil Gunton
            </font>
           </small>
           <p>
            <i>
             After 4 months of training I am reaching the Nirvana of " Buffness". My logbook shows 300 miles per week but that has been such a gradual buildup that there is little sense of progression. My pants are a size smaller, but I have been there before. People say I look 10 years younger but flattery will get you everywhere. The real manifestation of my progress came today on my 60 mile ride. Along my route there are a group of 8 slim, sleek, muscular riders, both male and female, on very fancy speedy bikes. My goal 4 months ago was to try to match their speed for a half mile or so, all be it 300-400 yards behind them so they would not see me struggling near death to keep pace. Today I spotted them a quarter mile ahead of me about 1 1/2 miles from my 45 mile "water- food" break spot. I decided to try to catch them and rapidly accelerated to "half dog speed"#. Not the least bit deterred that it was an uphill mile and a half I sped past gleefully saying " On your left!". I got to my spot 200 yards ahead of them and was already eating and drinking when they rode by with admiration for my burst of speed. I took a few second break to congratulate myself and finished the 60 miles with a twinkle in my eye , a smile on my face, and a realization that true "Buffness" was just around the corner. I can hardly wait to surprise my other 29 Northern Tier Riders in a few weeks with "Half Dog Speed"...if I happen to be in "Dog Speed" as I pass then I won't bother to say " On your left" but rather scream out " BIG DOG !!!". After all , we are all in this adventure together covering each other's gluteus maximus...
            </i>
