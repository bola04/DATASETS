		$('.sp_menu').click(function(){
			$(this).next('.h_navi').stop(true, true).slideToggle();

		$('.sp_btn').click(function(){
			$(this).parents('.h_navi').stop(true, true).slideToggle();

		});

	} else {
		$('.h_navi li').hover(function(){
			$("div:not(:animated)", this).slideDown(200);
		}, function(){
			$("div.child",this).slideUp(200);
		});
		
	var navigation = $('#mainNavi')
	var borderLine = 70
	$(document).ready(function(){
		var navHeight = navigation.outerHeight();
		navigation.before('<div id="navDmy" style="display:none;height:'+navHeight+'px;"></div>');
		hantei();
	});
	$(window).scroll(function(){
		hantei();
	});
	function hantei(){
		if($(window).scrollTop() > borderLine){
			navigation.addClass('active');
			$('#navDmy').show();
		}else{
			navigation.removeClass('active');
			$('#navDmy').hide();
		};
	};
	
   $(".hover").hover(function(){
       $(this).stop(true).fadeTo("normal", 0);
    },function(){
       $(this).fadeTo("normal", 1.0);
   });

	};
});
