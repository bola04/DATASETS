   var bannerRotatorId = "";
        $(function () {
            if ($(".rotator").children('.banner').length > 1) {
                startRotator(".rotator");
            }
        });

        function rotateBanners(elem) {
            var active = $(elem + " .banner.active");
            var activeIndicator = $(".indicator-button.active");

            var next = active.next();
            if (next.length == 0) {
                next = $(elem + " .banner:first");
            }

            var nextIndicator = activeIndicator.next();
            if (nextIndicator.length == 0) {
                nextIndicator = $(".indicator-button:first");
            }

            active.removeClass("active").hide();
            activeIndicator.removeClass("active");

            next.addClass("active").fadeIn(1000);
            nextIndicator.addClass("active");
        }

        function prepareRotator(elem) {
            $(elem + " .banner").fadeOut(0);
            $(elem + " .banner:first").fadeIn(0).addClass("active");
            $(".indicator-button:first").addClass("active");
        }

        function startRotator(elem) {
            prepareRotator(elem);
            bannerRotatorId = setInterval("rotateBanners('" + elem + "')", 8000);
        }

        $('.indicator-button').click(function (e) {
            e.preventDefault();

            clearInterval(bannerRotatorId);

            $('.indicator-button').removeClass("active");
            $(this).addClass("active");
            var index = $(this).data('slide-id');

            if (index > 0) {
                $(".rotator .banner").fadeOut(0);
                $(".banner:eq(" + (index - 1) + ")").fadeIn(0).addClass("active");
            }
        });
  </script>
