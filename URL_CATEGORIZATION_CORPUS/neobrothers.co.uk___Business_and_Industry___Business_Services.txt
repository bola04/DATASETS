function centreImg(c){
  $j(c).each(function(){
    var imgHeight = $j(this).find('img').height();
    var boxHeight = $j(this).height();
    if (imgHeight === 0) {
      boxHeight = 0;
    }
    var padTop = Math.floor((boxHeight-imgHeight)/2);
    $j(this).find('img').css('marginTop', padTop);
  });
}
