    		} else {
    			
    		}   		
    	});
    }
    
    // This related to an id (searchPath) in searchResultsBlock.
    function checkSearch() {
      var display = "";

      if ($("#filter1").val() != "") {
      
      	 if (display != "") {
      	 	display += " in ";
      	 }      
     	 display += $("#filter1 :selected").text();
      }
      if ($("#filter2").val() != "") {
     	 display += " | " + $("#filter2 :selected").text();
      }
            
	  $("#searchPath").text(display);

    }    
    
    $("#filter1").change(function(){
    	check = $("#filter1").val();
		$("#filter2 option").each(function(i) {
			$(this).removeAttr("selected");
		});
		
		if (check == "") {
			clearFilter2();
		} else {
			checkFilter2();
		}
    });
