                    var top = 0;

                                $('#newsFlash').append($(document).find('#newsFlash li').first());
                                top = 0;                                
                                $('#newsFlash').css({
                                    'margin-top': top + 'px'
                                });
                            } else {
                                top = top - 1;
                            }

                            $('#newsFlash').animate({
                                'margin-top': top + 'px'
                            }, 10);
                        }

                        marginInterval = setInterval(animateMargin, 100);
                        $('#newsFlash').hover(function () {
                            clearInterval(marginInterval);
                        },

                        function () {
                            marginInterval = setInterval(animateMargin, 100);
