	function get_fresh_delay(){
		var now = get_timestamp();
		var elapsed_secs = now - fresh_load_time;
		if (elapsed_secs<=30){
			return 5*1000;	
		}else if (elapsed_secs<=60){
			return 10*1000;

			return 15*1000;
