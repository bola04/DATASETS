dyn_menu = {
		"handle": $$(".b_small_media_menu")[0],
		"drawer": $("main_links"),
		"menu": $("main_links").getElement("ul"),
		"morph": {},
		"open": {},
		"close": {},
		"toggle": {},
		"state": false,
		"init": function() {
			if (!this.handle || !this.drawer || !this.menu) return false;
			dyn_menu.morph = new Fx.Morph(dyn_menu.drawer);
			dyn_menu.open = function() {
				if (dyn_menu.state) return false;
				dyn_menu.morph.start({
					"height": dyn_menu.menu.measure(function() {
						return this.getDimensions().y;
					}),
					"opacity": 1
				}).chain(function(){
					dyn_menu.state = true;
					window.addEvent("click", dyn_menu.close);
				});
			};
			dyn_menu.close = function(e) {
				if (!dyn_menu.state) return false;
				if (e.target != dyn_menu.handle
				 && dyn_menu.drawer.getElements("*").indexOf(e.target) != -1) {
					dyn_menu.drawer.setStyle("height", "auto");
					return true;
				}
				window.removeEvents("click");
				dyn_menu.morph.start({
					"height": 0,
					"opacity": 0
				}).chain(function(){
					dyn_menu.state = false;
				});
				return false;
			};
			dyn_menu.toggle = function() {
			console.log("toggle");
				if (dyn_menu.state) {
					dyn_menu.close();
				} else {
					dyn_menu.open();
				}
			};
			dyn_menu.handle.addEvent("click", dyn_menu.toggle);
	        dyn_menu.handle.addEvent("mouseover", dyn_menu.open);
		}
	};
	dyn_menu.init();
