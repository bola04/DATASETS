    function validate_form() {
    
        var fullname = $('.fldtxtName').val(); // document.frmancestor_search.txtName.value;
        var emailaddr =$('.fldtxtEmail').val(); // document.frmancestor_search.txtEmail.value;
        var errormsg = document.getElementById('errormssg');

        var errormssg = "";

        if (fullname == '') {
            errormssg += 'Please enter your name.';
            document.getElementById('errormssg').innerHTML = errormssg;
            $('.fldtxtName').focus();
            //document.frmancestor_search.txtName.focus();
            return false;
        }
        if (fullname.trim().indexOf(' ') == -1) {
            errormssg += 'Please enter your first and last name.';
            document.getElementById('errormssg').innerHTML = errormssg;
            $('.fldtxtName').focus();
            return false;
        }

        if (emailaddr == '') {
            errormssg += 'Please enter your email address.';
            document.getElementById('errormssg').innerHTML = errormssg;
            $('.fldtxtEmail').focus();
            return false;
        }
        else {
            if (!echeck(emailaddr)) {
                errormssg = 'Please enter a valid email address';
                document.getElementById('errormssg').innerHTML = errormssg;
            $('.fldtxtEmail').focus();
                return false;
            }
            var re=/^[_a-z0-9-]+((\.|\+)[_a-z0-9-]+)*@[a-z0-9-]+(\.[a-z0-9-]+)*(\.[a-z]{2,4})$/i;
            if (!re.test(emailaddr))
            {
                errormssg = 'Please enter a real email address.';
                document.getElementById('errormssg').innerHTML = errormssg;
                $('.fldtxtEmail').focus();
                return false;
            }
            if (emailaddr.toLowerCase().indexOf('myheritage.com')!=-1)
            {
                errormssg = 'Please enter a different email address.';
                document.getElementById('errormssg').innerHTML = errormssg;
                $('.fldtxtEmail').focus();
                return false;
            }
        }

        // the post is validated, so hide the button so they can't click it twice
        $(".fldbtnContinue").hide();
        $(".fldprocessingFT").show();
        
        return true;
    }

    function echeck(str) {
        var at = "@"
        var dot = "."
        var lat = str.indexOf(at)
        var lstr = str.length
        var ldot = str.indexOf(dot)
        if (str.indexOf(at) == -1) {
            return false
        }

        if (str.indexOf(at) == -1 || str.indexOf(at) == 0 || str.indexOf(at) == lstr) {
            return false
        }

        if (str.indexOf(dot) == -1 || str.indexOf(dot) == 0 || str.indexOf(dot) == lstr) {
            return false
        }

        if (str.indexOf(at, (lat + 1)) != -1) {
            return false
        }

        if (str.substring(lat - 1, lat) == dot || str.substring(lat + 1, lat + 2) == dot) {
            return false
        }

        if (str.indexOf(dot, (lat + 2)) == -1) {
            return false
        }

        if (str.indexOf(" ") != -1) {
            return false
        }

        return true
    }
