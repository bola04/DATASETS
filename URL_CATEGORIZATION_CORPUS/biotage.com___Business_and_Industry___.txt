     kickstart.addKickstart(
        function() {
    
            var slideshowHandle = null;
            var numRotItems = $('.rotation-item').length;
            var currentRotItem = 0;
    
            var displayRotItem = function() {
                $('.rotation-item').each(
                    function(i,e) {
                        if(currentRotItem === i){
                            $(e).show();
                        } else {
                            $(e).hide();
                        }
                    }
                );
            }
    
            var nextRotItem = function() {
                currentRotItem++;
                if(currentRotItem>=numRotItems) {
                    currentRotItem = 0;
                }
                displayRotItem();
            }
    
            var prevRotItem = function() {
                currentRotItem--;
                if(currentRotItem<0) {
                    currentRotItem = (numRotItems - 1);
                }
                displayRotItem();
            }
    
            $('#slideshow-left').click(
                function() {
                    startSlideshow();
                    prevRotItem();
                }
            );
            $('#slideshow-right').click(
                function() {
                    startSlideshow();
                    nextRotItem();
                }
            );
    
            var startSlideshow = function() {
                if(slideshowHandle!=null) {
                    clearInterval( slideshowHandle );
                }
                slideshowHandle = setInterval(
                    nextRotItem, 5000
                );
            }
            startSlideshow();
            
        }
    );
