         <article about="/node/1" class="node node-page view-mode-full" id="node-1" typeof="foaf:Document">
          <span class="rdf-meta element-hidden" content="Welcome To NALA" property="dc:title">

           <p>
           </p>
           <p>
            <span style="font-size:16px">
             For more information on how to use NALA's new

           <blockquote>
