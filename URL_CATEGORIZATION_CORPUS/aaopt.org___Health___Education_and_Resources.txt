           var searchURL = "";
    function searchClicked(event) {
        if (event && event.preventDefault) {
            event.preventDefault();
        } else {
            event.returnValue = false;
        }
        searchURL = "/fellows-search?startRow=0";
        var LastName = document.getElementById("FAF_Name").value.replace(/[^A-Za-z0-9, ]/, "").trim();
        var FirstName = "";

        var NameArray = LastName.match(/[^,]+/g);
        if (NameArray != null) {
            if (NameArray.length > 1) {
                // Last, First, more?
                LastName = NameArray.shift().trim();
                FirstName = NameArray.join(" ").trim();
            } else {
                NameArray = LastName.match(/[^ ]+/g);
                if (NameArray != null && NameArray.length > 1) {
                    // First Last more
                    FirstName = NameArray.shift();
                    LastName = NameArray.join(" ");
                }
            }
        }

        var City = document.getElementById("FAF_City").value.replace(/[^A-Za-z0-9 ]/, "").trim();
        var PostalCode = "";

        var PostalCodeArray = City.match(/(\d[-+ ]?){2,9}|[A-Za-z]\d([A-Za-z][-+ ]?\d?){0,2}/g);
        if (PostalCodeArray != null && PostalCodeArray.length > 0) {

            // Remove what looks like a postal code from the City field.
            City = City.replace(PostalCodeArray[0], "").trim();

            var ZipCodeArray = PostalCodeArray[0].match(/\d{2,5}/g);
            if (ZipCodeArray != null && ZipCodeArray.length > 0) {
                // US Zip Code
                PostalCode = ZipCodeArray[0];
            } else {
                PostalCodeArray = PostalCodeArray[0].match(/[A-Za-z]\d.*/g);
                if (PostalCodeArray != null && PostalCodeArray.length > 0) {
                    // Canadian Postal Code
                    PostalCode = PostalCodeArray[0].replace(/[^A-Za-z0-9]/, "");
                    if (PostalCode.length > 3) {
                        PostalCode = PostalCode.substring(0, 3) + " " + PostalCode.substring(3);
                    }
                }
            }
        }
        var State = "";
        var Country = "";

        var LocationArray = document.getElementById("FAF_State").value.match(/[^| ]+/g);
        if (LocationArray != null) {
            if (LocationArray.length > 1) {
                State = LocationArray[0];
                Country = LocationArray[1];
            }
            else {
                Country = LocationArray[0].trim();
            }
        }

        if (FirstName.length > 0) {
            searchURL += "&FirstName=" + FirstName;
        }
        if (LastName.length > 0) {
            searchURL += "&LastName=" + LastName;
        }
        if (City.length > 0) {
            searchURL += "&City=" + City;
        }
        if (State.length > 0) {
            searchURL += "&State=" + State;
        }
        if (PostalCode.length > 0) {
            searchURL += "&PostalCode=" + PostalCode;
        }
        if (Country.length > 0) {
            searchURL += "&Country=" + Country;
        }

        window.location = searchURL;
    }
