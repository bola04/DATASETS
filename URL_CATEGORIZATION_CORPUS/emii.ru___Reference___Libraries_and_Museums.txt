   require(['domReady'], function (domReady) {
        domReady(function () {
            require(['jquery'], function ($) {

                $('.header-top-toolbar-contacts__link').on({
                  mouseenter: function() {
                    $(this).addClass('active');
                    $('.header-top-toolbar-contacts-info').addClass('active');
                  }
                });
                $('.header-top-toolbar-contacts-info').on({
                  mouseenter: function() {
                    $(this).addClass('active');
                  },
                  mouseleave: function() {
                  $('.header-top-toolbar-contacts__link').removeClass('active');
                    $(this).removeClass('active');
                  }
                });


                $('.header-top-toolbar-contacts__close').on('click', function() {
                  if ($('.header-top-toolbar-contacts__link').hasClass('active')) {
                      $('.header-top-toolbar-contacts__link').removeClass('active');
                      $('.header-top-toolbar-contacts__info').removeClass('active');
                      $('.header-top-toolbar-contacts-info').removeClass('active');
                    }
                });

                $('.times-ico').on({
                  click: function() {
                    if (!$(this).hasClass('active')) {
                      $(this).addClass('active');
                      $('.header-top-toolbar-contacts-info__mobile').addClass('active');
                      return false;
                    } else {
                      $(this).removeClass('active');
                      $('.header-top-toolbar-contacts-info__mobile').removeClass('active');
                      return false;
                    }
                  }
                });
          
                $('.header-top-toolbar-contacts__close').on('click', function() {
                  if ($('.times-ico').hasClass('active')) {
                      $('.times-ico').removeClass('active');
                      $('.header-top-toolbar-contacts-info__mobile').removeClass('active');
                    }
                });

            }); // close require
        }); // close require dom ready
    });
  </script>
