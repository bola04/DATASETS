			var tiles = "";
			if (resp.topic_list.topics.length >8)
				var arraylength = 8;
			else
				var arraylength = resp.topic_list.topics.length;	

			for (var i = 1; i < arraylength; i++) {

				if (d2 - d1 < 864000000) {

				tiles += "<div class='item'>";
