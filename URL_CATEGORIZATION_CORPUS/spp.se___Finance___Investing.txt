    $("#backYes").on("click", Like);
    $("#backNo").on("click", Dislike);

    function Like() { GiveFeedback(true); }
    function Dislike() { GiveFeedback(false); }

    function GiveFeedback(feedback) {
        if($('#feedbackContainerReceipt').is(':visible') ) {
            $("#feedbackContainerReceipt").slideUp();
        }
        if(!$('#feedbackMsg').is(':visible') ) {
            $('#feedbackMsg').slideDown();
        }

        var dataLayerLabel;
        var dataLayerValue;
        if(feedback == true)
        {
            dataLayerValue = 1;
            dataLayerLabel = "positive";
        }
        else
        {
            dataLayerValue = -1;
            dataLayerLabel = "negative";
        }
        
        var data = {
            event: 'event',
            eventCategory: 'User feedback',
            eventAction: 'Page poll',
            eventLabel: dataLayerLabel,
            eventValue: dataLayerValue
        };
        dataLayer.push(data);

        var request = $.ajax({
            url: "/Feedback/GiveFeedback",
            type: 'POST',
            dataType: 'json',
            async: false,
            data: { pageId: 173, like: feedback }
        });
        request.done(function (data) {
            $("#feedbackId").val(data.Id.ExternalId);
            if (data.Like) {
                $("#backYes").addClass("active");
                $("#backNo").removeClass("active");
                $('#feedbackText').attr('placeholder', 'Vad bra - tala gärna om vad vi kan göra för att bli ännu bättre. Om du lämnar en kommentar, tänk på att inte skriva känsliga personuppgifter, till exempel person- och kontonummer.');
            }
            else
            {
                $("#backNo").addClass("active");
                $("#backYes").removeClass("active");
                $('#feedbackText').attr('placeholder', 'Vad synd - tala gärna om vad vi kan förbättra. Om du lämnar en kommentar, tänk på att inte skriva känsliga personuppgifter, till exempel person- och kontonummer.');
            }
            $("#feedbackMsg").slideDown();
            $("#backYes").prop("disabled",true);
            $("#backNo").prop("disabled",true);
        }).fail(function(jqXHR, textStatus) {
            alert( "Request failed: " + textStatus );
        });
    }

    function SaveComment() {
        var dataLayerLabel = "";
        var dataLayerValue = "";

        if($("#backYes").hasClass("active"))
        {
            dataLayerLabel = "yes";
        }
        else
        {
            dataLayerLabel = "no";
        }

        var data = {
            event: 'event',
            eventCategory: 'User feedback',
            eventAction: 'Page feedback',
            eventValue: '', 
            eventLabel: dataLayerLabel
        };
        dataLayer.push(data);

        var request = $.ajax({
            url: "/Feedback/SaveComment",
            type: 'POST',
            dataType: 'json',
            data: { id: $("#feedbackId").val(), name: $("#userFeedbackName").val(), email: $("#userFeedbackEmail").val(), comment: $("#feedbackText").val() }
        });

        request.done(function(data) {
            if($('#feedbackMsg').is(':visible') ) {
                $('#feedbackMsg').slideUp();
            }
            if(!$('#feedbackContainerReceipt').is(':visible') ) {
                $("#feedbackContainerReceipt").slideDown();
            }
            $("#feedbackText").val("");
        });
        return false;
    }
