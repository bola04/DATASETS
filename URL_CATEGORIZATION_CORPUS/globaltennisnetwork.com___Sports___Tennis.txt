	function geocode() {
		document.getElementById("gpsButton").disabled = true;
		document.getElementById("gpsButton").innerHTML = '<i class="fa fa-spinner fa-spin" aria-hidden="true"></i>';		
		document.getElementById("locationSearch").value = "";

		if(navigator.geolocation) {
			navigator.geolocation.getCurrentPosition(function(position) {
				var latitude = position.coords.latitude;
				var longitude = position.coords.longitude;
		
				document.getElementById("latitude").value = latitude;
				document.getElementById("longitude").value = longitude;

				getLinkByCoordinates(latitude, longitude, function(){
					countryChanged();
					getLocationsChanged();
				});
			}, function() {
				document.getElementById("gpsButton").disabled = false;
				document.getElementById("gpsButton").innerHTML = "GPS";
				
				alert("Could not find your location.");
			});
		} else {
			document.getElementById("gpsButton").disabled = false;
			document.getElementById("gpsButton").innerHTML = "GPS";
				
			alert("Could not find your location.");
		}
	}
