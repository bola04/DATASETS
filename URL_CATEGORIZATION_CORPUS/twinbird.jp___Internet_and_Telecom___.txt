       $(function(){

        var itemDetailIndex = 0;
        var fadeSpeed = 600;
        var itemDetail = $('div.fr_top_product_list div.pr_itemdetail');
        var btnMore = $('#pr_btn_more');
        var itemDetailHeight;

        itemDetailFade(itemDetailIndex);
        btnMore.click(function(){
            itemDetailFade(itemDetailIndex);
        });


        function itemDetailFade(i){

            btnMore.fadeOut(10);

            itemDetail.eq(i).show();
            if(i != 0){
                itemDetail.eq(i).animate({'height':itemDetailHeight});
            }

            itemDetail.eq(i).find('li').eq(0).delay(100).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(1).delay(200).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(2).delay(300).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(3).delay(400).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(4).delay(500).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(5).delay(600).fadeIn(fadeSpeed);
            itemDetail.eq(i).find('li').eq(6).delay(700).fadeIn(fadeSpeed);

            itemDetailIndex++;

            if(itemDetail.eq(itemDetailIndex).length > 0){
                btnMore.delay(800).fadeIn(fadeSpeed,function(){
                    if(i == 0){
                        itemDetailHeight = itemDetail.eq(i).height();
                    }
                });
            }
        }
    });
