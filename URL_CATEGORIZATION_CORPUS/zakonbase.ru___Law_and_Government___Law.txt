     $(function() {
            $('#rotator').jcarousel({
                auto: 7,
                wrap: 'last',
                scroll: 1,
                initCallback: rotator_initCallback,
                itemFirstInCallback: rotator_itemFirstInCallback
            });

            function rotator_itemFirstInCallback(carousel, element) {
                $('.jcarousel-pagination.rotator li.rotator-paging-x').removeClass('rotator-paging-x').addClass('rotator-paging-n')
                $('.jcarousel-pagination.rotator li.rotator-paging-n').each(function() {
                    if ($(this).attr('tabindex') == $(element).attr('jcarouselindex'))
                        $(this).removeClass('rotator-paging-n').addClass('rotator-paging-x')
                });
            }

            function rotator_initCallback(carousel) {
                $('.jcarousel-pagination.rotator a').bind('click', function() {
                    carousel.scroll($.jcarousel.intval(jQuery(this).attr('data-page')));
                    $('.jcarousel-pagination.rotator li.rotator-paging-x').removeClass('rotator-paging-x').addClass('rotator-paging-n');
                    $(this).parent().removeClass('rotator-paging-n').addClass('rotator-paging-x');
                    return false;
                });
            }
        })
    </script>
