        /********* Custom Daigger menu script *********/
        if( $("html").hasClass("ie8") || Modernizr.mq("only all and (min-width: 768px)") ) {
            var $navbarInner = $(".navbar-inner"),
                $daiggerDropdown = $(".daigger-dropdown"),
                $desktopSubmenu = $(".desktop-submenu"),
                daiggerfNav = false,
                iNavigationTopDistance = parseInt($(".section-container-main-navigation").offset().top);

            // Move top content in layout inside the last <li> of the menu
            $("#menuTxtTarget").html($(".topBanner .widget").html());

            // Open / close custom dropdown menu
            $daiggerDropdown.find(">a").click(function(){
                $daiggerDropdown.removeClass("open").toggleClass("opened");
                if( !$daiggerDropdown.hasClass("opened") ) {
                    $desktopSubmenu.removeClass("opened");
                }
  });

            // Close custom dropdown menu when f-nav appears
            $(window).scroll(function(){
                if (window.pageYOffset > iNavigationTopDistance && !daiggerfNav) {
                    daiggerfNav = true;

                    $daiggerDropdown.removeClass("open opened");
                    $desktopSubmenu.removeClass("opened");

                } else if(window.pageYOffset <= iNavigationTopDistance && daiggerfNav) {
                    daiggerfNav = false;
                }
            });

            // Open/close the alphabetical menu
            $(".daigger-dropdown .daigger-see-all>a,.desktop-submenu-close").click(function(){
                $desktopSubmenu.toggleClass("opened");
            });
            
            // Close the alphabetical menu on outside click
            $(document).mousedown(function(){
                if( $desktopSubmenu.hasClass("opened") && !Header.mouseTarget.is(".desktop-submenu") && !Header.mouseTarget.parents(".desktop-submenu").length ) {
                    $desktopSubmenu.removeClass("opened");
                }
            });

            // Calculate position and width of the alphabetical menu
            $(window).resize(function(){
                $desktopSubmenu.css({
                    top: $daiggerDropdown.height()
                });

                if( Modernizr.mq("only all and (min-width: 980px)") ) {
                    $desktopSubmenu.css({
                        left: 0,
                        width: $navbarInner.width() - $daiggerDropdown.width()
                    });
                } else {
                    $desktopSubmenu.css({
                        left: - $daiggerDropdown.width(),
                        width: $navbarInner.width()
                    });
                }
            }).resize();
        }

        /********* Custom Daigger menu script end *********/
    });
