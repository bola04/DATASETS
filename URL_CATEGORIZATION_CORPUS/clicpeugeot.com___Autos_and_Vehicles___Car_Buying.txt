       function equalWidth(group) {
        var tallest = 0;
        jQ("#ddcl-" + group + "-ddw .ui-dropdownchecklist-item").each(function () {
            var thisWidth = jQ(this).width();
            if (thisWidth > tallest) {
                tallest = thisWidth;
            }
        });

        jQ("#ddcl-" + group + "-ddw .ui-dropdownchecklist-item").each(function () {
            jQ(this).css('min-width', tallest);
        });
    }
