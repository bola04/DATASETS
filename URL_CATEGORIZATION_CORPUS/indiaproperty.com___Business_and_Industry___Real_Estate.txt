   function loadScriptURL(url, callback){
    var script = document.createElement("script")
    script.type = "text/javascript";

    if (script.readyState){  //IE
        script.onreadystatechange = function(){
            if (script.readyState == "loaded" ||
                    script.readyState == "complete"){
                script.onreadystatechange = null;
                callback();
            }
        };
    } else {  //Others
        script.onload = function(){
            callback();
        };
    }

    script.src = url;
    document.getElementsByTagName("head")[0].appendChild(script);
}

function get_AS_config_data(g,type){
    var selCityId = g("#sel_city_id").val();
    var selStateId = g("#sel_state_id").val();
    var filterItemParam = (!selCityId?["ci"]:[]);  
    var nlpEnabled = g("#nlpenabled").val();
    var inpWidth = (type=='nlp')?('682px'):('592px');
    var phText = "";
    if(type=='nlp'){
      phText = g("#localitycity_select1").data('placeholder');
    }else{
      phText = g("#localitycity_select").data('placeholder');
    }
    var AS_config_data = {
      placeHolder: phText,
      className: "headerSearchBox",
      dataUrl: "/ipautosearch/getdata-all.php",
      params: {'stateid':selStateId,'cityid':selCityId,'nlp':nlpEnabled},
      maxHeight: 65,
      inputWidth: inpWidth,
      isNlpEnabled: nlpEnabled,
      filterItems: filterItemParam,
      as_data_changed: function(obj){
        var data = obj.get_val();
        var data_zo = "", 
          data_cl = "", 
          data_lo = "", 
          data_bl = "", 
          data_pj = "", 
          data_ci = "";
        if(data['ad'].length>0) {
          g.each(data['ad'].join(",").split(","),function(k,v){
            v = parseInt(v);
            if(data['lo'].indexOf(v)<0)
              data['lo'].push(v);
          });
        }
        for(var i in data) {
          if(!data[i].length) 
            continue;
          if(i=="zo")
            data_zo = (g.map(data[i],function(v){return selCityId+"#"+v;})).join(",");
          else if(i=="cl")
            data_cl = (g.map(data[i],function(v){return selCityId+"~"+v;})).join(",");
          else if(i=="lo")
            data_lo = data[i].join(",");
          else if(i=="bl")
            data_bl = data[i].join(",");
          else if(i=="pj")
            data_pj = data[i].join(",");
          else if(i=="ci")
            data_ci = data[i].join(",");
        }
        var data_zoclloblpj = (g.grep([data_zo,data_cl,data_lo,data_bl,data_pj],function(v){
          return (v?1:0);
        })).join(",");
        if(g("#quicksearch").find("#ASCityId").length<=0)
          g("#quicksearch").append("<input type='hidden' id='ASCityId' name='ASCityId' />");
        g("#quicksearch").find("#ASCityId").val(data_ci);
        g("#reslocality").val(data_zoclloblpj);
        g("#zonename").val(data_zo);
        g("#clustername").val(data_cl);
        g("#qsbuildername").val(data_bl);
        g("#qsprojectname").val(data_pj);
      },
    };
    return AS_config_data;
  }
