          $(document).ready(function () {

            $(".userpasswordtext").removeClass("logintext").addClass("hidden");

            $(".userlogintext").removeClass("logintext").addClass("hidden");
        }).blur(function () {
            if (username.val().length == 0) {
                $(".userlogintext").removeClass("hidden").addClass("logintext");
            }
        });

        password.focus(function () {
            $(".userpasswordtext").removeClass("logintext").addClass("hidden");
        }).blur(function () {
            if (password.val().length == 0) $(".userpasswordtext").removeClass("hidden").addClass("logintext");
        });

        $(".userpasswordtext").click(function () {
            password.focus();
        });

        $(".userlogintext").click(function () {
            username.focus();
        });

        /*
        * Shows remember me and lost password dialog box
        */
        var login_help_hide = null;
        var login_help_delay = 2000;

        username.on("focusin", function () {
            $("#global-login form").addClass("help-mode");
            $("#login-help").show();
            clearTimeout(login_help_hide)
        });

        password.on("focusin", function () {
            $("#global-login form").addClass("help-mode");
            $("#login-help").show();
            clearTimeout(login_help_hide)
        });

        /*
        * Hides remember me and lost password dialog box
        */
        username.on("focusout", function () {
            login_help_hide = setTimeout(function () {
                $("#global-login form").removeClass("help-mode");
                $("#login-help").hide()
            }, login_help_delay);
        });
        password.on("focusout", function () {
            login_help_hide = setTimeout(function () {
                $("#global-login form").removeClass("help-mode");
                $("#login-help").hide()
            }, login_help_delay);
        });

        /*
        * Clear the timeout if the user clicks on the remember me checkbox
        */
        $("#global-login form input[type=checkbox]").on("click", function () {
            clearTimeout(login_help_hide)
        });
    });
