  function detectLogin(){

      if((Cookie.getCookie('wacsessionid') != null && Cookie.getCookie('wacsessionid') != 0)){
          //El cliente está logado en fijo o es convergente
          tipoLogin = "f";

      }else if(Cookie.getCookie('AUTH_SESSION_ID') != null || Cookie.getCookie('LESESSION') != null){
          //El cliente está logado en movil
          tipoLogin = "m";
      }else if((Cookie.getCookie('SMSESSION') != null || Cookie.getCookie('LESESSION') != null)){
          tipoLogin = "t";
      }

      if(tipoLogin != "not-set"){
          document.getElementById('a_acceso_satelites').style.display = "none";
          document.getElementById('a_desconectar').style.display = "block";
          document.getElementById('a_desconectarMobile').style.display = "block";

          $('#GW_conectedIcon').show();

    function logoutConvergente() {

        var urlLogoutCOLX = 'https://www.movistar.es/portal/site/COL10/template.COLX_LOGOUT/?logout=https://www.movistar.es/sr/sso/pub/servlet/logout';
        var urlLogoutNCCLI = 'https://www.movistar.es/mimovistar-cliente/publico/desconectar/colx';
        localStorage.removeItem("codigoOrden");
        localStorage.removeItem("numtlf");
        sessionStorage.removeItem("nameuser");
        sessionStorage.removeItem("urlhomeprivada");
        sessionStorage.removeItem("cclivr$$.api.misDatos");

        jQuery.ajax({
            url: urlLogoutNCCLI,
            method:'GET',
            async:false,
            cache: false,
            headers: {},
            success: function(data){},
            error: function(data){}
        });
        if(tipoLogin === 'm') {
            location.reload();
        } else {
            location.href= urlLogoutCOLX;
        }
    }

    function detectUserProperties(){  
        if(typeof(Storage) !== "undefined") {

            /*Buscamos los datos del saludo de COLX cargados en sessionStorage*/
            nameuser = sessionStorage.getItem('nameuser');


            if (nameuser !== null && ""!==nameuser && urlhomeprivada !== null && ""!==urlhomeprivada) {

                $( "#GW_username" ).html(nameuser);
                $('#GW_usernameMobile').html(nameuser);

                //$( "#usernamedesktop" ).prepend(nameuser);      
             }else{
                /*Buscamos los datos cargados desde CCLI*/
                datosCCLI= sessionStorage.getItem('cclivr$$.api.misDatos');
                if(datosCCLI !== null && '' !== datosCCLI){

                    var obj = jQuery.parseJSON(datosCCLI);
                    if( typeof(obj.datos) !== "undefined" && typeof(obj.datos.misDatos) !== "undefined" && typeof(obj.datos.misDatos.razonSocial) !== "undefined"){

                        $( "#GW_username" ).html(obj.datos.misDatos.razonSocial);      
                        $('#GW_usernameMobile').html(obj.datos.misDatos.razonSocial);
                        //$( "#usernamemovile" ).prepend(obj.datos.misDatos.razonSocial);      
                    }else if( typeof(obj.datos) !== "undefined" && typeof(obj.datos.misDatos) !== "undefined" && typeof(obj.datos.misDatos.apellido1) !== "undefined"
                          && typeof(obj.datos.misDatos.apellido2) !== "undefined" 
                       && typeof(obj.datos.misDatos.nombre) !== "undefined"){

                        nameuser = obj.datos.misDatos.apellido1 + " " + obj.datos.misDatos.apellido2 +", " + obj.datos.misDatos.nombre;
                        $( "#GW_username" ).html(nameuser); 
                        $('#GW_usernameMobile').html(nameuser);
                        //$( "#usernamemovile" ).prepend(nameuser);      
                        }
                }
             }				 
        }
    }

     function Utils(){};

  Utils.prototype.getCookie = function(c_name){
    var c_value = document.cookie;
    var c_start = c_value.indexOf(" " + c_name + "=");
    if (c_start == -1)
    {
      c_start = c_value.indexOf(c_name + "=");
    }
    if (c_start == -1)
    {
      c_value = null;
    }
    else
    {
      c_start = c_value.indexOf("=", c_start) + 1;
      var c_end = c_value.indexOf(";", c_start);
      if (c_end == -1)
      {
        c_end = c_value.length;
      }
      c_value = unescape(c_value.substring(c_start,c_end));
    }
    return c_value;
  };

  Utils.prototype.setCookie = function(c_name,value,exdays){

    var exdate= new Date();
    exdate.setDate(exdate.getDate() + exdays);
    var c_value=escape(value) + ((exdays==null) ? "" : "; expires="+exdate.toUTCString()+";path=/");
    document.cookie=c_name + "=" + c_value;

  };

  var utils = new Utils();
