   $(document).ready(function () {

            var token = $("input[name=__RequestVerificationToken]").val();

            function loadLatestContentwidget() {
                $.ajax({
                    method: "POST",
                    url: "/DoktorTV.Video/VideoProductionRepository/LatestVideosForWidget",
                    data: { size: 6, __RequestVerificationToken: token, videoType: "1" },
                    success: function (response) {
                        $("#news_tab").html(response);
                    }
                });

                $.ajax({
                    method: "POST",
                    url: "/DoktorTV.Video/VideoProductionRepository/LatestVideosForWidget",
                    data: { size: 6, __RequestVerificationToken: token, videoType: "3" },
                    success: function (response) {
                        $("#diseases_tab").html(response);
                    }
                });

                $.ajax({
                    method: "POST",
                    url: "/DoktorTV.Video/VideoProductionRepository/LatestVideosForWidget",
                    data: { size: 6, __RequestVerificationToken: token, videoType: "4" },
                    success: function (response) {
                        $("#doctors_tab").html(response);
                    }
                });

                $.ajax({
                    method: "POST",
                    url: "/DoktorTV.Video/VideoProductionRepository/LatestVideosForWidget",
                    data: { size: 6, __RequestVerificationToken: token, videoType: "2" },
                    success: function (response) {
                        $("#videoseries_tab").html(response);
                    }
                });
            }

            function searchTerms() {
                $(".latest-content-w-search-term").die("click");
                $(".latest-content-w-search-term").live("click", function () {
                    value = $(this).attr("data-value");
                    $("#lat_con_search_term_hidden").val(value);
                    $("#lat_con_search_term_hidden").closest("form").submit();

                    return false;
                });
            }

            loadLatestContentwidget();

            $(".navi-item").click(function () {

                tab = $(this).attr("data-tab");

                $(".tab_container").hide();
                $(".navi-item").removeClass("active");
                $(this).addClass("active");

                switch (tab) {
                    case "news":
                        $("#news_tab").show();
                        break;
                    case "diseases":
                        $("#diseases_tab").show();
                        break;
                    case "doctors":
                        $("#doctors_tab").show();
                        break;
                    case "videoseries":
                        $("#videoseries_tab").show();
                        break;
                }

                return false;

            });

            searchTerms();

        });
