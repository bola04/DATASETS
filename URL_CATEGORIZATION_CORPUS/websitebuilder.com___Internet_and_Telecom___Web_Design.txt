     browserAlert = new Object();

    browserAlert.runOnLoad = function() {

        if (Cookies.get('alert-box') == null) {
            Cookies.set('alert-box', 'open', { expires: 7, path: '/' });
            browserAlert.addBodyStyle();
        } else if(Cookies.get('alert-box') == 'close') {
            $(".browser-alert").hide();
            browserAlert.removeBodyStyle();
        }
    }

    browserAlert.hideMe = function() {
        Cookies.set('alert-box', 'close', {expires:7, path: '/' });
        $(".browser-alert").hide();
        browserAlert.removeBodyStyle();
    }

    browserAlert.addBodyStyle = function() {
        var doc = document.documentElement;
        doc.setAttribute('data-useragent', navigator.userAgent);

        var ieBAlert = $('html[data-useragent*="MSIE"] .browser-alert');
        ieBAlert.prependTo('body').removeClass('hidden');
        $('body').css({ 'top': ieBAlert.height(), 'position': 'relative' });
    }

    browserAlert.removeBodyStyle = function() {
        $('body').removeAttr('style');
    }

    browserAlert.runOnLoad();
    </script>

    <script type="text/javascript">
     $(function() {

function getCookie(cookieName) {
  var cookiesString = "; " + document.cookie;
  // split cookie to get key and value in 2 separate array elements
  var partsOfCookie = cookiesString.split("; " + cookieName + "=");

  // if there the requested cookie is present, the split array will have 2 string objects
  // 1st object will have cookie name (at the end), 2nd object will have cookie value.
  if (partsOfCookie.length === 2) {
    // take the second object if the array, split wrt ';' and return 
    // first value, which is the value of the cookie
    return partsOfCookie.pop().split(";").shift();
  }
}

  if( getCookie('cookieAccept') !== '1' ) {
    $('html, body').css('padding-bottom', '25px'); //set bottom padding
  }

  $('#hide-disclaimer').click(function (event) {
    event.preventDefault();
    Cookies.set('cookieAccept', '1');
    $( ".cookie-disclaimer" ).hide();
    $('html, body').css('padding-bottom', '0px'); //reset bottom padding
  });
});
    </script>
