        Heidelberg
       </a>
       ,
       <a href="http://www.embl-barcelona.es/" title="EMBL Barcelona">
        Barcelona
       </a>
       ,
       <a href="http://www.embl-hamburg.de" title="EMBL Hamburg">
        Hamburg
       </a>
       ,
       <a href="http://www.embl.fr" title="EMBL Grenoble">
        Grenoble
       </a>
       ,
       <a href="http://www.embl.it" title="EMBL Rome">
        Rome
       </a>
       <a href="http://www.embl.it" title="EMBL Monterotondo">
       </a>

        EMBL-EBI Hinxton
