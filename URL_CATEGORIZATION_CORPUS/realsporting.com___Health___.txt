    jQuery(document).ready(function ($) {
        jQuery(".menu > li").hover(function () {
            if ($(this).find("> a").next().hasClass("sub-menu")) {
                $(".line_nav_bottom").show();
            }
        }, function () {
            if (!$(".line_nav_bottom").hasClass("current") || $("body").attr("id") == "body-home") {
                $(".line_nav_bottom").hide();
            }
        });

        jQuery(".sub-menu,.sub-menu li,.sub-menu li a").mouseenter(function () {
            $(".line_nav_bottom").show();
            $(this).prev().addClass("currenthover")
        });

        jQuery(".sub-menu").mouseleave(function () {
            $(this).prev().removeClass("currenthover")
        });

        jQuery(".menu > li").hover(function () {
            if (!$(this).hasClass("li-current") && $(this).find("ul").length > 0) {
                $("header.menu4 .menu > li > a.current").not($(this)).addClass("hover");
            }
            else {
                if ($(this).find("> ul").length == 0) {
                    $(this).find("> a").addClass("nohover");
                }
            }
        }, function () {
            if (!$(this).hasClass("li-current")) {
                $("header.menu4 .menu > li > a.current").not($(this)).removeClass("hover");
            }
            $(this).find("> a").removeClass("nohover");
        });

        if (jQuery(".menu > li.li-current").find(".sub-menu").length > 0) {
            $(".line_nav_bottom").addClass("current");
        }

        if (jQuery("a.subseccion-desplegable").hasClass("current")) {
            jQuery("a.subseccion-desplegable.current").parent().addClass("li-current-2");
            jQuery("a.subseccion-desplegable.current").parent().find(".submenu-n3 li a").each(function () {
                if (jQuery(this).attr("class") === "current") {
                    jQuery(this).parent().parent().addClass("current");
                    return false;
                } else {
                    jQuery(this).parent().parent().removeClass("current");
                }
            });
        }

        jQuery(document).on("click touchend", ".sb-left ul li a.more", function (e) {
            e.preventDefault();
            jQuery(".submenu-n3").removeClass("current");
            jQuery(".sb-left ul li a.more").not($(this)).parent().removeClass("li-current", 200);
