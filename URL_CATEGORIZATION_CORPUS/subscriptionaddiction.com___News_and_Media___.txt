     function trashClick() {
			$.get("https://www.SubscriptionAddiction.com/AJAXremove.jsp?index="+$(this).data("index"),
				    function(rtnData, status){
				        if(status=="success"){
				        	cartJSON = jQuery.parseJSON(rtnData);
				        	$("#cartDesktop").find(".cartRespons").html(cartJSON.count);
				        	$("#cartMobileCount").html(cartJSON.count);
				        	$("#minicart .mCSB_container,#fullcart .mCSB_container").load("https://www.SubscriptionAddiction.com/AJAXCartContents.jsp");
				        	bindTrashCans();
				        	$(".cpnvalue").html(cartJSON.coupon!=null?cartJSON.coupon.savings:'0.00');
				        	if(cartJSON.coupon!=null)
   				        		$(".cpntotal").show();
				        	$(".subtotalvalue").html(cartJSON.total);
				        	$("#cartTotalValue").html(cartJSON.total);
				        	if(cartJSON.count<=0)
				        		$(".checkout-btn").hide();
				        }
				        else{
				        	
				       	}
			});
     }
