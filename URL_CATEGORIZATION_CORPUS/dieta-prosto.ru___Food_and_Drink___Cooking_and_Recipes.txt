                            var direction = 0;
                            var intervalId = setInterval(autoShift, 3000);
                            var prevButton = document.getElementById('prev_button');
                            var nextButton = document.getElementById('next_button');
                            var resizeFlag = false;
                            window.addEventListener('resize', onResize);
                            onResize();
                            complete();
                            function onResize() {
                              if ($(window).width() <= 900) {
                                if (resizeFlag) {
                                  resizeFlag = false;
                                  prevButton.removeEventListener('click', shift);
                                  nextButton.removeEventListener('click', shift);
                                  ul.removeEventListener('mouseover', disableAutoShift);
                                  ul.removeEventListener('mouseout', enableAutoShift);
                                }
                              } else {
                                if (!resizeFlag) {
                                  resizeFlag = true;
                                  prevButton.addEventListener('click', shift);
                                  nextButton.addEventListener('click', shift);
                                  ul.addEventListener('mouseover', disableAutoShift);
                                  ul.addEventListener('mouseout', enableAutoShift);
                                }
                              }
                            }
                            function enableAutoShift() {
                              intervalId = setInterval(autoShift, 3000);
                            }
                            function disableAutoShift() {
                              clearInterval(intervalId);
                            }
                            function autoShift() {
                              var pos = parseInt(ul.style.marginLeft || 0);
                              var posMin = -(imgs.length - count) * width;
                              var posMax = 0;
                              if (pos >= posMax) {
                                direction = 0;
                                nextButton.click();
                              } else if (pos <= posMin) {
                                direction = 1;
                                prevButton.click();
                              } else if (direction == 0) {
                                nextButton.click();
                              } else {
                                prevButton.click();
                              }
                            }
                            function shift() {
                              var pos = parseInt(ul.style.marginLeft || 0);
                              var posMin = -(imgs.length - count) * width;
                              var posMax = 0;
                              var end;
                              var duration = 70;
                              if (this == nextButton) {
                                end = Math.max(posMin, pos - width * count);
                                duration *= (pos - end) / width;
                              } else {
                                end = Math.min(posMax, pos + width * count);
                                duration *= (end - pos) / width;
                              }
                              animateProp({
                                elem: ul,
                                prop: 'marginLeft',
                                start: pos,
                                end: end,
                                duration: duration,
                                complete: complete,
                                delay: 1
                              });
                              return false;
                            }
                            function complete() {
                              var pos = parseInt(ul.style.marginLeft || 0);
                              var posMin = -(imgs.length - count) * width;
                              var posMax = 0;
                              if (pos >= posMax) {
                                prevButton.style.display = 'none';
                              } else {
                                prevButton.style.display = 'block';
                              }
                              if (pos <= posMin) {
                                nextButton.style.display = 'none';
                              } else {
                                nextButton.style.display = 'block';
                              }
                            }
                          })();
      </script>

    function setCookie(name, value, time, path) {
      var date = new Date();
      date.setTime(date.getTime() + (time * 1000));
      document.cookie = name + "=" + value + ";"
        + "expires=" + date.toUTCString() + ";"
        + "path=" + (path || "/");
    }
    function getCookie(name) {
      var nameEQ = name + "=";
      var ca = document.cookie.split(';');
      for(var i=0;i < ca.length;i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
      }
      return null;
