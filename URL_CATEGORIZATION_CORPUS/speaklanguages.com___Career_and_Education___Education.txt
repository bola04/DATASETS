   var select_language = document.getElementById("select_language");
        var disable_content = document.getElementById("disable_content");

        var currentLanguage = document.getElementById("language_menu");
        if (!currentLanguage.addEventListener) {
                currentLanguage.attachEvent("onclick", toggleSelector);
        } else {
                currentLanguage.addEventListener("click", toggleSelector);

        function toggleSelector(e) {
                if (hasClass(select_language, "open")) {
                        select_language.removeAttribute("class");
                        disable_content.removeAttribute("class");
                } else {
                        select_language.className = select_language.className + "open";
                        disable_content.className = "opacity";
                }
        }

        function hasClass(element, cls) {
                return (' ' + element.className + ' ').indexOf(' ' + cls + ' ') > -1;
        }

        document.onclick = function(event) {
                var target = getEventTarget(event);
                var hasParent = false;

                if (!target.id) {
                    return;
                }

                for(node = target; node != document.body; node = node.parentNode) {
                        if (node.id == 'select_language' || node.id == 'en') {
                                hasParent = true;
                                break;
                        }
                }
                if (!hasParent) {
                        var select_language = document.getElementById("select_language");
                        if (hasClass(select_language, "open")) {
                                select_language.removeAttribute("class");
                                disable_content.removeAttribute("class");
                        }
                }
        }

        function getEventTarget(e) {
                e = e || window.event;
                return e.target || e.srcElement;
        }

        document.addEventListener("DOMContentLoaded", function() {
                // Function to find the closest element, Source: http://bit.ly/1KQEUeF
                function closest(e,t){var n;for(["matches","webkitMatchesSelector","mozMatchesSelector","msMatchesSelector","oMatchesSelector"].some(function(e){return"function"==typeof document.body[e]?(n=e,!0):!1});null!==e;){if(parent=e.parentElement,null!==parent&&parent[n](t))return parent;e=parent}return null}
                // Function to save the user's locale
                function save_user_locale(e) {
                        xhr = new XMLHttpRequest();
                        xhr.open("GET", "/scripts/locale_selector_handler.cgi?locale_id=" + e.target.locale_id, true);
                        xhr.send();
                };
                var select_language_div = document.getElementById('select_language');
                var locale_selector_link = select_language_div.querySelectorAll('a');
                // Attach handlers to save user's locale
                for (var i = 0; i < locale_selector_link.length; i++) {
                        locale_selector_link[i].className = locale_selector_link[i].className + ' user_locale';
                        locale_selector_link[i].addEventListener('click', save_user_locale, false);
                        locale_selector_link[i].locale_id = closest(locale_selector_link[i], 'li').id;
                }
