   function getCookieVal (offset) {
	        var endstr = document.cookie.indexOf (";", offset);
	        if (endstr == -1)
	        endstr = document.cookie.length;
	        return unescape(document.cookie.substring(offset, endstr));
	    }
	    
	    function GetCookie (name) {
	        var arg = name + "=";
	        var alen = arg.length;
	        var clen = document.cookie.length;
	        var i = 0;
	        while (i < clen) {
	            var j = i + alen;
	            if (document.cookie.substring(i, j) == arg)
	            return getCookieVal (j);
	            i = document.cookie.indexOf(" ", i) + 1;
	            if (i == 0) 
	            break; 
	        }
	        return null;
	    }
	    
	    function SetCookie (name, value) {
	        var argv = SetCookie.arguments;
	        var argc = SetCookie.arguments.length;
	        var expires = (2 < argc) ? argv[2] : null;
	        var path = (3 < argc) ? argv[3] : null;
	        var domain = (4 < argc) ? argv[4] : null;
	        var secure = (5 < argc) ? argv[5] : false;
	        document.cookie = name + "=" + escape (value) +
	        ((expires == null) ? "" : ("; expires=" + expires.toGMTString())) +
	        ((path == null) ? "" : ("; path=" + path)) +
	        ((domain == null) ? "" : ("; domain=" + domain)) +
	        ((secure == true) ? "; secure" : "");
	    }

	    function TakeOver() {
	        var expdate = new Date();
	        var visitCount;
	        var takeoverThePage;

	        // set the cookie expiration date
	        expdate.setTime(expdate.getTime() + (expireDays * 1000 * 60 * 60 * 24)); 

	        // on the first visit, set visitCount to 0
	        if(!(visitCount = GetCookie("NumVisits"))) visitCount = 0;        
	        // otherwise get visit from the cookie
	        else visitCount = GetCookie("NumVisits"); 
	        
	        // count this visit and store the cookie
	        visitCount++;
	        SetCookie("NumVisits", visitCount, expdate, "/", null, false);

	        // check if we should takeoverThePage
	        if(!(takeoverThePage = GetCookie("ShowTakeOver")) ) {
	        	takeoverThePage = true;
	        	SetCookie("ShowTakeOver", takeoverThePage, expdate, "/", null, false);
	        }
	        else takeoverThePage = GetCookie('ShowTakeOver');
	        
	        // if we've reached the visitToShow and haven't opted out
	        if((visitCount >= visitToShow) && (takeoverThePage == 'true')) {
