  // Called when delivery form is submitted
  function submitDeliveryAddress(){
    console.log("submitting delivery address")

    // Hide search box on pressing submit
    $(".search-box").removeClass("search-box-open");
    hideAddressRow(addressInputRowDOM);

    var form = document.forms['addressForm'];
    var address = form['streetNumber'].value+' '+form['streetAddress'].value+', '+ form['city'].value+', '+form['state'].value+' '+form['zipCode'].value+', USA';

    var geocoder = new google.maps.Geocoder();
    geocoder.geocode({'address':address},function(results,status){

      var lat = parseFloat(results[0].geometry.location.lat());
      var lng = parseFloat(results[0].geometry.location.lng());

      if(!results[0].partial_match) {
        var pathname = window.location.pathname;

        if(pathname == '/location') { // If we're on location page show directly the results
          fetchData(lat, lng, form['streetNumber'].value, form['streetAddress'].value, form['address_line_2'].value, form['city'].value, form['state'].value, form['zipCode'].value);
          $("#btnShowMap").show(); // Display button to show the whole map, since opening delivery input hides it automatically
        }else{ // If we're not on locations page submit the hidden form and POST data to /locations
          console.log("delivery values set");
          $( "#json_place_lat" ).val(lat);
          $( "#json_place_lng" ).val(lng);
          $( "#json_place_street_num" ).val(form['streetNumber'].value);
          $( "#json_place_address" ).val(form['streetAddress'].value);
          $( "#json_place_address2" ).val(form['address_line_2'].value);
          $( "#json_place_city" ).val(form['city'].value);
          $( "#json_place_state" ).val(form['state'].value);
          $( "#json_place_zip" ).val(form['zipCode'].value);
          $('#form_locations_header').submit(); // Submit form with set input fields
        }

      } else {
        alert('Cant find the specified address, please try reviewing the fields');
      }
    })
  }

  var componentForm = {
    street_number: 'short_name',

  });

  autocomplete.addListener('place_changed', fillInAddress);

  function fillInAddress(place) {
    // Get the place details from the autocomplete object.

   // var place = autocomplete.getPlace();
    oldSelectValue = selectValue;
    for (var component in componentForm) {
      document.getElementById(component).value = '';
      document.getElementById(component).disabled = false;
    }

    // Get each component of the address from the place details
    // and fill the corresponding field on the form.
    for (var i = 0; i < place.address_components.length; i++) {
      var addressType = place.address_components[i].types[0];
      if (componentForm[addressType]) {
        var val = place.address_components[i][componentForm[addressType]];
        document.getElementById(addressType).value = val;
      }
    }

    var empty = false;
    $('.required').each(function() {
      if ($(this).val() == '') {
        empty = true;
      }
    });

    if (empty) {
      $('#submit-address-form').attr('disabled', 'disabled');
    } else {
      $('#submit-address-form').removeAttr('disabled');
    }

    selectValue = $('#administrative_area_level_1').val();
    if(oldSelectValue !== selectValue){
      if (!$('#administrative_area_level_1').val()) {
        $('#administrative_area_level_1').css('color', 'red');
      } else {
        $('#administrative_area_level_1').css('color', 'black');
      }
    }
  }
