           <script type="text/javascript">
            function initAccordion() {
	
	if (top.location.href != window.location.href) {
		return;
	}
	
	var queue = [];
	
	$('.accordion>ul>li>a').click(function(e){
			
			if($(this).next().hasClass('accordion') == false)
			{
				return true;
			}
			e.stopPropagation();
			var _this = this;

			topParent = $(_this).parent();

			topParent.attr('id', getUniqueId());

			if(!_this.id.match(/[a-z]+_\d+/))
			{	
				var uid = getUniqueId();
				_this.id =  uid;
			}
			
			var matches = 0;
			$.each(queue, function(i, val){
				len = $('#' + val).parent().find('.accordion').find('#'+_this.id).length;
				//if($(_this).parent().prev().find('a:first').attr('id') == val){
				 if(len <= 0){
					//if the link in the queue is in sibling li of the current
					//we shoud remove it and hide it
					queue.splice(i, 1);
				}
				matches += len > 0 ? 1 :0;
				
			})
			
			if(matches == 0) {//isNewBranch
				$('.accordion>ul').each(function(i){
					if(i == 0)
					{
						return;
					}
					$(this).slideUp(500);
				})
				queue = [];
			}
			else
			{
				$('.accordion>ul').each(function(i){
					if(i == 0)
					{
						return;
					}
					if($.inArray($(this).parent().prev().get(0).id, queue) == -1){
						$(this).slideUp();
					}
				})
			}
			
			var pos = $.inArray(_this.id, queue);

			if(pos < 0)
			{
				queue.push(_this.id)
			}
			var list =$(_this).parent().find('.accordion:first').find('ul:first');
			list.is(':visible') ? list.slideUp(500) : list.slideDown(500);;
			
			return false;
			
	});
	
	$('.accordion>ul>li.isOpen>a').click();
}

function getUniqueId()
{
	var classIndex = 0;
	var classPrefix = 'uid_';
	while($('#' + classPrefix.concat(classIndex)).length != 0){
		classIndex++;
	}
	return 	classPrefix.concat(classIndex);
}

$(function () {
	initAccordion();
});
