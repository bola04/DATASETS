                        }
                    },
                    error: function (responseError, t) {
                        if ("timeout" != t){
                            var suggestion;
                            if(responseError.hasOwnProperty('responseJSON')){
                                suggestion = responseError.responseJSON.suggestion;
                            }else if (responseError.hasOwnProperty('responseText')){
                                suggestion = JSON.parse(responseError.responseText).suggestion;
                            }else{
                                return;
                            }

                            var suggestionsSpans = [];
                            var i = 0;
                            suggestion.forEach(function (entry) {
                                i++;
                                suggestionsSpans.push('<span data-id="chatbox-suggestion-btn-link" data-suggestion-sources="username" data-num="'+i+'" class="btn-link chat-suggestion-btn-link">' + entry + '</span>');
                            });

                            var suggestionMessage = suggestionsSpans.join(' | ');
                            $("[data-js=chatbox_suggestions_nick]").removeClass('hide');
                            $("[data-js=suggestions_no_valid]").html(suggestionMessage);

                        }
                    }
                });
            };
