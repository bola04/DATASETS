      //reglementare legala de afisarea mesaj - acest site foloseste cookies.
function setCookie(cname, cvalue, exdays) {
	 var d = new Date();
    d.setTime(d.getTime() + (exdays*24*60*60*1000));
    var expires = "expires="+d.toUTCString();
		
		//var options = {path: '/', secure: true};
		var options = {path: '/'};
		
		return (document.cookie = [
				cname + "=" + cvalue + "; " + expires,
				options.path    ? '; path=' + options.path : '',
				options.domain  ? '; domain=' + options.domain : '',
				options.secure  ? '; secure' : ''
			].join(''));
		
    //document.cookie = cname + "=" + cvalue + "; " + expires;
}

// citim valoarea din cookies
function getCookie(cname) {
     var name = cname + "=";
		 //alert(document.cookie);
     var ca = document.cookie.split(';');
     for(var i=0; i<ca.length; i++) {
         var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1);
         if (c.indexOf(name) == 0) return c.substring(name.length,c.length);
     }
     return "";
}

// verificam daca avem acordul dat pentru cookies

{
	var AcordCookieEuPlatesc = getCookie('AcordCookieEuPlatesc');

		return true;
	else
		return false;
}

{

// verificam daca avem cererea deja data si daca da ascundem bannerul prin care cerem acordul
