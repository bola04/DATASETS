      jQuery(document).ready( function($) {
			var ajaxUrl = "https://www.hrf.net/wp-admin/admin-ajax.php";

			$('#faq_search').on('submit', function(e){
				e.preventDefault();
				var searchterm = $('#faq_search .faq_search_input').val();
				doSearch(searchterm);
			});

			function doSearch(searchterm){
				searchterm = typeof searchterm !== 'undefined' ? searchterm : 0;
				var result = 0;
				$('#faq_list').html('<li><a class="accordion-message">Söker...</a></li>');
				$.get('/wp-json/wp/v2/faq/?per_page=100&search='+searchterm)
				.success(function(posts) {
					$('#faq_list').html('');
					for (var i = 0; i < posts.length; i++) {
						var template = wp.template('aside-faq');
						if(contains(posts[i].faq_category, 221)) {
							$('#faq_list').append( template( posts[i] ) );
							result++;
						}
					}
					if (result === 0) {
						$('#faq_list').html('<li><a class="accordion-message">Hittade tyvärr inga frågor.</a></li>');
					}
					Foundation.reInit('accordion');
				});

				function contains(a, obj) {
					for (var i = 0; i < a.length; i++) {
						if (a[i] === obj) {
							return true;
						}
					}
					return false;
				}
			}
		});
