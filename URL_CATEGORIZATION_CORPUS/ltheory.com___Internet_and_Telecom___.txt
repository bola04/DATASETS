   $(window).scroll(function() {
		var scrollTop = $(this).scrollTop();
		$('#menu').css({
			function() {
				var elementHeight = 872,
				opacity = ((1 - (elementHeight - scrollTop) / elementHeight) * 0.25) + 0.75;
				$('#menu').css('background-color', 'rgba(0,0,0,' + opacity + ')');
			}
		});
		var picHeight = 1080.0;
		var fadeHeight = 500;
		var windowHeight = $( window ).height() - 60;
		var maxPic = (picHeight * -1) + fadeHeight;
		$('#backslider.bs1').css({
			function() {
				var fadeTop = $("div.bs1 > div#screendivide > div.fade").offset().top;
				var fadeDistance = (fadeTop - scrollTop) + fadeHeight - 60;
				var level = Math.max(0, 1 - (fadeDistance / (windowHeight + 500)));
				level = (Math.sqrt(.5 * level ) + level) / 2;
				$('#backslider.bs1').css('background-position', 'center ' + (maxPic * level) + 'px');
			}
		});
		
		$('#backslider.bs2').css({
			function() {
				var fadeTop = $("div.bs2 > div#screendivide > div.fade").offset().top;
				var fadeDistance = (fadeTop - scrollTop) + fadeHeight - 60;
				var level = Math.max(0, 1 - (fadeDistance / (windowHeight + 500)));
				level = (Math.sqrt(.5 * level ) + level) / 2;
				$('#backslider.bs2').css('background-position', 'center ' + (maxPic * level) + 'px');
			}
		});
		
		$('#backslider.bs3').css({
			function() {

			}
		});
	});
