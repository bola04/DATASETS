  <script type="text/javascript">
   var featuredEventsBtn = $("<a>").addClass("featured-events-btn").attr("href", "#").html("Back to Viewing Featured Events >").on("click", function (e) { showFeaturedEvents(); return false; });

        function transitionIn(el) {
            $(el).addClass("current_events events_start").removeClass("hidden").prependTo("#events-container");
            if (!$(el).is("#featured-events")) {
                $(el).prepend($("<a>").addClass("featured-events-btn").attr("href", "#").html("Back to Viewing Featured Events >").on("click", function (e) { showFeaturedEvents(); return false; }));
            }
            window.getComputedStyle($(".events_start").get(0)).left;
            $(".events_start").removeClass("events_start");
            if (typeof (useJSTransitions) != "undefined" && useJSTransitions) {
                $(".current_events").slideDown();
            }

            $(".events_end").addClass("hidden");
        }

        function transitionOut() {
            // clean up old hidden dates
            if ($("#featured-events").hasClass("events_end")) {
                $("#featured-events").removeClass("events_end").addClass("hidden").insertAfter("#events-container");
            }
            $(".events_end").remove();

            // start ending animation immediately
            $("#events-container").children(":last-child").removeClass("current_events").addClass("events_end");
        }

        function showFeaturedEvents() {
            transitionOut();
            transitionIn($("#featured-events"));
        }

        function gotEventsResult(data) {
            // start animation with pause

            transitionIn($(data));

        }

        var onSelectDate = function (dateText, inst) {
            transitionOut();

            dateText = dateText.split("/");

            $.get('/calendar/_day?byear=' + dateText[2] + '&bmonth=' + dateText[0] + '&bday=' + dateText[1] + '&filter=' + filter, gotEventsResult)
                .fail(function (e) { transitionIn($("<div />").html("<p>An error occured while loading performances</p>").prepend(featuredEventsBtn)); });

        };

        function initCal() {
            $("#calendar1").datepicker({ defaultDate: null, onSelect: onSelectDate });
        }

        $(document).ready(initCal);
  </script>

          <em>

Baltimore Symphony Orchestra, Knoxville Symphony Orchestra, Jacksonville Symphony, and Orpheus Chamber Orchestra (NY) will spotlight diverse repertoire that goes beyond traditional orchestral fare in multi-genre thematic collaborations—explorations of geography and community, meditations on “bridges” both literal and metaphorical, new takes on Shakespeare, and more. Performances will also feature commissioned works, dialogue and other vocal elements, and projections.
