    var viewCache = {lastView: ''};

    /**
     * Visszaadja a menü html-t cache-ből
     *
     * @param {string} viewType
     * @return string|undefinedfined
     */
    function getMenuHtmlFromCache(viewType) {
        return viewCache[viewType];
    }

    /**
     * Újraszámolja az oszlopokat
     */
    function recolumnizeIfNeeded() {
        var viewType = getViewType();
        var menuHtml = getMenuHtmlFromCache(viewType);

        if (viewType != viewCache.lastView) {
            if (typeof menuHtml != 'undefined') {
                // találat cache-ből
                $categories.html(menuHtml);
                viewCache.lastView = viewType;
                return;
            }

            rebuildMenu(viewType);
        }
    }

    /**
     * Újragenerálja a menüt
     *
     * @param {string} viewType
     */
    function rebuildMenu(viewType) {
        $categories.html(viewCache['phone']);
        $categories.find('.column > *').unwrap();

        $product_categories.addClass('freeze');
        $dds = $categories.find('.dropdown_flap');
        $dds.addClass('freeze');
        $dds.each(function () {
            // darabolás előtt kimentjük a képet
            var $img = $(this).find('.category_image');
            // Minden menüpont oszlopait úraszámoljuk
            switch (viewType) {
                case 'desktop':
                    $img = $img.detach();// detacheljük a képet, hogy ne számítódjon bele az utolsó oszlop magasságába
                    $(this).find('.main_types')
                            .bettercolumnize({columns: 4, lastNeverTallest: true})
                            .end()
                            .append($img);
                    break;
                case 'tablet':
                    $(this).find('.main_types')
                            .bettercolumnize({columns: 3, lastNeverTallest: true})
                            .find('.column').last()
                            .append($img);
                    break;
            }
        });
        $dds.removeClass('freeze');
        $product_categories.removeClass('freeze');

        viewCache[viewType] = $categories.html();
    }

    $(function(){
        viewCache['phone'] = $categories.html();
        viewCache.lastView = 'phone';
        $(window).resize(recolumnizeIfNeeded).trigger('resize');
    });
