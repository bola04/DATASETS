    // Gets the latest stock price
    function StartStockTicker() {		
      $.ajax({
        url: _alsn_stockquery,
        dataType: "json",
        success: function (data) {
          if (data) {
			//Convert Data into an array
			var dataArray = [];
			var keys = Object.keys(data);
			keys.forEach(function(key){
				dataArray.push(data[key]);
			});			
			//convert dataArray objects into Array objects
			var quote = [];
			var quoteKeys = Object.keys(dataArray[1]);
			quoteKeys.forEach(function(quoteKey){
				quote.push(dataArray[1][quoteKey]);
			});
			//declare current quote and previous quote
            var lastQuote = quote[0];
			var prevQuote = quote[1]
			//Get data from quotes to display 
			var low = parseFloat(lastQuote["3. low"]);
			var adjustedClose = parseFloat(lastQuote["5. adjusted close"]);
			var yesterdayClose = parseFloat(prevQuote["4. close"]);
			
			var change = (adjustedClose - yesterdayClose).toFixed(2);

            $(".stock-amount").html("$" + new Number(adjustedClose.toFixed(2)));
            $(".stock-points").html(new Number(change));

			$(".current").html($("#InvestorRelationsContent_C007_ctl00_ctl00_CurrentText").text());
			
            if (change >= 0) {
              $(".stock-arrow").removeClass("red-arrow").addClass("green-arrow");
              $(".stock-points").removeClass("red-points").addClass("green-points");
            }
            else {
              $(".stock-arrow").removeClass("green-arrow").addClass("red-arrow");
              $(".stock-points").removeClass("green-points").addClass("red-points");
            }
          }
        },
        error: function (request, status, error) { console.log(status + ", " + error); }
      });
    }
