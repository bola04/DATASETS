function isNumber(inputStr) {
        for (var i = 0; i < inputStr.length; i++) {
                var oneChar = inputStr.charAt(i)
                if (oneChar < "0" || oneChar > "9") {
                        return false
                }
        }
        return true
}

// function to determine if value is in acceptable range for this application
function inRangeDay(inputStr) {
       num = parseInt(inputStr)
        if (num < 1 || num > 31) {
                return false
        }
        return true
}


// function to determine if value is in acceptable range for this application
function inRangeYear(inputStr) {
        num = parseInt(inputStr)
        if (num < 1900 || num > 3000) {
                return false
        }
        return true
}

// Master value validator routine for day
function isValidDay(inputStr) {
        if (isEmpty(inputStr)) {
                alert("من فضلك ادخلي اليوم في حقل اليوم قبل ان تضغطي على الزر أحسب")
                return false
        } else {
                if (!isNumber(inputStr)) {

                        alert("من فضلك تأكدي من ان اليوم ادخل في صيغة رقمية")
                        return false
                } else {
                        if (!inRangeDay(inputStr)) {
                                alert("من فضلك ادخلي يوم صحيح")
                                return false
                        }
                }
        }
        return true
}


// Master value validator routine for year
function isValidYear(inputStr) {
        if (isEmpty(inputStr)) {
                alert("من فضلك ادخلي السنة في حقل السنة قبل ان تضغطي على الزر أحسب")
                return false
        } else {
                if (!isNumber(inputStr)) {
                        alert("من فضلك تأكدي من أن السنة ادخلت في صيغة رقمية")
                        return false
                } else {
                        if (!inRangeYear(inputStr)) {
                                alert("من فضلك ادخلي سنة صحيحة")
                                return false
                        }
                }
        }

        return true
}

        this.length = n
