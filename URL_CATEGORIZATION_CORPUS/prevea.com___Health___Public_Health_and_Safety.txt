        mylng=position.coords.longitude;
        var latlng = new google.maps.LatLng(mylat, mylng);
        
        geocoder.geocode({'latLng': latlng}, function(results, status) {
            if (status == google.maps.GeocoderStatus.OK) {
                myzipcode = extractFromAdress(results[0].address_components, "postal_code");
                doRedirect();
                
            }
            else {
                myzipcode = '';				
                doRedirect();
            }				
        });
    }
    
    function extractFromAdress(components, type){
        for (var i=0; i<components.length; i++)
            for (var j=0; j<components[i].types.length; j++)
                if (components[i].types[j]==type) return components[i].long_name;
        return "";
    }
    
    function processError(error) {
        myzipcode = '';		
        doRedirect();
    }
    
    function doRedirect()
    {
        switch (searchType) {
            case 'provider':
                location.href = "/providers?searchtype=provider&searchterm=" + $("#searchTerms").val() + "&zip=" + myzipcode + "&newsearch=1";
                break;
            case 'location':
                location.href = "/providers?searchtype=location&zip=" + myzipcode + "&newsearch=1";
                break;
        }
    }
