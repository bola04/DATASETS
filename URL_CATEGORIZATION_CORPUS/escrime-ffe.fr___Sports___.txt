				if($.isNumeric(tmp)){

	function initAutocompleteLocality(){
		$("#PCL_auto").show();
		$("#PCL_manu").hide();
		$("#form_pcl").attr('required', 'required');
		$("#form_pcl").attr('pattern', '\\d{5} - .{3,}');
		autocompleted = false;

		cache = {};
		$( "#form_pcl" ).autocomplete({
			minLength: 2,
			source: function( request, response ) {
				var term = request.term;

				if ( term in cache ) {
					response( cache[ term ] );
					return;
				}
				var route = "/autocomplete_CPVille/term"
				route = route.replace('term', term );
				$.getJSON( route, request, function( data, status, xhr ) {
					cache[ term ] = data;
					response( data );
				});
			},
			select: function( event, ui) {
				   cpLocality = ui.item.label.split(' - ');
				   locality = cpLocality[1];
				   cp = cpLocality[0];
				   $( "#form_postal_code" ).val(cp);
				   $( "#form_locality" ).val(locality);
				   autocompleted = true;
			}
		});
	}
