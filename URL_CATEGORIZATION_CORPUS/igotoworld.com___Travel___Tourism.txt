    function toggleSignInBlock()
    {
        var signInBlockObj = $('#signInBlock');
        signInBlockObj.toggleClass('displayNone');
        $('#signInLinks').toggleClass('displayNone');
        if(signInBlockObj.hasClass('displayNone'))
            $("#signInForm input.inputText").each(function(indx){
                $('div.'+this.id+'formError').remove();
            });
    }

    function SignIn()
    {
        $("div.formError").remove();
        var reg = /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,6})$/;
        var error = false;
        if(!reg.test($('#email').val())){
            $("#email").validationEngine("showPrompt","Invalid email address format", "pass", "", true);
            error = true;
        }
        if($('#password').val() == ""){
            $("#password").validationEngine("showPrompt","Necessary field", "pass", "", true);
            error = true;
        }

        if(!error)
        {
            var data = {
                login:		$('#email').val(),
                password:	$('#password').val(),
                rememberme: $('#remembermeinput:checked').length ? 1 : 0,
                method:		'SignIn',
                controller:	'SignInController'
            };

            $.ajax({
                type: 'POST',
                url: '',
                dataType: 'json',
                data: data,
                success: function (data) {
                    if(data.logined)
                        document.location = document.location;
                    else{
                        $("#email").validationEngine("showPrompt","Your login or password combination is invalid", "pass", "", true);
                    }
                }
            });

        }
    }
