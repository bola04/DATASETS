		if (tls_status == 'bad') { window.location.href = tls_redirect; }
	}
	
	function getTLSStatus () {
		var tls_status = 'good';

		// get browser info

		// passing the agent string from asp because compability mode is being forced
		// from the server, and it sees all IE as version 8 otherwise
		var ua_parser = new UAParser('Python-urllib/3.5');
		// var ua_parser = new UAParser();

		var ua_result = ua_parser.getResult();
		var ua_browser = ua_result.browser['name'];
		var ua_browser_ver = ua_result.browser['version'];
		ua_browser_ver = ua_browser_ver.replace(/\..*$/,''); // get major version only
		var ua_os = ua_result.os['name'];
		var ua_os_ver = ua_result.os['version'];
		ua_os_ver = ua_os_ver.replace(/\..*$/,''); // get major version only

		/*
		alert('Name: ' + ua_browser + ' -- Version: ' + ua_browser_ver);
		document.write('Name: ' + ua_browser + '<br />');
		document.write('Version: ' + ua_browser_ver + '<br />');
		document.write('OS Name: ' + ua_os + '<br />');
		document.write('OS Name: ' + ua_os_ver + '<br />');
		*/
		
		// determine browser status
		// TLS support reference used:
		// https://en.wikipedia.org/wiki/Template:TLS/SSL_support_history_of_web_browsers
		// Chrome
		if (ua_browser == 'Chrome' && ua_browser_ver < 22) { tls_status = 'bad' }
		// Google Android
		if (ua_browser == 'Android Browser' && ua_browser_ver < 5) { tls_status = 'bad' }
		// Internet Explorer (includes mobile)
		if (ua_browser.match(/IE/) && ua_browser_ver < 11) { tls_status = 'bad' }
		// Firefox (includes mobile)
		if (ua_browser == 'Firefox' && ua_browser_ver < 27) { tls_status = 'bad' }
		// Opera
		if (ua_browser.match(/Opera/) && (ua_browser_ver < 14 && ua_browser_ver != 9)) { tls_status = 'bad' }
		// Safari
		if (ua_browser == 'Safari' && (ua_browser_ver < 7 || ua_os != 'Mac OS')) { tls_status = 'bad' }
		// Safari Mobile
		if (ua_browser == 'Mobile Safari' && (ua_browser_ver < 5 || ua_os_ver < 5)) { tls_status = 'bad' }
		// return the status
		return tls_status;
	}
	
	function setCookie (name,value,expires,path,theDomain,secure) { 
		value = escape(value);
		var theCookie = name + "=" + value + 
		((expires)    ? "; expires=" + expires.toGMTString() : "") + 
		((path)       ? "; path="    + path   : "") + 
		((theDomain)  ? "; domain="  + theDomain : "") + 
		((secure)     ? "; secure"            : ""); 
		document.cookie = theCookie;
	} 

	function getCookie(Name) { 
		var search = Name + "=" 
		if (document.cookie.length > 0) { // if there are any cookies 
			var offset = document.cookie.indexOf(search) 
		  	if (offset != -1) { // if cookie exists 
				offset += search.length 
				// set index of beginning of value 
				var end = document.cookie.indexOf(";", offset) 
				// set index of end of cookie value 
				if (end == -1) end = document.cookie.length 
				return unescape(document.cookie.substring(offset, end)) 
			} 
		}
	} 

	function delCookie(name,path,domain) {
		if (getCookie(name)) document.cookie = name + "=" +
			((path)   ? ";path="   + path   : "") +
			((domain) ? ";domain=" + domain : "") +
			";expires=Thu, 01-Jan-70 00:00:01 GMT";
	}
