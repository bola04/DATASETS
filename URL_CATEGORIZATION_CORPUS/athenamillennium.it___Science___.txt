    <h1>
     Exegi monumentum aere perennius...
    </h1>
    <p>
     così cantava Orazio nella sua celebre Ode, sottolineando l'immortalità dell'opera poetica e del suo creatore.
     <br/>
     L'arte, in tutte le sue forme, rappresenta la sublimazione dell'uomo e di ciò che esso vuole esprimere.
     <br/>
     Tutto ciò che è
     <strong>
      Poiesis
     </strong>
     costituisce la linfa vitale di cui si nutre Athena sia che essa derivi dai grandi ingegni del passato sia da chi, seppur immerso nella frenesia del presente, provi il desiderio di trovare uno spazio in cui la contemplazione della realtà assuma forme simboliche ma non per questo meno vere.
    </p>
   </div>

	$("#addChat").on('click', '#sendChat', function() {
		var message = $("#message");
		//On key press
		message.keyup(validateMessage);
		if(validateMessage()){
			var operation = 'saveChat';
			var message = $('#message').val();
			var form_data = new FormData();                  
			form_data.append('operazione', operation);		
			form_data.append('message', message);			
			$.ajax({
				url: 'http://www.athenamillennium.it/config/save.php',
				dataType: 'text',  // what to expect back from the PHP script, if anything
				cache: false,
				contentType: false,
				processData: false,
				data: form_data,				
				type: 'post',
				success: function(data) {					
					$("#chat").load(location.href+" #chat>*","");
					$('#chat').show();
					$('#message').val('');
					$('#addMessage').show();		
					$('#addChat').hide();	
					
				}			
			});		
			return false;
		} else {			
			return false;	
		}			
		function validateMessage(){
			if(message.val().length > 0){			
				$("#message").css({"border":"1px solid #999999"});
				return true;
			} else{
				$("#message").css({"border":"1px solid #ff0000"});
				return false;
			}
		}						
	});
});
