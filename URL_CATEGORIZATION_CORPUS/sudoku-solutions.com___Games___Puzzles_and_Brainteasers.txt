  <script>
   $(document).ready( 
		// Initialise the popup
		function() {	
			$popupRoot = $("#popupComment");
			$popupRoot.find(".FAQButton").click(faqClickHandler);
		}
	);
	

	/**
	* Handle a FAQ click
	*/
	function faqClickHandler(e) {

		// Hide other FAQs
		$(".FAQButton").each(
			function() {
				if (this.id != e.target.id) {
					var contentId = this.id.replace("Button","Content");
					$("#" + contentId).hide();
				}
			}
		);

		// Toggle this FAQ
		var $contentSelector = $("#" + e.target.id.replace("Button","Content"));		
		if ($contentSelector.is(":visible")) {
			$contentSelector.hide();
		} else {
			$contentSelector.show();
		}
	}
	
	/**
	 * Send a comment request to the server and close the popup
	 * @param e
	 * @returns
	 */
	function sendClickHandler() {
		var gridContent = "";
		
		// Collect cell data for posting
		// TODO : Improve this. References the grid. 
		$gridCellSelector = $(".gridCell");
		$gridCellSelector.each( 
			function() {
				var cellContent = $(this).find(".gridCellValue").text();
				if (cellContent.length == 0) cellContent = ".";	
				gridContent += cellContent;			
			}
		);
		
		var request = {};
		request.comment = $("#COMMENT").val();
		request.gridContent = gridContent;

		$.ajax({
			type : "POST",
			url : "index.php",
			data : {
				service : "comment",
				request : JSON.stringify( request) 
			}
		});

		$("#popupComment").dialog("close");
	}

				modal : true
			}).parent().position(this.position);

		stepItemKeydownHandler : function (e) {
			var stepNumber = $(this).attr("tabIndex");
			switch (e.which) {
				// Up arrow
				case 38:
					var $previousStep = window.viewStepsPopup.$stepItems.filter("[tabIndex='" + --stepNumber + "']");
					if ($previousStep.length > 0) {
						$previousStep.focus();
						$previousStep.click();
					}
					break;

				// Down arrow
				case 40:
					var $nextStep = window.viewStepsPopup.$stepItems.filter("[tabIndex='" + ++stepNumber + "']");
					if ($nextStep.length > 0) {
						$nextStep.focus();
						$nextStep.click();
					}
					break;
			}
		},
		
		// Toggle info block
		infoBlockToggleHandler : function (e) {
			var popup = window.viewStepsPopup;
			if (popup.$viewStepsInfo.is(":visible")) {
				popup.$viewStepsInfo.hide();
				popup.$viewStepsToggleInfo.text(popup.$viewStepsToggleInfo.attr("data-more-info"));
			} else {
				popup.$viewStepsInfo.show();
				popup.$viewStepsToggleInfo.text(popup.$viewStepsToggleInfo.attr("data-less-info"));
			}			
		}, 

		stepItemClickHandler : function (e) {	
			var jsonDetails = $(this).attr("data-details");
			var jsonObj = JSON.parse(jsonDetails);
			
			// Fill the cells
			try {
				var gridEdits = utilMementoStart();
				for (var i=0; i < jsonObj.cells.length; i++) {
				     var cell = jsonObj.cells[i].cell;
				     var cellValue = (cell.val == ".") ? "" : cell.val;
				     var $cell = window.$gridSelector.find("#C" + cell.id);
				       
					 $cell.removeClass("inputcell hintcell");
					 $cell.addClass("cell");	
	
					 var $cellValueSelector = $cell.find(".gridCellValue");
					 utilMementoStore(gridEdits, $cellValueSelector, cellValue);		 
				     $cellValueSelector.text(cellValue);
				}
				utilMementoEnd(gridEdits);
		
				// Display the solved cell
				for (var i=0; i < jsonObj.hintCells.length; i++) {
				     var cell = jsonObj.hintCells[i].cell;
				     var cellValue = (cell.val == ".") ? "" : cell.val;
				     var $cell = window.$gridSelector.find("#C" + cell.id);
				       
					 $cell.removeClass("inputcell cell");
					 $cell.addClass("hintcell");
		
					 // If the cell was solved show the value
					 if (jsonObj.hintCells.length == 1) {
				     	$cell.find(".gridCellValue").text(cellValue);
					 }
				}
			}
			finally { 
				$(".stepItem").css("background","white");
				$(this).css("background","#D7D7D7");
			}
		}
	}

			saveRequest.identifier = this.$codeSelector.val();
			saveRequest.comment = this.$commentSelector.val();

			var cellContents = "";
			var candidateContents = "";
			window.$gridCellSelector.each(function() {
				var cellContent = $(this).find(".gridCellValue").text();
				if (cellContent.length == 0) {
					cellContents += ".";
				} else {
					cellContents += cellContent;
				}

				// Collect candidate info in a compressed form
				$(this).find(".gridCellCandidate").each(function() {
					var candidateContent = $(this).text();
					if (candidateContent.length == 0) {
						candidateContents += ".";
					} else {
						candidateContents += candidateContent;
					}
				});
			});

			saveRequest.cellContents = cellContents;
			saveRequest.candidateContents = candidateContents;

			// Client date
			var now = new Date();
			saveRequest.clientDate = $.datepicker.formatDate('dd M yy', now);

			// Client time
			var hours = now.getHours();
			var minutes = now.getMinutes();
			var seconds = now.getSeconds();
			saveRequest.clientTime = ((hours < 10) ? "0" + hours : hours) + ":"
					+ ((minutes < 10) ? "0" + minutes : minutes) + ":"
					+ ((seconds < 10) ? "0" + seconds : seconds);

			// Start the autosave service
			var saveService = Object.create(SaveService);
			saveService.request = saveRequest;
			saveService.execute();
		}
				
	}

	/**
 	* Load Popup Definition
	*/
	var LoadPuzzlePopup = {
		/** Attributes **/
		width : 0,

		height : 0,

		position : {},

		// Selector of the object which will be obscured by the popup
		$obscuredObject : null,

		$code : null,

		$resultContainer : null,

		$resultContainerHeader : null,

		$puzzleInfoTemplate : null,

		$errorText : null,
		
		/** Functions **/
		
		// Initialise the popup
		initialise : function () {
				this.$code = $("#loadIdentificationCode");
				this.$resultContainerHeader = $("#resultContainerHeader");
				this.$resultContainer = $("#resultContainer");
				this.$puzzleInfoTemplate = $("#puzzleInfoTemplate").find(".puzzleInfo");
				this.$errorText = $("#loadIdentificationCodeError");
			},

		// Display the popup		
		display : function () {
			var popup = this; 
			
			popup.$obscuredObject.hide();
									
			$("#popupLoadPuzzle").dialog({
				classes : {
					"ui-dialog" : "popupDialog rounded shadow",
					"ui-dialog-titlebar" : "noDisplay",
					"ui-dialog-buttonpane" : "popupDialog"
				},
				buttons : [ {
					text : "Close",
					icon : "ui-icon-close",
					iconPosition : "beginning",
					click : function() {
						popup.$obscuredObject.show();
						popup.reset();
						$(this).dialog("close");
					}
				} ],
				width : this.width,
				height : this.height,
				resizable : false,
				modal : true
			}).parent().position(this.position);
		},

		
		// Find puzzles
		findPuzzles : function () {
			// Runs in the context of the event so cannot use "this"
			var popup = window.loadPuzzlePopup;

			var identifier = popup.$code.val();
			if (identifier.length < 10) {
				popup.$errorText.show();
				return;
			}
			
			var request = {};
			request.identifier = identifier;

			// Execute the load service
			var findPuzzlesService = Object.create(FindPuzzlesService);
			findPuzzlesService.request = request;

			// Process the results
			var jsonPuzzles = JSON.parse(findPuzzlesService.execute());
			popup.$resultContainerHeader.show();
			popup.$resultContainer.show();
			
			for (var i = 0; i < jsonPuzzles.length; i++) {
				var jsonPuzzle = jsonPuzzles[i];
				$nextPuzzleInfo = popup.$puzzleInfoTemplate.clone();
				$nextPuzzleInfo.find(".puzzleComment").text(jsonPuzzle.comment);
				$nextPuzzleInfo.find(".puzzleTimestamp").text(jsonPuzzle.clientDate + " " + jsonPuzzle.clientTime);
				$nextPuzzleInfo.find(".puzzleCellContents").text(jsonPuzzle.cellContents);
				$nextPuzzleInfo.find(".puzzleCandidateContents").text(jsonPuzzle.candidateContents);
				$nextPuzzleInfo.appendTo(popup.$resultContainer);
			}
			
			// Attach a click handler after all objects have been created
			$(".puzzleInfo").click(puzzleInfoClickHandler);			
		},

		// Reset the popup
		reset : function () {
			this.$code.val("");
			this.$resultContainer.empty();
			this.$resultContainer.hide();
			this.$resultContainerHeader.hide();
			this.$errorText.hide();
		}
	}

   // Initialise the page
	$(document).ready( 

		function() {	
			
			// Do this first
			initialisePage($("#GRID_HISTORY").text());

			// Process the event
			var requestedEvent = "NONE";
			var $message = null;
			switch (requestedEvent) {
				case "SOLVE" :
				case "SOLVECELL" :
				case "SOLVE_PARTIALLY" :
					$messageBlock = window.$solvingMessageBlock;
					break;
				case "CHECK" :
				case "RATE" :
				case "HINT" :
					$messageBlock = window.$analysisMessageBlock;
					break;
				default:
					// Temporary. Shouldn't happen. Catch all other cases
					$messageBlock = window.$solvingMessageBlock;
					break;
			}

			
			// Process the response
			// TODO: $objGrid is null
			var responseType = "EmptyResponse";
			var responseText = "";
			var hintResponse = {};
			
			switch (responseType) {

				case "CompleteResponse" :
					utilShowSuccessMessage($messageBlock, "The puzzle is correctly solved");
					break;
			
				case "SuccessResponse" :
					utilShowSuccessMessage($messageBlock, responseText);
					break;

				case "ErrorResponse" :
					utilShowErrorMessage($messageBlock, responseText);
					break;

				case "SolveResponse" :
					$("#functionContainer").hide();
					window.viewStepsPopup.display();					
					break;

				case "MultipleSolutionsResponse" :
					$("#functionContainer").hide();
					window.multipleSolutionsPopup.display();					

					// Start the download service
					var downloadService = Object.create(DownloadService);
					downloadService.fileName = "duplicateSolutions" ;
					downloadService.content = $("#MULTIPLE_SOLUTIONS").text().replace(/\n/g,"\r\n");
					downloadService.trigger = "EVENT_DOWNLOAD_MULTIPLES";
					downloadService.start();
					
					break;

				case "HintResponse" :		
					$("#functionContainer").hide();				
					$("#VIEWED_HINTS").val(hintResponse.viewedHints); 
					window.hintResponsePopup.hintDescription = hintResponse.hintDescription;
					window.hintResponsePopup.hintDetails = hintResponse.hintDetails;
					window.hintResponsePopup.isRetry = hintResponse.isRetry;					
					window.hintResponsePopup.display();
		        	break;

				case "HintErrorResponse" :		
					$("#functionContainer").hide();
					window.hintErrorResponsePopup.display();
		        	break;
		        	
				default:
		        	break;
				
			} 

			// Start services
			
			// Start the autosave service
			window.autosaveService.desktopMode = true;
			window.autosaveService.referer = "78.53.180.134";
			window.autosaveService.puzzleType = "4";
			window.autosaveService.start();

			// Run the adbService
			window.adbDetectionService.client = "desktop";
			window.adbDetectionService.$detector = $("#detectorId");			
			window.adbDetectionService.referer = "78.53.180.134";			
			window.adbDetectionService.execute();


			// Facebook
			(function(d, s, id) {
				  var js, fjs = d.getElementsByTagName(s)[0];
				  if (d.getElementById(id)) return;
				  js = d.createElement(s); js.id = id;
				  js.src = "//connect.facebook.net/en_US/sdk.js#xfbml=1&version=v2.0";
				  fjs.parentNode.insertBefore(js, fjs);
				}(document, 'script', 'facebook-jssdk'));

			// Google Plus
			(function() {
			   var po = document.createElement('script'); po.type = 'text/javascript'; po.async = true;
			   po.src = 'https://apis.google.com/js/platform.js';
