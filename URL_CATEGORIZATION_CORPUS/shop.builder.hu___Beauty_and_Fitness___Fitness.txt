            function strip_tags(input, allowed) {
                allowed = (((allowed || "") + "").toLowerCase().match(/<[a-z][a-z0-9]*>/g) || []).join('');
                var tags = /<\/?([a-z][a-z0-9]*)\b[^>]*>/gi, commentsAndPhpTags = /<!--[\s\S]*?-->|<\?(?:php)?[\s\S]*?\?>/gi;
                return input.replace(commentsAndPhpTags, '').replace(tags, function ($0, $1) {
                    return allowed.indexOf('<' + $1.toLowerCase() + '>') > -1 ? $0 : '';
                });
            }

            /**
             * delete from list of featured (bookmarked) products

            function deleteFromFeatured(id, bid, type) {
                if (confirm('Valóban törölni szeretnéd a tételt a kedvencek közül?')) {
                    $.post("/_inc_shop/rendel_ajax.php", $.param({
                        "action": "deleteFeatured",
                        "product_id": id,
                        "bid": bid,
                        'type': type
                    }), function (data) {
                        $("div.bookmarkBox").html(data.return_favorite_block);
                        if (type == 'shop_item') {
                            var _id = $("#termek_reszletes #ID").val();
                        } else {
                            var _id = $("#article_id").val();
                        }
                        if (id == _id) {
                            if (type == 'shop_item') {
                                $(".addBookmark img").attr('src', $('.addBookmark img').attr('src').replace('_yellow', '_white'));
                            } else {
                                $(".addBookmark img").attr('src', '/images/star_boxhead.png');
                            }
                            $("a.bookmarkStar").attr('title', 'Kedvencek közé!');
                            $("a.bookmarkStar.orangeButton").html('Kedvencek közé!');
                            $("a.bookmarkStar").unbind('click').bind("click", function () {
                                addBookmark(id, type);
                            });
                            setTimeout(function () {
                                setQTip('a.bookmarkStar, .rightBookmarks img');
                            }, 500);
                        }
                        if ((parseInt($("#rightFeaturedNum").text())) == 0) {
                            $(".bookmarkBox").hide();
                        }
                    }, "json");
                }
            }

            function addBookmark(id, type) {
                $.post("/_inc_shop/rendel_ajax.php", $.param({
                    'action': 'addFeatured',
                    'product_id': id,
                    'type': type
                }), function (data) {
                    if ((parseInt($("#rightFeaturedNum").text())) == 0)
                        $(".bookmarkBox").show();
                    $("div.bookmarkBox").html(data.return_favorite_block);
                    if (type == 'shop_item') {
                        $(".addBookmark img").attr('src', $('.addBookmark img').attr('src').replace('_white', '_yellow'));
                    } else {
                        $(".addBookmark img").attr('src', '/images/star_boxhead_yellow.png');
                    }
                    $(".bookmarkStar").attr('title', 'Ki a kedvencekből');
                    $("a.bookmarkStar.orangeButton").html('Ki a kedvencekből');
                    $(".bookmarkStar").unbind('click').bind("click", function () {
                        deleteFromFeatured(id, data.bid, type);
                    });
                    alert(data.msg);
                    setTimeout(function () {
                        setQTip('a.bookmarkStar, .rightBookmarks img');
                    }, 500);
                }, "json");
                return true;
            }
