function rubrics_del(){
  var frm = document.forms["results"];
  var sel=new Array();
  var ind=0;
  for (var i=0; i<frm.rubrics.length; i++){
    if (frm.rubrics[i].selected){
      ind=sel.length;
      sel[ind]=new Object;
      sel[ind].value=frm.rubrics[i].value;
      sel[ind].index=i;
    }
  }
  for (var i=0; i<sel.length; i++){
    frm.rubrics[sel[i].index-i]=null;
    if (sel[i].index-i<frm.rubrics.length){
      frm.rubrics[sel[i].index-i].selected=true}
  }
}
