   function setActiveChecks(checkIDs) 
{
	$(".add_cart").attr("checked",false);
	$(".remove_cart").attr("checked",false);
	
	if(checkIDs!==null && checkIDs!=='') 
	{
		var checkIDsList = checkIDs.split('_');	
	
		$.each(checkIDsList, function() 
		{ 
			if (this.length > 1) 
			{
				$("#add_cart_"+ this).attr("checked","checked");
				$("#remove_cart_"+ this).attr("checked","checked");
				$("#cart").css("display","block");
				var ticketArtSrc = $("#"+this + " img").attr("src");
				var ticketName = $("#"+this + " h3").html();
				$("#cart_contents").append("<li class='cart_"+this+"'><img src='"+ticketArtSrc+"' /><h3>"+ticketName+"</h3></li>");
				$("#cart_contents .add_cart").attr("checked","checked");
			}
		});
		
	}
}

function checkCart() 
{
	if($.cookie('scratch_compare')!==null) 
	{

	}
}

/**** REMOVE SELECTION ****/
function removeSelection(myID) {
	if($.cookie('scratch_compare')!==null) 
	{
		var scratchItems = $.cookie('scratch_compare');
		$.cookie('scratch_compare', null, { path: '/' });
		var parseCartList = scratchItems.split("_");
		var newScratchItems = "";
		
		
		$.each(parseCartList, function(e) { 
			var myCartItem = parseCartList[e];
			if(myCartItem!==myID && myCartItem!=="" && myCartItem!==null) {
				if (parseCartList.length > 1)
					newScratchItems += (myCartItem + "_");
				else
					newScratchItems += (myCartItem);
			}
		});
		var newCartItems = newScratchItems;
		$.cookie('scratch_compare', newCartItems, { path: '/' });
		
		$("#add_cart_"+myID).removeAttr("checked");
		$("#cart_contents li.cart_"+myID).remove();
		
		$(".compare #"+myID).css("display","none");
		
		if($.cookie('scratch_compare')=="")
			$("#cart").css("display","none");
	}
}

	if($.cookie('scratch_compare')!==null) {
		$.cookie('scratch_compare', '', { path: '/' });

		$("#cart").css("display","none");
	}
}
