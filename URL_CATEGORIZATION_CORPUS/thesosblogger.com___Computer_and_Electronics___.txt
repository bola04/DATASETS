// Node object

function Node(id, pid, name, url, title, target, icon, iconOpen, open) {

	this.id = id;

	this.pid = pid;

	this.name = name;

	this.url = url;

	this.title = title;

	this.target = target;

	this.icon = icon;

	this.iconOpen = iconOpen;

	this._io = open || false;

	this._is = false;

	this._ls = false;

	this._hc = false;

	this._ai = 0;

	this._p;

};



// Tree object

function dTree(objName) {

	this.config = {

		target					: null,

		folderLinks			: true,

		useSelection		: true,

		useCookies			: true,

		useLines				: true,

		useIcons				: true,

		useStatusText		: false,

		closeSameLevel	: false,

		inOrder					: false

	}

	this.icon = {

		root				: 'http://3.bp.blogspot.com/-JY4rbWbvS_M/TbpRnykTHAI/AAAAAAAAAEA/KDuXD4WQHSo/s1600/base.gif',

		folder			: 'http://4.bp.blogspot.com/-XVXFkhLyY8U/TbpRonzDlfI/AAAAAAAAAEM/cvaWJCBdhGQ/s1600/folder.gif',

		folderOpen	: 'http://3.bp.blogspot.com/-ErheraIz2PE/TbpRpHZQfbI/AAAAAAAAAEQ/kM4jOvvNtgA/s1600/folderopen.gif',

		node				: 'http://2.bp.blogspot.com/-C9xCPRX01eo/TbpRsOmz1VI/AAAAAAAAAE8/6wwXEzTzrIw/s1600/page.gif',

		empty				: 'http://3.bp.blogspot.com/-3df66JY8wtw/TbpRobGPSnI/AAAAAAAAAEI/XuY3oudYOL4/s1600/empty.gif',

		line				: 'http://4.bp.blogspot.com/-UjZcQluCEHM/TbpRqdV8gBI/AAAAAAAAAEk/ziI6wRnr75Q/s1600/line.gif',

		join				: 'http://3.bp.blogspot.com/-TZfbM8l1YIg/TbpRp4R5MiI/AAAAAAAAAEc/LHvLRArRCbs/s1600/join.gif',

		joinBottom	: 'http://3.bp.blogspot.com/-npAizZylMZ8/TbpRqJxULuI/AAAAAAAAAEg/AZoxbhrEDAE/s1600/joinbottom.gif',

		plus				: 'http://2.bp.blogspot.com/-zk6IhkWihHc/TbpRsfRfOlI/AAAAAAAAAFA/MVUdXJQizVA/s1600/plus.gif',

		plusBottom	: 'http://2.bp.blogspot.com/-ICgnar4Z4PU/TbpRs6NW7fI/AAAAAAAAAFE/KhxTM5vZqFU/s1600/plusbottom.gif',

		minus				: 'http://2.bp.blogspot.com/-9-EtZ_BMrH0/TbpRq3e956I/AAAAAAAAAEo/zGX-6iS8DAQ/s1600/minus.gif',

		minusBottom	: 'http://2.bp.blogspot.com/-H_Hkv6pPIDE/TbpRrLLJqcI/AAAAAAAAAEs/7Azb3kqFho0/s1600/minusbottom.gif',

		nlPlus			: 'http://1.bp.blogspot.com/-EHNXkheGhsU/TbpRr-Zl3uI/AAAAAAAAAE4/zNAsa2F3Hd8/s1600/nolines_plus.gif',

		nlMinus			: 'http://2.bp.blogspot.com/-T7SzrwAL-gg/TbpRrhrxkSI/AAAAAAAAAE0/JxnnClFQYY4/s1600/nolines_minus.gif'

	};

	this.obj = objName;

	this.aNodes = [];

	this.aIndent = [];

	this.root = new Node(-1);

	this.selectedNode = null;

	this.selectedFound = false;

	this.completed = false;

};



// Adds a new node to the node array

dTree.prototype.add = function(id, pid, name, url, title, target, icon, iconOpen, open) {

	this.aNodes[this.aNodes.length] = new Node(id, pid, name, url, title, target, icon, iconOpen, open);

};



// Open/close all nodes

dTree.prototype.openAll = function() {

	this.oAll(true);

};

dTree.prototype.closeAll = function() {

	this.oAll(false);

};



// Outputs the tree to the page

dTree.prototype.toString = function() {

	var str = '<div class="dtree">\n';

	if (document.getElementById) {

		if (this.config.useCookies) this.selectedNode = this.getSelected();

		str += this.addNode(this.root);

	} else str += 'Browser not supported.';

	str += '</div>';

	if (!this.selectedFound) this.selectedNode = null;

	this.completed = true;

	return str;

};



// Creates the tree structure

dTree.prototype.addNode = function(pNode) {

	var str = '';

	var n=0;

	if (this.config.inOrder) n = pNode._ai;

	for (n; n<this.aNodes.length; n++) {

		if (this.aNodes[n].pid == pNode.id) {

			var cn = this.aNodes[n];

			cn._p = pNode;

			cn._ai = n;

			this.setCS(cn);

			if (!cn.target && this.config.target) cn.target = this.config.target;

			if (cn._hc && !cn._io && this.config.useCookies) cn._io = this.isOpen(cn.id);

			if (!this.config.folderLinks && cn._hc) cn.url = null;

			if (this.config.useSelection && cn.id == this.selectedNode && !this.selectedFound) {

					cn._is = true;

					this.selectedNode = n;

					this.selectedFound = true;

			}

			str += this.node(cn, n);

			if (cn._ls) break;

		}

	}

	return str;

};



// Creates the node icon, url and text

dTree.prototype.node = function(node, nodeId) {

	var str = '<div class="dTreeNode">' + this.indent(node, nodeId);

	if (this.config.useIcons) {

		if (!node.icon) node.icon = (this.root.id == node.pid) ? this.icon.root : ((node._hc) ? this.icon.folder : this.icon.node);

		if (!node.iconOpen) node.iconOpen = (node._hc) ? this.icon.folderOpen : this.icon.node;

		if (this.root.id == node.pid) {

			node.icon = this.icon.root;

			node.iconOpen = this.icon.root;

		}

		str += '<img id="i' + this.obj + nodeId + '" src="' + ((node._io) ? node.iconOpen : node.icon) + '" alt="" />';

	}

	if (node.url) {

		str += '<a id="s' + this.obj + nodeId + '" class="' + ((this.config.useSelection) ? ((node._is ? 'nodeSel' : 'node')) : 'node') + '" href="' + node.url + '"';

		if (node.title) str += ' title="' + node.title + '"';

		if (node.target) str += ' target="' + node.target + '"';

		if (this.config.useStatusText) str += ' onmouseover="window.status=\'' + node.name + '\';return true;" onmouseout="window.status=\'\';return true;" ';

		if (this.config.useSelection && ((node._hc && this.config.folderLinks) || !node._hc))

			str += ' onclick="javascript: ' + this.obj + '.s(' + nodeId + ');"';

		str += '>';

	}

	else if ((!this.config.folderLinks || !node.url) && node._hc && node.pid != this.root.id)

		str += '<a href="javascript: ' + this.obj + '.o(' + nodeId + ');" class="node">';

	str += node.name;

	if (node.url || ((!this.config.folderLinks || !node.url) && node._hc)) str += '</a>';

	str += '</div>';

	if (node._hc) {

		str += '<div id="d' + this.obj + nodeId + '" class="clip" style="display:' + ((this.root.id == node.pid || node._io) ? 'block' : 'none') + ';">';

		str += this.addNode(node);

		str += '</div>';

	}

	this.aIndent.pop();

	return str;

};



// Adds the empty and line icons

dTree.prototype.indent = function(node, nodeId) {

	var str = '';

	if (this.root.id != node.pid) {

		for (var n=0; n<this.aIndent.length; n++)

			str += '<img src="' + ( (this.aIndent[n] == 1 && this.config.useLines) ? this.icon.line : this.icon.empty ) + '" alt="" />';

		(node._ls) ? this.aIndent.push(0) : this.aIndent.push(1);

		if (node._hc) {

			str += '<a href="javascript: ' + this.obj + '.o(' + nodeId + ');"><img id="j' + this.obj + nodeId + '" src="';

			if (!this.config.useLines) str += (node._io) ? this.icon.nlMinus : this.icon.nlPlus;

			else str += ( (node._io) ? ((node._ls && this.config.useLines) ? this.icon.minusBottom : this.icon.minus) : ((node._ls && this.config.useLines) ? this.icon.plusBottom : this.icon.plus ) );

			str += '" alt="" /></a>';

		} else str += '<img src="' + ( (this.config.useLines) ? ((node._ls) ? this.icon.joinBottom : this.icon.join ) : this.icon.empty) + '" alt="" />';

	}

	return str;

};



// Checks if a node has any children and if it is the last sibling

dTree.prototype.setCS = function(node) {

	var lastId;

	for (var n=0; n<this.aNodes.length; n++) {

		if (this.aNodes[n].pid == node.id) node._hc = true;

		if (this.aNodes[n].pid == node.pid) lastId = this.aNodes[n].id;

	}

	if (lastId==node.id) node._ls = true;

};



// Returns the selected node

dTree.prototype.getSelected = function() {

	var sn = this.getCookie('cs' + this.obj);

	return (sn) ? sn : null;

};



// Highlights the selected node

dTree.prototype.s = function(id) {

	if (!this.config.useSelection) return;

	var cn = this.aNodes[id];

	if (cn._hc && !this.config.folderLinks) return;

	if (this.selectedNode != id) {

		if (this.selectedNode || this.selectedNode==0) {

			eOld = document.getElementById("s" + this.obj + this.selectedNode);

			eOld.className = "node";

		}

		eNew = document.getElementById("s" + this.obj + id);

		eNew.className = "nodeSel";

		this.selectedNode = id;

		if (this.config.useCookies) this.setCookie('cs' + this.obj, cn.id);

	}

};



// Toggle Open or close

dTree.prototype.o = function(id) {

	var cn = this.aNodes[id];

	this.nodeStatus(!cn._io, id, cn._ls);

	cn._io = !cn._io;

	if (this.config.closeSameLevel) this.closeLevel(cn);

	if (this.config.useCookies) this.updateCookie();

};



// Open or close all nodes

dTree.prototype.oAll = function(status) {

	for (var n=0; n<this.aNodes.length; n++) {

		if (this.aNodes[n]._hc && this.aNodes[n].pid != this.root.id) {

			this.nodeStatus(status, n, this.aNodes[n]._ls)

			this.aNodes[n]._io = status;

		}

	}

	if (this.config.useCookies) this.updateCookie();

};



// Opens the tree to a specific node

dTree.prototype.openTo = function(nId, bSelect, bFirst) {

	if (!bFirst) {

		for (var n=0; n<this.aNodes.length; n++) {

			if (this.aNodes[n].id == nId) {

				nId=n;

				break;

			}

		}

	}

	var cn=this.aNodes[nId];

	if (cn.pid==this.root.id || !cn._p) return;

	cn._io = true;

	cn._is = bSelect;

	if (this.completed && cn._hc) this.nodeStatus(true, cn._ai, cn._ls);

	if (this.completed && bSelect) this.s(cn._ai);

	else if (bSelect) this._sn=cn._ai;

	this.openTo(cn._p._ai, false, true);

};



// Closes all nodes on the same level as certain node

dTree.prototype.closeLevel = function(node) {

	for (var n=0; n<this.aNodes.length; n++) {

		if (this.aNodes[n].pid == node.pid && this.aNodes[n].id != node.id && this.aNodes[n]._hc) {

			this.nodeStatus(false, n, this.aNodes[n]._ls);

			this.aNodes[n]._io = false;

			this.closeAllChildren(this.aNodes[n]);

		}

	}

}



// Closes all children of a node

dTree.prototype.closeAllChildren = function(node) {

	for (var n=0; n<this.aNodes.length; n++) {

		if (this.aNodes[n].pid == node.id && this.aNodes[n]._hc) {

			if (this.aNodes[n]._io) this.nodeStatus(false, n, this.aNodes[n]._ls);

			this.aNodes[n]._io = false;

			this.closeAllChildren(this.aNodes[n]);		

		}

	}

}



// Change the status of a node(open or closed)

dTree.prototype.nodeStatus = function(status, id, bottom) {

	eDiv	= document.getElementById('d' + this.obj + id);

	eJoin	= document.getElementById('j' + this.obj + id);

	if (this.config.useIcons) {

		eIcon	= document.getElementById('i' + this.obj + id);

		eIcon.src = (status) ? this.aNodes[id].iconOpen : this.aNodes[id].icon;

	}

	eJoin.src = (this.config.useLines)?

	((status)?((bottom)?this.icon.minusBottom:this.icon.minus):((bottom)?this.icon.plusBottom:this.icon.plus)):

	((status)?this.icon.nlMinus:this.icon.nlPlus);

	eDiv.style.display = (status) ? 'block': 'none';

};





// [Cookie] Clears a cookie

dTree.prototype.clearCookie = function() {

	var now = new Date();

	var yesterday = new Date(now.getTime() - 1000 * 60 * 60 * 24);

	this.setCookie('co'+this.obj, 'cookieValue', yesterday);

	this.setCookie('cs'+this.obj, 'cookieValue', yesterday);

};



// [Cookie] Sets value in a cookie

dTree.prototype.setCookie = function(cookieName, cookieValue, expires, path, domain, secure) {

	document.cookie =

		escape(cookieName) + '=' + escape(cookieValue)

		+ (expires ? '; expires=' + expires.toGMTString() : '')

		+ (path ? '; path=' + path : '')

		+ (domain ? '; domain=' + domain : '')

		+ (secure ? '; secure' : '');

};



// [Cookie] Gets a value from a cookie

dTree.prototype.getCookie = function(cookieName) {

	var cookieValue = '';

	var posName = document.cookie.indexOf(escape(cookieName) + '=');

	if (posName != -1) {

		var posValue = posName + (escape(cookieName) + '=').length;

		var endPos = document.cookie.indexOf(';', posValue);

		if (endPos != -1) cookieValue = unescape(document.cookie.substring(posValue, endPos));

		else cookieValue = unescape(document.cookie.substring(posValue));

	}

	return (cookieValue);

};



// [Cookie] Returns ids of open nodes as a string

dTree.prototype.updateCookie = function() {

	var str = '';

	for (var n=0; n<this.aNodes.length; n++) {

		if (this.aNodes[n]._io && this.aNodes[n].pid != this.root.id) {

			if (str) str += '.';

			str += this.aNodes[n].id;

		}

	}

	this.setCookie('co' + this.obj, str);

};



// [Cookie] Checks if a node id is in a cookie

dTree.prototype.isOpen = function(id) {

	var aOpen = this.getCookie('co' + this.obj).split('.');

	for (var n=0; n<aOpen.length; n++)

		if (aOpen[n] == id) return true;

	return false;

};



// If Push and pop is not implemented by the browser

if (!Array.prototype.push) {

	Array.prototype.push = function array_push() {

		for(var i=0;i<arguments.length;i++)

			this[this.length]=arguments[i];

		return this.length;

	}

};

if (!Array.prototype.pop) {

	Array.prototype.pop = function array_pop() {

		lastElement = this[this.length-1];

		this.length = Math.max(this.length-1,0);

		return lastElement;

	}

};

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2018/05/liz-and-the-blue-bird-review-excellent-film.html">
                      《莉茲與青鳥》：京都動畫的頂級傑作！
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-8764352336091153058">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-RAedC4po2h8/Wvfi5mNRMkI/AAAAAAAA0BQ/Dq5L2L3cMzUbm---ZC5n1EZ27Ym-aOIvwCLcBGAs/s1600/liz%2Band%2Bblue%2Bbird%2Btitle.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="625" data-original-width="440" height="400" src="https://1.bp.blogspot.com/-RAedC4po2h8/Wvfi5mNRMkI/AAAAAAAA0BQ/Dq5L2L3cMzUbm---ZC5n1EZ27Ym-aOIvwCLcBGAs/s400/liz%2Band%2Bblue%2Bbird%2Btitle.jpg" width="281"/>
                      </a>
                     </div>
                     <div style="text-align: center;">
                      <b>
                       <span style="font-size: x-large;">
                        京都動畫的頂級作品！
                       </span>
                      </b>
                     </div>
                     <br/>
                     《莉茲與青鳥》（リズと青い鳥）早在公佈時，已經受多方期待，畢竟這是監督山田尚子和編劇吉田玲子合作的作品，她們的往績好，作品又是大受好評的《吹響吧！上低音號》的續篇，怎麼看都是夢幻組合，而最後的結果也沒有令大家失望，這絕對是今年最佳動畫之一！
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       雙簧管和長笛的抒情故事
                      </span>
                     </b>
                     <br/>
                     <br/>
                     《莉茲與青鳥》的背景基於《吹響吧！》動畫2期的兩位主要角色：鎧塚霙、傘木希美的故事。大家熟悉的麗奈、久美子等在這部只是偶爾出現（不過關係好得不得了，根本就是夫婦），動畫的風格也有很大的轉變。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-tOjmXGvlTkA/Wvf4cWPYQ0I/AAAAAAAA0Bg/rEkmulKfwE0Kc7UQcNlCIbViZ3-KbZc9ACLcBGAs/s1600/pic%2B2018-05-13%2B2.54.36%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="784" data-original-width="1600" height="195" src="https://2.bp.blogspot.com/-tOjmXGvlTkA/Wvf4cWPYQ0I/AAAAAAAA0Bg/rEkmulKfwE0Kc7UQcNlCIbViZ3-KbZc9ACLcBGAs/s400/pic%2B2018-05-13%2B2.54.36%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        熟悉的角色，但不是本作的重心！
                       </span>
                      </b>
                     </div>
                     <br class="Apple-interchange-newline"/>
                     《吹響吧！》是青春群像劇，從一班主要角色的角度看北宇治高校吹奏部進軍全國的過程。《莉茲與青鳥》則聚焦兩位主角，以樂曲來比喻兩人的關係轉變，比賽和吹奏部的變化只是背景。
                     <br/>
                     <br/>
                     兩種風格各有長處，喜歡熱血奮鬥的課外活動的觀眾，很可能會對這轉變感到不習慣，但如果你能夠走進山田監督表現的兩人世界，你將會感受到感動的90分鐘！
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       聲音的世界
                      </span>
                     </b>
                     <br/>
                     <br/>
                     觀看《莉茲與青鳥》的時候，首先會注意到的不是畫面，而是聲音。有留意制作名單的人，大概會發現負責音樂的
                     <a href="https://animeanime.jp/article/2016/09/16/30521.html" target="_blank">
                      牛尾憲輔就是之前擔任《聲之形》音樂制作的人
                     </a>
                     。
                     <br/>
                     <br/>
                     牛尾的風格在作品中表露無遺，開場時霙、希美在學校碰面，然後走到音樂室的時候，一段背景音樂「wind, glass, bluebird」配上腳步聲、學校的背景音、流水聲、開門聲等，構成一個豐富的聲音世界，記得在看《聲之形》的時候，留意到這種情況，以為是為了突出「聲音」這個主題在《聲之形》的重要性，想不到同樣的風格用在《莉茲與青鳥》也十分合適。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://4.bp.blogspot.com/-9pZKWmRkngM/Wvf4zhZDKhI/AAAAAAAA0Bo/2JuSwSitzpQH8lvyu46YmWqEmuARwPUcQCLcBGAs/s1600/pic%2B2018-05-13%2B2.53.01%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="785" data-original-width="1600" height="195" src="https://4.bp.blogspot.com/-9pZKWmRkngM/Wvf4zhZDKhI/AAAAAAAA0Bo/2JuSwSitzpQH8lvyu46YmWqEmuARwPUcQCLcBGAs/s400/pic%2B2018-05-13%2B2.53.01%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        劇中的背景聲音反映了角色的心情
                       </span>
                      </b>
                     </div>
                     <br/>
                     根據
                     <a href="https://mainichi.jp/articles/20180511/dyo/00m/200/041000c" target="_blank">
                      《每日新聞》的報導
                     </a>
                     ，在制作《莉茲與青鳥》時，為了做到
                     <a href="https://www.youtube.com/watch?v=f_yrCAZAiUA" target="_blank">
                      「物件聲構成的音樂劇」
                     </a>
                     ，制作組特意到動畫中出現的學校，移動椅子、擦窗、關儲物櫃，錄下這些真實的背景聲音用在動畫中。角色的心情、行動也會反映在背景聲音之上。
                     <br/>
                     <br/>
                     同時，京都動畫也對霙使用的樂器雙簧管有很詳細的描述，例如簧片需以水浸濕，樂手在試音時通常會吹A音（這也是雙簧管在樂團中的用途之一，為樂隊提供清脆準確的A），以至霙需要自製簧片（雙簧管高手必學技能）以確保聲音適合自己。
                     <br/>
                     <br/>
                     在以吹奏部為背景的動畫，提到聲音當然不可能不提到劇中的演奏，也就是《莉茲與青鳥》這個劇名的由來。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       誰是莉茲？誰是青鳥？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     《莉茲與青鳥》是動畫的名稱，但同時也是動畫中出現的劇中作的名稱。劇中的「莉茲與青鳥」是童話故事，講述少女莉茲遇上了化成人型的青鳥，她們過了一段快活的生活。可是莉茲必須要作出痛苦的決斷，青鳥應該要在天空飛翔，於是她放走了青鳥，她們走上了不同的道路。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-h9rwlWJq9Hk/Wvf53uwDO1I/AAAAAAAA0B0/oMAutbGlqP4wOuIglr4EgN2deK4YcTE6QCLcBGAs/s1600/pic%2B2018-05-13%2B4.32.20%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="688" data-original-width="1274" height="215" src="https://2.bp.blogspot.com/-h9rwlWJq9Hk/Wvf53uwDO1I/AAAAAAAA0B0/oMAutbGlqP4wOuIglr4EgN2deK4YcTE6QCLcBGAs/s400/pic%2B2018-05-13%2B4.32.20%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        作品的劇中劇，也是整部作品的中心思想
                       </span>
                      </b>
                     </div>
                     <br/>
                     動畫花了不少時間講述這個童話故事，因為北宇治吹奏部將要演奏的歌曲，就是以「莉茲與青鳥」為創作意念。這種敘述方式，也使觀眾明白到「莉茲與青鳥」就是表現了希美與霙在劇中的關係。
                     <br/>
                     <br/>
                     總是在人際關係中心的希美以及與他人保持距離的霙，兩人的對比非常鮮明，但同時她們也是最親密的朋友。霙在劇中多次說明自己「無法理解放走青鳥的想法」，因為她只想想緊緊抓住希美，同樣地，希美也想留在霙的身邊，甚至想與她一起進音樂大學。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://3.bp.blogspot.com/-nLDcS8VVtGs/Wvf6IyEX1fI/AAAAAAAA0CA/47Gtt15SncQuRZb23_rnSAQ3gHrMMB8rwCLcBGAs/s1600/pic%2B2018-05-13%2B2.51.50%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="782" data-original-width="1600" height="195" src="https://3.bp.blogspot.com/-nLDcS8VVtGs/Wvf6IyEX1fI/AAAAAAAA0CA/47Gtt15SncQuRZb23_rnSAQ3gHrMMB8rwCLcBGAs/s400/pic%2B2018-05-13%2B2.51.50%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-SARpdft9x1s/Wvf6IkckieI/AAAAAAAA0B8/p_dWacfSwIMjdwmUqRES7xUu52Z3QuTIQCLcBGAs/s1600/pic%2B2018-05-13%2B2.52.03%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="768" data-original-width="1600" height="191" src="https://2.bp.blogspot.com/-SARpdft9x1s/Wvf6IkckieI/AAAAAAAA0B8/p_dWacfSwIMjdwmUqRES7xUu52Z3QuTIQCLcBGAs/s400/pic%2B2018-05-13%2B2.52.03%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        兩位主角想一直在一起，但這是最好的選擇嗎？
                       </span>
                      </b>
                     </div>
                     <br/>
                     可是現實並不如想像中美好，希美對長笛的投入並不是很深，反而霙在雙簧管的才能卻得到老師的賞識，兩人需要各自走向不同的道路。於是我們得到了「莉茲與青鳥」的第3樂章 — 「愛的決斷」。
                     <br/>
                     <br/>
                     為了贏得比賽，希美和霙的雙人演奏部份必須要完美進行，可是如果希美不能放手，霙不能飛翔，演奏又如何成功？為了對方著想，兩人都必須學會放下執念，而這就是《莉茲與青鳥》的中心思想。為了對方好，有時需要作出痛苦的決擇。
                     <br/>
                     <br/>
                     劇中最高潮的一幕，就是演奏的部份，那部份的情感表現，甚至令在場觀眾也認不住落淚！
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       無論如何都應該一看的作品
                      </span>
                     </b>
                     <br/>
                     <br/>
                     《莉茲與青鳥》在日本上映後反響大很，在
                     <a href="https://www.japantimes.co.jp/culture/2018/04/18/films/film-reviews/liz-blue-bird-brilliantly-executed-return-anime-favorite/" target="_blank">
                      主流媒體的影評
                     </a>
                     上也得到了好評價。雖然現階段不知道《莉茲與青鳥》會否在日本以外放映，但如果有機會，請務必看一次！因為這可以說是京都動畫的頂級傑作！更是《吹響吧！上低音號》系列的最佳作品！
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-EBXBFStUk0A/Wvf6ckPbrkI/AAAAAAAA0CM/RAhdnT2QNp0ecSerPsZv1csi8Y6DryMZgCLcBGAs/s1600/pic%2B2018-05-13%2B4.33.08%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="688" data-original-width="1270" height="216" src="https://1.bp.blogspot.com/-EBXBFStUk0A/Wvf6ckPbrkI/AAAAAAAA0CM/RAhdnT2QNp0ecSerPsZv1csi8Y6DryMZgCLcBGAs/s400/pic%2B2018-05-13%2B4.33.08%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        因為愛，所以必須學會放手。
                       </span>
                      </b>
                     </div>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-eyor28bGUhQ/Wvf_oGgAakI/AAAAAAAA0Cg/7Q8WRKekVAkmf9QaZuUfZ_nhs2ryO3JkwCLcBGAs/s1600/thumb_IMG_9229_1024.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="1024" data-original-width="768" height="400" src="https://1.bp.blogspot.com/-eyor28bGUhQ/Wvf_oGgAakI/AAAAAAAA0Cg/7Q8WRKekVAkmf9QaZuUfZ_nhs2ryO3JkwCLcBGAs/s400/thumb_IMG_9229_1024.jpg" width="300"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         《莉茲與青鳥》在日本上映後受到各方好評！

                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: x-large;">
                        喜歡科幻的你不容錯過！
                       </span>
                      </b>
                     </div>
                     <div style="text-align: center;">
                      本文圖片來源：
                      <a href="http://www.project-aico.com/" target="_blank">
                       Bones
                      </a>
                     </div>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       日本動畫最有影響的題材是什麼？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     如果看近年的作品（特別是看銷量），最受歡迎的動畫題材大概是轉生、萌、偶像系等等。不過，曾經有那麼一個時期，日本科幻動畫是最受歡迎的題材，而且影響力遍及全球。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       為什麼動畫公司不願意製作科幻作品？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     1988年的動畫電影《Akira》是其中一個最成功的例子，其大膽和創意
                     <a href="https://www.theguardian.com/film/2011/jun/23/akira-film-review" target="_blank">
                      影響了很多後來的作品
                     </a>
                     。1995年的《攻殻機動隊》差不多是喜歡科幻的人必看的經典，作品預測的真假難分世界，也與我們現在的生活愈來愈相似。一些較少人知道的作品，如1998年的《玲音》也一直有一批忠實觀眾。
                     <br/>
                     <br class="Apple-interchange-newline"/>
                     可惜的是，科幻作品在商業上是高風險的，無論是科幻電影還是動畫，作品題材要求的世界觀建立、精細的設定、符合一定程度的科學考究，製作成本高，都令製作科幻題材變得困難，在日本經濟不好的年代，加上依賴實體（VHS/LD/DVD/BD）的分銷模式之下，動畫公司愈來愈不願意投入科幻（以及很多較認真的題材），轉為製作銷量更有保證的作品。
                     <br/>
                     <br/>
                     <br/>
                     <blockquote class="tr_bq">
                      <b>
                       <span style="font-size: large;">
                        "如果受委託的動畫公司能夠從短期銷量的壓力中解放，專心製作受好評的動畫，這模式可能從根本改變日本動畫業。"
                       </span>
                      </b>
                     </blockquote>
                     <br/>
                     <br/>
                     所以，當看到 Netflix 與動畫公司 Bones合作推出獨家的科幻動畫時， 實在很令人高興。不少觀眾喜歡看科幻作品，卻未必想購買實體光碟，這種情況下，串流動畫（streaming）是這類題材的出路。
                     <br/>
                     <br/>
                     Netflix 雖然有月費，但每套電影和動畫的邊際成本（marginal cost）是零，有利觀眾發掘新作品。相對之下，受委託的動畫公司能夠從短期銷量的壓力中解放，專心製作受好評的動畫，這模式可能從根本改變日本動畫業。
                     <br/>
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       《A.I.C.O. -Incarnation-》：
                      </span>
                     </b>
                     <b>
                      <span style="font-size: x-large;">
                       Bones的科幻究竟是怎樣的？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     回到這次要談的作品，《A.I.C.O. -Incarnation-》（下稱《A.I.C.O.》）是去年在
                     <a href="https://www.gizmodo.jp/2017/08/netflix-anime-slate-2017.html" target="_blank">
                      Netflix Anime Slate 2017
                     </a>
                     公佈的重點作品，並且在2018年3月公開，根據當日活動公佈的資料，半數日本的Netflix觀眾都在看動畫。加上成長中的歐美市場，投資在日式動畫實在是自然不過的決定。
                     <br/>
                     <br/>
                     <br/>
                     <blockquote class="tr_bq">
                      <b>
                       <span style="font-size: large;">
                        "隔離的城市、人工生命體佔據的地區、交錯的陰謀、被埋沒的真相，單是看設定已令人感到興奮。"
                       </span>
                      </b>
                     </blockquote>
                     <br/>
                     <br/>
                     《A.I.C.O.》發生在近未來的日本，人工生命體的暴走造成了毀滅性的「大爆發」，主角是在災難後存活下來的少女橘愛子（Aiko），本來是普通人的她發現了自己的身體出現異常的變化，為了尋找事件的真相，她必須要愈過重重障礙，到達造成事件發生的原初點。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-fATVIOLHGWM/Wq49GeuBiII/AAAAAAAAz_8/prqR_K7--ZYMfUhLmT8k-NEWCp8Q5hCSwCLcBGAs/s1600/aico%2Bchapter08%2B01.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="450" data-original-width="800" height="225" src="https://2.bp.blogspot.com/-fATVIOLHGWM/Wq49GeuBiII/AAAAAAAAz_8/prqR_K7--ZYMfUhLmT8k-NEWCp8Q5hCSwCLcBGAs/s400/aico%2Bchapter08%2B01.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-QPKOfanp_X4/Wq49TeagnjI/AAAAAAAA0AA/E97pwD_sdPcKxmEjoaFpKrvwdPBfq-iCwCLcBGAs/s1600/aico%2Bchapter06%2B01.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="450" data-original-width="800" height="225" src="https://1.bp.blogspot.com/-QPKOfanp_X4/Wq49TeagnjI/AAAAAAAA0AA/E97pwD_sdPcKxmEjoaFpKrvwdPBfq-iCwCLcBGAs/s400/aico%2Bchapter06%2B01.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         為了尋找事件的真相，
                        </span>
                       </b>
                      </span>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         主角必須到達造成事件發生的原初點。
                        </span>
                       </b>
                      </span>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <br/>
                     </div>
                     《A.I.C.O.》嚴格來說不能算是終末後故事（post apocalyptic），因其發生地點圍繞富山縣的黑部水壩及其周邊，並未影響全世界，但故事的氣氛卻有終末的絕望感覺：隔離的城市、龍蛇混雜的邊境區、人工生命體佔據的地區、為了解決災難而成立的官僚機構和衍生的特殊職業（打撈者）、交錯的陰謀、最後是被埋沒的真相，單是看設定已令人感到興奮。
                     <br/>
                     <br/>
                     作品在細節的描寫非常周到，從生活的細節、戰鬥的服裝和武器、以至入面採用的生物科技，《A.I.C.O.》世界看起來真實，令觀眾能夠代入一眾主要角色。
                     <br/>
                     <br/>
                     <span style="color: red;">
                      （文章的下半部有《A.I.C.O.》的劇透，請注意。）
                     </span>

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2017/12/kancolle-experiment-new-player-challenge.html">
                      大破解！究竟《艦隊收藏》的活動有多難？田中設計的平衡有問題嗎？
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-830489360353674250">
                     <div style="text-align: center;">
                      <div class="separator" style="clear: both; text-align: center;">
                       <a href="https://1.bp.blogspot.com/-PbhY_ttyPGw/WjYyM9lWx3I/AAAAAAAAz7o/h8YGve3_7FEM9gmdHs78z_SeYaxtB_kMwCLcBGAs/s1600/kancolle-2017-autumn-04.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                        <img border="0" data-original-height="479" data-original-width="800" height="238" src="https://1.bp.blogspot.com/-PbhY_ttyPGw/WjYyM9lWx3I/AAAAAAAAz7o/h8YGve3_7FEM9gmdHs78z_SeYaxtB_kMwCLcBGAs/s400/kancolle-2017-autumn-04.jpg" width="400"/>
                       </a>
                      </div>
                     </div>
                     <div style="text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        完全沒有準備活動用艦，能夠靠提督的實力
                       </span>
                      </b>
                     </div>
                     <div style="text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        達成不可能的任務嗎？
                       </span>
                      </b>
                     </div>
                     <br/>
                     《艦隊收藏》最近面對很大挑戰，我們早前
                     <a href="http://www.thesosblogger.com/2017/10/why-is-azurlane-popular-in-japan.html">
                      為大家分析了新遊戲如何搶走《艦隊收藏》的玩家
                     </a>
                     。事實上，SOS BLOG 的提督朋友們最近轉玩其他遊戲的人也不少。這不只是在說軍艦擬人遊戲，而是包括《千年戰爭》、《碧藍幻想》等手機遊戲。
                     <br/>
                     <br/>
                     「田中謙介是罪魁禍首。」
                     <br/>
                     「都是田中的錯。」
                     <br/>
                     「遊戲的難度很有問題。」
                     <br/>
                     <br/>
                     這是不少人的意見，也基本上與網路的討論流向相似。每到艦娘活動的季節，就會定期看到表達對《艦隊收藏》製作人田中謙介不滿的訊息，從要求製作人退職，到更加難聽的說話也會看到。
                     <br/>
                     <br/>
                     SOS BLOG 的甲蟲提督也感到很大的危機，如果這樣下去，新手都嚇跑了，老提督也退休了，還有人會玩《艦隊收藏》嗎？
                     <br/>
                     <br/>
                     所以他決定這次活動要作出新挑戰，在使用自己的帳號輕鬆突破甲難度後，他進行了一個實驗：用完全沒有準備這次活動需要的裝備和艦隻的提督帳號，挑戰2017年秋季活動。目標是把涼月帶回家，證明《艦隊收藏》的遊戲難度並沒有大家想像的高！
                     <br/>
                     <br/>
                     <div style="text-align: center;">
                      <b>
                       <span style="font-size: x-large;">
                        目標：證明遊戲的難度並沒有想像的高！
                       </span>
                      </b>
                     </div>

                     <a href="http://www.thesosblogger.com/2017/11/game-as-a-service-competition-and-longetivity.html">
                      課金遊戲的危機？談隨機性帶來的規管風險及遊戲服務化造成的競爭兩難
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-1565208640472767713">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-bF1b0jIMWE4/Whljii14UlI/AAAAAAAAz5M/JdUdLnDPRaAnDulONW8UPsfIPG5JLTr0ACLcBGAs/s1600/IMG_6750.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="750" data-original-width="1334" height="223" src="https://1.bp.blogspot.com/-bF1b0jIMWE4/Whljii14UlI/AAAAAAAAz5M/JdUdLnDPRaAnDulONW8UPsfIPG5JLTr0ACLcBGAs/s400/IMG_6750.jpg" width="400"/>
                      </a>
                     </div>
                     <br/>
                     <b>
                      遊戲業界已經永遠改變了。
                     </b>
                     <br/>
                     <br/>
                     隨著高速網絡的普及化，以及近年滲透至生活每一個部份的智能手機，遊戲作為娛樂產品與約10年大家認識的遊戲業已經變得非常不同。一隻遊戲，曾經對很多人來說，就是指包裝在盒中的光碟或盒帶，遊戲的長度不一，但遊戲的長度一般有限，玩家以往為自己完成了多少隻遊戲而自傲。
                     <br/>
                     <br/>
                     新類型的遊戲完成不同。
                     <br/>
                     <br/>
                     新遊戲講求
                     <b>
                      永遠玩不完
                     </b>
                     ，它們需要網路連線，把資料儲存在營運的服務器，活動存在時間性，需要玩家長期投入。相對地，營運會不斷更新內容吸引玩家繼續留在遊戲。玩家不會再自誇已完成遊戲，而是說自己「完成這次活動」、「拿到最新的裝備」、「抽到最新的五星角色」、「活動拿到高排名」等等。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://4.bp.blogspot.com/-cd2aZNleGX8/Whlu0eCxceI/AAAAAAAAz6w/vZ-Yhc0Mkn08EEPHqAcLvXIUGfImy_UWwCLcBGAs/s1600/cinderella%2Bgirls%2Btriad%2Bprimus.jpeg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="640" data-original-width="1136" height="225" src="https://4.bp.blogspot.com/-cd2aZNleGX8/Whlu0eCxceI/AAAAAAAAz6w/vZ-Yhc0Mkn08EEPHqAcLvXIUGfImy_UWwCLcBGAs/s400/cinderella%2Bgirls%2Btriad%2Bprimus.jpeg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        永遠玩不完的遊戲
                       </span>
                      </b>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        也可能是永遠享受的遊戲
                       </span>
                      </b>
                     </div>
                     <br/>
                     這當然就是在說外國近來最流行的業界用語「Games as a service」，簡稱「GaaS」。遊戲從以往的商品（product）轉化為服務（service），影響是深遠而難以逆轉的，現在所有大型主機遊戲廠商，以及所有手機課金遊戲的開發商，都是在吃這塊大餅，
                     <br/>
                     <br/>
                     遊戲服務化令開發遊戲的銷售凈利率上升，但同時亦衍生出各種問題。遊戲業依賴類賭博機制（課金）來支撐利潤，帶來的是前所有未有的政府規管風險，遊戲業界是否準備好這種衝擊？同時，遊戲服務化帶來龐大利潤的同時，長期來說對業界是否有利？這是這篇文章打算探討的問題。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       課金隨機性帶來的規管風險
                      </span>
                     </b>
                     <br/>
                     <br/>
                     寫這篇文章其中一個主要原因，當然是近月在歐美國家引起注意的
                     <a href="https://www.gamespot.com/articles/star-wars-battlefront-2s-loot-box-controversy-expl/1100-6455155/" target="_blank">
                      《Star Wars Battlefront II》課金機制事件
                     </a>
                     。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-DIixdQawbyc/Whlk92LZzNI/AAAAAAAAz5Y/kL5c5RZcL9Azc3uyX5WOb1fPAR51KmHdACLcBGAs/s1600/pic%2B2017-11-25%2B8.36.37%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="840" data-original-width="1346" height="248" src="https://2.bp.blogspot.com/-DIixdQawbyc/Whlk92LZzNI/AAAAAAAAz5Y/kL5c5RZcL9Azc3uyX5WOb1fPAR51KmHdACLcBGAs/s400/pic%2B2017-11-25%2B8.36.37%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        隨機抽獎的課金機制在外國遇上了挑戰
                       </span>
                      </b>
                     </div>
                     <br/>
                     如果你像團長一樣有留意主機遊戲，又或者是星戰迷，一定已經聽過這次事件，不過這裡也簡單地介紹一下來龍去脈。
                     <br/>
                     <br/>
                     簡單來說，《Star Wars Battlefront II》是一隻以《星球大戰》為主題的第3身射擊遊戲，放棄了前作單純以遊戲時間換取新裝備的系統，引入了隨機的課金機制，重課金玩家可以在遊戲中獲得絕對的優勢，導致大量玩家不滿，由於《星球大戰》在外國被視為老少咸宜的作品，新遊戲引起了連鎖反應。
                     <br/>
                     <br/>
                     大量
                     <a href="http://www.eurogamer.net/articles/2017-11-20-star-wars-battlefront-2-physical-sales-down-60-per-cent-on-battlefront-1" target="_blank">
                      玩家發起杯葛
                     </a>
                     ，銷量受到影響。團長本來也打算玩這遊戲，但也被課金系統嚇退。更嚴重的問題不是消費者的反應，而是政府和政治家的反應。
                     <br/>
                     <div style="clear: both;">
                     </div>

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2017/10/why-is-azurlane-popular-in-japan.html">
                      為什麼《碧藍航線》在日本受歡迎？對《艦隊收藏》有何影響？
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-8666682918746490790">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-6fzwNrhKWyk/Wd2zfWr2tkI/AAAAAAAAz1k/cx1Afoj21-UMju-99ylnHu5XxGkT7lCogCLcBGAs/s1600/IMG_6305.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="1108" data-original-width="1478" height="298" src="https://2.bp.blogspot.com/-6fzwNrhKWyk/Wd2zfWr2tkI/AAAAAAAAz1k/cx1Afoj21-UMju-99ylnHu5XxGkT7lCogCLcBGAs/s400/IMG_6305.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        《碧藍航線》的成功反映了市場的需求
                       </span>
                      </b>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="font-size: x-small;">
                       SOS BLOG 照片，10月9日攝於秋葉原Radio會館
                      </span>
                     </div>
                     <br/>
                     對於喜歡艦娘的社群來說，最近最多人討論的話題，當然就是中國遊戲《碧藍航線》（不少人稱為《艦B》）登陸日本，並且得到大量玩家的支持和同人畫師的支援，當看到一些本來在畫《艦隊收藏》的畫師轉投《碧藍航線》的懷抱，很多人都有既視感：
                     <br/>
                     <br/>
                     這不就與當年《東方Project》畫師轉畫艦娘的情況相似？風向是否已經改變？
                     <br/>
                     <br/>
                     <br/>
                     <b style="text-align: center;">
                      <span style="font-size: x-large;">
                       《碧藍航線》與
                      </span>
                     </b>
                     <b>
                      <span style="font-size: x-large;">
                       《艦隊收藏》的異同
                      </span>
                     </b>
                     <br/>
                     <br/>
                     雖然很多人說《碧藍航線》的冒起與《艦隊收藏》相似，但從商業的角度來看，兩者的成功方式並不相同。
                     <a href="http://www.thesosblogger.com/2013/10/kancolle.html" target="">
                      《艦隊收藏》的成功是意外
                     </a>
                     ，營運並沒有想像到作品擴散的速度，也沒有為此做好準備，當時甚至出現帳號需要抽選的奇怪情況。
                     <br/>
                     <br/>
                     <blockquote class="tr_bq">
                      <b>
                       <span style="font-size: large;">
                        "《艦隊收藏》是同人界從下而上創造的奇蹟，《碧藍航線》是企業從上而下製造的商業成功。"
                       </span>
                      </b>
                     </blockquote>
                     <br/>
                     相對之下，《碧藍航線》的營運從一開始就很清楚自己的目標顧客是誰（對軍事和萌有興趣的宅），而且採取了有效的方式獲取玩家。
                     <br/>
                     <br/>
                     如果你現在到動漫聖地秋葉原，在JR站外的Radio會館，就會看到一幅兩層樓高的廣告展示版，又例如當你打開YouTube，會發現與
                     <a href="https://www.youtube.com/watch?v=yfFjFvYLxyM" target="_blank">
                      有名的YouTuber的合作
                     </a>
                     ，宣傳精準而有效傳到目標客群。遊戲的日服於9月14日開始營運，
                     <a href="http://gamebiz.jp/?p=194786" target="_blank">
                      登錄數在一個月內突破100萬人
                     </a>
                     ，並不是太令人驚訝的事情。
                     <br/>
                     <br/>
                     《艦隊收藏》是同人界從下而上創造的奇蹟，《碧藍航線》是從上而下製造的商業成功。
                     <br/>
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://3.bp.blogspot.com/-Vq5gx8KVg3U/Wd23wDx_nRI/AAAAAAAAz1w/5ktqjvKKWJQjAiqDFOKEG9U8Y3jqBxP-ACLcBGAs/s1600/pic%2B2017-10-11%2B1.51.19%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="819" data-original-width="1600" height="202" src="https://3.bp.blogspot.com/-Vq5gx8KVg3U/Wd23wDx_nRI/AAAAAAAAz1w/5ktqjvKKWJQjAiqDFOKEG9U8Y3jqBxP-ACLcBGAs/s400/pic%2B2017-10-11%2B1.51.19%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         近日對《碧藍航線》的注意度（藍線）
                        </span>
                       </b>
                      </span>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="font-size: large;">
                       <span style="font-weight: bold; text-align: start;">
                        超過了
                       </span>
                       <b>
                        《艦隊收藏》（紅線）
                       </b>
                      </span>
                     </div>
                     <span style="font-size: x-small;">
                      （資料來源：Google Trend）
                     </span>
                     <br/>
                     <br/>
                     當然，純粹把遊戲的成功歸功於市場策略並不反映事實的全部，《碧藍航線》的確擁有優秀的設計，補充了日本玩家對艦娘的需求，人設也剌激了插畫師和玩家的想像。很多人留意到《碧藍航線》就是因為看到
                     <a href="https://www.pixiv.net/member_illust.php?mode=medium&amp;illust_id=65257645" target="_blank">
                      Pixiv
                     </a>
                     或 Twitter 上的愛宕或高雄的插畫的緣故。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-ATFKFiGKp48/Wd3M3mZzpsI/AAAAAAAAz2I/2O9AWTGGT7s8V2yHctkOiJtYYMgLMnZRgCLcBGAs/s1600/IMG_6281.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="750" data-original-width="1334" height="223" src="https://1.bp.blogspot.com/-ATFKFiGKp48/Wd3M3mZzpsI/AAAAAAAAz2I/2O9AWTGGT7s8V2yHctkOiJtYYMgLMnZRgCLcBGAs/s400/IMG_6281.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         《碧藍航線》的人設刺激了大家的想像力
                        </span>
                       </b>
                      </span>
                     </div>
                     <br/>
                     很多人指出了兩隻遊戲的相似之處，例如部份艤裝的設計、遠征（委託）、任務設計等，說《碧藍航線》參考了《艦隊收藏》相信沒有人多少人會反駁，可是同時兩者的遊戲核心體驗（core gameplay loop）並不相同，這也是《碧藍航線》的特別之處，作品補充了一些在《艦隊收藏》難以得到的體驗。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       玩家希望遊戲有更多操作性
                      </span>
                     </b>
                     <br/>
                     <br/>
                     從營運初期，《艦隊收藏》的玩家便一直對遊戲的隨機系統有很多意見，艦娘的攻擊對象並不能操控。起初大家還會幻想這點有所改變，不過隨著遊戲發展，大家開始明白RNG（隨機生成數字）才是《艦隊收藏》的戰鬥系統本質，留下來繼續玩的人都理解這點。
                     <br/>
                     <br/>
                     如果存在一隻能夠自行操作艦隊的《艦隊收藏》，相信會吸引到不少玩家.....
                     <br/>

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2017/09/re-creators-and-last-action-hero-a-world-of-crossover.html">
                      《Re:CREATORS》：設定優秀卻受劇本限制的動畫？
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-9153386504166981551">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-JrJfdN1nfO8/Wb4mbIuQQ5I/AAAAAAAAzxk/EZv48SZMhr8FeWoUdpziv9uHf2Uq0UyOACLcBGAs/s1600/pic%2B2017-09-17%2B3.32.46%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="640" data-original-width="1274" height="200" src="https://2.bp.blogspot.com/-JrJfdN1nfO8/Wb4mbIuQQ5I/AAAAAAAAzxk/EZv48SZMhr8FeWoUdpziv9uHf2Uq0UyOACLcBGAs/s400/pic%2B2017-09-17%2B3.32.46%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        《Re:CREATORS》的全面分析！
                       </span>
                      </b>
                     </div>
                     <br/>
                     <ul>
                      <li>
                       如果動畫作品的角色出現在現實世界會有什麼反應？
                      </li>
                      <li>
                       魔法少女與角色扮演世界的魔法師哪個比較強？
                      </li>
                      <li>
                       劍與魔法世界的騎士與巨大機械人對決又會發生什麼事？
                      </li>
                     </ul>
                     <br/>
                     相信所有動畫迷在某個時點都曾經有類似的幻想。畢竟，精彩的作品就是會刺激觀眾的想像力。《Re:CREATORS》就是在這樣的假設下出現的作品。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       美好的開始，然後....
                      </span>
                     </b>
                     <br/>
                     <br/>
                     《Re:CREATORS》自2017年4月開播以來，引起了不少討論，對角色的討論、對故事的討論、以及對設定的討論，這也難怪，由TROYCA製作，青木英作監督的《Re:CREATORS》在多個方面都表現了高超的水準。精美的作畫、澤野弘之的壯大音樂、令人回味的人設，都是吸引觀眾收看的原因。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://2.bp.blogspot.com/-yXo7r3ritS0/Wb4nSmHaBFI/AAAAAAAAzxs/Gppf2Ucm2v4O9A-Yy0ebU4yG-B_p76FcwCLcBGAs/s1600/pic%2B2017-09-17%2B3.41.46%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="838" data-original-width="1490" height="223" src="https://2.bp.blogspot.com/-yXo7r3ritS0/Wb4nSmHaBFI/AAAAAAAAzxs/Gppf2Ucm2v4O9A-Yy0ebU4yG-B_p76FcwCLcBGAs/s400/pic%2B2017-09-17%2B3.41.46%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        《Re:CREATORS》在背景設定花了很多心思
                       </span>
                      </b>
                     </div>
                     <br/>
                     可是隨著故事的推進，作品在後半的評價每況愈下，觀眾開始質疑廣江禮威筆下的故事方向，甚至認為作品沒有遵守自己訂下的規則。
                     <br/>
                     <br/>
                     究竟《Re:CREATORS》的創作過程是怎樣的？其角色的來源是什麼？為什麼會得到現在的評價？我們希望為大家找出這些答案。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       誰是《Re:CREATORS》的造物主？
                      </span>
                     </b>
                     <br/>

                       <img border="0" data-original-height="1200" data-original-width="1600" height="300" src="https://4.bp.blogspot.com/-60AkGlPb5m4/WY-nPVfiSKI/AAAAAAAAzvc/dG7jToreUEcF5Y3FAfmc3Ek-FjQ-deHwACLcBGAs/s400/IMG_5624.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <span style="text-align: start;">
                       <b>
                        <span style="font-size: large;">
                         埼玉現場報導！
                        </span>
                       </b>
                      </span>
                     </div>
                     <br/>
                     灰姑娘女孩的第5次演唱會終於到了最後一站的埼玉超級競技場（SSA），SOS BLOG這次也到了現場為大家報告！
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       愛梨回來了！
                      </span>
                     </b>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <br/>
                     </div>
                     十時愛梨作為第一次灰姑娘總選舉的冠軍，對不少P來說都有特別意義。可惜的是由於聲優原田ひとみ受傷的關係，被迫缺席之前的演唱會，所以當營運公佈原田會於埼玉復歸的時候，不少人都表現了興奮，同時也有點擔心她的身體狀況。
                     <div>
                      <br/>
                     </div>
                     <div>
                      <div class="separator" style="clear: both; text-align: center;">
                       <a href="https://4.bp.blogspot.com/-lD4B50aVt2Q/WY-l57So1WI/AAAAAAAAzvM/-XREddUueOs_z6TOkumzkrXcgueiK66UACLcBGAs/s1600/pic%2B2017-08-13%2B10.05.13%2BAM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                        <img border="0" data-original-height="852" data-original-width="680" height="320" src="https://4.bp.blogspot.com/-lD4B50aVt2Q/WY-l57So1WI/AAAAAAAAzvM/-XREddUueOs_z6TOkumzkrXcgueiK66UACLcBGAs/s320/pic%2B2017-08-13%2B10.05.13%2BAM.png" width="255"/>
                       </a>
                      </div>
                      <div class="separator" style="clear: both; text-align: center;">
                       <b>
                        <span style="font-size: large;">
                         第一代灰姑娘！
                        </span>
                       </b>
                      </div>
                      <div class="separator" style="clear: both; text-align: center;">
                       <a href="https://1.bp.blogspot.com/-QjHu-xHGzA8/WY-mLXRzQQI/AAAAAAAAzvU/dHzD-a7GDFkerZl2ABjiFI4YL2SrclwsgCEwYBhgL/s1600/IMG_5668.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                        <img border="0" data-original-height="1200" data-original-width="1600" height="240" src="https://1.bp.blogspot.com/-QjHu-xHGzA8/WY-mLXRzQQI/AAAAAAAAzvU/dHzD-a7GDFkerZl2ABjiFI4YL2SrclwsgCEwYBhgL/s320/IMG_5668.jpg" width="320"/>
                       </a>
                      </div>
                      <div class="separator" style="clear: both; text-align: center;">
                       <b>
                        <span style="font-size: large;">
                         看到原田的名字就有點感動
                        </span>
                       </b>
                      </div>
                      <div>
                       <br/>
                      </div>
                      <div>
                       <div>
                        營運似乎也考慮到大家的心情，所以在全體曲的「Yes! Party time!!」之後的立即讓原田唱「アップルパイ・プリンセス」！唱完後聽到的喝采和鼓掌，相信大家都理解這首歌對原田和一眾愛梨P的重要性！
                       </div>
                       <div>
                        <br/>
                        <br/>
                        <b>
                         <span style="font-size: x-large;">
                          精彩的remix！
                         </span>
                        </b>
                        <br/>
                        <br/>
                        Day1 最令觀眾興奮的環節應該就是「DJぴにゃスペシャルメドレー 」，多達20首歌曲（基本上都是早期的經典歌曲）以remix的方式連續唱出，而且營運很刻意讓不同屬性的偶像有機會唱到其他屬性的歌曲。
                       </div>
                      </div>
                     </div>

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2017/08/aqours-2nd-live-happy-party-train.html">
                      Aqours 2nd Live 名古屋現場報告！進化的 Aqours！（更新Day 2內容）
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-3901049528858792334">
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-pauyLib0DJ0/WYXpGK8BXqI/AAAAAAAAzt0/tLG4eh5b5pkLnwWoHTbFSoG0ny0b5lB-wCLcBGAs/s1600/IMG_5126.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="1600" data-original-width="1200" height="400" src="https://1.bp.blogspot.com/-pauyLib0DJ0/WYXpGK8BXqI/AAAAAAAAzt0/tLG4eh5b5pkLnwWoHTbFSoG0ny0b5lB-wCLcBGAs/s400/IMG_5126.jpg" width="300"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        Aqours 9人到了名古屋，展開了Live Tour！！
                       </span>
                      </b>
                     </div>
                     <br/>
                     <i>
                      （8月6日晚上更新
                      <span style="color: red;">
                       Day 2內容
                      </span>
                      ）
                     </i>
                     <br/>
                     <br/>
                     Aqours 的第二次演唱會的第一站到了名古屋！這次 SOS BLOG 也到了現場為大家報告！
                     <br/>
                     <div>
                      <br/>
                      <b>
                       <span style="font-size: x-large;">
                        進化的Aqours！
                       </span>
                      </b>
                     </div>
                     <div>
                      <br/>
                     </div>
                     <div>
                      相對於
                      <a href="http://www.thesosblogger.com/2017/03/love-live-sunshine-aqours-first-live-step-zero-to-one-report-in-yokohama.html">
                       第一次演唱會
                      </a>
                      ，這次的表演明顯看到Aqours成員的表現更加穩定，畢竟兩次演唱會中間有其他表演機會，而且多了數個月的練習，所以表現有所成長！
                     </div>
                     <div>
                      <br/>
                     </div>
                     <div>
                      這次的演唱會以「HAPPY PARTY TRAIN」為題，所以第一首的歌曲就是3單的「HAPPY PARTY TRAIN」！第一次聽到現場的版本，果然是夢幻的歌曲，舞台上更有火車出現，氣氛十足！
                      <br/>
                      <br/>
                      <div>
                       <b>
                        唱得最好的果然是.....
                       </b>
                      </div>
                      <div>
                       <br/>
                      </div>
                      <div>
                       Guilty Kiss 的歌唱力果然非同凡響！記得平日聽unit曲，聽得最多的是AZALEA的「GALAXY HidE and SeeK」，而事實上現場的版本也很好聽，加上服裝更有加乘。不過最令人眼前一亮的，果然還是Guilty Kiss的「コワレヤスキ」，這個組合真的很適合節奏激烈的歌曲！
                       <br/>
                       <br/>
                       另外第一次聽到現場的「Daydream Warrior」果然也是一首極有魅力的歌曲！
                      </div>
                      <br/>
                      <div>
                      </div>
                      <br/>
                      <div style='-webkit-text-stroke-width: 0px; color: black; font-family: "PingFang TC"; font-size: medium; font-style: normal; font-variant-caps: normal; font-variant-ligatures: normal; font-weight: normal; letter-spacing: normal; orphans: 2; text-align: start; text-decoration-color: initial; text-decoration-style: initial; text-indent: 0px; text-transform: none; white-space: normal; widows: 2; word-spacing: 0px;'>
                       <div style="margin: 0px;">
                        <br/>
                       </div>
                       <div class="separator" style="clear: both; text-align: center;">
                        <a href="https://2.bp.blogspot.com/-48roHWapiZ4/WYX00KSSwoI/AAAAAAAAzug/bD5rIONgufIcG8wrVTxptOfjUcgcisCQACLcBGAs/s1600/pic%2B2017-08-06%2B1.39.25%2BAM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                         <img border="0" data-original-height="700" data-original-width="1242" height="225" src="https://2.bp.blogspot.com/-48roHWapiZ4/WYX00KSSwoI/AAAAAAAAzug/bD5rIONgufIcG8wrVTxptOfjUcgcisCQACLcBGAs/s400/pic%2B2017-08-06%2B1.39.25%2BAM.png" width="400"/>
                        </a>
                       </div>
                       <div class="separator" style="clear: both; text-align: center;">
                        <span style="font-size: large;">
                         <b>
                          全場綠色的
                         </b>
                        </span>
                        <b>
                         <span style="font-size: large;">
                          「HAPPY PARTY TRAIN」！
                         </span>
                        </b>
                       </div>
                       <br/>
                       <div>
                       </div>
                       <br/>
                       <div style='-webkit-text-stroke-width: 0px; color: black; font-family: "PingFang TC"; font-size: medium; font-style: normal; font-variant-caps: normal; font-variant-ligatures: normal; font-weight: normal; letter-spacing: normal; orphans: 2; text-align: start; text-decoration-color: initial; text-decoration-style: initial; text-indent: 0px; text-transform: none; white-space: normal; widows: 2; word-spacing: 0px;'>
                        <div style="margin: 0px;">
                         <br/>
                        </div>
                       </div>
                       <div class="separator" style='-webkit-text-stroke-width: 0px; clear: both; color: black; font-family: "PingFang TC"; font-size: medium; font-style: normal; font-variant-caps: normal; font-variant-ligatures: normal; font-weight: normal; letter-spacing: normal; margin: 0px; orphans: 2; text-align: center; text-decoration-color: initial; text-decoration-style: initial; text-indent: 0px; text-transform: none; white-space: normal; widows: 2; word-spacing: 0px;'>
                       </div>
                      </div>
                      <br/>
                      <b>
                       <span style="color: red; font-size: large;">
                        Day 2 竟然唱BD特典歌曲！
                       </span>
                      </b>
                      <br/>
                      <br/>
                      這次演唱會最特別的選曲大概是第二天的「P.S.の向こう側」、「LONELY TUNING」、「Guilty eyes fever」，這3首unit歌曲都是特定店鋪的BD全卷購入特典，也就是說只有很少人可以從正常途徑入手這些歌曲。
                      <br/>
                      <br/>
                      <div class="separator" style="clear: both; text-align: center;">
                       <a href="https://3.bp.blogspot.com/-sAgN-3Vm-o8/WYcrwry1riI/AAAAAAAAzu4/3C-ZsvemY0A9YpFVJfqgnhw6OMPujg8TwCLcBGAs/s1600/pic%2B2017-08-06%2B11.45.38%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                        <img border="0" data-original-height="752" data-original-width="766" height="313" src="https://3.bp.blogspot.com/-sAgN-3Vm-o8/WYcrwry1riI/AAAAAAAAzu4/3C-ZsvemY0A9YpFVJfqgnhw6OMPujg8TwCLcBGAs/s320/pic%2B2017-08-06%2B11.45.38%2BPM.png" width="320"/>
                       </a>
                      </div>
                      <div class="separator" style="clear: both; text-align: center;">
                       <b>
                        神曲「Guilty eyes fever」！！
                       </b>
                      </div>
                      <br/>
                      一如所料，唱這3首新曲時現場不少人表現了不熟悉的反應，幸好3首都是好歌，現場反應也很好，特別是「Guilty eyes fever」再次表現了Guilty Kiss的歌唱力！大家叫 Guilty Fever！氣氛升至最高點！個人是希望這些歌曲以後也能從其他途徑發售，讓更多人欣賞到這些歌曲。
                      <br/>
                      <br/>
                      <br/>
                      <b>
                       <span style="color: red; font-size: large;">
                        愛香果然是製造氣氛的高手！
                       </span>
                      </b>
                      <br/>
                      <br/>
                      話說最近愛香在Aqours感覺就像是故意為大家製造吐嘈點，令現場的氣氛緩和，這次的名古屋演唱會也完成了以上的任務。愛香一有機會就說「墮天！」（例如展示二單服裝的時候），然後大家都會故意無視或稱她為善子，看著愛香「所以就說是夜羽！」的反應感覺真的很可愛！
                      <br/>
                      <br/>
                      <br/>
                      <b>
                       <span style="color: red; font-size: large;">
                        Aqours 2nd 有身份確認嗎？
                       </span>
                      </b>
                      <br/>
                      <br/>
                      由於不少讀者問到這問題，所以在這裡直接解釋，自從
                      <a href="http://www.thesosblogger.com/2017/07/how-eventer-is-affected-by-eplus-change.html">
                       日本各演唱會營運開始打擊門票轉售
                      </a>
                      ，很多演唱會都有作出身份確認的措施，查證入場者的名字。Aqours 2nd 有採取這種措施嗎？答案是有的，但只是抽查。
                      <br/>
                      <br/>
                      以名古屋為例，Arena席入場時有隨機抽查，另外入場後一些較重要的位置也有抽查。2階席入場基本上沒檢查，但入場後也有抽查，團長就親眼看著附近有2人被職員帶走後沒有再回來，也有坐Arena席的朋友看到有人被帶走。
                      <br/>
                      <br/>
                      所以這次 Aqours 2nd 購買轉售門票有一定風險，這是大家要理解的。
                      <br/>
                      <br/>
                      <br/>
                     </div>
                     <div>
                      <b>
                       <span style="font-size: large;">
                        最佳服裝是3年生？
                       </span>
                      </b>
                      <br/>

                    <h3 class="post-title entry-title">
                     <a href="http://www.thesosblogger.com/2017/07/tokyo-dolls-unexpected-good.html">
                      格鬥遊戲變身偶像遊戲？《Project 東京 Dolls》告訴我們推出時機的重要性
                     </a>
                    </h3>
                    <div class="post-header">
                     <div class="post-header-line-1">
                      <span class="post-author vcard">
                       張貼者：
                       <span class="fn">
                        團長
                       </span>
                      </span>
                     </div>
                    </div>
                    <div class="post-body entry-content" id="post-body-7908718158931938773">
                     <div class="separator" style="clear: both; text-align: center;">
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-RKbole2ahYs/WXx2gySI5MI/AAAAAAAAzsI/HkXzAv0T6g0GrBEEViFTv8lVwnNP8sHqACLcBGAs/s1600/IMG_4545.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="750" data-original-width="1334" height="223" src="https://1.bp.blogspot.com/-RKbole2ahYs/WXx2gySI5MI/AAAAAAAAzsI/HkXzAv0T6g0GrBEEViFTv8lVwnNP8sHqACLcBGAs/s400/IMG_4545.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        《Project 東京 Dolls》值得一玩嗎？
                       </span>
                      </b>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <br/>
                     </div>
                     近月實在有不少好玩的手機遊戲登場，除了不少讀者都有玩的《Million Live Theater Days》，還有《結城友奈是勇者》這隻意外地可以輕鬆玩的手遊。
                     <br/>
                     <br/>
                     不過今天要介紹的卻是一隻較少人留意的作品，團長發現這遊戲頗有故事性和遊戲性，而且本來被
                     <b>
                      設計作格鬥遊戲卻搖身一變加入偶像元素
                     </b>
                     ！那就是Square Enix的
                     <b>
                      《Project 東京 Dolls》
                     </b>
                     （プロジェクト東京ドールズ）。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       Square Enix 內戰的犧牲品？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     聽到 Square Enix（SE），不少玩家相信也會提高警戒，因為 SE 的手機遊戲給玩家的感覺是如果玩家人數未達預期，營運很快就會把遊戲關掉，
                     <a href="https://headlines.yahoo.co.jp/article?a=20170518-00010006-otapolz-game" target="_blank">
                      SE在2017年也關掉了數隻手遊
                     </a>
                     ，因此不少人體驗過後對 SE 敬而遠之，避免浪費時間和金錢。
                     <br/>
                     <br/>
                     這次介紹的《東京 Dolls》除了是SE出品，更神奇的是其推出日期。《東京 Dolls》的營運在2017年6月22日開始，可是SE與Pokelabo合作的另一隻手遊大作《SINoALICE》卻是在兩星期前的6月6日開始營運。
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://3.bp.blogspot.com/-Yahtm5s4HR0/WXx30FK8axI/AAAAAAAAzsU/tD4g6ZcQarcbzbW52_QO0Mqhiq0iNIAKgCLcBGAs/s1600/pic%2B2017-07-29%2B7.55.32%2BPM.png" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="764" data-original-width="1600" height="190" src="https://3.bp.blogspot.com/-Yahtm5s4HR0/WXx30FK8axI/AAAAAAAAzsU/tD4g6ZcQarcbzbW52_QO0Mqhiq0iNIAKgCLcBGAs/s400/pic%2B2017-07-29%2B7.55.32%2BPM.png" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        想不到同一發行商的遊戲會在同期推出！
                       </span>
                      </b>
                     </div>
                     <br/>
                     一般來說，這種情況是很少見的，手遊的推出時期往往決定了遊戲的生命，而發行商的職責之一就是為遊戲找到好檔期。同一發行商讓兩隻手遊同期推出，代表兩者必然會互相搶奪玩家和資源，日本玩家在討論遊戲時也對此情況也表達了疑問。
                     <br/>
                     <br/>
                     更令人不解的是，《SINoALICE》推出後
                     <a href="http://sinoalice.jp/collaboration/" target="_blank">
                      隨即與大熱的《NieR：Automata》合作
                     </a>
                     ，聲勢上完全壓倒了《東京 Dolls》。網路上的討論、課金排名等也明顯是《SINoALICE》佔優。
                     <br/>
                     <br/>
                     即使如此，《東京 Dolls》還是有些特別之處。
                     <br/>
                     <br/>
                     <br/>
                     <b>
                      <span style="font-size: x-large;">
                       《東京 Dolls》本來是格鬥遊戲？
                      </span>
                     </b>
                     <br/>
                     <br/>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/--ZcOm_7iYfs/WXx7VvMbyAI/AAAAAAAAzso/8t2CGbiRM8Ui5m4eJLRtUn-QiDavZQHygCLcBGAs/s1600/IMG_4908.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="750" data-original-width="1334" height="223" src="https://1.bp.blogspot.com/--ZcOm_7iYfs/WXx7VvMbyAI/AAAAAAAAzso/8t2CGbiRM8Ui5m4eJLRtUn-QiDavZQHygCLcBGAs/s400/IMG_4908.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        遊戲世界觀有點像現時的輕小說
                       </span>
                      </b>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <a href="https://1.bp.blogspot.com/-ZLoBf1YtgXY/WXx7fZQ16sI/AAAAAAAAzss/MMAon-OIhLUySR7BVCpUKM7w-ti1HZOrACLcBGAs/s1600/IMG_4806.jpg" imageanchor="1" style="margin-left: 1em; margin-right: 1em;">
                       <img border="0" data-original-height="750" data-original-width="1334" height="223" src="https://1.bp.blogspot.com/-ZLoBf1YtgXY/WXx7fZQ16sI/AAAAAAAAzss/MMAon-OIhLUySR7BVCpUKM7w-ti1HZOrACLcBGAs/s400/IMG_4806.jpg" width="400"/>
                      </a>
                     </div>
                     <div class="separator" style="clear: both; text-align: center;">
                      <b>
                       <span style="font-size: large;">
                        但突然加入偶像元素？
                       </span>
                      </b>
                     </div>
                     <div style="clear: both;">
                     </div>
                    </div>
