    /*** 롤링순위 ***/
function rankScroll(obj){
	obj.append($("div:first-child", obj).clone());
	window.setInterval(function(){		
		//console.log(obj.scrollTop() + " - " + obj.prop("scrollHeight"));
		if (obj.scrollTop()>=400){
			obj.scrollTop(0);
		}
		obj.animate({
			scrollTop: "+=40"
		});	
	}, 3000);
}
rankScroll($(".owl-top-keywords"));

$(".global_ranking").mouseover(function(){
	$(".box-top-keywords").show();
}).mouseout(function(){
	$(".box-top-keywords").hide();
});

/*** 추천검색어 노출 ***/
$("body").click(function(event){
	if (event.target.tagName=="INPUT") return;
	$(".box-top-search").removeClass('active');
});

$(document).ready(function(e){
	$('.box-top-search a').live("focusin", function(e){$(this).addClass("active");});
	$('.box-top-search a').live("focusout", function(e){$(this).removeClass("active");});
	$('.global_search input').keydown(function(event){
		//event.preventDefault();
		switch(event.keyCode){
			case 40:
				$(".box-top-search a").eq(0).focus();
				return false;
				break;
		}
	});
	$('body').keydown(function(event){
		var $obj = $('.box-top-search dd a.active');
		switch(event.keyCode){
			case 38:
				if ($obj.parent("dd").prev('dd').find('a').length>0) $obj.parent("dd").prev('dd').find('a').focus();
				else $obj.parents().parent().prev('li').find("dd a").focus();
				event.preventDefault();
				return false;
				break;
			case 40:
				if ($obj.parent("dd").next('dd').find('a').length>0) $obj.parent("dd").next('dd').find('a').focus();
				else $obj.parents().parent().next('li').find("dd a").eq(0).focus();
				event.preventDefault();
				return false;
				break;
		}
	});
});
