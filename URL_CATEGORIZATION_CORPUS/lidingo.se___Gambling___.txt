   String.prototype.shortenString = function(length){
      if(this.length > length) {
         return this.substring(0, length).concat('...');
      }else{
         return this;
      }
   }
   Array.prototype.chunk = function(chunkSize) {
       var R = [];
       for (var i=0; i<this.length; i+=chunkSize)
           R.push(this.slice(i,i+chunkSize));
       return R;
   }
     
   var fetchNews = function(start, num) {
      fetching = true;
      jq.get("/91.2e2f4c114d5435c2fd7f01/12.2e2f4c114d5435c2fd7f0b.json", {start: start, num: num, pathIds: config.pathIds, imgMaxWidth: config.imgMaxWidth}, "json")
         .done(function(data) {

               if(currNum >= (4*2)){

         }
         disableButton();
      }
      return html;
   }
   var addNews = function(hits){      
      var html = '';
      html += '<div class="sv-row c72" >';
      jq.each(hits, function(i, hit) { 
         html += '<div class="sv-column-' + config.colSize + '">';
        
         html += '   <a href="' + hit.uri + '">'
         if(hit.img){
            html += '   <div class="sol-startpage-news-img">';         
            html += '      <img src="' + hit.img +'">';
            html += '   </div>'
            html += '   <div class="sol-startpage-news-text normal">';   
         }else{
            html += '   <div class="sol-startpage-news-text normal sol-no-img">';   
         }  
        
         html += '         <span class="sol-news-title">' + hit.title + '</span>';
         html += '         <span class="sol-news-date-desc">' + hit.date + ' | ' + hit.desc.shortenString(config.descLength) + '</span>';
         html += '      </div>'
         html += '   </a>'
         html += '</div>';
      });
      
      html += getMissingCols(hits.length);

      html += '</div>'

      jq(newsContainer).append(html);

      if(currNum === 0) {
         jq(newsContainer).children('.sv-row').show();
      } else{
         jq(newsContainer).children('.sv-row').fadeIn();

      fetchNews(currNum, config.num);

   fetchNews(currNum, (config.startRows * config.num));

         });
         jq('.sol-startpage-news-row').fadeOut();
         jq('.sol-js-fetch').removeClass('sol-disabled').text('Visa fler nyheter');
         currNum = (config.num * config.startRows) - config.num;
      });
   });
   
   
})();
