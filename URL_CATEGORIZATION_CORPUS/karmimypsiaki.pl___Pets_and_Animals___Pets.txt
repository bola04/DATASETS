        });

        $(".cart-nr").keyup(function () {
            var el = $(this);
            if (el.val().length === 4 && el.data('id') < 4) {
                $("#cyclic-cart-nr-" + (el.data('id') + 1)).focus();
            }
        });

        var cyclic_inputs = [
            {el: $("#cyclic-first-name"), reg: /^[A-ZĄĆĘŁŃÓŚŹŻa-ząćęłńóśźż]+$/},
            {el: $("#cyclic-last-name"), reg: /^([A-ZĄĆĘŁŃÓŚŹŻa-ząćęłńóśźż]+[\-\']?)*([A-ZĄĆĘŁŃÓŚŹŻa-ząćęłńóśźż]+)?$/},
            {el: $("#cyclic-cart-nr-1"), reg: /^[0-9]{4}$/},
            {el: $("#cyclic-cart-nr-2"), reg: /^[0-9]{4}$/},
            {el: $("#cyclic-cart-nr-3"), reg: /^[0-9]{4}$/},
            {el: $("#cyclic-cart-nr-4"), reg: /^[0-9]{4}$/},

            {el: $("#cyclic-cart-day"), reg: /^[0-9]{2}$/},
            {el: $("#cyclic-cart-type"), reg: /^[A-Z]{3}$/},
            {el: $("#cyclic-desc"), reg: /^[0-9A-ZŻŹĄĆĘŁŃÓŚa-ząćęóśźż`!@#\$%\^&\*\(\)_\+\-=\[\]\{\};,\.\? ]{3,}$/}

        $("#zapisz-cykliczne").submit(function (e) {

            var error = false;
            for (var i = 0; i < cyclic_inputs.length; i++) {
                var input = cyclic_inputs[i];
                var parent = input.el.parents('.form-box');
                parent = parent.length ? parent : input.el.parent();
                parent.removeClass('error');
                if (!input.reg.test(input.el.val())) {
                    parent.addClass('error');
                    error = true;
                }
            }
            if (error) {
                e.preventDefault();
            }
        });
