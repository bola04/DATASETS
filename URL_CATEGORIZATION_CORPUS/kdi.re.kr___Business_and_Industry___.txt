    function kdicodesub(thisvalue,targetid) {
		$.ajax({
			async: true
			, url: '/api/kdicode.jsp'
			, type: 'post'
			, dataType: 'json'
			, data: { 'scode':thisvalue }
			, success: function(response) {
				var Json = response;
				$("#"+targetid).empty();
				$("#"+targetid).append("<option value=\"\">소분류 선택</option>");
				for(var k = 0; k < Json.data.length; k++) {
					$("#"+targetid).append("<option value=\""+Json.data[k].kdicode+"\">"+Json.data[k].codename+"</option>");				
				}
				$("#"+targetid).selectOrDie("update");
			}
			, error: function(request,status,error){
				alert("소분류 갱신에 실패 했습니다.");
			}
		});						
		
	}
   </script>
