   /* Custom JS Code */ function tm_custom_sticky(){
	
	
	if( jQuery('.masthead-header-stickyOnScroll').length > 0 ){
		
		// Top offset
		var offset_px = 0;
		if( jQuery('body').hasClass('admin-bar') ){
			offset_px = jQuery('#wpadminbar').height();
		}
		jQuery('.thememount-topbar').parent().attr('id','tm-topbar');  // adding ID to Topbar
		offset_px = offset_px + jQuery('.thememount-topbar').parent().height(); // We are adding Topbar height
		
		
		// Topbar offset
		var topbar_offset_px = 0;
		if( jQuery('body').hasClass('admin-bar') ){
			topbar_offset_px = jQuery('#wpadminbar').height();
		}
		
		
		// Returns width of browser viewport
		var pageWidth = jQuery( window ).width();
		var selector       = jQuery('.masthead-header-stickyOnScroll');
		var selectorParent = 'stickable-header-sticky-wrapper';
		
		if( jQuery('body').hasClass('thememount-header-style-3')){
			selector       = jQuery('#navbar');
			selectorParent = 'navbar-sticky-wrapper';
		}
		
		jQuery(selector).unstick();  // unstick first
		
		
		if( parseInt(pageWidth) > parseInt(tm_breakpoint) ){
			if( jQuery(selector).parent().attr('id')!=selectorParent ){

				jQuery('.thememount-topbar').parent().sticky({topSpacing:topbar_offset_px});  // Topbar
			}
		} else {
			if( jQuery(selector).parent().attr('id') == selectorParent ){
				jQuery(selector).unstick();
				jQuery('.thememount-topbar').parent().unstick();  // Topbar
			}
		}
	}
	
}
