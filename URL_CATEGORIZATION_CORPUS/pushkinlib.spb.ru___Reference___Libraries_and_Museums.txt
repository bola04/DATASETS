		if (!$(this).hasClass('disabled')) {
			zoom += .2;
			setZoom(zoom);
		};

		if (!$(this).hasClass('disabled')) {
			zoom -= .2;
			setZoom(zoom);
		};

	$('#cycle-slider ul').cycle({
		fx: 'fade',
		speed: 'slow', 
		timeout: 5000 
	});
});

function setZoom(zoom) {
	
	if (zoom >= 2) {
		$('.increase-button').addClass('disabled');
	}
	else if (zoom <= 1){
		$('.decrease-button').addClass('disabled');
	}
	else {
		$('.increase-button, .decrease-button').removeClass('disabled');
	};

	if (jQuery.browser.mozilla && jQuery.browser.version >= '3.6') {
		$('body').css({
			'-moz-transform': 'scale(' + zoom + ')',
			'-moz-transform-origin': '0 0'
		});
	}
	else {
		$('body').css({
			'zoom': zoom
		});
	}

	$.cookie('zoom', zoom);
};
