	function rep_login(json)
	{
		if( json.redirect ){
			document.location.href = json.redirect;
			return 1;
		}
		else if(json.errors.email_exists){
			jQuery("#frame_errors").empty().append(json.errors.email_exists+"<br />");
			jQuery("#frame_errors").slideDown();
		}
		else if(json.errors.send_mail_error){
			jQuery("#frame_errors").empty().append(json.errors.send_mail_error+"<br />");
			jQuery("#frame_errors").slideDown();
		}
		else if(json.valid){
			return 1;
		}
		else{}
		return 0;
	}

	function fb_login()
	{
		 FB.login(function(response){
			if (response.authResponse){
				FB.api('/me', {fields: 'last_name,first_name,id,email'}, function(response) {
					
					new Ajax.Request( site+"auth/checkformregisterfb/",

	}
