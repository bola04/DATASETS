  <script>
   var featuredLoaded = false;
    
    function closeNav(){
        $("#new-nav-menu").fadeOut(450);
        $('#new-search-results').fadeOut(450);
        $('body').removeClass('nav-open');
    }
    
    function openNav(){
        $("#new-nav-menu").fadeIn(450);
        $('#new-search-results').fadeIn(450);
        $('body').addClass('nav-open');
        if (!featuredLoaded) loadFeatured();
    }
    
    $(document).mouseup(function (e)
    {
        var container = $("#new-nav-menu");

        if (!container.is(e.target) // if the target of the click isn't the container...
            && container.has(e.target).length === 0) // ... nor a descendant of the container
        {
            closeNav();
        }
    });
    
 
    
    //algolia search UI function
    //show or hide search input
    $('#new-nav-menu #search-terms').focus(function() {
        $('#new-nav-menu #search-terms').autocomplete('open');
    });
    $('#new-nav-menu #search-terms').keyup(function() {
        if ($(this).val() != '')
            $('#new-search-results .row').hide();
        else
            $('#new-search-results .row').show();
        
        resizeMenuItems();
    });

    //execute search
    var client = algoliasearch('B5DSZCOIIX', '5e7623a12be3e483e0d72f62b31f5d6b')
    var index = client.initIndex('laravel_all');
    var debug = +"0" ? true : false;
    $('#new-nav-menu #search-terms').autocomplete({
            hint: false,
            debug: true, /* true */
            cssClasses: {
                suggestion: 'suggestion thumb-holder'
            },
            dropdownMenuContainer: '#new-search-results',
            templates: {
                header: '<div class="searchh1">' +
                '<div class="algolia-logo-holder">' +
                    '<span>Powered by </span><img src="https://d3kfjvhh435dey.cloudfront.net/_laravel/images/algolia-logo.png" />' +
                '</div>' +
                '<h3>Search Results</h3>' +
                '</div>',
                empty: '<span>No results found.</span>'
            }
        }, [{
            source: $.fn.autocomplete.sources.hits(index, { hitsPerPage: 36 }),
            displayKey: 'name',
            templates: {
                suggestion: function(hit) {
                    if(!hit.thumb) {
                        hit.thumb = 'https://d3kfjvhh435dey.cloudfront.net/_laravel/images/img_placeholder.png'
                    }
                    return '<a href="' + hit.href + '">' +
                            '<h4>' + hit._highlightResult.name.value + '</h4>' +
                            '<img src="' + hit.thumb + '"/>' +
                            '<span> ' + getHitType(hit.objectID) + '</span>' +
                            '</a>';
                }
            }
    }]);

    function getHitType(objectID) {
        var html, type = objectID.split('-').shift().toLowerCase();
        switch(type) {
            case 'map':
                html = '<i class="fa fa-globe" aria-hidden="true"></i> Map + trails';
                break;
            case 'trail':
                html = '<i class="fa fa-street-view" aria-hidden="true"></i> Trail';
                break;
            default:
                html = '<i class="fa fa-eye" aria-hidden="true"></i> Point of Interest';
                break;
        }
        return html;
    }
    
    function loadFeatured(){
        if (!featuredLoaded){ // Double check to make sure menu items aren't loaded twice
            $.ajax({
                url: '/api/menu-items',
                type: 'GET',
                dataType: 'JSON',
                success: function(data){
                    featuredLoaded = true;

                    $.each(data.featured_maps, function(i, v){
                        $('#fMaps-container').append('<div class="aa-suggestion thumb-holder"><a href="/map/'+ v.slug +'/visual"><h4>'+ v.name +'</h4><img src="'+ v.img +'"></a></div>');
                    });
                    $.each(data.featured_trails, function(i, v){
                        $('#fTrails-container').append('<div class="aa-suggestion thumb-holder"><a href="/trail/'+ v.slug +'"><h4>'+ v.name +'</h4><img src="'+ v.img +'"></a></div>');
                    });
                    $.each(data.last_maps, function(i, v){
                        $('#rMaps-container').append('<div class="aa-suggestion thumb-holder"><a href="/map/'+ v.slug +'/visual"><h4>'+ v.name +'</h4><img src="'+ v.img +'"></a></div>');
                    });
                    $.each(data.last_trails, function(i, v){
                        $('#rTrails-container').append('<div class="aa-suggestion thumb-holder"><a href="/trail/'+ v.slug +'"><h4>'+ v.name +'</h4><img src="'+ v.img +'"></a></div>');
                    });
                    resizeMenuItems();
                }
            });
        }
    }
