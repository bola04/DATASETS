function dropDownContainerSetPosition() {
	var left = dropDownLeft;
	var dropDownWidth = jQuery('.menuRow').position().left + jQuery('.menuRow').width() - left;

function liToDiv(target, rows) {
	var i = 0;
	var j = 0;
	var currentDiv = jQuery('<div></div>');
	currentDiv.addClass('locationVisible');
	currentDiv.attr('locationGroup',0);
	currentDiv.appendTo(target);
	jQuery(target + ' a').each(function() {
		var newButton = jQuery('<div class="menuButton"></div>');
		newButton.text(jQuery(this).text());
		newButton.attr('href', jQuery(this).attr('href'));
		newButton.click(divLinkClick);
		newButton.appendTo(currentDiv);
		i++;
		if (i > (rows - 1)) {
			j++;
			currentDiv = jQuery('<div></div>');
			if (Math.floor(j/3) == 0) currentDiv.addClass('locationVisible');
			currentDiv.attr('locationGroup',Math.floor(j/3));
			currentDiv.appendTo(target);
			i = 0;
		}
	});
	jQuery(target + ' a').remove();
}

function searchToggle(animate) {
	if (homepage) {
		if (animate != false)
			performSearch();
		return false;
	}
	jQuery('.dropDownContainerInner').hide();
	if (typeof animate == 'undefined') animate = true;
	if (!jQuery(this).parent().hasClass('searchClose') &&
		jQuery('.searchSlide input').val().length > 0 &&
		jQuery('.searchSlide input:visible').length > 0 &&
		animate) {
		performSearch();
		return false;
	}
	if (animating) return false;
	animating = true;
	if (jQuery('.searchClose:visible').length > 0) {
		jQuery('.searchClose').css('display','none');
	} else {
		jQuery('.searchClose').css('display','table-cell');
	}
	if (jQuery('.subMenu').width() > 1) { animating = false; menuToggle(false); }
	if (animate)
		jQuery('.searchSlide input').toggle("slide", { direction: "right" }, 300, function() {animating = false});
	else {
		jQuery('.searchSlide input').toggle();
		animating = false
	}
	if (jQuery('.searchSlide input:visible').length > 0 && animate != false) jQuery('.searchSlide input').focus();
}
menuToggling = false;
function menuToggle(animate) {
	menuToggling = true;
	if (typeof animate == 'undefined') animate = true;
	//jQuery('.locationsLeftArrow, .locationsRightArrow').hide();
	if (animating) return false;
	animating = true;
	if (jQuery('.menuClose:visible').length > 0) {
		jQuery('.menuClose').css('display','none');
	} else {
		jQuery('.menuClose').css('display','table-cell');
	}
	
	if (jQuery('.searchSlide input:visible').length > 0) { animating = false; searchToggle(false); }
	
	var opening = true;
	if (jQuery('.subMenu').width() > 1) opening = false;
	
	if (opening) jQuery('.menuToggle').css({'max-width': 0});
	
	if (animate) {
		if (!opening) {
			jQuery('.menuToggle').delay(300).animate({'max-width': 1000},300);
			jQuery('.subMenu').animate({'max-width': 0},600, function() { animating = false; });
		}
		else
			jQuery('.subMenu').animate({'max-width': 1000},500, function() { animating = false; });
	} else {
		if (jQuery('.subMenu').width() > 1) {
			jQuery('.subMenu').css({'max-width': 0});
			jQuery('.menuToggle').css({'max-width': 1000});
		} else {
			jQuery('.subMenu').css({'max-width': 1000});
			jQuery('.menuToggle').css({'max-width': 0});
		}
		animating = false
	}
}

function locationMove(i) {
	maxLocation = 0;
	jQuery('.locationLinksContainer > div > div').each(function() {
		if (parseInt(jQuery(this).attr('locationGroup')) > maxLocation) maxLocation = jQuery(this).attr('locationGroup');
	});
	if (i < 0 || i > maxLocation) return false;
	
	jQuery('.locationLinksContainer').attr('currentPage',i);
	
	jQuery('.locationVisible').removeClass('locationVisible');
	
	jQuery('.locationLinksContainer > div > div').each(function() {
		if (parseInt(jQuery(this).attr('locationGroup')) == i) jQuery(this).addClass('locationVisible');
	});
	
	if (i == 0)
		jQuery('.locationsLeftArrow img').css('opacity',0.2);
	else
		jQuery('.locationsLeftArrow img').css('opacity',1);
		
	if (i == maxLocation)
		jQuery('.locationsRightArrow img').css('opacity',0.2);
	else
		jQuery('.locationsRightArrow img').css('opacity',1);
}
