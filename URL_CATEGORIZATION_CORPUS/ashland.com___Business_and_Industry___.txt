       var params={};
window.location.search
  .replace(/[?&]+([^=&]+)=([^&;]*)/gi, function(str,key,value) {
    params[key] = decodeURIComponent(value).trim();
  }
);

var searchedTerm = params['q'];

var typeAheadEnableLength = 3;
var resultsDisplaySize = 8;
var suggestersDisplaySize = 5;
var productsDisplaySize = 3;

var TypeAheadViewModel = {
	typeAheadText: ko.observable(''),
	searchTextValue: ko.observable(''),
	typeAheadTextFocus : ko.observable(),

	// initial setup of SearchResults
    typeAheadResults: ko.observableArray([]),

	// visible
    typeAheadResultsAvailable: ko.observable(false),
    
    processTypeAheadResults: function(result){
		var self = this;
		self.typeAheadResults.removeAll();
		typeAheadValues = {};
		var products = [];
		var nonproducts = [];
		for (var field in result.suggest) {
			var suggestor = result.suggest[field];
			for (var term in suggestor) {
				var suggestions = suggestor[term].suggestions;
				suggestions.forEach(function(entry) {
					var lowercaseEntry = entry.term.toLowerCase();
					if(typeAheadValues[lowercaseEntry] === undefined){
						typeAheadValues[lowercaseEntry] = lowercaseEntry;
						if(entry.payload === "Product Detail"){
							products.push(lowercaseEntry);
						} else {
							nonproducts.push(lowercaseEntry);
						}
					}
				})
			}
		 }
		nonproducts.sort();
		products.sort();
		if((products.length + nonproducts.length) > resultsDisplaySize){
			if(nonproducts.length < suggestersDisplaySize){
				suggestersDisplaySize = nonproducts.length;
				productsDisplaySize = resultsDisplaySize - suggestersDisplaySize;
			}

			if(products.length < productsDisplaySize){
				productsDisplaySize = products.length;
				suggestersDisplaySize = resultsDisplaySize - productsDisplaySize;
			}
		}
		nonproducts = nonproducts.slice(0,suggestersDisplaySize);
		products = products.slice(0,productsDisplaySize);
		for(var i = 0; i < nonproducts.length ; i++){
			self.typeAheadResults.push(nonproducts[i]);
		}

		for(var i = 0; i < products.length ; i++){
			self.typeAheadResults.push(products[i]);
		}

		self.typeAheadResultsAvailable(self.typeAheadResults().length > 0);
		var typeaheadElement = document.getElementById("typeAhead");
		if(typeaheadElement){
			if(self.typeAheadResultsAvailable){
				$('#typeAhead').attr('style','display: block !important');
			} else {
				$('#typeAhead').attr('style','display: none');
			}
		}
	},

    executeTypeAhead: function () {
        var self = this; // Retain scope of view model
        var serviceUrl = '/ashland-cda/search/suggestservice/';
		serviceUrl = serviceUrl + self.searchTextValue();
		
        $.ajax({
            //url: 'search1.json',
            url: serviceUrl,
            cache: false,
            contentType: 'application/json',
            type: "GET",            
            success: function (result) {
				//process search results
				self.processTypeAheadResults(result);
            },
            error: function (jqXHR, status, error) {
            }
        });
    }
}
$(function () {
	ko.applyBindings(TypeAheadViewModel, document.getElementById('typeAheadSearch'));

	TypeAheadViewModel.typeAheadText = ko.pureComputed({
        read: function () {
        },
        write: function (value) {
			var self = this;
			if(value.trim().length >= 3){
				TypeAheadViewModel.searchTextValue(value);
				TypeAheadViewModel.executeTypeAhead();
			} else {
				TypeAheadViewModel.typeAheadResults.removeAll();
				TypeAheadViewModel.typeAheadResultsAvailable(false);
			}
        },
		owner: this
    });

	if(searchedTerm != undefined){
		document.getElementById("searchBar").value = searchedTerm.replace(/\+/g,' ');
	}

	TypeAheadViewModel.typeAheadTextFocus.subscribe(function(newValue) {
		if (!newValue) {
			setTimeout(function(){ 
				var typeaheadElement = document.getElementById("typeAhead");
				if(typeaheadElement){
					$('#typeAhead').attr('style','display: none');
				}
			}, 200);
		}
    });
})

      resins have an undeniable detection system to verify the integrity of designs wherever corrosion-resistant fiberglass reinforced plastic (FRP) pipes and tanks are specified. The resins have improved manufacturing efficiencies and lower emissions with an innovative styrene suppressant technology.
