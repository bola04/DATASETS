         * Ajaxで値を取得
         */
        $(function(){
            var langType = "japanese";
            // Ajax通信を開始
            $.ajax({
            url: '/timetable/ajax/getScheduleData.php',
                type: 'POST',
                dataType: 'json',
                data : {send_type : langType},
                timeout: 5000
            }).done(function(response) {
                // 取得したDOMデータをセット
                $(response.domestic_timetable).insertAfter( ".domestic_flight_info_navi" );
                $(response.international_timetable).insertAfter( ".international_flight_info_navi" );
            
                var dom_elem_str = response.domestic_timetable.replace(/国内線 /g , "");
                var int_elem_str = response.international_timetable.replace(/国際線 /g , "");
                $(dom_elem_str).insertAfter( ".domestic_flight_info" );
                $(int_elem_str).insertAfter( ".international_flight_info" );
            }).fail(function(XMLHttpRequest, textStatus, errorThrown) {
            })
        });
  </script>

    var now_datetime = new Date();
    var datetime_str = "";
    datetime_str =  + String(now_datetime.getFullYear()) 
                    + String(now_datetime.getMonth() + 1) 
                    + String(now_datetime.getDate()) 
                    + String(now_datetime.getHours()) 
                    + String(now_datetime.getMinutes());

    for (var i = 0; i < weather_img.length ; i++) {
        // 天気画像パスの取得

        weather_img[i].src = src_str + "?" + datetime_str;
