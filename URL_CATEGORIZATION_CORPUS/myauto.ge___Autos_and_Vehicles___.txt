      function search(obj){
        var filters = {}, hasValue, name;
        $.each($(obj).find('input, select'), function(){
            hasValue = !($.trim(this.value) == '' || ($(this).is(':checkbox') && !$(this).prop('checked')));
            if($(this).attr('name') && hasValue){
                if($(this).attr('name').indexOf('[]') >= 0){
                    name = $(this).attr('name').replace('[]', '');
                    filters[name] = filters[name] || [];
                    filters[name].push($(this).val());
                    $(this).removeAttr('name');
                }
            } else {
                $(this).removeAttr('name');
            }
        });
        for(var k in filters){
            var val = typeof filters[k] === 'object' ? filters[k].join(PAR_VALUES_SPLITTER) : filters[k];
            $(obj).append('<input type="hidden" name="' + k + '" value="' + val + '" />');
        };
        //return false;
    }
