   function getCookie(c_name){
					var c_value = document.cookie;
					var c_start = c_value.indexOf(" " + c_name + "=");
					if (c_start == -1) {
						c_start = c_value.indexOf(c_name + "=");
					}
					if (c_start == -1) {
						c_value = null;
						document.getElementById("REMPASS").checked = false;
					}
					else {
					c_start = c_value.indexOf("=", c_start) + 1;
					var c_end = c_value.indexOf(";", c_start);
						if (c_end == -1) {
						c_end = c_value.length;
						}
					c_value = unescape(c_value.substring(c_start,c_end));
					var c_unstart = c_value.indexOf("=")+1;
					var c_unend = c_value.indexOf("\&",c_unstart);
					var c_pwstart = c_value.indexOf("=",c_unstart)+1;
					var c_un = c_value.substring(c_unstart,c_unend);
					var c_pw = c_value.substring(c_pwstart);
					document.getElementById("ID").value = c_un;
					document.getElementById("PASSWORD").value = c_pw;
					document.getElementById("REMPASS").checked = true;
					}
				return c_value;
				}

                  Lawyers for Johnson &amp; Johnson reject claims widely-used baby powder contains asbestos
                 </i>
                </div>
                <div>
                 <div style="text-align: justify;">
                  <span style="background-color: transparent;">
                   A trial for a lawsuit alleging that Johnson &amp; Johnson Baby Powder was responsible for the death of a woman due to her exposure to cancer-causing asbestos began in South Carolina on Monday in the latest case against the healthcare conglomerate and a supplier over their talc-based products. J&amp;J said that its widely-used baby powder never contained asbestos, a known carcinogen linked to mesothelioma.

The case also names as a defendant a local unit of Rite Aid, one of the largest U.S. drugstore chains, which allegedly sold the baby powder used by the woman.
