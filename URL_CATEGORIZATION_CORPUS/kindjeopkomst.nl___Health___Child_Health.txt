// Check
function check_cookie() {

	var name = "cookieConsent=";
	var decodedCookie = decodeURIComponent( document.cookie );
	var ca = decodedCookie.split( ';' );

	for( var i = 0; i < ca.length; i++ ) {

		var c = ca[i];

		while( c.charAt(0) == ' ' ) {
			
			c = c.substring(1);

		}

		if( c.indexOf( name ) == 0 ) {

			return true;

		}

	}

	return false;

}

// Cookieconsent
function cookieConsent(preference) {

	var d = new Date();

	d.setTime( d.getTime() + ( 365 * 24 * 60 * 60 * 1000 ) ); // 365 days

	var expires = "expires="+ d.toUTCString();
	var domain = ".ikenmama.nl";

	document.cookie = "cookieConsent=" + preference + ";" + expires + ";path=/;domain=" + domain;

}

// Get value
function getCookieConsentValue() {

	var cname = 'cookieConsent';
	var name = cname + "=";
	var decodedCookie = decodeURIComponent( document.cookie );
	var ca = decodedCookie.split(';');

	for( var i = 0; i < ca.length; i++ ) {

		var c = ca[i];

		while( c.charAt(0) == ' ' ) {

			c = c.substring(1);

		}

		if( c.indexOf(name) == 0 ) {

			return c.substring( name.length, c.length );
		}

	}

	return "unknown";

}
