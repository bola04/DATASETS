      <script>
       $(document).ready(function(){
            var filters = {audience:"", keyword:"", ptURL:"", sort:"" };

            $("#activitiesFilterDiv").show();
            // ajax filter change
            $("#activitiesFilterDiv select").change(function(){

                var sF = $("option:selected", this).val();
                var category = $(this).attr("name");
                filters[category] = sF;

                if(filters["keyword"] != "" ) {
                    if(filters["audience"] != "" ){
                        if(filters["ptURL"] != "" ) {
                            window.location.href = "/topic?t="+encodeURI(filters["keyword"])+"&a="+encodeURI(filters["audience"])+"&pt="+encodeURI(filters["ptURL"]);
                        } else {
                            window.location.href = "/topic?t="+encodeURI(filters["keyword"])+"&a="+encodeURI(filters["audience"]);
                        }
                    } else if(filters["ptURL"] != "" ) {
                        if(filters["audience"] != "" ) {
                            window.location.href = "/topic?t="+encodeURI(filters["keyword"])+"&pt="+encodeURI(filters["ptURL"])+"&a="+encodeURI(filters["audience"]);
                        } else {
                            window.location.href = "/topic?t="+encodeURI(filters["keyword"])+"&pt="+encodeURI(filters["ptURL"]);
                        }
                    } else {
                        window.location.href = "/topic?t="+encodeURI(filters["keyword"]);
                    }
                } else if(filters["audience"] != "" ) {
                    if(filters["ptURL"] != "" ) {
                        window.location.href = "/topic?a="+encodeURI(filters["audience"])+"&pt="+encodeURI(filters["ptURL"]);
                    } else {
                        window.location.href = "/topic?a="+encodeURI(filters["audience"]);
                    }
                } else if(filters["ptURL"] != "" ) {
                    window.location.href = "/topic?pt="+encodeURI(filters["ptURL"]);
                } else {
                    window.location.href = "/topic";
                }
            });


            // load more button
            $(".load-more").click(function(){

                if ($(".load-more").attr("count")) {
                    var count = $(".load-more").attr("count");
                } else {
                    var count = 4;
                }

                var sF = $("option:selected", $("#activitiesFilterDiv select")).val();
                var category = $("#activitiesFilterDiv select").attr("name");
                filters[category] = sF;

                var pN = $("#ajaxContent").attr("pnumber");

                $.ajax({
                    url: "/ajaxTitan.php",
                    data: "ajax=titan&act=loadMore&count="+count+"&pNumber="+pN+"&filterJSON="+JSON.stringify(filters)+"&t=",
                    type: "POST",
                    cache: false,
                    success: function(data){

                        var ajaxObj = JSON.parse(data);
                        $("#ajaxContent").attr("pnumber",ajaxObj.pageNumber);
                        if (ajaxObj.data) {
                            $("#ajaxContent").append(ajaxObj.data);
                        } else {
                            $(".load-more").remove();
                        }

                    }
                });

            });
        });
      </script>
