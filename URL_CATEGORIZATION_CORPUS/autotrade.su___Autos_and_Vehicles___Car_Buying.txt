        <script>
            $().ready(function () {

                $('#button--article-clear').on('click',function () {
                    $('#input--oem').val('').focus();
                });

                $('#btn--oem-search').on('click',function (e) {
                    var rgx = /^[a-z0-9\-\s\/]+$/i;
                    var input_value = $('#input--oem').val();
                    if(false === rgx.test(input_value)) {
                        $('#fval--article').removeClass('d-none');
                        e.stopPropagation();
                    }
                    input_value = input_value.replace('/','');
                    var $form = $(this).closest('form');
                    var action = $form.attr('action');
                    action = action.replace(new RegExp('/([^/]*)$'), '/' + input_value);
                    $form.attr('action', action);
                });

                $('#search-form--button').on('click',function (e) {
                    var s_part = $('select[name="part"]').val();
                    if( s_part === undefined || s_part == 0 || s_part === null){
                        $('#validate--pt').addClass('has-danger');
                        $('#validate--pt-txt').removeClass('d-none');
                        e.preventDefault();
                    }

                    var model = $('#models-select').val();
                    if(model === undefined || model == 0 || model === null){
                        $('#fval--model').removeClass('d-none');
                        e.preventDefault();
                    }

                    var mark = $('#marks-select').val();
                    if(mark === undefined || mark == 0 || mark === null){
                        $('#fval--mark').removeClass('d-none');
                        e.preventDefault();
                    }
                });

                AppForms.init();
                            });
