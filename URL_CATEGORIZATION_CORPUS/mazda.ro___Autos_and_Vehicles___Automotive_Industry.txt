       function playVideo(self, pathString, playtag, stoptag, vttUrl) {
        window.pathArray = pathString.split('|');
        var videoObject = [];
        for (var i = 0, len = pathArray.length; i < len; i++) {
            videoObject.push({ 'file': pathArray[i] });
        }

        var getPlayerSize = function(){
            var winWidth = $('.video_container').width(),
                baseWidth = 468,
                baseHeight = 263;

            return {w: winWidth, h: Math.round((baseHeight / baseWidth) * winWidth)};
        };

        var size = getPlayerSize();

        function isPlayed() {
          return self.getAttribute('data-asset-auto');
        }

        if (videoObject.length) {
            jwplayer('player_95870c7c-e3ac-41b0-955d-bcece5c94a0d')
                .setup({
                    'modes': [{ type: 'flash', src: '/videoplayer/player.swf' }, { type: 'html5' }],
                    'levels': videoObject,
                    'skin': '/videoplayer/skins/mazda/mazda.xml',
                    'stretching': 'fill',
                    'width': size.w,
                    'height': size.h,
                    'autostart': isPlayed(),
                    'plugins': {

                            'file': vttUrl,
                            'default': vttUrl !== '' ? true : false,
                        }
                    },

                        },
                        onComplete: function () {
                          s3_log(stoptag);
                          destroy(videoObject, self);
                          self.setAttribute('data-asset-auto', 'false');
                        }
                    }
                })
            }
        }

    var getLiHeight = function () {
        var maxLiHeight = 0;

        $('.video_category ul li').each(function () {
            if (maxLiHeight < $(this).find('.video_column_preview_desc').height()) maxLiHeight = $(this).find('.video_column_preview_desc').height();
        });

        liHeight = maxLiHeight + 150;

        return liHeight;
    };

    function destroy(videoObject, self) {
      jwplayer('player_95870c7c-e3ac-41b0-955d-bcece5c94a0d').remove();
      setTimeout(function() {
        $(self).trigger('click');
      }, 1000);
    }

    function handleDescriptionLengths(orig_height) {

        $('.video_category li').each(function () {

            var desc = $('.description', $(this)),
                full_desc = desc.text(),
                short_desc = '',
                limit = 60;

            if (full_desc.length >= limit) {

                desc.html('<span class="full_desc" style="display:none">' + full_desc + '</span>');

                short_desc = full_desc
                    .trim()
                    .substring(0, limit)
                    .split(' ')
                    .slice(0, -1)
                    .join(' ') + '...';

                desc.append('<span class="short_desc">' + short_desc + '</span><br><a href="#" class="text_toggle" onclick="showDesc(this, \'more\', ' + orig_height + '); return false;">Arată mai mult</a>');
            }

        });

    }

    function showDesc(el, type, orig_height) {

        var el = $(el),
            parent = el.parent(),
            li = el.parents('li'),
            full_desc = $('.full_desc', parent),
            short_desc = $('.short_desc', parent),
            link = $('.text_toggle', parent),
            rowCount = li.data('row'),
            minHeight = li.data('height'),
            state = link.attr('data-state');

        if (typeof(state) === 'undefined' || state === 'more') {
            short_desc.hide();
            full_desc.show(100);

            $('.video_category li[data-row="' + rowCount + '"]').animate({ 'height': orig_height + 'px' }, 400);
            link.attr('data-state', 'less');
            link.attr('onclick', 'showDesc(this, \'less\', ' + orig_height + '); return false;').text('Arată mai puţin');
        } else {
            full_desc.hide();
            short_desc.show(100);

            if (!$('.full_desc:visible', '.video_category li[data-row="' + rowCount + '"]').length) {
                $('.video_category li[data-row="' + rowCount + '"]').animate({ 'height': minHeight + 'px' }, 200);
            }
            link.attr('data-state', 'more');
            link.attr('onclick', 'showDesc(this, \'more\', ' + orig_height + '); return false;').text('Arată mai mult');
        }

    }
