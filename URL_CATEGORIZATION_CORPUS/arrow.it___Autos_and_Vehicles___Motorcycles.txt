                    if (parseInt(jsonParameter[0].TipoAgente) != 3) {
                        var rePdf = new RegExp("_" + String(jsonParameter[0].Codice) + "_", "g");

                                var rePdf = new RegExp("_0920\\d{6}_", "g");
                                if (String(item).match(rePdf) != null)

                            if (parseInt(jsonParameter[0].TipoIspettore) == 3)//Ispettore Tutto
                            {
                                filesArray = item;
                            }

                            var rePdf = new RegExp("_" + String(jsonParameter[0].Codice) + ".", "g");

                                filesArray = item;
                        }
                    }

                    if (filesArray != null) {
                        var dataTemp = JsonFTP[el].Data;

                        dataTemp = dataTemp.replace(")/", "");

                    }
                }

                if (listOfFtpFile != "") {
                    $("#tabellaFileFTP").html("<div><h3>Area Documenti</h3></div>");
                    $("#tabellaFileFTP").append(listOfFtpFile)
                }



                if (JsonFTP.lenght == 0 && JsonPublic.lenght == 0 && JsonPrivate.lenght == 0) {
