   $(document).ready(function() {

         var insertSpeedbumpText = function(link, target) {
             // get content

             // move the currently attached disclaimer text element
             $('#speedbump .modal-body div.disclaimer-text').appendTo('#speedbumps');

             // get the speedbump name
             var classes = $(target).attr("class").split(" ");
             var name = "";

             if (classes.length == 2) {
                 name = classes[1];
             }

             // add speedbump disclaimer text element to modal
             $('#speedbump .modal-body').prepend($('#speedbump-' + name));

             $("#leave_this_page").attr('href', link.attr('href'));
             $('#speedbump').modal('show');
         };
         

		// display the speedbump when link is clicked
		$('body').on('click', 'span.speedbump a, a.speedbump, .speedbump a', function(e){
		    e.preventDefault();
            var link = $(this);
		    var target = link.closest('.speedbump');

		    insertSpeedbumpText(link, target);
		});

		$("a .speedbump").click(function (event){
			event.preventDefault();
			var link = $(this).closest("a");

            insertSpeedbumpText(link, this);
		});
		
		// Hide Modal
		$(document).on("click", ".hideSpeedbump", function(event) { 
			$('#speedbump').modal('toggle');
		});
    });
