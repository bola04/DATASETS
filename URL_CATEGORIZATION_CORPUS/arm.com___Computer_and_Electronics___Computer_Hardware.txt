            var _isExpanded = false;


            /**
             * Handles the form submission
             * param  {Event}   e   The dispatched event
             */
            function submitForm(e) {
                e.preventDefault();
                var searchPageUrl = searchForm.getAttribute('data-searchresultspageurl');
                var searchQueryString = [];
                for (var i = 0; i < searchForm.elements.length; i++) {
                    var el = searchForm.elements[i];
                    if (el.name) searchQueryString.push(encodeURIComponent(el.name) + '=' + encodeURIComponent(el.value));
                }
                window.location = searchPageUrl + '?' + searchQueryString.join('&');
            }

            /**
             * Handles user interaction events outside the search input
             * param  {Event}   e   The dispatched event
             */
            function onSearchFocusOut(e) {
                if (e.target !== searchInput) toggleSearch();
            }

            /**
             * Toggles the search component
             */
            function toggleSearch() {
                _isExpanded = !_isExpanded;

                if (_isExpanded) {
                    window.addEventListener('click', onSearchFocusOut);
                    window.addEventListener('touch', onSearchFocusOut);
                    searchElement.classList.add(EXPANDED_CLASS);
                    searchInput.focus();

                } else {
                    window.removeEventListener('click', onSearchFocusOut);
                    window.removeEventListener('touch', onSearchFocusOut);
                    searchElement.classList.remove(EXPANDED_CLASS);
                    searchForm.reset();
                    searchOpen.focus();
                }

                // Let other components know that the search was toggled
                var toggleEvent = document.createEvent('Event');
                toggleEvent.initEvent('search.toggle', false, false);
                document.dispatchEvent(toggleEvent);
            }

            function onSearchOpen(e) {
                e.preventDefault();
                e.stopPropagation();
                if (_isExpanded) return submitForm(e);
                toggleSearch();
            }

            /**
             * Handles search close events
             * param  {Event}   e   The dispatched event
             */
            function onSearchClose(e) {
                e.preventDefault();
                e.stopPropagation();
                toggleSearch();
