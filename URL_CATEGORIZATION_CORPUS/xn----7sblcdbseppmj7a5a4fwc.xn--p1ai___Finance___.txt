{
	var page_array = page_category.split(',');

function checkCity()
{
	var geolocation = ymaps.geolocation;
	var region = '';
	var city = '';
	if (typeof geolocation !== "undefined") {
		region = geolocation.region;
		city = geolocation.city;

		{
			geolocation.region = 'Москва';
		}

			if(geolocation.region == 'Москва и Московская область')

				geolocation.region = 'Московская область';
			}
		}
		if(geolocation.city == 'Санкт-Петербург')
		{
			geolocation.region = 'Санкт-Петербург';
		}
		else
		{
			if(geolocation.region == 'Санкт-Петербург и Ленинградская область')
			{
				geolocation.region = 'Ленинградская область';
			}
		}
		if(geolocation.country != "Россия")
		{
			geolocation.region = "Другой регион";
			geolocation.city = "Другой город";
		}
		yandex_city = geolocation.city;
		yandex_region = geolocation.region;
	} else {
		yandex_region = u.region.name_ru;

		if(c2key == user_region) found2 = true;

		user_region = "Другой регион";
		user_city = "Другой город";

	document.getElementById('region').innerHTML = sstring;
	document.getElementById('region').value=user_region;
