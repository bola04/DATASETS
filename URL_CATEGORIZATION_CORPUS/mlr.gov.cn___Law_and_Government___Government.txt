    function funLeft(){
      var bwidth = document.documentElement.clientWidth;
      var lwidth = bwidth-102;
      var icon_left = document.getElementById("site_right_icon");
      icon_left.style.right=lwidth+'px';
      params_rightIcon.right = lwidth;
    }

    var params_rightIcon = {
      right: 0,
      bottom: -15,
      currentX: 0,
      currentY: 0,
      flag: false
    };
    //获取相关CSS属性
    var getCss = function(o,key){
      return o.currentStyle? o.currentStyle[key] : document.defaultView.getComputedStyle(o,false)[key];
    };

    //拖拽的实现
    var startDrag = function(bar, target, callback){
      if(getCss(target, "right") !== "auto"){
        params_rightIcon.left = getCss(target, "right");
      }
      if(getCss(target, "bottom") !== "auto"){
        params_rightIcon.top = getCss(target, "bottom");
      }
      //o是移动对象
      bar.onmousedown = function(event){
        target.style.position="fixed";
        params_rightIcon.flag = true;
        if(!event){
          event = window.event;
          //防止IE文字选中
          bar.onselectstart = function(){
            return false;
          }
        }
        var e = event;
        params_rightIcon.currentX = e.clientX;
        params_rightIcon.currentY = e.clientY;
      };
      document.onmouseup = function(){
        target.style.position="fixed";
        params_rightIcon.flag = false;
        if(getCss(target, "right") !== "auto"){
          params_rightIcon.right = getCss(target, "right");
        }
        if(getCss(target, "bottom") !== "auto"){
          params_rightIcon.bottom = getCss(target, "bottom");
        }
      };
      document.onmousemove = function(event){
        var e = event ? event: window.event;
        if(params_rightIcon.flag){
          var nowX = e.clientX, nowY = e.clientY;
          var disX = nowX - params_rightIcon.currentX, disY = nowY - params_rightIcon.currentY;
          target.style.right = parseInt(params_rightIcon.right) - disX + "px";
          target.style.bottom = parseInt(params_rightIcon.bottom) - disY + "px";
        }

        if (typeof callback == "function") {
          callback(parseInt(params_rightIcon.right) + disX, parseInt(params_rightIcon.top) + disY);
        }
      }
    };
