			return $('#pref_roman').val() !== '' && $('#city_roman').val() !== '';
		}

		function enableSearch() {
			$('[data-button="submit"] button').each(function(i, el) {
				$(el).prop('disabled', false);
			});
		}

		function disableSearch() {
			$('[data-button="submit"] button').each(function(i, el) {
				$(el).prop('disabled', true);
			});
		}

		function setAction(callback) {
			var action = [];
			if ($('#pref_roman').val() !== '') action.push($('#pref_roman').val());
			if ($('#city_roman').val() !== '') action.push($('#city_roman').val());
			$('#search-form1').attr('action', '/' + action.join('/') + '/');

			if (typeof callback === 'function') {
				callback() ? enableSearch() : disableSearch();
			}
		}


		$('#pref_roman, #city_roman').on('change', function() {
			setAction(canSearch);
		});

		var $formCategory = $('[data-toggle="formCategory"]');

			$('.searchmenu-inner').animate({opacity: 'toggle', height: 'toggle'}, 'fast', 'swing', function() {

		});

	});
})();
