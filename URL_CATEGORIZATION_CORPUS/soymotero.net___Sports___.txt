    <script type="text/javascript">
     $(document).ready(botones_home);
  
  function botones_home(){
    $('.especiales').click(cambio_carrusel_especiales);
    $('.pruebas').click(cambio_carrusel_pruebas);
    $('.equipamientos').click(cambio_carrusel_equipamientos);
    
  }
  
  function ocultar_todos(){
    $('.especiales').removeClass('selected');
    $('.pruebas').removeClass('selected');
    $('.equipamientos').removeClass('selected');
    $('.bloque_especiales').addClass('hide');
    $('.bloque_pruebas').addClass('hide');
    $('.bloque_equipamientos').addClass('hide');
  }
  
  
  function cambio_carrusel_especiales(){
    if(!$(this).hasClass('selected')){
      ocultar_todos();
      $('.especiales').addClass('selected');
      $('.bloque_especiales').removeClass('hide');
    }
  }
  
  function cambio_carrusel_pruebas(){
    if(!$(this).hasClass('selected')){
      ocultar_todos();
      $('.pruebas').addClass('selected');
      $('.bloque_pruebas').removeClass('hide');
    }
  }
  
  function cambio_carrusel_equipamientos(){
    if(!$(this).hasClass('selected')){
      ocultar_todos();
      $('.equipamientos').addClass('selected');
      $('.bloque_equipamientos').removeClass('hide');
    }
  }
