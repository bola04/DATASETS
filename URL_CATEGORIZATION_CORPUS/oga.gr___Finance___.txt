function show()
{
	if (document.getElementById("mailList").style.display == "none")
	{
		document.getElementById("mailList").style.display = "block";
	}
	else
	{
		document.getElementById("mailList").style.display = "none";
	}
}

function check_request()
{
	if(document.getElementById('strMail').value=='')
		alert("Παρακαλούμε, εισάγετε το email σας!");
	else	
		makeRequest('mailList', 'http://www.oga.gr/newsletter/newsletter.php?name='+escape(document.getElementById('strMail').value)+'&id='+escape(document.getElementById('id').value));
}

function check_searchrequest()
{
	if(document.getElementById('txtsearch').value.length < 3)
		alert("Παρακαλούμε, εισάγετε τουλάχιστον 3 χαρακτήρες!!");
	else	
		makeRequest('contentdiv', 'http://www.oga.gr/search.php?txtsearch='+escape(document.getElementById('txtsearch').value)+'&sel_search='+escape(document.getElementById('sel_search').value));
}
