   var curIndex = 0;
	var blocks2show = ['first', 'second', 'third', 'fourth', 'fifth', 'sixth'];
	var timeStep = 30000;
	var playSlideShow = 1;
	var interval = setInterval(selectNext, timeStep);
	

	function selectBlock2Show(i) {
		if (i >= blocks2show.length) { 
			i = 0;
		}
		if (i < 0) {
			i = blocks2show.length - 1;
		}
		for(var j = 0; j < blocks2show.length; j++) {
		    if (j !== i ) {
		 	document.getElementById(blocks2show[j]).style.display = "none";
			document.getElementById("table-selector").rows[0].cells[j].classList.remove("active");  
		    } else {
			document.getElementById(blocks2show[j]).style.display = "";
			document.getElementById("table-selector").rows[0].cells[j].classList.add("active"); 
		    }
		}
		curIndex = i;
		clearInterval(interval);
		setTimeout(timeStep);
		if (playSlideShow == 1) {
			interval = setInterval(selectNext, timeStep);
		}
	}
	
	function selectNext() {
		 curIndex = curIndex + 1;
		 selectBlock2Show(curIndex); 
	}

	function selectPrevious() {
		 curIndex = curIndex - 1;
		 selectBlock2Show(curIndex); 
	}
	function stopSlideShow() {
		 clearInterval(interval);
	}
	function restartSlideShow() {
		 selectBlock2Show(curIndex);
	}
	function slideShow() {
		if (playSlideShow == 1) {
			playSlideShow = 0;
			stopSlideShow();
			document.getElementById("pauseTab").innerHTML = ">";
		} else {
			playSlideShow = 1;
			restartSlideShow();
			document.getElementById("pauseTab").innerHTML = "||";
		}
	}

                 <a href="http://www3.interscience.wiley.com/cgi-bin/fulltext/122523533/HTMLSTART">
                  [1]
                 </a>
                 ,
                 <a href="https://onlinelibrary.wiley.com/doi/abs/10.1002/prot.25005">
                  [2]
                 </a>
                 . 
			The latter paper also contains explanations of data handling procedures 
			and guidelines for navigating the data presented on this website.
                 <p>
                  Some of the best performing methods are implemented as
                  <a href="http://www.predictioncenter.org/index.cgi?page=links">
                   fully automated servers
                  </a>
                  and therefore can be used by public for protein structure modeling.
