        function listarNovedades() {
            $('#slider').nivoSlider();
            $.ajax({
                type: "POST",
                url: services + "ListarNovedadesEspanol.aspx",
                success: listaOK,
                error: listaNOOK
            });
        }
        function listarBanners() {
            $.ajax({
                type: "POST",
                url: services + "ListarBannersEspanol.aspx",
                success: listaBannersOK,
                error: listaBannersNOOK,
                complete: listarNovedades
            });

        function listaBannersOK(data) {
            var objBanners = jQuery.parseJSON(data);
            for (i = 0; i < objBanners.length; i++) {
                if (objBanners[i].id_tipo_banner == 1) {
                    agregarElementoBanner(objBanners[i]);
                }
                else if (objBanners[i].id_tipo_banner == 2) {
                    agregarBannerDerecha(objBanners[i]);
                }
            }
        }
        function agregarBannerDerecha(banner) {
            var divconte = document.getElementById("bannerDerecha");

            var link;
            var img = document.createElement("img");
            img.setAttribute("src", repos + banner.url_imagen);
            img.setAttribute("width", "300px");
            img.className = "imgDerecha"
            if (banner.link != "") {
                link = document.createElement("a");
                link.setAttribute("href", banner.link);
                link.setAttribute("target", "_blank");

                link.appendChild(img);
                divconte.appendChild(link);
            }
            else {
                divconte.appendChild(img);
            }

        }
        function listaBannersNOOK() {
            console.log("error banner");
        }
        function agregarElementoBanner(objBanner) {

            var div = document.getElementById("slider");

            var link = document.createElement("a");
            link.setAttribute("href", objBanner.link);
            link.setAttribute("target", "_blank");

            var img = document.createElement("img");
            img.setAttribute("src", repos + objBanner.url_imagen);
            img.setAttribute("data-thumb", repos + objBanner.url_imagen);
            img.setAttribute("alt", objBanner.nombre);
            img.setAttribute("data-transition", "fade");

            link.appendChild(img);

            div.appendChild(link);
        }

            var obj = jQuery.parseJSON(data);
            objNovedades = obj;
            mostrarNovedades();
        }
        function mostrarNovedades() {
            var limit = numNovedadesPagina > objNovedades.length ? objNovedades.length : numNovedadesPagina;
            for (i = 0; i < limit; i++) {
                agregarElemento(objNovedades[i]);
            }
            if (numNovedadesPagina < objNovedades.length) {
                crearVerMas();
            }            
        }

            alert("Error al cargar novedades");

            $("#Novedades").append(img);
            $("#Novedades").append(tit);
            $("#Novedades").append(des);
            $("#Novedades").append(enl);
            $("#Novedades").append(fin);
        }
        function quitarVerMas() {
            var elemento = document.getElementById("Novedades");
            if (elemento.contains(div)) {
                elemento.removeChild(div);
            }
        }

            var elemento = document.getElementById("Novedades");
            elemento.appendChild(div);
