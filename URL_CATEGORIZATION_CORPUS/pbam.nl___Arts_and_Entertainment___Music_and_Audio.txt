	function doSSOn_mod_rbg_162() {
		try {
			jQuery('#mod_rbg_162 input.login.button').hide();
			jQuery('#mod_rbg_162 img.ajax-loading-login').show();
			jQuery.ajax({
				url: "https://www.musidesk.nl/index.php?option=com_rbg&format=json&layout=mod_rbg&module_id=162&action=bibsys_login",
				type: "POST",
				cache: false,
				async: true,
				dataType: "json",
				data: jQuery('#mod_rbg_162 .mbLogin input').serialize(),
				success: function(data, textStatus, XHR) {
					var message = data.mesg;
											switch(data.stat) {
							case "1":
							  message = "Lener onbekend of leeg wachtwoord";
							  break;
							case "2":
							  message = "Gebruikersnaam of wachtwoord onjuist";
							  break;
							case "3":
							  message = "Aantal (meld_tel) overschreden";
							  break;
							case "5":
							  message = "Gebruik uw wachtwoord - niet uw PIN - om in te loggen";
							  break;
							case "6":
							  message = "Blokkade of abon verlopen";
							  break;
							case "7":
							  message = "Niet toegestaan in deze bibliotheek";
							  break;
							case "8":
							  message = "IP-nummer blokkade (aantal overschreden)";
							  break;
							case "9":
							  message = "Geen sessie/abon/instantie etc";
							  break;
							default:
							  message = "Onbekende fout. Probeer het later nog eens.";
						}
										if (data.stat == "0") {
						jQuery('#mod_rbg_162 .username').removeClass("error");
						jQuery('#mod_rbg_162 .password').removeClass("error");
						var scherm_meldingen = 0;
						if (typeof data.scherm_meldingen != "undefined") {
							jQuery.each(data.scherm_meldingen, function(key, melding) {	
								if (melding.id=='webcat.eerst_wachtwoord_aanmaken') {
									scherm_meldingen = 1;
									var redir_passw = jQuery('#mod_rbg_162 .redir_passw').attr('value');
									if (redir_passw != '' && redir_passw != "undefined") {
										document.location = redir_passw;
									} else {
										alert(melding.text);
									}
									return false; // stop each
								}
							});
						}
						if (scherm_meldingen == 0) {
							document.location = data.redir;
						}
					} else {
						jQuery('#mod_rbg_162 .password').val("");
						if (data.stat == "1") {
							jQuery('#mod_rbg_162 .username').addClass("error");
							jQuery('#mod_rbg_162 .password').addClass("error");
						} else if (data.stat == "2") {
							jQuery('#mod_rbg_162 .username').removeClass("error");
							jQuery('#mod_rbg_162 .password').addClass("error");
						} else {
							jQuery('#mod_rbg_162 .username').removeClass("error");
							jQuery('#mod_rbg_162 .password').removeClass("error");
						}
						jQuery('#mod_rbg_162 .error-popup').show().find("p").text(message);
						var t = setTimeout(function () {
							jQuery('#mod_rbg_162 .error-popup').hide()
						}, 5000);
						jQuery('#mod_rbg_162 input.login.button').show();
					}
					//RBG.log2console("Ajax succes; status: " + data.fullname);
				},
				error: function(XHR, textStatus, errorThrown) {
					RBG.log2console("Error with login: " + textStatus + "/" + errorThrown);
					RBG.log2console("Received XML: " + XHR.responseText + "\n\n" + "[" + XHR.responseXML + "]");
				},
				complete: function(XHR, textStatus) {
					//RBG.log2console("Ajax call complete; status: " + textStatus);
					jQuery('#mod_rbg_162 img.ajax-loading-login').hide();
				}
			});
		} catch (err) {
			RBG.log2console("Try/Catch Error: " + err);
		}
	};

	function doSSOff_mod_rbg_162() {
		try {
			jQuery('#mod_rbg_162 .logout.uitloggen').hide();
			jQuery('#mod_rbg_162 img.ajax-loading-logout').show();
			jQuery.ajax({
				url: "https://www.musidesk.nl/index.php?option=com_rbg&format=json&layout=mod_rbg&module_id=162&action=bibsys_logout",
				type: "POST",
				cache: false,
				async: true,
				dataType: "json",
				data: jQuery('#mod_rbg_162 .mbLogoff input').serialize(),
				success: function(data, textStatus, XHR) {
					window.location.href = data.redir;
				},
				error: function(XHR, textStatus, errorThrown) {
					RBG.log2console("Error with logout: " + textStatus + " / " + errorThrown);
					RBG.log2console("Received: " + XHR.responseText + " [" + XHR.responseXML + "]");
				},
				complete: function(XHR, textStatus) {
					//RBG.log2console("Ajax call complete; status: " + textStatus);
					jQuery('#mod_rbg_162 img.ajax-loading-logout').hide();
				}
			});
		} catch (err) {
			RBG.log2console("Try/Catch Error: " + err);
		}
	};
