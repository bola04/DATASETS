     * @param  {string} cookie Raw body of the cookie (see getCookie)
     * @return {object}        The cookie XML attributes as a JS object
     */
    function parseCookieIndi(cookie) {
        res = {};
        params = [];
        cookie = cleanArray(cookie.replaceAll('=', '·=·').split(/"| |·/));
        cookie.forEach(function(s) {
            if (s == '=') {
                params.push(cookie[cookie.indexOf(s) - 1]);
                delete cookie[cookie.indexOf(s)];
            }
        })
        params.forEach(function(p) {
            if (cookie.indexOf(p) != -1) {
                res[p] = cookie[cookie.indexOf(p) + 2].toLowerCase();
            }
        })
        return res;
