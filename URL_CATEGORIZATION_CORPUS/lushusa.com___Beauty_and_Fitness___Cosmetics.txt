        var timerContainer = $("#reserveCart");

function time_remaining(endtime){
	var t = Date.parse(endtime) - Date.parse(new Date());
	var seconds = Math.floor( (t/1000) % 60 );
	var minutes = Math.floor( (t/1000/60) % 60 );
	var hours = Math.floor( (t/(1000*60*60)) % 24 );
	var days = Math.floor( t/(1000*60*60*24) );
	return {'total':t, 'days':days, 'hours':hours, 'minutes':minutes, 'seconds':seconds};
}


 
function run_clock(id,idcount,endtime){
	
	var clockContainer = $('#'+id); 
	timerContainer.show();
	var clock = $('.'+idcount); 

	function update_clock(){
		var t = time_remaining(endtime);
		if(t.seconds < 10){ var preP = '0'; }else{var preP = '';}
		if(t.minutes < 1){var preM = '0';}else{var preM = '';}
		clock.text(preM+t.minutes+':'+preP+t.seconds);

		if( t.minutes > 4){
			timerContainer.removeClass('warn5');
			timerContainer.removeClass('warn6');
		}
		if( (t.minutes < 5 && t.minutes >= 1) && t.seconds <= 59 ){ 
			timerContainer.addClass('warn5');
			timerContainer.removeClass('warn6');
		}
		if( t.minutes < 1 && t.seconds <= 59  && !timerContainer.hasClass('warn6') ){ 
			timerContainer.removeClass('warn5');
			timerContainer.addClass('warn6');
		}
		if( t.minutes == 1 && t.seconds == 0 ){
			cartSession.extendDialog(); 
		}
		
		if(t.total<=0){ 
			window.clearInterval(window.timeInterval);
			cartSession.releaseDialog(); 
			timerContainer.hide();
		}
	}
	update_clock(); 
	window.timeInterval = setInterval(update_clock,1000);
}
