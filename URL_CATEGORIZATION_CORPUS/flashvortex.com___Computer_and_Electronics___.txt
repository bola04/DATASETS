    $( document ).ready( function() {
   var totalSamples = 42;
   var selectedSampleId = 0;
   // Load first sample
   loadSample( selectedSampleId );
   // If previous button is clicked
   $( '.previous' ).click( function() {
      if ( selectedSampleId > 0 ) { 
         selectedSampleId--;
      } else if ( selectedSampleId == 0 ) { 
         selectedSampleId = totalSamples - 1;
      }
      loadSample( selectedSampleId );
   });
   // If next button is clicked
   $( '.next' ).click( function() {
      if ( selectedSampleId < ( totalSamples - 1 ) ) {
         selectedSampleId++;
      } else if ( selectedSampleId == ( totalSamples - 1 ) ) {
         selectedSampleId = 0;
      }
      loadSample( selectedSampleId );
   });

         $( "#chrome" ).fadeTo( "fast", 0.99 ).fadeTo( "fast", 1 ); // Google Chrome bug that doesn't display sample
      });
   }
