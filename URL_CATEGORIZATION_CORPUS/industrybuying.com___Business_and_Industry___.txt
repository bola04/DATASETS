    function search_analytics(){

        var ref = document.referrer;
        if (ref.indexOf('/search/') <= -1)
            return -1;

        urlSplit = document.referrer.split('?');

        if (urlSplit.length < 2)
            return -1;


        params = urlSplit[1].split('&');

        var q = "";
        var cat = "";

        for (var i = 0; i < params.length; i++) {
            var name = params[i].split('=');
            if (name[0] == 'q')
                q = name[1];
            if (name[0] == 'cat')
                cat = name[1];
        }
        if (!q)
            return -1;

        if (q.indexOf('+') > -1)
            q= q.replace(/\+/gi, ' ');

            type: 'POST',
            url: url ,

                query: q,
                cat_id: cat,
                csrfmiddlewaretoken: getCookie('csrftoken')
            },
