function measurePageTiming(timeEvent){
    //can't do this on some older browsers.
    if(window.performance == undefined){
        return;
    }
    var page_load_time = null;
    
    var performance_data = window.performance.timing;
    if(performance_data[timeEvent] > 0){
        page_load_time = performance_data[timeEvent] - performance_data.navigationStart;
    }
    else{
        window.setTimeout(function(){
            measurePageTiming(timeEvent);
        }, 300);
        return;
    }

    //translate timeEvent to user readable
    var event_var = "not set";
    switch(timeEvent){
        case 'loadEventEnd':
            event_var = "Page Load Time";
            break;
        case 'domComplete':
            event_var = "Dom Complete Time";
            break;
        case 'responseEnd':
            event_var = "Server Response Time";
            break;
    }    

    //console.log(event_var+': '+page_load_time);

    //Send the page load time to GA
    //filter out errouneous timings greater than 100 seconds
    if( page_load_time > 0 && page_load_time < 100000 && typeof( dataLayer ) != 'undefined' ) {
        dataLayer.push({
              'event': 'UserTimePush',
              'eventVar': event_var,          
              'eventCategory': 'Page Timings v1',
              'eventValue': page_load_time,

            });    
    }
}

//Handle tracking promo views within viewport
var promo_counter = 0;
var promo_views_sent = [];


$(window).load(function(){

    window.setTimeout(trackPromoView, 10);
    //send promotions when they're scrolled into view.
    $(window).scrolled(trackPromoView);

    //track products initially loaded
    window.setTimeout(trackProductsInView, 1000);

    //send products when they're scrolled into view.
    $(window).scrolled(function() {
       trackProductsInView();
    });    

});

function trackPromoView(){
    //send promotions in view on page load
    $('.cbox:in-viewport').each(function(index, element){
                    //send promotion view to GA
            var promotion_element = $(element);
            if(!promotion_element.is(":visible")){        
                return false;
            }
            var map = promotion_element.next();
            getAndSendPromotion(map);
    });    
}

function getAndSendPromotion(map, index){
    
    var name = map.data('name');
    if(name == undefined){
        return false;
    }
    //if we're desktop, don't get mobile promos
    if(!isMobile()){
        if(name.indexOf('Mobile HP:') > -1 || name.indexOf('Mobile:') > -1){
            return false;
        }
    }
    else{
        //if mobile, only get mobile promos
        if(name.indexOf('Mobile HP:') < 0 && name.indexOf('Mobile:') < 0){
            return false;
        }   
    }    
    var id = map.data('id');
    var index = index != undefined ? "."+index : promo_counter++;
    var creative = map.data('creative');
    //only send that sucker if it's never been sent before.
    if(!promo_views_sent[id]){
        //console.log("sending promo: " + name);
        sendPromotionView(index, id, name, creative);
        promo_views_sent[id] = true;
    }            

}

var product_impression_queue = [];
var product_sent_registry = [];

function trackProductsInView(){

    $('.product-tile:in-viewport').each(function(index, element){
        //send promotion view to GA
        //console.log(element);
        product = window.products_for_ga[$(element).attr('data-product-idx')];
        product.position = $(element).attr('data-product-position');
        product.list = window.list_title;
        queueProductImpression(product);

    }).promise().done(sendAndClearProductImpressions('solr'));

    $('.product-recommendation-tile:in-viewport').each(function(index, element){
        product = window.product_recommendations_for_ga[$(element).attr('data-product-idx')];
        product.position = $(element).attr('data-product-position');
        product.list = 'Recommendations For: ' + window.recommendation_list_name;
        product.idx = product.id;
        queueProductImpression(product);
    }).promise().done(sendAndClearProductImpressions);
}

function queueProductImpression(product_obj){
    if(product_sent_registry[product_obj.idx] != undefined && product_sent_registry[product_obj.idx] == true){
        return;
    }   

    product_sent_registry[product_obj.idx] = true;
    product_impression_queue.push(product_obj); 
}

function sendAndClearProductImpressions(format){
    if(product_impression_queue.length > 0){
        sendProductListView(product_impression_queue, format);
        product_impression_queue = [];
    }

}

//Send Cart Add Event to Google Analytics
function sendProductCartAdd(order_line, redirect){

    var list = getListFromLocalStorage(order_line.id);

    var productData = {
		'id'       : order_line.id,
		'name'     : order_line.name,
		'category' : order_line.category,
		'brand'    : order_line.brand,
		'variant'  : order_line.variant,
		'price'    : order_line.price,
		'quantity' : order_line.quantity
		};

    // Measure adding a product to a shopping cart by using an 'add' actionFieldObject
    // and a list of productFieldObjects.
    if(redirect == undefined){  

    	///  GA reporting outside-of-GTM way
		ga('ec:addProduct', productData);
		ga('ec:setAction', 'add');
		ga('send', 'event', 'Ecommerce', 'Add to Cart', 'add to cart');     // Send data using an event.
	
		// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here
        dataLayer.push({
          'event': 'addToCart',
          'ecommerce': {
            'currencyCode': 'USD',
            'add': {
              'actionField': {'list': list},
              'products': [order_line]
            }
          }
        });
    }
    else{     

    	///  GA reporting outside-of-GTM way
		ga('ec:addProduct', productData);
		ga('ec:setAction', 'add');
		ga('send', 'event', 'Ecommerce', 'Add to Cart', 'add to cart', { hitCallback: function() {
            window.location = redirect;
		}});

		// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here
        dataLayer.push({
          'event': 'addToCart',
          'ecommerce': {
            'currencyCode': 'USD',
            'add': {  
              'actionField': {'list': list},
              'products': [order_line]
            }
          },
          //  DISABLED - done in non-GTM above
          // 'eventCallback': function() {
          //   window.location = redirect;
          // }
        });
    }
}

//Send CheckoutStep to Google Analytics
function sendCheckoutStep(step, order_lines, option ){
    if(order_lines.length < 1){
        return false;
    }

    var list = getListFromLocalStorage(order_lines[0].id);
    //get the step number
    var step_index;
    switch(step){
        case 'shopping cart': 
            step_index = 1;
            break;
        case 'auth':
            step_index = 2;
            break;
        case 'recipient':
            step_index = 3;
            break;
        case 'payment': 
            step_index = 4;
            break;
        case 'review':
            step_index = 5;
            break;
    }
    var actionField = (option == undefined) ? {'step': step_index, 'list': list} : {'step': step_index, 'option': option, 'list': list};

    if(step_index != undefined){

		///  GA reporting outside-of-GTM way
		for(var i = 0; i < order_lines.length; i++) {
		    var product = order_lines[i];
		    ga('ec:addProduct', {
				'id'       : product.id,
				'name'     : product.name,
				'category' : product.category,
				'brand'    : product.brand,
				'variant'  : product.variant,
				'price'    : product.price,
				'quantity' : product.quantity
		    });
		}

		// In the case of checkout actions, an additional actionFieldObject can
		// specify a checkout step and option.
		ga('ec:setAction','checkout', {
		    'step': step_index,            // A value of 1 indicates this action is first checkout step.
		    'option': option,      // Used to specify additional info about a checkout stage, e.g. payment method.
		    'actionFieldObject': actionField,
		});
		///  The pageview's are batched for the initial one, but once that had been sent,
		///    then send one for each of these events:
		if ( sentInitialPageView ) {
			ga('send', 'pageview');
		}

		// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here
        dataLayer.push({
        'event': 'checkout',
        'ecommerce': {
          'checkout': {
            'actionField': actionField,
            'products': order_lines,
         }
            },
        });
    }

}

//Send Transaction Data to Google Analytics 
function sendTransaction(order, order_lines){

    //add list that first item came from to actionField
    order.list = getListFromLocalStorage(order_lines[0].id);

	///  GA reporting outside-of-GTM way
	for(var i = 0; i < order_lines.length; i++) {
	    var product = order_lines[i];
	    ga('ec:addProduct', {
			'id'       : product.id,
			'name'     : product.name,
			'category' : product.category,
			'brand'    : product.brand,
			'variant'  : product.variant,
			'price'    : product.price,
			'quantity' : product.quantity
	    });
	}
	ga('ec:setAction', 'purchase', order ); // Transaction level information is provided via an actionFieldObject.
	///  The pageview's are batched for the initial one, but once that had been sent,
	///    then send one for each of these events:
	if ( sentInitialPageView ) {
		ga('send', 'pageview');
	}

	// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here

//Send Product View to Google Analytics 
function sendProductDetailView(product){
    // Measure a view of product details. This example assumes the detail view occurs on pageload,
    // and also tracks a standard pageview of the details page.
    var list = getListFromLocalStorage(product.id);
	if( typeof( dataLayer ) != 'undefined' ) {
		///  GA reporting outside-of-GTM way
		ga('ec:addProduct', {
			'id'       : product.id,
			'name'     : product.name,
			'category' : product.category,
			'brand'    : product.brand,
			'variant'  : product.variant,
			'price'    : product.price,
			'quantity' : product.qty
			});
		ga('ec:setAction', 'detail');
		///  The pageview's are batched for the initial one, but once that had been sent,
		///    then send one for each of these events:
		///  WHY? - because these are mostly driven by onScroll events
		if ( sentInitialPageView ) {
			ga('send', 'pageview');
		}

		// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here
		dataLayer.push({
		  'event': 'productDetailView',
		  'ecommerce': {
			'detail': {
			  'actionField': {'list': list},    // 'detail' actions have an optional list property.
			  'products': [product],
			 }
		   }
		});
	}
}

//Send Product List View to Google Analytics 
function sendProductListView(products, format){

    if(format == 'solr'){
        //normalize the solr products
        normalized_products = [];
        for(var index in products){
            product_obj = products[index];
            product = {};
            product.name = product_obj.title_short;
            product.id = typeof product_obj.item_ids != 'undefined' ? product_obj.item_ids[0] : product_obj.idx;
            product.price = typeof product_obj.price_min === 'number' ? product_obj.price_min.toFixed(2).toString() : product_obj.price_min,
            product.brand = product_obj.holiday;
            product.category = product_obj.packaging;
            product.variant = null;
            product.list = product_obj.list;
            product.position = parseInt(product_obj.position) + 1;

            normalized_products.push(product);
        }

        products = normalized_products;
    }


	///  GA reporting outside-of-GTM way
	for(var i = 0; i < products.length; i++) {
		ga('ec:addImpression', products[i]);
	}
	///  The pageview's are batched for the initial one, but once that had been sent,
	///    then send one for each of these events:
	///  WHY? - because these are mostly driven by onScroll/PCP-update events
	if ( sentInitialPageView ) {
		ga('send', 'pageview');
	}

	// NOTICE -- This is the using-GTM way.  But as other tags may refer to this dataLayer, leaving it here
    // Measure products shown on a list.
    dataLayer.push({
        'event': 'productListImpression',       
        'ecommerce': {
            'currencyCode': 'USD',                       // Local currency is optional.
            'impressions': products
        },
    });
}
