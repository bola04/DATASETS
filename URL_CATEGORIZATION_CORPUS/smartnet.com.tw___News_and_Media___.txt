var xmlHttp;
function checkLogin(url,kind,divId){ 
      if (divId.length==0){ 
          document.getElementById(divId).innerHTML="";
          return;
        }

      if (window.XMLHttpRequest) { // 如果可以取得XMLHttpRequest
          xmlHttp = new XMLHttpRequest();  // Mozilla、Firefox、Safari 
      }
      else if (window.ActiveXObject) { // 如果可以取得ActiveXObject
          xmlHttp = new ActiveXObject("Microsoft.XMLHTTP"); // Internet Explorer
      }  
      //xmlHttp=GetXmlHttpObject();
      if (xmlHttp==null){
          alert ("Your browser does not support AJAX!");
          return;
        }  
      var url=url;   
      if( kind=='doChkLogin'){
            url=url+"?sortWHProSn="+divId;     
      }
      xmlHttp.onreadystatechange=function(){  stateChanged(kind,divId); };
      xmlHttp.open("POST",url,true);
      xmlHttp.send(null); 
  }

}

function trim(str)
{
   var result = "";
   for(var i=0 ; i<str.length ; i++) {
     if(str.charCodeAt(i) != 9 && str.charCodeAt(i) != 10 && str.charCodeAt(i) != 13  && str.charCodeAt(i) != 26  && str.charCodeAt(i) != 32)
     	result = result + str.charAt(i);
   }
   return result;
}
