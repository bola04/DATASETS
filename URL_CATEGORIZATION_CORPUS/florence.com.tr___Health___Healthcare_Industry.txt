        var Term = $("#searchTerm").val();

                    url: "/tr-TR/SearchWidget/SiteSearchAutocomplete",
                    type: "POST",
                    dataType: "json",
                    data: { term: request.term },
                    success: function (data) {
                        if (data != "" && data != null) {

                event.preventDefault();

            function DoSelectAppointmentFilter() {
        var personelID = $("#PersonelId").val();
        var branchId = $("#BranchId").val();
        var hospitalId = $("#HospitalId").val();

        var link = '';

        if (hospitalId != 'Hastane Seçin')
            link = link + 'hastane=' + hospitalId + '&';
        if (branchId != 'Bölüm Seçin')
            link = link + 'bolum=' + branchId + '&';
        if (personelID != 'Doktor Seçin')
            link = link + 'doktor=' + personelID + '&';

        window.location.href = '/online-randevu?' + link;
    }

    function GetBracnhesShortApp() {
        var hospitalId = $('#HospitalId').val();
        var url = "/tr-TR/ShortAppointmentFilter/GetBranchByHospitalIdView";
        $.ajax({
            url: url,
            data: { hospitalId: hospitalId },
            cache: false,
            type: "POST",
            success: function (data) {
                $("#branchListDivSelect").html(data);
                $('#BranchId').trigger("chosen:updated");
                GetPersonelsListView();
            }
        });
    }

    function GetPersonelsListView() {
        var hospitalId = $('#HospitalId').val();
        var branchId = $('#BranchId').val();
        var url = "/tr-TR/ShortAppointmentFilter/GetPersonelByHospitalAndBranchIdView";
        $.ajax({
            url: url,
            data: { hospitalId: hospitalId, branchId: branchId },
            cache: false,
            type: "POST",
            success: function (data) {
                $("#personelListDivSelect").html(data);
                $('#PersonelId').trigger("chosen:updated");
            }
        });
    }


    

    function GetPersonelInformationShortApp() {
        var hospitalId = $('#HospitalId').val();
        var branchId = $('#BranchId').val();

        $.ajax({
            url: url,
            data: { hospitalId: hospitalId, branchId: branchId, personelId: personelId },
            cache: false,
            type: "POST",
            success: function (data) {
                $("#divPersonelInformation").html(data);
            }
        });
    }
