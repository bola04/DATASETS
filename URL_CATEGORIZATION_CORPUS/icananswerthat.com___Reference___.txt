        function setCookie(cname, cvalue, exdays) {
            if (exdays) {
                var d = new Date();
                d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
                var expire_date = d.toUTCString();
                var expires = "expires=" + expire_date;
                document.cookie = cname + '_dt' + "=" + expire_date + ";" + expires + ";path=/";
            } else {
                var expires = "expires=" + getCookie(cname + '_dt');
            }

            document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";


        }

        function getCookie(cname) {
            var name = cname + "=";
            var decodedCookie = decodeURIComponent(document.cookie);
            var ca = decodedCookie.split(';');
            for (var i = 0; i < ca.length; i++) {
                var c = ca[i];
                while (c.charAt(0) == ' ') {
                    c = c.substring(1);
                }
                if (c.indexOf(name) == 0) {
                    return c.substring(name.length, c.length);
                }
            }
            return "";
        }

// ================== Top Ads Alignment ==================//
function calcMinHeight(){
	var minHt = 0;
	$('.topEdBlock .ed-results li.gridLi:lt(4)').each(function() {
		$(this).css('min-height', 'auto')
   		if($(this).height() > minHt){
   			minHt = $(this).height();
   		}
	});
	$('.topEdBlock .ed-results li.gridLi:lt(4)').each(function() {
		$(this).css('min-height', minHt+2)
	});
}
// ================== QnA Alignment ==================//
function qnaMinHeight(){
	var qstnMinH = 0;
	var ansMinH = 0;
	$('.mostPopularQstnsWrap .qstn:lt(6)').each(function() {
		$(this).css('min-height', 'auto')
   		if($(this).height() > qstnMinH){
   			qstnMinH = $(this).height();
   		}
	});
	$('.mostPopularQstnsWrap .qstn:lt(6)').each(function() {
		$(this).css('min-height', qstnMinH+60)
	});
	$('.mostPopularQstnsWrap .ans:lt(6)').each(function() {
		$(this).css('min-height', 'auto')
   		if($(this).height() > ansMinH){
   			ansMinH = $(this).height();
   		}
	});
	$('.mostPopularQstnsWrap .ans:lt(6)').each(function() {
		$(this).css('min-height', ansMinH+50)
	});
}
