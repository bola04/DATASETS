            function ToJavaScriptDate(value) {
                var pattern = /Date\(([^)]+)\)/;
                var results = pattern.exec(value);
                var dt = new Date(parseFloat(results[1]));
                return new Date(dt.getFullYear(), (dt.getMonth() + 1), dt.getDate(), dt.getHours(), dt.getMinutes(), 0, 0);

            }

            function haritaYukle(locationData, divname) {
                LocationData = locationData;

                initialize();
            }
            function initialize() {

                if (LocationData.length > 0) {
                    var mapOptions = {
                        zoom: zoomDeger,
                        mapTypeId: google.maps.MapTypeId.HYBRID //Harita tipini yazabilirsin MapTypeId.ROADMAP MapTypeId.SATELLITE  MapTypeId.HYBRID 
                        ,
                        zoomControl: false //Soldaki zoom kontrolü
                        ,
                        mapTypeControl: false //Harita  tipi hybrid uydu gibi
                        ,
                        streetViewControl: false //Sol üstteki adam simgesi
                        //, disableDefaultUI: true //Tüm çzellikleri kapatır

                    };
                    map = new google.maps.Map(document.getElementById(Divname),
                        mapOptions);

                    var zoomXKoordinat;
                    var zoomYKoordinat;
                    var bounds = new google.maps.LatLngBounds();
                    var infowindow = new google.maps.InfoWindow();
                    var i = 0;
                    for (; i < LocationData.length; i++) {
                        var p = LocationData[i];
                        var latlng = new google.maps.LatLng(p.xKoordinat, p.yKoordinat);
                        bounds.extend(latlng);
                        if (p.xKoordinat != null && p.yKoordinat != null) {
                            zoomXKoordinat = p.xKoordinat;
                            zoomYKoordinat = p.yKoordinat;
                        }
                        var title = p.Adi + " <br/> Adres : " + p.Adres;
                        if (p.Telefon)
                            title += "<br/> Telefon : " + p.Telefon;
                        if (p.Kapasite)
                            title += "<br/> " + p.Kapasite;
                        var marker = new google.maps.Marker({
                            position: latlng,
                            map: map,
                            animation: google.maps.Animation.BOUNCE,
                            title: title
                        });

                        google.maps.event.addListener(marker, 'click', function () {
                            infowindow.setContent(this.title);
                            infowindow.open(map, this);
                        });
                    }

                    if (Divname == "map-GK")
                        map.setCenter(center_val);
                    else {


                        // Try HTML5 geolocation
                        if (navigator.geolocation) {
                            var currentPos = navigator.geolocation.getCurrentPosition(function (position) {
                                var pos;
                                if (zoomXKoordinat == "" && zoomYKoordinat == "") {
                                    pos = new google.maps.LatLng(position.coords.latitude, position.coords.longitude);
                                    new google.maps.Marker({
                                        position: pos,
                                        icon: 'files/google_map_man.gif',
                                        draggable: false,
                                        map: map,
                                        animation: google.maps.Animation.BOUNCE,
                                    });
                                } else {
                                    pos = new google.maps.LatLng(zoomXKoordinat, zoomYKoordinat);
                                }

                                map.setCenter(pos);
                            }, function () {
                                handleNoGeolocation(true);
                            });
                        } else {
                            // Browser doesn't support Geolocation
                            handleNoGeolocation(false);
                        }
                    }

                }
            }
