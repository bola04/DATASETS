     <script type="text/javascript">
      /*<![CDATA[*/
            $(function(){
                var form = $('form.SearchForm');
                if (!form.length){
                    return false;
                }
                var textField = form.find('input[name="text"]');
                if (!textField.length){
                    return false;
                }
                var submitField = form.find('input[type="submit"]');
                if (!submitField.length){
                    return false;
                }

                textField.keyup(function(event){
                    if (this.value.length){

                        submitField.removeClass('disabled')
                            .off('mousedown mouseup');
                    } else {
                        submitField.addClass('disabled')
                            .mousedown(submitMousedown)
                            .mouseup(submitMouseup)
                    }
                }).trigger('keyup')
                  .on('change keydown paste input', submitMouseup)
                  .keypress(function(event){
                        if (event.key == 'Enter'){
                            submitMousedown();
                        }
                    });

                function submitMousedown(){
                    textField.addClass('error');
                }
                function submitMouseup(){
                    textField.removeClass('error');
                }

                if (form.hasClass('no_ajax')){
                    form.submit(function(){
                        return submitField.hasClass('disabled') ? false : true;
                    });
                }
            });
        /*]]>*/
     </script>
