function createCookie(name,value,days) {
	if((usesCookie=="0") && (name != "cookieconsent_status")){		
		document.cookie = "location_activeTab" +'=; Path=/; Expires=Thu, 01 Jan 1970 00:00:01 GMT;';		
		document.cookie = "qtrans_front_language" +'=; Path=/; Expires=Thu, 01 Jan 1970 00:00:01 GMT;'; 		
		return;	
	}
  if (days) {
    var date = new Date();
    date.setTime(date.getTime() + (days * 24 * 60 * 60 *1000));
    var expires = ";secure; expires=" + date.toGMTString();
  }
  else {
    var expires = "";
  }
  document.cookie = name + "=" + value + expires + "; path=/";
}
function readCookie(name) {
	if((usesCookie=="0") && (name != "cookieconsent_status")){		
		document.cookie = "location_activeTab" +'=; Path=/; Expires=Thu, 01 Jan 1970 00:00:01 GMT;';		
		document.cookie = "qtrans_front_language" +'=; Path=/; Expires=Thu, 01 Jan 1970 00:00:01 GMT;'; 		
		return;	
	}
  var nameEQ = name + "=";
  var ca = document.cookie.split(';');
  for(var i=0;i < ca.length;i++) {
    var c = ca[i];
    while (c.charAt(0)==' ') c = c.substring(1,c.length);
    if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
  }
  return null;
}
