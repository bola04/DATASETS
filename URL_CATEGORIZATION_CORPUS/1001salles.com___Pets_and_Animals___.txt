	{
    	var matches, substrRegex;

		matches = [];

			if (substrRegex.test(str))

        		// the typeahead $ plugin expects suggestions to a

        		matches.push({ value: str });

        function recup() {
            _refresh();
            var place = autocomplete.getPlace();
            objGeo["dep"] = "";
            objGeo["region"] = "";
            objGeo["pays"] = "";

            // On regarde si il y a une locality (pour les points d'intérêt)
            has_locality = false;
            is_departement = false;
            is_region = false;
            for (var i=0; i < place.address_components.length; i++) {
                var details_adresse = place.address_components[i];
                for (var j=0; j<details_adresse.types.length; j++) {
                    if (details_adresse.types[j] == "locality") {
                        has_locality = true;
                    }
                    if (details_adresse.types[j] == "administrative_area_level_1") {
                        is_region = true;
                    }
                    if (details_adresse.types[j] == "administrative_area_level_2") {
                        is_departement = true;
                    }
                }
            }
            if (!has_locality && !is_region && !is_departement) {
                // Si il n'y a pas de ville, on recherche la ville dans laquelle est le point d'intérêt
                var geocoder = new google.maps.Geocoder();
                var latLng = new google.maps.LatLng(place.geometry.location.lat(), place.geometry.location.lng());
                geocoder.geocode({'latLng': latLng}, function(results, status){
                    if (status == google.maps.GeocoderStatus.OK) {
                        for (var i=0; i<results.length; i++) {
                            var details_adresse = results[i].address_components;
                            for (var j=0; j<details_adresse.length; j++) {
                                var details_type = details_adresse[j].types;
                                for (var k=0; k<details_type.length; k++) {
                                    // On prend le nom de la ville dans locality
                                    if (details_type[k] == 'locality') {
                                        $('#'+details_type[k]).val(details_adresse[j].long_name);
                                    }
                                }
                            }
                        }
                    }
                });
            };
