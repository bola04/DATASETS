        function showCountriesName(langKey, langDir) {
            $('.shipping-country-select option').each(function () {
                var $this = $(this);
                if (! $this.hasClass('display-none')) {
                    topCountriesId.push($this.val());
                }
                var countryName = $this.attr('data-country-name-' + langKey);
                $this.html(countryName);
            });
            makeSelect2(langKey, langDir);
            getCountryComboCounter = countryComboClosure();
        }

        function makeSelect2(langKey, langDir, isAllCountries) {
            var isAllCountriesParam = (typeof isAllCountries === 'undefined') ? false : true;
            $('.shipping-country-select').select2({
                language: langKey.toLowerCase(),
                dir: langDir,
                matcher: function (term, text) {
                    return select2Matcher(term, text, isAllCountriesParam);
                }
            });
        }

        function select2Matcher(term, text, isAllCountries) {
            if ((isAllCountries === false && topCountriesId.indexOf('' + text.id) === -1) || (isAllCountries === true && text.id === '1000')) {
                return false;
            }

            if (term.term == undefined) {
                return text;
            } else {
                var searchText = term.term.toLowerCase();
                var countryNameString = $(text.element).attr('data-country-name');
                if (countryNameString.search(searchText) != -1) {
                    return text;
                }
            }
            return true;
        }

        $('.select2-search__field').live('keyup', function() {
            var $this = $(this);
            var str = $this.val();
            if (getCountryComboCounter() === 1) {
                selectSearchFirstChar = str;
                makeSelect2('tr', 'ltr', true);
                $('.shipping-country-select').select2('open');
            } else if (getCountryComboCounter() === 3) {
                $this.val(selectSearchFirstChar + $this.val());
            }
        });

        $('.shipping-country-select').change(function() {
            var $this = $(this);
            if ($this.val() === '1000') {
                makeSelect2('tr', 'ltr', true);
                $this.select2('open');
            }
        });
