       function bind_form_1() {
var codigo_form = jQuery('form#frm-form-newsletter');
var sending_form = jQuery('#sending_form_1');
var resume_error = jQuery('#resume_error_1');
var result_info = jQuery('#result_info_1');
codigo_form.validate({
                    event: 'blur',
                    rules: {'form_1_field_1': {required:true},
'form_1_field_2': {required:true},
'form_1_field_3': {required:true, email:true},
'form_1_field_80': {required:true},
'form_1_field_81': {required:true},
'form_1_field_82': {required:true},
'form_1_field_83': {required:true}
},
messages: {
'form_1_field_1': 'Por favor ingresá tu nombre',
'form_1_field_2': 'Por favor ingresá tu apellido',
'form_1_field_3': 'Por favor ingresá un email válido',
'form_1_field_80': 'Por favor ingresá tu DNI',
'form_1_field_81': 'Por favor ingresá tu ciudad',
'form_1_field_82': 'Por favor ingresá tu provincia',
'form_1_field_83': 'Debe ingresar el código'
},
debug: true,
errorElement: 'label',
submitHandler: function(form) {
codigo_form.find('#overlay_1').hide();
                sending_form.show();
                
                resume_error.hide();
                result_info.hide();
				
                jQuery.ajax({
                    type: 'POST',
                    url: codigo_form.attr('action'),
                    data: codigo_form.serialize(),
                    datatype: 'TEXT',
                    success: function(msg) {
                       	sending_form.hide();
                        if(msg.status == '200'){
                            codigo_form.find('#overlay_1').fadeOut('slow', function() {
                                codigo_form[0].reset();
                                codigo_form.find('#result_info_1').html(msg.description).fadeIn('slow');
                                codigo_form[0].reset();
                            });                                            
                        }else{
                            
                            if(msg.status == '500'){
                                codigo_form.find('#overlay_1').fadeOut('slow', function() {
                                    codigo_form.find('#resume_error_1').html(msg.global).show();
                                });                                            
                            }
                            
                            if(msg.status == '503'){
                                codigo_form.find('#overlay_1').show();
                                codigo_form.find('#' + msg.description + '_1').show();
                                resume_error.html(msg.global).show();
                            }

                        }
                    },
                    error: function(XMLHttpRequest, textStatus, errorThrown) {
                        sending_form.hide();
                        codigo_form.find('#overlay_1').fadeOut('slow', function() {
                            resume_error.html('Ups!, ha ocurrido un error.').show();
                        });
                    }
                });

                return false;
                }
            });
            }
            
            jQuery(document).ready(function() {
                bind_form_1();
            });

      function initBt2() {
        var bt = document.querySelectorAll('#component-2')[0];
        var filter = document.querySelectorAll('#filter-goo-2 feGaussianBlur')[0];
        var particleCount = 12;
        var colors = ['#eb2154', '#f0e918', '#FFB300', '#17c8eb']

        bt.addEventListener('mouseover', function() {
          var particles = [];
          var tl = new TimelineLite({onUpdate: function() {
            filter.setAttribute('x', 0);
          }});
          
          tl.to(bt.querySelectorAll('.button__bg'), 0.6, { scaleX: 1.05 });
          tl.to(bt.querySelectorAll('.button__bg'), 0.9, { scale: 1, ease: Elastic.easeOut.config(1.2, 0.4) }, 0.6);

          for (var i = 0; i < particleCount; i++) {
            particles.push(document.createElement('span'));
            bt.appendChild(particles[i]);

            particles[i].classList.add(i % 2 ? 'left' : 'right');
            
            var dir = i % 2 ? '-' : '+';
            var r = i % 2 ? getRandom(-1, 1)*i/2 : getRandom(-1, 1)*i;
            var size = i < 2 ? 1 : getRandom(0.4, 0.8);
            var tl = new TimelineLite({ onComplete: function(i) {
              particles[i].parentNode.removeChild(particles[i]);
              this.kill();
            }, onCompleteParams: [i] });

            tl.set(particles[i], { scale: size });
            tl.to(particles[i], 0.6, { x: dir + 20, scaleX: 3, ease: SlowMo.ease.config(0.1, 0.7, false) });
            tl.to(particles[i], 0.1, { scale: size, x: dir +'=25' }, '-=0.1');
            if(i >= 2) tl.set(particles[i], { backgroundColor: colors[Math.round(getRandom(0, 3))] });
            tl.to(particles[i], 0.6, { x: dir + getRandom(60, 100), y: r*10, scale: 0.1, ease: Power3.easeOut });
            tl.to(particles[i], 0.2, { opacity: 0, ease: Power3.easeOut }, '-=0.2');
          }
