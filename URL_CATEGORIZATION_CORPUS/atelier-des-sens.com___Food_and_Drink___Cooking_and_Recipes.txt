  <script>
   $j('#form-footer-newsletter').on('submit', function(e){
        e.preventDefault();
        $j('.footer-newsletter-email').addClass('loader');
        $j.ajax({
            type: 'POST',
            url: 'https://www.atelier-des-sens.com/fr/newsletter/subscriber/newajax/',
            cache: false,
            data: { email : $j('.footer-newsletter-email').val() },
            beforeSend: function(){
                $j('.footer-newsletter-message').addClass('hidden').html('');
            },
            success: function(response, status) {
                var htmlClass = 'text-danger';
                if(response.status == 1)
                    htmlClass = 'text-success';
                $j('.footer-newsletter-message').addClass(htmlClass).html(response.text);
            },
            error: function(response, textStatus){
                $j('.footer-newsletter-message').addClass('text-danger').html(response.statusText);
            },
            complete: function(data){
                $j('.footer-newsletter-email').removeClass('loader');
                $j('.footer-newsletter-message').removeClass('hidden');
            }
        });
    });
