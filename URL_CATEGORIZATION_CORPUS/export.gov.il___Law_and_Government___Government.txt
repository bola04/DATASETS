        $(".search.calendar-widget-filtering").click(function (e) {
            e.preventDefault();
            $(".events-filter .preloader").addClass("sending");
            var filterobj = {
                FreeText: $("#freetext"),
                ItemType: $(".events-filter div.event-type input[type='hidden']"),
                Locations: $(".events-filter div.filter-event-location input[type='hidden']"),
                SectorId: $(".events-filter div.event-area input[type='hidden']"),
                ServiceId: $(".events-filter div.service-inpt input[type='hidden']"),
                CountryId: $(".events-filter div.event-country input[type='hidden']"),
                StartDate: $("#events-filter-start-date"),
                EndDate: $("#events-filter-end-date")
            }
            var url = "";
            for (var filter in filterobj) {
                if (filter == "FreeText") {
                    var temp = $(filterobj[filter]).val();
                    if (temp != "") {
                        url += "filter." + filter + "=" + temp + "&";
                    }
                }
                else {
                    if (filterobj[filter] != "" && $(filterobj[filter]).val() != -1) {
                        var temp = $(filterobj[filter]).val();
                        if (filterobj[filter] != null && filterobj[filter].parent() != null && filterobj[filter].parent().size() > 0 && filterobj[filter].parent().hasClass('ex-multi-select')) {
                            var values = temp.split(',');
                            for (ix = 0; ix < values.length; ix++) {
                                if (values[ix] != "") {
                                    url += "filter." + filter + "[" + ix + "]=" + values[ix] + "&"
                                }
                            }
                        } else if (temp != "") {
                            url += "filter." + filter + "=" + temp + "&"
                        }
                    }
                }
            }

            if ($('#members-free-event-filter').is(':checked')) {
                url = url + "filter.FreeMember=true&";
            }
            if ($('#public-free-event-filter').is(':checked')) {
                url = url + "filter.FreeEveryone=true&";
            }
            if ($('#public-open-registration-filter').is(':checked')) {
                url = url + "filter.OpenForRegistration=true&";
            }
            var areaFilter = '';
            if (window.areaId != null && window.areaId != '' && window.areaId > 0) {
                areaFilter = '&filter.MainObjectId[0]=' + window.areaId;// + '&filter.ServiceId[0]=' + window.areaId;
            }
            url = url.slice(0, -1);
            if (url.indexOf('ServiceId') == -1 && url.indexOf('SectorId') == -1 && url.indexOf('CountryId') == -1 && areaFilter != null && areaFilter != '') {
                url += areaFilter;
            }
            var search_calander = url.replace("filter=", "");
            localStorage.setItem('currentFilter', search_calander);
            calendar.Search(search_calander);

            var areaId = $('input[name="areaId"]').val();
            var areaName = $('input[name="areaName"]').val();
            var areaUrl = $('input[name="areaUrl"]').val();
            var areaType = $('input[name="areaType"]').val();
            var areaBannerImageUrl = $('input[name="areaBannerImageUrl"]').val();

            var requestUrl = fixUrl("/events/search");

            $.post(requestUrl, { filter: url, areaId: areaId, areaName: areaName, areaUrl: areaUrl, areaType: areaType, areaBannerImageUrl: areaBannerImageUrl }, function (data) {
                if (data != null && data.trim() != "") {
                    $(".events-list").html(data);
                    if ($('.events-list-inner .event-schedule-modal').length > 1) {
                        $('.list-veiw').trigger('click');
                    } else {
                        $('.calander-veiw').trigger('click');
                    }
                    page = 2;
                }
                else {
                    $(".events-list").html("<h2 class='err-text'>לא נמצאו אירועים</h2>");
                }
                $(".events-filter .preloader").removeClass("sending");
                ga('send', 'event', {
                    'eventCategory': 'Events',
                    'eventAction': 'Filter Events',
                    'eventLabel': url
                });

                exEllipsis('.events .item-wrap .event-description');
            }).error(function (err) {
                $(".events-filter .preloader").removeClass("sending");
            });
        });

        var requestUrl = fixUrl('/Events/SearchBar');

        $.get(requestUrl, function (data) {
            var searchBar = JSON.parse(data);
            parseTypesOptions(searchBar.ItemType, "#search_item");
            parseOptions(searchBar.SectorId, "#search_branch");
            parseOptions(searchBar.ServiceId, "#search_service");
            parseOptions(searchBar.CountryId, "#search_location");
            registerMultiSelects();
        }).error(function (err) {
