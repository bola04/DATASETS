function showHide(id) {
	var elem = document.getElementById(id);
	if(elem.style.visibility != 'visible'){
		elem.style.display = 'block'; /*saco el inline-block xq se corre todo el buscador hacia el centro al mostrar el msj "Ingrese texto..." - jp 07-12-2015*/		
		elem.style.visibility = 'visible';
	} else {
		elem.style.display = 'none';
		elem.style.visibility = 'hidden';
	}
}

function showHideBuscador(id, invocador) {
	if(invocador.innerHTML == 'Buscar'){
		invocador.innerHTML = 'Cerrar';
		//invocador.classList.add("lupaopen");
		invocador.className = invocador.className + ' lupaopen'; 
		showHide(id);
	} else {
		invocador.innerHTML = 'Buscar';
		//invocador.classList.remove("lupaopen");
		invocador.className = invocador.className.replace(' lupaopen','');
		showHide(id);
	}	
}

function _checkStandarSearchSubmit(queryId, queryMessageId) {
	var query = document.getElementById(queryId).value;
	var elem = document.getElementById(queryMessageId);		
	var a = query.substring(0,1);
	if (a=='*' || a=='?' || a=='') 
		{
			if(elem.style.visibility != 'visible'){					
				showHide(queryMessageId);	
			}				
			return false;
		}			
	else
	{
		if(elem.style.visibility == 'visible'){					
			showHide(queryMessageId);	
		}	
		return true;
	}			
}

function _formSearchSubmit(formId, queryId, queryMessageId) {
	if(_checkStandarSearchSubmit(queryId, queryMessageId)){
		document.getElementById(formId).submit();
	}
}

function destacoMinisitioSeleccionado(){
    $(document).ready(function () {
        var href = $(location).attr('href'); //obtengo url actual         
        var personajeActual = href.substring(href.lastIndexOf('/')+1); //obtengo el nombre del minisitio seleccionado        
     	var htmlBotonMinisitio = $("#"+ personajeActual +"_movil").html();
     	var tituloMinisitio = $("#"+ personajeActual +"_movil").text();

        $("#"+ personajeActual +"").addClass("seleccionada");
     	$("#"+ personajeActual +"_movil").addClass("seleccionada");
     	     	
     	if($("#"+ personajeActual +"_movil").hasClass("faceta")){         	
         	$("#"+ personajeActual +"_movil").empty(); 
         	$("#"+ personajeActual +"_movil").prepend(htmlBotonMinisitio);
        }else{
         	$("#"+ personajeActual +"_movil").empty();     	
         	$("#"+ personajeActual +"_movil").prepend(tituloMinisitio);
        }
    });    
}
