var ADAC = {};
ADAC.timestamp;

ADAC.openWindow = function(url,w,h) {
	var w = (parseInt(w)>1) ? w : 600;
	var h = (parseInt(h)>1) ? h : 400;
	popup = window.open(url, "adac_popup", "width="+w+",height="+h+",status=yes,scrollbars=yes,resizable=yes");
	popup.focus();

ADAC.updateCounter = function(){
	start = Math.round(new Date().getTime() / 1000)
	diff = ADAC.dateDiff(start, ADAC.timestamp);
}

ADAC.dateDiff = function(sdate, edate){
	var timeshift;
	time = edate - sdate;
	if(time >= 0 && time <= 59) {
		// Seconds
		timeshift = {
			'days': 0,
			'hours': 0,
			'minutes': 0,
			'seconds': time
		};
	} else if(time >= 60 && time <= 3599) {
		// Minutes + Seconds
		pmin = time / 60;
		premin = explode('.', pmin);
		
		presec = pmin - premin[0];
		sec = presec * 60;
		
		timeshift = {
			'days': 0,
			'hours': 0,
			'minutes': premin[0],
			'seconds': round(sec, 0)
		};
	} else if(time >= 3600 && time <= 86399) {
		// Hours + Minutes + Seconds
		phour = time / 3600;
		prehour = explode('.', phour);
		
		premin = phour - prehour[0];
		min = explode('.', premin * 60);
		
		presec = '0.' . min[1];
		sec = presec * 60;
		
		timeshift = {
			'days': 0,
			'hours': prehour[0],
			'minutes': min[0],
			'seconds': round(sec, 0)
	};
	} else if(time >= 86400) {
		// Days + Hours + Minutes + Seconds
		pday = time / 86400;
		preday = pday.split('.');
		
		phour = pday - preday[0];
		prehour = explode('.', phour * 24); 
		
		premin = (phour*24) - prehour[0];
		min = explode('.', premin * 60);
		
		presec = '0.' . min[1];
		sec = presec * 60;
		
		timeshift = {
			'days': preday[0],
			'hours': prehour[0],
			'minutes': min[0],
			'seconds': round(sec,0)
		};
	}
	return timeshift;
}
