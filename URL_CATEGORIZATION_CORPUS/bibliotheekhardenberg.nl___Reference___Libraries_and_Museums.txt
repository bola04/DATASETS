				} else {
					//self.selectedCat = '';
				}
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-select-cat");
				return false;
			});
			
			$('#' + block + ' .activ-submit').live('click', function() {
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-submit");
				return false;
			});
			$('#' + block + ' .activ-searchtext').keydown(function(event) {
				// Start the filter when the user hits the Enter key (DOM_VK_RETURN)
				if (event.keyCode == 13) {
					currentMonth = $('#' + block + ' .activ-current input.month').val();
					currentYear = $('#' + block + ' .activ-current input.year').val();
					self.getItems(currentYear, currentMonth, 0, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-searchtext");
					event.preventDefault();
				}
			});
			$('#' + block + ' .activ-clear').live('click', function() {
				$('#' + block + ' .activ-searchtext').val("").focus();
				currentMonth = $('#' + block + ' .activ-current input.month').val();
				currentYear = $('#' + block + ' .activ-current input.year').val();
				self.getItems(currentYear, currentMonth, 0, "", "", 0, "activ-clear");
				return false;
			});
			$('#' + block + ' .pager.pagenav a').live('click', function() {
				if (typeof $(this).attr('href') != "undefined") {
					var navOffset = $(this).children('input.nav-offset').val();
					var navYear = $(this).children('input.nav-year').val();
					var navMonth = $(this).children('input.nav-month').val();
					self.getItems(navYear, navMonth, navOffset, $('#' + block + ' .activ-searchtext').val(), self.selectedCat, 0, "activ-pagenav");
				}
				return false;
			});
	    },
		
		getItems: function(year, month, startPage, searchText, searchCat, useSession, caller) {
			
			var self = this;
			var block = this.block;
			var id = this.id;
			var module = this.module;
			
			this.currentMonth = month;
			this.currentYear = year;
			
			var extradata = '';
			if (caller != 'init') {
				extradata += "&searchtext=" + String(searchText);
				extradata += "&searchcat=" + String(searchCat);
			} else {
				// init routine
				if (this.catsearch=="0" || searchCat!="") {
					extradata += "&searchcat=" + String(searchCat);
				}
			}
			
			$('#' + block + " .activ-body .items").html("");
			$('#' + block + " .activ-body .waitaminute").show();
			
			$.ajax({
				url: "//www.bibliotheekhardenberg.nl/index.php?option=com_rbgcontent&view=activiteiten&format=raw&layout=" + this.module + "&mid=" + this.id + "&action=content" +
					"&year=" + String(year) +
					"&month=" + String(month) +
					"&limitstart=" + String(startPage) +
					"&useSession=" + String(useSession) + 
					extradata,
				type: "GET",
				dataType: "html",
				//async: false,
				success: function(data, textStatus, XMLHttpRequest) {
					$('#' + block + " .activ-body .waitaminute").hide();
					$('#' + block + " .activ-body .items").html(data);
					$('#' + block + " .activ-body .activity-item-type").each(function () {
						$(this).infoToolTip(module);
					});
				},
				error: function(XMLHttpRequest, textStatus, errorThrown) {},
				complete: function(XMLHttpRequest, textStatus) {
					$('#' + block + " .cat_id").selectbox();
					if (typeof $('#' + block + ' .activ-select-box .cat_id').val() != "undefined") {
						self.selectedCat = $('#' + block + ' .activ-select-box .cat_id').val();
					} else {
						//self.selectedCat = '';
					}
				}
			});
			
			if (useSession==0) {
				var el = $('#' + block).closest('h2'); // werkt niet, uitzoeken
				if (el.length == 0) el = $('#' + block).parents('.rbg-widget').find('h2:first');
				if (el.offset() != null) $('html,body').animate({scrollTop: el.offset().top}, 'fast');
			}
			return false; // Prevent default click action
		},
	
	};
	mod_rbg_activiteiten = mod_rbg_activiteiten$Class.constructor;
})(jQuery);
        </script>
