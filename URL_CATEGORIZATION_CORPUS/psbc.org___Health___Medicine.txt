/**
 * Blood Works Date Picker Mod
 * Destroys existing ui datepicker so we can rebuild with desired settings.
 * @see gravity forms
 * @see jquery ui datepicker
 * @example:
	 $(function() {
			BWDatePicker.init("#input_15_57");
	 });
 */
var BWDatePicker = {
	
	/** 
	 * Initialize object
	 * @param id - id of datepicker el
	 */
  init: function(id) {
    var dpEl = $(id);
    this.destroyExisting(dpEl);
    this.rebuild(dpEl);
  },
  
	/**	
	 * Destroy gravity forms datepicker
	 */
  destroyExisting: function(e) {
    e.datepicker('destroy');
  },
	
	/** 
	 * Rebuild datepicker, exclude weekends, begin at next 3 buisness days
	 */
  rebuild: function(e) {
    var fourWorkingDays = new Date();
    var adjustments = [0, 0, 2, 2, 2, 2, 1]; // Offsets by day of the week

    fourWorkingDays.setDate(fourWorkingDays.getDate() + 4 + adjustments[fourWorkingDays.getDay()]);

    e.datepicker({
      beforeShowDay: jQuery.datepicker.noWeekends, 
      minDate: fourWorkingDays, 
			dateFormat: "mm-dd-yy",
      showOn: 'both', 
      buttonImage: '/wp-content/plugins/gravityforms/images/calendar.png', 
      buttonImageOnly: true 
    });
  },
  
};
