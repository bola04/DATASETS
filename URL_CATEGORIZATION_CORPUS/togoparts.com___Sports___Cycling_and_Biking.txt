    });

    range2_element.change(function(){

        function initializeAutoComplete() {
            var input = document.getElementById('location-text-web');

                            var options = {
                    componentRestrictions: {country: "SG"}
                };
            
            var autocomplete = new google.maps.places.Autocomplete(input, options);

            google.maps.event.addListener(autocomplete, 'place_changed', function () {
                var place = autocomplete.getPlace();
                document.getElementById('latitude').value = place.geometry.location.lat();
                document.getElementById('longitude').value = place.geometry.location.lng();
            });

            
        }

        function initializeMap(searched) {
                            var pos = new google.maps.LatLng(1.294559944603731, 103.83831024169922);
            
            if(latitude_element.val() > 0) {
                pos = new google.maps.LatLng(latitude_element.val(), longitude_element.val());
                drawMap(pos);
            }  else {
                drawMap(pos);
            }
        }

        // Draws map using coordinates
        function drawMap(coords) {
            var range = range_element.val();

            var mapProp = {
                center: coords,
                zoom: 12,
                mapTypeId: google.maps.MapTypeId.ROADMAP
            };

            map = new google.maps.Map(document.getElementById("map-canvas"), mapProp);

            placeMarker(coords); // adds marker
            setLatLngLocation(coords); // sets location text values from lat and long
            map.setCenter(coords);
            marker.setMap(map);

            // if(searched && mpadmin) placeAdMarkers(); // To view ads within range

            // Upon clicking anywhere on the map, this removes the previous marker and circle and creates a new one where user has clicked. Sets location text again.
            google.maps.event.addListener(map, 'click', function(e) {
                var latlng = new google.maps.LatLng(e.latLng.lat(), e.latLng.lng());
                setLatLngLocation(latlng);
                marker.setMap(null);
                circle.setMap(null);
                placeMarker(e.latLng);
            });
        }

        // Sets marker
        function placeMarker(location) {
            marker = new google.maps.Marker({
                position: location, 
                map: map,
                draggable: true,
            });

            addMarkerCircle(range_element.val());
            latlong = marker.getPosition();

            latitude_element.val(latlong.lat());
            longitude_element.val(latlong.lng());

          //  setSortByLocation();

            // Upon dragging and dropping of marker, sets values again
            google.maps.event.addListener(marker, 'dragend', function (e) {
                var latlng = new google.maps.LatLng(e.latLng.lat(), e.latLng.lng());
                latitude_element.val(e.latLng.lat());
                longitude_element.val(e.latLng.lng());
                setLatLngLocation(latlng);
              //  setSortByLocation();
            });
        }

        // Sets marker circle, depending on the range which is the radius
        function addMarkerCircle(range) {
            if(range > 0) {
                var kms = range * 1000;

                circle = new google.maps.Circle({
                  map: map,
                  radius: kms,
                  fillColor: '#CC0000',
                  fillOpacity: 0.25,
                  strokeColor: '#FF0000',
                  strokeWeight: 0.5
                });
                 
                circle.bindTo('center', marker, 'position');
            }
        }

       

        // Sets the value of the nearest link


        // Obtains the location text from lat and long
        function setLatLngLocation(latlng) {
            geocoder.geocode({ 'latLng': latlng }, function (results, status) {
                if (status == google.maps.GeocoderStatus.OK) {
                    if (results[1]) {
                        $('.location-text').val(results[1].formatted_address);
                        $('#location').html(results[1].formatted_address);
                        $('.search-location-keyword').val(results[1].formatted_address);
                    }
                }
            });
        }
