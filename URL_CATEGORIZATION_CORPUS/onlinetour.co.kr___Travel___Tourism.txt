	function getHotelLink(hCode, nationCd) {
		
		var checkIn = "";
		var checkOut = "";
		var now = $.datepicker.formatDate( 'yy-mm-dd', new Date());
		var hotel_link = "/web/hotel/hotel_view.html?hCode=" + hCode;
		
		if(nationCd == "KR") {
			checkIn = $.datepicker.formatDate(  'yy-mm-dd', new Date(Date.parse(now) + 5 * 1000 * 60 * 60 *24));
			checkOut = $.datepicker.formatDate( 'yy-mm-dd', new Date(Date.parse(now) + 6 * 1000 * 60 * 60 *24));
			hotel_link += "&checkIn=" + checkIn + "&checkOut=" + checkOut + "&dbl=1";
		} else {
			checkIn = $.datepicker.formatDate(  'yy-mm-dd', new Date(Date.parse(now) + 15 * 1000 * 60 * 60 *24));
			checkOut = $.datepicker.formatDate( 'yy-mm-dd', new Date(Date.parse(now) + 16 * 1000 * 60 * 60 *24));
			hotel_link += "&checkIn=" + checkIn + "&checkOut=" + checkOut + "&dbl=1";
		}
		
		return hotel_link;
	}

				Date.prototype.format = function(f) {
					if (!this.valueOf()) return " ";

					var weekName = ["일요일", "월요일", "화요일", "수요일", "목요일", "금요일", "토요일"];
					var d = this;

						switch ($1) {
							case "yyyy": return d.getFullYear();
							case "yy": return (d.getFullYear() % 1000).zf(2);
							case "MM": return (d.getMonth() + 1).zf(2);

							case "hh": return ((h = d.getHours() % 12) ? h : 12).zf(2);
							case "mm": return d.getMinutes().zf(2);
							case "ss": return d.getSeconds().zf(2);
