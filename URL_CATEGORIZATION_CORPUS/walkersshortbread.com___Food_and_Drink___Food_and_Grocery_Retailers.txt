      <script type="text/javascript">
       $(function() {
		var selector = $('.country-selection');
		var picker = $('.country-selection-open', selector);
		$('.current-region', selector).click(function() {
			picker.show();
		});

		$('.country-selection-item', selector).click(function(){
			if ($(this).attr('data-href')) {
				window.location = $(this).attr('data-href');
			}
			else {
				var form = $('form', selector);
				var select = $('select', form);
				var slug = $(this).attr('data-slug');
				var current = select.val();
				if (slug != current) {
					select.val(slug);
					$('input[name=qs]', form).val(window.location.search);
					form.submit();
				}
				else {
					picker.hide();
				}
			}
		});

		$(document.body).click(function(e) {
			if (picker.is(':visible') && !$(e.target).parents().is(selector)) {
				picker.hide();
			}
		});
	});
      </script>
