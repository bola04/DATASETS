   function createCookie(name, value, minutes) {
		var expires = '';

		if(minutes) {
			var date = new Date();
			date.setTime(date.getTime() + (minutes * 60 * 1000));
			expires = '; expires=' + date.toGMTString();
		}

		document.cookie = name + '=' + value + expires + '; path=/';
	}

	function readCookie(name) {
		var nameEQ = name + '=',
			ca = document.cookie.split(';');

		for(var i = 0; i < ca.length; i++) {
			var c = ca[i];

			while(c.charAt(0) == ' ') {
				c = c.substring(1, c.length);
			}
			
			if(c.indexOf(nameEQ) === 0) {
				return decodeURI(c.substring(nameEQ.length, c.length));
			}
		}

		return null;
	}

	function urlParams(name, url) {
		var u = (url) ? url : window.location.href,
			regexS,
			regex,
			results;

		name = name.replace(/[\[]/, '\\\[').replace(/[\]]/, '\\\]');
		regexS = '[\\?&]' + name + '=([^&#]*)';
		regex = new RegExp(regexS);
		results = regex.exec(u);

		if (results === null) {
			return '';
		} else {
			return results[1];
		}
	}

	function urlParamMap() {
		if (window.location.search) {
			return window.location.search
				.substring(1)
				.split('&')
				.reduce(function(map, current, i, params) {
					var pair = current.split('=');

					if (pair.length === 2) {
						map[decodeURIComponent(pair[0])] = decodeURIComponent(pair[1]);
					} else if (pair.length === 1) {
						map[decodeURIComponent(pair[0])] = undefined;
					};

					return map;
				}, {});
		};

		return {};
	}

	function setUrlParams(paramMap) {
		var query = '?';

		if (Object.keys(paramMap).length) {
			var delimeter = '';

			for (var key in paramMap) {
				if (paramMap.hasOwnProperty(key)) {
					query += delimeter + encodeURIComponent(key); 

					if (paramMap[key]) {
						query += '=' + encodeURIComponent(paramMap[key]);
					};

					delimeter = '&';
				};
			};
		};

		window.location.search = query;
	}

	function setUrlParam(name, value) {
		var paramMap = urlParamMap();
		
		if (typeof value === 'undefined'
				&& paramMap.hasOwnProperty(name)) {
			delete paramMap[name];
		} else {
			paramMap[name] = value;
		};

		setUrlParams(paramMap);
	}

	function getCid() {
		var cid = urlParams('cid') || urlParams('CID');

		if(cid !== '') {
			var regex = /[\d|-]+/g,
				result = regex.exec(cid);

			if(result != null && result.length > 0) {
				createCookie('CampaignId', result[0].substring(0, 15), 60 * 24 * 60);
			}
		}

		var campaignId = readCookie('CampaignId');

		if (!campaignId) {

			if (window.location.pathname.indexOf('/accessories') >= 0) {
				// Set CID for Accessories organic
				campaignId = '52-0025-02';
				createCookie('CampaignId', campaignId, 60 * 24 * 60);
			} else {
				campaignId = '34-0007-03'; // Default CID
			}
			
		}

		return campaignId;
	}

	// Set CampaignId globally
	window.campaignId = getCid();
