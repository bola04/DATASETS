var _edithandle = function doEdithandle() {
    var _id = $(this).attr('id');
    var _action   = $(this).attr('name');
    var _editvals = $(this).attr('id').split('--');
    var _slotname = _editvals[1];

    var _actualcount = parseInt($('#editfield-'+_slotname+' #count--'+_slotname).html());

    if (_action !== 'close'
        && _action !== 'open'
        && _action !== 'upcount'
        && _action !== 'downcount'
        && _action !== 'save'
        && _actualcount == 0)
    {
        $('#editfield-'+_slotname+' #count--'+_slotname).html('3');
    }

    switch(_action) {
        case 'close':
            $('#edit--'+_slotname).attr('name', 'open');
            $('#close--'+_slotname).attr('name', 'open');
            $(this).addClass('red');
            $('#editfield-'+_slotname).parent().fadeIn('slow');
            return true;
            break;
        case 'open':
            $('#edit--'+_slotname).attr('name', 'close');
            $('#close--'+_slotname).attr('name', 'close');
            if ($('#edit--'+_slotname).hasClass('red')) {
                $('#edit--'+_slotname).removeClass('red');
            }
            $('#editfield-'+_slotname).parent().fadeOut('slow');
            return true;
            break;
        case 'upcount':
            _actualcount++;
            if (_actualcount > 10) {
                _actualcount = 10;
            }
            $('#editfield-'+_slotname+' #count--'+_slotname).html(_actualcount);
            break;
        case 'downcount':
            _actualcount--;
            if (_actualcount <= 1) {
                _actualcount = 1;
            }
            $('#editfield-'+_slotname+' #count--'+_slotname).html(_actualcount);
            break;
        case 'topicselect':
            var _key = $('#'+_id).attr('rel');
            var _label = $('#'+_id).text();
            var _readable = _editvals[2];
            var _html = '<li title="'+_label+'" id="'+_key+'" property="check">'+_label+'<a style="float:right;" id="topicdel--'+_slotname+'--'+_readable+'" class="btn-close edithandle" onmousedown="$(this).bind(\'click\', _edithandle);" name="topicdel">Del</a></li>';
            $(_html).prependTo('#topics--'+_slotname);
            $('#'+_id).css('display','none');
            break;
        case 'topicdel':
            if ($('#editfield-'+_slotname+' #'+_id).parent().attr('property') == 'check') {
                var _readable = _editvals[2];
                var _keyid = 'topicselect--'+_slotname+'--'+_readable;
                $('#editfield-'+_slotname+' #'+_keyid).css('display','block');
            }
            $('#editfield-'+_slotname+' #'+_id).parent().remove();
            break;
        case 'add':
            addBoxes(false);
            break;
        case 'remove':
            var _parent = $('#'+_slotname).parent().attr('id');
            $('#'+_slotname).remove();
            $.fancybox.close();
            checkColumns('remove', _slotname);
            break;
        case 'delete':
            $('#fancymessage').find('#submit-delurl').find('a').attr('id', 'remove--'+_slotname);
            $.fancybox(
                $('#fancymessage').html(),
                {
                    'autoDimensions'    : false,
                    'width'                 : 610,
                    'height'                : 230,
                    'transitionIn'      : 'none',
                    'transitionOut'     : 'none'
                },
                open
            );
            break;
        case 'reset':
            $('#editfield-'+_slotname+' input:checked').attr('checked', false);
            $('#topicselections--'+_slotname+' a').css('display','block');
            $('#topics--'+_slotname+' li').remove();
            $('#editfield-'+_slotname+' #count--'+_slotname).html('3');
            break;
        case 'save':
            var _ajaxQuery = buildAjaxQuery(_actualcount, _slotname);
            if (_ajaxQuery.RESET == true) {
                $('#edit--'+_slotname).addClass('red').removeClass('red2');
            } else {
                $('#edit--'+_slotname).addClass('red2').removeClass('red');
            }
            saveBox('pluginConf', _ajaxQuery.KEY, _ajaxQuery.VALUE, _slotname, _slotname, 'config');
            return true;
            break;
        default:
            break;
    }

    $('#close--'+_slotname).css('display','none');
    $('#save--'+_slotname).css('display','block');
};

function buildAjaxQuery(count, _slotname) {
    var _checked = $('#editfield-'+_slotname+' p input:checked').serializeArray();
    var _check = new Object;
    if (_checked.length > 0) {
        $('#editfield-'+_slotname+' p input').each(function() {
            if (typeof _check[$(this).attr('name')] !== 'object') {
                _check[$(this).attr('name')] = {
                        DATA : '',
                        ZAHL : 0
                };
            }

            if ($(this).is(':checked')) {
                if (_check[$(this).attr('name')].ZAHL > 0) {
                    _check[$(this).attr('name')].DATA += ';'+$(this).val();
                } else {
                    _check[$(this).attr('name')].DATA += $(this).val();
                }
                _check[$(this).attr('name')].ZAHL++;
            }
        });
    } else {
        $('#editfield-'+_slotname+' p input').each(function() {
            _check[$(this).attr('name')] = {
                    DATA : '',
                    ZAHL : 0
            };
        });
    }

    _check['topics'] = { DATA : '' };
    var _i = 0;
    $('#topics--'+_slotname+' li').each(function() {
        if (_i > 0) {
            _check['topics'].DATA += ';'+$(this).attr('id')+'~~'+$(this).attr('title');
        } else {
            _check['topics'].DATA += $(this).attr('id')+'~~'+$(this).attr('title');
        }
        _i++;
    });

    var _queue = false;
    var _key = 'anz';
    var _value = count;
    var _reset = true;

    jQuery.each(_check, function(index, value) {
        if (value.DATA!=='') { _reset = false; }
        _key += '|'+index;
        _value += '|'+value.DATA;
    });
    _queue = new Object({
        KEY   : _key,
        VALUE : _value,
        RESET : _reset
    });
    return _queue;
}

        var itemList=itemorder.toString();
        switch (columnId) {
            case 'col_1':
                _check.COL_1 = itemList;
                break;
            case 'col_2':
                _check.COL_2 = itemList;
                break;
            case 'col_3':
                _check.COL_3 = itemList;
                break;
            default:
                break;
        }
    });
    var _values = _check.COL_1+'|'+_check.COL_2+'|'+_check.COL_3;
    saveBox('pluginConf', 'col_1|col_2|col_3', _values, '_column', slotname, action);
}

function saveBox(cmd, keys, values, plugin, slotname, action) {
    if (action == 'config') {
        $('#save--'+slotname).fadeOut('fast');
        $('<a class="ajaxload_small" style="float:right;"></a>').prependTo('#editfield-'+slotname+' .btn-bar');
    }

    var _setup = $.ajaxSetup({
        url: 'http://www.chemie.de/visitor/send/',
        data: 'cmd='+cmd+'&key='+keys+'&value='+values+'&plugin='+plugin,
        dataType : "json",
        success: function(msg) {
            var obj = msg;
            if (obj.status == 'OK') {
                if (action !== false) {
                    if (action == 'remove') {
                        $('#'+slotname).fadeOut('slow').delay(500).remove();
                        $('#col_ghost-table #active--'+slotname).removeAttr('checked');
                    } else if (action == 'config') {
                        $.ajax({
                            url: 'http://www.chemie.de/home2/ajax/?slot='+slotname,
                            dataType: "html",
                            success: function(tpl) {
                                $('#replaceContent--'+slotname).fadeOut().html('');
                                $('#replaceContent--'+slotname).html(tpl).fadeIn('slow');
                                if ($('#replaceContent--'+slotname+' ul').hasClass('accordion')) {
                                    $('#replaceContent--'+slotname+' ul.accordion').accordion({
                                        active: ".selected",
                                        autoHeight: false,
                                        header: ".opener",
                                        collapsible: true,
                                        event: "click"
                                    });
                                }
                                if ($('#replaceContent--'+slotname+' div').hasClass('gallery')) {
                                    initGallery();
                                }
                                if ($('#replaceContent--'+slotname+' div.calendar-holder div').hasClass('datepicker')) {
                                    initDatePicker();
                                }
                                $('#close--'+slotname).css('display','block');
                                $('#save--'+slotname).css('display','none');
                                $('.ajaxload_small').remove();
                            }
                        });
                    } else if (action == 'move') {
                        var _splitted = values.split('|');
                        var _col1 = _splitted[0].split(',');
                        var _col2 = _splitted[1].split(',');
                        var _col3 = _splitted[2].split(',');
                        jQuery.each(_col1, function(index, slot) {
                            $('#col_ghost-table #active--'+slot).val('col_1');
                        });
                        jQuery.each(_col2, function(index, slot) {
                            $('#col_ghost-table #active--'+slot).val('col_2');
                        });
                        jQuery.each(_col3, function(index, slot) {
                            $('#col_ghost-table #active--'+slot).val('col_3');
                        });
                    } else if (action == 'reload') {
                        window.location.reload();
                    }
                }
            } else {
                alert('ERROR! '+obj.message);
                window.location.reload();
            }
        }
    });
    statusCheck(_setup, slotname);
}

function statusCheck(_setup, slotname) {
    $.ajax({
        url: 'http://www.chemie.de/visitor/send/',
        data: 'cmd=loginStatus',
        dataType : "json",
        success: function(msg) {
            var obj = msg;
            if(obj.loginStatus == 'anonymus' && obj.confcount > 8) {
                $('#close--'+slotname).fadeIn('slow');
                $.fancybox(
                    $('#fancybox-msg2').html(),
                    {
                        'autoDimensions'    : false,
                        'width'                 : 610,
                        'height'                : 230,
                        'transitionIn'      : 'none',
                        'transitionOut'     : 'none'
                    }
                );
                $('.ajaxload_small').remove();
            } else {
                $.ajax(_setup);
            }
        }
    });
}

function initAutocompleter(slotname, source) {
    $('#topic-search--'+slotname).keydown(function(e) {
        if (e.keyCode==13) {
            return false;
        }
    });

    $('#topic-search--'+slotname).autocomplete(
    "http://www.chemie.de/prototopix/json2/",
    {
            autosubmitform: "topic-form--"+slotname,
            width: '296',
            max: 10,
            highlight: false,
            scroll: true,
            scrollHeight: 300,
            multiple: false,
            autoFill: false,
            selectFirst: false,
            cacheLength: false,
            cache: false,
            minChars: 1,
            delay: 70,
            parse:function(data) {
                var parsed = [];
                var rows = data;
                if (rows.length>0) {
                    for (var i=0; i < rows.length; i++) {
                        var row = $.trim(rows[i]);
                        var content = row.split("|");
                        if (row) {
                            parsed[parsed.length] = { data: content, value: content[1], result: content[2] };
                        }
                    }
                } else {
                    parsed[0] = { data: new Array('noresult', 'Keine Themen vorhanden ...', 'Keine Themen vorhanden ...'), value: 'Keine Themen vorhanden ...', result : 'Keine Themen vorhanden ...' };
                }
                return parsed;
            },
            formatItem: function(data, i, n, value) { return value; },
            formatResult: function(data, value) { return value.split('.')[0]; },
            extraParams: {
                lang: "de" ,
                searchmode: "true",
                readable: "true",
                contenttype: source
            }
    }).result(function(event,item){
        if (item[0]!=='noresult') {
            var _exists = false;
            $('#topics--'+slotname+' li').each(function() {
                if ($(this).attr('title') == item[2]) {
                    _exists = true;
                }
            });

            if (_exists == false) {
                var _actualcount = parseInt($('#editfield-'+slotname+' #count--'+slotname).html());

                if (_actualcount == 0) {
                    $('#editfield-'+slotname+' #count--'+slotname).html('3');
                }

                var _property = '';
                if($('#topicselect--'+slotname+'--'+item[3]).length) {
                    $('#topicselect--'+slotname+'--'+item[3]).css('display','none');
                    _property = 'property="check"';
                }
                var _html = '<li title="'+item[2]+'" id="'+item[0]+'" '+_property+'>'+item[2]+'<a style="float:right;" id="topicdel--'+slotname+'--'+item[3]+'" class="btn-close edithandle" onmousedown="$(this).bind(\'click\', _edithandle);" name="topicdel">Del</a></li>';
                $(_html).prependTo('#topics--'+slotname);
                $('#close--'+slotname).css('display','none');
                $('#save--'+slotname).css('display','block');
            }
        }
    });

    $('#topic-search--'+slotname).focus(function() {
        if ($(this).val()=='Themensuche') {
            $(this).val('');
        }
    });

    $('#topic-search--'+slotname).blur(function() {
        if ($(this).val()=='') {
            $(this).val('Themensuche');
        }
    });
}
