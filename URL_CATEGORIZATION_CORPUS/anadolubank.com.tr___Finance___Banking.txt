                    tmp = window.interests.FAIZ[iobj][which_column];

                    if (typeof tmp != 'undefined'){
                        tmp = tmp.replace("%", "").trim();
                        int_rate = parseFloat(tmp)/100;

                        var calculated2 = (amount * int_rate * duration) / days;
                        //console.log ("c2-->" + calculated2);

                        var calculated  =  calculated2 - ( calculated2 * stopajOrani );
                        //console.log("c-->" + calculated);

                        //console.log("o-->" + stopajOrani);

                        $('#faizVade').val(tmp);

                        $('#toplam-tutar-k').val($.number(amount + calculated, 2, ',', '.'));
                        //console.log(amount+calculated);
                    }
                    
                }
            }

        } else {
            console.log("Data cannot be fetched.");
        }
    },

    inBetween:function(text_val, check){

        var res = text_val.replace(",","").split("-");

        var _begin = parseFloat(res[0].replace(",",""));
        var _end;

        if (res[1].trim() == ''){
            _end = Number.MAX_VALUE;
        } else {
            _end = parseFloat(res[1].replace(",",""));
        }
        if (check>=_begin && check<=_end){
            return true;
        }
        return false;
    },

    PMT:function(rate, nper, pv, fv, type) {
        if (!fv) fv = 0;
        if (!type) type = 0;
        if (rate == 0) return -(pv + fv) / nper;
        var pvif = Math.pow(1 + rate, nper);
        var pmt = rate / (pvif - 1) * -(pv * pvif + fv);
        if (type == 1) {
            pmt /= (1 + rate);
        }
        return pmt;
    },
