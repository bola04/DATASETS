   angular.module('recipesApp', ['ngSanitize'])
    .filter('unique', function() {
        return function(collection, keyname) {
            var output = [],
                keys = [];

            angular.forEach(collection, function(item) {
                var key = item[keyname];
                if(keys.indexOf(key) === -1) {
                    keys.push(key);
                    output.push(item);
                }
            });

            return output;
        };
    })
    .controller('RecipesController', ['$scope', '$location', '$http', function($scope, $location, $http) {

      var author = 'Chelsea Sugar';

      //if cached recipes we get them
      var cachedRecipes = JSON.parse(window.localStorage.getItem('recipesCache') || '[]');
      var recipesArray = cachedRecipes;

      //if no recipe id, means we dont come from a recipe so empty al cached recipes
      var recipeid = window.localStorage.getItem('recipeId');
      if(recipeid === null) {
        recipesArray = [];
        if(window.localStorage['selectedCategory1']){
        delete window.localStorage['selectedCategory1'];
        }
        if(window.localStorage['selectedCategory2']){
        delete window.localStorage['selectedCategory2'];
        }

        var initialLoad = 0;
      } else {
        var initialLoad = recipesArray.length;
      }



      //default categories values to get all recipes
      if(recipeid === null) {
          var category1 = 'All';
          var category2 = '';
      }
      else {
          var category1 = window.localStorage.getItem('selectedCategory1');
          var category2 = window.localStorage.getItem('selectedCategory2');
      }

      //default sort value
      if(recipeid === null) {
          var sortBy = 'popularity';
      }
      else {
          var sortBy = window.localStorage.getItem('sortBy');
      }


      function getRecipes() {

        $scope.displayMore = { 'display': 'none'};

        if(!category2) {
          category2 = 'no';
        }

        $('.spinny').addClass('loading');

        window.localStorage.setItem('selectedCategory1', category1);
        window.localStorage.setItem('selectedCategory2', category2);

        //                                              cat1     cat2     sort   limit    start
        $http.get('/recipes/get/' + category1 + '/' + category2 + '/' + sortBy + '/6/' + initialLoad + '/').
            success(function(data) {

                $('.spinny').removeClass('loading topPos');
                newData = data;
                recipesArray = recipesArray.concat(newData);

                initialLoad = (initialLoad + 6);

                $scope.recipes = recipesArray;

                //show 'load more' button only if new data
                $scope.displayMore = { 'display': 'inline-block'};
                if(newData.length < 6){
                  $scope.displayMore = { 'display': 'none'};
                }

                //re-initialize Share-this plugin
                setTimeout(function(){
                  stButtons.makeButtons()
                }, 1000);
            });
    }


      function getQueryVariable(variable) {
            var query = window.location.search.substring(1);
            var vars = query.split('&');

            for (var i = 0; i < vars.length; i++) {
                var pair = vars[i].split('=');
                if (decodeURIComponent(pair[0]) == variable) {
                    return decodeURIComponent(pair[1]);
                }
            }
        }

                var categoryquery1 = getQueryVariable('category1');
                var categoryquery2 = getQueryVariable('category2');

                if (categoryquery1){
                  category1 = categoryquery1;
                }
                if (categoryquery2){
                  category2 = categoryquery2;
                }


                getRecipes();

                if (categoryquery1 && recipeid === null) {
                    var offset = $(window).width() < 767 ? 300 : 100;
                    $('html, body').animate({
                        scrollTop: $("#recipes").offset().top + offset
                    }, 1000);
                }

                $scope.multipleCategories = function(value1, value2){
                  $scope.recipes = [];
                  recipesArray = [];
                  initialLoad = 0;
                  $scope.selected = value1;

                    category1 = value1;
                    category2 = value2;

                    $scope.recipes = getRecipes();

                    $scope.displayMore = { 'display': 'inline-block'};
                    var offset = $(window).width() < 767 ? 300 : 100;
                    $('html, body').animate({
                        scrollTop: $("#recipes").offset().top + offset
                    }, 1000);
                }

                $scope.sortRecipes = function(value){
                  initialLoad = 0;
                  $scope.recipes = [];
                  recipesArray = [];
                  sortBy = value;
                  $scope.sortedBy = value;

                    $scope.recipes = getRecipes();

                    $scope.displayMore = { 'display': 'inline-block'};
                }


                $scope.loadMore = function(){
                    getRecipes();
                }

                $scope.gotoRecipe = function(recipe){

                //caching all loaded recipes into localStorage
                var recipes = $scope.recipes;
                window.localStorage.setItem('recipesCache', JSON.stringify(recipes));

                //rememmber id to get scroll Position
                var recipeId =  recipe.cID;
                window.localStorage.setItem('recipeId', recipeId);

                if(sortBy) {
                window.localStorage.setItem('sortBy', sortBy);
                }

                //got to selected recipe
                window.location = recipe.cPath;
                }

                $scope.likeRecipe = function(recipe, event){
                if(logged){
                    var id = (oldUser && recipe.originalcID) ? recipe.originalcID : recipe.cID;
                //alert(id);
                $http.get('/recipes/like/' + id).
                  success(function(data) {
                      $(event.target).toggleClass('disable');
                  });

                } else {
                    $('#login-popup').modal('show');
                }
                }


                if(window.localStorage['recipeId']){
                    delete window.localStorage['recipeId'];
                }

  }]);
