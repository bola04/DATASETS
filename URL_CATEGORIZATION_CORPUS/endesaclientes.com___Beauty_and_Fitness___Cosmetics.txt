    function validateVersion(){
        // Convertimos en minusculas la cadena que devuelve userAgent
        var ie = navigator.userAgent.toLowerCase();

    	if(ie.indexOf('msie') > -1){
            // Extraemos de la cadena la version de IE
            version = parseInt(ie.split('msie')[1]);
            // Dependiendo de la version mostramos la modal
            if (version<=8) {
                fcNavigatorVersion();
            }
        } else if(ie.indexOf('firefox') > -1){
        	browsername = " Mozilla Firefox ";
       	   	version = parseInt(ie.split('firefox/')[1]);
        	if (version<=20) { //20
                fcNavigatorVersion();
            }

        } else if(ie.indexOf('chrome') > -1){
        	browsername = " Google Chrome ";
        	version = parseInt(ie.split('chrome/')[1]);
            if (version<=26) { //26
                fcNavigatorVersion();
            }

        } else if(ie.indexOf('safari') > -1){
            version = parseInt(ie.split('version/')[1]);
            browsername=" Safari ";
            if (version<=5){
                 fcNavigatorVersion();
            } 
        } 
    }

function focusMenuItem(focuselement) {
	focuselement.toggleClass("focusNavigation");
	if(focuselement.hasClass("focusNavigation")){
		$('.mga').not(focuselement)
			.removeClass("focusNavigation")
			.attr("aria-selected", "false")
			.attr("tabindex", "-1");
		focusMga.removeClass("focus");
	} else {
		focusMga.addClass("focus");
	}
}

$(".mga").click(function(){
	var nodo = $(this).attr("href");
	clickedItemLink = $(this);

	focusMenuItem(clickedItemLink);

	var nodoZindex = parseInt($(nodo).css("z-index"));

	if (OpenSection == null){ 
		$(nodo)
			.fadeIn()
			.attr("aria-expanded", "true")
			.attr("aria-hidden", "false");
		$(this).attr("tabindex", "0");
		$(this).attr("aria-selected", "true");
		$(nodo).css("z-index", nodoZindex - 1);
		OpenSection = $(nodo);
		return false;	
	} else if ($(nodo)[0] == OpenSection[0]){
		$(nodo)
			.fadeOut()
			.attr("aria-expanded", "false")
			.attr("aria-hidden", "true");
		$(nodo).css("z-index", nodoZindex + 1);
		OpenSection = null;
		return false;
	} else {
		$(nodo)
			.show()
			.attr("aria-expanded", "true")
			.attr("aria-hidden", "false");
		$(this).attr("tabindex", "0");
		$(this).attr("aria-selected", "true");		
		$(nodo).css("z-index", nodoZindex - 1);
		OpenSection
			.fadeOut()
			.attr("aria-expanded", "false")
			.attr("aria-hidden", "true");			
		var OpenSectionZindex = parseInt(OpenSection.css("z-index"))
		OpenSection.css("z-index", OpenSectionZindex + 1);
		OpenSection = $(nodo);
		return false;
	}
});

	var existeSeccionAbierta = OpenSection != null;

	if (clickFueraDelMenu && existeSeccionAbierta){
		clickedItemLink.click();
	}

function handleKeyDown(id){
	var keys = {
            	left:     37,
                right:    39,
                up: 38,
                down: 40
            };

	var numTabs = $(id+" li").length;
	var firsttab = $(id + " li:first-child");
	var lasttab = $(id + " li:last-child");

	$(id).keydown(function(event){
		switch(event.which){
			case keys.right||keys.up:

				var prevTab  = $(id + " li a[aria-selected='true']").parent();
				var nextTab;
				if(prevTab[0] == lasttab[0]){
					nextTab = firsttab;
				} else {
					nextTab = prevTab.next();
				}

				$(nextTab).find("a").focus().click();

				break;

			case keys.left||keys.down:

				var prevTab  = $(id + " li a[aria-selected='true']").parent();
				var nextTab;
				if(prevTab[0] == firsttab[0]){
					nextTab = lasttab;
				} else {
					nextTab = prevTab.prev();
				}

				$(nextTab).find("a").focus().click();

				break;

			break;
		}// end switch.
	}); // end keydown.
}// end handleKeyDown.

							$("#ui-id-"+parseIndex).click();
						}
				    } else{		    	
				    	setCookie(c+"-"+cid, parseInt(total) + 1,7000);
					}
			    }
			});

			function getCookie(cname) {
			    var name = cname + "=";
			    var ca = document.cookie.split(';');
			    for(var i=0; i<ca.length; i++) {
			        var c = ca[i];
			        while (c.charAt(0)==' ') c = c.substring(1);
			        if (c.indexOf(name) != -1) return c.substring(name.length,c.length);
			    }
			    return "";
			} 
			
			function setCookie(cname, cvalue, exdays) {
			    var d = new Date();
			    d.setTime(d.getTime() + (exdays*24*60*60*1000));
			    var expires = "expires="+d.toUTCString();
			    document.cookie = cname + "=" + cvalue + "; {path: '/ib', " + expires +"}";
			} 
		});
