        function getFastMenuItems() {
            var Urls = $("a[isfastmenu='1']");
            var list1 = "";
            var list2 = "";
            if (Urls.length > 20) {
                for (var i = 0; i < Urls.length; i++) {

                    if (i < 20) {
                        list1 += "<li>" + Urls[i] + "</li>";
                    } else {
                        list2 += "<li>" + Urls[i] + "</li>";
                    }
                }

            }
            if (list1 != "") {
                $("#ul_FastMenu").append(list1);
            }
            if (list2 != "") {
                $("#ul_FastMenu2").append(list2);
            }
        }

        function GetMobileMenu() {

                }

                MyLi.appendChild(MyLiSub);
            })

        function GetUnits(UnitID) {

            $.ajax({
                type: "POST",
                url: 'Default.aspx/GetUnits',
                data: '{UnitID: "' + UnitID + '",ParamLangID:"' + getParameter("LangID") + '"}',
                before: function () {
                    $('.load').show();
                    $('#fakulte').addClass("tab-pane fade in")
                },
                contentType: "application/json; charset=utf-8",
                dataType: "json",
                success: function (result) {
                    var MyTotalItem = $('.birim-liste ul li').length;
                    for (var i = 0; i < MyTotalItem; i++) {
                        $('.birim-liste ul').slick('slickRemove', 0);
                    }
                    $('#Ul_UnitList').html(result.d)
                    $('.load').hide();
                    $('#fakulte').addClass("tab-pane active fade in")
                    $('.birim-liste ul').slick("refresh");
                    UpdateResimAtt();
                },
                failure: function (response) {
                    alert(response.d);
                }
            });
        }

        function UpdateResimAtt() {
            $('.resimatt').each(function () {
                var src = $(this).attr('src');
                var klas = $(this).attr('class');
                var alt = $(this).attr('alt');
                if (klas = 'undefined') {
                    klas = 'image';
                }
                $(this).before('<div class="resim-att"><div class="resim-att-pre"><div class="resim-thumb"><div class="resim-centered"><img src="' + src + '" alt="' + alt + '" class="' + klas + '-att" /></div></div></div></div>');
                $(this).remove();
            });

        }
        function getParameter(paramName) {
            var searchString = window.location.search.substring(1),
                i, val, params = searchString.split("&");

            for (i = 0; i < params.length; i++) {
                val = params[i].split("=");
                if (val[0] == paramName) {
                    return val[1];
                }
            }
            return null;
        }
