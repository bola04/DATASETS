			var getUrlSegments = function (urlWithQueryString) { 
				//chars not allowed in values in DFP ---> " ' = ! + # * ~ ; ^ () < > [ ] , &

				var urlCharsRegex = /[:/?&=]/g;
				var spaceCharsRegex = /[ |\s]+/g;

				//separate segments
				segmentsString = url.replace(urlCharsRegex, " ");

				//sanitize the string. remove characters that are not allowed in DFP
				segmentsString = segmentsString.replace(dfpInvalidCharsRegex, " ");

				//replace multiple spaces with a single space

			function _getIosScreenOrientation() {
				//http://www.williammalone.com/articles/html5-javascript-ios-orientation/
				if (Math.abs(window.orientation) === 90) {

			function _getBrowserZoom() {	
				// Returns the scaling factor (for example 1.1 = 110%)
				// window.devicePixelRatio if exists (for Edge and maybe more browsers)
				// else when calculating to reduce the error we determine the need to increase the innerWidth by 16 (if open to full screen)

				var scale =  1;
				if (Edge) {
					scale = window.devicePixelRatio;
				} else if (iPhone || iPad) {
					// real devices do not change window.screen when orientation changes but Chrome emulator does
					var screenWidth = window.screen.width,
						innerWidth = window.innerWidth;

					switch (_getIosScreenOrientation()) {
						case 'landscape': 
							if (screenWidth < window.screen.height) screenWidth = window.screen.height;
							if (innerWidth < window.innerHeight) innerWidth = window.innerHeight;
							break;
						case 'portrait':
							if (screenWidth > window.screen.height) screenWidth = window.screen.height;
							if (innerWidth > window.innerHeight) innerWidth = window.innerHeight;
							break;
					}	
					scale = Math.round(screenWidth / innerWidth * 100) / 100;
				} else {
					scale = Math.round(window.outerWidth / (window.innerWidth + (window.outerWidth === window.screen.width ? 0 : 16)) * 100) / 100;
				}
				return scale;
			}

			function _addOrResizeBlock(block, metrics, parentNode) {
				var border = "";
				var size = "";
				var id = metrics['id'];
				var scrollable = id.endsWith('_panel_scroll');	// scrollable panel

				if (block == null) {
					var div = document.createElement('div');
					div.id = id;
					div.style.position = metrics['position'];
					if (! scrollable) {
						div.innerHTML = metrics["html"];
					}
					if (parentNode == null) {
						document.body.appendChild(div);
					} else {
						parentNode.appendChild(div);
					}
					block = document.getElementById(id);
				}

				block.style.left = metrics["x"] + "px";
				block.style.top = metrics["y"] + "px";
				block.style.width = metrics["w"] + "px";
				block.style.height = metrics["h"] + "px";
				block.style.overflowX = metrics["overflowX"];
				block.style.overflowY = metrics["overflowY"];

				if (scrollable) {
					var subBlocks = block.children;
					var subMetrics = _getSubBlockMetrics(metrics);
					for (var i = subBlocks.length - 1; i >= 0; i--) {
						var metricsId = _findMetricsIndex(subMetrics, subBlocks[i].id);
						if (metricsId != -1) {
							_addOrResizeBlock(subBlocks[i], subMetrics[metricsId], block);
							subMetrics.splice(metricsId, 1);
						} else {
							subBlocks[i].remove()
						}
					}

						_addOrResizeBlock(null, subMetrics[i], block);
					}
				}
			}

			function _getSubBlockMetrics(metrics) {
				var subMetrics = [];
				var div = document.createElement('div');
				div.innerHTML = metrics['html'];
				subBlocks = div.children;
				for(var i = 0; i < subBlocks.length; i++) {
					var subBlock = subBlocks[i];
					var m = {};
					m['id'] = subBlock.id;
					m['position'] = subBlock.style.position;
					m['x'] = parseFloat(subBlock.style.left);
					m['y'] = parseFloat(subBlock.style.top);
					m['w'] = parseFloat(subBlock.style.width);
					m['h'] = parseFloat(subBlock.style.height);
					m["overflowX"] = subBlock.style.overflowX;
					m["overflowY"] = subBlock.style.overflowY;
					m['html'] = subBlock.innerHTML;
					subMetrics.push(m);
				}
				return subMetrics;
			}

			//Function obtains the highest level block in document structure

			function _getLayoutBlock(childBlockId) {
				var current = $('#' + childBlockId);
				if(current != undefined && current.length != 0){
					var target = current.parent();
					while(target != undefined && target.length != 0) {
						var nextParent = target.parent();
						if(nextParent == undefined || nextParent.length == 0 || nextParent.prop('tagName').toLowerCase() == 'body') {
							break;
						} else {
							target = nextParent;
						}
					}
				}
				return target;
			}

			function _getLayoutBlockNotScroll(childBlockId) {

					var target = current.parent();
					while(target != undefined && target.length != 0) {

					}
				}

			function _manageUserFavoriteContent(element, userId) {
				var $element = $(element);
				var contentType = $element.attr('content_type');
				var contentId = $element.attr('content_id');
				var groupType = $element.attr('group_type');
				var groupIds = $element.attr('group_id');
				var shareId = $element.attr('share_id'); 
				var hasGroup = groupType != null && groupIds != null;
				var active = $element.hasClass('active');
				var action = active ? "remove" : "add";
				if (!hasGroup){
				// save content item
					var url = makeFavoriteUrl(action, userId, contentType, JSON.stringify([contentId])); 
					$.ajax(url).done(function( response ) {
						var success = false;
						try {
							success = JSON.parse(response);
						} catch (err) {
						}
						if (success == true) {
							if (active) {// delete from favorites
								$element.removeClass('active');
							} else { // add to favorites
								$element.addClass('active');
							}
						}
					});
				} else {
				// save group of content items if any
					url = makeFavoriteUrl(action, userId, groupType, groupIds); 
					$.ajax(url).done(function( response ) {
						var success = false;
						try {
							success = JSON.parse(response);
						} catch (err) {
						}
						if (success == true) {
							if (active) {// delete from favorites
								$element.removeClass('active');
							} else { // add to favorites
								$element.addClass('active');
							}
							ids = JSON.parse(groupIds);
							$.each(ids, function(i, id){
								if (active) {// delete from favorites
									$('.favorite[content_id="' + id + '"][content_type="' + groupType + '"]').removeClass('active');
								} else { // add to favorites
									$('.favorite[content_id="' + id + '"][content_type="' + groupType + '"]').addClass('active');
								}
							});
						}
					});
				}
			}

			function prepareSearchQuery(search_string) {
				// length of DB field for search query is 255. That is why we truncate query.
				var limit = 255;
				var query = search_string;
				if (search_string.length > limit) {
					query = query.substring(0, limit);
					var index = query.lastIndexOf(" ");
					query = query.substring(0, index);
				}
				return query;
			}

			function setAds() {
				// Check if googletag functions are defined
				adNotBlocked = (typeof(googletag.destroySlots) === "function" && typeof(googletag.pubads) === "function" && typeof(googletag.enableServices) === "function");

				if (!adNotBlocked) {
					console.error("Ad error: googletag functions not defined.");

					return;
				}

				var ads = $('.advertisment');
				if (ads.length && !gptAdsLoading && adNotBlocked) {
					gptAdsLoading = true;

					googletag.cmd.push(function () {
						//googletag.destroySlots();

						for (var i = 0, l = gptAdsInfo.length; i < l; i++) {
							var info = gptAdsInfo[i];

							var mapping = googletag.sizeMapping();
							for (var j = 0, k = info.sizeMap.length; j < k; j++) mapping = mapping.addSize(info.sizeMap[j].viewport, info.sizeMap[j].size)
							mapping = mapping.build();

							for (var id in info.slots) {
								if (info.slots[id].ref || !$('#' + id).length) continue;

								info.slots[id].ref = googletag.defineSlot(info.slots[id].path, info.slots[id].size, id);
								info.slots[id].ref.defineSizeMapping(mapping).addService(googletag.pubads());
							}
						}

						googletag.pubads().enableSingleRequest();
						googletag.pubads().setTargeting("UrlSegments", getUrlSegments());
						googletag.enableServices();
					});


					ads.each(function (i, item) { googletag.cmd.push(function () { googletag.display(item.id); }); });

					setTimeout(function () { gptAdsLoading = false; }, 1000);
				}
			}

			function makeShareButtons(content_id, content_type, url) {
				var result = '';
				var item = '';
				for (var name in shares) {
					if (name === 'popup') continue;
					if (name === 'endnote' && content_type !== 'article') continue;

					item = '<div class="share_button" onclick="shares.' + name + '.run(\'' + content_type + '\', \'' + url + '\', \'' + content_id + '\', this)">' +
						'<img src="images/' + shares[name].image + '">' +
						'<span>' + shares[name].title + '</span>' +
					'</div>';

					if (name === 'endnote') {
						var url = 'php/exporter.php?format=endnote&type=' + content_type + '&id=' + content_id + '&url=' + encodeURIComponent(url);
						item = '<a class="share_endnote" href="' + url + '" download="EndNote.enw">' + item + '</a>'
					}

					result += item;
				}

				return result;
			}

			function checkSendMailRequiredValues(id) {
			    if (id == '') return;

			    var modal = $("#" + id);
			    if (modal.length == 0) return false;

			    var result = true;

			    modal.find('.body .sendmail-required-value').each(function (i, item) { result = result && ($(item).val() !== '') });

			    var el = modal.find('.footer .sendmail-btn-send');
			    if (result) el.removeClass('disabled'); else el.addClass('disabled');
			}

			function sendMailMessage(id) {
			    if (id == '') return;

			    var modal = $("#" + id);
			    if (modal.length == 0) return;

			    $.ajax({
			        url: 'php/mailer.php',
			        type: 'post',
			        data: modal.find('form').serializeArray(),
			        success: function(result) { 
			            var ok = null;
			            try {
			                ok = JSON.parse(result);
			            } catch (err) {
			            }
			            if (ok != true) {
			                mes = 'Your e-mail has not been sent. Please, contact developers.';
			                if (ok != false) {
			                    mes += ' ' + result;
			                }
			                alert(mes);
			            }
			        }
			    });

			    closeSendMailPopUpModal(id);
			}

				showNotification = getUserId() != -1 ? true : showNotificationToAnonymUser(pageLoad);

				if (message != "" && showNotification && document.getElementById('notification') == null) {
					var isMobile = isMobileLayout();

					var notification = _generateNotification(message, link, isMobile);
					$('body').append(notification);

					if (isMobile) {
						$('#notification').animate({top : "-=150px"}, 100);
					} else {
						$('#notification').animate({top : "-=100px"}, 100);
					}

					setTimeout(function () {
						if (document.getElementById('notification') !== null)
							_removeNotification(isMobile, false)
					}, 10000);
				}
			}

			function _removeNotification (isMobile, onLinkClick) {
				if (!onLinkClick) {
					if (isMobile) {
						$('#notification').animate({top : "+=150px"}, 100);
					} else {
						$('#notification').animate({top : "+=100px"}, 100);
					}
					setTimeout(function () {$('#notification').remove();}, 100);
				}

				_saveNotificationAfterDisplay();
			}

			function _generateNotification (message, link, isMobile) {
				message = message.toUpperCase();

				if (link != '') {
					message = '<a onclick="_removeNotification(' + isMobile + ', true)" href ="' + link + '" style = "text-decoration: none;">' + message + '</a>';
				}

				if (isMobile) {
					var height = '6.3em';

					var footerLine = document.getElementById("mobileFooterLine");
					var footer = footerLine.getBoundingClientRect();
					var top = 'calc(' + footer.bottom + 'px - ' + height + ' + 149px)';

					var textStyle = 'text-align: center; margin-top: 15px; font-size: 16px;';
					var closeStyle = 'width:20px; height:20px; cursor:pointer';
					var logoStyle = 'position:absolute; left:-3px; top: -7px; width:4.5em;';			
					var style = 'opacity: 0.95; position: fixed; top : ' + top + '; height: ' + height + '; background-color: #E8E8E8; width: 100%;"';

					notification = 
					  '<div id="notification" style="' + style + '">' +
					  	'<div id ="closeNotification" style="width: auto; text-align: right; margin-top: 1%; margin-right: 1%;" onclick="_removeNotification(' + isMobile + ', false)">' + 
					  		'<img src="images/notification_logo.png" style="'+ logoStyle + '">' + 
							'<img src="images/close_notification.svg" style="' + closeStyle + '">' + 
						'</div>' + 
					    '<div class="notification-body" style= "' + textStyle + '">' +
					    	message +
					    '</div>' +
					'</div>';

				} else {
					var width = 728;
					var height = 90;

					var footerText = document.getElementById("desktopFooter");
					var footer = footerText.getBoundingClientRect();
					var top = 'calc(' + footer.bottom + 'px - ' + height + 'px + 100px)';
					var marginWidth = ($("#desktopFooter").width() - width) / 2 + footer.left;

					var textStyle = 'text-align: center; margin-left: 25%; margin-right: 20%; margin-top: -25px; font-size: 18px;';
					var closeStyle = 'width:20px; height:20px; cursor:pointer; margin-left: 93%; margin-top: 35px;';
					var logoStyle = 'position:absolute; margin-top: -15px; margin-left: 3px; width:9.5em;';	
					var style = 'position: fixed; opacity: 0.95; top: ' + top + '; height: ' + height + 'px; width: ' + width + 'px; background-color: #E8E8E8; margin-left: ' + marginWidth + ';';

					notification = 
					'<div id="notification" style="' + style + '">' +
						'<div class ="notification_logo">' +
							'<img src="images/notification_logo.png" style="'+ logoStyle + '">' + 
						'<div id ="closeNotification" onclick="_removeNotification(' + isMobile + ', false)">' + 
							'<img src="images/close_notification.svg" style="' + closeStyle + '">' + 
						'</div>' + 
					    '<div class="notification-body" style= "' + textStyle + '">' +
					    	message +
					    '</div>' +
					'</div>';
				}

				return notification;
			}

			function _saveNotificationAfterDisplay() {
				var message = "Subscribe Now: Special Offer";
				var md5 = "35baaa5e19c2900e1bb39d433c9fa7e3";
				var userId = getUserId();
				if (md5 == "" || userId == -1) {
					return;
				}

				var url = "php/notification.php?operation=save_displayed_notification&user_id=" + userId + "&md5=" + md5 + "&message=" + message;

				$.ajax(url).done(function( response ) {});
			}

			function showNotificationToAnonymUser(pageLoad){
				show = false;
				var isNotificationShownOnLoad = isCookieAvailable("notification_on_first_load");
				var userComesFromJBJSPage = isCookieAvailable("user_left_hub");
				if (!userComesFromJBJSPage && (!isNotificationShownOnLoad || !pageLoad)) {
					if (!isNotificationShownOnLoad)
						document.cookie = "notification_on_first_load=1;";
					show = true;
				}
				return show;
			}

			function _onUnloadPage() {
				clearCookie("notification_on_first_load");
			}
