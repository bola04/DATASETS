         function openinghours() 
{
    var todaystatus = "open";
    var d = new Date();
    var currentdate = d.getDate();
    var currentmonth = d.getMonth();
    var currentday = d.getDay();
    var currentyear = d.getFullYear();
    
// Test results of a specific date here: (does not change visible date, only open status)
//     currentdate = 4;
//    currentmonth = 5;
//    currentyear = 2018;
//   currentday = 1;
    
    
    // Overriding Closed dates, REMEMBER that GetDate uses 1-31, GetMonth uses 0-11, so 0 = January.
    
    // Christmas Day 25 Dec 2017
    if ((currentdate == 25) && (currentmonth == 11)) todaystatus = "closed";  
    // Good Friday 30 March 2018
    if ((currentdate == 30) && (currentmonth == 2)) todaystatus = "closed";  
    // Anzac Day 25 April 2018
    if ((currentdate == 25) && (currentmonth == 3)) todaystatus = "closed";  
    
     // Closing again every Monday from 1 April 2018
    if ((currentmonth > 2) && (currentday == 1) && (currentyear == 2018)) todaystatus = "closed";
    
    
    // Overriding Open dates, REMEMBER that GetDate uses 1-31, GetMonth uses 0-11, so 0 = January.
    
    // Easter Monday 2 April 2018
    if ((currentdate == 2) && (currentmonth == 3)) todaystatus = "open";  
    // Queen's Birthday 11 June 2017
    if ((currentdate == 11) && (currentmonth == 5)) todaystatus = "open";  
    // Monday after Christmas
    if ((currentdate == 31) && (currentmonth == 11) && (currentyear == 2018)) todaystatus = "open";  
    
    // Exclusion for 26 Dec 2016 - probably not needed now
    if ((currentdate == 26) && (currentmonth == 11) && (currentyear == 2017)) todaystatus = "open";  
    
    
    if (todaystatus == "open") 
    {
       document.getElementById('galleryopen').style.display = "block";
       document.getElementById('galleryclosed').style.display = "none";
    } else {
       document.getElementById('galleryopen').style.display = "none";
       document.getElementById('galleryclosed').style.display = "block";
    }
}
openinghours();
