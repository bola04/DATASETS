   // get url parameters
mapId = getUrlParameter('mapId');
initLatitude = getUrlParameter('lat');
initLongitude = getUrlParameter('lon');
initZoom = getUrlParameter('zoom');

// get session, if present
if( hasLocalStorage() ) {
    sessionId = localStorage.getItem( "serbiamap.session.id" );
}

// create map
map = L.map('map', { attributionControl: false, zoomControl: false });
if ( initLatitude && initLongitude && initZoom ) {
	map.setView( [ initLatitude, initLongitude ], initZoom );
} else {
	map.setView([44.8, 20.5], 10);
}

L.tileLayer('http://serbiamap.net/Tiles.ashx?z={z}&x={x}&y={y}', { maxZoom: 16, minZoom: 2 } ).addTo( map );

L.control.zoom( { position: 'topright' }).addTo( map );

markers = new L.FeatureGroup().addTo( map );
trackingMarkers = new L.FeatureGroup().addTo( map );

$( document ).ready(function() {
	if (!isMobile) {  
		$( "#dialog" ).dialog( { height: 480, width: 250 } ).dialogExtend( { collapsable: true} ).css( 'padding', '2px' );
		positionDialog();
	    $( "#accordion" ).accordion( { heightStyle: "fill" } );
		
		loadMaps('data/maps.xml', webMaps, '#sm-webmapspanel');
		setLoginState();
		focusInitLocation();
		getNotifications('');
	}
});

$(window).resize(function() {
	if (!isMobile) {
		positionDialog();
		positionNotificationDialog();
	}   	
});

function positionDialog() {
	$("#dialog").dialog("option", "position", { my: "left top", at: "left+5 top+5", of: "#map" } );
}

function positionNotificationDialog() {
	$("#notificationDialog").dialog("option", "position", { my: "right bottom", at: "right-5 bottom-5", of: "#map" } );
}


// AJAX callback functions
function onMapsLoad(){
	if ( mapId ) {
		loadMap( mapId );
	}
}

function setLoginState() {
	if ( sessionId !== null ) {
		$('#loginButton').val( 'Logout' );
		$('#sm-register').hide();
		loadUserTracks( sessionId, '#sm-mytrackspanel');
	} else {
		$('#loginButton').val( 'Login' );
		$('#sm-register').show();
		$('#sm-mytrackspanel').empty();
	}
}

function focusInitLocation(){
	if ( initLatitude && initLongitude) {
		markLocation( initLongitude, initLatitude );
	}
}
