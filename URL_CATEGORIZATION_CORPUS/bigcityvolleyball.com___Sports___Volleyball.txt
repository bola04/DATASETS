  <script>
   window.onload = function() {
        ga(function(tracker) {
            var linkerParam = tracker.get('linkerParam');
            var len = jQuery('a[href*="/account"],a[href*="/registrations"],a[href*="/cart"],a[href*="/teams"]').length;
            for (i = 0; i < len; i++) {
                var v = jQuery('a[href*="/account"],a[href*="/registrations"],a[href*="/cart"],a[href*="/teams"]')[i];
                if (v.getAttribute('href').indexOf("?") != -1) {
                    var h = v.getAttribute('href');
                    v.setAttribute('href', h + "&" + linkerParam);
                } else {
                    var h = v.getAttribute('href');
                    v.setAttribute('href', h + "?" + linkerParam);
                }
            }
        });
    }
  </script>
