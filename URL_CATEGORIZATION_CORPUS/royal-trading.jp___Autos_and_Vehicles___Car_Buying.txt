	var popovers = [];
	$('.country-selector').on('click',function(){
		
		
		
		var lastUrl = $(this).attr('data-link');

		
		if (typeof lastUrl == "undefined" || lastUrl == false)
		lastUrl = "";
		
		for(k in popovers)
		$(popovers[k]).popover('destroy');
		
		var placement = "right";
		if ($(this).attr('data-placement') != 'undefined' && $(this).attr('data-placement') != false){
			placement = $(this).attr('data-placement');
		} 
		
		$(this).popover({trigger:"manual", placement:placement,html:true,content:content,title:' -- Select your country first --'});
		$(this).popover('show');
		
		
		popovers.push(this);

		$("#country_pop").bind("change", function (event) {
			//alert("change");
			$.ajax(
				{async:true,
				 data:$("#country_pop").serialize(),
				 dataType:"html",
				 success:function (data, textStatus) {
				 	$("#ports_pop").html(data+"<option value='Other'>OTHER</option>");
				 	$("#ports_pop").attr("disabled",false);
				 	}, type:"post", url:"/site/get_ports"});
		return false;
		});
		
		//listener on validation button
		$("#validate-cp").on("click",function(){
			var country = $("#country_pop").val();
			var port = $("#ports_pop").val();
			
			
			if (country == "" || port == ""){
				alert("Please select country and port");
				return false;
			}
			
			$.cookie('royal_freight', country+"--"+port,{ expires: 365, path: '/' });
			
			//redirect to the usual vehicle
			if (lastUrl != "")
			window.location.href = lastUrl;
			else
			location.reload();
			
		});
		
	});
	
		$("#country").bind("change", function (event) {
			//alert("change");
			$.ajax(
				{async:true,
				 data:$("#country").serialize(),
				 dataType:"html",
				 success:function (data, textStatus) {
				 	$("#ports").html(data+"<option value='Other'>OTHER</option>");
				 	$("#ports").attr("disabled",false);
				 	}, type:"post", url:"/site/get_ports"});
		return false;
		});
	
		function killAllPopup(){
			for(k in popovers)
			$(popovers[k]).popover('destroy');		
		}
