				}, 500);

					document.querySelectorAll(".form-group.has-error .form-control").forEach(function(formElement){

						console.log(errorMessage)
						$( ".error-message" ).append( "<div class='alert alert-danger' role='alert'>" + errorMessage + "</div>" );
