	        called = true
	        handler()
	    }

	            ready()
	        }, false )
	    } else if ( document.attachEvent ) {

	                }
	            }

	                ready()
	            }

          <h2>
           Anthropologists and Their Worlds - a series
          </h2>
          <h3>
           <img alt="possedes" src="/images/M_images/possedes.png"/>
          </h3>
          <p style="text-align: justify;">
           <span style="font-family: arial, helvetica, sans-serif; font-size: 12pt;">
            This series presents audiovisual testimonies and reflections of Canadian anthropologists. Guests share their field experiences and their intellectual trajectories as well as the knowledge they have accumulated over the years, in different "worlds". The series offers long narrations of anthropologists, filmed in their environment, thinking back their disciplinary vocation and their work, talking about their intellectual contribution in methodological, empirical or theoretical fields. The films are open access. This series is produced by Anthropologie et Sociétés, the anthropology department of Université Laval, Anthropologica, and CASCA, and directed by Frédéric Benjamin Laugrand. The series received financial support from these institutions, and from the SHRCC and the IXE
            <span style="text-align: justify;">
             Commission mixte permanente Québec / Wallonie-Bruxelle.
            </span>
            To see the whole series, visit the
            <a href="http://www.anthropologie-societes.ant.ulaval.ca/les-possedes-et-leurs-mondes">
             Anthropologie et Sociétés website
            </a>
            , and follow the
            <a href="https://www.facebook.com/Les-Poss%C3%A9d%C3%A9s-et-leurs-mondes-1878987355727485/" rel="noopener noreferrer" target="_blank">
             Facebook page
            </a>
