					$(".slide").parent().each(function(){
						$(this).hide();

					$(".area"+areaNum).parent().show();
				}

				function determine_center(){
					var startcenter = 3;
					$(".item_"+startcenter).addClass("center");

				function showDatetime(){
					var FR = moment.tz(new Date(), "Europe/Berlin");
					var DE = moment.tz(new Date(), "America/New_York");
					var FRstatus = "업무종료";
					var DEstatus = "업무종료";
					FR.locale('ko');
					DE.locale('ko');
					if( FR.hour() > 7 && FR.hour() < 18 ) FRstatus = "업무중";
					if( DE.hour() > 7 && DE.hour() < 18 ) DEstatus = "업무중";
					$(".fr.time-section").html(FR.format("ll") + "<br>" + FR.format("LT") );
					$(".de.time-section").html(DE.format("ll") + "<br>" + DE.format("LT"));
					$(".fr.open-section").html(FRstatus);
                    $(".de.open-section").html(DEstatus);

					setTimeout(function(){ showDatetime(); }, 10000 );
				}

				function noticechange(){
				    var ncount = 3;
				    var current = $(".noticelink:visible").attr("id").replace("notice", "");
				    if( current == ncount ) current = 1;
				    else current++;

				    $(".noticelink").hide();
					$("#notice"+current).show();

				    setTimeout(function(){
				        noticechange();
				    }, 2000);
				}

                function searchzip(){
                    if ($(".zipcode-search").val() == "") {
                        alert('검색어를 입력하세요');
                        return;
                    }
                    $(".zipcode-list tbody").html("<td colspan='2'>검색중...</td>");
                    $.ajax({
                        //url: "//www.juso.go.kr/addrlink/addrLinkApiJsonp.do"  //인터넷망
                        url: "/include/zip.jsp"
                        , type: "post"
                        , data: {
                            keyword: $(".zipcode-search").val()
                        }
                        , dataType: "xml"
                        //, crossDomain: true
                        , success: function (xmlStr) {
                            $(".zipcode-list tbody").html("");
                            if (xmlStr != null) {
                                makeListnew(xmlStr);
                            }
                        }
                        , error: function (xhr, status, error) {
                            alert("에러발생");
                            console.log(xhr);
                        }
                    });
                }

                function makeListnew(xmlStr) {
                    var htmlStr = "";
                    if ($(xmlStr).find("totalCount").text() == "0") {
                        htmlStr += "<tr><td colspan='2'>------ 검색 결과가 없습니다. ------</td></tr>";
                    } else {
                        $(xmlStr).find("juso").each(function () {
                            htmlStr += "<tr>";
                            htmlStr += "<td>" + $(this).find('zipNo').text() + "</td>";
                            var zipNo = $(this).find('zipNo').text();
                            htmlStr += "<td><a href=\"javascript:setZipnew( '" + zipNo + "', '" + $(this).find('roadAddr').text().replace("'", "\\\'") + "' );\">" + $(this).find('roadAddr').text() + "</a></td>";
                            htmlStr += "</tr>";
                        });
                    }
                    $(".zipcode-list tbody").html(htmlStr);
                }

                    $(".zipcode-popup").hide();
