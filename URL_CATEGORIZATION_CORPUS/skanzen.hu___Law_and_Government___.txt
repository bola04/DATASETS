  <script type="text/javascript">
   function send_elfelejtett_data(){
			//$('#regFeedback').css('display', 'none');
			var errors = $('#addNewsletter').find('.error').css('display', 'none');
			$.ajax({
				type: 'POST',
				url: '/ajax_php/ajax_hirlevel.php',
				data: $('#addNewsletter').serialize(),
				success: function (content) {
					var a_content = jQuery.parseJSON(content);
					console.log(a_content);

					var errorObj = $('[name='+a_content.field+']').prev('.error');
					if(!$(errorObj).hasClass('error')){
						errorObj = $($('[name='+a_content.field+']').parent()).prev('.error');
					}

					if (a_content.type == 'hiba')
					{
						$(errorObj).css('display', 'block');
						$(errorObj).text(a_content.message);
						return false;
					}
					else if (a_content.type == 'info')
					{
						$(errorObj).css('display', 'none');
						$(errorObj).text('');
						myAlert.closeAlert();
						myAlert.Normal('middle', 'Sikeres feliratkozás', a_content.message);

						return false;
					}
				}
			});
		}
  </script>
