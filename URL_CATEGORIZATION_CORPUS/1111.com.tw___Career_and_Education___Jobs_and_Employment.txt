       <script language="javascript" type="text/javascript">
        $(function () {
         try {
             var SearchRecordLog = localStorage.getItem('SearchRecordLog')+"";
             var arrSearchRecordLog = SearchRecordLog.split('@1111@');
             if (location.pathname.toLowerCase() == '/job-bank/job-index.asp') {
                 var rulestr = '@1@' + '@1111@';
                 if (SearchRecordLog == 'underfined' || SearchRecordLog == null) { SearchRecordLog = ''; }
                 if (SearchRecordLog.indexOf(rulestr) == -1) {
                     var savedata = location.search + rulestr;
                     SearchRecordLog = savedata + SearchRecordLog;
                     localStorage.setItem('SearchRecordLog', SearchRecordLog);
                 }
             }
             listStorage(SearchRecordLog);
            
         } catch (e) { };
     });

     function removeStorage(StorageName) {
         localStorage.removeItem(StorageName);
         listStorage('');
     }

     function listStorage(SearchRecordLog) {
         $.post('/job-bank/job-index-searchlist.asp', { 'viewlist': SearchRecordLog }, function (data) {
              $('#searchRecord').hide();
              $('#searchRecord').html(data);
              $('#listSearchLog').hide();
              if (data.indexOf('找工作') >= 0 || data.indexOf('找公司') >= 0) {
                  var intC = data.split('找工作').length - 1 + data.split('找公司').length - 1;
                  $('#showSearchLog').html('最近搜尋條件(' + intC + ')▼');
                  //var dataTxt = data.replace(/<\/?[^>]+>/gi, "").replace(/\s+/g, "");
                  var dataTxt = data.replace(/<\/?[^>]+>/gi, "");
                  var arrTxt = dataTxt.split("│");
                  if (arrTxt.length > 1) {
                      dataTxt = arrTxt[1].replace("找工作", "").replace("找公司", "");
                      $('.recently').html('最近搜尋條件：<a href="javascript:void(0);" class="rcbtn" onclick="$(\'#searchRecord\').show();">' + dataTxt.replace('清除搜尋紀錄', '') + '<img src="/job-bank/images/jbank_arrow_down2.png"></a>');
                      $('#recentbox').html('<h4>最近搜尋條件：</h4><p><a href="javascript:void(0);" class="rcbtn" onclick="$(\'#searchRecord\').show();">' + dataTxt.replace('清除搜尋紀錄', '') + '</a></p><a href="javascript:void(0);" class="rcbtn itembtn" onclick="$(\'#searchRecord\').show();">查看紀錄<i class="fa fa-angle-down" aria-hidden="true"></i></a>');
                      var $rcbtn = $(".rcbtn"),
                        $showRecjob = $(".showRecjob");
                      $rcbtn.click(function () {
                          $showRecjob.slideToggle("fast");
                          return false;
                      });
                  } else {
                      $('.recently').html('');
                      $('#recentbox').html('');
                  }
                  
              } else {
                  $('#showSearchLog').html('無最近搜尋條件▼');
                  $('.recently').html('');
                  $('#recentbox').html('');
              }
              $('#listSearchLog').show();
          });
     }

   function setCookie(name,value,days){
  if (days) {
    var date = new Date();
    date.setTime(date.getTime()+(days*24*60*60*1000));
    var expires = "; expires="+date.toGMTString();
  }
  else var expires = "";
  document.cookie = name+"="+value+expires+"; path=/";
}


function getCookie(c_name){
  if (document.cookie.length>0){
      c_start=document.cookie.indexOf(c_name + "=")
      if (c_start!=-1){ 
        c_start=c_start + c_name.length+1 
        c_end=document.cookie.indexOf(";",c_start)
        if (c_end==-1) c_end=document.cookie.length
        return unescape(document.cookie.substring(c_start,c_end))
      } 
  }
  return ""
}
