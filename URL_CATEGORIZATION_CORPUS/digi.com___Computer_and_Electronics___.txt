$(document).ready(function(){
  if (is_storage_supported) {
    localStorage.getItem("noPopup");
    localStorage.getItem("noPopupDate");

    //if today is when they get the popup again, reset noPopup to false and make a new date
    if ((new Date()).getTime() >= localStorage.getItem("noPopupDate")) {
      localStorage.setItem("noPopup", "false");
      localStorage.setItem("noPopupDate", new Date(+new Date + 12096e5).getTime());

    //if they don't have a date set for the popup expiration, set it
    } else if (localStorage.getItem("noPopupDate") === null) {
      localStorage.setItem("noPopupDate", new Date(+new Date + 12096e5).getTime());
    }

    //if noPopup doesn't exist, set it to null
    if (localStorage.getItem("noPopup") === null) {
      localStorage.setItem("noPopup", "false");
    }
    //30 second popup timer
    window.setTimeout(function(){
      if (localStorage.getItem("noPopup") === "false") {
        open_chatbox();
        localStorage.setItem("noPopup", "true");
      }

    }, popdelay );

  }
    //when you click the corner, the popup opens. if you opened it manually before the 30 second limit, we won't open it again
    $(".chatpopup .corner").click(function(){
      if (opened) {
        close_chatbox();
      } else {
        open_chatbox();
      }
      if (is_storage_supported) { localStorage.setItem("noPopup", "true"); }
    });
});
