function initialize() {
  // Initialize the map
  map = new google.maps.Map(document.getElementById('map_canvas'), {
    center: center,
    zoom: zoom,
    mapTypeControl: true,
    mapTypeControlOptions: {
      style: google.maps.MapTypeControlStyle.DROPDOWN_MENU
    },
    mapTypeId: google.maps.MapTypeId.ROADMAP
  });

  layer = new google.maps.FusionTablesLayer(tableid);
  layer.setQuery("SELECT LATITUDE FROM  " + tableid);
  layer.setMap(map);
  var queryText = encodeURIComponent("SELECT 'CRIME_TYPE','MOTIVE_TYPE','WEAPON_TYPE','LATITUDE','LONGITUDE','PICTURE','DATE_YEAR','GENDER'  FROM 891757 WHERE 'DATE_YEAR' = 2012");

  query.send(getData);

  var legendDiv = document.createElement('DIV');
  var legend = new Legend(legendDiv, map);
  legendDiv.index = 1;
  map.controls[google.maps.ControlPosition.RIGHT_BOTTOM].push(legendDiv);

 
}

function zoomtoaddress() {

  // Use the geocoder to geocode the address
  geocoder.geocode( { 'address': document.getElementById("address").value }, function(results, status) {
    // If the status of the geocode is OK
    if (status == google.maps.GeocoderStatus.OK) {
      // Change the center and zoom of the map
      map.setCenter(results[0].geometry.location);
      map.setZoom(14);
    var contentString = 'You Zoomed to this map Location<BR>'+
			'Close this Info-window and<BR>'+
			' drag this blue marker if it get is your way';
    var infowindow = new google.maps.InfoWindow({
        content: contentString
    });
         var marker = new google.maps.Marker({
             map: map,
 icon: new google.maps.MarkerImage("place.png"),
    animation: google.maps.Animation.BOUNCE,
	title:"You Zoomed to this Address",
	draggable:true,
        position: results[0].geometry.location
         });  
    google.maps.event.addListener(marker, 'click', function() {
      infowindow.open(map,marker); });   
    } 
  });
}

 // Reset the zoom, center, and FusionTablesLayer query
function reset() {
  map.setCenter(center);
  map.setZoom(zoom);
}

function getData(response) {
  table = new google.visualization.Table(document.getElementById('visualization'));
  datatable = response.getDataTable();
  table.draw(datatable, {sortColumn:6,sortAscending:false,showRowNumber:true,allowHtml:true});
 
 
  // Add a select listener to the table. When someone clicks a row, 
  // the selectHandler function is called
  google.visualization.events.addListener(table, 'select', selectHandler);
}
 
 
function selectHandler() {
  // Get the lat/lng values from the Fusion Tables data table
  var selection = table.getSelection();
  var lat = datatable.getValue(selection[0].row, 3);
  var lng = datatable.getValue(selection[0].row, 4);
  var position = new google.maps.LatLng(lat, lng);
  var crime = datatable.getValue(selection[0].row, 0);
  var motive = datatable.getValue(selection[0].row, 1);
  var weapon = datatable.getValue(selection[0].row, 2);
  var picture = datatable.getValue(selection[0].row, 5);
 var gender = datatable.getValue(selection[0].row, 7);

  
  // Create an infowindow at the lat/long coordinates
  // Set the content and open on the map
 
  if(infowindow) infowindow.close();
  else infowindow = new google.maps.InfoWindow();
   infowindow.setContent('<IMG SRC='+picture+' width="100"><BR><FONT SIZE=-1>'+gender+' victim in an incident of:'+crime+'<BR>Thought to be motivated or linked to: '+motive+'<BR><br>'+weapon+' in the incident<hr><a href="javascript:map.setCenter(new google.maps.LatLng'+position+');map.setZoom(map.getZoom()+5);">Zoom in</a>&nbsp;&nbsp;&nbsp;&nbsp;<input type="button" onclick="reset()" value="Centralise">&nbsp;&nbsp;&nbsp;&nbsp;<a href="javascript:map.setCenter(new google.maps.LatLng'+position+');map.setZoom(map.getZoom()-3);">Zoom Out</a></FONT><HR><FONT SIZE=-2><i>Zoom in and click on map marker for more details<BR>Locations are approximate</i></FONT><BR><A HREF=http://www.citizensreportuk.org/news/2011/05/05/comment-on-crimes-reported-on-citizens-report/>Comment</A>');


 
 
 
  infowindow.setPosition(new google.maps.LatLng(lat, lng));
  infowindow.open(map);
}
