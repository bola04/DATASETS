  <script>
   function validateandsend() {
        if ($('#makeapaymentform').valid()) {
            $('#makeapaymentform').find('div.form-group').each(function () {
                if ($(this).find('span.field-validation-error').length == 0) {
                    $(this).removeClass('has-error');
                }
            });
            $('#CustRefID').val($('#DeceasedName').val());
            var GHKarray = $('#Location').val().split(',');
            $('#Gateway_ID').val(GHKarray[0]);
            $('#HostedKey').val(GHKarray[1]);
            $('#makeapayment').modal('hide');
            $('#makeapaymentform').submit();
            $('#DeceasedName').val('');
            $('#Location').val('');
            $('#CustRefID').val('');
            $('#Gateway_ID').val('');
            GHKarray = [];
        }
        else {
            $('#makeapaymentform').find('div.form-group').each(function () {
                if ($(this).find('span.field-validation-error').length == 0) {
                    $(this).removeClass('has-error');
                }
            });
            $('#makeapaymentform').find('div.form-group').each(function () {
                if ($(this).find('span.field-validation-error').length > 0) {
                    $(this).addClass('has-error');
                }
            });
        }
    }
  </script>
