        adLoadVastTimeout: 8000,        adLoadMediaTimeout: 15000,        adParser: 'ima',        flashFirst: !isHTML5Video    };
    var elementID = 'rmpPlayer';
    var rmpContainer = document.getElementById(elementID);
    var rmp = new RadiantMP(elementID);

    var onAdContentResumeRequested;
    var onAdDestroyed;
    var onAdError;
    var onAdStarted;

    // this function is called when the ad-cycle has completed and content is about to resume
    var unWireEvents = function () {
        if (debug) {
            console.log('RMP unWireEvents');
        }
        if (onAdContentResumeRequested && onAdDestroyed && onAdError && onAdStarted) {
            rmpContainer.removeEventListener('adcontentresumerequested', onAdContentResumeRequested);
            rmpContainer.removeEventListener('addestroyed', onAdDestroyed);
            rmpContainer.removeEventListener('aderror', onAdError);
            rmpContainer.removeEventListener('adstarted', onAdStarted);
        }
    };

    onAdContentResumeRequested = function () {
        if (debug) {
            console.log('RMP adcontentresumerequested');
        }
        if (adstarted < wantedPreroll && waterfall[waterfallIndex]) {
            rmp.pause();
            rmp.loadAds(waterfall[waterfallIndex]);
        } else {
            unWireEvents();
        }
    };

    onAdDestroyed = function () {
        if (debug) {
            console.log('RMP addestroyed');
        }
        if (adstarted < wantedPreroll && waterfall[waterfallIndex]) {
            rmp.pause();
            rmp.loadAds(waterfall[waterfallIndex]);
        } else {
            unWireEvents();
        }
    };

    onAdStarted = function () {
        waterfallIndex = 0;
        adstarted++;
        if (debug) {
            console.log('RMP adstarted: ' + adstarted);
        }
    };

    onAdError = function () {
        waterfallIndex++;
        if (debug) {
            console.log('RMP waterfallIndex: ' + waterfallIndex);
        }
        var errorType = rmp.getAdErrorType();
        var errorCode = rmp.getAdErrorCode();
        var errorMessage = rmp.getAdErrorMessage();
        var vastErrorCode = rmp.getAdVastErrorCode();
        if (debug) {
            console.log('RMP aderror');
            console.log('RMP type: ' + errorType);
            console.log('RMP code: ' + errorCode);
            console.log('RMP message: ' + errorMessage);
            console.log('RMP vastCode: ' + vastErrorCode); /* https://support.google.com/dfp_premium/answer/4442429?hl=en */
        }
    };

    // when player is ready wire app logic
    var onReadyWireEvents = function () {
        if (debug) {
            console.log('RMP ready');
        }
        rmpContainer.removeEventListener('ready', onReadyWireEvents);
        // depending on the ad parser the implementation changes a bit
        var adParser = rmp.getAdParser();
        if (adParser === 'ima') {
            rmpContainer.addEventListener('adcontentresumerequested', onAdContentResumeRequested);
        } else {
            rmpContainer.addEventListener('addestroyed', onAdDestroyed);
        }
        rmpContainer.addEventListener('aderror', onAdError);
        rmpContainer.addEventListener('adstarted', onAdStarted);
    };

    // when ready wire event handlers
    rmpContainer.addEventListener('ready', onReadyWireEvents);

    rmp.init(settings);

})();
