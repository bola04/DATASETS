	    function getVideoPlayer(videoId) {
	        for (var i = 0; i < videoPlayers.length; i++) {
	            if (videoPlayers[i].options.video_id === videoId) {
	                return videoPlayers[i];
	            }
	        }
	    }

	    function pauseAllVideos() {
	        for (var i = 0; i < videoPlayers.length; i++) {
	            videoPlayers[i].player.pauseVideo();
	        }
	    }

	        for (var i = 0; i < imgWrappers.length; i++) {

	    function setUpVideos() {
	        if (slideVideos.length) {
	            buildVideos($(slideVideos[0]));
	        }
	    }

	    function buildVideos(slideVideo) {
	        var autoplay = 1;
	        var tempactiveSlide = getActiveSlide();
	        if (!tempactiveSlide) {
	            return;
	        }

	        if (!tempactiveSlide.hasClass('video-slide') || videoPlayers.length >= 1) {
	            autoplay = 0;
	        }

	        var videoId = slideVideo.attr("data-sv-video-id");
	        var videoType =  slideVideo.attr("data-sv-video-type");
	        var videoAudio =  slideVideo.attr("data-sv-video-audio");
	        var volume = 0;

	        if (videoAudio === "true") {
	            volume = 50;
	        }

	        new vp().init({
	            options: {
	                parent: slideVideo, 
	                type: videoType,
	                video_id : videoId,
	                iframe_id: 'video-iframe-' + videoId,
	                frameborder: 0,
	                scrolling: 'no',
	                width: "100%",
	                height: "1080",
	                volume: volume,
	                settings: {
	                   autoplay: autoplay,
	                    loop: 1,
	                    fullscreen: 1,
	                    controls : 0,
	                    modestbranding : 1,
	                    background: 1,
	                    vq: '1080',
	                    rel: 0,
	                    showinfo : 0,
	                    wmode: "transparent",
	                },
	                events: {
	                    ready: onReady,
	                    play: onPlay,
	                    pause: onPause,
	                },
	            }
	        }, onPlayerInit);
	    }

	    function onPlayerInit(_videoPlayer) {
	        videoPlayers.push(_videoPlayer);
	        [].shift.call(slideVideos);
	        setUpVideos();
	    }

	    function onPlay() {
	        if ($(window).width() > 1024) {
	            setTimeout(function() {
	                var activeSlide = getActiveSlide()
	                var videoPlayer = getVideoPlayer(activeSlide.attr("data-sv-video-id"));
	                if (videoPlayer) {
	                    activeSlide.find(".imageOuter").css('opacity', '0');
	                }
	            }, 2000)
	        }
	    }

	    function onReady() {
	        //setImage(getActiveSlide());
	        if ($(window).width() > 1024) {
	            setTimeout(function() {
	                var activeSlide = getActiveSlide()
	                var videoPlayer = getVideoPlayer(activeSlide.attr("data-sv-video-id"));
	                if (videoPlayer) {
	                    activeSlide.find(".imageOuter").css('opacity', '0');
	                }
	            }, 2000)
	        }
	    }

	    function onPause() {}

	    $('.video-mute').on('click', function () {
	        var activeSlide = getActiveSlide()
	        var videoPlayer = getVideoPlayer(activeSlide.attr("data-sv-video-id"));
	        if (!videoPlayer) {
	            return;
	        }
	        if ($(this).hasClass('mute')) {
	            $(this).removeClass('mute');
	            if ($(this).hasClass('youtube')) {
	                videoPlayer.player.setVolume(50);
	            } else {
	                videoPlayer.player.setVolume(1);
	            }
	        } else {
	            $(this).addClass('mute');
	            videoPlayer.player.setVolume(0);
	        }
	    });

	    $(window).on('resize', lodash.debounce(function() {
		    if(!disable_video_autoplay) {
		    	manageVideoState(getActiveSlide());
		    }		    
	    }, 150));

	        }
