    "setHiddenDataWhere": function (fieldValue, results) {
        var fieldName = '';
        var domFieldInitialName = 'homelink_homebundle_searchtype';
        if (fieldValue == "location") {
            fieldName = '_location';
        }

        $.each(results[0].address_components, function (index, addressComponent) {
            addressComponentType = addressComponent.types[0];
            addressComponentShortName = addressComponent.short_name;

            var arrayAddressType = {
                'neighborhood': "neighbor",
                'administrative_area_level_5': "adminLevel5",
                'administrative_area_level_4': "adminLevel4",
                'administrative_area_level_3': "adminLevel3"
            };

            if (addressComponentType in arrayAddressType) {
                addressComponentType = arrayAddressType[addressComponentType];
            }
            if (addressComponentType === 'locality' || addressComponentType === 'colloquial_area') {
                addressComponentType = "locality";
                if (fieldValue === "location") {
                    addressComponentType = "city";
                }
            }
            if (addressComponentType === 'administrative_area_level_1') {
                addressComponentType = 'adminLevel1';
                if (fieldValue === "location") {
                    addressComponentType = 'state';
                }
            }
            if (addressComponentType === 'administrative_area_level_2') {
                if (fieldValue === "location") {
                    addressComponentType = 'county';
                } else {
                    addressComponentType = 'adminLevel2';
                }
            }
            if (addressComponent.short_name === "County Galway") {
                document.getElementById(domFieldInitialName + fieldName + '_type').value = "administrative_area_level_1";
            }
            if (addressComponentType === 'establishment') {
                addressComponentType = 'country';
                if (addressComponentShortName == 'Ireland') {
                    addressComponentShortName = 'IE';
                }
            }

            if (document.getElementById(domFieldInitialName + fieldName + '_' + addressComponentType) != null)
                document.getElementById(domFieldInitialName + fieldName + '_' + addressComponentType).value = addressComponentShortName;
        });
    },
    "setRadius": function (locationGeometry) {
        var centerLng = locationGeometry.location.lng;
        var centerLat = locationGeometry.location.lat;
        var NELat, NELng = null;

        $('#homelink_homebundle_searchtype_longitude').val(centerLng);
        $('#homelink_homebundle_searchtype_latitude').val(centerLat);
        $('#homelink_homebundle_searchtype_centerMainLat').val(centerLat);
        $('#homelink_homebundle_searchtype_centerMainLng').val(centerLng);

        if (locationGeometry.bounds) {
            NELat = locationGeometry.bounds.northeast.lat / 57.2958
            NELng = locationGeometry.bounds.northeast.lng / 57.2958
        } else {
            NELat = locationGeometry.viewport.northeast.lat / 57.2958
            NELng = locationGeometry.viewport.northeast.lng / 57.2958
        }
        centerLat = centerLat / 57.2958
        centerLng = centerLng / 57.2958

        var radiusKm = getRadius(NELat, NELng, centerLat, centerLng);
        $('#homelink_homebundle_searchtype_radius').val(radiusKm);
        $('#homelink_homebundle_searchtype_radiusMain').val(radiusKm);
    },
