var minAnimation, minAnimationS, minAnimationCadreCel

// Fonction démarrant l'animation
function DemarrerAnim()
{
	// Début de l'animation
	minAnimation = window.setInterval("ChangerCitation()", delai)
	minAnimationS = window.setInterval("ChangerSocietes()", delaiSocietes)
	minAnimationCadreCel = window.setInterval("ChangerCadreCel()", delai)
}

// Fonction changeant la citation
function ChangerCitation()
{
	if(indiceCitation == 3)
		indiceCitation = 0
	
	document.getElementById('txtCitations').innerHTML = vectCitations[indiceCitation]
	document.getElementById('txtCitationsB').innerHTML = vectCitationsB[indiceCitation]
	indiceCitation++	
}

function ChangerSocietes()
{
	var i = 1;
	if(indiceSocietes == TotalSocietes + 1)
		indiceSocietes = 1
	
	for (i=1; i<TotalSocietes+1; i++)
	{
		if(indiceSocietes == i)
			document.getElementById('societe' + String(i)).style.display = "inline";
		else
			document.getElementById('societe' + String(i)).style.display = "none";
	}
	indiceSocietes++
	
	i = 1;
	if(indicePartenaires == TotalPartenaires + 1)
		indicePartenaires = 1
	
	for (i=1; i<TotalPartenaires+1; i++)
	{
		if(indicePartenaires == i)
			document.getElementById('partenaire' + String(i)).style.display = "inline";
		else
			document.getElementById('partenaire' + String(i)).style.display = "none";
	}
	indicePartenaires++
}

function ChangerCadreCel()
{
	if(document.getElementById('CadreHaut0')) {
		if(indiceCadreCel == 2)
			indiceCadreCel = 0
		
		for (i=0; i<2; i++)
		{
			if(indiceCadreCel == i)
				document.getElementById('CadreHaut' + String(i)).style.display = "inline";
			else
				document.getElementById('CadreHaut' + String(i)).style.display = "none";
		}
		indiceCadreCel++	
	}
}

// Fonction arrêtant l'animation
function ArreterAnim()
{
	// Arrêt de l'animation
	window.clearInterval(minAnimation);
	window.clearInterval(minAnimationS);
	window.clearInterval(minAnimationCadreCel);
}
