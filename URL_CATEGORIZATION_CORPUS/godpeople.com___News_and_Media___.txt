			'top':(pos.top-10)+'px'
		});

		var pos = $("#section_zzim").offset();
		var dH = $(document).height();
		$('.gnb_login_layer').hide();
		$("#zzim_bottom_layer").hide().css({
			'display':'block',

		});

			'display':'block',
			'left':pos.left+'px',

		});

		$("#zzim_bottom_layer").hide();
		$("#zzim_login_layer").hide();
	}
}
