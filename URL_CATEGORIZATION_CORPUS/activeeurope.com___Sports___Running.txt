   <script type="text/javascript">
    var SSAP = window.SSAP = window.SSAP || {};
SSAP.instant_ads_js_url = 'https://marketinghub.active.com/js/ads/instantAds2.js?2375'
SSAP.searchConfig = {"page_id":14353,"ad_unit_1":335353,"ad_unit_2":335354,"ad_unit_3":335355};

(function(window, SSAP){
  var loadJs = function() {
    var deferred = $.Deferred();

    ACTIVE.loadJsFile(SSAP.instant_ads_js_url, false, true, function() {
      SSAP.featuredJsLoaded = true;
      deferred.resolve();
    });

    return deferred.promise();
  };

  var requestJsonAds = function(keywords) {
    var adPlaceholders = [
      'autosuggest-ia-placeholder-0',
      'autosuggest-ia-placeholder-1',
      'autosuggest-ia-placeholder-2'
    ];

    var addPlaceholders = function() {
      for(var index = 0; index < adPlaceholders.length; index++) {
        $('<div id="'+ adPlaceholders[index] +'"></div>').appendTo('body');
      }
    };

    var deferred = $.Deferred();

    if (!SSAP.adPlaceholdersAdded) {
      addPlaceholders();
      SSAP.adPlaceholdersAdded = true;
    }

    window.ssap_ads1 = new SSAP_ADS({
      // The item1...item5 will match splited asset name from IA
      item1: keywords || '',
      pageId: SSAP.searchConfig.page_id,
      meta_interest: ACTIVE.isKidsPage()? 'kids': '',
      config:[
        {
          listings_div: adPlaceholders[0],
          adUnits : [SSAP.searchConfig.ad_unit_1]
        },
        {
          listings_div: adPlaceholders[1],
          adUnits : [SSAP.searchConfig.ad_unit_2]
        },
        {
          listings_div: adPlaceholders[2],
          adUnits : [SSAP.searchConfig.ad_unit_3]
        }
      ],
      callback: function() {
        var jsonAds = ssap_ads1.getJsonAds();
        jsonAds = jsonAds.slice(0, 2);
        deferred.resolve(jsonAds);
      }
    });
    ssap_ads1.initAds();

    return deferred.promise();
  };

  SSAP.getSuggestedAds = function(keywords) {
    // Pages that IA has already been there
    if (SSAP.featuredJsLoaded) {
      return requestJsonAds(keywords);
    }
    // Pages that IA are not implemented
    else {
      var deferred = $.Deferred();
      loadJs().then(function() {
        requestJsonAds(keywords).then(function(jsonAds) {
          deferred.resolve(jsonAds);
        });
      });
      return deferred.promise();
    }
  };

})(window, SSAP);
