    function SetActiveDiv() {
    $('.divtop').each(function() {
        var imgsrc = $(this).attr('src');
        $(this).attr("src", imgsrc.replace("-on.png", ".png"));
    });
    var $div = "none";
    if($div == "none") { $div = ""; }
    if($div) {
        $srz = $("#link"+$div).find('.divtop').attr("src");
        if ($srz.indexOf("-on.png") == -1) $("#link" + $div).find('.divtop').attr("src", $srz.replace(".png", "-on.png"));
    }
}

$(document).ready(function() { SetActiveDiv(); });

$(function(){
    $('.divtop').click(function() {
        if($(this).parent().parent().find('div[class^="container-"]').css('display') != 'none'){

            $(this).parent().parent().find('div[class^="container-"]').hide();
            $srz = $(this).attr("src");
            $(this).attr("src", $srz.replace("-on.png", ".png"));

            $srz = $(this).attr("src");
            if ($srz.indexOf("-on.png") == -1) $(this).attr("src", $srz.replace(".png", "-on.png"));
        }
        return false;
    });
