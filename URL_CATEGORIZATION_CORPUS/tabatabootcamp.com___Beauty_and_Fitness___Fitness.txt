        var svg = d3.select("#mapEvents")
          .attr("width", width)
          .attr("height", height);

          .projection(projection);

        d3.json("/www.tabatabootcamp.com_templates/json/us.json", function(error, us) {

          svg.append("path")
          .attr("class", "states")
          .datum(topojson.feature(us, us.objects.states))
          .attr("d", path);

          // add circles to svg
          var locations = svg.append("svg:g")
            .attr("class", "locations");

          // Read the paths of the Events from a JSON file - Plot the points of the Events within the SVG


          var locations = d3.select(".locations").selectAll('circle')
            .data(coordinates);

          locations.enter().append("svg:circle")
            .attr("cy", function(d) { return projection(d.coordinates)[1];})
            .attr("cx", function(d) { return projection(d.coordinates)[0];})
            .attr("id", function(d) { return d.label})
            .attr("r", 2.5);


        });
