    //<![CDATA[
    var promotion = {};
    (function() {
      var updatePeriodically = true;
      var promotionArray = ['promotion-0','promotion-1','promotion-2','promotion-3','promotion-4','promotion-5','promotion-6','promotion-7'];
      var show = Math.floor(Math.random()*promotionArray.length);
      //var show = 0;  // show the first banner first
      
  
      promotion.previous = function()
      {
        decrementShow();
        changePromotion(show);
      };
      promotion.next = function()
      {
        incrementShow();
        changePromotion(show);
      };
      promotion.go = function(index)
      {
        show = index
        updatePeriodically = false;
        changePromotion(index);
        return false;
      };
      
  
      function incrementShow() {
        show +=1;
        if (show >= promotionArray.length)
        {
          show = 0;
        }
      }
      
  
      function decrementShow() {
        show -=1;
        if (show < 0)
        {
          show = promotionArray.length;
        }
      }
      
  
      function changePromotion(promotionIndex) {
        for (var i=0; i < promotionArray.length; i++)
        {
          if (i == promotionIndex)
          {
            document.getElementById(promotionArray[i]).style.display = "";
            document.getElementById("promotion-"+i+"-selector").className = "promotion-number-selector-active";
          }
          else
          {
            document.getElementById(promotionArray[i]).style.display = "none";
            document.getElementById("promotion-"+i+"-selector").className = "promotion-number-selector";
          }
        }
      }
  
      function scheduleUpdatePromotionPeriodically()
      {
         setTimeout(updatePromotionPeriodically, 10000);
      }
  
      function updatePromotionPeriodically() {
        if (!updatePeriodically)
          return;
        promotion.next();
        scheduleUpdatePromotionPeriodically()
      }
      
  
      document.getElementById("promotion-selectors").style.display = "";
      if (!updatePeriodically) 
  
        promotion.go(show);
      else
      {
        changePromotion(show);
        scheduleUpdatePromotionPeriodically()
      }
  })();
  //]]>
