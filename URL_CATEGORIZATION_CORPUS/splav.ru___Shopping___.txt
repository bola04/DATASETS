function getFilterFlags(){
    var cook=new SplavControls.Cookie("SplavFilter");
    flags=cook["flags"];
    if(!flags)
        return 0;
    return flags;
    }

var btnTourism, btnCamouflage, btnFishing, btnGuard,btnClearFilters;
var mTourism=1;
var mCamouflage=8;
var mFishing=16;
var mGuard=6;

function getFilterButtonsState(){
    var flags=0;
    if(btnTourism.getChecked())    flags|=mTourism;
    if(btnCamouflage.getChecked()) flags|=mCamouflage;
    if(btnFishing.getChecked())    flags|=mFishing;
    if(btnGuard.getChecked())      flags|=mGuard; 
    return flags;
    }

function clearFilterFlags(){
    var changed=false;
    if(btnTourism.getChecked()){btnTourism.setChecked(0); changed=true;}
    if(btnCamouflage.getChecked()){btnCamouflage.setChecked(0); changed=true;}
    if(btnFishing.getChecked()){btnFishing.setChecked(0); changed=true;}
    if(btnGuard.getChecked()){btnGuard.setChecked(0); changed=true;}
    return changed;
    }

function clearFiltersButtonClick(e){
    this.setChecked(false);
    var cleared=clearFilterFlags();
    if(cleared)
        buttonClick(e);
    if(bFiltersEnabledThisPage)
        e.preventDefault();
}
 
 
function filterButtonClick(e){
    if(!bFiltersEnabledThisPage)
        this.setChecked(true);
    //radio buttons - begin
    if(this.getChecked()){
       var  buttons=[btnTourism,btnCamouflage,btnFishing,btnGuard];
        var i;
        for(i=0;i<buttons.length;i++)
            if(buttons[i]!=this)
                buttons[i].setChecked(false);
        }
    //radio buttons - end
        
    buttonClick.call(this,e);
    }; 

function buttonClick(e){
    var flags=getFilterButtonsState(); 
    var cl=document.getElementById("clrFiltersBtn");
    if(btnClearFilters)
        btnClearFilters.setChecked(!!flags);
    var cook=new SplavControls.Cookie("SplavFilter");
    cook["flags"]=flags;
    cook.store(null,'/');

    if(bFiltersEnabledThisPage){
        e.preventDefault();
        SplavControls.showProcessing(e,function(){
            if(typeof toggleTreeFilters == 'function')  toggleTreeFilters(flags);
            if(typeof toggleDetailFilters == 'function')toggleDetailFilters(flags);
            });
        }
    }; 

$(document).ready(function(){

    btnTourism     = SplavControls.jClassButton(".mod_tourism",   "on","off",filterButtonClick);
    btnCamouflage  = SplavControls.jClassButton(".mod_camouflage","on","off",filterButtonClick);
    btnFishing     = SplavControls.jClassButton(".mod_fishing",   "on","off",filterButtonClick);
    btnGuard       = SplavControls.jClassButton(".mod_guard",     "on","off",filterButtonClick);
    btnClearFilters= SplavControls.jsClassButton("#clrFiltersBtn", "on","off",clearFiltersButtonClick);

    var cook=new SplavControls.Cookie("SplavFilter");
    var flags=parseInt(cook["flags"]);
    if(!flags || !bFiltersEnabledThisPage)
        flags=0;
    btnTourism.setChecked(!!(flags&mTourism));
    btnCamouflage.setChecked(!!(flags&mCamouflage));
    btnFishing.setChecked(!!(flags&mFishing));
    btnGuard.setChecked(!!(flags&mGuard));
    if(btnClearFilters)
        btnClearFilters.setChecked(!!flags);
