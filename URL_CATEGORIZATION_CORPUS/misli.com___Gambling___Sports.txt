        function SearchControl(v) {

            var sval = $("#txtSearch").val();

            if (sval == "ARAMA" || sval == "" || sval.length < 3) {
                return false;
            }
            else {
                if (v != undefined) {
                    $("#dvSearchInner").html("");            
                }
                $("#dvSearchInner").show();
                setTimeout(function() { $.ajax({
                    type: "POST",
                    url: "/Default.aspx/Search",
                    data: "{key: '" + sval + "'}",
                    contentType: "application/json; charset=utf-8",
                    dataType: "json",
                    async: true,
                    cache: false,
                    success: function(msg) {
                        if (msg.d != "") {
                            $("#dvSearchInner").show();
                            $("#dvSearchInner").html(msg.d);
                        }
                    },
                    error: function(xmlHttpRequest, status, err) {
                        ShowModalPopUp("", "Lütfen tekrar deneyiniz!", "0");
                    }
                });
                },1);
            }
        }
