				}
				else
				{				    
					$(".borderedContent").html(xmlhttp.responseText);
					$(".slider").find(".active").removeClass("active");			  			  
					$(e.target).parent().parent().addClass("active");
				}	

				if ($('#btnShowMenuOptions').is(":visible"))
				{
					$('body').removeClass('active');

				else
				{
					$("#templateContent").css("height", $("#menu_content").height());

	}
