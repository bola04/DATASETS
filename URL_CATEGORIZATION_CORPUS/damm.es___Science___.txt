function quitar_acentos(cadena) {
	cadena = cadena.replace(/[ÁÀÂÃÄ]/g, "A");
	cadena = cadena.replace(/[ÉÈÊË]/g, "E");
	cadena = cadena.replace(/[ÍÌÎÏ]/g, "I");
	cadena = cadena.replace(/[ÓÒÔÕÖ]/g, "O");
	cadena = cadena.replace(/[ÚÙÛÜ]/g, "U");
	cadena = cadena.replace(/[áàâãªä]/g, "a");
	cadena = cadena.replace(/[éèêë]/g, "e");
	cadena = cadena.replace(/[íìîï]/g, "i");
	cadena = cadena.replace(/[óòôõºö]/g, "o");
	cadena = cadena.replace(/[úùûü]/g, "u");
	cadena = cadena.replace(/Ñ/g, "N");
	cadena = cadena.replace(/ñ/g, "n");
	cadena = cadena.replace(/ç/g, "c");
	cadena = cadena.replace(/Ç/g, "C");
	return cadena;
}

function primeros_caracteres (n, texto)
{
	/* Función primeros_caracteres(n, texto) */
	/* Devuelve los n primeros caracteres de una cadena de texto */

	/* quitamos los <br> que pueda haber en el texto */
	texto=texto.replace(/<[^<>]+>/g, " ");
	
	/* mira si la longitud del texto es mayor que primeros_caracteres */
	if (n<texto.length)
	{
		/* a partir de la posición 'primeros_caracteres' busca la posición del primer espacio en blanco */
		parte_final_texto= texto.substring(n);
		posicion_espacio=parte_final_texto.indexOf(' ');
		/* escribe el texto desde el inicio a la posición del primer espacio en blanco después de la posición de 'primeros_caracteres' */
		return(texto.substring(0,n+posicion_espacio)+ '...');
	}
	else
	{
		/* la longitud del texto es menor que 'primeros_caracteres' */
		return(texto);	
	};
}

function mostrar_imagen_con_alt (tag_imagen)
{
  /* FUNCIÓN PARA PONER EL ALT A LAS IMÁGENES */

	/* recorta el nombre a partir de la extensión del archivo de imagen*/

	punto=tag_imagen.lastIndexOf('.');
	alt=tag_imagen.substring(0,punto);

	/* busca la última /  */
	barra=alt.lastIndexOf('/');
	alt=alt.substring(barra+1);

	/* sustituye los _ por espacio en blanco */
	alt=alt.replace(/_/g, " ");

	/* construye el tag imagen con el alt */
		/* ADVERTENCIAS: el alt tiene que ir sin espacio entre el = y la ' o el  \. 
		   El alt tiene que ir en minúscula. 
		   Los nombres de las imágenes tienen que ir con las palabras separadas por _ */
	posicion_inicial_alt=tag_imagen.indexOf("alt=");
	
	if (tag_imagen.charAt(posicion_inicial_alt+4) == '\\')
	{
	posicion_final_alt=tag_imagen.indexOf("'", posicion_inicial_alt+6);
	}
	else {
	posicion_final_alt=tag_imagen.indexOf("'", posicion_inicial_alt+4);
	}
	
	/* el if es para que no escriba el alt='' cuando no hay imagen a mostrar*/

	if(tag_imagen){
	tag_imagen=tag_imagen.substring(0,posicion_inicial_alt) + "alt='"+ alt + "'" + "title='" + alt + "'" + tag_imagen.substring(posicion_final_alt);
	/* */
	}
	 
	document.write(tag_imagen);
}

		var search = Name + "="
		var returnvalue = "";
		if (document.cookie.length > 0) {
			offset = document.cookie.indexOf(search)
			if (offset != -1) { // if cookie exists
				offset += search.length

				// set index of beginning of value
				end = document.cookie.indexOf(";", offset);

				// set index of end of cookie value
				if (end == -1)
					end = document.cookie.length;

				returnvalue=unescape(document.cookie.substring(offset, end))
			}
		}
		return returnvalue;

                </p>

                 <br style="FONT-WEIGHT: bold"/>
                 <span style="FONT-WEIGHT: bold">
                  - COOKIES PERMANENTES
                 </span>
                </p>
                <p dir="ltr">

                </p>

                 </span>

Por ejemplo, el idioma, el tipo de navegador a través del cual accede al servicio, el diseño de contenidos seleccionado, geolocalización del terminal y la configuración regional desde donde se accede al servicio.
                </p>
                <p dir="ltr">
                 <br style="FONT-WEIGHT: bold"/>
                 <span style="FONT-WEIGHT: bold">
                  - COOKIES PUBLICITARIAS
                 </span>
                </p>
                <p dir="ltr">

Permiten adecuar el contenido de la publicidad para que esta sea relevante para el usuario y para evitar mostrar anuncios que el usuario ya haya visto.
                </p>
                <p dir="ltr">
                 <br style="FONT-WEIGHT: bold"/>
                 <span style="FONT-WEIGHT: bold">
                  - COOKIES DE ANÁLISIS ESTADÍSTICO
                 </span>
                </p>
                <p dir="ltr">

La información recogida mediante este tipo de cookies se utiliza en la medición de la actividad de los sitios Web, aplicación o plataforma y para la elaboración de perfiles de navegación de los usuarios de dichos sitios, con el fin de introducir mejoras en el servicio en función de los datos de uso que hacen los usuarios.

   function getCookie(c_name){
	var c_value = document.cookie;
	var c_start = c_value.indexOf(" " + c_name + "=");
	if (c_start == -1){
		c_start = c_value.indexOf(c_name + "=");
	}
	if (c_start == -1){
		c_value = null;
	}else{
		c_start = c_value.indexOf("=", c_start) + 1;
		var c_end = c_value.indexOf(";", c_start);
		if (c_end == -1){

		}

	}
	return c_value;
}
