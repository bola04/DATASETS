        (function() {
        var lc = document.createElement('script'); lc.type = 'text/javascript'; lc.async = true;
        lc.src = ('https:' == document.location.protocol ? 'https://' : 'http://') + 'cdn.livechatinc.com/tracking.js';
        //add an onload function to livechat
        lc.setAttribute('onload', 'watchForLiveChat()');
        var s = document.getElementsByTagName('script')[0]; s.parentNode.insertBefore(lc, s);
        })();

        function watchForLiveChat(){

                // using setTimeout to give livechat a sec to add its DOM elements
                setTimeout(function(){

                    // watch the livechat conatainer for style changes...

                    var reportsButton = document.querySelector('.button-today');
                    var livechatContainer = document.getElementById('livechat-compact-container');
                    var livechatFull = document.getElementById('livechat-full');

                    var observer = new MutationObserver(function(mutationsList){
                        for(var mutation of mutationsList) {
                            if(mutation.attributeName == 'style'){
                                if (livechatFull.style.transform == 'translateY(0%)') {
                                    reportsButton.style.display = 'none';
                                }else{
                                    reportsButton.style.display = '';
                                }
                            }
                        }
                    });

                    observer.observe(livechatFull,{attributes:true});

                    // function that controls when to hide/show live chat

                    function hideLivechat() {
                        if (reportsButton.classList.contains('not-today')) {
                            livechatContainer.style.display = '';
                        }else if (!window.matchMedia("(min-width: 1024px)").matches){
                            livechatContainer.style.display = 'none';
                        } else {
                            livechatContainer.style.display = '';
                        }
                    }

                    // watch the reports button for style changes...

                    var observer = new MutationObserver(function(mutationsList){
                        for(var mutation of mutationsList) {
                            if(mutation.attributeName == 'class'){
                                hideLivechat();
                            }
                        }
                    });

                    observer.observe(reportsButton,{attributes:true});

                    // check whether we should hide live chat initially.

                    hideLivechat();

                    window.addEventListener('resize', function() {
                        hideLivechat();
                    });

                },1000);
            }
