    function Reset() {
        $('#divLoading').fadeOut('ease', function () {
            $('#home-bright').fadeIn('ease');
            $('#location-search').fadeIn('ease');
        });
        $('#location-search').fadeOut('ease', function () {
            $('#home-bright').fadeIn('ease');
            $('#location-search').fadeIn('ease');
        });
    }

    function LocationFound(locId) {
        if (locId && locId > 0) {
            $.ajax({
                url: '/Ajax/NearestLocationFoundAjax',
                contentType: 'application/html; charset=utf-8',
                data: {
                    intStoreID: locId
                },
                type: 'GET',
                dataType: 'html'
            }).success(function (result) {
                if (result == 'error') {
                    Reset();
                } else
                    $('#divLoading').fadeOut('ease', function () {
                        $('#location-search').html(result);
                        $('#location-search').fadeIn('slow');
                    });
            });
        }
    }
