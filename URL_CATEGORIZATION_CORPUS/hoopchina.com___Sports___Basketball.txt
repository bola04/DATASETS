    var $bottomTools = $('.bottom_tools');
    $(window).scroll(function() {
        var scrollHeight = $(document).height();
        var scrollTop = $(window).scrollTop();
        var $windowHeight = $(window).innerHeight();
        scrollTop > 100 ? $("#scrollUp").fadeIn(200).css("display", "block") : $("#scrollUp").fadeOut(200);
        $bottomTools.css("bottom", scrollHeight - scrollTop > $windowHeight ? 100 : $windowHeight + scrollTop + 100 - scrollHeight);
    });
    if (screen.width <= 1366) {
        $("body .content-wrap .bottom_tools").css("right", "3%");
    }
    $('#scrollUp').click(function(e) {
        e.preventDefault();
        $('body,html').animate({
                scrollTop: 0
            },
            500);
        return false;
    });
