    var Sresults;
            $.fn.typeahead.Constructor.prototype.blur = function () {
                var that = this;
                setTimeout(function () { that.hide(); }, 250);
            };

            var that = this;
            var url = "";
            $("#product-keyWords").typeahead({
                autoSelect: false,
                source: function (e) {
                    e = $.trim(e);
                    var result = [];

                    if (e.length < 2) {
                        return result;
                    }
                    var type = $("#dp_searchType").val();
                    $.ajax({
                        type: "get",
                        async: false,
                        timeout: 5000,
                        url: type == "article" ? "/Handlers/Catalog/GetAritlesByKey.ashx" : type == "question" ? "/Handlers/Catalog/GetQuestionsByKey.ashx" :
                            type == "video" ? "/Handlers/Catalog/GetVideosByKey.ashx" : type == "demoOrSample" ? "/Handlers/Catalog/GetdemoOrSamplesByKey.ashx" :
                            "/Handlers/Catalog/GetProductsByKey.ashx",
                        data: { k: e },
                        dataType: "json",
                        cache: true,
                        success: function (results) {
                            Sresults = results;
                            result = isNullOrEmpty(results) ? [] : _.map(results, function (r) {
                                return r.Name;
                            });

                        }
                    });

                    return result;
                },
                updater: function (name) {

                    var product = _.find(Sresults, function (p) {
                        return p.Name == name;
                    });
                    url = product.url;
                    $("#product-keyWords").val(product.Name);
                    that.setSelectedProduct(product);

                    return product.Name;
                }
            }).keyup(function (e) {
                if (e.keyCode == 13) {
                    if (url != "") {
                        AddKeywords($("#product-keyWords").val(), $("#dp_searchType").val());
                        location.href = url;
                    } else {
                        $("#product-btn-search").click();
                    }

                }

            });

            this.setSelectedProduct = function (product) {
                AddKeywords($("#product-keyWords").val(), $("#dp_searchType").val());
                location.href = product.url;
            };
   </script>
