               <script type="text/javascript">
                function shouldRedirectCanada(element) {        
        return (isCanadaSelected(element) && agencyHasNoCanadianProduct() && redirectCanadaAvailable() && interceptCanadaSell(element));
    }

    function isCanadaSelected(element) {
        var country = $(element).val();        
        return (country == '37');
    }

    function agencyHasNoCanadianProduct() {
        
        //return false if no agency, or agency has no products assigned for Canada
        var agencyNumber = $('input[type=text][id*=txtAgencyNumber]').val();

        if (typeof agencyNumber === 'undefined') {
            agencyNumber = $('input[type=hidden][id*=AgencyNumberHdn]').val();
        }

        if (typeof agencyNumber === 'undefined') {            
            agencyNumber = Cookies.get('AgentLink_AgencyNumber');

            if (typeof agencyNumber === 'undefined') {                
                agencyNumber = '';                
            }            

            return (agencyNumber == null || agencyNumber.length < 5) || !anyProductsAvailableInCanada(agencyNumber);
        }

        return (agencyNumber == null || agencyNumber == '' || agencyNumber.length < 5) ? true : !anyProductsAvailableInCanada(agencyNumber);        
    }

    function redirectCanadaAvailable() {
        var urlCanada = 'https://canada.travelinsured.com/';
        return urlCanada.length > 0;
    }

    function interceptCanadaSell(element) {
        if ($(element).val() == 37) {
            $(element).val(224);
            $('#redirectCanadaModal').modal('show'); 
            return true;
        }
        return false;
    }
    
    function anyProductsAvailableInCanada(agencyNumber) {

        var anyCanadianProduct = false;

        $.ajax({
            type: 'POST',
            url: '/Services/Agency/Agency.asmx/AnyProductsAvailableInCanada',
            async: false,
            data: '{"agencyNumber":\'' + agencyNumber + '\'}',
            contentType: 'application/json; charset=utf-8',
            dataType: 'json',
            success: function (result) {                
                anyCanadianProduct = eval('(' + result.d + ')');     
            }
        });

        return anyCanadianProduct;
    }

    var canadaTimeout = null;

    $(document).ready(function() {
        $('#redirectCanadaModal').appendTo($('#MainWrapper'));

        var urlCanada = 'https://canada.travelinsured.com/';
        var isInternal = false;
        
        if (isInternal) {
            $('#divHeader').hide();
            $('#btnOk').show();
            $('#btnCancel').hide();
            $('#divMessage1').hide();
            $('#divMessage2').show();
        } else {
            $('#btnOk').hide();
            $('#btnCancel').show();
            $('#redirectCanadaModal').on('show.bs.modal', function() { canadaTimeout = setTimeout(function() { window.location.href = urlCanada; }, 10000); });
            $('#divMessage1').show();
            $('#divMessage2').hide();
        }
    });

    function cancelCanadaRedirect() {
        if (canadaTimeout != null) {
            clearTimeout(canadaTimeout);
            canadaTimeout = null;
        }
    }
