			if ($("#barra_busqueda #contenedor > ul").hasClass('marcas'))
			{
				$("#barra_busqueda #contenedor > div").hide('fast');
				$("#barra_busqueda #contenedor > ul").removeClass('marcas')
				break;
			}
			
			//mostramos el loader
			$('#barra_busqueda #loader').show();

			$.getJSON('https://automovilescolombia.com/' + 'ajax/marcas', '', function(marcas) {
				var opciones = '';
				for (var i = 0; i < marcas.length; i++)
				{
					opciones += '<li><a href="https://automovilescolombia.com/vehiculos/' + marcas[i].titulo + '">' + marcas[i].nombre + '</a></li>' + "\n";
				}
				$("#barra_busqueda #contenedor #listados ul").html(opciones);
				$("#barra_busqueda #contenedor > ul").removeClass().addClass('marcas');
				$("#barra_busqueda #contenedor #listados h3").html('Listado de Marcas');
				
				$("#barra_busqueda #contenedor #listados").show('fast');
				//escondemos el loader
				$('#barra_busqueda #loader').hide();
				
			});

			break;

		case 'vehiculos/carrocerias':

			if ($("#barra_busqueda #contenedor > ul").hasClass('carrocerias'))
			{
				$("#barra_busqueda #contenedor > div").hide('fast');
				$("#barra_busqueda #contenedor > ul").removeClass('carrocerias')
				break;
			}
			
			//mostramos el loader
			$('#barra_busqueda #loader').show();
			
			$.getJSON('https://automovilescolombia.com/' + 'ajax/carrocerias_simples', '', function(carrocerias) {
				var opciones = '';
				for (var i = 0; i < carrocerias.length; i++)
				{
					opciones += '<li><a href="https://automovilescolombia.com/vehiculos/carroceria/' + carrocerias[i].titulo + '">' + carrocerias[i].nombre + '</a></li>' + "\n";
				}
				$("#barra_busqueda #contenedor #listados ul").html(opciones);
				$("#barra_busqueda #contenedor > ul").removeClass().addClass('carrocerias');
				$("#barra_busqueda #contenedor #listados h3").html('Listado de Carrocerías');
				$("#barra_busqueda #contenedor #listados").show('fast');

				//escondemos el loader
				$('#barra_busqueda #loader').hide();
			});

			
			break;

		case 'vehiculos/necesidades':
		case 'vehiculos':
			return;
			break;

		default:

			$("#barra_busqueda #contenedor > div").hide('fast');
			$('#barra_busqueda #loader').hide();
		}

		event.preventDefault();

	});
	
});
