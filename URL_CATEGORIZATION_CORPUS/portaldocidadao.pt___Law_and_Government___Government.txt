            var liSelected = "";   
var serveResourcesREST = "https://www.portaldocidadao.pt/home;jsessionid=4DFEA1C3E4757B8875CCFFE3FF924973.node-01?p_p_id=PortalAmaFOSearch_WAR_ptamaportalfoportlet&p_p_lifecycle=2&p_p_state=normal&p_p_mode=view&p_p_cacheability=cacheLevelPage&p_p_col_id=column-1&p_p_col_count=5";

function refreshHightlightEventSuggestionItems(){	// para activar e desactivar o evento de highlight na drop down de sugestoes
	
		$("li[id*='suggestionsItemContainer']").hover( function(){
								$("li[id*='suggestionsItemContainer']").css("background-color","");
								$(this).css("background-color", "lightgrey");
								
								liSelected = $("li[id*='suggestionsItemContainer']").index( $(this) ); 
					},
					function(){
							$(this).css("background-color", "");
							liSelected = "";
					});
}

function showSuggestions(obj){ // shows drop down with the suggestions
	
	var searchQuery = obj.value;

	if (searchQuery.length >= 3){ // at least 3 characters
		
		var keywords = searchQuery.replace( new RegExp("[?=&|!{}<>!@$%\\\\/\\\[\\\]\'\`]","g")," ");
		keywords = keywords.replace( new RegExp("\\s","g"),"+");//replace white spaces with +
		getSuggestions(keywords);// asks for suggestions
	
		if ( $( "ul[id*='suggestionsContainers']" ).children().length > 0){ // se existirem sugestoes
			$("#searchSuggestions").switchClass('suggestions-drop-down-hidden','suggestions-drop-down');
		
			refreshHightlightEventSuggestionItems(); 
		}
	
	} else { // nothing happens/ removes drop down
		$("[id*='suggestionsItemContainer']").hide();
		$("li[id*='suggestionsItemContainer']").remove();// remove content
		$("#searchSuggestions").switchClass('suggestions-drop-down','suggestions-drop-down-hidden');
		liSelected = "";
	}
}

$(document).keydown(function(e){ // highlights the selected suggestion in the drop down  (38=up, 40=down)
	
	if ( $(".suggestions-drop-down") != null ){// only if the drop down menu is opened	
		
	   if (e.keyCode == 40) {  // keydown --> ArrowDown  switch (e.key) {case "ArrowDown":  //do smt break;}
	        if(liSelected === "") {
	        	liSelected = 0;
	        } else if( (liSelected+1) < $("li[id*='suggestionsItemContainer']").length ) {
	        	liSelected++; 
	        }
	        $("li[id*='suggestionsItemContainer']").css("background-color", "");
	        $("li[id*='suggestionsItemContainer']:eq("+liSelected+")").css("background-color", "lightgrey"); 
	        return false;
	    }
	    if (e.keyCode == 38) { // keyup --> ArrowUp
	        if(liSelected === "") {
	        	liSelected = 0;
	        } else if(liSelected > 0) {
	        	liSelected--;            
	        }
	        $("li[id*='suggestionsItemContainer']").css("background-color", "");
	        $("li[id*='suggestionsItemContainer']:eq("+liSelected+")").css("background-color", "lightgrey"); 
	        return false;
	    }
	    
	    if (e.keyCode == 13){ // enter
			if (liSelected !== "")
				replaceQueriedText(  $( $("li[id*='suggestionsItemContainer']")[liSelected] )  ); // feeds the selected element into the input box
	    }  
	    
	}
});

function checkSuggestions( existSuggestions, keyWord){ // checks if any keyword correction was suggested
	
	if ( existSuggestions ) {
		jQuery("#didYouMean").html(keyWord);			
		jQuery("#didYouMeanBlock").show();
	} else {
		jQuery("#didYouMeanBlock").hide();
	}
}

function replaceQueriedText(obj){ // when a suggestion is clicked, it replaces the content of the search input box and makes a new search

	var didYouMean = $( obj ).text();

	$("[id*='formSearchInput']").val(didYouMean);
	
	$("#formSearchGo").click();
}

function updateSuggestionList(array){ // populates up to 3 suggestions

	$("[id*='suggestionsItemContainer']").hide(); // hide all previous suggestions
	$("li[id*='suggestionsItemContainer']").remove();

	var i = 0;
	var j = 1;
	for (i = 0, j = 1; i < array.length; i++,j++) {
		$( "ul[id*='suggestionsContainers']" ).append("<li id=\"suggestionsItemContainer"+j+"\" class=\"\"><a id=\"suggestionItem"+j+"\" class=\"\"  href=\"#\" onclick=\"replaceQueriedText(this);\" >"+array[i]+" </a></li>");
	}
	
	$("[id*='suggestionsItemContainer']").show(); 
	refreshHightlightEventSuggestionItems(); 
	
}

function getSuggestions(userInput){ // makes an ajax request to retrieve suggested search terms
	
	var url_final = setResourceRequestURL_lang(serveResourcesREST);
	
	var url = url_final + '&action=' + 'suggestions ' + '&searchKeyWords=' + userInput;
	
	jQuery.ajax({
		  dataType: "json",
		  url: url,
		  success: function(data){
			  
			  			var suggestionArray = new Array();
			  			for (var i = 0; i < data.length; i++){
			  				suggestionArray.push(data[i]);
		  				}
						 
						 updateSuggestionList(suggestionArray);
		  			},
	}); 
}





function setResourceRequestURL_lang(resourceReqURL){ 

	var base_url = null;
	
	if (!window.location.origin) { 
		base_url = window.location.protocol + "//" + window.location.hostname + (window.location.port ? ':' + window.location.port: '');
	} else {
		base_url = window.location.origin; 
	}

	var queryString_part = resourceReqURL.split("?")[1];
	var url_final = null;
	
	url_final = base_url + '/' + localLanguage + '/searchresult' + '?' + queryString_part;
	
	return url_final;

}
