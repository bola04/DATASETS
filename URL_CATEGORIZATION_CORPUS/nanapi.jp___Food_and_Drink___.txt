    singleFieldDelimiter: ' ',
    autocomplete: {
      source: tags, delay: 0
    }
  });

  $('#contact_send_request_ajax').on('ajax:beforeSend', function (xhr, setting) {
    body = $('#contact_body').val();
    if(!body.trim()) {
      $('#request-answer').html('内容を入れてお送りください');
      $('#request-answer').addClass('c-caution--text');
      return false;
    }

    $('#send_request_button').prop("disabled", true);
    $('#send_request_button').val('送信中です...');
    $.ajax({
      type: 'GET',
      url: '/contact/get_token',
      dataType: 'json',
    }).done(function (result) {
      if (result['token']) {
        $('input:hidden[name="authenticity_token"]').val(result['token']);
      }
    });
  });

  $('#contact_send_request_ajax').on('ajax:success', function (event, xhr, status, response) {
    if (xhr['message']) {
      $('#request-answer').html(xhr['message']);
      if (xhr['message'].match(/リクエストありがとうございました/)) { $('#contact_body').val(''); }
      $('#request-answer').addClass('c-caution--text');
      disabledFalseButton();
    } else {
      $('#request-answer').text('');
      disabledFalseButton();
    }
  });

  $('#contact_send_request_ajax').on('ajax:error', function () {
    $('#contact_body').val('');
    $('#request-answer').text('');
    disabledFalseButton();
  });

  function disabledFalseButton() {
    $('#send_request_button').prop("disabled", false);
    $('#send_request_button').val('送信する');
  }
});

function openHistories() {
  $(".c-pager--button-open").toggle();
  $(".c-pager--button-close").toggle();
  $("#disableHistories").slideToggle();
}
function renderHistories(histories) {
  if (histories.length == 0) return;
  $(".c-sidebar--history").show();
  $.each(histories, function(i, val) {
    var time = Math.floor((new Date)/1000) - val.date
    if (time/3600/24 >= 1) {
      time = Math.floor(time/3600/24) + "日前"
    } else if (time/3600 >= 1) {
      time = Math.floor(time/3600) + "時間前"
    } else {
      time = Math.ceil((time/60)%60)　+ "分前"
    }
    var title_url = "https://nanapi.com/ja/123".replace("123", val.id);
    var parent = $("#enableHistories");
    if (i >= 3) {
      parent = $("#disableHistories");
      $(".c-pager--button-open").show();
    }
    $('<a>').attr({class: 'c-media--side__item', href: title_url, id: 'gtm-history'}).append(
      $('<div>').attr({class: 'c-media__image'}).append($('<img>').attr({src: "//image.nanapi.com/100_100/dummy.jpg".replace("/dummy.jpg", val.image)}))
    ).append(
      $('<div>').attr({class: 'c-media__text'}).append(
        $('<div>').attr({class: 'c-media__title'}).append($('<h3>').text(val.title))
      ).append($('<div>').attr({class: 'c-media__date'}).text(time))
    ).appendTo(parent);
  });
}
