    function f_tranc(X) {
        u = ""
        var re = /^\s+/
        var re1 = /\s+$/

        u = u.replace(re1, "");
        return u

    function catchEnter(e,id) {
        var theKey = 0;
        e = (window.event) ? event : e;
        theKey = (e.keyCode) ? e.keyCode : e.charCode;
        if (theKey == "13") { // 13 is the key code for ENTER
            if (f_validatetext(id)) {
                GenSubmitSearchForm(id);
            }
        }
    }



    function f_validatetext(id) {

        var strmsg = "Δεν είναι δυνατή η αναζήτηση λέξης με λιγότερους από 2 χαρακτήρες ή κανέναν!";

        oSearchField = document.getElementById(id);

        if (!oSearchField)
            return false;
        else {

            var strsearch = f_tranc(oSearchField.value)

            if ((strsearch == '') || (strsearch.length < 3) || (strsearch == 'Αναζήτηση') || (strsearch == 'Search')) {
                alert(strmsg);
                return false;
            }

        }
        return true;

    }


    function GenSubmitSearchForm(id) {

        var frm = document.forms[0];

    }
