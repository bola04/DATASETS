      function GetParameter(parameter) {

        var loc = self.parent.document.location;
        var line = String(loc).split("?");
        if (line.length < 2) {
          return "";
        }
        var query = line[1];

        // convert to array of individual arguments
        var arglist = query.split("&");

        // process each argument
        var j;
        for (j in arglist) {
          var arg = arglist[j].split("=");
          if (arg[0] == parameter) {
            return unescape(arg[1]);
          }
        }
        return "";
      }

        var lastState = " ";

          for (var j=0; j<valueArray.length; j+=2) {
            for (var i in pages) {
              if (pages[i][shortname].substr(0,5) == valueArray[j].substr(0,5) &&
                  pages[i][level] == 0) {
                pages[i][state] = valueArray[j+1];
                break;
              }
            }
          }
        } else { // if cookies are disabled, open all folders

          if (GetCookie("cookietest") != 1) {

              if (pages[i][state] == "+") {

              }
            }
          }
          SetCookie("cookietest", 0);
        }

        // override updated states if there is a query-string argument
        var folder = GetParameter("folder").toLowerCase();
        if (folder != "") {
          for (var i in pages) {
            if (pages[i][shortname].substr(0,folder.length).toLowerCase() == folder) {
              pages[i][state] = "-";
            } else {
              pages[i][state] = "+";
            }
          }
        }

        // show only selected lines if there is a query-string argument directing us to do so
        var lineprefix = GetParameter("lineprefix").toLowerCase();

        // process each line in javascript table
        var lastLineIsFolder = false;
        for (var i in pages) { // process each line in javascript table

          // skip lines not satisfying the specified prefix
          if (pages[i][shortname].substr(0,lineprefix.length).toLowerCase() != lineprefix &&
              pages[i][shortname].substr(0,folder.length).toLowerCase() != folder) {
            continue;
          }

          // skip contents of closed folders
          if (pages[i][level] > 0 && !open[pages[i][level]-1]) {
            continue;
          } else {
            open[pages[i][level]] = (pages[i][state] != "+");
          }

          var line = "<tr>";
          var indent = "";
          for (var j=1; j<=pages[i][level]; j++) { // add indentation
            indent += "&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp";

          // for n4: use <img so there is no blue box around the image
          // for newer browsers: use <input type="image" so the hand appears when mouseover
          var imgTag = n4 ? "<img" : "<input type='image'";

          if (pages[i][shortname] == "<hr>") {
            line += "<td></td><td><hr></td>";
            lastLineIsFolder = false;
          } else if (pages[i][localurl] != "" && pages[i][localurl] != "-") { // file
            // adjust for some pages not being on my server
            var url = (IsFileProtocol() || pages[i][remoteurl] == "") ? localurl : remoteurl;
            if (pages[i][url] == "-") { // this file is for local use only

            }
            var img = (pages[i][state]=="$") ? "$" : " ";

            lastLineIsFolder = false;
          } else { // folder
            if (!IsFileProtocol() && pages[i][remoteurl] == "-") { // this folder is for local use only

            }
            if (folder=="" || pages[i][state]=="-") { // hide folder if query-string arg was passed in
              var img = (pages[i][state]=="+") ? "closed" : "open";
              var anchor = pages[i][shortname]; // get folder name from which we can build an anchor
              var endPosition = "";
              if ((endPosition=anchor.indexOf(" ")) != -1) { // stop before the first space
                anchor = anchor.substr(0,endPosition);
              }
              if ((endPosition=anchor.indexOf(",")) != -1) { // stop before the first comma
                anchor = anchor.substr(0,endPosition);
              }
              anchor = "<a name='" + anchor.toLowerCase() + "'></a>"; // build the anchor

                line += "<td>&nbsp;</td></tr><tr>";

            }

          }
          line += "</td></tr>\n";

          window.scrollTo(position[0], position[1]);
        }
      }

      function ChangeAll(newstate) {
        for (var i in pages) { // process each line in javascript table
          if (pages[i][state] != "") {
            pages[i][state] = newstate;
          }

        url = url.split("#")[0]; // remove anchor if any
        document.location.replace(url);
      }

      function GetScrollWidth() {
        var w = 0;
        if (document.body == undefined) { // N4 browser
          w = window.pageXOffset;
        } else {
          w = window.pageXOffset || // non IE browsers
              document.body.scrollLeft || // IE pre 6.0, and 6.0 in non-standards-complient mode
              document.documentElement.scrollLeft; // IE 6.0 in standards-compliant mode
        }
        return w ? w : 0;
      }

      function GetScrollHeight() {
        var h = 0;
        if (document.body == undefined) { // N4 browser
          h = window.pageXOffset;
        } else {
          h = window.pageYOffset || // non IE browsers
              document.body.scrollTop || // IE pre 6.0, and 6.0 in non-standards-complient mode
              document.documentElement.scrollTop; // IE 6.0 in standards-compliant mode
        }
        return h ? h : 0;
      }

      function Unload() {
        // save screen positions so we can restore it next time page is loaded
        if (!retoggled) {
        SetCookie("position", ""); // get rid of position cookie so that next reload is to top
        }
      }

      function UpdateCookie() {
        var value = "";
        for (var i in pages) { // process each line in javascript table
          if (pages[i][level] == 0 && pages[i][state] != "") {
            value += pages[i][shortname].substr(0,5) + "|" + pages[i][state] + "|"; 
          }
        }
        SetCookie("openfolder", value);
      }

      function SetCookie(cookieName, cookieValue) {
        var expires = new Date();
        var today = new Date();
        expires.setTime(today.getTime() + 1000*60*60*24*365);
        document.cookie = cookieName + "=" + cookieValue + "; expires=" + expires.toGMTString();
      }

      function GetCookie(cookieName) {
        var cookies = document.cookie.split(";");
        for (var i=0; i<cookies.length; i++) {
          while (cookies[i].charAt(0) == " ") {
            cookies[i] = cookies[i].slice(1);
          }
          var thisCookie = cookies[i].split("=");
          if (thisCookie[0] == cookieName) {
            if (thisCookie.length > 1) {
              return thisCookie[1];
            } else {
              return "";
            }
          }
        }
        return "";
      }
