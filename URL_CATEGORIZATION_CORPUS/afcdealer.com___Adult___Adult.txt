function intializeMenuWidth() {
	var nav = $('ul.sf-menu'),
		nav_item_height = 0,
		numMenuItems = $('> li', nav).size(),
		totalMenuItemWidth = 0,
		menuWidthRemainder = 0,
		wrapperWidth = 980,
		maxNavItemWidth = 200,
		priNavItems = new Array();
	
	/* First, determine the total width of each item in the navigation. */
	$('> li', nav).each(function (i) {
		totalMenuItemWidth += $(this).width();
		priNavItems.push(($(this).width() > maxNavItemWidth) ? maxNavItemWidth : $(this).width());
	});


	/* Primary navigation items (combined) are less width than the
	wrapper. */
	if(totalMenuItemWidth < wrapperWidth) {
		menuWidthRemainder = wrapperWidth - totalMenuItemWidth;

		$('> li', nav).each(function() {
			var tmp_width = $(this).width();
			$(this).width(tmp_width + parseInt(menuWidthRemainder / numMenuItems));
		});

		tmp_width = $('> li:first', nav).width();
		$('> li:first', nav).width(tmp_width + (menuWidthRemainder % numMenuItems));
	}

}

/****************************************************************
 * Disable all input and submit buttons within a given form.  Used
 * to prohibit double submits.
 * @param formObj - Form Object
 ****************************************************************/
function preventDoubleSubmit(formObj) { 
	 formObj.submit(function() {
		$('input[type=submit]', this).attr('disabled', 'disabled');
		$('input[type=button]', this).attr('disabled', 'disabled');
	});
}	
 
 /****************************************************************
  * Generic jQuery dialog to substitute for the alert dialog
  * @param title - The title to be shown for the dialog
  * @param message - The message to be displayed
  * @param curCtrl - the form control to return focus
  ****************************************************************/
 function afcAlert(title, message, curCtrl) {
	if(curCtrl != null && typeof curCtrl == "string"){ 
		jQuery(curCtrl).focus();
	}
	jQuery('<div style="text-align:center; margin-left:auto; margin-right:auto; width:70%;"></div>').html(message)
	.dialog({autoOpen:true,
			title: title,
			modal: true,
            buttons: { "OK": function() { 
				jQuery(this).dialog("close"); 
				if(curCtrl != null && typeof curCtrl == "string"){ 
					jQuery(curCtrl).focus();
				}
			}},
            resizable:false,
	});		
 }	 
   
  /****************************************************************
   * Sets focus to the input control specified.  A small delay is
   * used to ensure DOM is loaded.  Without the delay, some
   * browsers may loose focus on control.
   * @param control - The control which focus should be set to
   ****************************************************************/
  function setDefaultFocus(control) {
		if(control != null && typeof control == "string") { 
			setTimeout(function(){jQuery(control).focus();}, 500);
		}
		else {
			jQuery("input:text:visible:first").focus();
		}
  }
  
 /****************************************************************
  * Initializes the question mark bubble the popups up when
  * question mark icon is hovered.
  ****************************************************************/
  function initQuestionMarkBubbles() {  
   $('.questionMarkBubble').each(function () {
     // options
     var distance = 10;
     var time = 250;
     var hideDelay = 500;

     var hideDelayTimer = null;

     // tracker
     var beingShown = false;
     var shown = false;
     
     var questionMark = $('.questionMark', this);
     var questionMarkPopup = $('.questionMarkPopup', this).css('opacity', 0);
     var left = questionMark.position().left;
     var top = questionMark.position().top;


     // set the mouseover and mouseout on both element
     $([questionMark.get(0), questionMarkPopup.get(0)]).mouseover(function () {
       // stops the hide event if we move from the trigger to the popup element
       if (hideDelayTimer) clearTimeout(hideDelayTimer);

       // don't trigger the animation again if we're being shown, or already visible
       if (beingShown || shown) {
         return;
       } else {
         beingShown = true;

         // reset position of popup box
         questionMarkPopup.css({
           top: top - 55,
           left: left + 10,
           width: 150,
           display: 'block' // brings the popup back in to view
         })

         // (we're using chaining on the popup) now animate it's opacity and position
         .animate({
           top: '-=' + distance + 'px',
           opacity: 1
         }, time, 'swing', function() {
           // once the animation is complete, set the tracker variables
           beingShown = false;
           shown = true;
         });
       }
     }).mouseout(function () {
       // reset the timer if we get fired again - avoids double animations
       if (hideDelayTimer) clearTimeout(hideDelayTimer);
       
       // store the timer so that it can be cleared in the mouseover if required
       hideDelayTimer = setTimeout(function () {
         hideDelayTimer = null;
         questionMarkPopup.animate({
           top: '-=' + distance + 'px',
           opacity: 0
         }, time, 'swing', function () {
           // once the animate is complete, set the tracker variables
           shown = false;
           // hide the popup entirely after the effect (opacity alone doesn't do the job)
           questionMarkPopup.css('display', 'none');
         });
       }, hideDelay);
     });
   });
  }

 /****************************************************************
  * Initializes the accordion toggle container.
  ****************************************************************/
 function initAccordion(autoHide) {
	if (autoHide != false) {
		// Hide (Collapse) the toggle containers on load
		$(".toggle_container").hide(); 
	}
	// Switch the "Open" and "Close" state per click then slide up / down (depending on open / close state)
	$("h2.toggle_trigger").click(function() {
		$(this).toggleClass("active").next().slideToggle("slow");
		return false; // Prevent the browser jump to the link anchor
	});
	
 }
 
 /********************************************
 * Used to perform Javascript navigation to Portal friendly URLs.
 * This is necessary because the same code can be called from the production and demo versions
 * of AFC Dealer, and we need to make sure we land on the right virtual portal.
 ********************************************/
 function navigateToPortalURL(targetURL) {
	var href = window.location.href;
	var portalRootIndex = href.indexOf("/wps/", 0);
	var authenticatedPortalIndex = href.indexOf("/", portalRootIndex + 1);
	var virtualPortalRootIndex = href.indexOf("/", authenticatedPortalIndex + 1);
	var virtualPortalEndIndex = href.indexOf("/", virtualPortalRootIndex + 1);
	href = href.substring(0, virtualPortalEndIndex);
	href = href + targetURL;
	window.location.href = href;
 }
