  <script type="text/javascript">
   var $rso2 = $("#rso2");
        var $currentSlide;
        var $slider;
        $(function () {
            if (getUrlParameter("test") != null) {
                reloadRsoInfos();

                $(window).scroll(function () {
                    refreshSlider();
                });
                $(window).resize(function () {
                    refreshSlider();
                });
            }
        });
        var CategoriesInfo = {
            "ogolne": false,
            "meteorologiczne": false,
            "informacje-drogowe": false
        };
        function getUrlParameter(sParam) {
            var sPageURL = decodeURIComponent(window.location.search.substring(1)),
                sURLVariables = sPageURL.split('&'),
                sParameterName,
                i;

            for (i = 0; i < sURLVariables.length; i++) {
                sParameterName = sURLVariables[i].split('=');

                if (sParameterName[0] === sParam) {
                    return sParameterName[1] === undefined ? true : sParameterName[1];
                }
            }
        }
        function refreshSlider() {
            var sh = 130;
            if ($slider != null) {
                sh = $slider[0].scrollHeight;
            }
            if ($currentSlide != null) {
                sh = $currentSlide.height();
            }
            $rso2
                .stop()
                .animate({ "marginTop": ($(window).scrollTop() + $(window).height() - sh - 20) + "px" }, "slow");
        }
        function reloadRsoInfos() {
            $.each(CategoriesInfo, function (index, item) {
                getRsoInfo(index);
            });
        }
        function drawRsoSlider() {
            var allLoaded = true;
            $.each(CategoriesInfo, function (index, item) {
                allLoaded = allLoaded & item;
            });
            if (allLoaded) {
                $("#rso2>ul").css("max-height", "0");
                $slider = $("#rso2>ul").bxSlider({
                    mode: 'fade',
                    pager: true,
                    wrapperClass: "rso-wrapper bg-danger",
                    //preloadImages: 'all',
                    controls: true,
                    auto: true,
                    pause: 10000,
                    autoDelay: 0,
                    autoHover: true,
                    adaptiveHeight: true,
                    onSlideAfter: function (slide, oldIndex, newIndex) {
                        $currentSlide = slide;
                        refreshSlider();
                    }
                });
                $("button.rso_close").on("click", function (e) {
                    $.cookie("rso_hide", "1");
                    $(".rso-wrapper").animate({ "left": ($(window).width() - 40) + "px" }, "slow", function (e) {
                        $(".rso-wrapper").on("click", function (e) {
                            $(".rso-wrapper").animate({ "left": "0px" }, "slow", function () { $.cookie("rso_hide", "0");$(".rso-wrapper").off("click");});
                        });
                    });
                });
            }
        }
        function getRsoInfo(category) {
            var $category = category;
            return $.ajax({
                type: "GET",
                dataType: "jsonp",
                url: "https://app.malopolska.uw.gov.pl/ws/proxy/jsonp?url=http://komunikaty.tvp.pl/komunikaty/malopolskie/" + category + "/1?_format=json",
                success: function (data) {
                    var rso = $.parseJSON(data);
                    if (rso.pagination.totalitems > 0) {
                        $.each(rso.newses, function (index, item) {
                            var cl = $category;
                            if ($category == 'meteorologiczne') {
                                cl = "burza";
                                if (item.title.indexOf("wiatr") > -1) {
                                    cl = "wiatr";
                                }
                                if (item.title.indexOf("deszcz") > -1) {
                                    cl = "deszcz";
                                }

                                if (item.title.indexOf("śnieg") > -1) {
                                    cl = "snieg";
                                }

                            } else if ($category == 'informacje-drogowe') {
                                cl = "korek";
                            }
                            $("#rso2>ul").append("<li id=\"ID_" + item.id + "\"><button type=\"button\" class=\"close rso_close\"><span >&times;</span></button><div class=\"rso_info " + cl + "\"><div class=\"rso_lead\">" + item.title + "</div><div class=\"rso_text\">" + item.content + "</div></div></li>");   //shortcut==lead
                        });
                    }
                },
                error: function (e, a, b) {
                    //alert("error");
                },
                complete: function () {
                    CategoriesInfo[category] = true;
                    drawRsoSlider();
                }
            });
        }
  </script>
