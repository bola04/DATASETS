function getFbPermissions()
{
	
	FB.login(function(response) {
		if (response.authResponse) {
				//console.log(response);
				var request_ids = [];
				FB.api('/me/apprequests', function(r){
					if (r && r.data) {
						for (var i = 0; i < r.data.length; i++) {
							var id = r.data[i].id.split('_')[0]; // this is the request id
								//console.log(id, r.data[i]);
								request_ids.push(id);
						};
					}
					//console.log("REQUEST_IDS="+ request_ids.join(','));
					//console.log(Routing.generate('mainPage', { request_ids: request_ids.join(',')}));

					$.ajax(
					{
							url:  Routing.generate('user_fb_canvaslogin', { accessToken: response.authResponse.accessToken}),							
							cache: false,	                        
							success: function(data) 
							{
								//console.log(data);
								window.location.href= Routing.generate('mainPage', { request_ids: request_ids.join(',')});     					
							},
							error: function(data)
							{
								//console.log(data);
								//TODO:: Display error message
						    }                
					});
		    	    
				});

		    

		     
		} 
		else 
		{
			//console.log('User cancelled login or did not fully authorize.');
			//window.top.location.href = Routing.generate('mainPage');
		}
	}, {scope: 'public_profile, email, publish_actions, user_friends'});
	
}
