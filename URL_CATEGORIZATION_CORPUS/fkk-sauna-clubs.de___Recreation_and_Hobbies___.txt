    function addMarker(location, title, market, region, infotext, pincolor) {

        if( pincolor == "") {pincolor = 'black'; }

        var icon = 'http://www.fkk-sauna-clubs.de/images/icon/dot_' + pincolor + '.png' ;
        var shadow = 'http://www.fkk-sauna-clubs.de/images/icon/shadow.png' ;

        var marker = new google.maps.Marker({
            position: location,
            title: title,
            icon: icon,
            map: map,
            shadow: shadow
        });
        google.maps.event.addListener(marker, 'click', function () {
            infowindow.setContent(infotext);
            infowindow.open(map, marker);
        });
        markers.push({'marker': marker, 'region': region, 'market': market});
    }


    function hideMarkers(region, market) {
        for (var i=0; i<markers.length; i++) {
            if ((!region || markers[i].region == region) && (!market || markers[i].market == market) && markers[i].marker.getMap() != null) {
                markers[i].marker.setMap(null);
            }
        }
    }


    function loadMap(langCode) {
        google.load('maps', '3.7', {
            'other_params' :
                'sensor=false&libraries=places&language=' + langCode,
            'callback' : initialize
        });
    }

    function panTo(region, markets) {
        var marketsSearch = ','+markets+',';
        var currentZoom = map.getZoom();
        var bounds = new google.maps.LatLngBounds();
        for (var i=0; i<markers.length; i++) {
            if ((!region || markers[i].region == region) && (!markets || (marketsSearch.indexOf(','+markers[i].market+',') > -1))) {
                bounds.extend(markers[i].marker.getPosition());
            }
        }
        map.panTo(bounds.getCenter());
        var zoomInterval = setInterval(
            function() {
                map.fitBounds(bounds);
                if (map.getZoom() > (currentZoom+1)) {
                    map.setZoom(currentZoom+1);
                    currentZoom = currentZoom+1;
                } else window.clearInterval(zoomInterval);
            },
            500
        );
    }
