   jQuery(document).ready(function( $ ){

     // set up the configuration for the slideshow

            var mySiema = new Siema({
        selector: '.siema', // the div we are using
        duration: 250, // the animation duration NOT the autoplay duration (See setInterval function below)
        loop: true, // do we loop
        perPage: {
         1: 1,
         767: 2
       } // how many divs do we show at one time
      });
    
    // pagination is also a bit of a custom roll since we are advancing based on perPage.
    Siema.prototype.addPagination = function() {
      var _this = this;

      // mySiema.perPage should report back the correct value based on breakpoints
      var wrapper = document.createElement('div');
      jQuery(wrapper).addClass('slide_nav');
      wrapper.setAttribute("aria-hidden", "true");

      var _loop = function _loop(i) {
        // console.log('perPage is %O', mySiema.perPage);
        // create a button
        var btn = document.createElement('button');
        jQuery(btn).attr('tabindex', -1);

        if (jQuery(_this.innerElements[i]).hasClass('no-image')) {
          jQuery(btn).addClass('no-image');
        }
        if (i == 0) {
          jQuery(btn).addClass('active');
        }
        // btn.textContent = i+1;
        // add the click handler
        // this probably needs to add an active class
        // and remove the active class from the other buttons.
        btn.addEventListener('click', function () {
          jQuery('.slide_nav button').removeClass('active');
          jQuery(this).addClass('active');
          mySiema.goTo(i);
        });
        // add it to the slides div
        wrapper.appendChild(btn);
      };

      for (var i = 0; i < this.innerElements.length; i += mySiema.perPage) {
        _loop(i);
      }
      this.selector.appendChild(wrapper);
    }

    // Trigger pagination creator
    mySiema.addPagination();

    //console.log('Siema is as Siema does %O', mySiema);

    // This sets a value of true which we use in our interval to keep the slide show from moving when the mouse is hovering.
    Siema.prototype.mouseoverHandler = function(e) {
      //console.log('over it! ');
      mySiema.hover = true;
    }
    mySiema.selector.addEventListener('mouseover', Siema.prototype.mouseoverHandler);

    // This sets a value of false when the mouse exits so we start the slide show again.
    mySiema.selector.addEventListener('mouseleave', function() {
      //console.log('Mouse has left the building');
      mySiema.hover = false;
     }, false);

    // the autoplay code
    // commented out for debugging styles
    // this probably also needs to add an active class
    // and remove the active class from the other buttons.

      getNextSlideIdx = function(idx) {
        //idx = idx - 1;
        var idx2, count;
        //console.log('IDX: ', idx);
        if (mySiema.perPage == 1) {

        if (idx > mySiema.innerElements.length - mySiema.perPage && mySiema.config.loop) {
          return [0, 1];
        } else {
            idx2 =  ((idx + mySiema.perPage) / mySiema.perPage);
            count = idx2;
            if (mySiema.isHidden(idx2)) {
              return getNextSlideIdx(idx2);
            } else {
              return [count, count + 1];
            }
        }
          } else {
          if (idx >= mySiema.innerElements.length - mySiema.perPage && mySiema.config.loop) {
            return 0;
          } else {
            idx2 =  ((idx + mySiema.perPage) / mySiema.perPage) + 1;

            return [idx + mySiema.perPage, idx2];
          }
         }
      }


 setInterval(function() {
      if(!mySiema.hover) {
        if(mySiema.currentSlide >= mySiema.innerElements.length - mySiema.perPage && mySiema.config.loop) {
          // if we reached the end of the show, go back to the beginning
          mySiema.goTo(0);
          jQuery('.slide_nav button').removeClass('active');
          jQuery( ".slide_nav button:nth-child(1)" ).addClass( "active" );
        } else {
          var count = getNextSlideIdx(mySiema.currentSlide);

          jQuery('.slide_nav button').removeClass('active');

          //mySiema.goTo(mySiema.currentSlide + mySiema.perPage);
          mySiema.goTo(count[0]);
        }
      }
    }, 6000); // interval is 6 seconds

    jQuery('.siema ul').attr('aria-labelledby', 'siema_label');

   <script>
    /**
   * Opens window screen centered.
   * @param windowWidth the window width in pixels (integer)
   * @param windowHeight the window height in pixels (integer)
   * @param windowOuterHeight the window outer height in pixels (integer)
   * @param url the url to open
   * @param wname the name of the window
   * @param features the features except width and height (status, toolbar, location, menubar, directories, resizable, scrollbars)
   */
  function CenterWindow(windowWidth, windowHeight, windowOuterHeight, url, wname, features) {
    var centerLeft = parseInt((window.screen.availWidth - windowWidth) / 2);
    var centerTop = parseInt(((window.screen.availHeight - windowHeight) / 2) - windowOuterHeight);

    var misc_features;
    if (features) {
      misc_features = ', ' + features;
    }
    else {
      misc_features = ', status=no, location=no, scrollbars=yes, resizable=yes';
    }
    var windowFeatures = 'width=' + windowWidth + ',height=' + windowHeight + ',left=' + centerLeft + ',top=' + centerTop + misc_features;
    var win = window.open(url, wname, windowFeatures);
    win.focus();
    return win;
  }
