          jQuery(function($) {

	var lastRotateIndex = 1;
	var bRotate = true;
	var bRotateStop = false;

	if ($('#rankList li').length > 0)
		$('.keyword_r_no1').clone().appendTo('#rankList ul');

	function rotateRank(index) {

		setTimeout(function(){

			lastRotateIndex = index;

			if (!bRotate) {
				bRotateStop = true;
				return;
			}

			$('#rankList ul').animate({
				top: -$('#rankList li').height() * index
			}, 500, function() {
				if (index == $('#rankList li').length -1) {
					$('#rankList ul').css('top', '0px');
					index = 0;
				}
				rotateRank(index + 1);
			});
		}, 2000);
	}

    rotateRank(1);

	$("#rankList").click(function(){
		bRotateStop = false;
		bRotate = false;
		$("#rankLayer").fadeIn();
	});

	$("#rankLayer").hover(function(){}, function(){
		$("#rankLayer").fadeOut();
		bRotate = true;
		if(bRotateStop) {
			bRotateStop = false;
			rotateRank(lastRotateIndex);
		}
	});
});

   function setCookie(cName, cValue, cDay){
	var expire = new Date();
    expire.setDate(expire.getDate() + cDay);
    cookies = cName + '=' + escape(cValue) + '; path=/ '; // 한글 깨짐을 막기위해 escape(cValue)를 합니다.
    if(typeof cDay != 'undefined') cookies += ';expires=' + expire.toGMTString() + ';';
    document.cookie = cookies;
}

function getCookie(cName) {
    cName = cName + '=';
    var cookieData = document.cookie;
    var start = cookieData.indexOf(cName);
    var cValue = '';
    if(start != -1){
        start += cName.length;
        var end = cookieData.indexOf(';', start);
        if(end == -1)end = cookieData.length;
        cValue = cookieData.substring(start, end);
    }
    return unescape(cValue);
}
