    function downloadUrl2(url, callback) {
        var request = new XMLHttpRequest();
        request.onreadystatechange = function () {
            if (request.readyState == 4) {
                callback(request.responseText, request.status);
                request.onreadystatechange = doNothing;
            }
        };

        request.open('GET', url, true);
        request.send(null);
    }

    function ban_lonuevo(ruta, elem1, elem2, elem3) {
        downloadUrl2(ruta, function (data) {
            var xml = parseXml(data);
            var markers = xml.documentElement.getElementsByTagName(elem1);
            var convec = 0;
            var imgcon = 0;
            var alt_img = "";
            for (var recel = 0; recel < markers.length; recel++) {
                for (var nodhijo = 0; nodhijo < markers[recel].childNodes.length; nodhijo++) {
                    var nodo = markers[recel].childNodes[nodhijo].nodeName;
                    switch (nodo) {
                        case "url":
                            var enlace = null;
                            enlace = markers[recel].childNodes[nodhijo].textContent;
                            if (enlace == undefined) {
                                enlace = markers[recel].childNodes[nodhijo].nodeTypedValue;
                            }
                            var param = enlace.split('?')[1];
                            if (param != undefined) {
                                param = param.replace(/[#]/gi, '&');
                                concatenacad = enlace.split('?')[0];
                                enlace = concatenacad + "?" + param;
                            }
                            var imghref = document.getElementById(elem2 + convec);
                            if (imghref != null)
                                imghref.href = enlace;
                            convec++;
                            break;
                        case "alt":
                            var alt_img = markers[recel].childNodes[nodhijo].textContent;
                            if (alt_img == undefined) {
                                alt_img = markers[recel].childNodes[nodhijo].nodeTypedValue;
                            }

                            imgalt = document.getElementById(elem3 + imgcon);
                            if (imgalt != null)
                                imgalt.alt = alt_img;
                            imgcon++;
                            break;
                    }

                }
            }
        });
    }
